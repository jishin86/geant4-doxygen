138 //--------------------------------------------------------------------------
139 void G4VDivisionParameterisation::CheckOffset( G4double maxPar )
140 {
141  if( foffset >= maxPar )

143  std::ostringstream message;
144  message << "Configuration not supported." << G4endl

153 //--------------------------------------------------------------------------
154 void G4VDivisionParameterisation::CheckNDivAndWidth( G4double maxPar )
155 {
156  if( (fDivisionType == DivNDIVandWIDTH)
157  && (foffset + fwidth*fnDiv - maxPar > kCarTolerance ) )
158  {
159  std::ostringstream message;
160  message << "Configuration not supported." << G4endl
161  << "Division of solid " << fmotherSolid->GetName()
162  << " has too big offset + width*nDiv = " << G4endl
163  << " " << foffset + fwidth*fnDiv << " > "
164  << foffset << ". Width = "
165  << G4endl
166  << " " << fwidth << ". nDiv = " << fnDiv << " !";
167  G4Exception("G4VDivisionParameterisation::CheckNDivAndWidth()",
168  "GeomDiv0001", FatalException, message);
169  }
170 }
