90 void B1ConRunAction::EndOfRunAction(const G4Run* aRun)
91 {
92  G4int nofEvents = aRun->GetNumberOfEvent();
93  if (nofEvents == 0) return;
94 
95  const B1ConRun* b1ConRun = static_cast<const B1ConRun*>(aRun);
96 
97  // Compute dose
98  //
99  G4double edep = b1ConRun->GetEdep();
100  G4double edep2 = b1ConRun->GetEdep2();
101  G4double rms = edep2 - edep*edep/nofEvents;
102  if (rms > 0.) rms = std::sqrt(rms); else rms = 0.;
103 
104  const B1DetectorConstruction* detectorConstruction
105  = static_cast<const B1DetectorConstruction*>
106  (G4RunManager::GetRunManager()->GetUserDetectorConstruction());
107  G4double mass = detectorConstruction->GetScoringVolume()->GetMass();
108  G4double dose = edep/mass;
109  G4double rmsDose = rms/mass;
110 
111  // Run conditions
112  // note: There is no primary generator action object for "master"
113  // run manager for multi-threaded mode.
114  const B1PrimaryGeneratorAction* generatorAction
115  = static_cast<const B1PrimaryGeneratorAction*>
116  (G4RunManager::GetRunManager()->GetUserPrimaryGeneratorAction());
117  G4String runCondition;
118  if (generatorAction)
119  {
120  const G4ParticleGun* particleGun = generatorAction->GetParticleGun();
121  runCondition += particleGun->GetParticleDefinition()->GetParticleName();
122  runCondition += " of ";
123  G4double particleEnergy = particleGun->GetParticleEnergy();
124  runCondition += G4BestUnit(particleEnergy,"Energy");
125  }
126 
127  // Print
128  //
129  if (IsMaster())
130  {
131 
132  for ( G4int i = 0 ; i != b1ConRun->GetNumberOfEvent(); i++ ) {
133  G4double aDose = b1ConRun->GetEdepPerEvent(i)/mass/gray;
134  fdose_tally->AddScore( aDose );
135  }
136 
137  fdose_tally->ShowResult();
138  fdose_tally->ShowHistory();
139  delete fdose_tally;
140 
141  G4cout
142  << "\n--------------------End of Global Run-----------------------";
143  }
144  else
145  {
146  G4cout
147  << "\n--------------------End of Local Run------------------------";
148  }
149  G4cout
150  << "\n The run consists of " << nofEvents << " "<< runCondition
151  << "\n Dose in scoring volume : "
152  << G4BestUnit(dose,"Dose") << " +- " << G4BestUnit(rmsDose,"Dose")
153  << "\n------------------------------------------------------------\n"
154  << G4endl;
155 
156 }
