303  theDNApChargeDecreaseProcess->SetEmModel(new G4DummyModel());
304  pman->AddDiscreteProcess(theDNApChargeDecreaseProcess);
305 
306  G4DNAChargeDecrease* theDNAa2ChargeDecreaseProcess =
307  new G4DNAChargeDecrease("alpha_G4DNAChargeDecrease");
308  theDNAa2ChargeDecreaseProcess->SetEmModel(new G4DummyModel());
309  a2man->AddDiscreteProcess(theDNAa2ChargeDecreaseProcess);
310 
311  G4DNAChargeDecrease* theDNAa1ChargeDecreaseProcess =
312  new G4DNAChargeDecrease("alpha+_G4DNAChargeDecrease");
313  theDNAa1ChargeDecreaseProcess->SetEmModel(new G4DummyModel());
314  a1man->AddDiscreteProcess(theDNAa1ChargeDecreaseProcess);
315 
316  G4DNAChargeIncrease* theDNAa1ChargeIncreaseProcess =
317  new G4DNAChargeIncrease("alpha+_G4DNAChargeIncrease");
318  theDNAa1ChargeIncreaseProcess->SetEmModel(new G4DummyModel());
319  a1man->AddDiscreteProcess(theDNAa1ChargeIncreaseProcess);
320 
321  G4DNAChargeIncrease* theDNAa0ChargeIncreaseProcess =
322  new G4DNAChargeIncrease("helium_G4DNAChargeIncrease");
323  theDNAa0ChargeIncreaseProcess->SetEmModel(new G4DummyModel());
324  a0man->AddDiscreteProcess(theDNAa0ChargeIncreaseProcess);
325 
326  G4DNAChargeIncrease* theDNAh0ChargeIncreaseProcess =
327  new G4DNAChargeIncrease("hydrogen_G4DNAChargeIncrease");
328  theDNAh0ChargeIncreaseProcess->SetEmModel(new G4DummyModel());
329  h0man->AddDiscreteProcess(theDNAh0ChargeIncreaseProcess);
330 
331  // limits for DNA model applicability
332  static const G4double elowest= 7.4 * eV;
333  static const G4double elimel = 1 * MeV;
334  static const G4double pminbb = 2 * MeV;
335  static const G4double pmin = 0.1 * keV;
336  static const G4double pmax = 100 * MeV;
337  static const G4double hemin = 1 * keV;
338  static const G4double ionmin = 0.5 * MeV;
339 
340  // low-energy capture
341  G4LowECapture* ecap = nullptr;
342  if(G4DNAChemistryManager::IsActivated() == false)
343  {
344  // Note: G4DNAElectronSolvation could also be used instead of G4LowECapture
345  ecap = new G4LowECapture(elowest);
346  // ecap->SetVerboseLevel(1);
347  eman->AddDiscreteProcess(ecap);
348  }
349  else
350  {
351  // When chemistry is activated: G4DNAElectronSolvation turns the electron
352  // to a solvated electron, otherwise it kills the electron at the
353  // corresponding high energy limit of the model
354  G4DNAElectronSolvation* pSolvatation =
355  new G4DNAElectronSolvation("e-_G4DNAElectronSolvation");
356  pSolvatation->SetEmModel(G4DNASolvationModelFactory::GetMacroDefinedModel());
357  eman->AddDiscreteProcess(pSolvatation);
358  }
359 
360  G4LowECapture* pcap = new G4LowECapture(pmin);
361  pman->AddDiscreteProcess(pcap);
362  G4LowECapture* icap = new G4LowECapture(ionmin);
363  iman->AddDiscreteProcess(icap);
364  G4LowECapture* a2cap = new G4LowECapture(hemin);
365  a2man->AddDiscreteProcess(a2cap);
366  G4LowECapture* a1cap = new G4LowECapture(hemin);
367  a1man->AddDiscreteProcess(a1cap);
368  G4LowECapture* a0cap = new G4LowECapture(hemin);
369  a0man->AddDiscreteProcess(a0cap);
370  G4LowECapture* h0cap = new G4LowECapture(ionmin);
371  h0man->AddDiscreteProcess(h0cap);
372 
373  // loop over regions
374  for(G4int i = 0; i < nreg; ++i)
375  {
376  G4String reg = regnamesDNA[i];
377  if(IsVerbose())
378  {
379  G4cout << "### DNA models type " << typesDNA[i]
380  << " are activated for G4Region " << reg << G4endl;
381  }
382 
383  // type of DNA physics
384  G4int itype = 0;
385 
386  if(0 == itype) {
387  AddElectronModels0(reg, ecap, emsc, elowest, elimel);
388  AddProtonModels0(reg, pmsc, elimel, pminbb, pmax);
389  AddHeliumModels0(reg, a1msc, a2msc, elimel, pminbb, pmax);
390  AddGenericIonModels0(reg, pminbb);
391  DeactivateNuclearStopping(pman, elimel);
392  DeactivateNuclearStopping(a1man, elimel);
393  DeactivateNuclearStopping(a2man, elimel);
394  }
395  }
396  G4LossTableManager::Instance()->EmConfigurator()->AddModels();
397 }

401 void G4EmDNAPhysicsActivator::AddElectronModels0(const G4String& reg,
402  G4LowECapture* ecap,
403  G4bool emsc,
404  G4double elowest,
405  G4double elimel)
406 {
407  G4EmConfigurator* em_config =
408  G4LossTableManager::Instance()->EmConfigurator();
409  G4VEmModel* mod;
410 
411  static const G4double elimin = 1 * MeV;
412  static const G4double elimvb = 100 * eV;
413  static const G4double elimat = 13 * eV;
414  static const G4double elim1 = 10 * keV;
415 
416  // for e- 100 MeV is a limit between different msc models
417  G4double emax = theParameters->MaxKinEnergy();
418 
419  if(emsc) {
420  G4UrbanMscModel* msc = new G4UrbanMscModel();
421  msc->SetActivationLowEnergyLimit(elimel);
422  G4double emaxmsc = std::min(100*MeV, emax);
423  em_config->SetExtraEmModel("e-", "msc", msc, reg, 0.0, emaxmsc);
424  } else {
425  mod = new G4eCoulombScatteringModel();
426  mod->SetActivationLowEnergyLimit(elimel);
427  em_config->SetExtraEmModel("e-", "CoulombScat", mod, reg, 0.0, emax);
428  }
429 
430  // cuts and solvation
431  if(ecap) {
432  ecap->AddRegion(reg);
433  } else {
434  // For this condition to work, the chemistry list has to be called
435  // before any standard EM physics list
436  mod = new G4DNAOneStepThermalizationModel();
437  em_config->SetExtraEmModel("e-", "e-_G4DNAElectronSolvation",
438  mod, reg, 0., elowest);
439  }
440 
441  // elastic
442  mod = new G4DNAChampionElasticModel();
443  em_config->SetExtraEmModel("e-", "e-_G4DNAElastic",
444  mod, reg, 0.0, elimel);
445  // ionisation
446  mod = new G4MollerBhabhaModel();
447  mod->SetActivationLowEnergyLimit(elimin);
448  em_config->SetExtraEmModel("e-", "eIoni",
449  mod, reg, 0.0, emax,
450  new G4UniversalFluctuation());
451 
452  mod = new G4DNABornIonisationModel();
453  em_config->SetExtraEmModel("e-", "e-_G4DNAIonisation",
454  mod, reg, elim1, elimin);
455 
456  mod = new G4DNAEmfietzoglouIonisationModel();
457  em_config->SetExtraEmModel("e-", "e-_G4DNAIonisation",
458  mod, reg, 0.0, elim1);
459 
460  // exc
461  mod = new G4DNAEmfietzoglouExcitationModel();
462  em_config->SetExtraEmModel("e-", "e-_G4DNAExcitation",
463  mod, reg, 0.0, elim1);
464 
465  mod = new G4DNABornExcitationModel();
466  em_config->SetExtraEmModel("e-", "e-_G4DNAExcitation",
467  mod, reg, elim1, elimin);
468 
469  mod = new G4DNASancheExcitationModel();
470  em_config->SetExtraEmModel("e-", "e-_G4DNAVibExcitation",
471  mod, reg, 0.0, elimvb);
472 
473  mod = new G4DNAMeltonAttachmentModel();
474  em_config->SetExtraEmModel("e-", "e-_G4DNAAttachment",
475  mod, reg, 0.0, elimat);
476 
477 }

481 void G4EmDNAPhysicsActivator::AddProtonModels0(const G4String& reg,
482  G4bool pmsc, G4double elimel,
483  G4double pminbb, G4double pmax)
484 {
485  G4EmConfigurator* em_config =
486  G4LossTableManager::Instance()->EmConfigurator();
487  G4VEmModel* mod;
488 
489  static const G4double gmmax = 500 * keV;
490 
491  G4double emax = theParameters->MaxKinEnergy();
492 
493  // proton
494 
495  // if SS physics list msc process does not exist
496  if(pmsc) {

498  msc->SetActivationLowEnergyLimit(elimel);
499  em_config->SetExtraEmModel("proton", "msc", msc, reg, 0.0, emax);
500  }

508  em_config->SetExtraEmModel("proton", "hIoni",
509  mod, reg, 0.0, pminbb,
510  new G4UniversalFluctuation());
511 
512  mod = new G4BetheBlochModel();
513  mod->SetActivationLowEnergyLimit(pmax);
514  em_config->SetExtraEmModel("proton", "hIoni",
515  mod, reg, pminbb, emax,
516  new G4UniversalFluctuation());
517 
518  mod = new G4DNARuddIonisationModel();
519  em_config->SetExtraEmModel("proton", "proton_G4DNAIonisation",
520  mod, reg, 0.0, gmmax);
521 
522  mod = new G4DNABornIonisationModel();
523  em_config->SetExtraEmModel("proton", "proton_G4DNAIonisation",
524  mod, reg, gmmax, pmax);
525 
526  mod = new G4DNAMillerGreenExcitationModel();
527  em_config->SetExtraEmModel("proton", "proton_G4DNAExcitation",
528  mod, reg, 0.0, gmmax);
529 
530  mod = new G4DNABornExcitationModel();
531  em_config->SetExtraEmModel("proton", "proton_G4DNAExcitation",
532  mod, reg, gmmax, pmax);
533 
534  mod = new G4DNADingfelderChargeDecreaseModel();
535  em_config->SetExtraEmModel("proton", "proton_G4DNAChargeDecrease",
536  mod, reg, 0.0, pmax);
537 
538  mod = new G4DNAIonElasticModel();
539  em_config->SetExtraEmModel("proton", "proton_G4DNAElastic",
540  mod, reg, 0.0, elimel);
