66 G4bool ExN04CalorimeterSD::ProcessHits(G4Step*aStep,G4TouchableHistory*ROhist)
67 {
68  if(!ROhist) return false;
69  G4double edep = aStep->GetTotalEnergyDeposit();
70  if(verboseLevel>1) G4cout << "Next step edep(MeV) = " << edep/MeV << G4endl;
71  if(edep==0.) return false;
72 
73  G4VPhysicalVolume* physVol = ROhist->GetVolume();
74  //ROhist->MoveUpHistory();
75  //G4VPhysicalVolume* mothVol = ROhist->GetVolume(1);
76  G4int copyIDinZ = ROhist->GetReplicaNumber();
77  G4int copyIDinPhi = ROhist->GetReplicaNumber(1);
78 
79  if(CellID[copyIDinZ][copyIDinPhi]==-1)
80  {
81  ExN04CalorimeterHit* calHit
82  = new ExN04CalorimeterHit
83  (physVol->GetLogicalVolume(),copyIDinZ,copyIDinPhi);
84  calHit->SetEdep( edep );
85  G4AffineTransform aTrans = ROhist->GetHistory()->GetTopTransform();
86  aTrans.Invert();
87  calHit->SetPos(aTrans.NetTranslation());
88  calHit->SetRot(aTrans.NetRotation());
89  G4int icell = CalCollection->insert( calHit );
90  CellID[copyIDinZ][copyIDinPhi] = icell - 1;
91  if(verboseLevel>0)
92  { G4cout << " New Calorimeter Hit on CellID "
93  << copyIDinZ << " " << copyIDinPhi << G4endl; }
94  }
95  else
96  {
97  (*CalCollection)[CellID[copyIDinZ][copyIDinPhi]]->AddEdep(edep);
98  if(verboseLevel>0)
99  { G4cout << " Energy added to CellID "
100  << copyIDinZ << " " << copyIDinPhi << G4endl; }
101  }
102 
103  return true;
104 }
