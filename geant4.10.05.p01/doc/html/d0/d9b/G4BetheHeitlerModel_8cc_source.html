99 // Calculates the microscopic cross section in GEANT4 internal units.
100 // A parametrized formula from L. Urban is used to estimate
101 // the total cross section.
102 // It gives a good description of the data from 1.5 MeV to 100 GeV.
103 // below 1.5 MeV: sigma=sigma(1.5MeV)*(GammaEnergy-2electronmass)
104 // *(GammaEnergy-2electronmass)
105 G4double
106 G4BetheHeitlerModel::ComputeCrossSectionPerAtom(const G4ParticleDefinition*,
107  G4double gammaEnergy, G4double Z,
108  G4double, G4double, G4double)
109 {
110  G4double xSection = 0.0 ;
111  // short versions
112  static const G4double kMC2 = CLHEP::electron_mass_c2;
113  // zero cross section below the kinematical limit: Eg<2mc^2
114  if (Z < 0.9 || gammaEnergy <= 2.0*kMC2) { return xSection; }
115  //
116  static const G4double gammaEnergyLimit = 1.5*CLHEP::MeV;
117  // set coefficients a, b c
118  static const G4double a0 = 8.7842e+2*CLHEP::microbarn;
119  static const G4double a1 = -1.9625e+3*CLHEP::microbarn;
120  static const G4double a2 = 1.2949e+3*CLHEP::microbarn;
121  static const G4double a3 = -2.0028e+2*CLHEP::microbarn;
122  static const G4double a4 = 1.2575e+1*CLHEP::microbarn;
123  static const G4double a5 = -2.8333e-1*CLHEP::microbarn;
124 
125  static const G4double b0 = -1.0342e+1*CLHEP::microbarn;
126  static const G4double b1 = 1.7692e+1*CLHEP::microbarn;
127  static const G4double b2 = -8.2381 *CLHEP::microbarn;
128  static const G4double b3 = 1.3063 *CLHEP::microbarn;
129  static const G4double b4 = -9.0815e-2*CLHEP::microbarn;
130  static const G4double b5 = 2.3586e-3*CLHEP::microbarn;
131 
132  static const G4double c0 = -4.5263e+2*CLHEP::microbarn;
133  static const G4double c1 = 1.1161e+3*CLHEP::microbarn;
134  static const G4double c2 = -8.6749e+2*CLHEP::microbarn;
135  static const G4double c3 = 2.1773e+2*CLHEP::microbarn;
136  static const G4double c4 = -2.0467e+1*CLHEP::microbarn;
137  static const G4double c5 = 6.5372e-1*CLHEP::microbarn;
138  // check low energy limit of the approximation (1.5 MeV)
139  G4double gammaEnergyOrg = gammaEnergy;
140  if (gammaEnergy < gammaEnergyLimit) { gammaEnergy = gammaEnergyLimit; }
141  // compute gamma energy variables
142  const G4double x = G4Log(gammaEnergy/kMC2);
143  const G4double x2 = x *x;
144  const G4double x3 = x2*x;
145  const G4double x4 = x3*x;
146  const G4double x5 = x4*x;
147  //
148  const G4double F1 = a0 + a1*x + a2*x2 + a3*x3 + a4*x4 + a5*x5;
149  const G4double F2 = b0 + b1*x + b2*x2 + b3*x3 + b4*x4 + b5*x5;
150  const G4double F3 = c0 + c1*x + c2*x2 + c3*x3 + c4*x4 + c5*x5;
151  // compute the approximated cross section
152  xSection = (Z + 1.)*(F1*Z + F2*Z*Z + F3);
153  // check if we are below the limit of the approximation and apply correction
154  if (gammaEnergyOrg < gammaEnergyLimit) {
155  const G4double dum = (gammaEnergyOrg-2.*kMC2)/(gammaEnergyLimit-2.*kMC2);
156  xSection *= dum*dum;
157  }
158  // make sure that the cross section is never negative
159  xSection = std::max(xSection, 0.);
160  return xSection;
161 }

163 // The secondaries e+e- energies are sampled using the Bethe - Heitler
164 // cross sections with Coulomb correction.
165 // A modified version of the random number techniques of Butcher & Messel
166 // is used (Nuc Phys 20(1960),15).
167 //
168 // GEANT4 internal units.
169 //
170 // Note 1 : Effects due to the breakdown of the Born approximation at
171 // low energy are ignored.
172 // Note 2 : The differential cross section implicitly takes account of
173 // pair creation in both nuclear and atomic electron fields.
174 // However triplet prodution is not generated.
175 void G4BetheHeitlerModel::SampleSecondaries(std::vector<G4DynamicParticle*>* fvect,
176  const G4MaterialCutsCouple* couple,
177  const G4DynamicParticle* aDynamicGamma,
178  G4double, G4double)
179 {
180  // set some constant values
181  const G4double gammaEnergy = aDynamicGamma->GetKineticEnergy();
182  const G4double eps0 = CLHEP::electron_mass_c2/gammaEnergy;
183  //
184  // check kinematical limit: gamma energy(Eg) must be at least 2 e- rest mass
185  if (eps0 > 0.5) { return; }
186  //
187  // select target element of the material (probs. are based on partial x-secs)
188  const G4Element* anElement = SelectRandomAtom(couple,fTheGamma,gammaEnergy);
189  //
190  // get the random engine
191  CLHEP::HepRandomEngine* rndmEngine = G4Random::getTheEngine();
192  //
193  // 'eps' is the total energy transferred to one of the e-/e+ pair in initial
194  // gamma energy units Eg. Since the corresponding DCS is symmetric on eps=0.5,
195  // the kinematical limits for eps0=mc^2/Eg <= eps <= 0.5
196  // 1. 'eps' is sampled uniformly on the [eps0, 0.5] inteval if Eg<Egsmall
197  // 2. otherwise, on the [eps_min, 0.5] interval according to the DCS (case 2.)
198  G4double eps;
199  // case 1.
200  static const G4double Egsmall = 2.*CLHEP::MeV;
201  if (gammaEnergy < Egsmall) {
202  eps = eps0 + (0.5-eps0)*rndmEngine->flat();
203  } else {
204  // case 2.
205  // get the Coulomb factor for the target element (Z) and gamma energy (Eg)
206  // F(Z) = 8*ln(Z)/3 if Eg <= 50 [MeV] => no Coulomb correction
207  // F(Z) = 8*ln(Z)/3 + 8*fc(Z) if Eg > 50 [MeV] => fc(Z) is the Coulomb cor.
208  //
209  // The screening variable 'delta(eps)' = 136*Z^{-1/3}*eps0/[eps(1-eps)]
210  // Due to the Coulomb correction, the DCS can go below zero even at
211  // kinematicaly allowed eps > eps0 values. In order to exclude this eps
212  // range with negative DCS, the minimum eps value will be set to eps_min =
213  // max[eps0, epsp] with epsp is the solution of SF(delta(epsp)) - F(Z)/2 = 0
214  // with SF being the screening function (SF1=SF2 at high value of delta).
215  // The solution is epsp = 0.5 - 0.5*sqrt[ 1 - 4*136*Z^{-1/3}eps0/deltap]
216  // with deltap = Exp[(42.038-F(Z))/8.29]-0.958. So the limits are:
217  // - when eps=eps_max = 0.5 => delta_min = 136*Z^{-1/3}*eps0/4
218  // - epsp = 0.5 - 0.5*sqrt[ 1 - delta_min/deltap]
219  // - and eps_min = max[eps0, epsp]
220  static const G4double midEnergy = 50.*CLHEP::MeV;
221  const G4int iZet = std::min(gMaxZet, anElement->GetZasInt());
222  const G4double deltaFactor = 136.*eps0/anElement->GetIonisation()->GetZ3();
223  G4double deltaMax = gElementData[iZet]->fDeltaMaxLow;
224  G4double FZ = 8.*anElement->GetIonisation()->GetlogZ3();
225  if (gammaEnergy > midEnergy) {
226  FZ += 8.*(anElement->GetfCoulomb());
227  deltaMax = gElementData[iZet]->fDeltaMaxHigh;
228  }
229  const G4double deltaMin = 4.*deltaFactor;
230  //
231  // compute the limits of eps
232  const G4double epsp = 0.5 - 0.5*std::sqrt(1. - deltaMin/deltaMax) ;
233  const G4double epsMin = std::max(eps0,epsp);
234  const G4double epsRange = 0.5 - epsMin;
235  //
236  // sample the energy rate (eps) of the created electron (or positron)
237  G4double F10, F20;
238  ScreenFunction12(deltaMin, F10, F20);
239  F10 -= FZ;
240  F20 -= FZ;
241  const G4double NormF1 = std::max(F10 * epsRange * epsRange, 0.);
242  const G4double NormF2 = std::max(1.5 * F20 , 0.);
243  const G4double NormCond = NormF1/(NormF1 + NormF2);
244  // we will need 3 uniform random number for each trial of sampling
245  G4double rndmv[3];
246  G4double greject = 0.;
247  do {
248  rndmEngine->flatArray(3, rndmv);
249  if (NormCond > rndmv[0]) {
250  eps = 0.5 - epsRange * fG4Calc->A13(rndmv[1]);
251  const G4double delta = deltaFactor/(eps*(1.-eps));
252  greject = (ScreenFunction1(delta)-FZ)/F10;
253  } else {
254  eps = epsMin + epsRange*rndmv[1];
255  const G4double delta = deltaFactor/(eps*(1.-eps));
256  greject = (ScreenFunction2(delta)-FZ)/F20;
257  }
258  // Loop checking, 03-Aug-2015, Vladimir Ivanchenko
259  } while (greject < rndmv[2]);
260  } // end of eps sampling
261  //
262  // select charges randomly
263  G4double eTotEnergy, pTotEnergy;
264  if (rndmEngine->flat() > 0.5) {
265  eTotEnergy = (1.-eps)*gammaEnergy;
266  pTotEnergy = eps*gammaEnergy;
267  } else {
268  pTotEnergy = (1.-eps)*gammaEnergy;
269  eTotEnergy = eps*gammaEnergy;
270  }
271  //
272  // sample pair kinematics
273  const G4double eKinEnergy = std::max(0.,eTotEnergy - CLHEP::electron_mass_c2);
274  const G4double pKinEnergy = std::max(0.,pTotEnergy - CLHEP::electron_mass_c2);
275  //
276  G4ThreeVector eDirection, pDirection;
277  //
278  GetAngularDistribution()->SamplePairDirections(aDynamicGamma,
279  eKinEnergy, pKinEnergy,
280  eDirection, pDirection);
281  // create G4DynamicParticle object for the particle1
282  G4DynamicParticle* aParticle1= new G4DynamicParticle(
283  fTheElectron,eDirection,eKinEnergy);
284  // create G4DynamicParticle object for the particle2
285  G4DynamicParticle* aParticle2= new G4DynamicParticle(
286  fThePositron,pDirection,pKinEnergy);
287  // Fill output vector
288  fvect->push_back(aParticle1);
289  fvect->push_back(aParticle2);
290  // kill incident photon
291  fParticleChange->SetProposedKineticEnergy(0.);
292  fParticleChange->ProposeTrackStatus(fStopAndKill);
293 }

295 // should be called only by the master and at initialisation
296 void G4BetheHeitlerModel::InitialiseElementData()
297 {
298  G4int size = gElementData.size();
299  if (size < gMaxZet+1) {
300  gElementData.resize(gMaxZet+1, nullptr);
301  }
302  // create for all elements that are in the detector
303  const G4ElementTable* elemTable = G4Element::GetElementTable();
304  size_t numElems = (*elemTable).size();
305  for (size_t ie = 0; ie < numElems; ++ie) {
306  const G4Element* elem = (*elemTable)[ie];
307  const G4int iz = std::min(gMaxZet, elem->GetZasInt());
308  if (!gElementData[iz]) { // create it if doesn't exist yet
309  G4double FZLow = 8.*elem->GetIonisation()->GetlogZ3();
310  G4double FZHigh = FZLow + 8.*elem->GetfCoulomb();
311  ElementData* elD = new ElementData();
312  elD->fDeltaMaxLow = G4Exp((42.038 - FZLow )/8.29) - 0.958;
313  elD->fDeltaMaxHigh = G4Exp((42.038 - FZHigh)/8.29) - 0.958;
314  gElementData[iz] = elD;
315  }
316  }
317 }
