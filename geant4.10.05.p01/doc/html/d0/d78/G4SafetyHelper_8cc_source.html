113 G4double G4SafetyHelper::ComputeSafety( const G4ThreeVector& position, G4double maxLength )
114 {
115  G4double newSafety;
116 
117  // Only recompute (calling Navigator/PathFinder) if 'position'
118  // is *not* the safety location and has moved 'significantly'
119  //
120  G4double moveLengthSq = (position-fLastSafetyPosition).mag2();
121  if( (moveLengthSq > 0.0 ) )
122  {
123  if( !fUseParallelGeometries )
124  {
125  // Safety for mass geometry
126  newSafety = fpMassNavigator->ComputeSafety(position, maxLength, true);
127 
128  // Only store a 'true' safety - one that was not restricted by maxLength
129  if( newSafety < maxLength )
130  {
131  fLastSafety= newSafety;
132  fLastSafetyPosition = position;
133  }
134  }
135  else
136  {
137  // Safety for all geometries
138  newSafety = fpPathFinder->ComputeSafety(position);
139 
140  fLastSafety= newSafety;
141  fLastSafetyPosition = position;
142  }
143 
144  }
145  else
146  {
147  // return last value if position is not (significantly) changed
148  //
149  // G4double moveLength = 0;
150  // if( moveLengthSq > 0.0 ) { moveLength= std::sqrt(moveLengthSq); }
151  newSafety = fLastSafety; // -moveLength;
152  }
153 
154  return newSafety;
155 }

157 void G4SafetyHelper::ReLocateWithinVolume( const G4ThreeVector &newPosition )
158 {
159  // G4cout << " G4SafetyHelper::ReLocateWithinVolume called at " << newPosition << G4endl;
160 #ifdef G4VERBOSE
161  if( fVerbose > 0 ) {
162  // There is an opportunity - and need - to check whether the proposed move is safe
163  G4ThreeVector moveVec= newPosition - fLastSafetyPosition;
164  if( moveVec.mag2() > sqr(fLastSafety) )
165  {
166  // A problem exists - we are proposing to move outside 'Safety Sphere'
167  G4ExceptionDescription ed;
168  ed << "Unsafe Move> Asked to relocate beyond 'Safety sphere'. Details: " << G4endl;
169  ed << " Safety Sphere: Radius = " << fLastSafety;
170  ed << " Center = " << fLastSafetyPosition << G4endl;
171  ed << " New Location : Move = " << moveVec.mag();
172  ed << " Position = " << newPosition << G4endl;
173  G4Exception("G4SafetyHelper::ReLocateWithinVolume", "GeomNav999", JustWarning, ed);
174  }
175  }
176 #endif
177 
178  if( !fUseParallelGeometries )
179  {
180  fpMassNavigator->LocateGlobalPointWithinVolume( newPosition );
181  }
182  else
183  {
184  fpPathFinder->ReLocate( newPosition );
185  }
186 }
