129 void HadronPhysicsCRMC_FTFP_BERT::Neutron() {
130  // General schema:
131  // 1) Create a builder
132  // 2) Call AddBuilder
133  // 3) Configure the builder, possibly with sub-builders
134  // 4) Call builder->Build()
135  auto neu = new G4NeutronBuilder;
136  AddBuilder( neu );
137  auto epos_n = new CRMCNeutronBuilder;
138  AddBuilder( epos_n );
139  epos_n->SetMinEnergy( minCRMC );
140  neu->RegisterMe( epos_n );
141  auto ftfp_n = new G4FTFPNeutronBuilder( true );
142  AddBuilder( ftfp_n );
143  ftfp_n->SetMinEnergy( minFTFP );
144  ftfp_n->SetMaxEnergy( maxFTFP );
145  neu->RegisterMe( ftfp_n );
146  auto bert_n = new G4BertiniNeutronBuilder;
147  AddBuilder( bert_n );
148  bert_n->SetMinEnergy( minBERT );
149  bert_n->SetMaxEnergy( maxBERT );
150  neu->RegisterMe( bert_n );
151  neu->Build();
152 }

233 void HadronPhysicsCRMC_FTFP_BERT::ExtraConfiguration() {
234  // Modify cross sections for kaons
235  auto xsk = new G4ComponentGGHadronNucleusXsc;
236  xs_k.Put( xsk );
237  G4VCrossSectionDataSet* kaonxs = new G4CrossSectionInelastic( xsk );
238  xs_ds.Push_back( kaonxs );
239  G4PhysListUtil::FindInelasticProcess( G4KaonMinus::KaonMinus() )->AddDataSet( kaonxs );
240  G4PhysListUtil::FindInelasticProcess( G4KaonPlus::KaonPlus() )->AddDataSet( kaonxs );
241  G4PhysListUtil::FindInelasticProcess( G4KaonZeroShort::KaonZeroShort() )->AddDataSet( kaonxs );
242  G4PhysListUtil::FindInelasticProcess( G4KaonZeroLong::KaonZeroLong() )->AddDataSet( kaonxs );
243  // Modify Neutrons
244  const G4ParticleDefinition* neutron = G4Neutron::Neutron();
245  G4HadronicProcess* inel = G4PhysListUtil::FindInelasticProcess( neutron );
246  if ( inel ) inel->AddDataSet( new G4NeutronInelasticXS );
247  G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess( neutron );
248  if ( capture ) capture->RegisterMe( new G4NeutronRadCapture );
249 }
