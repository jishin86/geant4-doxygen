69 G4bool B5DriftChamberSD::ProcessHits(G4Step* step, G4TouchableHistory*)
70 {
71  auto charge = step->GetTrack()->GetDefinition()->GetPDGCharge();
72  if (charge==0.) return true;
73 
74  auto preStepPoint = step->GetPreStepPoint();
75 
76  auto touchable = step->GetPreStepPoint()->GetTouchable();
77  auto motherPhysical = touchable->GetVolume(1); // mother
78  auto copyNo = motherPhysical->GetCopyNo();
79 
80  auto worldPos = preStepPoint->GetPosition();
81  auto localPos
82  = touchable->GetHistory()->GetTopTransform().TransformPoint(worldPos);
83 
84  auto hit = new B5DriftChamberHit(copyNo);
85  hit->SetWorldPos(worldPos);
86  hit->SetLocalPos(localPos);
87  hit->SetTime(preStepPoint->GetGlobalTime());
88 
89  fHitsCollection->insert(hit);
90 
91  return true;
92 }
