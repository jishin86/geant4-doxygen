128 void F04Trajectory::MergeTrajectory(G4VTrajectory* secondTrajectory)
129 {
130  if(!secondTrajectory) return;
131 
132  F04Trajectory* second = (F04Trajectory*)secondTrajectory;
133  G4int ent = second->GetPointEntries();
134  // initial point of the second trajectory should not be merged
135  for(G4int i=1; i<ent; ++i) {
136  fpPointsContainer->push_back((*(second->fpPointsContainer))[i]);
137  }
138  delete (*second->fpPointsContainer)[0];
139  second->fpPointsContainer->clear();
140 }
