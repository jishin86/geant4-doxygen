74 void G4IonElasticPhysics::ConstructProcess()
75 {
76  // Elastic process for other ions
77  G4HadronElasticProcess* ionElasticProcess = new G4HadronElasticProcess("ionElastic");
78 
79  //Model
80  G4NuclNuclDiffuseElastic* ionElastic = new G4NuclNuclDiffuseElastic;
81  ionElastic->SetMinEnergy(0.0);
82  ionElasticProcess->RegisterMe(ionElastic);
83 
84  //Cross Section
85  G4ComponentGGNuclNuclXsc* ionElasticXS = new G4ComponentGGNuclNuclXsc;
86  G4VCrossSectionDataSet* ionElasticXSDataSet = new G4CrossSectionElastic(ionElasticXS);
87  ionElasticXSDataSet->SetMinKinEnergy(0.0);
88  ionElasticProcess->AddDataSet(ionElasticXSDataSet);
89 
90  G4ProcessManager* ionManager = G4GenericIon::GenericIon()->GetProcessManager();
91  ionManager->AddDiscreteProcess( ionElasticProcess );
92 
93  if ( verbose > 1 ) {
94  G4cout << "### IonElasticPhysics: " << ionElasticProcess->GetProcessName()
95  << " added for " << G4GenericIon::GenericIon()->GetParticleName() << G4endl;
96  }
97 }
