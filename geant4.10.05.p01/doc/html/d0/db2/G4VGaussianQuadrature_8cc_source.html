75 G4double
76 G4VGaussianQuadrature::GammaLogarithm(G4double xx)
77 {
78 
79 // Returns the value ln(Gamma(xx) for xx > 0. Full accuracy is obtained for
80 // xx > 1. For 0 < xx < 1. the reflection formula (6.1.4) can be used first.
81 // (Adapted from Numerical Recipes in C)
82 
83  static const G4double cof[6] = { 76.18009172947146, -86.50532032941677,
84  24.01409824083091, -1.231739572450155,
85  0.1208650973866179e-2, -0.5395239384953e-5 };
86  G4double x = xx - 1.0;
87  G4double tmp = x + 5.5;
88  tmp -= (x + 0.5) * std::log(tmp);
89  G4double ser = 1.000000000190015;
90 
91  for ( size_t j = 0; j <= 5; j++ )
92  {
93  x += 1.0;
94  ser += cof[j]/x;
95  }
96  return -tmp + std::log(2.5066282746310005*ser);
97 }
