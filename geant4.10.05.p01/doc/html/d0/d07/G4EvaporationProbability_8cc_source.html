88 G4double G4EvaporationProbability::TotalProbability(
89  const G4Fragment & fragment, G4double minEnergy, G4double maxEnergy,
90  G4double CoulombBarrier)
91 {
92  fragA = fragment.GetA_asInt();
93  fragZ = fragment.GetZ_asInt();
94  resA = fragA - theA;
95  resZ = fragZ - theZ;
96 
97  G4double fragMass = fragment.GetGroundStateMass();
98  U = fragment.GetExcitationEnergy();
99  Mass = fragMass + U;
100  delta0 = std::max(0.0, fPairCorr->GetPairingCorrection(fragA,fragZ));
101  delta1 = std::max(0.0, fPairCorr->GetPairingCorrection(resA,resZ));
102  //delta0 = fLevelData->GetPairingCorrection(fragZ,fragA);
103  //delta1 = fLevelData->GetPairingCorrection(resZ,resA);
104  resMass = G4NucleiProperties::GetNuclearMass(resA, resZ);
105  resA13 = fG4pow->Z13(resA);
106  a0 = fLevelData->GetLevelDensity(fragZ,fragA,U);
107  /*
108  G4cout << "G4EvaporationProbability: Z= " << theZ << " A= " << theA
109  << " resZ= " << resZ << " resA= " << resA
110  << " fragZ= " << fragZ << " fragA= " << fragA
111  << "\n U= " << U << " d0= " << delta0 << " a0= " << a0
112  << " OPT= " << OPTxs << G4endl;
113  */
114  if(U < delta0 || maxEnergy <= minEnergy) { return 0.0; }
115 
116  G4double Width = 0.0;
117  if (OPTxs==0 || (OPTxs==4 && U < 10. + delta0)) {
118 
119  G4double SystemEntropy = 2.0*std::sqrt(a0*(U-delta0));
120 
121  static const G4double RN2 =
122  2.25*fermi*fermi/(twopi* hbar_Planck*hbar_Planck);
123 
124  G4double Alpha = CalcAlphaParam(fragment);
125  G4double Beta = CalcBetaParam(fragment);
126 
127  // to be checked where to use a0, where - a1
128  G4double a1 = fLevelData->GetLevelDensity(resZ,resA,U);
129  G4double GlobalFactor = Gamma*Alpha*partMass*RN2*resA13*resA13/(a1*a1);
130 
131  G4double maxea = maxEnergy*a1;
132  G4double Term1 = Beta*a1 - 1.5 + maxea;
133  G4double Term2 = (2.0*Beta*a1-3.0)*std::sqrt(maxea) + 2*maxea;
134 
135  G4double ExpTerm1 = (SystemEntropy <= explim) ? G4Exp(-SystemEntropy) : 0.0;
136 
137  G4double ExpTerm2 = 2.*std::sqrt(maxea) - SystemEntropy;
138  ExpTerm2 = std::min(ExpTerm2, explim);
139  ExpTerm2 = G4Exp(ExpTerm2);
140 
141  Width = GlobalFactor*(Term1*ExpTerm1 + Term2*ExpTerm2);
142 
143  } else {
144 
145  // compute power once
146  if(0 < index) {
147  muu = G4KalbachCrossSection::ComputePowerParameter(resA, index);
148  }
149  // if Coulomb barrier cutoff is superimposed for all cross sections
150  // then the limit is the Coulomb Barrier
151  Width = IntegrateProbability(minEnergy, maxEnergy, CoulombBarrier);
152  }
153  return Width;
154 }

156 G4double G4EvaporationProbability::ComputeProbability(G4double K, G4double cb)
157 {
158  //G4cout << "### G4EvaporationProbability::ProbabilityDistributionFunction"
159  // << G4endl;
160 
161  G4double E0 = U - delta0;
162  //G4double E1 = Mass - partMass - resMass - delta1 - K;
163  G4double E1 = std::sqrt((Mass - partMass)*(Mass - partMass) - 2*Mass*K)
164  - resMass - delta1;
165  /*
166  G4cout << "PDF: FragZ= " << fragZ << " FragA= " << fragA
167  << " Z= " << theZ << " A= " << theA
168  << " K= " << K << " E0= " << E0 << " E1= " << E1 << G4endl;
169  */
170  if(E1 < 0.0) { return 0.0; }
171 
172  G4double a1 = fLevelData->GetLevelDensity(resZ,resA,U);
173  G4double xs = CrossSection(K, cb);
174  G4double prob = pcoeff*G4Exp(2.0*(std::sqrt(a1*E1) - std::sqrt(a0*E0)))*K*xs;
175  /*
176  G4cout << "Evap prob: " << prob << " FragZ= " << fragZ << " FragA= " << fragA
177  << " Z= " << theZ << " A= " << theA
178  << "\n K= " << K << " E0= " << E0 << " E1= " << E1
179  << " cb= " << cb << " xsec= " << xs << G4endl;
180  */
181  return prob;
182 }

199 G4double
200 G4EvaporationProbability::SampleKineticEnergy(G4double minKinEnergy,
201  G4double maxKinEnergy,
202  G4double)
203 {
204  /*
205  G4cout << "### Sample probability Emin= " << minKinEnergy
206  << " Emax= " << maxKinEnergy
207  << " Z= " << theZ << " A= " << theA << G4endl;
208  */
209  G4double T = 0.0;
210  CLHEP::HepRandomEngine* rndm = G4Random::getTheEngine();
211  static const G4int nmax = 100;
212  if (OPTxs==0 || (OPTxs==4 && U < 10. + delta0)) {
213  // JMQ:
214  // It uses Dostrovsky's approximation for the inverse reaction cross
215  // in the probability for fragment emission
216  // MaximalKineticEnergy energy in the original version (V.Lara) was
217  // calculated at the Coulomb barrier.
218 
219  G4double Rb = 4.0*a0*maxKinEnergy;
220  G4double RbSqrt = std::sqrt(Rb);
221  G4double PEX1 = (RbSqrt < explim) ? G4Exp(-RbSqrt) : 0.0;
222  G4double Rk = 0.0;
223  G4double FRk = 0.0;
224  G4int nn = 0;
225  do {
226  G4double RandNumber = rndm->flat();
227  Rk = 1.0 + (1./RbSqrt)*G4Log(RandNumber + (1.0-RandNumber)*PEX1);
228  G4double Q1 = 1.0;
229  G4double Q2 = 1.0;
230  if (theZ == 0) { // for emitted neutron
231  G4double Beta = (2.12/(resA13*resA13) - 0.05)*MeV/(0.76 + 2.2/resA13);
232  Q1 = 1.0 + Beta/maxKinEnergy;
233  Q2 = Q1*std::sqrt(Q1);
234  }
235 
236  static const G4double ssqr3 = 1.5*std::sqrt(3.0);
237  FRk = ssqr3 * Rk * (Q1 - Rk*Rk)/Q2;
238  if(nn > nmax) { break; }
239  ++nn;
240  // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
241  } while (FRk < rndm->flat());
242 
243  T = std::max(maxKinEnergy * (1.0-Rk*Rk), 0.0) + minKinEnergy;
244 
245  } else {
246 
247  if(fVerbose > 1) {
248  G4cout << "###=== SampleEnergy: " << " Z= " << theZ
249  << " A= " << theA << " FragZ= " << fragZ
250  << " FragA= " << fragA << G4endl;
251  }
252  T = SampleEnergy();
253 
254  }
255  //G4cout << "-- new Z= " << theZ << " A= " << theA << " ekin= " << T << G4endl;
256  return T;
257  //return fLevelData->FindLevel(resZ, resA, resMass, Mass, partMass, T);
258 }
