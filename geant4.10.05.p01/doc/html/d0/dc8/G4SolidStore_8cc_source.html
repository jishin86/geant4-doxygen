68 // ***************************************************************************
69 // Delete all elements from the store
70 // ***************************************************************************
71 //
72 void G4SolidStore::Clean()
73 {
74  // Do nothing if geometry is closed
75  //
76  if (G4GeometryManager::IsGeometryClosed())
77  {
78  G4cout << "WARNING - Attempt to delete the solid store"
79  << " while geometry closed !" << G4endl;
80  return;
81  }
82 
83  // Locks store for deletion of solids. De-registration will be
84  // performed at this stage. G4VSolids will not de-register themselves.
85  //
86  locked = true;
87 
88  size_t i=0;
89  G4SolidStore* store = GetInstance();
90 
91 #ifdef G4GEOMETRY_VOXELDEBUG
92  G4cout << "Deleting Solids ... ";
93 #endif
94 
95  for(iterator pos=store->begin(); pos!=store->end(); pos++)
96  {
97  if (fgNotifier) { fgNotifier->NotifyDeRegistration(); }
98  if (*pos) { delete *pos; }
99  i++;
100  }
101 
102 #ifdef G4GEOMETRY_VOXELDEBUG
103  if (store->size() < i-1)
104  { G4cout << "No solids deleted. Already deleted by user ?" << G4endl; }
105  else
106  { G4cout << i-1 << " solids deleted !" << G4endl; }
107 #endif
108 
109  locked = false;
110  store->clear();
111 }

133 // ***************************************************************************
134 // Remove Solid from container
135 // ***************************************************************************
136 //
137 void G4SolidStore::DeRegister(G4VSolid* pSolid)
138 {
139  if (!locked) // Do not de-register if locked !
140  {
141  if (fgNotifier) { fgNotifier->NotifyDeRegistration(); }
142  for (reverse_iterator i=GetInstance()->rbegin();
143  i!=GetInstance()->rend(); i++)
144  {
145  if (**i==*pSolid)
146  {
147  GetInstance()->erase(std::next(i).base());
148  break;
149  }
150  }
151  }
152 }
