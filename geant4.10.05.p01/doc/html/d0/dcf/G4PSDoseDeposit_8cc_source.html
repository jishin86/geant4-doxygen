61 G4bool G4PSDoseDeposit::ProcessHits(G4Step* aStep,G4TouchableHistory*)
62 {
63  G4double edep = aStep->GetTotalEnergyDeposit();
64  if ( edep == 0. ) return FALSE;
65 
66  G4int idx = ((G4TouchableHistory*)
67  (aStep->GetPreStepPoint()->GetTouchable()))
68  ->GetReplicaNumber(indexDepth);
69  G4double cubicVolume = ComputeVolume(aStep, idx);
70 
71 
72  G4double density = aStep->GetTrack()->GetStep()->GetPreStepPoint()->GetMaterial()->GetDensity();
73  G4double dose = edep / ( density * cubicVolume );
74  dose *= aStep->GetPreStepPoint()->GetWeight();
75  G4int index = GetIndex(aStep);
76  EvtMap->add(index,dose);
77  return TRUE;
78 }

119 G4double G4PSDoseDeposit::ComputeVolume(G4Step* aStep, G4int idx){
120 
121  G4VPhysicalVolume* physVol = aStep->GetPreStepPoint()->GetPhysicalVolume();
122  G4VPVParameterisation* physParam = physVol->GetParameterisation();
123  G4VSolid* solid = 0;
124  if(physParam)
125  { // for parameterized volume
126  if(idx<0)
127  {
128  G4ExceptionDescription ED;
129  ED << "Incorrect replica number --- GetReplicaNumber : " << idx << G4endl;
130  G4Exception("G4PSDoseDeposit::ComputeVolume","DetPS0004",JustWarning,ED);
131  }
132  solid = physParam->ComputeSolid(idx, physVol);
133  solid->ComputeDimensions(physParam,idx,physVol);
134  }
135  else
136  { // for ordinary volume
137  solid = physVol->GetLogicalVolume()->GetSolid();
138  }
139 
140  return solid->GetCubicVolume();
141 }
