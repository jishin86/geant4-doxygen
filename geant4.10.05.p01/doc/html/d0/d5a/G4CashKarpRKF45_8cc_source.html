28 // The Cash-Karp Runge-Kutta-Fehlberg 4/5 method is an embedded fourth
29 // order method (giving fifth-order accuracy) for the solution of an ODE.
30 // Two different fourth order estimates are calculated; their difference
31 // gives an error estimate. [ref. Numerical Recipes in C, 2nd Edition]
32 // It is used to integrate the equations of the motion of a particle
33 // in a magnetic field.
34 //
35 // [ref. Numerical Recipes in C, 2nd Edition]

110 //
111 // Given values for n = 6 variables yIn[0,...,n-1]
112 // known at x, use the fifth-order Cash-Karp Runge-
113 // Kutta-Fehlberg-4-5 method to advance the solution over an interval
114 // Step and return the incremented variables as yOut[0,...,n-1]. Also
115 // return an estimate of the local truncation error yErr[] using the
116 // embedded 4th-order method. The user supplies routine
117 // RightHandSide(y,dydx), which returns derivatives dydx for y .
118 
119 void
120 G4CashKarpRKF45::Stepper(const G4double yInput[],
121  const G4double dydx[],
122  G4double Step,
123  G4double yOut[],
124  G4double yErr[])
125 {
126  // const G4int nvar = 6 ;
127  // const G4double a2 = 0.2 , a3 = 0.3 , a4 = 0.6 , a5 = 1.0 , a6 = 0.875;
128  G4int i;
129 
130  const G4double b21 = 0.2 ,
131  b31 = 3.0/40.0 , b32 = 9.0/40.0 ,
132  b41 = 0.3 , b42 = -0.9 , b43 = 1.2 ,
133 
134  b51 = -11.0/54.0 , b52 = 2.5 , b53 = -70.0/27.0 ,
135  b54 = 35.0/27.0 ,
136 
137  b61 = 1631.0/55296.0 , b62 = 175.0/512.0 ,
138  b63 = 575.0/13824.0 , b64 = 44275.0/110592.0 ,
139  b65 = 253.0/4096.0 ,
140 
141  c1 = 37.0/378.0 , c3 = 250.0/621.0 , c4 = 125.0/594.0 ,
142  c6 = 512.0/1771.0 ,
143  dc5 = -277.0/14336.0 ;
144 
145  const G4double dc1 = c1 - 2825.0/27648.0 , dc3 = c3 - 18575.0/48384.0 ,
146  dc4 = c4 - 13525.0/55296.0 , dc6 = c6 - 0.25 ;
147 
148  // Initialise time to t0, needed when it is not updated by the integration.
149  // [ Note: Only for time dependent fields (usually electric)
150  // is it neccessary to integrate the time.]
151  yOut[7] = yTemp[7] = yIn[7] = yInput[7];
152 
153  const G4int numberOfVariables= this->GetNumberOfVariables();
154  // The number of variables to be integrated over
155 
156  // Saving yInput because yInput and yOut can be aliases for same array
157 
158  for(i=0;i<numberOfVariables;i++)
159  {
160  yIn[i]=yInput[i];
161  }
162  // RightHandSide(yIn, dydx) ; // 1st Step
163 
164  for(i=0;i<numberOfVariables;i++)
165  {
166  yTemp[i] = yIn[i] + b21*Step*dydx[i] ;
167  }
168  RightHandSide(yTemp, ak2) ; // 2nd Step
169 
170  for(i=0;i<numberOfVariables;i++)
171  {
172  yTemp[i] = yIn[i] + Step*(b31*dydx[i] + b32*ak2[i]) ;
173  }
174  RightHandSide(yTemp, ak3) ; // 3rd Step
175 
176  for(i=0;i<numberOfVariables;i++)
177  {
178  yTemp[i] = yIn[i] + Step*(b41*dydx[i] + b42*ak2[i] + b43*ak3[i]) ;
179  }
180  RightHandSide(yTemp, ak4) ; // 4th Step
181 
182  for(i=0;i<numberOfVariables;i++)
183  {
184  yTemp[i] = yIn[i] + Step*(b51*dydx[i] + b52*ak2[i] + b53*ak3[i] +
185  b54*ak4[i]) ;
186  }
187  RightHandSide(yTemp, ak5) ; // 5th Step
188 
189  for(i=0;i<numberOfVariables;i++)
190  {
191  yTemp[i] = yIn[i] + Step*(b61*dydx[i] + b62*ak2[i] + b63*ak3[i] +
192  b64*ak4[i] + b65*ak5[i]) ;
193  }
194  RightHandSide(yTemp, ak6) ; // 6th Step
195 
196  for(i=0;i<numberOfVariables;i++)
197  {
198  // Accumulate increments with proper weights
199 
200  yOut[i] = yIn[i] + Step*(c1*dydx[i] + c3*ak3[i] + c4*ak4[i] + c6*ak6[i]) ;
201 
202  // Estimate error as difference between 4th and
203  // 5th order methods
204 
205  yErr[i] = Step*(dc1*dydx[i] + dc3*ak3[i] + dc4*ak4[i] +
206  dc5*ak5[i] + dc6*ak6[i]) ;
207 
208  // Store Input and Final values, for possible use in calculating chord
209  fLastInitialVector[i] = yIn[i] ;
210  fLastFinalVector[i] = yOut[i];
211  fLastDyDx[i] = dydx[i];
212  }
213  // NormaliseTangentVector( yOut ); // Not wanted
214 
215  fLastStepLength =Step;
216 
217  return ;
218 }

239 G4double G4CashKarpRKF45::DistChord() const
240 {
241  G4double distLine, distChord;
242  G4ThreeVector initialPoint, finalPoint, midPoint;
243 
244  // Store last initial and final points (they will be overwritten in self-Stepper call!)
245  initialPoint = G4ThreeVector( fLastInitialVector[0],
246  fLastInitialVector[1], fLastInitialVector[2]);
247  finalPoint = G4ThreeVector( fLastFinalVector[0],
248  fLastFinalVector[1], fLastFinalVector[2]);
249 
250  // Do half a step using StepNoErr
251 
252  fAuxStepper->Stepper( fLastInitialVector, fLastDyDx, 0.5 * fLastStepLength,
253  fMidVector, fMidError );
254 
255  midPoint = G4ThreeVector( fMidVector[0], fMidVector[1], fMidVector[2]);
256 
257  // Use stored values of Initial and Endpoint + new Midpoint to evaluate
258  // distance of Chord
259 
260 
261  if (initialPoint != finalPoint)
262  {
263  distLine = G4LineSection::Distline( midPoint, initialPoint, finalPoint );
264  distChord = distLine;
265  }
266  else
267  {
268  distChord = (midPoint-initialPoint).mag();
269  }
270  return distChord;
271 }
