53 G4bool VoxelSD::ProcessHits(G4Step* astep, G4TouchableHistory*)
54 {

57 
58  G4int tid = astep-> GetTrack()-> GetTrackID();

60 
61  Analysis* myana = Analysis::GetAnalysis();
62 
63  // incident position
64  if ( tid == 1 && iz == 1000 ) { // scored @ first layer
65  myana-> FillIncident(x0);
66  }
67 
68  // energy deposit
69  G4ThreeVector p = G4UniformRand()*(x1-x0) + x0; // position sampling
70  G4double edep = astep-> GetTotalEnergyDeposit();
71  myana-> FillDose(p, edep);
72 
73  return true;
74 }
