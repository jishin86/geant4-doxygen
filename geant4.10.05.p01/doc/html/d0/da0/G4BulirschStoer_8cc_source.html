50 G4BulirschStoer::G4BulirschStoer(G4EquationOfMotion* equation,
51  G4int nvar, G4double eps_rel, G4double max_dt)
52  : fnvar(nvar), m_eps_rel(eps_rel), m_midpoint(equation,nvar),
53  m_last_step_rejected(false), m_first(true), m_dt_last(0.0), m_max_dt(max_dt)
54 {
55  /* initialize sequence of stage numbers and work */
56 
57  for(G4int i = 0; i < m_k_max + 1; ++i)
58  {
59  m_interval_sequence[i] = 2 * (i + 1);
60  if (i == 0)
61  {
62  m_cost[i] = m_interval_sequence[i];
63  }
64  else
65  {
66  m_cost[i] = m_cost[i-1] + m_interval_sequence[i];
67  }
68  for(G4int k = 0; k < i; ++k)
69  {
70  const G4double r = static_cast<G4double>(m_interval_sequence[i])
71  / static_cast<G4double>(m_interval_sequence[k]);
72  m_coeff[i][k] = 1.0 / (r * r - 1.0); // coefficients for extrapolation
73  }
74 
75  // crude estimate of optimal order
76  m_current_k_opt = 4;
77 
78  // no calculation because log10 might not exist for value_type!
79 
80  //const G4double logfact = -log10(std::max(eps_rel, 1.0e-12)) * 0.6 + 0.5;
81  //m_current_k_opt = std::max(1.,
82  // std::min(static_cast<G4double>(m_k_max-1), logfact));
83  }
84 }

86 G4BulirschStoer::step_result
87 G4BulirschStoer::try_step( const G4double in[], const G4double dxdt[],
88  G4double& t, G4double out[], G4double& dt)
89 {
90  if(m_max_dt < dt)
91  {
92  // given step size is bigger then max_dt set limit and return fail
93  //
94  dt = m_max_dt;
95  return step_result::fail;
96  }
97 
98  if (dt != m_dt_last)
99  {
100  reset(); // step size changed from outside -> reset
101  }
102 
103  G4bool reject = true;
104 
105  G4double new_h = dt;
106 
107  /* m_current_k_opt is the estimated current optimal stage number */
108 
109  for(G4int k = 0; k <= m_current_k_opt+1; ++k)
110  {
111  // the stage counts are stored in m_interval_sequence
112  //
113  m_midpoint.SetSteps(m_interval_sequence[k]);
114  if(k == 0)
115  {
116  m_midpoint.DoStep(in, dxdt, out, dt);
117  /* the first step, nothing more to do */
118  }
119  else
120  {
121  m_midpoint.DoStep(in, dxdt, m_table[k-1], dt);
122  extrapolate(k, out);
123  // get error estimate
124  for (G4int i = 0; i < fnvar; ++i)
125  {
126  m_err[i] = out[i] - m_table[0][i];
127  }
128  const G4double error =
129  field_utils::relativeError(out, m_err, dt, m_eps_rel);
130  h_opt[k] = calc_h_opt(dt, error, k);
131  work[k] = static_cast<G4double>(m_cost[k]) / h_opt[k];
132 
133  if( (k == m_current_k_opt-1) || m_first) // convergence before k_opt ?
134  {
135  if(error < 1.0)
136  {
137  // convergence
138  reject = false;
139  if( (work[k] < KFAC2 * work[k-1]) || (m_current_k_opt <= 2) )
140  {
141  // leave order as is (except we were in first round)
142  m_current_k_opt = std::min(m_k_max - 1 , std::max(2 , k + 1));
143  new_h = h_opt[k];
144  new_h *= static_cast<G4double>(m_cost[k + 1])
145  / static_cast<G4double>(m_cost[k]);
146  }
147  else
148  {
149  m_current_k_opt = std::min(m_k_max - 1, std::max(2, k));
150  new_h = h_opt[k];
151  }
152  break;
153  }
154  else if(should_reject(error , k) && !m_first)
155  {
156  reject = true;
157  new_h = h_opt[k];
158  break;
159  }
160  }
161  if(k == m_current_k_opt) // convergence at k_opt ?
162  {
163  if(error < 1.0)
164  {
165  // convergence
166  reject = false;
167  if(work[k-1] < KFAC2 * work[k])
168  {
169  m_current_k_opt = std::max( 2 , m_current_k_opt-1 );
170  new_h = h_opt[m_current_k_opt];
171  }
172  else if( (work[k] < KFAC2 * work[k-1]) && !m_last_step_rejected )
173  {
174  m_current_k_opt = std::min(m_k_max - 1, m_current_k_opt + 1);
175  new_h = h_opt[k];
176  new_h *= static_cast<G4double>(m_cost[m_current_k_opt])
177  / static_cast<G4double>(m_cost[k]);
178  }
179  else
180  {
181  new_h = h_opt[m_current_k_opt];
182  }
183  break;
184  }
185  else if(should_reject(error, k))
186  {
187  reject = true;
188  new_h = h_opt[m_current_k_opt];
189  break;
190  }
191  }
192  if(k == m_current_k_opt + 1) // convergence at k_opt+1 ?
193  {
194  if(error < 1.0) // convergence
195  {
196  reject = false;
197  if(work[k-2] < KFAC2 * work[k-1])
198  {
199  m_current_k_opt = std::max(2, m_current_k_opt - 1);
200  }
201  if((work[k] < KFAC2 * work[m_current_k_opt]) && !m_last_step_rejected)
202  {
203  m_current_k_opt = std::min(m_k_max - 1 , k);
204  }
205  new_h = h_opt[m_current_k_opt];
206  }
207  else
208  {
209  reject = true;
210  new_h = h_opt[m_current_k_opt];
211  }
212  break;
213  }
214  }
215  }
216 
217  if(!reject)
218  {
219  t += dt;
220  }
221 
222  if(!m_last_step_rejected || new_h < dt)
223  {
224  // limit step size
225  new_h = std::min(m_max_dt, new_h);
226  m_dt_last = new_h;
227  dt = new_h;
228  }
229 
230  m_last_step_rejected = reject;
231  m_first = false;
232 
233  return reject ? step_result::fail : step_result::success;
234 }

242 void G4BulirschStoer::extrapolate(size_t k , G4double xest[])
243 {
244  /* polynomial extrapolation, see http://www.nr.com/webnotes/nr3web21.pdf
245  * uses the obtained intermediate results to extrapolate to dt->0 */
246 
247  for(G4int j = k - 1 ; j > 0; --j)
248  {
249  for (G4int i = 0; i < fnvar; ++i)
250  {
251  m_table[j-1][i] = m_table[j][i] * (1. + m_coeff[k][j])
252  - m_table[j-1][i] * m_coeff[k][j];
253  }
254  }
255  for (G4int i = 0; i < fnvar; ++i)
256  {
257  xest[i] = m_table[0][i] * (1. + m_coeff[k][0]) - xest[i] * m_coeff[k][0];
258  }
259 }

261 G4double
262 G4BulirschStoer::calc_h_opt(G4double h , G4double error , size_t k) const
263 {
264  /* calculates the optimal step size for a given error and stage number */
265 
266  const G4double expo = 1.0 / (2 * k + 1);
267  const G4double facmin = std::pow(STEPFAC3, expo);
268  G4double fac;
269 
270  G4double facminInv= 1.0 / facmin;
271  if (error == 0.0)
272  {
273  fac = facminInv;
274  }
275  else
276  {
277  fac = STEPFAC2 * std::pow(error * inv_STEPFAC1 , -expo);
278  fac = std::max(facmin * inv_STEPFAC4, std::min( facminInv, fac));
279  }
280 
281  return h * fac;
282 }

285 G4bool G4BulirschStoer::set_k_opt(size_t k, G4double& dt)
286 {
287  /* calculates the optimal stage number */
288 
289  if(k == 1)
290  {
291  m_current_k_opt = 2;
292  return true;
293  }
294  if( (work[k-1] < KFAC1 * work[k]) || (k == m_k_max) ) // order decrease
295  {
296  m_current_k_opt = k - 1;
297  dt = h_opt[ m_current_k_opt ];
298  return true;
299  }
300  else if( (work[k] < KFAC2 * work[k-1])
301  || m_last_step_rejected || (k == m_k_max-1) )
302  { // same order - also do this if last step got rejected
303  m_current_k_opt = k;
304  dt = h_opt[m_current_k_opt];
305  return true;
306  }
307  else { // order increase - only if last step was not rejected
308  m_current_k_opt = k + 1;
309  dt = h_opt[m_current_k_opt - 1] * m_cost[m_current_k_opt]
310  / m_cost[m_current_k_opt - 1];
311  return true;
312  }

325 G4bool G4BulirschStoer::should_reject(G4double error, G4int k) const
326 {
327  if(k == m_current_k_opt - 1)
328  {
329  const G4double d = G4double(m_interval_sequence[m_current_k_opt]
330  * m_interval_sequence[m_current_k_opt+1]);
331  const G4double e = G4double(m_interval_sequence[0]);
332  const G4double e2 = e*e;
333  // step will fail, criterion 17.3.17 in NR
334  return error * e2 * e2 > d * d; // was return error > dOld * dOld; (where dOld= d/e; )
335  }
336  else if(k == m_current_k_opt)
337  {
338  const G4double d = G4double(m_interval_sequence[m_current_k_opt]);
339  const G4double e = G4double(m_interval_sequence[0]);
340  return error * e * e > d * d; // was return error > dOld * dOld; (where dOld= d/e; )
341  }
342  else
343  {
344  return error > 1.0;
345  }
