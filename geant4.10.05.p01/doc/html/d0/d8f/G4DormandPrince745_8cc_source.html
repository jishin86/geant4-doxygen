181 void G4DormandPrince745::Stepper(const G4double yInput[],
182  const G4double DyDx[],
183  G4double Step,
184  G4double yOut[],
185  G4double yErr[] )
186 {
187  G4int i;
188 
189  //The various constants defined on the basis of butcher tableu
190  const G4double //G4double - only once
191  b21 = 0.2 ,
192 
193  b31 = 3.0/40.0, b32 = 9.0/40.0 ,
194 
195  b41 = 44.0/45.0, b42 = -56.0/15.0, b43 = 32.0/9.0,
196 
197  b51 = 19372.0/6561.0, b52 = -25360.0/2187.0, b53 = 64448.0/6561.0,
198  b54 = -212.0/729.0 ,
199 
200  b61 = 9017.0/3168.0 , b62 = -355.0/33.0,
201  b63 = 46732.0/5247.0 , b64 = 49.0/176.0 ,
202  b65 = -5103.0/18656.0 ,
203 
204  b71 = 35.0/384.0, b72 = 0.,
205  b73 = 500.0/1113.0, b74 = 125.0/192.0,
206  b75 = -2187.0/6784.0, b76 = 11.0/84.0,
207 
208  //Sum of columns, sum(bij) = ei
209 // e1 = 0. ,
210 // e2 = 1.0/5.0 ,
211 // e3 = 3.0/10.0 ,
212 // e4 = 4.0/5.0 ,
213 // e5 = 8.0/9.0 ,
214 // e6 = 1.0 ,
215 // e7 = 1.0 ,
216 
217 // Difference between the higher and the lower order method coeff. :
218  // b7j are the coefficients of higher order
219 
220  dc1 = -( b71 - 5179.0/57600.0),
221  dc2 = -( b72 - .0),
222  dc3 = -( b73 - 7571.0/16695.0),
223  dc4 = -( b74 - 393.0/640.0),
224  dc5 = -( b75 + 92097.0/339200.0),
225  dc6 = -( b76 - 187.0/2100.0),
226  dc7 = -( - 1.0/40.0 ); //end of declaration
227 
228  const G4int numberOfVariables= this->GetNumberOfVariables();
229 
230  // The number of variables to be integrated over
231  yOut[7] = yTemp[7] = yInput[7];
232  // Saving yInput because yInput and yOut can be aliases for same array
233 
234  for(i=0;i<numberOfVariables;i++)
235  {
236  yIn[i]=yInput[i];
237  }
238 
239  // RightHandSide(yIn, DyDx) ;
240  // 1st Step - Not doing, getting passed
241 
242  for(i=0;i<numberOfVariables;i++)
243  {
244  yTemp[i] = yIn[i] + b21*Step*DyDx[i] ;
245  }
246  RightHandSide(yTemp, ak2) ; // 2nd stage
247 
248  for(i=0;i<numberOfVariables;i++)
249  {
250  yTemp[i] = yIn[i] + Step*(b31*DyDx[i] + b32*ak2[i]) ;
251  }
252  RightHandSide(yTemp, ak3) ; // 3rd stage
253 
254  for(i=0;i<numberOfVariables;i++)
255  {
256  yTemp[i] = yIn[i] + Step*(b41*DyDx[i] + b42*ak2[i] + b43*ak3[i]) ;
257  }
258  RightHandSide(yTemp, ak4) ; // 4th stage
259 
260  for(i=0;i<numberOfVariables;i++)
261  {
262  yTemp[i] = yIn[i] + Step*(b51*DyDx[i] + b52*ak2[i] + b53*ak3[i] +
263  b54*ak4[i]) ;
264  }
265  RightHandSide(yTemp, ak5) ; // 5th stage
266 
267  for(i=0;i<numberOfVariables;i++)
268  {
269  yTemp[i] = yIn[i] + Step*(b61*DyDx[i] + b62*ak2[i] + b63*ak3[i] +
270  b64*ak4[i] + b65*ak5[i]) ;
271  }
272  RightHandSide(yTemp, ak6) ; // 6th stage
273 
274  for(i=0;i<numberOfVariables;i++)
275  {
276  yOut[i] = yIn[i] + Step*(b71*DyDx[i] + b72*ak2[i] + b73*ak3[i] +
277  b74*ak4[i] + b75*ak5[i] + b76*ak6[i] );
278  }
279  RightHandSide(yOut, ak7); //7th and Final stage
280 
281  for(i=0;i<numberOfVariables;i++)
282  {
283  yErr[i] = Step*(dc1*DyDx[i] + dc2*ak2[i] + dc3*ak3[i] + dc4*ak4[i] +
284  dc5*ak5[i] + dc6*ak6[i] + dc7*ak7[i] ) + 1.5e-18 ;
285 
286  // Store Input and Final values, for possible use in calculating chord
287  fLastInitialVector[i] = yIn[i] ;
288  fLastFinalVector[i] = yOut[i];
289  fInitialDyDx[i] = DyDx[i];
290  }
291 
292  fLastStepLength = Step;
293 
294  return ;
295 }

298 // Calculate DistChord given start, mid and end-point of step
299 G4double G4DormandPrince745::DistLine( G4double yStart[], G4double yMid[], G4double yEnd[] ) const
300 {
301  G4double distLine, distChord;
302  G4ThreeVector initialPoint, finalPoint, midPoint;
303 
304  initialPoint = G4ThreeVector( yStart[0], yStart[1], yStart[2]);
305  finalPoint = G4ThreeVector( yEnd[0], yEnd[1], yEnd[2]);
306  midPoint = G4ThreeVector( yMid[0], yMid[1], yMid[2]);
307 
308  // Use stored values of Initial and Endpoint + new Midpoint to evaluate
309  // distance of Chord
310  if (initialPoint != finalPoint)
311  {
312  distLine = G4LineSection::Distline( midPoint, initialPoint, finalPoint );
313  distChord = distLine;
314  }
315  else
316  {
317  distChord = (midPoint-initialPoint).mag();
318  }
319  return distChord;
320 }

336 G4double G4DormandPrince745::DistChord() const
337 {
338  // Coefficients were taken from Some Practical Runge-Kutta Formulas by Lawrence F. Shampine, page 149, c*
339  const G4double
340  hf1 = 6025192743.0 / 30085553152.0,
341  hf2 = 0.0,
342  hf3 = 51252292925.0 / 65400821598.0,
343  hf4 = - 2691868925.0 / 45128329728.0,
344  hf5 = 187940372067.0 / 1594534317056.0,
345  hf6 = - 1776094331.0 / 19743644256.0,
346  hf7 = 11237099.0 / 235043384.0;
347 
348  for(G4int i = 0; i < 3; ++i) {
349  fMidVector[i] = fLastInitialVector[i] + 0.5 * fLastStepLength *
350  (hf1 * fInitialDyDx[i] + hf2 * ak2[i] + hf3 * ak3[i] +
351  hf4 * ak4[i] + hf5 * ak5[i] + hf6 * ak6[i] + hf7 * ak7[i]);
352  }
353 
354  // Use stored values of Initial and Endpoint + new Midpoint to evaluate
355  // distance of Chord
356  return DistLine( fLastInitialVector, fMidVector, fLastFinalVector);
357 }
