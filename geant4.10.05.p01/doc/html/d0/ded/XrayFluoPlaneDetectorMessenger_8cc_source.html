49 XrayFluoPlaneDetectorMessenger::XrayFluoPlaneDetectorMessenger(XrayFluoPlaneDetectorConstruction * Det)

55  UpdateCmd = new G4UIcmdWithoutParameter("/apparate/update",this);

59 
60  UpdateCmd->AvailableForStates(G4State_Idle);
61

63  sampleCmd->SetGuidance("select a diferent material for the plane");
64  sampleCmd->SetParameterName("material",true);
65  sampleCmd->SetDefaultValue("mars1");
66  sampleCmd->SetCandidates("Dolorite Anorthosite Mars1 IceBasalt");
67  sampleCmd->AvailableForStates(G4State_Idle);
68 
69  detectorCmd = new G4UIcmdWithAString("/apparate/detector",this);
70  detectorCmd->SetGuidance("select a diferent detectorType");
71  detectorCmd->SetParameterName("detector",true);
72  detectorCmd->SetDefaultValue("sili");
73  detectorCmd->SetCandidates("sili hpge");
74  detectorCmd->AvailableForStates(G4State_Idle);
75 
76  grainDiaCmd = new G4UIcmdWithADoubleAndUnit( "/apparate/GrainDiameter",this );
77  grainDiaCmd->SetGuidance( "Set diameter of grains" );

104 void XrayFluoPlaneDetectorMessenger::SetNewValue(G4UIcommand* command,G4String newValue)
105 {
106  if( command == UpdateCmd )
107  {
108  //This triggers a full re-build of the geometry. The method in the
109  //geometry will take care of that.
110  Detector->UpdateGeometry();
111  return;
112  }
113  else if ( command == sampleCmd )
114  {
115  Detector->SetPlaneMaterial(newValue);
116  }
117  else if ( command == detectorCmd )
118  {
119  Detector->SetDetectorType(newValue);
120  }
121  else if ( command == grainDiaCmd )
122  {
123  G4double newSize = grainDiaCmd->GetNewDoubleValue(newValue);
124  Detector->SetGrainDia(newSize);
125  }
126  else if ( command == granularityFlagCmd )
127  {
128  Detector->DeleteGrainObjects();
129  G4bool newGranFlag = granularityFlagCmd->GetNewBoolValue(newValue);
130  Detector->SetPlaneGranularity(newGranFlag);
131  }
132  //Notify the run manager that the geometry has been modified
133  G4RunManager::GetRunManager()->GeometryHasBeenModified();
134 
135 }
