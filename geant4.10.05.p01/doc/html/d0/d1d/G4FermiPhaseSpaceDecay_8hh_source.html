78 G4FermiPhaseSpaceDecay::PtwoBody(G4double E, G4double P1, G4double P2) const
79 {
80  G4double P = (E+P1+P2)*(E+P1-P2)*(E-P1+P2)*(E-P1-P2)/(4.0*E*E);
