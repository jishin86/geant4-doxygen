91 G4bool G4DNAMolecularReaction::TestReactibility(const G4Track& trackA,
92  const G4Track& trackB,
93  const double currentStepTime,
94  const double /*previousStepTime*/,
95  bool userStepTimeLimit) /*const*/
96 {
97  G4MolecularConfiguration* moleculeA =
98  GetMolecule(trackA)->GetMolecularConfiguration();
99  G4MolecularConfiguration* moleculeB =
100  GetMolecule(trackB)->GetMolecularConfiguration();
101 
102  if (!fReactionModel)
103  {
104  G4ExceptionDescription exceptionDescription(
105  "You have to give a reaction model to the molecular reaction process");
106  G4Exception("G4DNAMolecularReaction::TestReactibility",
107  "MolecularReaction001", FatalErrorInArgument,
108  exceptionDescription);
109  return false; // makes coverity happy
110  }
111  if (fMolReactionTable == 0)
112  {
113  G4ExceptionDescription exceptionDescription(
114  "You have to give a reaction table to the molecular reaction process");
115  G4Exception("G4DNAMolecularReaction::TestReactibility",
116  "MolecularReaction002", FatalErrorInArgument,
117  exceptionDescription);
118  return false; // makes coverity happy
119  }
120 
121  // Retrieve reaction range
122  fReactionRadius = -1; // reaction Range
123  fReactionRadius = fReactionModel->GetReactionRadius(moleculeA, moleculeB);
124 
125  fDistance = -1; // separation distance
126 
127  if (currentStepTime == 0.)
128  {
129  userStepTimeLimit = false;
130  }
131 
132  G4bool output = fReactionModel->FindReaction(trackA, trackB, fReactionRadius,
133  fDistance, userStepTimeLimit);
134 
135 #ifdef G4VERBOSE
136  // DEBUG
137  if (fVerbose > 1)
138  {
139  G4cout << "\033[1;39;36m" << "G4MolecularReaction " << G4endl;
140  G4cout << "FindReaction returned : " << G4BestUnit(output,"Length") << G4endl;
141  G4cout << " reaction radius : " << G4BestUnit(fReactionRadius,"Length")
142  << " real distance : " << G4BestUnit((trackA.GetPosition() - trackB.GetPosition()).mag(), "Length")
143  << " calculated distance by model (= -1 if real distance > reaction radius and the user limitation step is not reached) : "
144  << G4BestUnit(fDistance,"Length")
145  << G4endl;
146 
147  G4cout << "TrackID A : " << trackA.GetTrackID()
148  << ", TrackID B : " << trackB.GetTrackID()
149  << " | MolA " << moleculeA->GetName()
150  << ", MolB " << moleculeB->GetName()
151  <<"\033[0m\n"
152  << G4endl;
153  G4cout << "--------------------------------------------" << G4endl;
154  }
155 #endif
156  return output;
157 }

159 G4ITReactionChange* G4DNAMolecularReaction::MakeReaction(const G4Track& trackA,
160  const G4Track& trackB)
161 {
162  fpChanges = new G4ITReactionChange();
163  fpChanges->Initialize(trackA, trackB);
164 
165  G4MolecularConfiguration* moleculeA =
166  GetMolecule(trackA)->GetMolecularConfiguration();
167  G4MolecularConfiguration* moleculeB =
168  GetMolecule(trackB)->GetMolecularConfiguration();
169 
170 #ifdef G4VERBOSE
171  // DEBUG
172  if (fVerbose)
173  {
174  G4cout << "G4DNAMolecularReaction::MakeReaction" << G4endl;
175  G4cout<<"TrackA n°" << trackA.GetTrackID()
176  <<"\t | Track B n°" << trackB.GetTrackID() << G4endl;
177 
178  G4cout<<"Track A : Position : " << G4BestUnit(trackA.GetPosition(),"Length")
179  <<"\t Global Time : " << G4BestUnit(trackA.GetGlobalTime(), "Time")<< G4endl;
180 
181  G4cout<<"Track B : Position : " << G4BestUnit(trackB.GetPosition() ,"Length")
182  <<"\t Global Time : " << G4BestUnit(trackB.GetGlobalTime(), "Time")<< G4endl;
183 
184  G4cout<<"Reaction range : " << G4BestUnit(fReactionRadius,"Length")
185  << " \t Separation distance : " << G4BestUnit(fDistance,"Length")<< G4endl;
186  G4cout << "--------------------------------------------" << G4endl;
187  }
188 #endif
189 
190  const G4DNAMolecularReactionData* reactionData = fMolReactionTable
191  ->GetReactionData(moleculeA, moleculeB);
192 
193  G4int nbProducts = reactionData->GetNbProducts();
194 
195  if (nbProducts)
196  {
197  G4double D1 = moleculeA->GetDiffusionCoefficient();
198  G4double D2 = moleculeB->GetDiffusionCoefficient();
199  G4double sqrD1 = sqrt(D1);
200  G4double sqrD2 = sqrt(D2);
201  G4double numerator = sqrD1 + sqrD2;
202  G4ThreeVector reactionSite = sqrD1 / numerator * trackA.GetPosition()
203  + sqrD2 / numerator * trackB.GetPosition();
204 
205  for (G4int j = 0; j < nbProducts; j++)
206  {
207  G4Molecule* product = new G4Molecule(reactionData->GetProduct(j));
208  G4Track* productTrack = product->BuildTrack(trackA.GetGlobalTime(),
209  reactionSite);
210 
211 // G4cout << ">> G4DNAMolecularReaction::MakeReaction " << G4endl
212 // << "\t track A ("<< trackA.GetTrackID() <<"): " << trackA.GetGlobalTime() << G4endl
213 // << "\t track B ("<< trackB.GetTrackID() <<"): " << trackB.GetGlobalTime() << G4endl;
214 
215  productTrack->SetTrackStatus(fAlive);
216 
217  fpChanges->AddSecondary(productTrack);
218  G4MoleculeFinder::Instance()->Push(productTrack);
219  }
220  }
221 
222  fpChanges->KillParents(true);
223 
224  return fpChanges;
225 }
