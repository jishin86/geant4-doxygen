85 G4double
86 G4UPiNuclearCrossSection::GetElasticCrossSection(const G4DynamicParticle* dp,
87  G4int Z, G4int A)
88 {
89  G4double cross = 0.0;
90  const G4ParticleDefinition* part = dp->GetDefinition();
91  G4PhysicsTable* table = (part == piPlus) ? piPlusElastic : piMinusElastic;
92  if(table) {
93  cross = Interpolate(Z, A, dp->GetKineticEnergy(),table);
94  }
95  return cross;
96 }

98 G4double
99 G4UPiNuclearCrossSection::GetInelasticCrossSection(const G4DynamicParticle* dp,
100  G4int Z, G4int A)
101 {
102  G4double cross = 0.0;
103  G4double fact = 1.0;
104  G4double ekin = dp->GetKineticEnergy();
105  G4PhysicsTable* table = 0;
106  const G4ParticleDefinition* part = dp->GetDefinition();
107 
108  // Coulomb barrier
109  if(part == piPlus) {
110  if(ekin > elowest) {
111  table = piPlusInelastic;
112  if(ekin < elow) {
113  fact = std::sqrt((ekin-elowest)/(elow-elowest));
114  ekin = elow;
115  }
116  }
117  } else if(part == piMinus) {
118  table = piMinusInelastic;
119  if(ekin < elow) { ekin = elow; }
120  }
121  if(table) {
122  cross = fact*Interpolate(Z, A, ekin, table);
123  }
124  return cross;
125 }

127 G4double G4UPiNuclearCrossSection::Interpolate(
128  G4int Z, G4int A, G4double ekin, G4PhysicsTable* table)
129 {
130  G4double res = 0.0;
131  G4int idx;
132  G4int iz = Z;
133  if(iz > 92) iz = 92;
134  for(idx=0; idx<NZ; idx++) {if(theZ[idx] >= iz) break;}
135  if(idx >= NZ) idx = NZ - 1;
136  G4int iz2 = theZ[idx];
137  // G4cout << "U: iz= " << iz << " iz2= " << iz2 << " "
138  // << APower[iz] << " " << APower[iz2]<<G4endl;
139  G4double x2 = (((*table)[idx])->Value(ekin))*APower[iz]/APower[iz2];
140 
141  // use only one Z
142  if(iz >= theZ[idx] || idx == 0) {
143  res = x2;
144 
145  // Interpolation between Z
146  } else {
147 
148  G4int iz1 = theZ[idx-1];
149  G4double x1 = (((*table)[idx-1])->Value(ekin))*APower[iz]/APower[iz1];
150  G4double w1 = G4double(A) - theA[idx-1];
151  G4double w2 = theA[idx] - G4double(A);
152  res = (w1*x2 + w2*x1)/(w1 + w2);
153  }
154  return res;
155 }

157 void G4UPiNuclearCrossSection::AddDataSet(const G4String& p,
158  const G4double* tot,
159  const G4double* in,
160  const G4double* e,
161  G4int n)
162 {
163  G4LPhysicsFreeVector* pvin = new G4LPhysicsFreeVector(n,e[0]*GeV,e[n-1]*GeV);
164  //pvin->SetSpline(true);
165  G4LPhysicsFreeVector* pvel = new G4LPhysicsFreeVector(n,e[0]*GeV,e[n-1]*GeV);
166  //pvel->SetSpline(true);
167  for(G4int i=0; i<n; ++i) {
168  pvin->PutValues(i,e[i]*GeV,in[i]*millibarn);
169  pvel->PutValues(i,e[i]*GeV,std::max(0.0,(tot[i]-in[i])*millibarn));
170  }
171  if(p == "pi+") {
172  piPlusInelastic->push_back(pvin);
173  piPlusElastic->push_back(pvel);
174  } else {
175  piMinusInelastic->push_back(pvin);
176  piMinusElastic->push_back(pvel);
177  }
178 }
