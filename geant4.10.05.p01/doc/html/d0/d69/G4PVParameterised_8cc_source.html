149 // ----------------------------------------------------------------------
150 // CheckOverlaps
151 //
152 G4bool
153 G4PVParameterised::CheckOverlaps(G4int res, G4double tol,
154  G4bool verbose, G4int maxErr)
155 {
156  if (res<=0) { return false; }
157 
158  G4int trials = 0;
159  G4bool retval = false;
160  G4VSolid *solidA = 0, *solidB = 0;
161  G4LogicalVolume *motherLog = GetMotherLogical();
162  G4VSolid *motherSolid = motherLog->GetSolid();
163  std::vector<G4ThreeVector> points;
164 
165  if (verbose)
166  {
167  G4cout << "Checking overlaps for parameterised volume "
168  << GetName() << " ... ";
169  }
170 
171  for (G4int i=0; i<GetMultiplicity(); i++)
172  {
173  solidA = fparam->ComputeSolid(i, this);
174  solidA->ComputeDimensions(fparam, i, this);
175  fparam->ComputeTransformation(i, this);
176 
177  // Create the transformation from daughter to mother
178  //
179  G4AffineTransform Tm( GetRotation(), GetTranslation() );
180 
181  // Generate random points on surface according to the given resolution,
182  // transform them to the mother's coordinate system and if no overlaps
183  // with the mother volume, cache them in a vector for later use with
184  // the daughters
185  //
186  for (G4int n=0; n<res; n++)
187  {
188  G4ThreeVector mp = Tm.TransformPoint(solidA->GetPointOnSurface());
189 
190  // Checking overlaps with the mother volume
191  //
192  if (motherSolid->Inside(mp)==kOutside)
193  {
194  G4double distin = motherSolid->DistanceToIn(mp);
195  if (distin > tol)
196  {
197  trials++; retval = true;
198  std::ostringstream message;
199  message << "Overlap with mother volume !" << G4endl
200  << " Overlap is detected for volume "
201  << GetName() << ", parameterised instance: " << i << G4endl
202  << " with its mother volume "
203  << motherLog->GetName() << G4endl
204  << " at mother local point " << mp << ", "
205  << "overlapping by at least: "
206  << G4BestUnit(distin, "Length");
207  if (trials>=maxErr)
208  {
209  message << G4endl
210  << "NOTE: Reached maximum fixed number -" << maxErr
211  << "- of overlaps reports for this volume !";
212  }
213  G4Exception("G4PVParameterised::CheckOverlaps()",
214  "GeomVol1002", JustWarning, message);
215  if (trials>=maxErr) { return true; }
216  }
217  }
218  points.push_back(mp);
219  }
220 
221  // Checking overlaps with each other parameterised instance
222  //
223  std::vector<G4ThreeVector>::iterator pos;
224  for (G4int j=i+1; j<GetMultiplicity(); j++)
225  {
226  solidB = fparam->ComputeSolid(j,this);
227  solidB->ComputeDimensions(fparam, j, this);
228  fparam->ComputeTransformation(j, this);
229 
230  // Create the transformation for daughter volume
231  //
232  G4AffineTransform Td( GetRotation(), GetTranslation() );
233 
234  for (pos=points.begin(); pos!=points.end(); pos++)
235  {
236  // Transform each point according to daughter's frame
237  //
238  G4ThreeVector md = Td.InverseTransformPoint(*pos);
239 
240  if (solidB->Inside(md)==kInside)
241  {
242  G4double distout = solidB->DistanceToOut(md);
243  if (distout > tol)
244  {
245  trials++; retval = true;
246  std::ostringstream message;
247  message << "Overlap within parameterised volumes !" << G4endl
248  << " Overlap is detected for volume "
249  << GetName() << ", parameterised instance: " << i << G4endl
250  << " with parameterised volume instance: " << j
251  << G4endl
252  << " at local point " << md << ", "
253  << "overlapping by at least: "
254  << G4BestUnit(distout, "Length")
255  << ", related to volume instance: " << j << ".";
256  if (trials>=maxErr)
257  {
258  message << G4endl
259  << "NOTE: Reached maximum fixed number -" << maxErr
260  << "- of overlaps reports for this volume !";
261  }
262  G4Exception("G4PVParameterised::CheckOverlaps()",
263  "GeomVol1002", JustWarning, message);
264  if (trials>=maxErr) { return true; }
265  }
266  }
267  }
268  }
269  }
270  if (verbose)
271  {
272  G4cout << "OK! " << G4endl;
273  }
274 
275  return retval;
276 }
