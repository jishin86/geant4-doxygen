43 // Class Description:
44 //
45 // Generates the conversion of a high-energy photon to an e+e- pair, either in the field of an
46 // atomic electron (triplet) or nucleus (nuclear).
47 // Samples the five-dimensional (5D) differential cross-section analytical expression:
48 // . Non polarized conversion:
49 // H.A. Bethe, W. Heitler, Proc. R. Soc. Lond. Ser. A 146 (1934) 83.
50 // . Polarized conversion:
51 // T. H. Berlin and L. Madansky, Phys. Rev. 78 (1950) 623,
52 // M. M. May, Phys. Rev. 84 (1951) 265,
53 // J. M. Jauch and F. Rohrlich, The theory of photons and electrons, 1976.
54 //
55 // All the above expressions are named "Bethe-Heitler" here.
56 //
57 // Bethe & Heitler, put in Feynman diagram parlance, compute only the two dominant diagrams of
58 // the first order Born development, which is an excellent approximation for nuclear conversion
59 // and for high-energy triplet conversion.
60 //
61 // Only the linear polarisation of the incoming photon takes part in these expressions.
62 // The circular polarisation of the incoming photon does not (take part) and no polarisation
63 // is transfered to the final leptons.
64 //
65 // In case conversion takes place in the field of an isolated nucleus or electron, the bare
66 // Bethe-Heitler expression is used.
67 //
68 // In case the nucleus or the electron are part of an atom, the screening of the target field
69 // by the other electrons of the atom is described by a simple form factor, function of q2:
70 // . nuclear: N.F. Mott, H.S.W. Massey, The Theory of Atomic Collisions, 1934.
71 // . triplet: J.A. Wheeler and W.E. Lamb, Phys. Rev. 55 (1939) 858.
72 //
73 // The nuclear form factor that affects the probability of very large-q2 events, is not considered.
74 //
75 // In principle the code is valid from threshold, that is from 2 * m_e c^2 for nuclear and from
76 // 4 * m_e c^2 for triplet, up to infinity, while in pratice the divergence of the differential
77 // cross section at small q2 and, at high-energy, at small polar angle, make it break down at
78 // some point that depends on machine precision.
79 //
80 // Very-high-energy (above a few tens of TeV) LPM suppression effects in the normalized differential
81 // cross-section are not considered.
82 //
83 // The 5D differential cross section is sampled without any high-energy nor small
84 // angle approximation(s).
85 // The generation is strictly energy-momentum conserving when all particles in the final state
86 // are taken into account, that is, including the recoiling target.
87 // (In contrast with the BH expressions taken at face values, for which the electron energy is
88 // taken to be EMinus = GammaEnergy - EPlus)
89 //
90 // Tests include the examination of 1D distributions: see TestEm15
91 //
92 // Total cross sections are not computed (we inherit from other classes).
93 // We just convert a photon on a target when asked to do so.
94 //
95 // Pure nuclear, pure triplet and 1/Z triplet/nuclear mixture can be generated.
96 //

165 void
166 G4BetheHeitler5DModel::BoostG4LorentzVector(const G4LorentzVector& p,
167  const G4LorentzVector& q,
168  G4LorentzVector& res) const
169 {
170 // p : 4-vector which will be boosted
171 // q : 4-vector of new origin in the old coordinates
172  const G4double pq = p.x()*q.x() + p.y()*q.y() + p.z()*q.z();
173  const G4double qq = q.x()*q.x() + q.y()*q.y() + q.z()*q.z();
174  const G4double mass2 = q.t()*q.t()-qq;
175  if ( mass2 > 0.0 ) {
176  const G4double mass = std::sqrt(q.t()*q.t()-qq);
177  const G4double lf = ((q.t()-mass)*pq/qq+p.t())/mass;
178  res.set( (p.x()+q.x()*lf), (p.y()+q.y()*lf), (p.z()+q.z()*lf),
179  ((p.t()*q.t()+pq)/mass) );
180  } else {
181  res = p;
182  if ( fVerbose > 3 ) {
183  G4cout << "G4BetheHeitler5DModel::BoostG4LorentzVector Warning point not converted"
184  << G4endl << "secondary in " << p
185  << G4endl << "Pair1 " << q << G4endl;
186  }
187  }
188 }

190 // assuming that q.x=q.y=0.0
191 void
192 G4BetheHeitler5DModel::BoostG4LorentzVector(const G4LorentzVector& p,
193  const G4double qz,
194  const G4double qt,
195  const G4double lffac,
196  const G4double imass,
197  G4LorentzVector& res) const
198 {
199 // p : 4-vector which will be boosted
200 // q : 4-vector of new origin in the old coordinates
201  const G4double pq = p.z()*qz;
202  const G4double lf = (lffac*pq+p.t())*imass;
203  res.setZ(p.z()+qz*lf);
204  res.setT((p.t()*qt+pq)*imass);
205 }

209 G4double G4BetheHeitler5DModel::MaxDiffCrossSection(const G4double* par,
210  G4double Z,
211  G4double e,
212  G4double loge) const
213 {
214  const G4double Q = e/par[9];
215  return par[0] * G4Exp((par[2]+loge*par[4])*loge)
216  / (par[1]+ G4Exp(par[3]*loge)+G4Exp(par[5]*loge))
217  * (1+par[7]*G4Exp(par[8]*G4Log(Z))*Q/(1+Q));
218 }

222 void
223 G4BetheHeitler5DModel::SampleSecondaries(std::vector<G4DynamicParticle*>* fvect,
224  const G4MaterialCutsCouple* couple,
225  const G4DynamicParticle* aDynamicGamma,
226  G4double, G4double)
227 {
228  // MeV
229  static const G4double ElectronMass = CLHEP::electron_mass_c2;
230  static const G4double ElectronMass2 = ElectronMass*ElectronMass;
231  static const G4double alpha0 = CLHEP::fine_structure_const;
232  // mm
233  static const G4double r0 = CLHEP::classic_electr_radius;
234  // mbarn
235  static const G4double r02 = r0*r0*1.e+25;
236  static const G4double twoPi = CLHEP::twopi;
237  static const G4double factor = alpha0 * r02 / (twoPi*twoPi);
238  // static const G4double factor1 = pow((6.0 * pi),(1.0/3.0))/(8.*alpha0*ElectronMass);
239  static const G4double factor1 = 2.66134007899/(8.*alpha0*ElectronMass);
240  //
241  static const G4double PairInvMassMin = 2.*ElectronMass;
242  //
243  static const G4double nu[10] = { 0.0227436, 0.0582046, 3.0322675, 2.8275065,
244  -0.0034004, 1.1212766, 1.8989468, 68.3492750,
245  0.0211186, 14.4 };
246  static const G4double tr[10] = { 0.0332350, 4.3942537, 2.8515925, 2.6351695,
247  -0.0031510, 1.5737305, 1.8104647, 20.6434021,
248  -0.0272586, 28.9};
249  //
250  static const G4double para[3][2] = { {11., -16.},{-1.17, -2.95},{-2., -0.5} };
251  //
252  static const G4double correctionIndex = 1.4;
253  //
254  const G4double GammaEnergy = aDynamicGamma->GetKineticEnergy();
255  const G4double GammaEnergy2 = GammaEnergy*GammaEnergy;
256  // Will not be true tot cross section = 0
257  if ( GammaEnergy <= 2.0*ElectronMass) { return; }
258  //
259  const G4ParticleMomentum GammaDirection = aDynamicGamma->GetMomentumDirection();
260  G4ThreeVector GammaPolarization = aDynamicGamma->GetPolarization();
261 
262  // The protection polarization perpendicular to the direction vector,
263  // as it done in G4LivermorePolarizedGammaConversionModel,
264  // assuming Direction is unitary vector
265  // (projection to plane) p_proj = p - (p o d)/(d o d) x d
266  if ( GammaPolarization.howOrthogonal(GammaDirection) != 0) {
267  GammaPolarization -= GammaPolarization.dot(GammaDirection) * GammaDirection;
268  }
269  // End of Protection
270  //
271  const G4double GammaPolarizationMag = GammaPolarization.mag();
273  // target element
274  // select randomly one element constituting the material
275  const G4Element* anElement = SelectRandomAtom(couple, fTheGamma, GammaEnergy);
276  // Atomic number
277  const G4int Z = anElement->GetZasInt();
278  const G4int A = SelectIsotopeNumber(anElement);
279  const G4double iZ13 = 1./anElement->GetIonisation()->GetZ3();
280  const G4double targetMass = G4NucleiProperties::GetNuclearMass(A, Z);
281 
282  CLHEP::HepRandomEngine* rndmEngine = G4Random::getTheEngine();
283 
284  // itriplet : true -- triplet, false -- nuclear.
285  G4bool itriplet = false;
286  if (fConversionType == 1) {
287  itriplet = false;
288  } else if (fConversionType == 2) {
289  itriplet = true;
290  if ( GammaEnergy <= 4.0*ElectronMass ) return;
291  } else if ( GammaEnergy > 4.0*ElectronMass ) {
292  // choose triplet or nuclear from a triplet/nuclear=1/Z
293  // total cross section ratio.
294  // approximate at low energies !
295  if(rndmEngine->flat()*(Z+1) < 1.) {
296  itriplet = true;
297  }
298  }
299  //
300  const G4double RecoilMass = itriplet ? ElectronMass : targetMass;
301  const G4double RecoilMass2 = RecoilMass*RecoilMass;
302  const G4double sCMS = 2.*RecoilMass*GammaEnergy + RecoilMass2;
303  const G4double sCMSPlusRM2 = sCMS + RecoilMass2;
304  const G4double sqrts = std::sqrt(sCMS);
305  const G4double isqrts2 = 1./(2.*sqrts);
306  //
307  const G4double PairInvMassMax = sqrts-RecoilMass;
308  const G4double PairInvMassRange = PairInvMassMax/PairInvMassMin;
309  const G4double lnPairInvMassRange = G4Log(PairInvMassRange);
310 
311  // initial state. Defines z axis of "0" frame as along photon propagation.
312  // create 4-vectors: gamma0 + target0 and CMS=gamma0+target0
313  // Since CMS(0., 0., GammaEnergy, GammaEnergy+RecoilMass) set some constants
314  // for the special boost that makes use of the form of CMS 4-vector
315  const G4double CMSqz = GammaEnergy;
316  const G4double CMSt = GammaEnergy+RecoilMass;
317  const G4double iCMSmass = 1./std::sqrt(RecoilMass*(RecoilMass+2.*GammaEnergy));
318  const G4double CMSfact = (CMSt-1./iCMSmass)/(CMSqz*CMSqz);
319  // maximum value of pdf
320  const G4double EffectiveZ = iraw ? 0.5 : Z;
321  const G4double Threshold = itriplet ? 4.*ElectronMass : 2.*ElectronMass;
322  const G4double AvailableEnergy = GammaEnergy - Threshold;
323  const G4double LogAvailableEnergy = G4Log(AvailableEnergy);
324  //
325  const G4double MaxDiffCross = itriplet
326  ? MaxDiffCrossSection(tr, EffectiveZ, AvailableEnergy, LogAvailableEnergy)
327  : MaxDiffCrossSection(nu, EffectiveZ, AvailableEnergy, LogAvailableEnergy);
328  //
329  // 50% safety marging factor
330  const G4double ymax = 1.5 * MaxDiffCross;
331  // x1 bounds
332  const G4double xu1 = (LogAvailableEnergy > para[2][0])
333  ? para[0][0] + para[1][0]*LogAvailableEnergy
334  : para[0][0] + para[2][0]*para[1][0];
335  const G4double xl1 = (LogAvailableEnergy > para[2][1])
336  ? para[0][1] + para[1][1]*LogAvailableEnergy
337  : para[0][1] + para[2][1]*para[1][1];
338  //
339  G4LorentzVector Recoil0;
340  G4LorentzVector Positron0;
341  G4LorentzVector Electron0;
342  G4double pdf = 0.;
343  // START Sampling
344  do {
346  // pdf pow(x,c) with c = 1.4
347  // integral y = pow(x,(c+1))/(c+1) @ x = 1 => y = 1 /(1+c)
348  // invCdf exp( log(y /* *( c + 1.0 )/ (c + 1.0 ) */ ) /( c + 1.0) )
350  const G4double X1 =
351  G4Exp(G4Log(rndmEngine->flat())/(correctionIndex + 1.0));
352 
353  const G4double x0 = G4Exp(xl1 + (xu1 - xl1)*rndmEngine->flat());
354  const G4double dum0 = 1./(1.+x0);
355  const G4double cosTheta = (x0-1.)*dum0;
356  const G4double sinTheta = std::sqrt(4.*x0)*dum0;
357 
358  const G4double PairInvMass = PairInvMassMin*G4Exp(X1*X1*lnPairInvMassRange);
359 
360  G4double rndmv3[3];
361  rndmEngine->flatArray(3, rndmv3);
362  //--------------------------------------------------------------------------
363  // const G4double ThetaLept = pi*rndmv3[0];
364  // const G4double cosThetaLept = std::cos(ThetaLept);
365  // const G4double sinThetaLept = std::sin(ThetaLept);
366  //
367  // const G4double PhiLept = twoPi*rndmv3[1]-pi;
368  // const G4double cosPhiLept = std::cos(PhiLept);
369  // const G4double sinPhiLept = std::sin(PhiLept);
370  //
371  // const G4double Phi = twoPi*rndmv3[2]-pi;
372  // const G4double cosPhi = std::cos(Phi);
373  // const G4double sinPhi = std::sin(Phi);
374  //---------------------------------------------------------------------------
375  // cos and sin theta-lepton
376  const G4double cosThetaLept = std::cos(pi*rndmv3[0]);
377  // sin(ThetaLept) is always in [0,+1] if ThetaLept is in [0,pi]
378  const G4double sinThetaLept = std::sqrt((1.-cosThetaLept)*(1.+cosThetaLept));
379  // cos and sin phi-lepton
380  const G4double cosPhiLept = std::cos(twoPi*rndmv3[1]-pi);
381  // sin(PhiLept) is in [-1,0] if PhiLept in [-pi,0) and
382  // is in [0,+1] if PhiLept in [0,+pi]
383  const G4double sinPhiLept = std::copysign(std::sqrt((1.-cosPhiLept)*(1.+cosPhiLept)),rndmv3[1]-0.5);
384  // cos and sin phi
385  const G4double cosPhi = std::cos(twoPi*rndmv3[2]-pi);
386  const G4double sinPhi = std::copysign(std::sqrt((1.-cosPhi)*(1.+cosPhi)),rndmv3[2]-0.5);
387 
388  // frames:
389  // 0 : the laboratory Lorentz frame, axes along photon direction and polarisation
390  // 1 : the center-of-mass Lorentz frame
391  // 2 : the pair Lorentz frame
392  // 3 : the laboratory Lorentz frame, Geant4 axes definition
393 
394  // in the center-of-mass frame
395  const G4double RecEnergyCMS = (sCMSPlusRM2-PairInvMass*PairInvMass)*isqrts2;
396  const G4double LeptonEnergy2 = PairInvMass*0.5;
397  // Denis ** correction
398  // const G4double thePRecoil = std::sqrt( (RecEnergyCMS-RecoilMass)
399  // *(RecEnergyCMS+RecoilMass));
400  const G4double ap1 = 2.0*GammaEnergy*RecoilMass -
401  PairInvMass*PairInvMass + 2.0*PairInvMass*RecoilMass;
402  const G4double bp1 = 2.0*GammaEnergy*RecoilMass -
403  PairInvMass*PairInvMass - 2.0*PairInvMass*RecoilMass;
404 
405  if (bp1 <= 0.0 ) {
406  if ( fVerbose > 3 ) {
407  G4cout
408  << "G4BetheHeitler5DModel::SampleSecondaries Warning bp1 "
409  << bp1 << "point rejected" << G4endl
410  << "GammaEnergy " << GammaEnergy << G4endl
411  << "PairInvMass " << PairInvMass << G4endl;
412  }
413  pdf = -1.0; // force next iteration
414  continue;
415  }
416  const G4double thePRecoil = std::sqrt(ap1 * bp1) * isqrts2;
417  // Denis ** correction
418 
419  // back to the center-of-mass frame
420  const G4LorentzVector Recoil1( thePRecoil*sinTheta*cosPhi,
421  thePRecoil*sinTheta*sinPhi,
422  thePRecoil*cosTheta,
423  RecEnergyCMS);
424 
425  const G4LorentzVector Pair1(-Recoil1.x(),
426  -Recoil1.y(),
427  -Recoil1.z(),
428  sqrts-RecEnergyCMS);
429 
430  // in the pair frame
431  const G4double thePLepton = std::sqrt( (LeptonEnergy2-ElectronMass)
432  *(LeptonEnergy2+ElectronMass));
433 
434  const G4LorentzVector Positron2( thePLepton*sinThetaLept*cosPhiLept,
435  thePLepton*sinThetaLept*sinPhiLept,
436  thePLepton*cosThetaLept,
437  LeptonEnergy2);
438 
439  const G4LorentzVector Electron2(-Positron2.x(),
440  -Positron2.y(),
441  -Positron2.z(),
442  LeptonEnergy2);
443 
444 
445  // Normalisation of final state phase space:
446  // Section 47 of Particle Data Group, Chin. Phys. C, 40, 100001 (2016)
447  const G4double Norme = Recoil1.vect().mag() * Positron2.vect().mag();
448  //
449  G4LorentzVector Positron1;
450  G4LorentzVector Electron1;
451  BoostG4LorentzVector(Positron2, Pair1, Positron1);
452  BoostG4LorentzVector(Electron2, Pair1, Electron1);
453 
454  // back to the laboratory frame (make use of the CMS(0,0,Eg,Eg+RM)) form
455  Recoil0.setX(Recoil1.x());
456  Recoil0.setY(Recoil1.y());
457  BoostG4LorentzVector(Recoil1 , CMSqz, CMSt, CMSfact, iCMSmass, Recoil0);
458 
459  Positron0.setX(Positron1.x());
460  Positron0.setY(Positron1.y());
461  BoostG4LorentzVector(Positron1, CMSqz, CMSt, CMSfact, iCMSmass, Positron0);
462 
463  Electron0.setX(Electron1.x());
464  Electron0.setY(Electron1.y());
465  BoostG4LorentzVector(Electron1, CMSqz, CMSt, CMSfact, iCMSmass, Electron0);
466 
467  // Jacobian factors
468  const G4double Jacob0 = x0*dum0*dum0;
469  const G4double Jacob1 = 2.*X1*lnPairInvMassRange*PairInvMass;
470  const G4double Jacob2 = std::abs(sinThetaLept);
471 
472  const G4double EPlus = Positron0.t();
473  const G4double PPlus = Positron0.vect().mag();
474  const G4double sinThetaPlus = Positron0.vect().perp()/PPlus;
475  const G4double cosThetaPlus = Positron0.vect().cosTheta();
476 
477  const G4double pPX = Positron0.x();
478  const G4double pPY = Positron0.y();
479  const G4double dum1 = 1./std::sqrt( pPX*pPX + pPY*pPY );
480  const G4double cosPhiPlus = pPX*dum1;
481  const G4double sinPhiPlus = pPY*dum1;
482 
483  // denominators:
484  // the two cancelling leading terms for forward emission at high energy, removed
485  const G4double elMassCTP = ElectronMass*cosThetaPlus;
486  const G4double ePlusSTP = EPlus*sinThetaPlus;
487  const G4double DPlus = (elMassCTP*elMassCTP + ePlusSTP*ePlusSTP)
488  /(EPlus + PPlus*cosThetaPlus);
489 
490  const G4double EMinus = Electron0.t();
491  const G4double PMinus = Electron0.vect().mag();
492  const G4double sinThetaMinus = Electron0.vect().perp()/PMinus;
493  const G4double cosThetaMinus = Electron0.vect().cosTheta();
494 
495  const G4double ePX = Electron0.x();
496  const G4double ePY = Electron0.y();
497  const G4double dum2 = 1./std::sqrt( ePX*ePX + ePY*ePY );
498  const G4double cosPhiMinus = ePX*dum2;
499  const G4double sinPhiMinus = ePY*dum2;
500 
501  const G4double elMassCTM = ElectronMass*cosThetaMinus;
502  const G4double eMinSTM = EMinus*sinThetaMinus;
503  const G4double DMinus = (elMassCTM*elMassCTM + eMinSTM*eMinSTM)
504  /(EMinus + PMinus*cosThetaMinus);
505 
506  // cos(phiMinus-PhiPlus)
507  const G4double cosdPhi = cosPhiPlus*cosPhiMinus + sinPhiPlus*sinPhiMinus;
