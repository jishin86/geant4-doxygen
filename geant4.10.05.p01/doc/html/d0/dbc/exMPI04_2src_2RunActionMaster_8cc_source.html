98 RunActionMaster::EndOfRunAction(const G4Run* arun)
99 {
100  G4cout << "RunActionMaster::EndOfRunAction" << G4endl;
101 
102  //This is executed by master thread only. Worker threads have already
103  //merged their results into this master threads.
104 
105  //We are going to merge the results via MPI for:
106  // 1. User-defined "Run" object
107  // 2. Command line scorers, if exists
108  // 3. G4Analysis objects
109 
110  // For debugging purposes in the following we write out results twice:
111  // BEFORE and AFTER the merging, so that rank 0 actually
112  // writes two files, the one called "*rank000*" will contain the partial
113  // results only from rank #0,
114  // the file *merged* contains the reduction from all ranks.
115  // It should be very easy to adapt this code
116 
117 
118  const G4int rank = G4MPImanager::GetManager()-> GetRank();
119 
120  G4cout << "=====================================================" << G4endl;
121  G4cout << "Start EndOfRunAction for master thread in rank: " << rank<<G4endl;
122  G4cout << "=====================================================" << G4endl;
123 
124  if ( ! G4MPImanager::GetManager()->IsExtraWorker() ) {
125  G4cout << "Go to merge on processing workers" << G4endl;
126 
127  //Merging of G4Run object:
128  //All ranks > 0 merge to rank #0
129  RunMerger rm(static_cast<const Run*>(arun), G4MPImanager::kRANK_MASTER, 5);
130  G4int ver = 0 ; //Use 4 for lots of info
131  if ( rank == 0 && ver == 0) ver = 1;
132 
133  if ( ver > 1 ) {
134  G4cout<<"Before merge the Run object has test counter value: "
135  <<static_cast<const Run*>(arun)->GetCounter()<<G4endl;
136  }
137  rm.SetVerbosity( ver );
138  rm.Merge();
139  if ( ver > 1 ) {
140  G4cout<<"After merge the Run object has test counter value: "
141  <<static_cast<const Run*>(arun)->GetCounter()
142  <<" (with 1 thread== number of ranks)"<<G4endl;
143  }
144 
145  //Merge of scorers
146  ver = 0;
147  if (G4ScoringManager::GetScoringManagerIfExist()) {
148  const auto scor = G4ScoringManager::GetScoringManager();
149  G4MPIscorerMerger sm(scor);
150  sm.SetVerbosity(ver);
151  //Debug!
152  auto debugme = [&scor](){
153  for ( size_t idx = 0 ; idx < scor->GetNumberOfMesh() ; ++idx) {
154  const auto m = scor->GetMesh(idx);
155  const MeshScoreMap& map = m->GetScoreMap();
156  std::for_each(map.begin(),map.end(),
157  [](const MeshScoreMap::value_type& e) {
158  G4cout<<e.first<<"("<<e.second<<"):"<<G4endl;
159  const auto data = e.second->GetMap();
160  for( auto it = data->begin() ; it != data->end() ; ++it ) {
161  G4cout<<it->first<<" => G4StatDouble(n,sum_w,sum_w2,sum_wx,sum_wx2): "
162  <<it->second->n()<<" "<<it->second->sum_w()<<" "
163  <<it->second->sum_w2()<<" "<<it->second->sum_wx()<<" "
164  <<it->second->sum_wx2()<<G4endl;
165  }
166  });
167  }
168  };
169  //Debug!
170  if ( ver > 4 ) {
171  G4cout<<"Before merging: Meshes dump"<<G4endl;
172  debugme();
173  }
174  //Write partial scorers from single ranks *before* merging
175  //Do not rely on UI command to write out scorers, because rank-specific
176  //files will have same file name: need to add rank # to file name
177  if ( true ) {
178  for ( size_t idx = 0 ; idx < scor->GetNumberOfMesh() ; ++idx) {
179  const auto m = scor->GetMesh(idx);
180  const auto& mn = m->GetWorldName();
181  std::ostringstream fname;
182  fname<<"scorer-"<<mn<<"-rank"<<rank<<".csv";
183  scor->DumpAllQuantitiesToFile(mn,fname.str());
184  }
185  }
186 
187  //Now reduce all scorers to rank #0
188  sm.Merge();
189 
190  //Debug!
191  if ( ver > 4 ) {
192  G4cout<<"After merging: Meshes dump"<<G4endl;
193  debugme();
194  }
195  //For rank #0 write out the merged files
196  if ( rank == 0 ) {
197  for ( size_t idx = 0 ; idx < scor->GetNumberOfMesh() ; ++idx) {
198  const auto m = scor->GetMesh(idx);
199  const auto& mn = m->GetWorldName();
200  std::ostringstream fname;
201  fname<<"scorer-"<<mn<<"-merged.csv";
202  scor->DumpAllQuantitiesToFile(mn,fname.str());
203  }
204  }
205  }
206 
207  //Save histograms *before* MPI merging for rank #0
208  if (rank == 0)
209  {
210  Analysis* myana = Analysis::GetAnalysis();
211  myana-> Save();
212  myana-> Close(false); // close without resetting histograms
213  }
214  //Merge of g4analysis objects
215  G4cout << "Go to merge histograms " << G4endl;
216  ver=1;
217  G4MPIhistoMerger hm(G4AnalysisManager::Instance());
218  hm.SetVerbosity(ver);
219  hm.Merge();
220  G4cout << "Done merge histograms " << G4endl;
221  }
222 
223  //Save g4analysis objects to a file
224  //NB: It is important that the save is done *after* MPI-merging of histograms
225 
226  //One can save all ranks or just rank0, chane the if
227  if (true /*rank == 0*/)
228  // if (rank == 0)
229  {
230  Analysis* myana = Analysis::GetAnalysis();
231  if (rank == 0) {
232  myana->OpenFile("dose-merged");
233  }
234  myana-> Save();
235  myana-> Close();
236  }
237 
238  G4cout << "===================================================" << G4endl;
239  G4cout << "End EndOfRunAction for master thread in rank: " << rank << G4endl;
240  G4cout << "===================================================" << G4endl;
241 }
