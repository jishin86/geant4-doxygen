47 void G4PhysicalVolumesSearchScene::ProcessVolume (const G4VSolid&)
48 {
49  G4VPhysicalVolume* pCurrentPV = fpSearchVolumesModel->GetCurrentPV();
50  const G4String& name = pCurrentPV->GetName();
51  G4int copyNo = fpSearchVolumesModel->GetCurrentPVCopyNo();
52  // Match the name with the required physical volume name. The latter can be of
53  // the form "/regexp/", where regexp is a regular expression (see C++ regex),
54  // or a plain name, in which case there must be an exact match.
55  if (fMatcher.Match(name)) {
56  if ((fRequiredCopyNo < 0 || // I.e., ignore negative request
57  fRequiredCopyNo == copyNo)) {
58  auto path = fpSearchVolumesModel->GetFullPVPath();
59  path.pop_back(); // Base node is one up from found node
60  fFindings.push_back
61  (Findings
62  (fpSearchVolumesModel->GetTopPhysicalVolume(),
63  pCurrentPV,
64  copyNo,
65  fpSearchVolumesModel->GetCurrentDepth(),
66  path,
67  *fpCurrentObjectTransformation));
68  }
69  }
70 }

72 G4PhysicalVolumesSearchScene::Matcher::Matcher(const G4String& requiredMatch)
73 : fRegexFlag(false)
74 {
75  if (requiredMatch.size()) {
76  size_t last = requiredMatch.size() - 1;
77  // If required name begins and ends with '/', treat as a regular expression.
78  // 0 causes a conversion ambiguity that upsets the Windows compiler, so use 0U.
79  if (requiredMatch[0U] == '/' && requiredMatch[last] == '/') {
80  if (last > 1) { // Non-null regexp
81  // regex match required
82  fRegexFlag = true;
83  // Extract the required regex
84  fRequiredMatch = requiredMatch.substr(1,last-1);
85  }
86  } else {
87  // Exact match required
88  fRequiredMatch = requiredMatch;
89  }
90  }
91  if (fRequiredMatch.isNull()) {
92  G4Exception
93  ("G4PhysicalVolumesSearchScene::Matcher::Matcher",
94  "modeling0013", JustWarning, "Required match is null");
95  }
96 }

98 G4bool G4PhysicalVolumesSearchScene::Matcher::Match(const G4String& s)
99 // Match the string with the required match. The latter can be of the form
100 // "/regexp/", where regexp is a regular expression (see C++ regex),
101 // or a plain string, in which case there must be an exact match.
102 {
103  G4bool found = false;
104  if (fRequiredMatch.size()) {
105  if (fRegexFlag) { // Use extracted regex
106  std::regex requiredRegex(fRequiredMatch);
107  std::cmatch match;
108  std::regex_search(s.c_str(), match, requiredRegex);
109  if (match.size() > 0) found = true;
110  } else { // Require complete match
111  if (s == fRequiredMatch) found = true;
112  }
113  }
114  return found;
115 }
