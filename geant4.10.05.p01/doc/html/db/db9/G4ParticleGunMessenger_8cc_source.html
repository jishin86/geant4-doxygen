203 void G4ParticleGunMessenger::SetNewValue(G4UIcommand * command,G4String newValues)
204 {
205  G4ExceptionDescription ed;
206  if (command==listCmd) {
207  particleTable->DumpTable();
208  } else if (command==particleCmd) {
209  if (newValues =="ion") {
210  fShootIon = true;
211  } else {
212  fShootIon = false;
213  G4ParticleDefinition* pd = particleTable->FindParticle(newValues);
214  if(pd != 0) {
215  fParticleGun->SetParticleDefinition( pd );
216  } else {
217  ed << "Particle [" << newValues << "] is not found.";
218  command->CommandFailed(ed);
219  }
220  }
221 
222  } else if( command==directionCmd )
223  { fParticleGun->SetParticleMomentumDirection(directionCmd->GetNew3VectorValue(newValues)); }
224  else if( command==energyCmd )
225  { fParticleGun->SetParticleEnergy(energyCmd->GetNewDoubleValue(newValues)); }
226  else if( command==momCmd )
227  { fParticleGun->SetParticleMomentum(momCmd->GetNew3VectorValue(newValues)); }
228  else if( command==momAmpCmd )
229  { fParticleGun->SetParticleMomentum(momAmpCmd->GetNewDoubleValue(newValues)); }
230  else if( command==positionCmd )
231  { fParticleGun->SetParticlePosition(positionCmd->GetNew3VectorValue(newValues)); }
232  else if( command==timeCmd )
233  { fParticleGun->SetParticleTime(timeCmd->GetNewDoubleValue(newValues)); }
234  else if( command==polCmd )
235  { fParticleGun->SetParticlePolarization(polCmd->GetNew3VectorValue(newValues)); }
236  else if( command==numberCmd )
237  { fParticleGun->SetNumberOfParticles(numberCmd->GetNewIntValue(newValues)); }
238  else if( command==ionCmd )
239  {
240  if (fShootIon) {
241  IonCommand(newValues);
242  } else {
243  ed << "Set /gun/particle to ion before using /gun/ion command";
244  command->CommandFailed(ed);
245  }
246  }
247  else if( command==ionLvlCmd )
248  {
249  G4ExceptionDescription depWarn;
250  depWarn << "\nCommand /gun/ionL is deprecated and will be removed in release 10.5.\n"
251  << "Use /gun/ion instead.\n";
252  G4Exception("G4ParticleGunMessenger::SetNewValue","IonLWarn",JustWarning,depWarn);
253 
254  if (fShootIon) {
255  IonLevelCommand(newValues);
256  } else {
257  ed << "Set /gun/particle to ion before using /gun/ion command";
258  command->CommandFailed(ed);
259  }
260  }
261 }

263 G4String G4ParticleGunMessenger::GetCurrentValue(G4UIcommand * command)
264 {
265  G4String cv;
266 
267  if( command==directionCmd )
268  { cv = directionCmd->ConvertToString(fParticleGun->GetParticleMomentumDirection()); }
269  else if( command==particleCmd )
270  { cv = fParticleGun->GetParticleDefinition()->GetParticleName(); }
271  else if( command==energyCmd )
272  {
273  G4double ene = fParticleGun->GetParticleEnergy();
274  if(ene == 0.)
275  { G4cerr << " G4ParticleGun: was defined in terms of momentum." << G4endl; }
276  else
277  { cv = energyCmd->ConvertToString(ene,"GeV"); }
278  }
279  else if( command==momCmd || command==momAmpCmd )
280  {
281  G4double mom = fParticleGun->GetParticleMomentum();
282  if(mom == 0.)
283  { G4cerr << " G4ParticleGun: was defined in terms of kinetic energy." << G4endl; }
284  else
285  {
286  if( command==momCmd )
287  { cv = momCmd->ConvertToString(mom*(fParticleGun->GetParticleMomentumDirection()),"GeV"); }
288  else
289  { cv = momAmpCmd->ConvertToString(mom,"GeV"); }
290  }
291  }
292  else if( command==positionCmd )
293  { cv = positionCmd->ConvertToString(fParticleGun->GetParticlePosition(),"cm"); }
294  else if( command==timeCmd )
295  { cv = timeCmd->ConvertToString(fParticleGun->GetParticleTime(),"ns"); }
296  else if( command==polCmd )
297  { cv = polCmd->ConvertToString(fParticleGun->GetParticlePolarization()); }
298  else if( command==numberCmd )
299  { cv = numberCmd->ConvertToString(fParticleGun->GetNumberOfParticles()); }
300  else if( command==ionCmd )
301  {
302  if (fShootIon) {
303  cv = ItoS(fAtomicNumber) + " " + ItoS(fAtomicMass) + " ";
304  cv += ItoS(fIonCharge);
305  } else {
306  cv = "";
307  }
308  }
309  return cv;
310 }

314 void G4ParticleGunMessenger::IonLevelCommand(G4String newValues)
315 {
316  G4Tokenizer next( newValues );
317  // check argument
318  fAtomicNumber = StoI(next());
319  fAtomicMass = StoI(next());
320  G4String sQ = next();
321  if (sQ.isNull() || StoI(sQ)<0) {
322  fIonCharge = fAtomicNumber;
323  } else {
324  fIonCharge = StoI(sQ);
325  }
326  sQ = next();
327  if (sQ.isNull()) {
328  fIonEnergyLevel = 0;
329  } else {
330  fIonEnergyLevel = StoI(sQ);
331  }
332  G4ParticleDefinition* ion = 0;
333  ion = G4IonTable::GetIonTable()->GetIon(fAtomicNumber,fAtomicMass,fIonEnergyLevel);
334  if (ion == 0) {
335  G4ExceptionDescription ed;
336  ed << "Ion with Z = " << fAtomicNumber << ", A = " << fAtomicMass
337  << ", I = " << fIonEnergyLevel << " is not defined ";
338  ionLvlCmd->CommandFailed(ed);
339  } else {
340  fParticleGun->SetParticleDefinition(ion);
341  fParticleGun->SetParticleCharge(fIonCharge*eplus);
342  }
343 }

345 void G4ParticleGunMessenger::IonCommand(G4String newValues)
346 {
347  G4Tokenizer next( newValues );
348  // check argument
349  fAtomicNumber = StoI(next());
350  fAtomicMass = StoI(next());
351  fIonCharge = fAtomicNumber;
352  fIonExciteEnergy = 0.0;
353  fIonFloatingLevelBase = '\0';
354  G4String sQ = next();
355  if (!(sQ.isNull()))
356  {
357  if (StoI(sQ)>=0)
358  fIonCharge = StoI(sQ);
359 
360  sQ = next();
361  if (!(sQ.isNull()))
362  {
363  fIonExciteEnergy = StoD(sQ) * keV;
364 
365  sQ = next();
366  if (sQ.isNull()||sQ=="noFloat")
367  { fIonFloatingLevelBase = '\0'; }
368  else
369  { fIonFloatingLevelBase = sQ[(size_t)0]; }
370  }
371  }
372  G4ParticleDefinition* ion = 0;
373  ion = G4IonTable::GetIonTable()->GetIon( fAtomicNumber, fAtomicMass,
374  fIonExciteEnergy, fIonFloatingLevelBase);
375  if (ion==0) {
376  G4ExceptionDescription ed;
377  ed << "Ion with Z=" << fAtomicNumber;
378  ed << " A=" << fAtomicMass << "is not defined";
379  ionCmd->CommandFailed(ed);
380  } else {
381  fParticleGun->SetParticleDefinition(ion);
382  fParticleGun->SetParticleCharge(fIonCharge*eplus);
383  }
384 }
