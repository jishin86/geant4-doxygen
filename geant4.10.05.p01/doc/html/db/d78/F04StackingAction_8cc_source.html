50  F04StackingAction::ClassifyNewTrack(const G4Track * aTrack)
51 {
52  G4ParticleDefinition* particleType = aTrack->GetDefinition();
53  const G4String name = particleType->GetParticleName();
54 
55  // keep primary particle
56  if (aTrack->GetParentID() == 0) return fUrgent;
57 
58  if (particleType != G4Proton::ProtonDefinition() &&
59  particleType != G4Neutron::NeutronDefinition() &&
60  particleType != G4KaonPlus::KaonPlusDefinition() &&
61  particleType != G4PionPlus::PionPlusDefinition() &&
62  particleType != G4MuonPlus::MuonPlusDefinition() &&
63  particleType != G4Positron::PositronDefinition()) return fKill;
64 
65  if (name != "pi+" && name != "mu+") return fKill;
66 
67 // if (name == "mu+") G4RunManager::GetRunManager()->rndmSaveThisEvent();
68
