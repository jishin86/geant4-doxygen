186 G4double
187 G4MolecularDissociationChannel::
188  GetRMSRadialDisplacementOfProduct(const G4Molecule* product)
189 {
190  if(!fProductsVector) return -1.;
191 
192  G4int sz = fProductsVector->size();
193  G4double value = DBL_MAX;
194  for(G4int i = 0; i < sz; i++)
195  {
196  if(product->GetMolecularConfiguration() != ((*fProductsVector)[i]))
197  {
198  value = fRMSProductsDisplacementVector[i];
199  }
200  }
201  return value;
202 }
