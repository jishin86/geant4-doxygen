114  manager->SetProduceFissionFragments( bValue );
115  }

117  manager->SetUseNRESP71Model( bValue );
118  }
119  if ( command == VerboseCmd ) {
120  manager->SetVerboseLevel( VerboseCmd->ConvertToInt( newValue ) );
