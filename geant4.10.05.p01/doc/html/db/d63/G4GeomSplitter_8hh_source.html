48 template <class T> // T is the private data from the object to be split
49 class G4GeomSplitter
50 {
51  public:

53  G4GeomSplitter()
54  : totalobj(0), totalspace(0), sharedOffset(0)
55  {
56  G4MUTEXINIT(mutex);

59  void *operator new(size_t)
60  // Override "new" for "G4Allocator"
61  {
62  if (!anOffsetAllocator) anOffsetAllocator = new G4Allocator<T>;
63  return (void *) anOffsetAllocator->MallocSingle();
64  }

66  void operator delete(void *anOffset)
67  // Override "delete" for "G4Allocator".
68  {
69  anOffsetAllocator->FreeSingle((T*) anOffset);
70  }

72  T* Reallocate(G4int size)
73  {
74  T* noffset = new T[size];
75  if (offset)
76  {
77  for (G4int i=0; i<totalspace; ++i)
78  {
79  noffset[i] = offset[i];
80  }
81  delete [] offset;
82  }
83  totalspace = size;
84 
85  return noffset;
86  }
87 
88  G4int CreateSubInstance()
89  // Invoked by the master or work thread to create a new subinstance
90  // whenever a new split class instance is created.
91  {
92  G4AutoLock l(&mutex);
93  totalobj++;
94  if (totalobj > totalspace)
95  {
96  offset = Reallocate(totalspace+512);
97  if (offset == 0)
98  {
99  G4Exception("G4GeomSPlitter::CreateSubInstance()",
100  "OutOfMemory", FatalException, "Cannot malloc space!");
101  }
102  sharedOffset = offset;
103  }
104  return (totalobj - 1);
105  }
106 
107  void CopyMasterContents()
108  {
109  G4AutoLock l(&mutex);
110  for (G4int i=0; i<totalspace; ++i)
111  {
112  offset[i] = sharedOffset[i];
113  }
114  }
115 
116  void SlaveCopySubInstanceArray()
117  // Invoked by each worker thread to copy all the subinstance array
118  // from the master thread.
119  {
120  G4AutoLock l(&mutex);
121  if (offset) { return; }
122  offset = Reallocate(totalspace);
123  if (offset == 0)
124  {
125  G4Exception("G4GeomSplitter::SlaveCopySubInstanceArray()",
126  "OutOfMemory", FatalException, "Cannot malloc space!");
127  }
128  l.unlock();
129  CopyMasterContents();
130  }
131 
132  void SlaveInitializeSubInstance()
133  // Invoked by each worker thread to create the subinstance array and
134  // initialize each subinstance using a particular method defined by
135  // the subclass.
136  {
137  G4AutoLock l(&mutex);
138  if (offset) { return; }
139  offset = Reallocate(totalspace);
140 
141  if (offset == 0)
142  {
143  G4Exception("G4GeomSplitter::SlaveInitializeSubInstance()",
144  "OutOfMemory", FatalException, "Cannot malloc space!");
145  }
146 
147  for (G4int i=0 ; i<totalspace; ++i)
148  {
149  offset[i].initialize();
150  }
151  }
152 
153  void SlaveReCopySubInstanceArray()
154  // Invoked by each worker thread at start of a run (2nd or later)
155  // to copy again all the subinstance array from the master thread.
156  // To cope with user's changes in Geometry - e.g. change of material
157  // in a volume
158  {
159  if (!offset)
160  {
161  SlaveInitializeSubInstance();
162  G4Exception("G4GeomSPlitter::SlaveReCopySubInstance()",
163  "MissingInitialisation", JustWarning,
164  "Must be called after Initialisation or first Copy.");
165  }
166  CopyMasterContents();
167  }
168 
169  void FreeSlave()
170  // Invoked by all threads to free the subinstance array.
171  {
172  if (!offset) { return; }
173  delete [] offset;
174  offset = 0;
175  }
176 
177  // Extension - to allow sharing of workspaces
178 
179  T* GetOffset() { return offset; }
180 
181  void UseWorkArea( T* newOffset )
182  // Use recycled work area - which was created previously
183  {
184  if( offset && offset!=newOffset )
185  {
186  G4Exception("G4GeomSplitter::UseWorkspace()",
187  "TwoWorkspaces", FatalException,
188  "Thread already has workspace - cannot use another.");
189  }
190  offset= newOffset;
191  }
192 
193  T* FreeWorkArea()
194  // Detach this thread from this Location.
195  // The object which calls this method is responsible for it.
196  {
197  T* offsetRet = offset;
198  offset = 0;
199  return offsetRet;
200  }
