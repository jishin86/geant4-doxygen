142  G4int G4ExcitedNucleonConstructor::GetQuarkContents(G4int iQ, G4int iIso3)
143 {
144  // Quark contents
145  // iIso3 = -1 : udd
146  // iIso3 = +1 : uud
147  G4int quark=0;
148  if ( iQ == 0 ){
149  // u-quark
150  quark = 2;
151  } else if ( iQ == 2 ){
152  // d-quark
153  quark = 1;
154  } else {
155  if ( iIso3 == -1 ){
156  // d-quark
157  quark = 1;
158  } else {
159  // u-quark
160  quark = 2;
161  }
162  }
163  return quark;
164 }
