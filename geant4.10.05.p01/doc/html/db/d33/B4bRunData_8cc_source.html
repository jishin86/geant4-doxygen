57 void B4bRunData::FillPerEvent()
58 {
59  // get analysis manager
60  auto analysisManager = G4AnalysisManager::Instance();
61 
62  // accumulate statistic
63  // in the order od the histograms, ntuple columns declarations
64  G4int counter = 0;
65  for ( auto edep : fEdep ) {
66  analysisManager->FillH1(counter, edep);
67  analysisManager->FillNtupleDColumn(counter++, edep);
68  }
69  for ( auto trackLength : fTrackLength ) {
70  analysisManager->FillH1(counter, trackLength);
71  analysisManager->FillNtupleDColumn(counter++, trackLength);
72  }
73  analysisManager->AddNtupleRow();
74 }
