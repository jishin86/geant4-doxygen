56 void OpNoviceSteppingAction::UserSteppingAction(const G4Step* step)
57 {
58  G4int eventNumber = G4RunManager::GetRunManager()->
59  GetCurrentEvent()->GetEventID();
60 
61  if (eventNumber != fEventNumber) {
62  fEventNumber = eventNumber;
63  fScintillationCounter = 0;
64  fCerenkovCounter = 0;
65  }
66 
67  G4Track* track = step->GetTrack();
68 
69  G4String ParticleName = track->GetDynamicParticle()->
70  GetParticleDefinition()->GetParticleName();
71 
72  if (ParticleName == "opticalphoton") return;
73 
74  const std::vector<const G4Track*>* secondaries =
75  step->GetSecondaryInCurrentStep();
76 
77  if (secondaries->size()>0) {
78  for(unsigned int i=0; i<secondaries->size(); ++i) {
79  if (secondaries->at(i)->GetParentID()>0) {
80  if(secondaries->at(i)->GetDynamicParticle()->GetParticleDefinition()
81  == G4OpticalPhoton::OpticalPhotonDefinition()){
82  if (secondaries->at(i)->GetCreatorProcess()->GetProcessName()
83  == "Scintillation")fScintillationCounter++;
84  if (secondaries->at(i)->GetCreatorProcess()->GetProcessName()
85  == "Cerenkov")fCerenkovCounter++;
86  }
87  }
88  }
89  }
90 }
