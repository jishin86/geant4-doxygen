33 // Utility class for adapting VecGeom solids API to Geant4 solids.
34 // NOTE: Using protected inheritance since the Adapter is supposed to

37 // to propagate functions further down in the inheritance hierarchy.

363 template <class UnplacedVolume_t>
364 G4double G4UAdapter<UnplacedVolume_t>::
365 DistanceToOut(const G4ThreeVector& pt, const G4ThreeVector& d,
366  const G4bool calcNorm, G4bool* validNorm,
367  G4ThreeVector* norm) const
368 {
369  U3Vector p(pt.x(), pt.y(), pt.z());
370  U3Vector v(d.x(), d.y(), d.z());
371 
372  G4double dist = UnplacedVolume_t::DistanceToOut(p, v, kInfinity);
373  if(calcNorm) // *norm=n, but only after calcNorm check and if convex volume
374  {
375  if (UnplacedVolume_t::IsConvex())
376  {
377  U3Vector n, hitpoint = p + dist * v;
378  UnplacedVolume_t::Normal(hitpoint, n);
379  *validNorm = true;
380  norm->set(n.x(), n.y(), n.z());
381  }
382  else
383  {
384  *validNorm = false;
385  }
386  }
387 
388  // Apply Geant4 distance conventions
389  //
390  if (dist < kHalfTolerance) return 0.0;
391  return (dist > kInfinity) ? kInfinity : dist;
392 }
