68 G4double XrayFluoNormalization::Integrate(G4double minIntExtreme, G4double maxIntExtreme, G4int nBins, XrayFluoDataSet* dataSet)
69 {
70  G4double partialHeight = 0;
71  G4int id = 0;
72  G4double lenghtOfBins = (maxIntExtreme - minIntExtreme)/nBins;
73 
74  for (G4int i = 0; i<nBins; i++)
75  {
76  G4double midPoint = minIntExtreme + i*lenghtOfBins+lenghtOfBins/2;
77 
78  G4double heightOfRectangle = dataSet->FindValue(midPoint,id);
79 
80  partialHeight += heightOfRectangle;
81 
82  }
83 
84  G4double integral = lenghtOfBins * partialHeight;
85 
86  delete dataSet;
87  dataSet = 0;
88  return integral;
89 }
