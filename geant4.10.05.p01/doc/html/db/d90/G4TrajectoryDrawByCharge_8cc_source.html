58 {
59  G4Colour colour;
60 
61  const G4double charge = traj.GetCharge();
62 
63  if(charge>0.) fMap.GetColour(Positive, colour);
64  else if(charge<0.) fMap.GetColour(Negative, colour);
65  else fMap.GetColour(Neutral, colour);
66 
67  G4VisTrajContext myContext(GetContext());
68 
69  myContext.SetLineColour(colour);
70  myContext.SetVisible(visible);
71 
72  if (GetVerbose()) {
73  G4cout<<"G4TrajectoryDrawByCharge drawer named "<<Name();
74  G4cout<<", drawing trajectory with charge, "<<charge<<G4endl;
75  G4cout<<", with configuration:"<<G4endl;
76  myContext.Print(G4cout);
77  }
78 
79  G4TrajectoryDrawerUtils::DrawLineAndPoints(traj, myContext);
80 }

134 G4bool
135 G4TrajectoryDrawByCharge::ConvertToCharge(const G4String& string, Charge& myCharge)
136 {
137  bool result(true);
138 
139  G4int charge;
140  std::istringstream is(string.c_str());
141  is >> charge;
142 
143  switch (charge) {
144  case 1:
145  myCharge = G4TrajectoryDrawByCharge::Positive;
146  break;
147  case 0:
148  myCharge = G4TrajectoryDrawByCharge::Neutral;
149  break;
150  case -1:
151  myCharge = G4TrajectoryDrawByCharge::Negative;
152  break;
153  default:
154  result = false;
155  }
156 
157  return result;
158 }
