83  fNtupleMergeMode(G4NtupleMergeMode::kNone),
84  fNtupleManager(nullptr),
85  fSlaveNtupleManager(nullptr),
86  fFileManager(nullptr)
87 {
88  if ( ( isMaster && fgMasterInstance ) || ( fgInstance ) ) {
89  G4ExceptionDescription description;
90  description
91  << " "
92  << "G4RootAnalysisManager already exists."
93  << "Cannot create another instance.";
94  G4Exception("G4RootAnalysisManager::G4RootAnalysisManager()",
95  "Analysis_F001", FatalException, description);
96  }
97  if ( isMaster ) fgMasterInstance = this;
98  fgInstance = this;
99 
100  // File manager
101  fFileManager = std::make_shared<G4RootFileManager>(fState);
102  SetFileManager(fFileManager);
103  fFileManager->SetBasketSize(fgkDefaultBasketSize);
104 
105  // Do not merge ntuples by default
106  // Merging may require user code migration as analysis manager
107  // must be created both on master and workers.
108  auto mergeNtuples = false;
109  SetNtupleMergingMode(mergeNtuples, fNofNtupleFiles, fNtupleRowWise);
110 
111  // Create ntuple managers
112  CreateNtupleManagers();
113 }

127 void G4RootAnalysisManager::SetNtupleMergingMode(G4bool mergeNtuples,
128  G4int nofNtupleFiles,
129  G4bool rowWise)
130 
131 {
132 #ifdef G4VERBOSE
133  if ( fState.GetVerboseL4() )
134  fState.GetVerboseL4()
135  ->Message("set", "ntuple merging mode", "");
136 #endif
137 
138  auto canMerge = true;
139 
140  // Illegal situations
141  if ( mergeNtuples && ( ! G4Threading::IsMultithreadedApplication() ) ) {
142  if ( nofNtupleFiles > 0 ) {
143  G4ExceptionDescription description;
144  description
145  << " " << "Merging ntuples is not applicable in sequential application."
146  << G4endl
147  << " " << "Setting was ignored.";
148  G4Exception("G4RootAnalysisManager::SetNtupleMergingMode()",
149  "Analysis_W013", JustWarning, description);
150  }
151  canMerge = false;
152  }
153 
154  // Illegal situations
155  if ( mergeNtuples && G4Threading::IsMultithreadedApplication() &&
156  ( ! fgMasterInstance ) ) {
157  G4ExceptionDescription description;
158  description
159  << " " << "Merging ntuples requires G4AnalysisManager instance on master."
160  << G4endl
161  << " " << "Setting was ignored.";
162  G4Exception("G4RootAnalysisManager::SetNtupleMergingMode()",
163  "Analysis_W013", JustWarning, description);
164  canMerge = false;
165  }
166 
167  G4String mergingMode;
168  if ( ( ! mergeNtuples ) || ( ! canMerge ) ) {
169  fNtupleMergeMode = G4NtupleMergeMode::kNone;
170  mergingMode = "G4NtupleMergeMode::kNone";
171  }
172  else {
173  // Set the number of reduced ntuple files
174  fNofNtupleFiles = nofNtupleFiles;
175  fNtupleRowWise = rowWise;
176 
177  // Check the number of reduced ntuple files
178  // if ( fNofNtupleFiles < 0 || fNofNtupleFiles > nofThreads ) {
179  if ( fNofNtupleFiles < 0 ) {
180  G4ExceptionDescription description;
181  description
182  << " " << "Number of reduced files must be [0, nofThreads]."
183  << G4endl
184  << " " << "Cannot set " << nofNtupleFiles
185  // << " files when nofThreads is " << nofThreads << G4endl
186  << " files" << G4endl
187  << " " << "Ntuples will be merged in a single file.";
188  G4Exception("G4RootAnalysisManager::SetNtupleMergingMode()",
189  "Analysis_W013", JustWarning, description);
190  fNofNtupleFiles = 0;
191  }
192 
193  // if ( fNofNtupleFiles == nofThreads ) {
194  // // add warning that no merging will be applied
195  // fNtupleMergeMode = G4NtupleMergeMode::kNone;
196  // fNofNtupleFiles = 0;
197  // mergingMode = "G4NtupleMergeMode::kNone";
198  // }
199  // else {
200  // G4bool isMaster = ! G4Threading::IsWorkerThread();
201  // if ( isMaster ) {
202  // fNtupleMergeMode = G4NtupleMergeMode::kMain;
203  // mergingMode = "G4NtupleMergeMode::kMain";
204  // } else {
205  // fNtupleMergeMode = G4NtupleMergeMode::kSlave;
206  // mergingMode = "G4NtupleMergeMode::kSlave";
207  // }
208  // }
209 
210  // Forced merging mode
211  G4bool isMaster = ! G4Threading::IsWorkerThread();
212  if ( isMaster ) {
213  fNtupleMergeMode = G4NtupleMergeMode::kMain;
214  mergingMode = "G4NtupleMergeMode::kMain";
215  } else {
216  fNtupleMergeMode = G4NtupleMergeMode::kSlave;
217  mergingMode = "G4NtupleMergeMode::kSlave";
218  }
219  }
220 
221 #ifdef G4VERBOSE
222  if ( fState.GetVerboseL2() )
223  fState.GetVerboseL2()
224  ->Message("set", "ntuple merging mode", mergingMode);
225 #endif

253 void G4RootAnalysisManager::CreateNtupleManagers()
254 {
255 #ifdef G4VERBOSE
256  if ( fState.GetVerboseL4() )
257  fState.GetVerboseL4()->Message("create", "ntuple managers", "");
258 #endif
259 
260  switch ( fNtupleMergeMode )
261  {
262  case G4NtupleMergeMode::kNone:
263  fNtupleManager = new G4RootNtupleManager(fState, 0, fNtupleRowWise);
264  fNtupleManager->SetFileManager(fFileManager);
265  SetNtupleManager(fNtupleManager);
266  break;
267 
268  case G4NtupleMergeMode::kMain: {
269  G4int nofMainManagers = fNofNtupleFiles;
270  if ( ! nofMainManagers ) nofMainManagers = 1;
271  // create one manager if merging required into the histos & profiles files
272  fNtupleManager = new G4RootNtupleManager(fState, nofMainManagers, fNtupleRowWise);
273  fNtupleManager->SetFileManager(fFileManager);
274  SetNtupleManager(fNtupleManager);
275  break;
276  }
277 
278  case G4NtupleMergeMode::kSlave:
279  fNtupleManager = fgMasterInstance->fNtupleManager;
280  // The master class is used only in Get* functions
281  auto mainNtupleManager
282  = fNtupleManager->GetMainNtupleManager(GetNtupleFileNumber());
283  fSlaveNtupleManager = new G4RootPNtupleManager(mainNtupleManager, fState);
284  SetNtupleManager(fSlaveNtupleManager);
285  break;
286  }
287 
288 #ifdef G4VERBOSE
289  if ( fState.GetVerboseL3() )
290  fState.GetVerboseL3()->Message("create", "ntuple managers", "");
291 #endif
292 }

295 G4int G4RootAnalysisManager::GetNtupleFileNumber()
296 {
297  if ( ! fNofNtupleFiles ) return 0;
298 
299  G4int nofMainManagers = fNofNtupleFiles;
300  if ( ! nofMainManagers ) nofMainManagers = 1;
301 
302  // Debug - check G4Threading::GetNumberOfRunningWorkerThreads()
303  G4cout << "In GetNtupleFileNumber: "

307  return fileNumber;

311 G4bool G4RootAnalysisManager::WriteH1()
312 {
313  auto h1Vector = fH1Manager->GetH1Vector();
314  auto hnVector = fH1Manager->GetHnVector();
315 
316  if ( ! h1Vector.size() ) return true;
317 
318  auto result = true;
319 
320  if ( ! G4Threading::IsWorkerThread() ) {
321  auto directory = fFileManager->GetHistoDirectory();
322  result = WriteT(h1Vector, hnVector, directory, "h1");
323  }
324  else {
325  // The worker manager just adds its histograms to the master
326  // This operation needs a lock
327  G4AutoLock lH1(&mergeH1Mutex);
328  fgMasterInstance->fH1Manager->AddH1Vector(h1Vector);
329  lH1.unlock();
330  }
331 
332  return result;
333 }

564 G4bool G4RootAnalysisManager::WriteImpl()
565 {
566 
567  auto finalResult = true;
568 
569  if ( ! fgMasterInstance &&
570  ( ( ! fH1Manager->IsEmpty() ) || ( ! fH2Manager->IsEmpty() ) ||
571  ( ! fH3Manager->IsEmpty() ) || ( ! fP1Manager->IsEmpty() ) ||
572  ( ! fP2Manager->IsEmpty() ) ) ) {
573  G4ExceptionDescription description;
574  description
575  << " " << "No master G4RootAnalysisManager instance exists."
576  << G4endl
577  << " " << "Histogram/profile data will not be merged.";
578  G4Exception("G4RootAnalysisManager::Write()",
579  "Analysis_W031", JustWarning, description);
580  }
581 
582  // H1
583  auto result = WriteH1();
584  finalResult = finalResult && result;
585 
586  // H2
587  result = WriteH2();
588  finalResult = finalResult && result;
589 
590  // H3
591  result = WriteH3();
592  finalResult = finalResult && result;
593 
594  // P1
595  result = WriteP1();
596  finalResult = finalResult && result;
597 
598  // P2
599  result = WriteP2();
600  finalResult = finalResult && result;
601 
602  // Ntuples
603  result = WriteNtuple();
604  finalResult = finalResult && result;
605 
606  // File
607  if ( fNtupleMergeMode != G4NtupleMergeMode::kSlave ) {
608  result = fFileManager->WriteFile();
609  finalResult = finalResult && result;
610  }
611 
612  // Write ASCII if activated
613  if ( IsAscii() ) {
614  result = WriteAscii(fFileManager->GetFileName());
615  finalResult = finalResult && result;
616  }
617 
618  return finalResult;
619 }

622 G4bool G4RootAnalysisManager::CloseFileImpl(G4bool reset)
623 {
624  auto finalResult = true;
625 
626  G4bool isNtupleManagerEmpty = fNtupleManager->IsEmpty();
627  // the ntuple decription vector is cleared on Reset()
628  // in kNoMergeAfterOpen ntuple manager mode
629 
630  auto result = true;
631  if ( reset ) {
632  result = Reset();
633  } else {
634  // ntuple must be reset
635  result = ResetNtuple();
636  }
637  if ( ! result ) {
638  G4ExceptionDescription description;
639  description << " " << "Resetting data failed";
640  G4Exception("G4RootAnalysisManager::CloseFile()",
641  "Analysis_W021", JustWarning, description);
642  }
643  finalResult = finalResult && result;
644 
645  if ( fNtupleMergeMode != G4NtupleMergeMode::kSlave ) {
646  // close file
647  fFileManager->CloseFile();
648  }
649 
650  // No files clean-up in sequential mode
651  if ( ! G4Threading::IsMultithreadedApplication() ) return finalResult;
652 
653  // Delete files if empty in MT mode
654  if ( ( fState.GetIsMaster() &&
655  fH1Manager->IsEmpty() && fH2Manager->IsEmpty() && fH3Manager->IsEmpty() &&
656  fP1Manager->IsEmpty() && fP2Manager->IsEmpty() && isNtupleManagerEmpty ) ||
657  ( ( ! fState.GetIsMaster() ) && isNtupleManagerEmpty &&
658  fNtupleMergeMode == G4NtupleMergeMode::kNone ) ) {
659  result = ! std::remove(fFileManager->GetFullFileName());
660  // std::remove returns 0 when success
661  if ( ! result ) {
662  G4ExceptionDescription description;
663  description << " " << "Removing file "
664  << fFileManager->GetFullFileName() << " failed";
665  G4Exception("G4XmlAnalysisManager::CloseFile()",
666  "Analysis_W021", JustWarning, description);
667  }
668  finalResult = finalResult && result;
669 #ifdef G4VERBOSE
670  if ( fState.GetVerboseL1() )
671  fState.GetVerboseL1()
672  ->Message("delete", "empty file", fFileManager->GetFullFileName());
673 #endif
674  }
675 
676  return finalResult;
677 }
