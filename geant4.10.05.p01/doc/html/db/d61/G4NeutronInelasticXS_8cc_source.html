143 G4double G4NeutronInelasticXS::GetElementCrossSection(
144  const G4DynamicParticle* aParticle,
145  G4int ZZ, const G4Material*)
146 {
147  G4double xs = 0.0;
148  G4double ekin = aParticle->GetKineticEnergy();
149 
150  G4int Z = (ZZ >= MAXZINEL) ? MAXZINEL - 1 : ZZ;
151 
152  G4PhysicsVector* pv = data->GetElementData(Z);
153  // G4cout << "G4NeutronInelasticXS::GetCrossSection e= " << ekin
154  // << " Z= " << Z << G4endl;
155 
156  // element was not initialised
157  if(!pv || ekin <= pv->Energy(0)) { return xs; }
158 
159  if(ekin <= pv->GetMaxEnergy()) {
160  xs = pv->Value(ekin);
161  } else if(1 == Z) {
162  fNucleon->GetHadronNucleonXscNS(aParticle, proton);
163  xs = coeff[1]*fNucleon->GetInelasticHadronNucleonXsc();
164  } else {
165  G4int Amean =
166  G4lrint(G4NistManager::Instance()->GetAtomicMassAmu(Z));
167  ggXsection->GetIsoCrossSection(aParticle, Z, Amean);
168  xs = coeff[Z]*ggXsection->GetInelasticGlauberGribovXsc();
169  }
170 
171  if(verboseLevel > 0) {
172  G4cout << "Z= " << Z << " Ekin(MeV)= " << ekin/CLHEP::MeV
173  << ", nElmXSinel(bn)= " << xs/CLHEP::barn
174  << G4endl;
175  }
176  return xs;
177 }

188 G4double
189 G4NeutronInelasticXS::IsoCrossSection(G4double ekin, G4int ZZ, G4int A)
190 {
191  G4double xs = 0.0;
192  G4int Z = (ZZ >= MAXZINEL) ? MAXZINEL - 1 : ZZ;
193 
194  /*
195  G4cout << "IsoCrossSection Z= " << Z << " A= " << A
196  << " Amin= " << amin[Z] << " Amax= " << amax[Z]
197  << " E(MeV)= " << ekin << G4endl;
198  */
199  // first compute isotope cross section
200  if(ekin <=emax && amin[Z]>0 && A >= amin[Z] && A <= amax[Z]) {
201  G4PhysicsVector* pviso = data->GetComponentDataByIndex(Z, A - amin[Z]);
202  if(pviso) {
203  xs = pviso->Value(ekin);
204  if(verboseLevel > 0) {
205  G4cout << "IsoXS: Z= " << Z << " A= " << A
206  << " Ekin(MeV)= " << ekin/CLHEP::MeV
207  << ", nElmXSinel(bn)= " << xs/CLHEP::barn << G4endl;
208  }
209  return xs;
210  }
211  }
212  // isotope data are not available or applicable
213  G4PhysicsVector* pv = data->GetElementData(Z);
214  if(pv) { xs = pv->Value(ekin); }
215  if(verboseLevel > 0) {
216  G4cout << "IsoXS: Z= " << Z << " A= " << A
217  << " Ekin(MeV)= " << ekin/CLHEP::MeV
218  << ", nElmXSinel(bn)= " << xs/CLHEP::barn << G4endl;
219  }
220  return xs;
221 }

223 const G4Isotope* G4NeutronInelasticXS::SelectIsotope(
224  const G4Element* anElement, G4double kinEnergy)
225 {
226  size_t nIso = anElement->GetNumberOfIsotopes();
227  const G4Isotope* iso = anElement->GetIsotope(0);
228 
229  //G4cout << "SelectIsotope NIso= " << nIso << G4endl;
230  if(1 == nIso) { return iso; }
231 
232  // more than 1 isotope
233  G4int Z = anElement->GetZasInt();
234  //G4cout << "SelectIsotope Z= " << Z << G4endl;
235 
236  const G4double* abundVector = anElement->GetRelativeAbundanceVector();
237  G4double q = G4UniformRand();
238  G4double sum = 0.0;
239 
240  // isotope wise cross section not available
241  size_t j;
242  if(kinEnergy > emax || 0 == amin[Z] || Z >= MAXZINEL) {
243  for (j = 0; j<nIso; ++j) {
244  sum += abundVector[j];
245  if(q <= sum) {
246  iso = anElement->GetIsotope(j);
247  break;
248  }
249  }
250  return iso;
251  }
252 
253  // use isotope cross sections
254  size_t nn = temp.size();
255  if(nn < nIso) { temp.resize(nIso, 0.); }
256 
257  for (j=0; j<nIso; ++j) {
258  //G4cout << j << "-th isotope " << (*isoVector)[j]->GetN()
259  // << " abund= " << abundVector[j] << G4endl;
260  sum += abundVector[j]*IsoCrossSection(kinEnergy, Z,
261  anElement->GetIsotope(j)->GetN());
262  temp[j] = sum;
263  }
264  sum *= q;
265  for (j = 0; j<nIso; ++j) {
266  if(temp[j] >= sum) {
267  iso = anElement->GetIsotope(j);
268  break;
269  }
270  }
271  return iso;
272 }

274 void
275 G4NeutronInelasticXS::BuildPhysicsTable(const G4ParticleDefinition& p)
276 {
277  if(verboseLevel > 0){
278  G4cout << "G4NeutronInelasticXS::BuildPhysicsTable for "
279  << p.GetParticleName() << G4endl;
280  }
281  if(p.GetParticleName() != "neutron") {
282  G4ExceptionDescription ed;
283  ed << p.GetParticleName() << " is a wrong particle type -"
284  << " only neutron is allowed";
285  G4Exception("G4NeutronInelasticXS::BuildPhysicsTable(..)","had012",
286  FatalException, ed, "");
287  return;
288  }
289 
290  if(!data) {
291 #ifdef G4MULTITHREADED
292  G4MUTEXLOCK(&neutronInelasticXSMutex);
293  if(!data) {
294 #endif
295  isMaster = true;
296  data = new G4ElementData();
297  data->SetName("NeutronInelastic");
298  temp.resize(13,0.0);
299 #ifdef G4MULTITHREADED
300  }
301  G4MUTEXUNLOCK(&neutronInelasticXSMutex);
302 #endif
303  }
304 
305  // it is possible re-initialisation for the new run
306  if(isMaster) {
307 
308  // check environment variable
309  // Build the complete string identifying the file with the data set
310  char* path = getenv("G4PARTICLEXSDATA");
311 
312  G4DynamicParticle* dynParticle =
313  new G4DynamicParticle(G4Neutron::Neutron(),G4ThreeVector(1,0,0),1);
314 
315  // Access to elements
316  const G4ElementTable* theElmTable = G4Element::GetElementTable();
317  size_t numOfElm = G4Element::GetNumberOfElements();
318  for(size_t i=0; i<numOfElm; ++i) {
319  G4int Z = ((*theElmTable)[i])->GetZasInt();
320  if(Z >= MAXZINEL) { Z = MAXZINEL-1; }
321  //G4cout << "Z= " << Z << G4endl;
322  // Initialisation
323  if(!(data->GetElementData(Z))) {
324  Initialise(Z, dynParticle, path);
325  }
326  }
327  delete dynParticle;
328  }
329 }

331 void
332 G4NeutronInelasticXS::Initialise(G4int Z, G4DynamicParticle* dp,
333  const char* p)
334 {
335  if(data->GetElementData(Z) || Z < 1 || Z >= MAXZINEL) { return; }
336  const char* path = p;
337  if(!p) {
338  // check environment variable
339  // Build the complete string identifying the file with the data set
340  path = getenv("G4PARTICLEXSDATA");
341  if (!path) {
342  G4Exception("G4NeutronInelasticXS::Initialise(..)","had013",
343  FatalException,
344  "Environment variable G4PARTICLEXSDATA is not defined");
345  return;
346  }
347  }
348  // upload element data
349  std::ostringstream ost;
350  ost << path << "/neutron/inel" << Z ;
351  G4PhysicsVector* v = RetrieveVector(ost, true);
352  data->InitialiseForElement(Z, v);
353  /*
354  G4cout << "G4NeutronInelasticXS::Initialise for Z= " << Z
355  << " A= " << Amean << " Amin= " << amin[Z]
356  << " Amax= " << amax[Z] << G4endl;
357  */
358  // upload isotope data
359  if(amin[Z] > 0) {
360  size_t nmax = (size_t)(amax[Z]-amin[Z]+1);
361  data->InitialiseForComponent(Z, nmax);
362 
363  for(G4int A=amin[Z]; A<=amax[Z]; ++A) {
364  std::ostringstream ost1;
365  ost1 << path << "/neutron/inel" << Z << "_" << A;
366  G4PhysicsVector* v1 = RetrieveVector(ost1, false);
367  data->AddComponent(Z, A, v1);
368  }
369  }
370 
371  // smooth transition
372  G4double sig1 = (*v)[v->GetVectorLength()-1];
373  dp->SetKineticEnergy(v->GetMaxEnergy());
374  G4double sig2 = 0.0;
375  if(1 == Z) {
376  fNucleon->GetHadronNucleonXscNS(dp, proton);
377  sig2 = fNucleon->GetInelasticHadronNucleonXsc();
378  } else {
379  G4int Amean =
380  G4lrint(G4NistManager::Instance()->GetAtomicMassAmu(Z));
381  ggXsection->GetIsoCrossSection(dp, Z, Amean);
382  sig2 = ggXsection->GetInelasticGlauberGribovXsc();
383  }
384  if(sig2 > 0.) { coeff[Z] = sig1/sig2; }
385 }
