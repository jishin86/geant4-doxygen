58 void PrimaryGenerator::GeneratePrimaryVertex(G4Event* event)
59 {
60  //vertex A uniform on a cylinder
61  //
62  const G4double r = 2*mm;
63  const G4double zmax = 8*mm;
64  //
65  G4double alpha = twopi*G4UniformRand(); //alpha uniform in (0, 2*pi)
66  G4double ux = std::cos(alpha);
67  G4double uy = std::sin(alpha);
68  G4double z = zmax*(2*G4UniformRand() - 1); //z uniform in (-zmax, +zmax)
69  G4ThreeVector positionA(r*ux,r*uy,z);
70  G4double timeA = 0*s;
71  //
72  G4PrimaryVertex* vertexA = new G4PrimaryVertex(positionA, timeA);
73 
74  //particle 1 at vertex A
75  //
76  G4ParticleDefinition* particleDefinition
77  = G4ParticleTable::GetParticleTable()->FindParticle("geantino");
78  G4PrimaryParticle* particle1 = new G4PrimaryParticle(particleDefinition);
79  particle1->SetMomentumDirection(G4ThreeVector(ux,uy,0));
80  particle1->SetKineticEnergy(1*MeV);
81  //
82  vertexA->SetPrimary(particle1);
83  event->AddPrimaryVertex(vertexA);
84 
85  //vertex (B) symetric to vertex A
86  //
87  alpha += pi;
88  ux = std::cos(alpha);
89  uy = std::sin(alpha);
90  G4ThreeVector positionB(r*ux,r*uy,z);
91  G4double timeB = 1*s;
92  //
93  G4PrimaryVertex* vertexB = new G4PrimaryVertex(positionB, timeB);
94 
95  //particles 2 and 3 at vertex B
96  //
97  const G4double dalpha = 10*deg;
98  ux = std::cos(alpha + dalpha);
99  uy = std::sin(alpha + dalpha);
100  G4PrimaryParticle* particle2 = new G4PrimaryParticle(particleDefinition);
101  particle2->SetMomentumDirection(G4ThreeVector(ux,uy,0));
102  particle2->SetKineticEnergy(1*keV);
103  //
104  ux = std::cos(alpha - dalpha);
105  uy = std::sin(alpha - dalpha);
106  G4PrimaryParticle* particle3 = new G4PrimaryParticle(particleDefinition);
107  particle3->SetMomentumDirection(G4ThreeVector(ux,uy,0));
108  particle3->SetKineticEnergy(1*GeV);
109  //
110  vertexB->SetPrimary(particle2);
111  vertexB->SetPrimary(particle3);
112  event->AddPrimaryVertex(vertexB);
113 }
