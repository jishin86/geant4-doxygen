109 void G4VRML2SceneHandler::connectPort(G4int max_trial)
110 {
111  G4int trial = 0 ;
112  int port = fSystem.getPort();
113  for (trial = 0; !fDest.isConnected()&& trial < max_trial; trial++, port++ ) {
114  if (fDest.connect( (const char * )fSystem.getHostName(), port)) {
115  // INET domain connection is established
116  if (G4VisManager::GetVerbosity() >= G4VisManager::errors) {
117  G4cout << "*** GEANT4 is connected to port ";
118  G4cout << fDest.getPort();
119  G4cout << " of server " << fSystem.getHostName() << G4endl;
120  }
121  break;
122  } else {
123  // Connection failed. Try the next port.
124  if (G4VisManager::GetVerbosity() >= G4VisManager::errors) {
125  G4cout << "*** GEANT4 incremented targeting port to ";
126  G4cout << port << G4endl;
127  }
128  }
129 
130  sleep (1);
131 
132  } // for
133 
134  if (!fDest.isConnected()) {
135  if (G4VisManager::GetVerbosity() >= G4VisManager::errors) {
136  G4cout << "*** INET Connection failed. " << G4endl;
137  G4cout << " Maybe, you have not invoked viewer g4vrmlview yet, " << G4endl;
138  G4cout << " or too many viewers are already running in the " << G4endl;
139  G4cout << " server host(" << fSystem.getHostName() << "). " << G4endl;
140  }
141  }
142 }
