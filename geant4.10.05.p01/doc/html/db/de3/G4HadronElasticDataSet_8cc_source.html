68 G4double G4HadronElasticDataSet::GetElementCrossSection(
69  const G4DynamicParticle* aParticle, G4int Z, const G4Material*)
70 {
71  G4double ekin = aParticle->GetKineticEnergy();
72  const G4ParticleDefinition* pd = aParticle->GetDefinition();
73  if(Z != theZ || ekin != fKinEnergy || pd != fParticle) {
74  theZ = Z;
75  fKinEnergy = ekin;
76  fParticle = pd;
77  G4int A = G4lrint(fNIST->GetAtomicMassAmu(Z));
78  fElasticXS = fGheishaXS->GetElasticCrossSection(aParticle, Z, A);
79  }
80  return fElasticXS;
81 }
