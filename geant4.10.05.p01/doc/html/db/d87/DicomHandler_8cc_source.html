78 G4String DicomHandler::GetDicomDataPath()
79 {
80  // default is current directory
81  G4String driverPath = ".";
82  // check environment
83  const char* path = getenv("DICOM_PATH");
84 
85  if(path)
86  {
87  // if is set in environment
88  return G4String(path);
89  }
90  else
91  {
92  // if DICOM_USE_HEAD, look for data installation
93 #ifdef DICOM_USE_HEAD
94  G4cerr << "Warning! DICOM was compiled with DICOM_USE_HEAD option but "
95  << "the DICOM_PATH was not set!" << G4endl;
96  G4String datadir = G4GetEnv<G4String>("G4ENSDFSTATEDATA", "");
97  if(datadir.length() > 0)
98  {
99  auto _last = datadir.rfind("/");
100  if(_last != std::string::npos)
101  datadir.erase(_last);
102  driverPath = datadir + "/DICOM1.1/DICOM_HEAD_TEST";
103  int rc = setenv("DICOM_PATH", driverPath.c_str(), 0);
104  G4cerr << "\t --> Using '" << driverPath << "'..." << G4endl;
105  G4ConsumeParameters(rc);
106  }
107 #endif
108  }
109  return driverPath;
110 }

164 G4int DicomHandler::ReadFile(FILE* dicom, char* filename2)
165 {
166  G4cout << " ReadFile " << filename2 << G4endl;
167 
168  G4int returnvalue = 0; size_t rflag = 0;
169  char * buffer = new char[LINEBUFFSIZE];
170 
171  fImplicitEndian = false;
172  fLittleEndian = true;
173 
174  rflag = std::fread( buffer, 1, 128, dicom ); // The first 128 bytes
175  //are not important
176  // Reads the "DICOM" letters
177  rflag = std::fread( buffer, 1, 4, dicom );
178  // if there is no preamble, the FILE pointer is rewinded.
179  if(std::strncmp("DICM", buffer, 4) != 0) {
180  std::fseek(dicom, 0, SEEK_SET);
181  fImplicitEndian = true;
182  }
183 
184  short readGroupId; // identify the kind of input data
185  short readElementId; // identify a particular type information
186  short elementLength2; // deal with element length in 2 bytes
187  //unsigned int elementLength4;
188  // deal with element length in 4 bytes
189  unsigned long elementLength4; // deal with element length in 4 bytes
190 
191  char * data = new char[DATABUFFSIZE];
192 
193  // Read information up to the pixel data
194  while(true) {
195 
196  //Reading groups and elements :
197  readGroupId = 0;
198  readElementId = 0;
199  // group ID
200  rflag = std::fread(buffer, 2, 1, dicom);
201  GetValue(buffer, readGroupId);
202  // element ID
203  rflag = std::fread(buffer, 2, 1, dicom);
204  GetValue(buffer, readElementId);
205 
206  // Creating a tag to be identified afterward
207  G4int tagDictionary = readGroupId*0x10000 + readElementId;
208 
209  // beginning of the pixels
210  if(tagDictionary == 0x7FE00010) {
211  // Following 2 fread's are modifications to original DICOM example
212  // (Jonathan Madsen)
213  if(!fImplicitEndian)
214  rflag = std::fread(buffer,2,1,dicom); // Reserved 2 bytes
215  // (not used for pixels)
216  rflag = std::fread(buffer,4,1,dicom); // Element Length
217  // (not used for pixels)
218  break; // Exit to ReadImageData()
219  }
220 
221  // VR or element length
222  rflag = std::fread(buffer,2,1,dicom);
223  GetValue(buffer, elementLength2);
224 
225  // If value representation (VR) is OB, OW, SQ, UN, added OF and UT
226  //the next length is 32 bits
227  if((elementLength2 == 0x424f || // "OB"
228  elementLength2 == 0x574f || // "OW"
229  elementLength2 == 0x464f || // "OF"
230  elementLength2 == 0x5455 || // "UT"
231  elementLength2 == 0x5153 || // "SQ"
232  elementLength2 == 0x4e55) && // "UN"
233  !fImplicitEndian ) { // explicit VR
234 
235  rflag = std::fread(buffer, 2, 1, dicom); // Skip 2 reserved bytes
236 
237  // element length
238  rflag = std::fread(buffer, 4, 1, dicom);
239  GetValue(buffer, elementLength4);
240 
241  if(elementLength2 == 0x5153)
242  {
243  if(elementLength4 == 0xFFFFFFFF)
244  {
245  read_undefined_nested( dicom );
246  elementLength4=0;
247  } else{
248  if(read_defined_nested( dicom, elementLength4 )==0){
249  G4Exception("DicomHandler::ReadFile",
250  "DICOM001",
251  FatalException,
252  "Function read_defined_nested() failed!");
253  }
254  }
255  } else {
256  // Reading the information with data
257  rflag = std::fread(data, elementLength4,1,dicom);
258  }
259 
260  } else {
261 
262  // explicit with VR different than previous ones
263  if(!fImplicitEndian || readGroupId == 2) {
264 
265  //G4cout << "Reading DICOM files with Explicit VR"<< G4endl;
266  // element length (2 bytes)
267  rflag = std::fread(buffer, 2, 1, dicom);
268  GetValue(buffer, elementLength2);
269  elementLength4 = elementLength2;
270 
271  rflag = std::fread(data, elementLength4, 1, dicom);
272 
273  } else { // Implicit VR
274 
275  //G4cout << "Reading DICOM files with Implicit VR"<< G4endl;
276 
277  // element length (4 bytes)
278  if(std::fseek(dicom, -2, SEEK_CUR) != 0) {
279  G4Exception("DicomHandler::ReadFile",
280  "DICOM001",
281  FatalException,
282  "fseek failed");
283  }
284 
285  rflag = std::fread(buffer, 4, 1, dicom);
286  GetValue(buffer, elementLength4);
287 
288  //G4cout << std::hex<< elementLength4 << G4endl;
289 
290  if(elementLength4 == 0xFFFFFFFF)
291  {
292  read_undefined_nested(dicom);
293  elementLength4=0;
294  } else{
295  rflag = std::fread(data, elementLength4, 1, dicom);
296  }
297 
298  }
299  }
300 
301  // NULL termination
302  data[elementLength4] = '\0';
303 
304  // analyzing information
305  GetInformation(tagDictionary, data);
306  }
307 
308  G4String fnameG4DCM = G4String(filename2) + ".g4dcm";
309 
310  // Perform functions originally written straight to file
311  DicomPhantomZSliceHeader* zslice = new DicomPhantomZSliceHeader(fnameG4DCM);
312  std::map<G4float,G4String>::const_iterator ite;
313  for( ite = fMaterialIndices.begin(); ite != fMaterialIndices.end(); ++ite){
314  zslice->AddMaterial(ite->second);
315  }
316 
317  zslice->SetNoVoxelX(fColumns/fCompression);
318  zslice->SetNoVoxelY(fRows/fCompression);
319  zslice->SetNoVoxelZ(1);
320 
321  zslice->SetMinX(-fPixelSpacingX*fColumns/2.);
322  zslice->SetMaxX(fPixelSpacingX*fColumns/2.);
323 
324  zslice->SetMinY(-fPixelSpacingY*fRows/2.);
325  zslice->SetMaxY(fPixelSpacingY*fRows/2.);
326 
327  zslice->SetMinZ(fSliceLocation-fSliceThickness/2.);
328  zslice->SetMaxZ(fSliceLocation+fSliceThickness/2.);
329 
330  //===
331 
332  ReadData( dicom, filename2 );
333 
334  // DEPRECIATED
335  //StoreData( foutG4DCM );
336  //foutG4DCM.close();
337 
338  StoreData( zslice );
339 
340  // Dumped 2 file after DicomPhantomZSliceMerged has checked for consistency
341  //zslice->DumpToFile();
342 
343  fMergedSlices->AddZSlice(zslice);
344 
345  //
346  delete [] buffer;
347  delete [] data;
348 
349  if (rflag) return returnvalue;
350  return returnvalue;
351 }

496 void DicomHandler::StoreData(DicomPhantomZSliceHeader* dcmPZSH)
497 {
498  G4int mean;
499  G4double density;
500  G4bool overflow = false;
501 
502  if(!dcmPZSH) { return; }
503 
504  dcmPZSH->SetSliceLocation(fSliceLocation);
505 
506  //----- Print indices of material
507  if(fCompression == 1) { // no fCompression: each pixel has a density value)
508  for( G4int ww = 0; ww < fRows; ww++) {
509  dcmPZSH->AddRow();
510  for( G4int xx = 0; xx < fColumns; xx++) {
511  mean = fTab[ww][xx];
512  density = Pixel2density(mean);
513  dcmPZSH->AddValue(density);
514  dcmPZSH->AddMateID(GetMaterialIndex(density));
515  }
516  }
517 
518  } else {
519  // density value is the average of a square region of
520  // fCompression*fCompression pixels
521  for(G4int ww = 0; ww < fRows ;ww += fCompression ) {
522  dcmPZSH->AddRow();
523  for(G4int xx = 0; xx < fColumns ;xx +=fCompression ) {
524  overflow = false;
525  mean = 0;
526  for(int sumx = 0; sumx < fCompression; sumx++) {
527  for(int sumy = 0; sumy < fCompression; sumy++) {
528  if(ww+sumy >= fRows || xx+sumx >= fColumns) overflow = true;
529  mean += fTab[ww+sumy][xx+sumx];
530  }
531  if(overflow) break;
532  }
533  mean /= fCompression*fCompression;
534 
535  if(!overflow) {
536  density = Pixel2density(mean);
537  dcmPZSH->AddValue(density);
538  dcmPZSH->AddMateID(GetMaterialIndex(density));
539  }
540  }
541  }
542  }
543 
544  dcmPZSH->FlipData();
545 }
