52 ChromosomeParameterisation::ChromosomeParameterisation(const char* filename):
53  G4VPVParameterisation()
54 {
55  ifstream f(filename, ios::in);
56  if (!f)
57  return;
58 
59  fPositions.reserve(100);
60  fRotations.reserve(100);
61 
62  while (!f.eof())
63  {
64  double x, y, z, rot;
65  f >> x >> y >> z >> rot;
66  fPositions.push_back(new G4ThreeVector(x * nanometer,
67  y * nanometer,
68  z * nanometer));
69  fRotations.push_back(new G4RotationMatrix(0, 0, rot * degree));
70  }
71 }
