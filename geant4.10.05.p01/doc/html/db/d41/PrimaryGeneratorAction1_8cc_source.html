58 void PrimaryGeneratorAction1::GeneratePrimaries(G4Event* anEvent)
59 {
60  const G4double r = 2*mm;
61  const G4double zmax = 8*mm;
62 
63  //vertex 1 uniform on cylinder
64  //
65  G4double alpha = twopi*G4UniformRand(); //alpha uniform in (0, 2*pi)
66  G4double ux = std::cos(alpha);
67  G4double uy = std::sin(alpha);
68  G4double z = zmax*(2*G4UniformRand() - 1); //z uniform in (-zmax, +zmax)
69 
70  fParticleGun->SetParticlePosition(G4ThreeVector(r*ux,r*uy,z));
71  fParticleGun->SetParticleMomentumDirection(G4ThreeVector(ux,uy,0));
72  fParticleGun->SetParticleEnergy(1*MeV);
73  fParticleGun->GeneratePrimaryVertex(anEvent);
74 
75  //vertex 2 at opposite
76  //
77  alpha += pi;
78  ux = std::cos(alpha);
79  uy = std::sin(alpha);
80  fParticleGun->SetParticlePosition(G4ThreeVector(r*ux,r*uy,z));
81 
82  //particle 2 at vertex 2
83  //
84  const G4double dalpha = 10*deg;
85  ux = std::cos(alpha + dalpha);
86  uy = std::sin(alpha + dalpha);
87 
88  fParticleGun->SetParticleMomentumDirection(G4ThreeVector(ux,uy,0));
89  fParticleGun->SetParticleEnergy(1*keV);
90  fParticleGun->GeneratePrimaryVertex(anEvent);
91 
92  //particle 3 at vertex 3 (same as vertex 2)
93  //
94  ux = std::cos(alpha - dalpha);
95  uy = std::sin(alpha - dalpha);
96 
97  fParticleGun->SetParticleMomentumDirection(G4ThreeVector(ux,uy,0));
98  fParticleGun->SetParticleEnergy(1*GeV);
99  fParticleGun->GeneratePrimaryVertex(anEvent);
100 
101  // randomize time zero of anEvent
102  //
103  G4double tmin = 0*s, tmax = 10*s;
104  G4double t0 = tmin + (tmax - tmin)*G4UniformRand();
105  G4PrimaryVertex* aVertex = anEvent->GetPrimaryVertex();
106  while (aVertex) {
107  aVertex->SetT0(t0);
108  aVertex = aVertex->GetNext();
109  }
110 }
