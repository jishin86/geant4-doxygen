71 (const G4int copyNo, G4VPhysicalVolume* physVol) const
72 {
73  // Note: copyNo will start with zero!
74  G4double Zposition = fStartZ + copyNo * fSpacing;
75  G4ThreeVector origin(0,0,Zposition);
