462 inline void G4VEmProcess::DefineMaterial(const G4MaterialCutsCouple* couple)
463 {
464  if(couple != currentCouple) {
465  currentCouple = couple;
466  currentMaterial = couple->GetMaterial();
467  baseMaterial = (currentMaterial->GetBaseMaterial())
468  ? currentMaterial->GetBaseMaterial() : currentMaterial;
469  currentCoupleIndex = couple->GetIndex();
470  basedCoupleIndex = (*theDensityIdx)[currentCoupleIndex];
471  fFactor = biasFactor*(*theDensityFactor)[currentCoupleIndex];
472  mfpKinEnergy = DBL_MAX;
473  preStepKinEnergy = 0.0;
474  idxLambda = idxLambdaPrim = 0;
475  }
476 }

523 inline G4double G4VEmProcess::GetCurrentLambda(G4double e)
524 {
525  G4double x(0.0);
526  if(e >= minKinEnergyPrim) { x = GetLambdaFromTablePrim(e); }
527  else if(theLambdaTable) { x = GetLambdaFromTable(e); }
528  else if(currentModel) { x = ComputeCurrentLambda(e); }
529  return fFactor*x;
530 }
