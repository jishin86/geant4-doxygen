167 G4VParticleChange*
168 G4Cerenkov::PostStepDoIt(const G4Track& aTrack, const G4Step& aStep)
169 
170 // This routine is called for each tracking Step of a charged particle
171 // in a radiator. A Poisson-distributed number of photons is generated
172 // according to the Cerenkov formula, distributed evenly along the track
173 // segment and uniformly azimuth w.r.t. the particle direction. The
174 // parameters are then transformed into the Master Reference System, and
175 // they are added to the particle change.
176 
177 {
179  // Should we ensure that the material is dispersive?
181 
182  aParticleChange.Initialize(aTrack);
183 
184  const G4DynamicParticle* aParticle = aTrack.GetDynamicParticle();
185  const G4Material* aMaterial = aTrack.GetMaterial();
186 
187  G4StepPoint* pPreStepPoint = aStep.GetPreStepPoint();
188  G4StepPoint* pPostStepPoint = aStep.GetPostStepPoint();
189 
190  G4ThreeVector x0 = pPreStepPoint->GetPosition();
191  G4ThreeVector p0 = aStep.GetDeltaPosition().unit();
192  G4double t0 = pPreStepPoint->GetGlobalTime();
193 
194  G4MaterialPropertiesTable* aMaterialPropertiesTable =
195  aMaterial->GetMaterialPropertiesTable();
196  if (!aMaterialPropertiesTable) return pParticleChange;
197 
198  G4MaterialPropertyVector* Rindex =
199  aMaterialPropertiesTable->GetProperty(kRINDEX);
200  if (!Rindex) return pParticleChange;
201 
202  // particle charge
203  G4double charge = aParticle->GetDefinition()->GetPDGCharge();
204 
205  // particle beta
206  G4double beta = (pPreStepPoint->GetBeta() + pPostStepPoint->GetBeta())*0.5;
207 
208  //fNumPhotons = 0; // in PostStepGetPhysicalInteractionLength()
209 
210  G4double MeanNumberOfPhotons =
211  GetAverageNumberOfPhotons(charge,beta,aMaterial,Rindex);
212 
213  if (MeanNumberOfPhotons <= 0.0) {
214 
215  // return unchanged particle and no secondaries
216 
217  aParticleChange.SetNumberOfSecondaries(0);
218 
219  return pParticleChange;
220 
221  }
222 
223  G4double step_length = aStep.GetStepLength();
224 
225  MeanNumberOfPhotons = MeanNumberOfPhotons * step_length;
226 
227  fNumPhotons = (G4int) G4Poisson(MeanNumberOfPhotons);
228 
229  if ( fNumPhotons <= 0 || !fStackingFlag ) {
230 
231  // return unchanged particle and no secondaries
232 
233  aParticleChange.SetNumberOfSecondaries(0);
234 
235  return pParticleChange;
236 
237  }
238 
240 
241  aParticleChange.SetNumberOfSecondaries(fNumPhotons);
242 
243  if (fTrackSecondariesFirst) {
244  if (aTrack.GetTrackStatus() == fAlive )
245  aParticleChange.ProposeTrackStatus(fSuspend);
246  }
247 
249 
250  G4double Pmin = Rindex->GetMinLowEdgeEnergy();
251  G4double Pmax = Rindex->GetMaxLowEdgeEnergy();
252  G4double dp = Pmax - Pmin;
253 
254  G4double nMax = Rindex->GetMaxValue();
255 
256  G4double BetaInverse = 1./beta;
257 
258  G4double maxCos = BetaInverse / nMax;
259  G4double maxSin2 = (1.0 - maxCos) * (1.0 + maxCos);
260 
261  G4double beta1 = pPreStepPoint ->GetBeta();
262  G4double beta2 = pPostStepPoint->GetBeta();
263 
264  G4double MeanNumberOfPhotons1 =
265  GetAverageNumberOfPhotons(charge,beta1,aMaterial,Rindex);
266  G4double MeanNumberOfPhotons2 =
267  GetAverageNumberOfPhotons(charge,beta2,aMaterial,Rindex);
268 
269  for (G4int i = 0; i < fNumPhotons; i++) {
270 
271  // Determine photon energy
272 
273  G4double rand;
274  G4double sampledEnergy, sampledRI;
275  G4double cosTheta, sin2Theta;
276 
277  // sample an energy
278 
279  do {
280  rand = G4UniformRand();
281  sampledEnergy = Pmin + rand * dp;
282  sampledRI = Rindex->Value(sampledEnergy);
283  cosTheta = BetaInverse / sampledRI;
284 
285  sin2Theta = (1.0 - cosTheta)*(1.0 + cosTheta);
286  rand = G4UniformRand();
287 
288  // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
289  } while (rand*maxSin2 > sin2Theta);
290 
291  // Generate random position of photon on cone surface
292  // defined by Theta
293 
294  rand = G4UniformRand();
295 
296  G4double phi = twopi*rand;
297  G4double sinPhi = std::sin(phi);
298  G4double cosPhi = std::cos(phi);
299 
300  // calculate x,y, and z components of photon energy
301  // (in coord system with primary particle direction
302  // aligned with the z axis)
303 
304  G4double sinTheta = std::sqrt(sin2Theta);
305  G4double px = sinTheta*cosPhi;
306  G4double py = sinTheta*sinPhi;
307  G4double pz = cosTheta;
308 
309  // Create photon momentum direction vector
310  // The momentum direction is still with respect
311  // to the coordinate system where the primary
312  // particle direction is aligned with the z axis
313 
314  G4ParticleMomentum photonMomentum(px, py, pz);
315 
316  // Rotate momentum direction back to global reference
317  // system
318 
319  photonMomentum.rotateUz(p0);
320 
321  // Determine polarization of new photon
322 
323  G4double sx = cosTheta*cosPhi;
324  G4double sy = cosTheta*sinPhi;
325  G4double sz = -sinTheta;
326 
327  G4ThreeVector photonPolarization(sx, sy, sz);
328 
329  // Rotate back to original coord system
330 
331  photonPolarization.rotateUz(p0);
332 
333  // Generate a new photon:
334 
335  G4DynamicParticle* aCerenkovPhoton =
336  new G4DynamicParticle(G4OpticalPhoton::OpticalPhoton(),photonMomentum);
337 
338  aCerenkovPhoton->SetPolarization(photonPolarization.x(),
339  photonPolarization.y(),
340  photonPolarization.z());
341 
342  aCerenkovPhoton->SetKineticEnergy(sampledEnergy);
343 
344  // Generate new G4Track object:
345 
346  G4double NumberOfPhotons, N;
347 
348  do {
349  rand = G4UniformRand();
350  NumberOfPhotons = MeanNumberOfPhotons1 - rand *
351  (MeanNumberOfPhotons1-MeanNumberOfPhotons2);
352  N = G4UniformRand() *
353  std::max(MeanNumberOfPhotons1,MeanNumberOfPhotons2);
354  // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
355  } while (N > NumberOfPhotons);
356 
357  G4double delta = rand * aStep.GetStepLength();
358 
359  G4double deltaTime = delta / (pPreStepPoint->GetVelocity()+
360  rand*(pPostStepPoint->GetVelocity()-
361  pPreStepPoint->GetVelocity())*0.5);
362 
363  G4double aSecondaryTime = t0 + deltaTime;
364 
365  G4ThreeVector aSecondaryPosition = x0 + rand * aStep.GetDeltaPosition();
366 
367  G4Track* aSecondaryTrack =
368  new G4Track(aCerenkovPhoton,aSecondaryTime,aSecondaryPosition);
369 
370  aSecondaryTrack->SetTouchableHandle(
371  aStep.GetPreStepPoint()->GetTouchableHandle());
372 
373  aSecondaryTrack->SetParentID(aTrack.GetTrackID());
374 
375  aParticleChange.AddSecondary(aSecondaryTrack);
376  }
377 
378  if (verboseLevel>0) {
379  G4cout <<"\n Exiting from G4Cerenkov::DoIt -- NumberOfSecondaries = "
380  << aParticleChange.GetNumberOfSecondaries() << G4endl;
381  }
382 
383  return pParticleChange;
384 }

390 void G4Cerenkov::BuildThePhysicsTable()
391 {
392  if (thePhysicsTable) return;
393 
394  const G4MaterialTable* theMaterialTable=
395  G4Material::GetMaterialTable();
396  G4int numOfMaterials = G4Material::GetNumberOfMaterials();
397 
398  // create new physics table
399 
400  thePhysicsTable = new G4PhysicsTable(numOfMaterials);
401 
402  // loop for materials
403 
404  for (G4int i=0 ; i < numOfMaterials; i++) {
405 
406  G4PhysicsOrderedFreeVector* aPhysicsOrderedFreeVector = 0;
407 
408  // Retrieve vector of refraction indices for the material
409  // from the material's optical properties table
410 
411  G4Material* aMaterial = (*theMaterialTable)[i];
412 
413  G4MaterialPropertiesTable* aMaterialPropertiesTable =
414  aMaterial->GetMaterialPropertiesTable();
415 
416  if (aMaterialPropertiesTable) {
417  aPhysicsOrderedFreeVector = new G4PhysicsOrderedFreeVector();
418  G4MaterialPropertyVector* theRefractionIndexVector =
419  aMaterialPropertiesTable->GetProperty(kRINDEX);
420 
421  if (theRefractionIndexVector) {
422 
423  // Retrieve the first refraction index in vector
424  // of (photon energy, refraction index) pairs
425 
426  G4double currentRI = (*theRefractionIndexVector)[0];
427 
428  if (currentRI > 1.0) {
429 
430  // Create first (photon energy, Cerenkov Integral)
431  // pair
432 
433  G4double currentPM = theRefractionIndexVector->Energy(0);
434  G4double currentCAI = 0.0;
435 
436  aPhysicsOrderedFreeVector->InsertValues(currentPM , currentCAI);
437 
438  // Set previous values to current ones prior to loop
439 
440  G4double prevPM = currentPM;
441  G4double prevCAI = currentCAI;
442  G4double prevRI = currentRI;
443 
444  // loop over all (photon energy, refraction index)
445  // pairs stored for this material
446 
447  for (size_t ii = 1;
448  ii < theRefractionIndexVector->GetVectorLength();
449  ++ii) {
450  currentRI = (*theRefractionIndexVector)[ii];
451  currentPM = theRefractionIndexVector->Energy(ii);
452 
453  currentCAI = 0.5*(1.0/(prevRI*prevRI) +
454  1.0/(currentRI*currentRI));
455 
456  currentCAI = prevCAI + (currentPM - prevPM) * currentCAI;
457 
458  aPhysicsOrderedFreeVector->
459  InsertValues(currentPM, currentCAI);
460 
461  prevPM = currentPM;
462  prevCAI = currentCAI;
463  prevRI = currentRI;
464  }
465 
466  }
467  }
468  }
469 
470  // The Cerenkov integral for a given material
471  // will be inserted in thePhysicsTable
472  // according to the position of the material in
473  // the material table.
474 
475  thePhysicsTable->insertAt(i,aPhysicsOrderedFreeVector);
476 
477  }
478 }

491 G4double G4Cerenkov::PostStepGetPhysicalInteractionLength(
492  const G4Track& aTrack,
493  G4double,
494  G4ForceCondition* condition)
495 {
496  *condition = NotForced;
497  G4double StepLimit = DBL_MAX;
498  fNumPhotons = 0;
499 
500  const G4Material* aMaterial = aTrack.GetMaterial();
501  G4int materialIndex = aMaterial->GetIndex();
502 
503  // If Physics Vector is not defined no Cerenkov photons
504  // this check avoid string comparison below
505 
506  if(!(*thePhysicsTable)[materialIndex]) { return StepLimit; }
507 
508  const G4DynamicParticle* aParticle = aTrack.GetDynamicParticle();
509  const G4MaterialCutsCouple* couple = aTrack.GetMaterialCutsCouple();
510 
511  G4double kineticEnergy = aParticle->GetKineticEnergy();
512  const G4ParticleDefinition* particleType = aParticle->GetDefinition();
513  G4double mass = particleType->GetPDGMass();
514 
515  // particle beta
516  G4double beta = aParticle->GetTotalMomentum() /
517  aParticle->GetTotalEnergy();
518  // particle gamma
519  G4double gamma = aParticle->GetTotalEnergy()/mass;
520 
521  G4MaterialPropertiesTable* aMaterialPropertiesTable =
522  aMaterial->GetMaterialPropertiesTable();
523 
524  G4MaterialPropertyVector* Rindex = NULL;
525 
526  if (aMaterialPropertiesTable)
527  Rindex = aMaterialPropertiesTable->GetProperty(kRINDEX);
528 
529  G4double nMax;
530  if (Rindex) {
531  nMax = Rindex->GetMaxValue();
532  } else {
533  return StepLimit;
534  }
535 
536  G4double BetaMin = 1./nMax;
537  if ( BetaMin >= 1. ) return StepLimit;
538 
539  G4double GammaMin = 1./std::sqrt(1.-BetaMin*BetaMin);
540 
541  if (gamma < GammaMin ) return StepLimit;
542 
543  G4double kinEmin = mass*(GammaMin-1.);
544 
545  G4double RangeMin = G4LossTableManager::Instance()->GetRange(particleType,
546  kinEmin,
547  couple);
548  G4double Range = G4LossTableManager::Instance()->GetRange(particleType,
549  kineticEnergy,
550  couple);
551 
552  G4double Step = Range - RangeMin;
553 
554  // If the step is smaller than 1e-16 mm, it may happen that the particle
555  // does not move. See bug 1992.
556  // 2019-03-11: change to 1e-15
557  if (Step < 1.e-15*mm) return StepLimit;
558 
559  if (Step < StepLimit) StepLimit = Step;
560  // If user has defined an average maximum number of photons to
561  // be generated in a Step, then calculate the Step length for
562  // that number of photons.
563 
564  if (fMaxPhotons > 0) {
565 
566  // particle charge
567  const G4double charge = aParticle->GetDefinition()->GetPDGCharge();
568 
569  G4double MeanNumberOfPhotons =
570  GetAverageNumberOfPhotons(charge,beta,aMaterial,Rindex);
571 
572  Step = 0.;
573  if (MeanNumberOfPhotons > 0.0) Step = fMaxPhotons / MeanNumberOfPhotons;
574 
575  if (Step > 0. && Step < StepLimit) StepLimit = Step;
576  }
577 
578  // If user has defined an maximum allowed change in beta per step
579  if (fMaxBetaChange > 0.) {
580 
581  G4double dedx = G4LossTableManager::Instance()->GetDEDX(particleType,
582  kineticEnergy,
583  couple);
584 
585  G4double deltaGamma = gamma - 1./std::sqrt(1.-beta*beta*
586  (1.-fMaxBetaChange)*
587  (1.-fMaxBetaChange));
588 
589  Step = mass * deltaGamma / dedx;
590 
591  if (Step > 0. && Step < StepLimit) StepLimit = Step;
592 
593  }
594 
595  *condition = StronglyForced;
596  return StepLimit;
597 }

599 // GetAverageNumberOfPhotons
600 // -------------------------
601 // This routine computes the number of Cerenkov photons produced per
602 // GEANT-unit (millimeter) in the current medium.
603 // ^^^^^^^^^^
604 
605 G4double
606  G4Cerenkov::GetAverageNumberOfPhotons(const G4double charge,
607  const G4double beta,
608  const G4Material* aMaterial,
609  G4MaterialPropertyVector* Rindex) const
610 {
611  const G4double Rfact = 369.81/(eV * cm);
612 
613  if(beta <= 0.0)return 0.0;
614 
615  G4double BetaInverse = 1./beta;
616 
617  // Vectors used in computation of Cerenkov Angle Integral:
618  // - Refraction Indices for the current material
619  // - new G4PhysicsOrderedFreeVector allocated to hold CAI's
620 
621  G4int materialIndex = aMaterial->GetIndex();
622 
623  // Retrieve the Cerenkov Angle Integrals for this material
624 
625  G4PhysicsOrderedFreeVector* CerenkovAngleIntegrals =
626  (G4PhysicsOrderedFreeVector*)((*thePhysicsTable)(materialIndex));
627 
628  if(!(CerenkovAngleIntegrals->IsFilledVectorExist()))return 0.0;
629 
630  // Min and Max photon energies
631  G4double Pmin = Rindex->GetMinLowEdgeEnergy();
632  G4double Pmax = Rindex->GetMaxLowEdgeEnergy();
633 
634  // Min and Max Refraction Indices
635  G4double nMin = Rindex->GetMinValue();
636  G4double nMax = Rindex->GetMaxValue();
637 
638  // Max Cerenkov Angle Integral
639  G4double CAImax = CerenkovAngleIntegrals->GetMaxValue();
640 
641  G4double dp, ge;
642 
643  // If n(Pmax) < 1/Beta -- no photons generated
644 
645  if (nMax < BetaInverse) {
646  dp = 0.0;
647  ge = 0.0;
648  }
649 
650  // otherwise if n(Pmin) >= 1/Beta -- photons generated
651 
652  else if (nMin > BetaInverse) {
653  dp = Pmax - Pmin;
654  ge = CAImax;
655  }
656 
657  // If n(Pmin) < 1/Beta, and n(Pmax) >= 1/Beta, then
658  // we need to find a P such that the value of n(P) == 1/Beta.
659  // Interpolation is performed by the GetEnergy() and
660  // Value() methods of the G4MaterialPropertiesTable and
661  // the GetValue() method of G4PhysicsVector.
662 
663  else {
664  Pmin = Rindex->GetEnergy(BetaInverse);
665  dp = Pmax - Pmin;
666 
667  // need boolean for current implementation of G4PhysicsVector
668  // ==> being phased out
669  G4bool isOutRange;
670  G4double CAImin = CerenkovAngleIntegrals->GetValue(Pmin, isOutRange);
671  ge = CAImax - CAImin;
672 
673  if (verboseLevel>0) {
674  G4cout << "CAImin = " << CAImin << G4endl;
675  G4cout << "ge = " << ge << G4endl;
676  }
677  }
678 
679  // Calculate number of photons
680  G4double NumPhotons = Rfact * charge/eplus * charge/eplus *
681  (dp - ge * BetaInverse*BetaInverse);
682 
683  return NumPhotons;
684 }
