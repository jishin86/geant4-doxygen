111 G4ReactionProductVector* G4GeneratorPrecompoundInterface::
112 Propagate(G4KineticTrackVector* theSecondaries, G4V3DNucleus* theNucleus)
113 {
114  #ifdef debugPrecoInt
115  G4cout<<G4endl<<"G4GeneratorPrecompoundInterface::Propagate"<<G4endl;
116  G4cout<<"Target A and Z "<<theNucleus->GetMassNumber()<<" "<<theNucleus->GetCharge()<<G4endl;
117  G4cout<<"Directly produced particles number "<<theSecondaries->size()<<G4endl;
118  #endif
119 
120  G4ReactionProductVector * theTotalResult = new G4ReactionProductVector;
121 
122  // decay the strong resonances
123  G4DecayKineticTracks decay(theSecondaries);
124  #ifdef debugPrecoInt
125  G4cout<<"Final stable particles number "<<theSecondaries->size()<<G4endl;
126  #endif
127 
128  // prepare the fragment
129  G4int anA=theNucleus->GetMassNumber();
130  G4int aZ=theNucleus->GetCharge();
131 // G4double TargetNucleusMass = G4NucleiProperties::GetNuclearMass(anA, aZ);
132 
133  G4int numberOfEx = 0;
134  G4int numberOfCh = 0;
135  G4int numberOfHoles = 0;
136 
137  G4double R = theNucleus->GetNuclearRadius();
138 
139  G4LorentzVector captured4Momentum(0.,0.,0.,0.);
140  G4LorentzVector Residual4Momentum(0.,0.,0.,0.); // TargetNucleusMass is not need at the moment
141  G4LorentzVector Secondary4Momentum(0.,0.,0.,0.);
142 
143  // loop over secondaries
144  G4KineticTrackVector::iterator iter;
145  for(iter=theSecondaries->begin(); iter !=theSecondaries->end(); ++iter)
146  {
147  const G4ParticleDefinition* part = (*iter)->GetDefinition();
148  G4double e = (*iter)->Get4Momentum().e();
149  G4double mass = (*iter)->Get4Momentum().mag();
150  G4ThreeVector mom = (*iter)->Get4Momentum().vect();
151  if((part != proton && part != neutron) ||
152 // Uzhi 2.05.2015 (e > mass + CaptureThreshold) ||
153  ((*iter)->GetPosition().mag() > R)) {
154  G4ReactionProduct * theNew = new G4ReactionProduct(part);
155  theNew->SetMomentum(mom);
156  theNew->SetTotalEnergy(e);
157  theTotalResult->push_back(theNew);
158  Secondary4Momentum += (*iter)->Get4Momentum(); // Uzhi 29 April
159  #ifdef debugPrecoInt
160  G4cout<<"Secondary 4Mom "<<part->GetParticleName()<<" "<<(*iter)->Get4Momentum()<<" "
161  <<(*iter)->Get4Momentum().mag()<<G4endl;
162  #endif
163  } else {
164  if( e-mass > -CaptureThreshold*G4Log( G4UniformRand()) ) { // Added by Uzhi 2.05.2015
165  G4ReactionProduct * theNew = new G4ReactionProduct(part);
166  theNew->SetMomentum(mom);
167  theNew->SetTotalEnergy(e);
168  theTotalResult->push_back(theNew);
169  Secondary4Momentum += (*iter)->Get4Momentum(); // Uzhi 29 April
170  #ifdef debugPrecoInt
171  G4cout<<"Secondary 4Mom "<<part->GetParticleName()<<" "<<(*iter)->Get4Momentum()<<" "
172  <<(*iter)->Get4Momentum().mag()<<G4endl;
173  #endif
174  } else {
175  // within the nucleus, neutron or proton
176  // now calculate A, Z of the fragment, momentum, number of exciton states
177  ++anA;
178  ++numberOfEx;
179  G4int Z = G4int(part->GetPDGCharge()/eplus + 0.1);
180  aZ += Z;
181  numberOfCh += Z;
182  captured4Momentum += (*iter)->Get4Momentum();
183  #ifdef debugPrecoInt
184  G4cout<<"Captured 4Mom "<<part->GetParticleName()<<(*iter)->Get4Momentum()<<G4endl;
185  #endif
186  }
187  }
188  delete (*iter);
189  }
190  delete theSecondaries;
191 
192  // loop over wounded nucleus
193  G4Nucleon * theCurrentNucleon =
194  theNucleus->StartLoop() ? theNucleus->GetNextNucleon() : 0;
195  while(theCurrentNucleon) /* Loop checking, 31.08.2015, G.Folger */
196  {
197  if(theCurrentNucleon->AreYouHit()) {
198  ++numberOfHoles;
199  ++numberOfEx;
200  --anA;
201  aZ -= G4int(theCurrentNucleon->GetDefinition()->GetPDGCharge()/eplus + 0.1);
202 
203  Residual4Momentum -= theCurrentNucleon->Get4Momentum();
204  }
205  theCurrentNucleon = theNucleus->GetNextNucleon();
206  }
207 
208  #ifdef debugPrecoInt
209  G4cout<<G4endl;
210  G4cout<<"Secondary 4Mom "<<Secondary4Momentum<<G4endl;
211  G4cout<<"Captured 4Mom "<<captured4Momentum<<G4endl;
212  G4cout<<"Sec + Captured "<<Secondary4Momentum+captured4Momentum<<G4endl;
213  G4cout<<"Residual4Mom "<<Residual4Momentum<<G4endl;
214  G4cout<<"Sum 4 momenta "
215  <<Secondary4Momentum + captured4Momentum + Residual4Momentum <<G4endl;
216  #endif
217 
218  // Check that we use QGS model; loop over wounded nucleus
219  G4bool QGSM(false);
220  theCurrentNucleon = theNucleus->StartLoop() ? theNucleus->GetNextNucleon() : 0;
221  while(theCurrentNucleon) /* Loop checking, 31.08.2015, G.Folger */
222  {
223  if(theCurrentNucleon->AreYouHit())
224  {
225  if(theCurrentNucleon->Get4Momentum().mag() <
226  theCurrentNucleon->GetDefinition()->GetPDGMass()) QGSM=true;
227  }
228  theCurrentNucleon = theNucleus->GetNextNucleon();
229  }
230 
231  #ifdef debugPrecoInt
232  if(!QGSM){
233  G4cout<<G4endl;
234  G4cout<<"Residual A and Z "<<anA<<" "<<aZ<<G4endl;
235  G4cout<<"Residual 4Mom "<<Residual4Momentum<<G4endl;
236  if(numberOfEx == 0)
237  {G4cout<<"Residual 4Mom = 0 means that there were not wounded and captured nucleons"<<G4endl;}
238  }
239  #endif
240 
241  if(anA == 0) return theTotalResult;
242 
243  G4LorentzVector exciton4Momentum(0.,0.,0.,0.); // Uzhi 29 April
244  if(anA >= aZ)
245  {
246  if(!QGSM)
247  { // FTF model was used
248  G4double fMass = G4NucleiProperties::GetNuclearMass(anA, aZ);
249 
250 // G4LorentzVector exciton4Momentum = Residual4Momentum + captured4Momentum;
251  exciton4Momentum = Residual4Momentum + captured4Momentum;
252 //exciton4Momentum.setE(std::sqrt(exciton4Momentum.vect().mag2()+sqr(fMass)));
253  G4double ActualMass = exciton4Momentum.mag();
254  if(ActualMass <= fMass ) { //E*<=0, Uzhi 5.05.2015
255  exciton4Momentum.setE(std::sqrt(exciton4Momentum.vect().mag2()+sqr(fMass))); // Uzhi 13.05.2015
256  }
257 
258  #ifdef debugPrecoInt
259  G4double exEnergy = 0.0;
260  if(ActualMass <= fMass ) {exEnergy = 0.;} // Uzhi 5.05.2015
261  else {exEnergy = ActualMass - fMass;}
262  G4cout<<"Ground state residual Mass "<<fMass<<" E* "<<exEnergy<<G4endl;
263  #endif
264  }
265  else
266  { // QGS model was used
267  G4double InitialTargetMass =
268  G4NucleiProperties::GetNuclearMass(theNucleus->GetMassNumber(), theNucleus->GetCharge());
269 
270  exciton4Momentum =
271  GetPrimaryProjectile()->Get4Momentum() + G4LorentzVector(0.,0.,0.,InitialTargetMass)
272  -Secondary4Momentum;
273 
274  G4double fMass = G4NucleiProperties::GetNuclearMass(anA, aZ);
275  G4double ActualMass = exciton4Momentum.mag();
276 
277  #ifdef debugPrecoInt
278  G4cout<<G4endl;
279  G4cout<<"Residual A and Z "<<anA<<" "<<aZ<<G4endl;
280  G4cout<<"Residual4Momentum "<<exciton4Momentum<<G4endl;
281  G4cout<<"ResidualMass, GroundStateMass and E* "<<ActualMass<<" "<<fMass<<" "
282  <<ActualMass - fMass<<G4endl;
283  #endif
284 
285  if(ActualMass - fMass < 0.)
286  {
287  G4double ResE = std::sqrt(exciton4Momentum.vect().mag2() + sqr(fMass+10*MeV));
288  exciton4Momentum.setE(ResE);
289  #ifdef debugPrecoInt
290  G4cout<<"ActualMass - fMass < 0. "<<ActualMass<<" "<<fMass<<" "<<ActualMass - fMass<<G4endl;
291  G4int Uzhi; G4cin>>Uzhi;
292  #endif
293  }
294  }
295  // Need to de-excite the remnant nucleus only if excitation energy > 0.
296  G4Fragment anInitialState(anA, aZ, exciton4Momentum);
297  anInitialState.SetNumberOfParticles(numberOfEx-numberOfHoles);
298  anInitialState.SetNumberOfCharged(numberOfCh);
299  anInitialState.SetNumberOfHoles(numberOfHoles);
300 
301  G4ReactionProductVector * aPrecoResult =
302  theDeExcitation->DeExcite(anInitialState);
303  // fill pre-compound part into the result, and return
304  #ifdef debugPrecoInt
305  G4cout<<"Target fragment number "<<aPrecoResult->size()<<G4endl;
306  #endif
307  for(unsigned int ll=0; ll<aPrecoResult->size(); ++ll)
308  {
309  theTotalResult->push_back(aPrecoResult->operator[](ll));
310  #ifdef debugPrecoInt
311  G4cout<<"Fragment "<<ll<<" "
312  <<aPrecoResult->operator[](ll)->GetDefinition()->GetParticleName()<<" "
313  <<aPrecoResult->operator[](ll)->GetMomentum()<<" "
314  <<aPrecoResult->operator[](ll)->GetTotalEnergy()<<" "
315  <<aPrecoResult->operator[](ll)->GetDefinition()->GetPDGMass()<<G4endl;
316  #endif
317  }
318  delete aPrecoResult;
319  }
320 
321  return theTotalResult;
322 }

348 G4ReactionProductVector* G4GeneratorPrecompoundInterface::
349 PropagateNuclNucl(G4KineticTrackVector* theSecondaries, G4V3DNucleus* theNucleus,
350  G4V3DNucleus* theProjectileNucleus)
351 {
352 #ifdef debugPrecoInt
353  G4cout<<G4endl<<"G4GeneratorPrecompoundInterface::PropagateNuclNucl "<<G4endl;
354  G4cout<<"Projectile A and Z "<<theProjectileNucleus->GetMassNumber()<<" "
355  <<theProjectileNucleus->GetCharge()<<G4endl;
356  G4cout<<"Target A and Z "<<theNucleus->GetMassNumber()<<" "
357  <<theNucleus->GetCharge()<<G4endl;
358  G4cout<<"Directly produced particles number "<<theSecondaries->size()<<G4endl;
359  G4cout<<"Projectile 4Mom and mass "<<GetPrimaryProjectile()->Get4Momentum()<<" "
360  <<GetPrimaryProjectile()->Get4Momentum().mag()<<G4endl<<G4endl;
361 #endif
362 
363  // prepare the target residual
364  G4int anA=theNucleus->GetMassNumber();
365  G4int aZ=theNucleus->GetCharge();
366  G4int numberOfEx = 0;
367  G4int numberOfCh = 0;
368  G4int numberOfHoles = 0;
369  G4double exEnergy = 0.0;
370  G4double R = theNucleus->GetNuclearRadius();
371  G4LorentzVector Target4Momentum(0.,0.,0.,0.);
372 
373  // loop over wounded target nucleus
374  G4Nucleon * theCurrentNucleon =
375  theNucleus->StartLoop() ? theNucleus->GetNextNucleon() : 0;
376  while(theCurrentNucleon) /* Loop checking, 31.08.2015, G.Folger */
377  {
378  if(theCurrentNucleon->AreYouHit()) {
379  ++numberOfHoles;
380  ++numberOfEx;
381  --anA;
382  aZ -= G4int(theCurrentNucleon->GetDefinition()->GetPDGCharge()/
383  eplus + 0.1);
384  exEnergy += theCurrentNucleon->GetBindingEnergy();
385  Target4Momentum -=theCurrentNucleon->Get4Momentum();
386  }
387  theCurrentNucleon = theNucleus->GetNextNucleon();
388  }
389 
390 #ifdef debugPrecoInt
391  G4cout<<"Residual Target A Z E* 4mom "<<anA<<" "<<aZ<<" "<<exEnergy<<" "
392  <<Target4Momentum<<G4endl;
393 #endif
394 
395  // prepare the projectile residual
396 
397  G4bool ProjectileIsAntiNucleus=
398  GetPrimaryProjectile()->GetDefinition()->GetBaryonNumber() < -1;
399 
400  G4ThreeVector bst = GetPrimaryProjectile()->Get4Momentum().boostVector();
401 
402  G4int anAb=theProjectileNucleus->GetMassNumber();
403  G4int aZb=theProjectileNucleus->GetCharge();
404  G4int numberOfExB = 0;
405  G4int numberOfChB = 0;
406  G4int numberOfHolesB = 0;
407  G4double exEnergyB = 0.0;
408  G4double Rb = theProjectileNucleus->GetNuclearRadius();
409  G4LorentzVector Projectile4Momentum(0.,0.,0.,0.);
410 
411  // loop over wounded projectile nucleus
412  theCurrentNucleon =
413  theProjectileNucleus->StartLoop() ? theProjectileNucleus->GetNextNucleon() : 0;
414  while(theCurrentNucleon) /* Loop checking, 31.08.2015, G.Folger */
415  {
416  if(theCurrentNucleon->AreYouHit()) {
417  ++numberOfHolesB;
418  ++numberOfExB;
419  --anAb;
420  if(!ProjectileIsAntiNucleus) {
421  aZb -= G4int(theCurrentNucleon->GetDefinition()->GetPDGCharge()/
422  eplus + 0.1);
423  } else {
424  aZb += G4int(theCurrentNucleon->GetDefinition()->GetPDGCharge()/
425  eplus - 0.1);
426  }
427  exEnergyB += theCurrentNucleon->GetBindingEnergy();
428  Projectile4Momentum -=theCurrentNucleon->Get4Momentum();
429  }
430  theCurrentNucleon = theProjectileNucleus->GetNextNucleon();
431  }
432 
433  G4bool ExistTargetRemnant = G4double (numberOfHoles) <
434  0.3* G4double (numberOfHoles + anA);
435  G4bool ExistProjectileRemnant= G4double (numberOfHolesB) <
436  0.3*G4double (numberOfHolesB + anAb);
437 
438 #ifdef debugPrecoInt
439  G4cout<<"Projectile residual A Z E* 4mom "<<anAb<<" "<<aZb<<" "<<exEnergyB<<" "
440  <<Projectile4Momentum<<G4endl;
441  G4cout<<" ExistTargetRemnant ExistProjectileRemnant "
442  <<ExistTargetRemnant<<" "<< ExistProjectileRemnant<<G4endl;
443 #endif
444  //-----------------------------------------------------------------------------
445  // decay the strong resonances
446  G4ReactionProductVector * theTotalResult = new G4ReactionProductVector;
447  G4DecayKineticTracks decay(theSecondaries);
448  #ifdef debugPrecoInt
449  G4cout<<"Secondary stable particles number "<<theSecondaries->size()<<G4endl;
450  #endif
451 
452 #ifdef debugPrecoInt
453  G4LorentzVector secondary4Momemtum(0,0,0,0);
454  G4int SecondrNum(0);
455 #endif
456 
457  // loop over secondaries
458  G4KineticTrackVector::iterator iter;
459  for(iter=theSecondaries->begin(); iter !=theSecondaries->end(); ++iter)
460  {
461  const G4ParticleDefinition* part = (*iter)->GetDefinition();
462  G4LorentzVector aTrack4Momentum=(*iter)->Get4Momentum();
463 
464  if( part != proton && part != neutron &&
465  (part != ANTIproton && ProjectileIsAntiNucleus) &&
466  (part != ANTIneutron && ProjectileIsAntiNucleus) )
467  {
468  G4ReactionProduct * theNew = new G4ReactionProduct(part);
469  theNew->SetMomentum(aTrack4Momentum.vect());
470  theNew->SetTotalEnergy(aTrack4Momentum.e());
471  theTotalResult->push_back(theNew);
472 #ifdef debugPrecoInt
473  SecondrNum++;
474  secondary4Momemtum += (*iter)->Get4Momentum();
475  G4cout<<"Secondary "<<SecondrNum<<" "
476  <<theNew->GetDefinition()->GetParticleName()<<" "
477  <<theNew->GetMomentum()<<" "<<theNew->GetTotalEnergy()<<G4endl;
478 
479 #endif
480  delete (*iter);
481  continue;
482  }
483 
484  G4bool CanBeCapturedByTarget = false;
485  if( part == proton || part == neutron)
486  {
487  CanBeCapturedByTarget = ExistTargetRemnant &&
488  (-CaptureThreshold*G4Log( G4UniformRand()) >
489  (aTrack4Momentum + Target4Momentum).mag() -
490  aTrack4Momentum.mag() - Target4Momentum.mag()) &&
491  ((*iter)->GetPosition().mag() < R);
492  }
493  // ---------------------------
494  G4LorentzVector Position((*iter)->GetPosition(), (*iter)->GetFormationTime());
495  Position.boost(bst);
496 
497  G4bool CanBeCapturedByProjectile = false;
498 
499  if( !ProjectileIsAntiNucleus &&
500  ( part == proton || part == neutron))
501  {
502  CanBeCapturedByProjectile = ExistProjectileRemnant &&
503  (-CaptureThreshold*G4Log( G4UniformRand()) >
504  (aTrack4Momentum + Projectile4Momentum).mag() -
505  aTrack4Momentum.mag() - Projectile4Momentum.mag()) &&
506  (Position.vect().mag() < Rb);
507  }
508 
509  if( ProjectileIsAntiNucleus &&
510  ( part == ANTIproton || part == ANTIneutron))
511  {
512  CanBeCapturedByProjectile = ExistProjectileRemnant &&
513  (-CaptureThreshold*G4Log( G4UniformRand()) >
514  (aTrack4Momentum + Projectile4Momentum).mag() -
515  aTrack4Momentum.mag() - Projectile4Momentum.mag()) &&
516  (Position.vect().mag() < Rb);
517  }
518 
519  if(CanBeCapturedByTarget && CanBeCapturedByProjectile)
520  {
521  if(G4UniformRand() < 0.5)
522  { CanBeCapturedByTarget = true; CanBeCapturedByProjectile = false;}
523  else
524  { CanBeCapturedByTarget = false; CanBeCapturedByProjectile = true;}
525  }
526 
527  if(CanBeCapturedByTarget)
528  {
529  // within the target nucleus, neutron or proton
530  // now calculate A, Z of the fragment, momentum,
531  // number of exciton states
532 #ifdef debugPrecoInt
533  G4cout<<"Track is CapturedByTarget "<<" "<<part->GetParticleName()<<" "
534  <<aTrack4Momentum<<" "<<aTrack4Momentum.mag()<<G4endl;
535 #endif
536  ++anA;
537  ++numberOfEx;
538  G4int Z = G4int(part->GetPDGCharge()/eplus + 0.1);
539  aZ += Z;
540  numberOfCh += Z;
541  Target4Momentum +=aTrack4Momentum;
542  delete (*iter);
543  } else if(CanBeCapturedByProjectile)
544  {
545  // within the projectile nucleus, neutron or proton
546  // now calculate A, Z of the fragment, momentum,
