302 void G4VITProcess::SubtractNumberOfInteractionLengthLeft(G4double previousStepSize)
303 {
304  if (fpState->currentInteractionLength > 0.0)
305  {
306  fpState->theNumberOfInteractionLengthLeft -= previousStepSize
307  / fpState->currentInteractionLength;
308  if (fpState->theNumberOfInteractionLengthLeft < 0.)
309  {
310  fpState->theNumberOfInteractionLengthLeft = CLHEP::perMillion;
311  }
312 
313  }
314  else
315  {
316 #ifdef G4VERBOSE
317  if (verboseLevel > 0)
318  {
319  G4cerr << "G4VITProcess::SubtractNumberOfInteractionLengthLeft()";
320  G4cerr << " [" << theProcessName << "]" << G4endl;
321  G4cerr << " currentInteractionLength = "
322  << fpState->currentInteractionLength << " [mm]";
323  G4cerr << " previousStepSize = " << previousStepSize << " [mm]";
324  G4cerr << G4endl;
325  }
326 #endif
327  G4String msg = "Negative currentInteractionLength for ";
328  msg += theProcessName;
329  G4Exception("G4VITProcess::SubtractNumberOfInteractionLengthLeft()",
330  "ProcMan201",EventMustBeAborted,
331  msg);
332  }
