58 G4bool G4PSMinKinEAtGeneration::ProcessHits(G4Step* aStep,G4TouchableHistory*)
59 {
60  // ===============
61  // First Step,
62  // Confirm this is a newly produced secondary.
63  //
64  //- check for newly produced particle. e.g. Step number is 1.
65  if ( aStep->GetTrack()->GetCurrentStepNumber() != 1) return FALSE;
66  //- check for this is not a primary particle. e.g. ParentID != 0 .
67  if ( aStep->GetTrack()->GetParentID() == 0 ) return FALSE;
68 
69  //===============================================
70  //- This is a newly produced secondary particle.
71  //===============================================
72 
73  // ===============
74  // Second Step,
75  // Confirm this track has lower energy than previous one.
76  //
77 
78  // -Kinetic energy of this particle at the starting point.
79  G4double kinetic = aStep->GetPreStepPoint()->GetKineticEnergy();
80 
81  // -Stored value in the current HitsMap.
82  G4int index = GetIndex(aStep);
83  G4double* mapValue= ((*EvtMap)[index]);
84 
85  // -
86  // If mapValue exits (e.g not NULL ), compare it with
87  // current track's kinetic energy.
88  if ( mapValue && ( kinetic > *mapValue ) ) return FALSE;
89 
90  // -
91  // Current Track is a newly produced secondary and has lower
92  // kinetic energy than previous one in this geometry.
93  //
94  EvtMap->set(index, kinetic);
95  return TRUE;
96 }
