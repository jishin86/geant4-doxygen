137 void G4PartialPhantomParameterisation::
138 ComputeVoxelIndices(const G4int copyNo, size_t& nx,
139  size_t& ny, size_t& nz ) const
140 {
141  CheckCopyNo( copyNo );
142 
143  std::multimap<G4int,G4int>::const_iterator ite =
144  fFilledIDs.lower_bound(size_t(copyNo));
145  G4int dist = std::distance( fFilledIDs.begin(), ite );
146  nz = size_t(dist/fNoVoxelY);
147  ny = size_t( dist%fNoVoxelY );
148 
149  G4int ifmin = (*ite).second;
150  G4int nvoxXprev;
151  if( dist != 0 ) {
152  ite--;
153  nvoxXprev = (*ite).first;
154  } else {
155  nvoxXprev = -1;
156  }
157 
158  nx = ifmin+copyNo-nvoxXprev-1;
159 }

162 //------------------------------------------------------------------
163 G4int G4PartialPhantomParameterisation::
164 GetReplicaNo( const G4ThreeVector& localPoint, const G4ThreeVector& localDir )
165 {
166  // Check the voxel numbers corresponding to localPoint
167  // When a particle is on a surface, it may be between -kCarTolerance and
168  // +kCartolerance. By a simple distance as:
169  // G4int nx = G4int( (localPoint.x()+)/fVoxelHalfX/2.);
170  // those between -kCartolerance and 0 will be placed on voxel N-1 and those
171  // between 0 and kCarTolerance on voxel N.
172  // To avoid precision problems place the tracks that are on the surface on
173  // voxel N-1 if they have negative direction and on voxel N if they have
174  // positive direction.
175  // Add +kCarTolerance so that they are first placed on voxel N, and then
176  // if the direction is negative substract 1
177 
178  G4double fx = (localPoint.x()+fContainerWallX+kCarTolerance)/(fVoxelHalfX*2.);
179  G4int nx = G4int(fx);
180 
181  G4double fy = (localPoint.y()+fContainerWallY+kCarTolerance)/(fVoxelHalfY*2.);
182  G4int ny = G4int(fy);
183 
184  G4double fz = (localPoint.z()+fContainerWallZ+kCarTolerance)/(fVoxelHalfZ*2.);
185  G4int nz = G4int(fz);
186 
187  // If it is on the surface side, check the direction: if direction is
188  // negative place it on the previous voxel (if direction is positive it is
189  // already in the next voxel...).
190  // Correct also cases where n = -1 or n = fNoVoxel. It is always traced to be
191  // due to multiple scattering: track is entering a voxel but multiple
192  // scattering changes the angle towards outside
193  //
194  if( fx - nx < kCarTolerance/fVoxelHalfX )
195  {
196  if( localDir.x() < 0 )
197  {
198  if( nx != 0 )
199  {
200  nx -= 1;
201  }
202  }
203  else
204  {
205  if( nx == G4int(fNoVoxelX) )
206  {
207  nx -= 1;
208  }
209  }
210  }
211  if( fy - ny < kCarTolerance/fVoxelHalfY )
212  {
213  if( localDir.y() < 0 )
214  {
215  if( ny != 0 )
216  {
217  ny -= 1;
218  }
219  }
220  else
221  {
222  if( ny == G4int(fNoVoxelY) )
223  {
224  ny -= 1;
225  }
226  }
227  }
228  if( fz - nz < kCarTolerance/fVoxelHalfZ )
229  {
230  if( localDir.z() < 0 )
231  {
232  if( nz != 0 )
233  {
234  nz -= 1;
235  }
236  }
237  else
238  {
239  if( nz == G4int(fNoVoxelZ) )
240  {
241  nz -= 1;
242  }
243  }
244  }
245 
246  // Check if there are still errors
247  //
248  G4bool isOK = true;
249  if( nx < 0 )
250  {
251  nx = 0;
252  isOK = false;
253  }
254  else if( nx >= G4int(fNoVoxelX) )
255  {
256  nx = fNoVoxelX-1;
257  isOK = false;
258  }
259  if( ny < 0 )
260  {
261  ny = 0;
262  isOK = false;
263  }
264  else if( ny >= G4int(fNoVoxelY) )
265  {
266  ny = fNoVoxelY-1;
267  isOK = false;
268  }
269  if( nz < 0 )
270  {
271  nz = 0;
272  isOK = false;
273  }
274  else if( nz >= G4int(fNoVoxelZ) )
275  {
276  nz = fNoVoxelZ-1;
277  isOK = false;
278  }
279  if( !isOK )
280  {
281  std::ostringstream message;
282  message << "Corrected the copy number! It was negative or too big."
283  << G4endl
284  << " LocalPoint: " << localPoint << G4endl
285  << " LocalDir: " << localDir << G4endl
286  << " Voxel container size: " << fContainerWallX
287  << " " << fContainerWallY << " " << fContainerWallZ << G4endl
288  << " LocalPoint - wall: "
289  << localPoint.x()-fContainerWallX << " "
290  << localPoint.y()-fContainerWallY << " "
291  << localPoint.z()-fContainerWallZ;
292  G4Exception("G4PartialPhantomParameterisation::GetReplicaNo()",
293  "GeomNav1002", JustWarning, message);
294  }
295 
296  G4int nyz = nz*fNoVoxelY+ny;
297  std::multimap<G4int,G4int>::iterator ite = fFilledIDs.begin();
298 /*
299  for( ite = fFilledIDs.begin(); ite != fFilledIDs.end(); ite++ )
300  {
301  G4cout << " G4PartialPhantomParameterisation::GetReplicaNo filled "
302  << (*ite).first << " , " << (*ite).second << std::endl;
303  }
304 */
305  ite = fFilledIDs.begin();
306 
307  advance(ite,nyz);
308  std::multimap<G4int,G4int>::iterator iteant = ite; iteant--;
309  G4int copyNo = (*iteant).first + 1 + ( nx - (*ite).second );
310 /*
311  G4cout << " G4PartialPhantomParameterisation::GetReplicaNo getting copyNo "
312  << copyNo << " nyz " << nyz << " (*iteant).first "
313  << (*iteant).first << " (*ite).second " << (*ite).second << G4endl;
314 
315  G4cout << " G4PartialPhantomParameterisation::GetReplicaNo " << copyNo
316  << " nx " << nx << " ny " << ny << " nz " << nz
317  << " localPoint " << localPoint << " localDir " << localDir << G4endl;
318 */
319  return copyNo;
320 }
