97 void G4FermiBreakUpVI::BreakFragment(G4FragmentVector* theResult,
98  G4Fragment* theNucleus)
99 {
100  if(verbose > 0) {
101  G4cout << "### G4FermiBreakUpVI::BreakFragment start new fragment " << G4endl;
102  G4cout << *theNucleus << G4endl;
103  }
104  frag.clear();
105  lvect.clear();
106 
107  // initial fragment
108  Z = theNucleus->GetZ_asInt();
109  A = theNucleus->GetA_asInt();
110  excitation = theNucleus->GetExcitationEnergy();
111  mass = theNucleus->GetGroundStateMass() + excitation;
112  spin = -1;
113  G4double time = theNucleus->GetCreationTime();
114 
115  lv0 = theNucleus->GetMomentum();
116  rndmEngine = G4Random::getTheEngine();
117 
118  // sample first decay of an initial state
119  if(!SampleDecay()) {
120  theResult->push_back(theNucleus);
121  return;
122  }
123  delete theNucleus;
124 
125  static const G4int imax = 100;
126 
127  // loop over vector of Fermi fragments
128  // vector may grow at each iteraction
129  for(size_t i=0; i<frag.size(); ++i) {
130  Z = frag[i]->GetZ();
131  A = frag[i]->GetA();
132  spin = frag[i]->GetSpin();
133  mass = frag[i]->GetTotalEnergy();
134  excitation = 0.0;
135  if(thePool->IsPhysical(Z, A)) {
136  excitation = frag[i]->GetExcitationEnergy();
137  }
138  lv0 = lvect[i];
139  if(verbose > 0) {
140  G4cout << "# FermiFrag " << i << ". Z= " << Z << " A= " << A
141  << " mass= " << mass << " exc= " << excitation << G4endl;
142  }
143  // stable fragment
144  if(!SampleDecay()) {
145  if(verbose > 0) { G4cout << " New G4Fragment" << G4endl; }
146  G4Fragment* f = new G4Fragment(A, Z, lv0);
147  f->SetSpin(0.5*spin);
148  f->SetCreationTime(time);
149  theResult->push_back(f);
150  }
151  // limit the loop
152  if(i == imax) {
153  break;
154  }
155  }
156 }

158 G4bool G4FermiBreakUpVI::SampleDecay()
159 {
160  const G4FermiChannels* chan = thePool->ClosestChannels(Z, A, mass);
161  if(!chan) { return false; }
162  size_t nn = chan->GetNumberOfChannels();
163  if(verbose > 0) {
164  G4cout << "== SampleDecay " << nn << " channels Eex= "
165  << chan->GetExcitation() << G4endl;
166  }
167  if(0 == nn) { return false; }
168 
169  const G4FermiPair* fpair = nullptr;
170 
171  // one unstable fragment
172  if(1 == nn) {
173  fpair = chan->GetPair(0);
174 
175  // more pairs
176  } else {
177 
178  // in static probabilities may be used
179  if(std::abs(excitation - chan->GetExcitation()) < tolerance) {
180  fpair = chan->SamplePair(rndmEngine->flat());
181 
182  } else {
183 
184  // recompute probabilities
185  const std::vector<const G4FermiPair*>& pvect = chan->GetChannels();
186  if(nn > 12) { prob.resize(nn, 0.0); }
187  G4double ptot = 0.0;
188  if(verbose > 1) {
189  G4cout << "Start recompute probabilities" << G4endl;
190  }
191  for(size_t i=0; i<nn; ++i) {
192  ptot += theDecay->ComputeProbability(Z, A, -1, mass,
193  pvect[i]->GetFragment1(),
194  pvect[i]->GetFragment2());
195  prob[i] = ptot;
196  if(verbose > 1) {
197  G4cout << i << ". " << prob[i]
198  << " Z1= " << pvect[i]->GetFragment1()->GetZ()
199  << " A1= " << pvect[i]->GetFragment1()->GetA()
200  << " Z2= " << pvect[i]->GetFragment2()->GetZ()
201  << " A2= " << pvect[i]->GetFragment2()->GetA()
202  << G4endl;
203  }
204  }
205  ptot *= rndmEngine->flat();
206  for(size_t i=0; i<nn; ++i) {
207  if(ptot <= prob[i] || i+1 == nn) {
208  fpair = pvect[i];
209  break;
210  }
211  }
212  }
213  }
214  if(!fpair) { return false; }
215 
216  frag1 = fpair->GetFragment1();
217  frag2 = fpair->GetFragment2();
218 
219  G4double mass1 = frag1->GetTotalEnergy();
220  G4double mass2 = frag2->GetTotalEnergy();
221  if(verbose > 1) {
222  G4cout << " M= " << mass << " M1= " << mass1 << " M2= " << mass2
223  << " Exc1= " << frag1->GetExcitationEnergy()
224  << " Exc2= " << frag2->GetExcitationEnergy() << G4endl;
225  }
226  // sample fragment1
227  G4double e1 = 0.5*(mass*mass - mass2*mass2 + mass1*mass1)/mass;
228  //G4cout << "ekin1= " << e1 - mass1 << G4endl;
229  G4double p1(0.0);
230  if(e1 > mass1) {
231  p1 = std::sqrt((e1 - mass1)*(e1 + mass1));
232  } else {
233  e1 = mass1;
234  }
235  G4ThreeVector v = G4RandomDirection();
236  G4LorentzVector lv1 = G4LorentzVector(v*p1, e1);
237 
238  // compute kinematics
239  boostVector = lv0.boostVector();
240  lv1.boost(boostVector);
241 
242  G4double e2 = mass - e1;
243  if(e2 < mass2) {
244  e2 = mass2;
245  p1 = 0.0;
246  }
247  lv0.set(-v*p1, e2);
248  lv0.boost(boostVector);
249 
250  frag.push_back(frag1);
251  frag.push_back(frag2);
252  lvect.push_back(lv1);
253  lvect.push_back(lv0);
254 
255  return true;
256 }
