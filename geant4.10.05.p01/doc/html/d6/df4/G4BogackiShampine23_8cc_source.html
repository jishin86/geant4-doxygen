77 void G4BogackiShampine23::makeStep(const G4double yInput[],
78  const G4double dydx[],
79  const G4double hstep,
80  G4double yOutput[],
81  G4double* dydxOutput,
82  G4double* yError) const
83 {
84 
85  G4double yTemp[G4FieldTrack::ncompSVEC];
86  for(G4int i = GetNumberOfVariables(); i < GetNumberOfStateVariables(); ++i)
87  yOutput[i] = yTemp[i] = yInput[i];
88 
89 
90  G4double ak2[G4FieldTrack::ncompSVEC],
91  ak3[G4FieldTrack::ncompSVEC];
92 
93  const G4double b21 = 0.5 ,
94  b31 = 0., b32 = 3.0 / 4.0,
95  b41 = 2.0 / 9.0, b42 = 1.0 / 3.0, b43 = 4.0 / 9.0;
96 
97  const G4double dc1 = b41 - 7.0 / 24.0, dc2 = b42 - 1.0 / 4.0,
98  dc3 = b43 - 1.0 / 3.0, dc4 = - 1.0 / 8.0;
99 
100  // RightHandSide(yInput, dydx);
101  for(G4int i = 0; i < GetNumberOfVariables(); ++i)
102  yTemp[i] = yInput[i] + b21 * hstep * dydx[i];
103 
104  RightHandSide(yTemp, ak2);
105  for(G4int i = 0; i < GetNumberOfVariables(); ++i)
106  yTemp[i] = yInput[i] + hstep * (b31 * dydx[i] + b32 * ak2[i]);
107 
108  RightHandSide(yTemp, ak3);
109  for(G4int i = 0; i < GetNumberOfVariables(); ++i)
110  yOutput[i] = yInput[i] + hstep * (b41 * dydx[i] + b42 * ak2[i] + b43 * ak3[i]);
111 
112  if (dydxOutput && yError) {
113  RightHandSide(yOutput, dydxOutput);
114  for(G4int i = 0; i < GetNumberOfVariables(); ++i)
115  yError[i] = hstep * (dc1 * dydx[i] + dc2 * ak2[i] +
116  dc3 * ak3[i] + dc4 * dydxOutput[i]);
117 
118  }
119 }
