88 void G4LowECapture::BuildPhysicsTable(const G4ParticleDefinition& part)
89 {
90  G4RegionStore* store = G4RegionStore::GetInstance();
91  for(G4int i=0; i<nRegions; ++i) {
92  const G4Region* r = store->GetRegion(regionName[i]);
93  if(r && verboseLevel > 0) {
94  G4cout << "### G4LowECapture: new G4Region <"
95  << regionName[i] << "> with tracking cut "
96  << kinEnergyThreshold/keV << " keV" << G4endl;
97  }
98  if(r) { region.push_back(r); }
99  }
100  nRegions = region.size();
101 
102  // ions reusing G4GenericIon parameters
103  if(part.GetParticleType() == "nucleus") {
104  G4String pname = part.GetParticleName();
105  if(pname != "deuteron" && pname != "triton" &&
106  pname != "alpha" && pname != "He3" &&
107  pname != "alpha+" && pname != "helium" &&
108  pname != "hydrogen") { isIon = true; }
109  }
110 }

119 G4double G4LowECapture::PostStepGetPhysicalInteractionLength(
120  const G4Track& aTrack, G4double, G4ForceCondition* condition)
121 {
122  *condition = NotForced;
123  G4double limit = DBL_MAX;
124  G4double eLimit = kinEnergyThreshold;
125  if(isIon) {
126  eLimit *= aTrack.GetDefinition()->GetPDGMass()/CLHEP::proton_mass_c2;
127  }
128  if(aTrack.GetKineticEnergy() < eLimit) {
129  for(G4int i=0; i<nRegions; ++i) {
130  if(aTrack.GetVolume()->GetLogicalVolume()->GetRegion() == region[i]) {
131  limit = 0.0;
132  break;
133  }
134  }
135  }
136  return limit;
137 }
