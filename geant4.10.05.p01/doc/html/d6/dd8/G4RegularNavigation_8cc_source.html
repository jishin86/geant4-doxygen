59 G4double G4RegularNavigation::
60  ComputeStep(const G4ThreeVector& localPoint,
61  const G4ThreeVector& localDirection,
62  const G4double currentProposedStepLength,
63  G4double& newSafety,
64  G4NavigationHistory& history,
65  G4bool& validExitNormal,
66  G4ThreeVector& exitNormal,
67  G4bool& exiting,
68  G4bool& entering,
69  G4VPhysicalVolume *(*pBlockedPhysical),
70  G4int& blockedReplicaNo)
71 {
72  // This method is never called because to be called the daughter has to be
73  // a regular structure. This would only happen if the track is in the mother
74  // of voxels volume. But the voxels fill completely their mother, so when a
75  // track enters the mother it automatically enters a voxel. Only precision
76  // problems would make this method to be called
77 
78  G4ThreeVector globalPoint =
79  history.GetTopTransform().InverseTransformPoint(localPoint);
80  G4ThreeVector globalDirection =
81  history.GetTopTransform().InverseTransformAxis(localDirection);
82 
83  G4ThreeVector localPoint2 = localPoint; // take away constantness
84 
85  LevelLocate( history, *pBlockedPhysical, blockedReplicaNo,
86  globalPoint, &globalDirection, true, localPoint2 );
87 
88 
89  // Get in which voxel it is
90  //
91  G4VPhysicalVolume *motherPhysical, *daughterPhysical;
92  G4LogicalVolume *motherLogical;
93  motherPhysical = history.GetTopVolume();
94  motherLogical = motherPhysical->GetLogicalVolume();
95  daughterPhysical = motherLogical->GetDaughter(0);
96 
97  G4PhantomParameterisation * daughterParam =
98  (G4PhantomParameterisation*)(daughterPhysical->GetParameterisation());
99  G4int copyNo = daughterParam ->GetReplicaNo(localPoint,localDirection);
100 
101  G4ThreeVector voxelTranslation = daughterParam->GetTranslation( copyNo );
102  G4ThreeVector daughterPoint = localPoint - voxelTranslation;
103 
104 
105  // Compute step in voxel
106  //
107  return fnormalNav->ComputeStep(daughterPoint,
108  localDirection,
109  currentProposedStepLength,
110  newSafety,
111  history,
112  validExitNormal,
113  exitNormal,
114  exiting,
115  entering,
116  pBlockedPhysical,
117  blockedReplicaNo);
118 }

121 //------------------------------------------------------------------
122 G4double G4RegularNavigation::ComputeStepSkippingEqualMaterials(
123  G4ThreeVector& localPoint,
124  const G4ThreeVector& localDirection,
125  const G4double currentProposedStepLength,
126  G4double& newSafety,
127  G4NavigationHistory& history,
128  G4bool& validExitNormal,
129  G4ThreeVector& exitNormal,
130  G4bool& exiting,
131  G4bool& entering,
132  G4VPhysicalVolume *(*pBlockedPhysical),
133  G4int& blockedReplicaNo,
134  G4VPhysicalVolume* pCurrentPhysical)
135 {
136  G4RegularNavigationHelper::Instance()->ClearStepLengths();
137 
138  G4PhantomParameterisation *param =
139  (G4PhantomParameterisation*)(pCurrentPhysical->GetParameterisation());
140 
141  if( !param->SkipEqualMaterials() )
142  {
143  return fnormalNav->ComputeStep(localPoint,
144  localDirection,
145  currentProposedStepLength,
146  newSafety,
147  history,
148  validExitNormal,
149  exitNormal,
150  exiting,
151  entering,
152  pBlockedPhysical,
153  blockedReplicaNo);
154  }
155 
156 
157  G4double ourStep = 0.;
158 
159  // To get replica No: transform local point to the reference system of the
160  // param container volume
161  //
162  G4int ide = history.GetDepth();
163  G4ThreeVector containerPoint = history.GetTransform(ide).InverseTransformPoint(localPoint);
164 
165  // Point in global frame
166  //
167  containerPoint = history.GetTransform(ide).InverseTransformPoint(localPoint);
168 
169  // Point in voxel parent volume frame
170  //
171  containerPoint = history.GetTransform(ide-1).TransformPoint(containerPoint);
172 
173  // Store previous voxel translation to move localPoint by the difference
174  // with the new one
175  //
176  G4ThreeVector prevVoxelTranslation = containerPoint - localPoint;
177 
178  // Do not use the expression below: There are cases where the
179  // fLastLocatedPointLocal does not give the correct answer
180  // (particle reaching a wall and bounced back, particle travelling through
181  // the wall that is deviated in an step, ...; these are pathological cases
182  // that give wrong answers in G4PhantomParameterisation::GetReplicaNo()
183  //
184  // G4ThreeVector prevVoxelTranslation = param->GetTranslation( copyNo );
185 
186  G4int copyNo = param->GetReplicaNo(containerPoint,localDirection);
187 
188  G4Material* currentMate = param->ComputeMaterial( copyNo, 0, 0 );
189  G4VSolid* voxelBox = pCurrentPhysical->GetLogicalVolume()->GetSolid();
190 
191  G4VSolid* containerSolid = param->GetContainerSolid();
192  G4Material* nextMate;
193  G4bool bFirstStep = true;
194  G4double newStep;
195  G4double totalNewStep = 0.;
196 
197  // Loop while same material is found
198  //
199  for( ;; )
200  {
201  newStep = voxelBox->DistanceToOut( localPoint, localDirection );
202 
203  if( (bFirstStep) && (newStep < currentProposedStepLength) )
204  {
205  exiting = true;
206  }
207  bFirstStep = false;
208 
209  newStep += kCarTolerance; // Avoid precision problems
210  ourStep += newStep;
211  totalNewStep += newStep;
212 
213  // Physical process is limiting the step, don't continue
214  //
215  if(std::fabs(totalNewStep-currentProposedStepLength) < kCarTolerance)
216  {
217  return currentProposedStepLength;
218  }
219  if(totalNewStep > currentProposedStepLength)
220  {
221  G4RegularNavigationHelper::Instance()->
222  AddStepLength(copyNo, newStep-totalNewStep+currentProposedStepLength);
223  return currentProposedStepLength;
224  }
225  else
226  {
227  G4RegularNavigationHelper::Instance()->AddStepLength( copyNo, newStep );
228  }
229 
230 
231  // Move container point until wall of voxel
232  //
233  containerPoint += newStep*localDirection;
234  if( containerSolid->Inside( containerPoint ) != kInside )
235  {
236  break;
237  }
238 
239  // Get copyNo and translation of new voxel
240  //
241  copyNo = param->GetReplicaNo(containerPoint,localDirection);
242  G4ThreeVector voxelTranslation = param->GetTranslation( copyNo );
243 
244  // G4cout << " copyNo " << copyNo << " = " << pCurrentPhysical->GetCopyNo() << G4endl;
245  // Move local point until wall of voxel and then put it in the new voxel
246  // local coordinates
247  //
248  localPoint += newStep*localDirection;
249  localPoint += prevVoxelTranslation - voxelTranslation;
250 
251  prevVoxelTranslation = voxelTranslation;
252 
253  // Check if material of next voxel is the same as that of the current voxel
254  nextMate = param->ComputeMaterial( copyNo, 0, 0 );
255 
256  if( currentMate != nextMate ) { break; }
257  }
258 
259  return ourStep;
260 }

284 G4bool
285 G4RegularNavigation::LevelLocate( G4NavigationHistory& history,
286  const G4VPhysicalVolume* ,
287  const G4int ,
288  const G4ThreeVector& globalPoint,
289  const G4ThreeVector* globalDirection,
290  const G4bool, // pLocatedOnEdge,
291  G4ThreeVector& localPoint )
292 {
293  G4VPhysicalVolume *motherPhysical, *pPhysical;
294  G4PhantomParameterisation *pParam;
295  G4LogicalVolume *motherLogical;
296  G4ThreeVector localDir;
297  G4int replicaNo;
298 
299  motherPhysical = history.GetTopVolume();
300  motherLogical = motherPhysical->GetLogicalVolume();
301 
302  pPhysical = motherLogical->GetDaughter(0);
303  pParam = (G4PhantomParameterisation*)(pPhysical->GetParameterisation());
304 
305  // Save parent history in touchable history
306  // ... for use as parent t-h in ComputeMaterial method of param
307  //
308  G4TouchableHistory parentTouchable( history );
309 
310  // Get local direction
311  //
312  if( globalDirection )
313  {
314  localDir = history.GetTopTransform().TransformAxis(*globalDirection);
315  }
316  else
317  {
318  localDir = G4ThreeVector(0.,0.,0.);
319  }
320 
321  // Enter this daughter
322  //
323  replicaNo = pParam->GetReplicaNo( localPoint, localDir );
324 
325  if( replicaNo < 0 || replicaNo >= G4int(pParam->GetNoVoxel()) )
326  {
327  return false;
328  }
329 
330  // Set the correct copy number in physical
331  //
332  pPhysical->SetCopyNo(replicaNo);
333  pParam->ComputeTransformation(replicaNo,pPhysical);
334 
335  history.NewLevel(pPhysical, kParameterised, replicaNo );
336  localPoint = history.GetTopTransform().TransformPoint(globalPoint);
337 
338  // Set the correct solid and material in Logical Volume
339  //
340  G4LogicalVolume *pLogical = pPhysical->GetLogicalVolume();
341 
342  pLogical->UpdateMaterial(pParam->ComputeMaterial(replicaNo,
343  pPhysical, &parentTouchable) );
344  return true;
345 }
