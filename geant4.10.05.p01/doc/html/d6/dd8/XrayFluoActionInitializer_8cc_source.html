52 void XrayFluoActionInitializer::Build() const
53 {
54  //Actual version will depend on the geometry
55  XrayFluoEventAction* eventAction = 0;
56  XrayFluoSteppingAction* stepAction = new XrayFluoSteppingAction();
57 
58  //Selecting the PrimaryGenerator depending upon Geometry setup selected
59  if (fGeometryFlag == 1 || fGeometryFlag == 4)
60  {
61  const XrayFluoDetectorConstruction* testBeamDetector =
62  static_cast<const XrayFluoDetectorConstruction*>
63  (G4RunManager::GetRunManager()->GetUserDetectorConstruction());
64 
65  eventAction = new XrayFluoEventAction(testBeamDetector);
66  SetUserAction(new XrayFluoPrimaryGeneratorAction(testBeamDetector));
67  }
68  else if (fGeometryFlag == 2)
69  {
70  const XrayFluoPlaneDetectorConstruction* planeDetector =
71  static_cast<const XrayFluoPlaneDetectorConstruction*>
72  (G4RunManager::GetRunManager()->GetUserDetectorConstruction());
73 
74  eventAction = new XrayFluoEventAction(planeDetector);
75  SetUserAction(new XrayFluoPlanePrimaryGeneratorAction(planeDetector));
76  }
77  else if (fGeometryFlag == 3)
78  {
79  const XrayFluoMercuryDetectorConstruction* mercuryDetector =
80  static_cast<const XrayFluoMercuryDetectorConstruction*>
81  (G4RunManager::GetRunManager()->GetUserDetectorConstruction());
82  eventAction = new XrayFluoEventAction(mercuryDetector);
83  stepAction->SetMercuryFlag(true);
84  SetUserAction(new XrayFluoMercuryPrimaryGeneratorAction(mercuryDetector));
85  }
86  SetUserAction(eventAction);
87  SetUserAction(new XrayFluoRunAction());
88  SetUserAction(stepAction);
89 
90 }
