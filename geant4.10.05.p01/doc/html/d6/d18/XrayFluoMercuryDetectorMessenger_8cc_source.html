49 XrayFluoMercuryDetectorMessenger::XrayFluoMercuryDetectorMessenger(XrayFluoMercuryDetectorConstruction * Det)

53  detDir->SetGuidance("detector control.");

55  UpdateCmd = new G4UIcmdWithoutParameter("/apparate/update",this);
56  UpdateCmd->SetGuidance("Update apparate geometry.");

58  UpdateCmd->SetGuidance("if you changed geometrical value(s): /apparate/GrainDiameter and /apparate/sampleGranularity");
59 
60  UpdateCmd->AvailableForStates(G4State_Idle);
61 
62  sampleCmd = new G4UIcmdWithAString("/apparate/mercuryMaterial",this);
63  sampleCmd->SetGuidance("select a diferent material for the mercury");
64  sampleCmd->SetParameterName("material",true);
65  sampleCmd->SetDefaultValue("mars1");
66  sampleCmd->SetCandidates("Dolorite Anorthosite Mars1 IceBasalt");
67  sampleCmd->AvailableForStates(G4State_Idle);
68 
69  detectorCmd = new G4UIcmdWithAString("/apparate/detector",this);
70  detectorCmd->SetGuidance("select a diferent detectorType");
71  detectorCmd->SetParameterName("detector",true);
72  detectorCmd->SetDefaultValue("sili");
73  detectorCmd->SetCandidates("sili hpge");
74  detectorCmd->AvailableForStates(G4State_Idle);
75 
76  latitudeAngleCmd = new G4UIcmdWithADoubleAndUnit( "/apparate/latitude",this );
77  latitudeAngleCmd->SetGuidance( "Set latitude angle of the spacecraft" );
78  latitudeAngleCmd->SetGuidance( "After this, /apparate/update must be executed before BeamOn" );

105 void XrayFluoMercuryDetectorMessenger::SetNewValue(G4UIcommand* command,G4String newValue)
106 {
107  if( command == UpdateCmd )
108  {
109  //This triggers a full re-build of the geometry. The method in the
110  //geometry will take care of that.
111  Detector->UpdateGeometry();
112  return;
113  }
114  else if ( command == sampleCmd )
115  { Detector->SetMercuryMaterial(newValue);}
116 
117  else if ( command == detectorCmd )
118  { Detector->SetDetectorType(newValue);}
119 
120  else if ( command == latitudeAngleCmd )
121  {
122  G4double newAngle = latitudeAngleCmd->GetNewDoubleValue(newValue);
123  Detector->SetLatitude(newAngle);
124  }
125 
126  else if ( command == orbitHeightCmd )
127  {
128  G4double newAngle = orbitHeightCmd->GetNewDoubleValue(newValue);
129  Detector->SetOribitHeight(newAngle);
130  }
131  //Notify the run manager that the geometry has been modified
132  G4RunManager::GetRunManager()->GeometryHasBeenModified();
133 
134 }
