252 void G4EmExtraPhysics::ConstructProcess()
253 {
254  G4ParticleDefinition* gamma = G4Gamma::Gamma();
255  G4ParticleDefinition* electron = G4Electron::Electron();
256  G4ParticleDefinition* positron = G4Positron::Positron();
257  G4ParticleDefinition* muonplus = G4MuonPlus::MuonPlus();
258  G4ParticleDefinition* muonminus = G4MuonMinus::MuonMinus();
259 
260  G4ParticleDefinition* anuelectron = G4AntiNeutrinoE::AntiNeutrinoE();
261  G4ParticleDefinition* nuelectron = G4NeutrinoE::NeutrinoE();
262  G4ParticleDefinition* anumuon = G4AntiNeutrinoMu::AntiNeutrinoMu();
263  G4ParticleDefinition* numuon = G4NeutrinoMu::NeutrinoMu();
264  G4ParticleDefinition* anutau = G4AntiNeutrinoTau::AntiNeutrinoTau();
265  G4ParticleDefinition* nutau = G4NeutrinoTau::NeutrinoTau();
266 
267  G4PhysicsListHelper* ph = G4PhysicsListHelper::GetPhysicsListHelper();
268  if(gnActivated) {
269  if ( gLENDActivated != true ) {
270  theGNPhysics = new G4BertiniElectroNuclearBuilder(eActivated);
271  } else {
272  theGNPhysics = new G4LENDBertiniGammaElectroNuclearBuilder(eActivated);
273  }
274  theGNPhysics->Build();
275  }
276  if(munActivated) {
277  G4MuonNuclearProcess* muNucProcess = new G4MuonNuclearProcess();
278  G4MuonVDNuclearModel* muNucModel = new G4MuonVDNuclearModel();
279  muNucProcess->RegisterMe(muNucModel);
280  ph->RegisterProcess( muNucProcess, muonplus);
281  ph->RegisterProcess( muNucProcess, muonminus);
282  }
283  if(gmumuActivated) {
284  theGammaToMuMu = new G4GammaConversionToMuons();
285  theGammaToMuMu->SetCrossSecFactor(gmumuFactor);
286  G4GammaGeneralProcess* sp =
287  (G4GammaGeneralProcess*)G4LossTableManager::Instance()->GetGammaGeneralProcess();
288  if(sp) {
289  sp->AddMMProcess(theGammaToMuMu);
290  } else {
291  ph->RegisterProcess(theGammaToMuMu, gamma);
292  }
293  }
294  if(pmumuActivated) {
295  thePosiToMuMu = new G4AnnihiToMuPair();
296  thePosiToMuMu->SetCrossSecFactor(pmumuFactor);
297  ph->RegisterProcess(thePosiToMuMu, positron);
298  }
299  if(phadActivated) {
300  thePosiToHadrons = new G4eeToHadrons();
301  thePosiToHadrons->SetCrossSecFactor(phadFactor);
302  ph->RegisterProcess(thePosiToHadrons, positron);
303  }
304  if(synActivated) {
305  theSynchRad = new G4SynchrotronRadiation();
306  ph->RegisterProcess( theSynchRad, electron);
307  ph->RegisterProcess( theSynchRad, positron);
308  if(synActivatedForAll) {
309  auto myParticleIterator=GetParticleIterator();
310  myParticleIterator->reset();
311  G4ParticleDefinition* particle = nullptr;
312 
313  while( (*myParticleIterator)() ) {
314  particle = myParticleIterator->value();
315  if( particle->GetPDGStable() && particle->GetPDGCharge() != 0.0) {
316  if(verbose > 1) {
317  G4cout << "### G4SynchrotronRadiation for "
318  << particle->GetParticleName() << G4endl;
319  }
320  ph->RegisterProcess( theSynchRad, particle);
321  }
322  }
323  }
324  }
325  if( fNuActivated )
326  {
327  theNuEleProcess = new G4NeutrinoElectronProcess(fNuDetectorName);
328  theNuEleTotXsc = new G4NeutrinoElectronTotXsc();
329 
330  if(fNuETotXscActivated)
331  {
332  G4double bftot = std::max(fNuEleCcBias,fNuEleNcBias);
333  theNuEleProcess->SetBiasingFactor(bftot);
334  }
335  else
336  {
337  theNuEleProcess->SetBiasingFactors(fNuEleCcBias,fNuEleNcBias);
338  theNuEleTotXsc->SetBiasingFactors(fNuEleCcBias,fNuEleNcBias);
339  }
340  theNuEleProcess->AddDataSet(theNuEleTotXsc);
341 
342  G4NeutrinoElectronCcModel* ccModel = new G4NeutrinoElectronCcModel();
343  G4NeutrinoElectronNcModel* ncModel = new G4NeutrinoElectronNcModel();
344  theNuEleProcess->RegisterMe(ccModel);
345  theNuEleProcess->RegisterMe(ncModel);
346 
347  ph->RegisterProcess(theNuEleProcess, anuelectron);
348  ph->RegisterProcess(theNuEleProcess, nuelectron);
349  ph->RegisterProcess(theNuEleProcess, anumuon);
350  ph->RegisterProcess(theNuEleProcess, numuon);
351  ph->RegisterProcess(theNuEleProcess, anutau);
352  ph->RegisterProcess(theNuEleProcess, nutau);
353  }
354 }
