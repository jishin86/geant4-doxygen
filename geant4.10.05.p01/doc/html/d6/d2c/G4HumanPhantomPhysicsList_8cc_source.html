84 void G4HumanPhantomPhysicsList::SetCuts()
85 {
86  // The production threshold is fixed to 0.1 mm for all the particles
87  // Secondary particles with a range bigger than 0.1 mm
88  // are generated; otherwise their energy is considered deposited locally
89 
90  defaultCutValue = 1. * mm;
91 
92  const G4double cutForGamma = defaultCutValue;
93  const G4double cutForElectron = defaultCutValue;
94  const G4double cutForPositron = defaultCutValue;
95 
96  SetCutValue(cutForGamma, "gamma");
97  SetCutValue(cutForElectron, "e-");
98  SetCutValue(cutForPositron, "e+");
99 
100  // Set the secondary production cut lower than 990. eV
101  // Very important for high precision of lowenergy processes at low energies
102 
103  G4double lowLimit = 250. * eV;
104  G4double highLimit = 100. * GeV;
105  G4ProductionCutsTable::GetProductionCutsTable()->SetEnergyRange(lowLimit, highLimit);
106 
107  if (verboseLevel>0) DumpCutValuesTable();
108 }
