108 //
109 // Return approximate memory use

111 G4long G4SmartVoxelStat::GetMemoryUse() const
112 {
113  static const G4long headSize = sizeof(G4SmartVoxelHeader)
114  + sizeof(G4SmartVoxelProxy);
115 
116  static const G4long nodeSize = sizeof(G4SmartVoxelNode)
117  + sizeof(G4SmartVoxelProxy);
118 
119  static const G4long pointerSize = sizeof(G4SmartVoxelProxy*);
120 
121  return nodes*nodeSize + heads*headSize + pointers*pointerSize;
122 }

126 //
127 // CountHeadsAndNodes
128 //
129 // Recursively count the number of voxel headers and nodes,
130 // updating class member variables heads and nodes on the way.
131 //
132 void G4SmartVoxelStat::CountHeadsAndNodes( const G4SmartVoxelHeader *head )
133 {
134  G4int numSlices = head->GetNoSlices();
135 
136  pointers += numSlices;
137 
138  const G4SmartVoxelProxy *lastProxy = 0;
139 
140  for(G4int i=0;i<numSlices;++i) {
141  const G4SmartVoxelProxy *proxy = head->GetSlice(i);
142  if (proxy == lastProxy) continue;
143 
144  lastProxy = proxy;
145 
146  if (proxy->IsNode()) {
147  nodes++;
148  }
149  else {
150  heads++;
151  CountHeadsAndNodes(proxy->GetHeader());
152  }
153  }
154 }
