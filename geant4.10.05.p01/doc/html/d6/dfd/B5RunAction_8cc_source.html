40 B5RunAction::B5RunAction(B5EventAction* eventAction)
41  : G4UserRunAction(),
42  fEventAction(eventAction)
43 {
44  // Create analysis manager
45  // The choice of analysis technology is done via selectin of a namespace
46  // in B5Analysis.hh

48  G4cout << "Using " << analysisManager->GetType() << G4endl;
49 
50  // Default settings
51  analysisManager->SetNtupleMerging(true);
52  // Note: merging ntuples is available only with Root output
53  analysisManager->SetVerboseLevel(1);
54  analysisManager->SetFileName("B5");
55 
56  // Book histograms, ntuple
57  //
58 
59  // Creating 1D histograms
60  analysisManager
61  ->CreateH1("Chamber1","Drift Chamber 1 # Hits", 50, 0., 50); // h1 Id = 0
62  analysisManager
63  ->CreateH1("Chamber2","Drift Chamber 2 # Hits", 50, 0., 50); // h1 Id = 1
64 
65  // Creating 2D histograms
66  analysisManager
67  ->CreateH2("Chamber1 XY","Drift Chamber 1 X vs Y", // h2 Id = 0
68  50, -1000., 1000, 50, -300., 300.);
69  analysisManager
70  ->CreateH2("Chamber2 XY","Drift Chamber 2 X vs Y", // h2 Id = 1
71  50, -1500., 1500, 50, -300., 300.);
72 
73  // Creating ntuple
74  //
75  if ( fEventAction ) {
76  analysisManager->CreateNtuple("B5", "Hits");
77  analysisManager->CreateNtupleIColumn("Dc1Hits"); // column Id = 0
78  analysisManager->CreateNtupleIColumn("Dc2Hits"); // column Id = 1
79  analysisManager->CreateNtupleDColumn("ECEnergy"); // column Id = 2
80  analysisManager->CreateNtupleDColumn("HCEnergy"); // column Id = 3
81  analysisManager->CreateNtupleDColumn("Time1"); // column Id = 4
82  analysisManager->CreateNtupleDColumn("Time2"); // column Id = 5
83  analysisManager // column Id = 6
84  ->CreateNtupleDColumn("ECEnergyVector", fEventAction->GetEmCalEdep());
85  analysisManager // column Id = 7
