228 inline G4double G4IonisParamMat::DensityCorrection(G4double x)
229 {
230  // x = log10(beta*gamma)
231  G4double y = 0.0;
232  if(x < fX0density) {
233  if(fD0density > 0.0) { y = fD0density*G4Exp(twoln10*(x - fX0density)); }
234  } else if(x >= fX1density) { y = twoln10*x - fCdensity; }
235  else {y = twoln10*x - fCdensity + fAdensity*G4Exp(G4Log(fX1density - x)*fMdensity);}
236  return y;
237 }
