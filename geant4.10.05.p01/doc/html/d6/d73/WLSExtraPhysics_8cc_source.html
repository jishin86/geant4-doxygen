56 void WLSExtraPhysics::ConstructProcess()
57 {
58  G4cout << "WLSExtraPhysics:: Add Extra Physics Processes"
59  << G4endl;
60 
61  auto particleIterator=GetParticleIterator();
62  particleIterator->reset();
63 
64  while ((*particleIterator)()) {
65  G4ParticleDefinition* particle = particleIterator->value();
66  G4ProcessManager* pmanager = particle->GetProcessManager();
67  G4String particleName = particle->GetParticleName();
68  G4double charge = particle->GetPDGCharge();
69 
70  if (!pmanager) {
71  std::ostringstream o;
72  o << "Particle " << particleName << "without a Process Manager";
73  G4Exception("WLSExtraPhysics::ConstructProcess()","",
74  FatalException,o.str().c_str());
75  }
76 
77  if (particleName == "opticalphoton") break;
78 
79  if (charge != 0.0) {
80  // All charged particles should have a step limiter
81  // to make sure that the steps do not get too long.
82  pmanager->AddDiscreteProcess(new G4StepLimiter());
83  pmanager->AddDiscreteProcess(new G4UserSpecialCuts());
84  } else if (particleName == "neutron") {
85  // time cuts for ONLY neutrons:
86  pmanager->AddDiscreteProcess(new G4UserSpecialCuts());
87  } else {
88  // Energy cuts for all other neutral particles
89  pmanager->AddDiscreteProcess(new G4UserSpecialCuts());
90  }
91  }
92 }
