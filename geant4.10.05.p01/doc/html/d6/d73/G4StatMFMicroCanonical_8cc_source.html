57 void G4StatMFMicroCanonical::Initialize(const G4Fragment & theFragment)
58 {
59 
60  std::vector<G4StatMFMicroManager*>::iterator it;
61 
62  // Excitation Energy
63  G4double U = theFragment.GetExcitationEnergy();
64 
65  G4int A = theFragment.GetA_asInt();
66  G4int Z = theFragment.GetZ_asInt();
67  G4double x = 1.0 - 2.0*Z/G4double(A);
68  G4Pow* g4calc = G4Pow::GetInstance();
69 
70  // Configuration temperature
71  G4double TConfiguration = std::sqrt(8.0*U/G4double(A));
72 
73  // Free internal energy at Temperature T = 0
74  __FreeInternalE0 = A*(
75  // Volume term (for T = 0)
76  -G4StatMFParameters::GetE0() +
77  // Symmetry term
78  G4StatMFParameters::GetGamma0()*x*x
79  ) +
80  // Surface term (for T = 0)
81  G4StatMFParameters::GetBeta0()*g4calc->Z23(A) +
82  // Coulomb term
83  elm_coupling*0.6*Z*Z/(G4StatMFParameters::Getr0()*g4calc->Z13(A));
84 
85  // Statistical weight
86  G4double W = 0.0;
87 
88  // Mean breakup multiplicity
89  __MeanMultiplicity = 0.0;
90 
91  // Mean channel temperature
92  __MeanTemperature = 0.0;
93 
94  // Mean channel entropy
95  __MeanEntropy = 0.0;
96 
97  // Calculate entropy of compound nucleus
98  G4double SCompoundNucleus = CalcEntropyOfCompoundNucleus(theFragment,TConfiguration);
99 
100  // Statistical weight of compound nucleus
101  _WCompoundNucleus = 1.0;
102 
103  W += _WCompoundNucleus;
104 
105  // Maximal fragment multiplicity allowed in direct simulation
106  G4int MaxMult = G4StatMFMicroCanonical::MaxAllowedMultiplicity;
107  if (A > 110) MaxMult -= 1;
108 
109  for (G4int im = 2; im <= MaxMult; im++) {
110  G4StatMFMicroManager * aMicroManager =
111  new G4StatMFMicroManager(theFragment,im,__FreeInternalE0,SCompoundNucleus);
112  _ThePartitionManagerVector.push_back(aMicroManager);
113  }
114 
115  // W is the total probability
116  W = std::accumulate(_ThePartitionManagerVector.begin(),
117  _ThePartitionManagerVector.end(),
118  W, SumProbabilities());
119 
120  // Normalization of statistical weights
121  for (it = _ThePartitionManagerVector.begin(); it != _ThePartitionManagerVector.end(); ++it)
122  {
123  (*it)->Normalize(W);
124  }
125 
126  _WCompoundNucleus /= W;
127 
128  __MeanMultiplicity += 1.0 * _WCompoundNucleus;
129  __MeanTemperature += TConfiguration * _WCompoundNucleus;
130  __MeanEntropy += SCompoundNucleus * _WCompoundNucleus;
131 
132  for (it = _ThePartitionManagerVector.begin(); it != _ThePartitionManagerVector.end(); ++it)
133  {
134  __MeanMultiplicity += (*it)->GetMeanMultiplicity();
135  __MeanTemperature += (*it)->GetMeanTemperature();
136  __MeanEntropy += (*it)->GetMeanEntropy();
137  }
138 
139  return;
140 }

142 G4double G4StatMFMicroCanonical::CalcFreeInternalEnergy(const G4Fragment & theFragment,
143  G4double T)
144 {
145  G4int A = theFragment.GetA_asInt();
146  G4int Z = theFragment.GetZ_asInt();
147  G4double A13 = G4Pow::GetInstance()->Z13(A);
148 
149  G4double InvLevelDensityPar = G4StatMFParameters::GetEpsilon0()
150  *(1.0 + 3.0/G4double(A-1));
151 
152  G4double VolumeTerm = (-G4StatMFParameters::GetE0()+T*T/InvLevelDensityPar)*A;
153 
154  G4double SymmetryTerm = G4StatMFParameters::GetGamma0()
155  *(A - 2*Z)*(A - 2*Z)/G4double(A);
156 
157  G4double SurfaceTerm = (G4StatMFParameters::Beta(T)
158  - T*G4StatMFParameters::DBetaDT(T))*A13*A13;
159 
160  G4double CoulombTerm = elm_coupling*0.6*Z*Z/(G4StatMFParameters::Getr0()*A13);
161 
162  return VolumeTerm + SymmetryTerm + SurfaceTerm + CoulombTerm;
163 }

165 G4double
166 G4StatMFMicroCanonical::CalcEntropyOfCompoundNucleus(const G4Fragment & theFragment,
167  G4double & TConf)
168  // Calculates Temperature and Entropy of compound nucleus
169 {
170  G4int A = theFragment.GetA_asInt();
171  G4double U = theFragment.GetExcitationEnergy();
172  G4double A13 = G4Pow::GetInstance()->Z13(A);
173 
174  G4double Ta = std::max(std::sqrt(U/(0.125*A)),0.0012*MeV);
175  G4double Tb = Ta;
176 
177  G4double ECompoundNucleus = CalcFreeInternalEnergy(theFragment,Ta);
178  G4double Da = (U+__FreeInternalE0-ECompoundNucleus)/U;
179  G4double Db = 0.0;
180 
181  G4double InvLevelDensity = CalcInvLevelDensity(A);
182 
183  // bracketing the solution
184  if (Da == 0.0) {
185  TConf = Ta;
186  return 2*Ta*A/InvLevelDensity - G4StatMFParameters::DBetaDT(Ta)*A13*A13;
187  } else if (Da < 0.0) {
188  do {
189  Tb -= 0.5*Tb;
190  ECompoundNucleus = CalcFreeInternalEnergy(theFragment,Tb);
191  Db = (U+__FreeInternalE0-ECompoundNucleus)/U;
192  } while (Db < 0.0);
193  } else {
194  do {
195  Tb += 0.5*Tb;
196  ECompoundNucleus = CalcFreeInternalEnergy(theFragment,Tb);
197  Db = (U+__FreeInternalE0-ECompoundNucleus)/U;
198  } while (Db > 0.0);
199  }
200 
201  G4double eps = 1.0e-14 * std::abs(Tb-Ta);
202 
203  for (G4int i = 0; i < 1000; i++) {
204  G4double Tc = (Ta+Tb)*0.5;
205  if (std::abs(Ta-Tb) <= eps) {
206  TConf = Tc;
207  return 2*Tc*A/InvLevelDensity - G4StatMFParameters::DBetaDT(Tc)*A13*A13;
208  }
209  ECompoundNucleus = CalcFreeInternalEnergy(theFragment,Tc);
210  G4double Dc = (U+__FreeInternalE0-ECompoundNucleus)/U;
211 
212  if (Dc == 0.0) {
213  TConf = Tc;
214  return 2*Tc*A/InvLevelDensity - G4StatMFParameters::DBetaDT(Tc)*A13*A13;
215  }
216 
217  if (Da*Dc < 0.0) {
218  Tb = Tc;
219  Db = Dc;
220  } else {
221  Ta = Tc;
222  Da = Dc;
223  }
224  }
225 
226  G4cout <<
227  "G4StatMFMicrocanoncal::CalcEntropyOfCompoundNucleus: I can't calculate the temperature"
228  << G4endl;
229 
230  return 0.0;
231 }

233 G4StatMFChannel * G4StatMFMicroCanonical::ChooseAandZ(const G4Fragment & theFragment)
234  // Choice of fragment atomic numbers and charges
235 {
236  // We choose a multiplicity (1,2,3,...) and then a channel
237  G4double RandNumber = G4UniformRand();
238 
239  if (RandNumber < _WCompoundNucleus) {
240 
241  G4StatMFChannel * aChannel = new G4StatMFChannel;
242  aChannel->CreateFragment(theFragment.GetA_asInt(),theFragment.GetZ_asInt());
243  return aChannel;
244 
245  } else {
246 
247  G4double AccumWeight = _WCompoundNucleus;
248  std::vector<G4StatMFMicroManager*>::iterator it;
249  for (it = _ThePartitionManagerVector.begin(); it != _ThePartitionManagerVector.end(); ++it) {
250  AccumWeight += (*it)->GetProbability();
251  if (RandNumber < AccumWeight) {
252  return (*it)->ChooseChannel(theFragment.GetA_asInt(),theFragment.GetZ_asInt(),__MeanTemperature);
253  }
254  }
255  throw G4HadronicException(__FILE__, __LINE__, "G4StatMFMicroCanonical::ChooseAandZ: wrong normalization!");
256  }
257 
258  return 0;
