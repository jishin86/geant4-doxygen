147 void Run::Merge(const G4Run* run)
148 {
149  const Run* localRun = static_cast<const Run*>(run);
150 
151  //primary particle info
152  //
153  fParticle = localRun->fParticle;
154  fEkin = localRun->fEkin;
155 
156  // accumulate sums
157  //
158  fEdepTarget += localRun->fEdepTarget;
159  fEdepTarget2 += localRun->fEdepTarget2;
160  fEdepDetect += localRun->fEdepDetect;
161  fEdepDetect2 += localRun->fEdepDetect2;
162 
163  //map: processes count in target
164 
165  std::map<G4String,G4int>::const_iterator itp1;
166  for ( itp1 = localRun->fProcCounter1.begin();
167  itp1 != localRun->fProcCounter1.end(); ++itp1 ) {
168 
169  G4String procName = itp1->first;
170  G4int localCount = itp1->second;
171  if ( fProcCounter1.find(procName) == fProcCounter1.end()) {
172  fProcCounter1[procName] = localCount;
173  }
174  else {
175  fProcCounter1[procName] += localCount;
176  }
177  }
178 
179  //map: processes count in detector
180 
181  std::map<G4String,G4int>::const_iterator itp2;
182  for ( itp2 = localRun->fProcCounter2.begin();
183  itp2 != localRun->fProcCounter2.end(); ++itp2 ) {
184 
185  G4String procName = itp2->first;
186  G4int localCount = itp2->second;
187  if ( fProcCounter2.find(procName) == fProcCounter2.end()) {
188  fProcCounter2[procName] = localCount;
189  }
190  else {
191  fProcCounter2[procName] += localCount;
192  }
193  }
194 
195  //map: created particles in target
196  std::map<G4String,ParticleData>::const_iterator itc;
197  for (itc = localRun->fParticleDataMap1.begin();
198  itc != localRun->fParticleDataMap1.end(); ++itc) {
199 
200  G4String name = itc->first;
201  const ParticleData& localData = itc->second;
202  if ( fParticleDataMap1.find(name) == fParticleDataMap1.end()) {
203  fParticleDataMap1[name]
204  = ParticleData(localData.fCount,
205  localData.fEmean,
206  localData.fEmin,
207  localData.fEmax);
208  }
209  else {
210  ParticleData& data = fParticleDataMap1[name];
211  data.fCount += localData.fCount;
212  data.fEmean += localData.fEmean;
213  G4double emin = localData.fEmin;
214  if (emin < data.fEmin) data.fEmin = emin;
215  G4double emax = localData.fEmax;
216  if (emax > data.fEmax) data.fEmax = emax;
217  }
218  }
219 
220  //map: created particle in detector
221  std::map<G4String,ParticleData>::const_iterator itn;
222  for (itn = localRun->fParticleDataMap2.begin();
223  itn != localRun->fParticleDataMap2.end(); ++itn) {
224 
225  G4String name = itn->first;
226  const ParticleData& localData = itn->second;
227  if ( fParticleDataMap2.find(name) == fParticleDataMap2.end()) {
228  fParticleDataMap2[name]
229  = ParticleData(localData.fCount,
230  localData.fEmean,
231  localData.fEmin,
232  localData.fEmax);
233  }
234  else {
235  ParticleData& data = fParticleDataMap2[name];
236  data.fCount += localData.fCount;
237  data.fEmean += localData.fEmean;
238  G4double emin = localData.fEmin;
239  if (emin < data.fEmin) data.fEmin = emin;
240  G4double emax = localData.fEmax;
241  if (emax > data.fEmax) data.fEmax = emax;
242  }
243  }
244 
245  G4Run::Merge(run);
246 }

385 void Run::WriteActivity(G4int nevent)
386 {
387  G4ProcessTable *pTable = G4ProcessTable::GetProcessTable();
388  G4RadioactiveDecay * rDecay = (G4RadioactiveDecay *)
389  pTable->FindProcess("RadioactiveDecay", "GenericIon");
390 
391  // output the induced radioactivities (in VR mode only)
392  //
393  if ((rDecay == 0) || (rDecay->IsAnalogueMonteCarlo())) return;
394 
395  G4String fileName = G4AnalysisManager::Instance()->GetFileName() + ".activity";
396  std::ofstream outfile (fileName, std::ios::out );
397 
398  std::vector<G4RadioactivityTable*> theTables =
399  rDecay->GetTheRadioactivityTables();
400 
401  for (size_t i = 0 ; i < theTables.size(); i++) {
402  G4double rate, error;
403  outfile << "Radioactivities in decay window no. " << i << G4endl;
404  outfile << "Z \tA \tE \tActivity (decays/window) \tError (decays/window) "
405  << G4endl;
406 
407  map<G4ThreeVector,G4TwoVector> *aMap = theTables[i]->GetTheMap();
408  map<G4ThreeVector,G4TwoVector>::iterator iter;
409  for (iter=aMap->begin(); iter != aMap->end(); iter++) {
410  rate = iter->second.x()/nevent;
411  error = std::sqrt(iter->second.y())/nevent;
412  if (rate < 0.) rate = 0.; // statically it can be < 0.
413  outfile << iter->first.x() <<"\t"<< iter->first.y() <<"\t"
414  << iter->first.z() << "\t" << rate <<"\t" << error << G4endl;
415  }
416  outfile << G4endl;
417  }
418  outfile.close();
419 }
