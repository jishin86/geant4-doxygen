99 void DicomRun::ConstructMFD(const std::vector<G4String>& mfdName)
100 {
101 
102  G4SDManager* SDman = G4SDManager::GetSDMpointer();
103  //=================================================
104  // Initalize RunMaps for accumulation.
105  // Get CollectionIDs for HitCollections.
106  //=================================================
107  G4int Nmfd = mfdName.size();
108  for ( G4int idet = 0; idet < Nmfd ; idet++){ // Loop for all MFD.
109  G4String detName = mfdName[idet];
110  //--- Seek and Obtain MFD objects from SDmanager.
111  G4MultiFunctionalDetector* mfd =
112  (G4MultiFunctionalDetector*)(SDman->FindSensitiveDetector(detName));
113  //
114  if ( mfd ){
115  //--- Loop over the registered primitive scorers.
116  for (G4int icol = 0; icol < mfd->GetNumberOfPrimitives(); icol++){
117  // Get Primitive Scorer object.
118  G4VPrimitiveScorer* scorer = mfd->GetPrimitive(icol);
119  // collection name and collectionID for HitsCollection,
120  // where type of HitsCollection is G4THitsMap in case
121  // of primitive scorer.
122  // The collection name is given by <MFD name>/<Primitive
123  // Scorer name>.
124  G4String collectionName = scorer->GetName();
125  G4String fullCollectionName = detName+"/"+collectionName;
126  G4int collectionID = SDman->GetCollectionID(fullCollectionName);
127  //
128  if ( collectionID >= 0 ){
129  G4cout << "++ "<<fullCollectionName<< " id " << collectionID
130  << G4endl;
131  // Store obtained HitsCollection information into data members.
132  // And, creates new G4THitsMap for accumulating quantities during RUN.
133  fCollName.push_back(fullCollectionName);
134  fCollID.push_back(collectionID);
135  fRunMap.push_back(new G4THitsMap<G4double>(detName,collectionName));
136  } else {
137  G4cout << "** collection " << fullCollectionName << " not found. "
138  <<G4endl;
139  }
140  }
141  }
142  }
143 }

151 {
152  // disable below because this is done in G4Run::RecordEvent(aEvent);
153  //numberOfEvent++; // This is an original line.

155  //G4cout << "Dicom Run :: Recording event " << aEvent->GetEventID()

157  //=============================
158  // HitsCollection of This Event
159  //============================
160  G4HCofThisEvent* HCE = aEvent->GetHCofThisEvent();
161  if (!HCE) return;
162 
163  //=======================================================
164  // Sum up HitsMap of this Event into HitsMap of this RUN
165  //=======================================================
166  G4int Ncol = fCollID.size();
167  for ( G4int i = 0; i < Ncol ; i++ ){ // Loop over HitsCollection
168  G4THitsMap<G4double>* EvtMap = 0;
169  if ( fCollID[i] >= 0 ){ // Collection is attached to HCE
170  EvtMap = static_cast<G4THitsMap<G4double>*>(HCE->GetHC(fCollID[i]));
171  }else{
172  G4cout <<" Error EvtMap Not Found "<< i << G4endl;
173  }
174  if ( EvtMap ) {
175  //=== Sum up HitsMap of this event to HitsMap of RUN.===
176  *fRunMap[i] += *EvtMap;
177  //G4cout << "Summing EvtMap into RunMap at " << i << "..." << G4endl;
178  //======================================================
179  } //else {
180  //G4cout << "Null pointer to EvtMap at " << i << "..." << G4endl;
181  //}
182  }
183 
184  G4Run::RecordEvent(aEvent);

190 void DicomRun::Merge(const G4Run* aRun)
191 {
192  const DicomRun* localRun = static_cast<const DicomRun*>(aRun);
193  Copy(fCollName, localRun->fCollName);
194  Copy(fCollID, localRun->fCollID);
195  unsigned ncopies = Copy(fRunMap, localRun->fRunMap);
196  // copy function returns the fRunMap size if all data is copied
197  // so this loop isn't executed the first time around
198  for(unsigned i = ncopies; i < fRunMap.size(); ++i) {
199  *fRunMap[i] += *localRun->fRunMap[i];
200  }
201  G4Run::Merge(aRun);
202 }
