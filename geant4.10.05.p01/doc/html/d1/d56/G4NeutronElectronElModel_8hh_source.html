103 
104 inline G4double G4NeutronElectronElModel::CalculateAm( G4double Tkin)
105 {
106  fee = (Tkin+fM)*fme/fM;
107  // G4cout<<"fee = "<<fee<<" MeV"<<G4endl;
108  fee2 = fee*fee;
109  G4double momentum = sqrt( fee2 - fme2 );
110  G4double k = momentum/CLHEP::hbarc;
111  G4double ch = 1.13;
112  G4double zn = 1.77*k*CLHEP::Bohr_radius;
113  G4double zn2 = zn*zn;
114  fAm = ch/zn2;
115 
116  return fAm;
117 }
