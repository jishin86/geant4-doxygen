308 inline
309 G4double G4VTwistedFaceted::GetCubicVolume()
310 {
311  if(fCubicVolume != 0.) ;
312  else fCubicVolume = 2 * fDz
313  * ( ( fDx1 + fDx2 ) * fDy1 + ( fDx3 + fDx4 ) * fDy2 );
314  return fCubicVolume;
315 }

325 inline
326 G4double G4VTwistedFaceted::GetValueA(G4double phi) const
327 {
328  return ( fDx4 + fDx2 + ( fDx4 - fDx2 ) * ( 2 * phi ) / fPhiTwist ) ;
329 }
330 
331 inline
332 G4double G4VTwistedFaceted::GetValueD(G4double phi) const
333 {
334  return ( fDx3 + fDx1 + ( fDx3 - fDx1 ) * ( 2 * phi ) / fPhiTwist ) ;
335 }
336 
337 inline
338 G4double G4VTwistedFaceted::GetValueB(G4double phi) const
339 {
340  return ( fDy2 + fDy1 + ( fDy2 - fDy1 ) * ( 2 * phi ) / fPhiTwist ) ;
341 }

343 inline
344 G4double G4VTwistedFaceted::Xcoef(G4double u, G4double phi, G4double ftg) const
345 {
346  return GetValueA(phi)/2. + (GetValueD(phi)-GetValueA(phi))/4.
347  - u*( ( GetValueD(phi)-GetValueA(phi) ) / ( 2 * GetValueB(phi) ) - ftg );
348 }
