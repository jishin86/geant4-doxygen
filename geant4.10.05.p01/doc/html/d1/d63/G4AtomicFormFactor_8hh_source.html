70  //
71  // theCoefficientsMap stores the coefficients for the form factor
72  // calculations. It can be loaded only by LoadCoefficiencts()
73  // and accessed by theCoefficients[].
74  //
75 private:
76  std::map<G4int,std::vector<G4double> > theCoefficientsMap;
77  G4double theCoefficients[9];
78  G4int loadedIndex;
79 
80  //
81  // LoadCoefficiencts() method allows the evaluation of the atomic form
82  // factor coefficients and the storage in theCoefficients.
83  // If theCoefficients are already correct, no need to get new ones
84  // Reference: International Tables for Crystallography (2006).
85  // Vol. C, ch. 6.1, pp. 554-595
86  // doi: 10.1107/97809553602060000600
87  // Chapter 6.1. Intensity of diffracted intensities
88  // IUCr Eq. 6.1.1.15, Coefficients Table 6.1.1.4
89  //
90 private:
91  void InsertCoefficients(G4int index,std::vector<G4double> aDoubleVec){
92  theCoefficientsMap.insert(std::pair<G4int,std::vector<G4double>>(index,aDoubleVec));
93  }
94 
95  void LoadCoefficiencts(G4int index){
96  loadedIndex = index;
97  for(unsigned int i0=0;i0<9;i0++){
98  theCoefficients[i0] = theCoefficientsMap[index][i0];
99  }
100 
101  }
102 
103  inline
104  G4int GetIndex(G4int Z, G4int charge = 0) {return Z*100 + charge;}
105 
106 
107  //
108  // Get() function gives back the Atomic Form Factor of the Z material
109  //
110 public:
111  G4double Get(G4double kScatteringVector, G4int Z, G4int charge = 0){
112  if(loadedIndex != GetIndex(Z,charge)){
113  LoadCoefficiencts(GetIndex(Z,charge));
114  }
115  G4double result = 0.;
116  G4double kVecOn4PiSquared = (kScatteringVector / 1.e-7 / 3.1415926536) * 0.125; // (k/(4pi))/ angstrom
117  kVecOn4PiSquared= kVecOn4PiSquared * kVecOn4PiSquared; // (k/(4pi))^2
118 
119  for(unsigned int i0=0;i0<4;i0++){
120  result += theCoefficients[i0*2] * G4Exp(- theCoefficients[i0*2+1] * kVecOn4PiSquared);
121  }
122  result += theCoefficients[8];
123  return result;
124  }
