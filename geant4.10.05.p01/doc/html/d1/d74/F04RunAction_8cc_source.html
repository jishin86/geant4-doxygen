59 void F04RunAction::BeginOfRunAction(const G4Run* aRun)
60 {
61  G4cout << "### Run " << aRun->GetRunID() << " start." << G4endl;
62 
63  G4RunManager::GetRunManager()->SetRandomNumberStore(true);
64  G4RunManager::GetRunManager()->SetRandomNumberStoreDir("random/");
65 
66  if (fAutoSeed) {
67  // automatic (time-based) random seeds for each run
68  G4cout << "*******************" << G4endl;
69  G4cout << "*** AUTOSEED ON ***" << G4endl;
70  G4cout << "*******************" << G4endl;
71  long seeds[2];
72  time_t systime = time(NULL);
73  seeds[0] = (long) systime;
74  seeds[1] = (long) (systime*G4UniformRand());
75  G4Random::setTheSeeds(seeds);
76  G4Random::showEngineStatus();
77  } else {
78  G4Random::showEngineStatus();
79  }
80 
81  if (fSaveRndm > 0)
82  {
83  std::ostringstream os;
84  os<<"beginOfRun_"<<G4Threading::G4GetThreadId()<<".rndm";
85  G4Random::saveEngineStatus(os.str().c_str());
86  }
87 }
