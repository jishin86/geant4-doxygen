74 void Run::RecordEvent(const G4Event* event)
75 {
76  if(event->IsAborted()) return;
77 
78  G4int CollectionID =
79  G4SDManager::GetSDMpointer()->GetCollectionID("mfDetector/Species");
80 
81  //Hits collections
82  G4HCofThisEvent* HCE = event->GetHCofThisEvent();
83  if(!HCE) return;
84 
85  G4THitsMap<G4double>* evtMap =
86  static_cast<G4THitsMap<G4double>*>(HCE->GetHC(CollectionID));
87 
88  std::map<G4int,G4double*>::iterator itr;
89  for (itr = evtMap->GetMap()->begin(); itr != evtMap->GetMap()->end(); itr++){
90  G4double edep = *(itr->second);
91  fSumEne+=edep;
92  }
93 
94  G4Run::RecordEvent(event);
95 }

99 void Run::Merge(const G4Run* aRun)

110 
111  ScoreSpecies* localScorer =

113 
114  masterScorer->AbsorbResultsFromWorkerScorer(localScorer);
115  G4Run::Merge(aRun);
