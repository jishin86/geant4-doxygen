49 G4FTFPNeutronBuilder::
50 G4FTFPNeutronBuilder(G4bool quasiElastic)
51 {
52  theMin = 4*GeV;
53  theMax = G4HadronicParameters::Instance()->GetMaxEnergy();
54  theModel = new G4TheoFSGenerator("FTFP");
55 
56  theStringModel = new G4FTFModel;
57  theStringDecay = new G4ExcitedStringDecay(theLund = new G4LundStringFragmentation);
58  theStringModel->SetFragmentationModel(theStringDecay);
59 
60  theCascade = new G4GeneratorPrecompoundInterface();
61 
62  theModel->SetTransport(theCascade);
63 
64  theModel->SetHighEnergyGenerator(theStringModel);
65  if (quasiElastic)
66  {
67  theQuasiElastic=new G4QuasiElasticChannel;
68  theModel->SetQuasiElasticChannel(theQuasiElastic);
69  } else
70  { theQuasiElastic=0;}
71 
72  theModel->SetMinEnergy(theMin);
73  theModel->SetMaxEnergy(theMax);
74 }

76 G4FTFPNeutronBuilder::
77 ~G4FTFPNeutronBuilder()
78 {
79  delete theStringDecay;

82  delete theLund;
