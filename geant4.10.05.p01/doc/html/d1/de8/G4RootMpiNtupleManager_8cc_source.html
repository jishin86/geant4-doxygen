69 G4bool G4RootMpiNtupleManager::Send(G4int id, tools::wroot::ntuple* ntuple)
70 {
71 // Pack and send the main ntuple data to the slave ranks
72 
73  // G4cout << "Going to send main ntuple data " << G4endl;
74 
75  // Get basket sizes
76  std::vector<tools::wroot::branch*> mainBranches;
77  ntuple->get_branches(mainBranches);
78  std::vector<tools::uint32> basketSizes;
79  tools_vforcit(tools::wroot::branch*, mainBranches, it) {
80  basketSizes.push_back((*it)->basket_size());
81  }
82 
83  auto ntupleFile = fFileManager->GetNtupleFile(id);
84  tools::uint32 basketSize = fFileManager->GetBasketSize();
85 
86  for ( auto slaveRank : fSlaveRanks ) {
87  G4cout << "Going to send main ntuple data to slave rank " << slaveRank << G4endl;
88 
89  fImpi->pack_reset();
90  if ( ! fImpi->pack(id)) {
91  G4cerr << "pack(id) failed." << G4endl;
92  return false;
93  }
94 
95  if ( ! fImpi->bpack(ntupleFile->byte_swap())) {
96  G4cerr << "bpack(byte_swap) failed." << G4endl;
97  return false;
98  }
99 
100  if ( ! fImpi->pack(ntupleFile->compression()) ) {
101  G4cerr << "pack(compression) failed." << G4endl;
102  return false;
103  }
104 
105  if ( ! fImpi->pack(ntupleFile->dir().seek_directory())) {
106  // Should be used fNtupleDirectory ??
107  G4cerr << "pack(seek) failed." << G4endl;
108  return false;
109  }
110 
111  if ( fRowWise ) {
112  if ( ! fImpi->pack(basketSize) ) {
113  G4cerr << "pack(basket_sizes) failed." << G4endl;
114  return false;
115  }
116  } else {
117  if ( ! fImpi->vpack(basketSizes) ) {
118  G4cerr << "vpack(basket_sizes) failed." << G4endl;
119  return false;
120  }
121  }
122 
123  if ( ! fImpi->send_buffer(slaveRank, kTAG_NTUPLE )) {
124  G4cerr << "send_buffer() failed." << G4endl;
125  return false;
126  }
127  fImpi->pack_reset();
128 
129  G4cout << "Sent ntuple description to slave on rank " << slaveRank << G4endl;
130  }
131 
132  // G4cout << "Done: send main ntuple data " << G4endl;
133  return true;
134 }

137 G4bool G4RootMpiNtupleManager::InitializeRanks()
138 {
139  // G4cout << "G4RootMpiNtupleManager::InitializeRanks" << G4endl;
140 
141  auto finalResult = true;
142 
143  auto counter = 0;
144  for ( auto ntupleDescription : fNtupleDescriptionVector ) {
145 
146  // Do not create ntuple if it is inactivated
147  if ( fState.GetIsActivation() && ( ! ntupleDescription->fActivation ) ) continue;
148 
149  auto result = Send(counter++, ntupleDescription->fNtuple);
150  finalResult = finalResult && result;
151  }
152 
153  return finalResult;
154 }

157 G4bool G4RootMpiNtupleManager::WaitBuffer()
158 {
159 // Receive the pntuple data from the slave ranks
160 // For the time being only one ntuple
161 
162  G4cout << "G4RootMpiNtupleManager::WaitBuffer" << G4endl;
163 
164  unsigned long numberOfEndFill = 0;
165 
166  G4bool verbose = ( fState.GetVerboseL2() );
167 
168  while ( true ) {
169  fImpi->pack_reset();
170 
171  // loop until receiving end_fill from all ranks
172  // G4cout << "G4RootMpiNtupleManager::WaitBuffer entering loop" << G4endl;
173  int probe_src;
174  if ( ! fImpi->wait_buffer(fMainRank, kTAG_NTUPLE, probe_src, verbose)) {
175  G4cerr << "!!! wait_buffer() failed." << std::endl;
176  return EXIT_FAILURE;
177  }
178 
179  tools::uint32 protocol;
180  if ( ! fImpi->unpack(protocol)) {
181  G4cerr << "sunpack(protocol) failed."<< G4endl;
182  return false;
183  }
184 
185  if ( protocol == tools::wroot::mpi_protocol_basket() ) {
186 
187  // G4cout << "G4RootMpiNtupleManager::WaitBuffer got protocol_basket" << G4endl;
188 
189  // get ntuple Id
190  tools::uint32 ntupleId;
191  if ( ! fImpi->unpack(ntupleId) ) {
192  G4cerr << "unpack(ntuple_id) failed."<< std::endl;
193  return false;
194  }
195 
196  if ( ntupleId >= fNtupleVector.size() ) {
197  std::cerr << "!!! unknown ntupleId " << ntupleId << std::endl;
198  return false;
199  }
200 
201  // Main ntuple
202  auto mainNtuple = fNtupleVector[ntupleId];
203 
204  // add basket to main ntuple
205  if ( ! mainNtuple->mpi_add_basket(*fImpi)) {
206  std::cerr << "mainNtuple->mpi_add_basket() failed." << std::endl;
207  return EXIT_FAILURE;
208  }
209  }
210  else if ( protocol==tools::wroot::mpi_protocol_end_fill() ) {
211 
212  // G4cout << "G4RootMpiNtupleManager::WaitBuffer got protocol_end_fill" << G4endl;
213 
214  // get ntuple Id
215  tools::uint32 ntupleId;
216  if ( ! fImpi->unpack(ntupleId) ) {
217  G4cerr << "unpack(ntuple_id) failed."<< std::endl;
218  return false;
219  }
220 
221  if ( ntupleId >= fNtupleVector.size() ) {
222  std::cerr << "!!! unknown ntupleId " << ntupleId << std::endl;
223  return false;
224  }
225 
226  // Main ntuple
227  auto mainNtuple = fNtupleVector[ntupleId];
228 
229  // end_fill in main ntuple
230  if ( ! mainNtuple->mpi_end_fill(*fImpi) ) {
231  G4cerr << "main_ntuple->mpi_end_fill() failed." << std::endl;
232  return false;
233  }
234 
235  numberOfEndFill++;
236 
237  if ( numberOfEndFill == fSlaveRanks.size() ) break;
238  }
239  else {
240  G4cerr << "unknown protocol " << protocol << G4endl;
241  return false;
242  }
243  }
244 
245  return true;
246 }
