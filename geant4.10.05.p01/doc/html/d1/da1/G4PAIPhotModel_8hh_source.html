169 inline G4int G4PAIPhotModel::FindCoupleIndex(const G4MaterialCutsCouple* couple)
170 {
171  G4int idx = -1;
172  size_t jMatMax = fMaterialCutsCoupleVector.size();
173  for(size_t jMat = 0;jMat < jMatMax; ++jMat) {
174  if(couple == fMaterialCutsCoupleVector[jMat]) {
175  idx = jMat;
176  break;
177  }
178  }
179  return idx;
180 }
