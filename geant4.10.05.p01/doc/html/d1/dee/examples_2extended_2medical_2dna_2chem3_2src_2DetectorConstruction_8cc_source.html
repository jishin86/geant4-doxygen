97 G4VPhysicalVolume* DetectorConstruction::ConstructDetector()
98 {
99  G4Material *water = OtherMaterial("G4_WATER");
100 
101  // WORLD VOLUME = an G4ORB FULL OF LIQUID WATER
102 
103  double worldRadius = 0.3*micrometer;
104 
105  G4Orb* solidWorld = new G4Orb("World", worldRadius);
106 
107  G4LogicalVolume* logicWorld =
108  new G4LogicalVolume(solidWorld, //its solid
109  water, //its material
110  "World"); //its name
111 
112  G4VPhysicalVolume* physiWorld =
113  new G4PVPlacement(0, //no rotation
114  G4ThreeVector(), //at (0,0,0)
115  "World", //its name
116  logicWorld, //its logical volume
117  0, //its mother volume
118  false, //no boolean operation
119  0); //copy number
120 
121  // Visualization attributes = semitransparent navy blue
122  G4VisAttributes* worldVisAtt =
123  new G4VisAttributes(G4Colour(0.0,0.4,0.8,0.5));
124  worldVisAtt->SetForceSolid(true);
125  worldVisAtt->SetVisibility(true);
126  logicWorld->SetVisAttributes(worldVisAtt);
127 
128  return physiWorld;
129 }
