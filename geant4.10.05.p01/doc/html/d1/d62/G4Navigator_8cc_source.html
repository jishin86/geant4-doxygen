111 // ********************************************************************
112 // LocateGlobalPointAndSetup
113 //
114 // Locate the point in the hierarchy return 0 if outside
115 // The direction is required
116 // - if on an edge shared by more than two surfaces
117 // (to resolve likely looping in tracking)
118 // - at initial location of a particle
119 // (to resolve potential ambiguity at boundary)
120 //
121 // Flags on exit: (comments to be completed)
122 // fEntering - True if entering `daughter' volume (or replica)
123 // whether daughter of last mother directly
124 // or daughter of that volume's ancestor.
125 // fExiting - True if exited 'mother' volume
126 // (always ? - how about if going back down ? - tbc)
127 // ********************************************************************
128 //
129 G4VPhysicalVolume*
130 G4Navigator::LocateGlobalPointAndSetup( const G4ThreeVector& globalPoint,
131  const G4ThreeVector* pGlobalDirection,
132  const G4bool relativeSearch,
133  const G4bool ignoreDirection )
134 {
135  G4bool notKnownContained=true, noResult;
136  G4VPhysicalVolume *targetPhysical;
137  G4LogicalVolume *targetLogical;
138  G4VSolid *targetSolid=0;
139  G4ThreeVector localPoint, globalDirection;
140  EInside insideCode;
141 
142  G4bool considerDirection = (!ignoreDirection) || fLocatedOnEdge;
143 
144  fLastTriedStepComputation= false;
145  fChangedGrandMotherRefFrame= false; // For local exit normal
146 
147  if( considerDirection && pGlobalDirection != 0 )
148  {
149  globalDirection=*pGlobalDirection;
150  }
151 
152 #ifdef G4VERBOSE
153  if( fVerbose > 2 )
154  {
155  G4int oldcoutPrec = G4cout.precision(8);
156  G4cout << "*** G4Navigator::LocateGlobalPointAndSetup: ***" << G4endl;
157  G4cout << " Called with arguments: " << G4endl
158  << " Globalpoint = " << globalPoint << G4endl
159  << " RelativeSearch = " << relativeSearch << G4endl;
160  if( fVerbose >= 4 )
161  {
162  G4cout << " ----- Upon entering:" << G4endl;
163  PrintState();
164  }
165  G4cout.precision(oldcoutPrec);
166  }
167 #endif
168 
169  G4int noLevelsExited=0 ;
170  G4int noLevelsEntered= 0;
171 
172  if ( !relativeSearch )
173  {
174  ResetStackAndState();
175  }
176  else
177  {
178  if ( fWasLimitedByGeometry )
179  {
180  fWasLimitedByGeometry = false;
181  fEnteredDaughter = fEntering; // Remember
182  fExitedMother = fExiting; // Remember
183  if ( fExiting )
184  {
185  noLevelsExited++; // count this first level entered too
186 
187  if ( fHistory.GetDepth() )
188  {
189  fBlockedPhysicalVolume = fHistory.GetTopVolume();
190  fBlockedReplicaNo = fHistory.GetTopReplicaNo();
191  fHistory.BackLevel();
192  }
193  else
194  {
195  fLastLocatedPointLocal = localPoint;
196  fLocatedOutsideWorld = true;
197  fBlockedPhysicalVolume = 0; // to be sure
198  fBlockedReplicaNo = -1;
199  fEntering = false; // No longer
200  fEnteredDaughter = false;
201  fExitedMother = true; // ??
202 
203  return 0; // Have exited world volume
204  }
205  // A fix for the case where a volume is "entered" at an edge
206  // and a coincident surface exists outside it.
207  // - This stops it from exiting further volumes and cycling
208  // - However ReplicaNavigator treats this case itself
209  //
210  // assert( fBlockedPhysicalVolume!=0 );
211 
212  // Expect to be on edge => on surface
213  //
214  if ( fLocatedOnEdge && (VolumeType(fBlockedPhysicalVolume)!=kReplica ))
215  {
216  fExiting= false;
217  // Consider effect on Exit Normal !?
218  }
219  }
220  else
221  if ( fEntering )
222  {
223  // assert( fBlockedPhysicalVolume!=0 );
224 
225  noLevelsEntered++; // count the first level entered too
226 
227  switch (VolumeType(fBlockedPhysicalVolume))
228  {
229  case kNormal:
230  fHistory.NewLevel(fBlockedPhysicalVolume, kNormal,
231  fBlockedPhysicalVolume->GetCopyNo());
232  break;
233  case kReplica:
234  freplicaNav.ComputeTransformation(fBlockedReplicaNo,
235  fBlockedPhysicalVolume);
236  fHistory.NewLevel(fBlockedPhysicalVolume, kReplica,
237  fBlockedReplicaNo);
238  fBlockedPhysicalVolume->SetCopyNo(fBlockedReplicaNo);
239  break;
240  case kParameterised:
241  if( fBlockedPhysicalVolume->GetRegularStructureId() == 0 )
242  {
243  G4VSolid *pSolid;
244  G4VPVParameterisation *pParam;
245  G4TouchableHistory parentTouchable( fHistory );
246  pParam = fBlockedPhysicalVolume->GetParameterisation();
247  pSolid = pParam->ComputeSolid(fBlockedReplicaNo,
248  fBlockedPhysicalVolume);
249  pSolid->ComputeDimensions(pParam, fBlockedReplicaNo,
250  fBlockedPhysicalVolume);
251  pParam->ComputeTransformation(fBlockedReplicaNo,
252  fBlockedPhysicalVolume);
253  fHistory.NewLevel(fBlockedPhysicalVolume, kParameterised,
254  fBlockedReplicaNo);
255  fBlockedPhysicalVolume->SetCopyNo(fBlockedReplicaNo);
256  //
257  // Set the correct solid and material in Logical Volume
258  //
259  G4LogicalVolume *pLogical;
260  pLogical = fBlockedPhysicalVolume->GetLogicalVolume();
261  pLogical->SetSolid( pSolid );
262  pLogical->UpdateMaterial(pParam ->
263  ComputeMaterial(fBlockedReplicaNo,
264  fBlockedPhysicalVolume,
265  &parentTouchable));
266  }
267  break;
268  }
269  fEntering = false;
270  fBlockedPhysicalVolume = 0;
271  localPoint = fHistory.GetTopTransform().TransformPoint(globalPoint);
272  notKnownContained = false;
273  }
274  }
275  else
276  {
277  fBlockedPhysicalVolume = 0;
278  fEntering = false;
279  fEnteredDaughter = false; // Full Step was not taken, did not enter
280  fExiting = false;
281  fExitedMother = false; // Full Step was not taken, did not exit
282  }
283  }
284  //
285  // Search from top of history up through geometry until
286  // containing volume found:
287  // If on
288  // o OUTSIDE - Back up level, not/no longer exiting volumes
289  // o SURFACE and EXITING - Back up level, setting new blocking no.s
290  // else
291  // o containing volume found
292  //
293 
294  while (notKnownContained) // Loop checking, 07.10.2016, J.Apostolakis
295  {
296  if ( fHistory.GetTopVolumeType()!=kReplica )
297  {
298  targetSolid = fHistory.GetTopVolume()->GetLogicalVolume()->GetSolid();
299  localPoint = fHistory.GetTopTransform().TransformPoint(globalPoint);
300  insideCode = targetSolid->Inside(localPoint);
301 #ifdef G4VERBOSE
302  if(( fVerbose == 1 ) && ( fCheck ))
303  {
304  G4String solidResponse = "-kInside-";
305  if (insideCode == kOutside)
306  solidResponse = "-kOutside-";
307  else if (insideCode == kSurface)
308  solidResponse = "-kSurface-";
309  G4cout << "*** G4Navigator::LocateGlobalPointAndSetup(): ***" << G4endl
310  << " Invoked Inside() for solid: " << targetSolid->GetName()
311  << ". Solid replied: " << solidResponse << G4endl
312  << " For local point p: " << localPoint << G4endl;
313  }
314 #endif
315  }
316  else
317  {
318  insideCode = freplicaNav.BackLocate(fHistory, globalPoint, localPoint,
319  fExiting, notKnownContained);
320  // !CARE! if notKnownContained returns false then the point is within
321  // the containing placement volume of the replica(s). If insidecode
322  // will result in the history being backed up one level, then the
323  // local point returned is the point in the system of this new level
324  }
325 
326 
327  if ( insideCode==kOutside )
328  {
329  noLevelsExited++;
330  if ( fHistory.GetDepth() )
331  {
332  fBlockedPhysicalVolume = fHistory.GetTopVolume();
333  fBlockedReplicaNo = fHistory.GetTopReplicaNo();
334  fHistory.BackLevel();
335  fExiting = false;
336 
337  if( noLevelsExited > 1 )
338  {
339  // The first transformation was done by the sub-navigator
340  //
341  const G4RotationMatrix* mRot = fBlockedPhysicalVolume->GetRotation();
342  if( mRot )
343  {
344  fGrandMotherExitNormal *= (*mRot).inverse();
345  fChangedGrandMotherRefFrame= true;
346  }
347  }
348  }
349  else
350  {
351  fLastLocatedPointLocal = localPoint;
352  fLocatedOutsideWorld = true;
353  // No extra transformation for ExitNormal - is in frame of Top Volume
354  return 0; // Have exited world volume
355  }
356  }
357  else
358  if ( insideCode==kSurface )
359  {
360  G4bool isExiting = fExiting;
361  if( (!fExiting)&&considerDirection )
362  {
363  // Figure out whether we are exiting this level's volume
364  // by using the direction
365  //
366  G4bool directionExiting = false;
367  G4ThreeVector localDirection =
368  fHistory.GetTopTransform().TransformAxis(globalDirection);
369 
370  // Make sure localPoint in correct reference frame
371  // ( Was it already correct ? How ? )
372  //
373  localPoint= fHistory.GetTopTransform().TransformPoint(globalPoint);
374  if ( fHistory.GetTopVolumeType()!=kReplica )
375  {
376  G4ThreeVector normal = targetSolid->SurfaceNormal(localPoint);
377  directionExiting = normal.dot(localDirection) > 0.0;
378  isExiting = isExiting || directionExiting;
379  }
380  }
381  if( isExiting )
382  {
383  noLevelsExited++;
384  if ( fHistory.GetDepth() )
385  {
386  fBlockedPhysicalVolume = fHistory.GetTopVolume();
387  fBlockedReplicaNo = fHistory.GetTopReplicaNo();
388  fHistory.BackLevel();
389  //
390  // Still on surface but exited volume not necessarily convex
391  //
392  fValidExitNormal = false;
393 
394  if( noLevelsExited > 1 )
395  {
396  // The first transformation was done by the sub-navigator
397  //
398  const G4RotationMatrix* mRot =
399  fBlockedPhysicalVolume->GetRotation();
400  if( mRot )
401  {
402  fGrandMotherExitNormal *= (*mRot).inverse();
403  fChangedGrandMotherRefFrame= true;
404  }
405  }
406  }
407  else
408  {
409  fLastLocatedPointLocal = localPoint;
410  fLocatedOutsideWorld = true;
411  // No extra transformation for ExitNormal, is in frame of Top Vol
412  return 0; // Have exited world volume
413  }
414  }
415  else
416  {
417  notKnownContained=false;
418  }
419  }
420  else
421  {
422  notKnownContained=false;
423  }
424  } // END while (notKnownContained)
425  //
426  // Search downwards until deepest containing volume found,
427  // blocking fBlockedPhysicalVolume/BlockedReplicaNum
428  //
429  // 3 Cases:
430  //
431  // o Parameterised daughters
432  // =>Must be one G4PVParameterised daughter & voxels
433  // o Positioned daughters & voxels
434  // o Positioned daughters & no voxels
435 
436  noResult = true; // noResult should be renamed to
437  // something like enteredLevel, as that is its meaning.
438  do
439  {
440  // Determine `type' of current mother volume
441  //
442  targetPhysical = fHistory.GetTopVolume();
443  if (!targetPhysical) { break; }
444  targetLogical = targetPhysical->GetLogicalVolume();
445  switch( CharacteriseDaughters(targetLogical) )
446  {
447  case kNormal:
448  if ( targetLogical->GetVoxelHeader() ) // use optimised navigation
449  {
450  noResult = fvoxelNav.LevelLocate(fHistory,
451  fBlockedPhysicalVolume,
452  fBlockedReplicaNo,
453  globalPoint,
454  pGlobalDirection,
455  considerDirection,
456  localPoint);
457  }
458  else // do not use optimised navigation
459  {
460  noResult = fnormalNav.LevelLocate(fHistory,
461  fBlockedPhysicalVolume,
462  fBlockedReplicaNo,
463  globalPoint,
464  pGlobalDirection,
465  considerDirection,
466  localPoint);
467  }
468  break;
469  case kReplica:
470  noResult = freplicaNav.LevelLocate(fHistory,
471  fBlockedPhysicalVolume,
472  fBlockedReplicaNo,
473  globalPoint,
474  pGlobalDirection,
475  considerDirection,
476  localPoint);
477  break;
478  case kParameterised:
479  if( GetDaughtersRegularStructureId(targetLogical) != 1 )
480  {
481  noResult = fparamNav.LevelLocate(fHistory,
482  fBlockedPhysicalVolume,
483  fBlockedReplicaNo,
484  globalPoint,
485  pGlobalDirection,
486  considerDirection,
487  localPoint);
488  }
489  else // Regular structure
490  {
491  noResult = fregularNav.LevelLocate(fHistory,
492  fBlockedPhysicalVolume,
493  fBlockedReplicaNo,
494  globalPoint,
495  pGlobalDirection,
496  considerDirection,
497  localPoint);
498  }
499  break;
500  }
501 
502  // LevelLocate returns true if it finds a daughter volume
503  // in which globalPoint is inside (or on the surface).
504 
505  if ( noResult )
506  {
507  noLevelsEntered++;
508 
509  // Entering a daughter after ascending
510  //
511  // The blocked volume is no longer valid - it was for another level
512  //
513  fBlockedPhysicalVolume = 0;
514  fBlockedReplicaNo = -1;
515 
516  // fEntering should be false -- else blockedVolume is assumed good.
517  // fEnteredDaughter is used for ExitNormal
518  //
519  fEntering = false;
520  fEnteredDaughter = true;
521 
522  if( fExitedMother )
523  {
524  G4VPhysicalVolume* enteredPhysical = fHistory.GetTopVolume();
525  const G4RotationMatrix* mRot = enteredPhysical->GetRotation();
526  if( mRot )
527  {
528  // Go deeper, i.e. move 'down' in the hierarchy
529  // Apply direct rotation, not inverse
530  //
531  fGrandMotherExitNormal *= (*mRot);
532  fChangedGrandMotherRefFrame= true;
533  }
534  }
535 
536 #ifdef G4DEBUG_NAVIGATION
537  if( fVerbose > 2 )
538  {
539  G4VPhysicalVolume* enteredPhysical = fHistory.GetTopVolume();
540  G4cout << "*** G4Navigator::LocateGlobalPointAndSetup() ***" << G4endl;
541  G4cout << " Entering volume: " << enteredPhysical->GetName()
542  << G4endl;
543  }
544 #endif
545  }
546  } while (noResult); // Loop checking, 07.10.2016, J.Apostolakis
547 
548  fLastLocatedPointLocal = localPoint;
549 
550 #ifdef G4VERBOSE
551  if( fVerbose >= 4 )
552  {
553  G4int oldcoutPrec = G4cout.precision(8);
554  G4String curPhysVol_Name("None");
555  if (targetPhysical) { curPhysVol_Name = targetPhysical->GetName(); }
556  G4cout << " Return value = new volume = " << curPhysVol_Name << G4endl;
557  G4cout << " ----- Upon exiting:" << G4endl;
558  PrintState();
559  if( fVerbose >= 5 )
560  {
561  G4cout << "Upon exiting LocateGlobalPointAndSetup():" << G4endl;
562  G4cout << " History = " << G4endl << fHistory << G4endl << G4endl;
563  }
564  G4cout.precision(oldcoutPrec);
565  }
566 #endif
567 
568  fLocatedOutsideWorld= false;
569 
570  return targetPhysical;
571 }

573 // ********************************************************************
574 // LocateGlobalPointWithinVolume
575 //
576 // -> the state information of this Navigator and its subNavigators
577 // is updated in order to start the next step at pGlobalpoint
578 // -> no check is performed whether pGlobalpoint is inside the
579 // original volume (this must be the case).
580 //
581 // Note: a direction could be added to the arguments, to aid in future
582 // optional checking (via the old code below, flagged by OLD_LOCATE).
583 // [ This would be done only in verbose mode ]
584 // ********************************************************************
585 //
586 void
587 G4Navigator::LocateGlobalPointWithinVolume(const G4ThreeVector& pGlobalpoint)
588 {
589 #ifdef G4DEBUG_NAVIGATION
590  // Check: Either step was not limited by a boundary
591  // or else the full step is no longer being taken
592  assert( !fWasLimitedByGeometry );
593 #endif
594 
595  fLastLocatedPointLocal = ComputeLocalPoint(pGlobalpoint);
596  fLastTriedStepComputation= false;
597  fChangedGrandMotherRefFrame= false; // Frame for Exit Normal
598 
599 #ifdef G4DEBUG_NAVIGATION
600  if( fVerbose > 2 )
601  {
602  G4cout << "Entering LocateGlobalWithinVolume(): History = " << G4endl;
603  G4cout << fHistory << G4endl;
604  }
605 #endif
606 
607  // For the case of Voxel (or Parameterised) volume the respective
608  // Navigator must be messaged to update its voxel information etc
609 
610  // Update the state of the Sub Navigators
611  // - in particular any voxel information they store/cache
612  //
613  G4VPhysicalVolume* motherPhysical = fHistory.GetTopVolume();
614  G4LogicalVolume* motherLogical = motherPhysical->GetLogicalVolume();
615  G4SmartVoxelHeader* pVoxelHeader = motherLogical->GetVoxelHeader();
616 
617  if ( fHistory.GetTopVolumeType()!=kReplica )
618  {
619  switch( CharacteriseDaughters(motherLogical) )
620  {
621  case kNormal:
622  if ( pVoxelHeader )
623  {
624  fvoxelNav.VoxelLocate( pVoxelHeader, fLastLocatedPointLocal );
625  }
626  break;
627  case kParameterised:
628  if( GetDaughtersRegularStructureId(motherLogical) != 1 )
629  {
630  // Resets state & returns voxel node
631  //
632  fparamNav.ParamVoxelLocate( pVoxelHeader, fLastLocatedPointLocal );
633  }
634  break;
635  case kReplica:
636  G4Exception("G4Navigator::LocateGlobalPointWithinVolume()",
637  "GeomNav0001", FatalException,
638  "Not applicable for replicated volumes.");
639  break;
640  }
641  }
642 
643  // Reset the state variables
644  // - which would have been affected
645  // by the 'equivalent' call to LocateGlobalPointAndSetup
646  // - who's values have been invalidated by the 'move'.
647  //
648  fBlockedPhysicalVolume = 0;
649  fBlockedReplicaNo = -1;
650  fEntering = false;
651  fEnteredDaughter = false; // Boundary not encountered, did not enter
652  fExiting = false;
653  fExitedMother = false; // Boundary not encountered, did not exit
654 }

1117  //

1119  }
1120  else

1122  fCalculatedExitNormal = false;

1145  //

1158  }

1163  }

1166  {
1167  // This if Step is not really limited by the geometry.
1168  // The Navigator is obliged to return "infinity"
1169  //
1170  Step = kInfinity;
1171  }
1172 
1173 #ifdef G4VERBOSE
1174  if( fVerbose > 1 )
1175  {
1176  if( fVerbose >= 4 )
1177  {
1178  G4cout << " ----- Upon exiting :" << G4endl;
1179  PrintState();
1180  }
1181  G4cout << " Returned step= " << Step;
1182  if( fVerbose > 5 ) G4cout << G4endl;

1184  {
1185  G4cout << " Requested step= " << pCurrentProposedStepLength ;
1186  if( fVerbose > 5) G4cout << G4endl;
1187  }
1188  G4cout << " Safety = " << pNewSafety << G4endl;
1189  }
1190 #endif
1191 
1192  return Step;
1193 }
