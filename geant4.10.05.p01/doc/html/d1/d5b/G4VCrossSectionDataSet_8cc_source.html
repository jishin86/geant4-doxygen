82 G4double
83 G4VCrossSectionDataSet::ComputeCrossSection(const G4DynamicParticle* part,
84  const G4Element* elm,
85  const G4Material* mat)
86 {
87  G4int Z = elm->GetZasInt();
88 
89  if (IsElementApplicable(part, Z, mat)) {
90  return GetElementCrossSection(part, Z, mat);
91  }
92 
93  // isotope-wise cross section making sum over available
94  // isotope cross sections, which may be incomplete, so
95  // the result is corrected
96  G4int nIso = elm->GetNumberOfIsotopes();
97  G4double fact = 0.0;
98  G4double xsec = 0.0;
99  const G4Isotope* iso = nullptr;
100 
101  // user-defined isotope abundances
102  const G4IsotopeVector* isoVector = elm->GetIsotopeVector();
103  const G4double* abundVector = elm->GetRelativeAbundanceVector();
104 
105  for (G4int j = 0; j<nIso; ++j) {
106  iso = (*isoVector)[j];
107  G4int A = iso->GetN();
108  if(abundVector[j] > 0.0 && IsIsoApplicable(part, Z, A, elm, mat)) {
109  fact += abundVector[j];
110  xsec += abundVector[j]*GetIsoCrossSection(part, Z, A, iso, elm, mat);
111  }
112  }
113  if(fact > 0.0) { xsec /= fact; }
114  return xsec;
115 }

150 const G4Isotope*
151 G4VCrossSectionDataSet::SelectIsotope(const G4Element* anElement, G4double)
152 {
153  G4int nIso = anElement->GetNumberOfIsotopes();
154  const G4Isotope* iso = anElement->GetIsotope(0);
155 
156  // more than 1 isotope
157  if(1 < nIso) {
158  const G4double* abundVector = anElement->GetRelativeAbundanceVector();
159  G4double sum = 0.0;
160  G4double q = G4UniformRand();
161  for (G4int j = 0; j<nIso; ++j) {
162  sum += abundVector[j];
163  if(q <= sum) {
164  iso = anElement->GetIsotope(j);
165  break;
166  }
167  }
168  }
169  return iso;
170 }
