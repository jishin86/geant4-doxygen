28 // class G4SmartVoxelHeader
29 //
30 // Class description:
31 //
32 // Represents a set of voxels, created by a single axis of virtual division.
33 // Contains the individual voxels, which are potentially further divided
34 // along different axes.
35 //
36 // Member data:
37 //
38 // EAxis faxis
39 // - The (cartesian) slicing/division axis
40 // G4double fmaxExtent
41 // G4double fminExtent
42 // - Minimum and maximum coordiantes along the axis
43 // std::vector<G4SmartVoxelProxy*> fslices
44 // - The slices along the axis
45 //
46 // G4int fminEquivalent
47 // G4int fmaxEquivalent
48 // - Minimum and maximum equivalent slice nos.
49 // [Applies to the level of the header, not its nodes]
