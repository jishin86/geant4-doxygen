64 void WLSOpticalPhysics::ConstructProcess()
65 {
66  G4cout << "WLSOpticalPhysics:: Add Optical Physics Processes"
67  << G4endl;
68 
69  fWLSProcess = new G4OpWLS();
70 
71  fScintProcess = new G4Scintillation();
72  fScintProcess->SetScintillationYieldFactor(1.);
73  fScintProcess->SetTrackSecondariesFirst(true);
74 
75  fCerenkovProcess = new G4Cerenkov();
76  fCerenkovProcess->SetMaxNumPhotonsPerStep(300);
77  fCerenkovProcess->SetTrackSecondariesFirst(true);
78 
79  fAbsorptionProcess = new G4OpAbsorption();
80  fRayleighScattering = new G4OpRayleigh();
81  fMieHGScatteringProcess = new G4OpMieHG();
82  fBoundaryProcess = new G4OpBoundaryProcess();
83 
84  G4ProcessManager* pManager =
85  G4OpticalPhoton::OpticalPhoton()->GetProcessManager();
86 
87  if (!pManager) {
88  std::ostringstream o;
89  o << "Optical Photon without a Process Manager";
90  G4Exception("WLSOpticalPhysics::ConstructProcess()","",
91  FatalException,o.str().c_str());
92  }
93 
94  if (fAbsorptionOn) pManager->AddDiscreteProcess(fAbsorptionProcess);
95 
96  //pManager->AddDiscreteProcess(fRayleighScattering);
97  //pManager->AddDiscreteProcess(fMieHGScatteringProcess);
98 
99  pManager->AddDiscreteProcess(fBoundaryProcess);
100 
101  fWLSProcess->UseTimeProfile("delta");
102  //fWLSProcess->UseTimeProfile("exponential");
103 
104  pManager->AddDiscreteProcess(fWLSProcess);
105 
106  fScintProcess->SetScintillationYieldFactor(1.);
107  fScintProcess->SetScintillationExcitationRatio(0.0);
108  fScintProcess->SetTrackSecondariesFirst(true);
109 
110  // Use Birks Correction in the Scintillation process
111 
112  G4EmSaturation* emSaturation = G4LossTableManager::Instance()->EmSaturation();
113  fScintProcess->AddSaturation(emSaturation);
114 
115  auto particleIterator=GetParticleIterator();
116  particleIterator->reset();
117  while ( (*particleIterator)() ){
118 
119  G4ParticleDefinition* particle = particleIterator->value();
120  G4String particleName = particle->GetParticleName();
121 
122  pManager = particle->GetProcessManager();
123  if (!pManager) {
124  std::ostringstream o;
125  o << "Particle " << particleName << "without a Process Manager";
126  G4Exception("WLSOpticalPhysics::ConstructProcess()","",
127  FatalException,o.str().c_str());
128  }
129 
130  if(fCerenkovProcess->IsApplicable(*particle)){
131  pManager->AddProcess(fCerenkovProcess);
132  pManager->SetProcessOrdering(fCerenkovProcess,idxPostStep);
133  }
134  if(fScintProcess->IsApplicable(*particle)){
135  pManager->AddProcess(fScintProcess);
136  pManager->SetProcessOrderingToLast(fScintProcess,idxAtRest);
137  pManager->SetProcessOrderingToLast(fScintProcess,idxPostStep);
138  }
139 
140  }
141 }
