145 void G4StatMFChannel::CoulombImpulse(G4int anA, G4int anZ, G4double T)
146 // Aafter breakup, fragments fly away under Coulomb field.
147 // This method calculates asymptotic fragments momenta.
148 {
149  // First, we have to place the fragments inside of the original nucleus volume
150  PlaceFragments(anA);
151 
152  // Second, we sample initial charged fragments momenta. There are
153  // _NumOfChargedFragments charged fragments and they start at the begining
154  // of the vector _theFragments (i.e. 0)
155  FragmentsMomenta(_NumOfChargedFragments, 0, T);
156 
157  // Third, we have to figure out the asymptotic momenta of charged fragments
158  // For taht we have to solve equations of motion for fragments
159  SolveEqOfMotion(anA,anZ,T);
160 
161  return;
162 }

164 void G4StatMFChannel::PlaceFragments(G4int anA)
165 // This gives the position of fragments at the breakup instant.
166 // Fragments positions are sampled inside prolongated ellipsoid.
167 {
168  G4Pow* g4calc = G4Pow::GetInstance();
169  const G4double R0 = G4StatMFParameters::Getr0();
170  G4double Rsys = 2.0*R0*g4calc->Z13(anA);
171 
172  G4bool TooMuchIterations;
173  do
174  {
175  TooMuchIterations = false;
176 
177  // Sample the position of the first fragment
178  G4double R = (Rsys - R0*g4calc->Z13(_theFragments[0]->GetA()))*
179  g4calc->A13(G4UniformRand());
180  _theFragments[0]->SetPosition(R*G4RandomDirection());
181 
182 
183  // Sample the position of the remaining fragments
184  G4bool ThereAreOverlaps = false;
185  std::deque<G4StatMFFragment*>::iterator i;
186  for (i = _theFragments.begin()+1; i != _theFragments.end(); ++i)
187  {
188  G4int counter = 0;
189  do
190  {
191  R = (Rsys - R0*g4calc->Z13((*i)->GetA()))*g4calc->A13(G4UniformRand());
192  (*i)->SetPosition(R*G4RandomDirection());
193 
194  // Check that there are not overlapping fragments
195  std::deque<G4StatMFFragment*>::iterator j;
196  for (j = _theFragments.begin(); j != i; ++j)
197  {
198  G4ThreeVector FragToFragVector =
199  (*i)->GetPosition() - (*j)->GetPosition();
200  G4double Rmin = R0*(g4calc->Z13((*i)->GetA()) +
201  g4calc->Z13((*j)->GetA()));
202  if ( (ThereAreOverlaps = (FragToFragVector.mag2() < Rmin*Rmin)))
203  { break; }
204  }
205  counter++;
206  // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
207  } while (ThereAreOverlaps && counter < 1000);
208 
209  if (counter >= 1000)
210  {
211  TooMuchIterations = true;
212  break;
213  }
214  }
215  // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
216  } while (TooMuchIterations);
217  return;
218 }

220 void G4StatMFChannel::FragmentsMomenta(G4int NF, G4int idx,
221  G4double T)
222 // Calculate fragments momenta at the breakup instant
223 // Fragment kinetic energies are calculated according to the
224 // Boltzmann distribution at given temperature.
225 // NF is number of fragments
226 // idx is index of first fragment
227 {
228  G4double KinE = 1.5*T*NF;
229  G4ThreeVector p(0.,0.,0.);
230 
231  if (NF <= 0) return;
232  else if (NF == 1)
233  {
234  // We have only one fragment to deal with
235  p = std::sqrt(2.0*_theFragments[idx]->GetNuclearMass()*KinE)
236  *G4RandomDirection();
237  _theFragments[idx]->SetMomentum(p);
238  }
239  else if (NF == 2)
240  {
241  // We have only two fragment to deal with
242  G4double M1 = _theFragments[idx]->GetNuclearMass();
243  G4double M2 = _theFragments[idx+1]->GetNuclearMass();
244  p = std::sqrt(2.0*KinE*(M1*M2)/(M1+M2))*G4RandomDirection();
245  _theFragments[idx]->SetMomentum(p);
246  _theFragments[idx+1]->SetMomentum(-p);
247  }
248  else
249  {
250  // We have more than two fragments
251  G4double AvailableE;
252  G4int i1,i2;
253  G4double SummedE;
254  G4ThreeVector SummedP(0.,0.,0.);
255  do
256  {
257  // Fisrt sample momenta of NF-2 fragments
258  // according to Boltzmann distribution
259  AvailableE = 0.0;
260  SummedE = 0.0;
261  SummedP.setX(0.0);SummedP.setY(0.0);SummedP.setZ(0.0);
262  for (G4int i = idx; i < idx+NF-2; ++i)
263  {
264  G4double E;
265  G4double RandE;
266  do
267  {
268  E = 9.0*G4UniformRand();
269  RandE = std::sqrt(0.5/E)*G4Exp(E-0.5)*G4UniformRand();
270  }
271  // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
272  while (RandE > 1.0);
273  E *= T;
274  p = std::sqrt(2.0*E*_theFragments[i]->GetNuclearMass())
275  *G4RandomDirection();
276  _theFragments[i]->SetMomentum(p);
277  SummedE += E;
278  SummedP += p;
279  }
280  // Calculate momenta of last two fragments in such a way
281  // that constraints are satisfied
282  i1 = idx+NF-2; // before last fragment index
283  i2 = idx+NF-1; // last fragment index
284  p = -SummedP;
285  AvailableE = KinE - SummedE;
286  // Available Kinetic Energy should be shared between two last fragments
287  }
288  // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
289  while (AvailableE <= p.mag2()/(2.0*(_theFragments[i1]->GetNuclearMass()+
290  _theFragments[i2]->GetNuclearMass())));
291  G4double H = 1.0 + _theFragments[i2]->GetNuclearMass()
292  /_theFragments[i1]->GetNuclearMass();
293  G4double CTM12 = H*(1.0 - 2.0*_theFragments[i2]->GetNuclearMass()
294  *AvailableE/p.mag2());
295  G4double CosTheta1;
296  G4double Sign;
297 
298  if (CTM12 > 1.) {CosTheta1 = 1.;}
299  else {
300  do
301  {
302  do
303  {
304  CosTheta1 = 1.0 - 2.0*G4UniformRand();
305  }
306  // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
307  while (CosTheta1*CosTheta1 < CTM12);
308  }
309  // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
310  while (CTM12 >= 0.0 && CosTheta1 < 0.0);
311  }
312 
313  if (CTM12 < 0.0) Sign = 1.0;
314  else if (G4UniformRand() <= 0.5) Sign = -1.0;
315  else Sign = 1.0;
316 
317  G4double P1 = (p.mag()*CosTheta1+Sign*std::sqrt(p.mag2()
318  *(CosTheta1*CosTheta1-CTM12)))/H;
319  G4double P2 = std::sqrt(P1*P1+p.mag2() - 2.0*P1*p.mag()*CosTheta1);
320  G4double Phi = twopi*G4UniformRand();
321  G4double SinTheta1 = std::sqrt(1.0 - CosTheta1*CosTheta1);
322  G4double CosPhi1 = std::cos(Phi);
323  G4double SinPhi1 = std::sin(Phi);
324  G4double CosPhi2 = -CosPhi1;
325  G4double SinPhi2 = -SinPhi1;
326  G4double CosTheta2 = (p.mag2() + P2*P2 - P1*P1)/(2.0*p.mag()*P2);
327  G4double SinTheta2 = 0.0;
328  if (CosTheta2 > -1.0 && CosTheta2 < 1.0) {
329  SinTheta2 = std::sqrt(1.0 - CosTheta2*CosTheta2);
330  }
331  G4ThreeVector p1(P1*SinTheta1*CosPhi1,P1*SinTheta1*SinPhi1,P1*CosTheta1);
332  G4ThreeVector p2(P2*SinTheta2*CosPhi2,P2*SinTheta2*SinPhi2,P2*CosTheta2);
333  G4ThreeVector b(1.0,0.0,0.0);
334 
335  p1 = RotateMomentum(p,b,p1);
336  p2 = RotateMomentum(p,b,p2);
337 
338  SummedP += p1 + p2;
339  SummedE += p1.mag2()/(2.0*_theFragments[i1]->GetNuclearMass()) +
340  p2.mag2()/(2.0*_theFragments[i2]->GetNuclearMass());
341 
342  _theFragments[i1]->SetMomentum(p1);
343  _theFragments[i2]->SetMomentum(p2);
344 
345  }
346  return;
347 }

349 void G4StatMFChannel::SolveEqOfMotion(G4int anA, G4int anZ, G4double T)
350 // This method will find a solution of Newton's equation of motion
351 // for fragments in the self-consistent time-dependent Coulomb field
352 {
353  G4Pow* g4calc = G4Pow::GetInstance();
354  G4double CoulombEnergy = 0.6*elm_coupling*anZ*anZ*
355  g4calc->A13(1.0+G4StatMFParameters::GetKappaCoulomb())/
356  (G4StatMFParameters::Getr0()*g4calc->Z13(anA)) - GetFragmentsCoulombEnergy();
357  if (CoulombEnergy <= 0.0) return;
358 
359  G4int Iterations = 0;
360  G4double TimeN = 0.0;
361  G4double TimeS = 0.0;
362  G4double DeltaTime = 10.0;
363 
364  G4ThreeVector * Pos = new G4ThreeVector[_NumOfChargedFragments];
365  G4ThreeVector * Vel = new G4ThreeVector[_NumOfChargedFragments];
366  G4ThreeVector * Accel = new G4ThreeVector[_NumOfChargedFragments];
367 
368  G4int i;
369  for (i = 0; i < _NumOfChargedFragments; i++)
370  {
371  Vel[i] = (1.0/(_theFragments[i]->GetNuclearMass()))*
372  _theFragments[i]->GetMomentum();
373  Pos[i] = _theFragments[i]->GetPosition();
374  }
375 
376  G4ThreeVector distance(0.,0.,0.);
377  G4ThreeVector force(0.,0.,0.);
378  G4ThreeVector SavedVel(0.,0.,0.);
379  do {
380  for (i = 0; i < _NumOfChargedFragments; i++)
381  {
382  force.set(0.,0.,0.);
383  for (G4int j = 0; j < _NumOfChargedFragments; j++)
384  {
385  if (i != j)
386  {
387  distance = Pos[i] - Pos[j];
388  force += (elm_coupling*_theFragments[i]->GetZ()
389  *_theFragments[j]->GetZ()/
390  (distance.mag2()*distance.mag()))*distance;
391  }
392  }
393  Accel[i] = (1./(_theFragments[i]->GetNuclearMass()))*force;
394  }
395 
396  TimeN = TimeS + DeltaTime;
397 
398  for ( i = 0; i < _NumOfChargedFragments; i++)
399  {
400  SavedVel = Vel[i];
401  Vel[i] += Accel[i]*(TimeN-TimeS);
402  Pos[i] += (SavedVel+Vel[i])*(TimeN-TimeS)*0.5;
403  }
404  TimeS = TimeN;
405 
406  // Loop checking, 05-Aug-2015, Vladimir Ivanchenko
407  } while (Iterations++ < 100);
408 
409  // Summed fragment kinetic energy
410  G4double TotalKineticEnergy = 0.0;
411  for (i = 0; i < _NumOfChargedFragments; i++)
412  {
413  TotalKineticEnergy += _theFragments[i]->GetNuclearMass()*
414  0.5*Vel[i].mag2();
415  }
416  // Scaling of fragment velocities
417  G4double KineticEnergy = 1.5*_theFragments.size()*T;
418  G4double Eta = ( CoulombEnergy + KineticEnergy ) / TotalKineticEnergy;
419  for (i = 0; i < _NumOfChargedFragments; i++)
420  {
421  Vel[i] *= Eta;
422  }
423 
424  // Finally calculate fragments momenta
425  for (i = 0; i < _NumOfChargedFragments; i++)
426  {
427  _theFragments[i]->SetMomentum(_theFragments[i]->GetNuclearMass()*Vel[i]);
428  }
429 
430  // garbage collection
431  delete [] Pos;
432  delete [] Vel;
433  delete [] Accel;
434 
435  return;
436 }

438 G4ThreeVector G4StatMFChannel::RotateMomentum(G4ThreeVector Pa,
439  G4ThreeVector V, G4ThreeVector P)
440  // Rotates a 3-vector P to close momentum triangle Pa + V + P = 0
441 {
442  G4ThreeVector U = Pa.unit();
443 
444  G4double Alpha1 = U * V;
445 
446  G4double Alpha2 = std::sqrt(V.mag2() - Alpha1*Alpha1);
447 
448  G4ThreeVector N = (1./Alpha2)*U.cross(V);
449 
450  G4ThreeVector RotatedMomentum(
451  ( (V.x() - Alpha1*U.x())/Alpha2 ) * P.x() + N.x() * P.y() + U.x() * P.z(),
452  ( (V.y() - Alpha1*U.y())/Alpha2 ) * P.x() + N.y() * P.y() + U.y() * P.z(),
453  ( (V.z() - Alpha1*U.z())/Alpha2 ) * P.x() + N.z() * P.y() + U.z() * P.z()
454  );
455  return RotatedMomentum;
456 }
