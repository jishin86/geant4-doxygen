44 //----------------------------------------------------------------------------
45 G4ErrorTrackLengthTarget::
46 G4ErrorTrackLengthTarget(const G4double maxTrkLength )
47  : G4VDiscreteProcess ("G4ErrorTrackLengthTarget"),
48  theMaximumTrackLength( maxTrkLength )
49 {
50  theType = G4ErrorTarget_TrkL;
51 
52  G4ParticleTable::G4PTblDicIterator* theParticleIterator
53  = G4ParticleTable::GetParticleTable()->GetIterator();
54 
55  // loop over all particles in G4ParticleTable
56 
57  theParticleIterator->reset();
58  while( (*theParticleIterator)() ) // Loop checking, 06.08.2015, G.Cosmo
59  {
60  G4ParticleDefinition* particle = theParticleIterator->value();
61  G4ProcessManager* pmanager = particle->GetProcessManager();
62  if (!particle->IsShortLived())
63  {
64  // Add transportation process for all particles other than "shortlived"
65  if ( pmanager == 0)
66  {
67  // Error !! no process manager
68  G4String particleName = particle->GetParticleName();
69  G4Exception("G4ErrorTrackLengthTarget::G4ErrorTrackLengthTarget",
70  "No process manager", RunMustBeAborted, particleName );
71  }
72  else
73  {
74  G4ProcessVector* procvec = pmanager->GetProcessList();
75  size_t isiz = procvec->size();
76 
77  for( size_t ii=0; ii < isiz; ii++ )
78  {
79  if( ((*procvec)[ii])->GetProcessName() == "G4ErrorTrackLengthTarget")
80  {
81  pmanager->RemoveProcess( (*procvec)[ii] );
82  }
83  }
84  pmanager ->AddDiscreteProcess(this,4);
85  isiz = procvec->size();
86  }
87  }
88  else
89  {
90  // shortlived particle case
91  }
92  }
93 }
