80 void B5PrimaryGeneratorAction::GeneratePrimaries(G4Event* event)
81 {
82  G4ParticleDefinition* particle;
83  if (fRandomizePrimary) {
84  G4int i = (int)(5.*G4UniformRand());
85  switch(i) {
86  case 0:
87  particle = fPositron;
88  break;
89  case 1:
90  particle = fMuon;
91  break;
92  case 2:
93  particle = fPion;
94  break;
95  case 3:
96  particle = fKaon;
97  break;
98  default:
99  particle = fProton;
100  break;
101  }
102  fParticleGun->SetParticleDefinition(particle);
103  }
104  else {
105  particle = fParticleGun->GetParticleDefinition();
106  }
107 
108  auto pp = fMomentum + (G4UniformRand()-0.5)*fSigmaMomentum;
109  auto mass = particle->GetPDGMass();
110  auto ekin = std::sqrt(pp*pp+mass*mass)-mass;
111  fParticleGun->SetParticleEnergy(ekin);
112 
113  auto angle = (G4UniformRand()-0.5)*fSigmaAngle;
114  fParticleGun->SetParticleMomentumDirection(
115  G4ThreeVector(std::sin(angle),0.,std::cos(angle)));
116 
117  fParticleGun->GeneratePrimaryVertex(event);
118 }
