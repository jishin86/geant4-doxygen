114 G4double
115 G4BGGPionInelasticXS::GetElementCrossSection(const G4DynamicParticle* dp,
116  G4int ZZ, const G4Material*)
117 {
118  // this method should be called only for Z > 1
119 
120  G4double cross = 0.0;
121  G4double ekin = dp->GetKineticEnergy();
122  G4int Z = std::min(ZZ, 92);
123  if(1 == Z) {
124  cross = 1.0115*GetIsoCrossSection(dp,1,1);
125  } else {
126  if(ekin <= fLowEnergy && !isPiplus) {
127  cross = theCoulombFac[Z];
128  } else if(ekin <= 2*MeV && isPiplus) {
129  cross = theCoulombFac[Z]*CoulombFactor(ekin, Z);
130  } else if(ekin > fGlauberEnergy) {
131  cross = theGlauberFac[Z]*fGlauber->GetInelasticGlauberGribov(dp, Z, theA[Z]);
132  } else {
133  cross = fPion->GetInelasticCrossSection(dp, Z, theA[Z]);
134  }
135  }
136  if(verboseLevel > 1) {
137  G4cout << "G4BGGPionInelasticXS::GetCrossSection for "
138  << dp->GetDefinition()->GetParticleName()
139  << " Ekin(GeV)= " << dp->GetKineticEnergy()
140  << " in nucleus Z= " << Z << " A= " << theA[Z]
141  << " XS(b)= " << cross/barn
142  << G4endl;
143  }
144  return cross;
145 }

149 G4double
150 G4BGGPionInelasticXS::GetIsoCrossSection(const G4DynamicParticle* dp,
151  G4int Z, G4int A,
152  const G4Isotope*,
153  const G4Element*,
154  const G4Material*)
155 {
156  // this method should be called only for Z = 1
157 
158  G4double cross = 0.0;
159  if(1 == Z) {
160  fHadron->GetHadronNucleonXscNS(dp, theProton);
161  cross = fHadron->GetInelasticHadronNucleonXsc();
162  }
163  cross *= A;
164 
165  if(verboseLevel > 1) {
166  G4cout << "G4BGGPionInelasticXS::GetCrossSection for "
167  << dp->GetDefinition()->GetParticleName()
168  << " Ekin(GeV)= " << dp->GetKineticEnergy()
169  << " in nucleus Z= " << Z << " A= " << A
170  << " XS(b)= " << cross/barn
171  << G4endl;
172  }
173  return cross;
174 }

178 void G4BGGPionInelasticXS::BuildPhysicsTable(const G4ParticleDefinition& p)
179 {
180  if(&p == G4PionPlus::PionPlus() || &p == G4PionMinus::PionMinus()) {
181  particle = &p;
182  } else {
183  G4cout << "### G4BGGPionInelasticXS WARNING: is not applicable to "
184  << p.GetParticleName()
185  << G4endl;
186  throw G4HadronicException(__FILE__, __LINE__,
187  "G4BGGPionInelasticXS::BuildPhysicsTable is used for wrong particle");
188  return;
189  }
190 
191  if(isInitialized) { return; }
192  isInitialized = true;
193 
194  fPion = new G4UPiNuclearCrossSection();
195  fGlauber = new G4ComponentGGHadronNucleusXsc();
196  fHadron = new G4HadronNucleonXsc();
197 
198  fPion->BuildPhysicsTable(*particle);
199  fGlauber->BuildPhysicsTable(*particle);
200 
201  G4ThreeVector mom(0.0,0.0,1.0);
202  G4DynamicParticle dp(particle, mom, fGlauberEnergy);
203 
204  G4NistManager* nist = G4NistManager::Instance();
205 
206  G4double csup, csdn;
207 
208  if(verboseLevel > 0) {
209  G4cout << "### G4BGGPionInelasticXS::Initialise for "
210  << particle->GetParticleName()
211  << " isPiplus: " << isPiplus
212  << G4endl;
213  }
214 
215  for(G4int iz=2; iz<93; ++iz) {
216  G4int A = G4lrint(nist->GetAtomicMassAmu(iz));
217  theA[iz] = A;
218 
219  csup = fGlauber->GetInelasticGlauberGribov(&dp, iz, A);
220  csdn = fPion->GetInelasticCrossSection(&dp, iz, A);
221 
222  theGlauberFac[iz] = csdn/csup;
223  if(verboseLevel > 0) {
224  G4cout << "Z= " << iz << " A= " << A
225  << " factor= " << theGlauberFac[iz] << G4endl;
226  }
227  }
228  theCoulombFac[1] = 1.0;
229  G4double elim = (isPiplus) ? 2*MeV : fLowEnergy;
230  dp.SetKineticEnergy(elim);
231 
232  for(G4int iz=2; iz<93; ++iz) {
233  if(isPiplus) {
234  theCoulombFac[iz] = fPion->GetInelasticCrossSection(&dp, iz, theA[iz])
235  /CoulombFactor(elim, iz);
236  } else {
237  theCoulombFac[iz] = fPion->GetInelasticCrossSection(&dp, iz, theA[iz]);
238  }
239  if(verboseLevel > 0) {
240  G4cout << "Z= " << iz << " A= " << theA[iz]
241  << " CoulombFactor= " << theCoulombFac[iz] << G4endl;
242  }
243  }
244 }

248 G4double G4BGGPionInelasticXS::CoulombFactor(G4double kinEnergy, G4int Z)
249 {
250  G4int A = theA[Z];
251  G4double res= 0.0;
252  if(kinEnergy <= DBL_MIN) { return res; }
253  else if(A < 2) { return kinEnergy*kinEnergy; }
254 
255  G4double elog = fG4pow->log10A(6.7*kinEnergy/GeV);
256  G4double aa = A;
257 
258  // from G4ProtonInelasticCrossSection
259  G4double ff1 = 0.70 - 0.002*aa; // slope of the drop at medium energies.
260  G4double ff2 = 1.00 + 1/aa; // start of the slope.
261  G4double ff3 = 0.8 + 18/aa - 0.002*aa; // stephight
262  res = 1.0 + ff3*(1.0 - (1.0/(1+fG4pow->expA(-8*ff1*(elog + 1.37*ff2)))));
263 
264  ff1 = 1. - 1./aa - 0.001*aa; // slope of the rise
265  ff2 = 1.17 - 2.7/aa-0.0014*aa; // start of the rise
266  res /= (1 + fG4pow->expA(-8.*ff1*(elog + 2*ff2)));
267  return res;
268 }
