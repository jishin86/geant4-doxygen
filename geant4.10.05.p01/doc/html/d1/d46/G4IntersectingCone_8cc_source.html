45 G4IntersectingCone::G4IntersectingCone( const G4double r[2],
46  const G4double z[2] )
47 {
48  const G4double halfCarTolerance
49  = 0.5 * G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
50 
51  // What type of cone are we?
52  //
53  type1 = (std::abs(z[1]-z[0]) > std::abs(r[1]-r[0]));
54 
55  if (type1) // tube like
56  {
57  B = (r[1] - r[0]) / (z[1] - z[0]);
58  A = (r[0]*z[1] - r[1]*z[0]) / (z[1] -z[0]);
59  }
60  else // disk like
61  {
62  B = (z[1] - z[0]) / (r[1] - r[0]);
63  A = (z[0]*r[1] - z[1]*r[0]) / (r[1] - r[0]);
64  }
65 
66  // Calculate extent
67  //
68  rLo = std::min(r[0], r[1]) - halfCarTolerance;
69  rHi = std::max(r[0], r[1]) + halfCarTolerance;
70  zLo = std::min(z[0], z[1]) - halfCarTolerance;
71  zHi = std::max(z[0], z[1]) + halfCarTolerance;
72 }

99 G4bool G4IntersectingCone::HitOn( const G4double r,
100  const G4double z )
101 {
102  //
103  // Be careful! The inequalities cannot be "<=" and ">=" here without
104  // punching a tiny hole in our shape!
105  //
106  if (type1)
107  {
108  if (z < zLo || z > zHi) return false;
109  }
110  else
111  {
112  if (r < rLo || r > rHi) return false;
113  }
114 
115  return true;
116 }

140 //
141 // LineHitsCone1
142 //
143 // Calculate the intersections of a line with a conical surface. Only
144 // suitable if zPlane[0] != zPlane[1].
145 //
146 // Equation of a line:
147 //
148 // x = x0 + s*tx y = y0 + s*ty z = z0 + s*tz
149 //
150 // Equation of a conical surface:
151 //
152 // x**2 + y**2 = (A + B*z)**2
153 //
154 // Solution is quadratic:
155 //
156 // a*s**2 + b*s + c = 0
157 //
158 // where:
159 //
160 // a = tx**2 + ty**2 - (B*tz)**2
161 //
162 // b = 2*( px*vx + py*vy - B*(A + B*pz)*vz )
163 //
164 // c = x0**2 + y0**2 - (A + B*z0)**2
165 //
166 // Notice, that if a < 0, this indicates that the two solutions (assuming
167 // they exist) are in opposite cones (that is, given z0 = -A/B, one z < z0
168 // and the other z > z0). For our shapes, the invalid solution is one
169 // which produces A + Bz < 0, or the one where Bz is smallest (most negative).
170 // Since Bz = B*s*tz, if B*tz > 0, we want the largest s, otherwise,
171 // the smaller.
172 //
173 // If there are two solutions on one side of the cone, we want to make
174 // sure that they are on the "correct" side, that is A + B*z0 + s*B*tz >= 0.
175 //
176 // If a = 0, we have a linear problem: s = c/b, which again gives one solution.
177 // This should be rare.
178 //
179 // For b*b - 4*a*c = 0, we also have one solution, which is almost always
180 // a line just grazing the surface of a the cone, which we want to ignore.
181 // However, there are two other, very rare, possibilities:
182 // a line intersecting the z axis and either:
183 // 1. At the same angle std::atan(B) to just miss one side of the cone, or
184 // 2. Intersecting the cone apex (0,0,-A/B)
185 // We *don't* want to miss these! How do we identify them? Well, since
186 // this case is rare, we can at least swallow a little more CPU than we would
187 // normally be comfortable with. Intersection with the z axis means
188 // x0*ty - y0*tx = 0. Case (1) means a==0, and we've already dealt with that
189 // above. Case (2) means a < 0.
190 //
191 // Now: x0*tx + y0*ty = 0 in terms of roundoff error. We can write:
192 // Delta = x0*tx + y0*ty
193 // b = 2*( Delta - B*(A + B*z0)*tz )
194 // For:
195 // b*b - 4*a*c = epsilon
196 // where epsilon is small, then:
197 // Delta = epsilon/2/B
198 //
199 G4int G4IntersectingCone::LineHitsCone1( const G4ThreeVector &p,
200  const G4ThreeVector &v,
201  G4double *s1, G4double *s2 )
202 {
203  static const G4double EPS = DBL_EPSILON; // Precision constant,
204  // originally it was 1E-6
205  G4double x0 = p.x(), y0 = p.y(), z0 = p.z();
206  G4double tx = v.x(), ty = v.y(), tz = v.z();
207 
208  // Value of radical can be inaccurate due to loss of precision
209  // if to calculate the coefficiets a,b,c like the following:
210  // G4double a = tx*tx + ty*ty - sqr(B*tz);
211  // G4double b = 2*( x0*tx + y0*ty - B*(A + B*z0)*tz);
212  // G4double c = x0*x0 + y0*y0 - sqr(A + B*z0);
213  //
214  // For more accurate calculation of radical the coefficients
215  // are splitted in two components, radial and along z-axis
216  //
217  G4double ar = tx*tx + ty*ty;
218  G4double az = sqr(B*tz);
219  G4double br = 2*(x0*tx + y0*ty);
220  G4double bz = 2*B*(A + B*z0)*tz;
221  G4double cr = x0*x0 + y0*y0;
222  G4double cz = sqr(A + B*z0);
223 
224  // Instead radical = b*b - 4*a*c
225  G4double arcz = 4*ar*cz;
226  G4double azcr = 4*az*cr;
227  G4double radical = (br*br - 4*ar*cr) + ((std::max(arcz,azcr) - 2*bz*br) + std::min(arcz,azcr));
228 
229  // Find the coefficients
230  G4double a = ar - az;
231  G4double b = br - bz;
232  G4double c = cr - cz;
233 
234  if (radical < -EPS*std::fabs(b)) { return 0; } // No solution
235 
236  if (radical < EPS*std::fabs(b))
237  {
238  //
239  // The radical is roughly zero: check for special, very rare, cases
240  //
241  if (std::fabs(a) > 1/kInfinity)
242  {
243  if(B==0.) { return 0; }
244  if ( std::fabs(x0*ty - y0*tx) < std::fabs(EPS/B) )
245  {
246  *s1 = -0.5*b/a;
247  return 1;
248  }
249  return 0;
250  }
251  }
252  else
253  {
254  radical = std::sqrt(radical);
255  }
256 
257  if (a > 1/kInfinity)
258  {
259  G4double sa, sb, q = -0.5*( b + (b < 0 ? -radical : +radical) );
260  sa = q/a;
261  sb = c/q;
262  if (sa < sb) { *s1 = sa; *s2 = sb; } else { *s1 = sb; *s2 = sa; }
263  if (A + B*(z0+(*s1)*tz) < 0) { return 0; }
264  return 2;
265  }
266  else if (a < -1/kInfinity)
267  {
268  G4double sa, sb, q = -0.5*( b + (b < 0 ? -radical : +radical) );
269  sa = q/a;
270  sb = c/q;
271  *s1 = (B*tz > 0)^(sa > sb) ? sb : sa;
272  return 1;
273  }
274  else if (std::fabs(b) < 1/kInfinity)
275  {
276  return 0;
277  }
278  else
279  {
280  *s1 = -c/b;
281  if (A + B*(z0+(*s1)*tz) < 0) { return 0; }
282  return 1;
283  }
284 }

287 //
288 // LineHitsCone2
289 //
290 // See comments under LineHitsCone1. In this routine, case2, we have:
291 //
292 // Z = A + B*R
293 //
294 // The solution is still quadratic:
295 //
296 // a = tz**2 - B*B*(tx**2 + ty**2)
297 //
298 // b = 2*( (z0-A)*tz - B*B*(x0*tx+y0*ty) )
299 //
300 // c = ( (z0-A)**2 - B*B*(x0**2 + y0**2) )
301 //
302 // The rest is much the same, except some details.
303 //
304 // a > 0 now means we intersect only once in the correct hemisphere.
305 //
306 // a > 0 ? We only want solution which produces R > 0.
307 // since R = (z0+s*tz-A)/B, for tz/B > 0, this is the largest s
308 // for tz/B < 0, this is the smallest s
309 // thus, same as in case 1 ( since sign(tz/B) = sign(tz*B) )
310 //
311 G4int G4IntersectingCone::LineHitsCone2( const G4ThreeVector &p,
312  const G4ThreeVector &v,
313  G4double *s1, G4double *s2 )
314 {
315  static const G4double EPS = DBL_EPSILON; // Precision constant,
316  // originally it was 1E-6
317  G4double x0 = p.x(), y0 = p.y(), z0 = p.z();
318  G4double tx = v.x(), ty = v.y(), tz = v.z();
319 
320  // Special case which might not be so rare: B = 0 (precisely)
321  //
322  if (B==0)
323  {
324  if (std::fabs(tz) < 1/kInfinity) { return 0; }
325 
326  *s1 = (A-z0)/tz;
327  return 1;
328  }
329 
330  // Value of radical can be inaccurate due to loss of precision
331  // if to calculate the coefficiets a,b,c like the following:
332  // G4double a = tz*tz - B2*(tx*tx + ty*ty);
333  // G4double b = 2*( (z0-A)*tz - B2*(x0*tx + y0*ty) );
334  // G4double c = sqr(z0-A) - B2*( x0*x0 + y0*y0 );
335  //
336  // For more accurate calculation of radical the coefficients
337  // are splitted in two components, radial and along z-axis
338  //
339  G4double B2 = B*B;
340 
341  G4double az = tz*tz;
342  G4double ar = B2*(tx*tx + ty*ty);
343  G4double bz = 2*(z0-A)*tz;
344  G4double br = 2*B2*(x0*tx + y0*ty);
345  G4double cz = sqr(z0-A);
346  G4double cr = B2*(x0*x0 + y0*y0);
347 
348  // Instead radical = b*b - 4*a*c
349  G4double arcz = 4*ar*cz;
350  G4double azcr = 4*az*cr;
351  G4double radical = (br*br - 4*ar*cr) + ((std::max(arcz,azcr) - 2*bz*br) + std::min(arcz,azcr));
352 
353  // Find the coefficients
354  G4double a = az - ar;
355  G4double b = bz - br;
356  G4double c = cz - cr;
357 
358  if (radical < -EPS*std::fabs(b)) { return 0; } // No solution
359 
360  if (radical < EPS*std::fabs(b))
361  {
362  //
363  // The radical is roughly zero: check for special, very rare, cases
364  //
365  if (std::fabs(a) > 1/kInfinity)
366  {
367  if ( std::fabs(x0*ty - y0*tx) < std::fabs(EPS/B) )
368  {
369  *s1 = -0.5*b/a;
370  return 1;
371  }
372  return 0;
373  }
374  }
375  else
376  {
377  radical = std::sqrt(radical);
378  }
379 
380  if (a < -1/kInfinity)
381  {
382  G4double sa, sb, q = -0.5*( b + (b < 0 ? -radical : +radical) );
383  sa = q/a;
384  sb = c/q;
385  if (sa < sb) { *s1 = sa; *s2 = sb; } else { *s1 = sb; *s2 = sa; }
386  if ((z0 + (*s1)*tz - A)/B < 0) { return 0; }
387  return 2;
388  }
389  else if (a > 1/kInfinity)
390  {
391  G4double sa, sb, q = -0.5*( b + (b < 0 ? -radical : +radical) );
392  sa = q/a;
393  sb = c/q;
394  *s1 = (tz*B > 0)^(sa > sb) ? sb : sa;
395  return 1;
396  }
397  else if (std::fabs(b) < 1/kInfinity)
398  {
399  return 0;
400  }
401  else
402  {
403  *s1 = -c/b;
404  if ((z0 + (*s1)*tz - A)/B < 0) { return 0; }
405  return 1;
406  }
407 }
