230 //
231 // Calculates total and inelastic Xsc, derives elastic as total
232 // inelastic accordong to Glauber model with Gribov correction calculated
233 // in the dipole approximation on light cone. Gaussian density of point-like
234 // nucleons helps to calculate rest integrals of the model.
235 // [1] B.Z. Kopeliovich, nucl-th/0306044 + simplification above
236 
237 void G4ComponentGGHadronNucleusXsc::ComputeCrossSections(
238  const G4ParticleDefinition* aParticle,
239  G4double kinEnergy, G4int Z, G4int A)
240 {
241  // check cache
242  if(aParticle == fParticle && fZ == Z && fA == A && kinEnergy == fEnergy)
243  { return; }
244  fParticle = aParticle;
245  fZ = Z;
246  fA = A;
247  fEnergy = kinEnergy;
248 
249  //
250  G4double sigma(0.0), cofInelastic(2.2), cofTotal(2.0);
251  G4double hpInXsc(0.), hnInXsc(0.);
252  G4double R = GetNucleusRadius(A);
253 
254  G4int N = std::max(A - Z, 0); // number of neutrons
255 
256  if( aParticle == theProton ||
257  aParticle == theNeutron ||
258  aParticle == thePiPlus ||
259  aParticle == thePiMinus ||
260  aParticle == theL || aParticle == theAntiL || aParticle == theSPlus || aParticle == theASPlus ||
261  aParticle == theSMinus || aParticle == theASMinus || aParticle == theS0 || aParticle == theAS0 ||
262  aParticle == theXiMinus || aParticle == theXi0 || aParticle == theAXiMinus || aParticle == theAXi0 ||
263  aParticle == theOmega || aParticle == theAOmega
264  )
265  {
266  sigma = Z*hnXsc->HadronNucleonXscNS(aParticle, theProton, kinEnergy);
267  hpInXsc = hnXsc->GetInelasticHadronNucleonXsc();
268 
269  if(N > 0) {
270  sigma += N*hnXsc->HadronNucleonXscNS(aParticle, theNeutron, kinEnergy);
271  hnInXsc = hnXsc->GetInelasticHadronNucleonXsc();
272  }
273  cofInelastic = 2.4;
274 
275  } else if( aParticle == theKPlus ||
276  aParticle == theKMinus ||
277  aParticle == theK0S ||
278  aParticle == theK0L)
279  {
280  sigma = (1 == Z)
281  ? hnXsc->KaonNucleonXscNS(aParticle, theProton, kinEnergy)
282  : Z*hnXsc->KaonNucleonXscGG(aParticle, theProton, kinEnergy);
283  hpInXsc = hnXsc->GetInelasticHadronNucleonXsc();
284 
285  if(N > 0) {
286  sigma += N*hnXsc->KaonNucleonXscGG(aParticle, theNeutron, kinEnergy);
287  hnInXsc = hnXsc->GetInelasticHadronNucleonXsc();
288  }
289  R = 1.3*fermi*g4calc->Z13(A);
290  }
291  else
292  {
293  sigma = Z*hnXsc->HadronNucleonXscNS(aParticle, theProton, kinEnergy);
294  hpInXsc = hnXsc->GetInelasticHadronNucleonXsc();
295 
296  if(N > 0) {
297  sigma += N*hnXsc->HadronNucleonXscNS(aParticle, theNeutron, kinEnergy);
298  hnInXsc = hnXsc->GetInelasticHadronNucleonXsc();
299  }
300  }
301  G4double nucleusSquare = cofTotal*pi*R*R; // basically 2piRR
302  G4double ratio = sigma/nucleusSquare;
303  G4double difratio = ratio/(1.+ratio);
304  fDiffractionXsc = 0.5*nucleusSquare*( difratio - G4Log( 1. + difratio ) );
305 
306  if( A > 1 )
307  {
308  fTotalXsc = nucleusSquare*G4Log( 1. + ratio )
309  *GetParticleBarCorTot(aParticle, Z);
310 
311  // inelastic xsc
312  fAxsc2piR2 = cofInelastic*ratio;
313  fModelInLog = G4Log( 1. + fAxsc2piR2 );
314  fInelasticXsc = nucleusSquare*fModelInLog/cofInelastic;
315  G4double barcorr = GetParticleBarCorIn(aParticle, Z);
316  fInelasticXsc *= barcorr;
317  fElasticXsc = std::max(fTotalXsc - fInelasticXsc, 0.);
318 
319  G4double xratio = (Z*hpInXsc + N*hnInXsc)/nucleusSquare;
320  fProductionXsc =
321  nucleusSquare*G4Log(1. + cofInelastic*xratio)*barcorr/cofInelastic;
322  fProductionXsc = std::min(fProductionXsc, fInelasticXsc);
323  }
324  else // H
325  {
326  fTotalXsc = sigma;
327  fInelasticXsc = hpInXsc;
328  fElasticXsc = std::max(fTotalXsc - fInelasticXsc, 0.);
329  G4double xratio = hpInXsc/nucleusSquare;
330  fProductionXsc = nucleusSquare*G4Log(1. + cofInelastic*xratio)/cofInelastic;
331  fProductionXsc = std::min(fProductionXsc, fInelasticXsc);
332  }
333  /*
334  G4cout << "GGXsc: Z= " << Z << " A= " << A << " E= " << kinEnergy
335  << " xtot(b)= " << fTotalXsc/barn
336  << " xel(b)= " << fElasticXsc/barn << " xinel(b)= " << fInelasticXsc/barn
337  << G4endl;
338  */
339 }

524 G4double G4ComponentGGHadronNucleusXsc::GetNucleusRadius(
525  const G4DynamicParticle*, const G4Element* anElement)
526 {
527  G4int At = G4lrint(anElement->GetN());
528  G4double R = fRadiusConst*g4calc->Z13(At);
529 
530  static const G4double meanA = 21.;
531  static const G4double tauA1 = 40.;
532  static const G4double tauA2 = 10.;
533  static const G4double tauA3 = 5.;
534 
535  static const G4double a1 = 0.85;
536  static const G4double b1 = 1. - a1;
537 
538  static const G4double b2 = 0.3;
539  static const G4double b3 = 4.;
540 
541  if (At > 20)
542  {
543  R *= ( a1 + b1*G4Exp( -(At - meanA)/tauA1) );
544  }
545  else if (At > 3)
546  {
547  R *= ( 1.0 + b2*( 1. - G4Exp( (At - meanA)/tauA2) ) );
548  }
549  else
550  {
551  R *= ( 1.0 + b3*( 1. - G4Exp( (At - meanA)/tauA3) ) );
552  }
553  return R;
