29 // class G4PropagatorInField Implementation
30 //
31 // This class implements an algorithm to track a particle in a
32 // non-uniform magnetic field. It utilises an ODE solver (with
33 // the Runge - Kutta method) to evolve the particle, and drives it
34 // until the particle has traveled a set distance or it enters a new
35 // volume.
36 //
37 // 14.10.96 John Apostolakis, design and implementation
38 // 17.03.97 John Apostolakis, renaming new set functions being added
39 //

103  fVerbTracePiF = true;

111  }

143 G4double
144 G4PropagatorInField::ComputeStep(
145  G4FieldTrack& pFieldTrack,
146  G4double CurrentProposedStepLength,
147  G4double& currentSafety, // IN/OUT
148  G4VPhysicalVolume* pPhysVol)
149 {
150  GetChordFinder()->OnComputeStep();
151 
152  // If CurrentProposedStepLength is too small for finding Chords
153  // then return with no action (for now - TODO: some action)
154  //
155  const char* methodName="G4PropagatorInField::ComputeStep";
156  if(CurrentProposedStepLength<kCarTolerance)
157  {
158  return kInfinity;
159  }
160 
161  // Introducing smooth trajectory display (jacek 01/11/2002)
162  //
163  if (fpTrajectoryFilter)
164  {
165  fpTrajectoryFilter->CreateNewTrajectorySegment();
166  }
167 
168  fFirstStepInVolume = fNewTrack ? true : fLastStepInVolume;
169  fLastStepInVolume= false;
170  fNewTrack= false;
171 
172  if( fVerboseLevel > 2 )
173  {
174  G4cout << methodName << " called" << G4endl;
175  G4cout << " Starting FT: " << pFieldTrack;
176  G4cout << " Requested length = " << CurrentProposedStepLength << G4endl;
177  G4cout << " PhysVol = ";
178  if( pPhysVol )
179  G4cout << pPhysVol->GetName() << G4endl;
180  else
181  G4cout << " N/A ";
182  G4cout << G4endl;
183  }
184 
185  // Parameters for adaptive Runge-Kutta integration
186 
187  G4double h_TrialStepSize; // 1st Step Size
188  G4double TruePathLength = CurrentProposedStepLength;
189  G4double StepTaken = 0.0;
190  G4double s_length_taken, epsilon ;
191  G4bool intersects;
192  G4bool first_substep = true;
193 
194  G4double NewSafety;
195  fParticleIsLooping = false;
196 
197  // If not yet done,
198  // Set the field manager to the local one if the volume has one,
199  // or to the global one if not
200  //
201  if( !fSetFieldMgr ) fCurrentFieldMgr= FindAndSetFieldManager( pPhysVol );
202  // For the next call, the field manager must again be set
203  fSetFieldMgr= false;
204 
205  G4FieldTrack CurrentState(pFieldTrack);
206  G4FieldTrack OriginalState = CurrentState;
207 
208  // If the Step length is "infinite", then an approximate-maximum Step
209  // length (used to calculate the relative accuracy) must be guessed.
210  //
211  if( CurrentProposedStepLength >= fLargestAcceptableStep )
212  {
213  G4ThreeVector StartPointA, VelocityUnit;
214  StartPointA = pFieldTrack.GetPosition();
215  VelocityUnit = pFieldTrack.GetMomentumDir();
216 
217  G4double trialProposedStep = 1.e2 * ( 10.0 * cm +
218  fNavigator->GetWorldVolume()->GetLogicalVolume()->
219  GetSolid()->DistanceToOut(StartPointA, VelocityUnit) );
220  CurrentProposedStepLength= std::min( trialProposedStep,
221  fLargestAcceptableStep );
222  }
223  epsilon = fCurrentFieldMgr->GetDeltaOneStep() / CurrentProposedStepLength;
224  // G4double raw_epsilon= epsilon;
225  G4double epsilonMin= fCurrentFieldMgr->GetMinimumEpsilonStep();
226  G4double epsilonMax= fCurrentFieldMgr->GetMaximumEpsilonStep();
227  if( epsilon < epsilonMin ) epsilon = epsilonMin;
228  if( epsilon > epsilonMax ) epsilon = epsilonMax;
229  SetEpsilonStep( epsilon );
230 
231 
232  // Values for Intersection Locator has to be updated on each call for the
233  // case that CurrentFieldManager has changed from the one of previous step
234  RefreshIntersectionLocator();
235 
236  // G4cout << "G4PiF: Epsilon of current step - raw= " << raw_epsilon
237  // << " final= " << epsilon << G4endl;
238 
239  // Shorten the proposed step in case of earlier problems (zero steps)
240  //
241  if( fNoZeroStep > fActionThreshold_NoZeroSteps )
242  {
243  G4double stepTrial;
244 
245  stepTrial= fFull_CurveLen_of_LastAttempt;
246  if( (stepTrial <= 0.0) && (fLast_ProposedStepLength > 0.0) )
247  {
248  stepTrial= fLast_ProposedStepLength;
249  }
250 
251  G4double decreaseFactor = 0.9; // Unused default
252  if( (fNoZeroStep < fSevereActionThreshold_NoZeroSteps)
253  && (stepTrial > 100.0*fZeroStepThreshold) )
254  {
255  // Attempt quick convergence
256  //
257  decreaseFactor= 0.25;
258  }
259  else
260  {
261  // We are in significant difficulties, probably at a boundary that
262  // is either geometrically sharp or between very different materials.
263  // Careful decreases to cope with tolerance are required.
264  //
265  if( stepTrial > 100.0*fZeroStepThreshold )
266  decreaseFactor = 0.35; // Try decreasing slower
267  else if( stepTrial > 30.0*fZeroStepThreshold )
268  decreaseFactor= 0.5; // Try yet slower decrease
269  else if( stepTrial > 10.0*fZeroStepThreshold )
270  decreaseFactor= 0.75; // Try even slower decreases
271  else
272  decreaseFactor= 0.9; // Try very slow decreases
273  }
274  stepTrial *= decreaseFactor;
275 
276 #ifdef G4DEBUG_FIELD
277  if( fVerboseLevel > 2
278  || (fNoZeroStep >= fSevereActionThreshold_NoZeroSteps)
279  )
280  {
281  G4cerr << " " << methodName
282  << " Decreasing step after " << fNoZeroStep << " zero steps "
283  << " - in volume " << pPhysVol;
284  if( pPhysVol )
285  G4cerr << " with name " << pPhysVol->GetName();
286  else
287  G4cerr << " i.e. *unknown* volume.";
288  G4cerr << G4endl;
289  PrintStepLengthDiagnostic(CurrentProposedStepLength, decreaseFactor,
290  stepTrial, pFieldTrack);
291  }
292 #endif
293  if( stepTrial == 0.0 ) // Change to make it < 0.1 * kCarTolerance ??
294  {
295  std::ostringstream message;
296  message << "Particle abandoned due to lack of progress in field."
297  << G4endl
298  << " Properties : " << pFieldTrack << G4endl
299  << " Attempting a zero step = " << stepTrial << G4endl
300  << " while attempting to progress after " << fNoZeroStep
301  << " trial steps. Will abandon step.";
302  G4Exception(methodName, "GeomNav1002", JustWarning, message);
303  fParticleIsLooping= true;
304  return 0; // = stepTrial;
305  }
306  if( stepTrial < CurrentProposedStepLength )
307  {
308  CurrentProposedStepLength = stepTrial;
309  }
310  }
311  fLast_ProposedStepLength = CurrentProposedStepLength;
312 
313  G4int do_loop_count = 0;
314  do // Loop checking, 07.10.2016, J.Apostolakis
315  {
316  G4FieldTrack SubStepStartState = CurrentState;
317  G4ThreeVector SubStartPoint = CurrentState.GetPosition();
318 
319  if(!first_substep)
320  {
321  if( fVerboseLevel > 4 )
322  {
323  G4cout << " PiF: Calling Nav/Locate Global Point within-Volume "
324  << G4endl;
325  }
326  fNavigator->LocateGlobalPointWithinVolume( SubStartPoint );
327  }
328 
329  // How far to attempt to move the particle !
330  //
331  h_TrialStepSize = CurrentProposedStepLength - StepTaken;
332 
333  // Integrate as far as "chord miss" rule allows.
334  //
335  s_length_taken = GetChordFinder()->AdvanceChordLimited(
336  CurrentState, // Position & velocity
337  h_TrialStepSize,
338  fEpsilonStep,
339  fPreviousSftOrigin,
340  fPreviousSafety
341  );
342  // CurrentState is now updated with the final position and velocity.
343 
344  fFull_CurveLen_of_LastAttempt = s_length_taken;
345 
346  G4ThreeVector EndPointB = CurrentState.GetPosition();
347  G4ThreeVector InterSectionPointE;
348  G4double LinearStepLength;
349 
350  // Intersect chord AB with geometry
351  intersects= IntersectChord( SubStartPoint, EndPointB,
352  NewSafety, LinearStepLength,
353  InterSectionPointE );
354  // E <- Intersection Point of chord AB and either volume A's surface
355  // or a daughter volume's surface ..
356 
357  if( first_substep )
358  {
359  currentSafety = NewSafety;
360  } // Updating safety in other steps is potential future extention
361 
362  if( intersects )
363  {
364  G4FieldTrack IntersectPointVelct_G(CurrentState); // FT-Def-Construct
365 
366  // Find the intersection point of AB true path with the surface
367  // of vol(A), if it exists. Start with point E as first "estimate".
368  G4bool recalculatedEndPt= false;
369 
370  G4bool found_intersection = fIntersectionLocator->
371  EstimateIntersectionPoint( SubStepStartState, CurrentState,
372  InterSectionPointE, IntersectPointVelct_G,
373  recalculatedEndPt, fPreviousSafety,
374  fPreviousSftOrigin);
375  intersects = found_intersection;
376  if( found_intersection )
377  {
378  End_PointAndTangent= IntersectPointVelct_G; // G is our EndPoint ...
379  StepTaken = TruePathLength = IntersectPointVelct_G.GetCurveLength()
380  - OriginalState.GetCurveLength();
381  }
382  else
383  {
384  // Either "minor" chords do not intersect
385  // or else stopped (due to too many steps)
386  //
387  if( recalculatedEndPt )
388  {
389  G4double endAchieved = IntersectPointVelct_G.GetCurveLength();
390  G4double endExpected = CurrentState.GetCurveLength();
391 
392  // Detect failure - due to too many steps
393  G4bool shortEnd = endAchieved
394  < (endExpected*(1.0-CLHEP::perMillion));
395 
396  G4double stepAchieved = endAchieved
397  - SubStepStartState.GetCurveLength();
398 
399  // Update remaining state - must work for 'full' step or
400  // abandonned intersection
401  //
402  CurrentState= IntersectPointVelct_G;
403  s_length_taken = stepAchieved;
404  if( shortEnd )
405  {
406  fParticleIsLooping = true;
407  }
408  }
409  }
410  }
411  if( !intersects )
412  {
413  StepTaken += s_length_taken;
414 
415  if (fpTrajectoryFilter) // For smooth trajectory display (jacek 1/11/2002)
416  {
417  fpTrajectoryFilter->TakeIntermediatePoint(CurrentState.GetPosition());
418  }
419  }
420  first_substep = false;
421 
422 #ifdef G4DEBUG_FIELD
423  if( fNoZeroStep > fActionThreshold_NoZeroSteps )
424  {
425  if( fNoZeroStep > fSevereActionThreshold_NoZeroSteps )
426  G4cout << " Above 'Severe Action' threshold -- for Zero steps. ";
427  else
428  G4cout << " Above 'action' threshold -- for Zero steps. ";
429  G4cout << " Number of zero steps = " << fNoZeroStep << G4endl;
430  printStatus( SubStepStartState, // or OriginalState,
431  CurrentState, CurrentProposedStepLength,
432  NewSafety, do_loop_count, pPhysVol );
433  }
434  if( (fVerboseLevel > 1) && (do_loop_count > fMax_loop_count-10 ))
435  {
436  if( do_loop_count == fMax_loop_count-9 )
437  {
438  G4cout << " G4PropagatorInField::ComputeStep(): " << G4endl
439  << " Difficult track - taking many sub steps." << G4endl;
440  printStatus( SubStepStartState, SubStepStartState, CurrentProposedStepLength,
441  NewSafety, 0, pPhysVol );
442  }
443  printStatus( SubStepStartState, CurrentState, CurrentProposedStepLength,
444  NewSafety, do_loop_count, pPhysVol );
445  }
446 #endif
447 
448  do_loop_count++;
449 
450  } while( (!intersects )
451  && (!fParticleIsLooping)
452  && (StepTaken + kCarTolerance < CurrentProposedStepLength)
453  && ( do_loop_count < fMax_loop_count ) );
454 
455  if( do_loop_count >= fMax_loop_count
456  && (StepTaken + kCarTolerance < CurrentProposedStepLength)
457  )
458  {
459  fParticleIsLooping = true;
460  }
461  if ( ( fParticleIsLooping ) && (fVerboseLevel > 0) )
462  {
463  ReportLoopingParticle( do_loop_count, StepTaken,
464  CurrentProposedStepLength, methodName,
465  CurrentState.GetMomentum(), pPhysVol );
466  }
467 
468  if( !intersects )
469  {
470  // Chord AB or "minor chords" do not intersect
471  // B is the endpoint Step of the current Step.
472  //
473  End_PointAndTangent = CurrentState;
474  TruePathLength = StepTaken; // Original code
475  // Tried the following to avoid potential issue with round-off error
476  // - but has issues... Suppressing this change JA 2015/05/02
477  // TruePathLength = CurrentProposedStepLength;
478  }
479  fLastStepInVolume = intersects;
480 
481  // Set pFieldTrack to the return value
482  //
483  pFieldTrack = End_PointAndTangent;
484 
485 #ifdef G4VERBOSE
486  // Check that "s" is correct
487  //
488  if( std::fabs(OriginalState.GetCurveLength() + TruePathLength
489  - End_PointAndTangent.GetCurveLength()) > 3.e-4 * TruePathLength )
490  {
491  std::ostringstream message;
492  message << "Curve length mis-match between original state "
493  << "and proposed endpoint of propagation." << G4endl
494  << " The curve length of the endpoint should be: "
495  << OriginalState.GetCurveLength() + TruePathLength << G4endl
496  << " and it is instead: "
497  << End_PointAndTangent.GetCurveLength() << "." << G4endl
498  << " A difference of: "
499  << OriginalState.GetCurveLength() + TruePathLength
500  - End_PointAndTangent.GetCurveLength() << G4endl
501  << " Original state = " << OriginalState << G4endl
502  << " Proposed state = " << End_PointAndTangent;
503  G4Exception(methodName, "GeomNav0003", FatalException, message);
504  }
505 #endif
506 
507  if( TruePathLength+kCarTolerance >= CurrentProposedStepLength )
508  {
509  fNoZeroStep = 0;
510  }
511  else
512  {
513  // In particular anomalous cases, we can get repeated zero steps
514  // We identify these cases and take corrective action when they occur.
515  //
516  if( TruePathLength < std::max( fZeroStepThreshold, 0.5*kCarTolerance ) )
517  {
518  fNoZeroStep++;
519  }
520  else
521  {
522  fNoZeroStep = 0;
523  }
524  }
525  if( fNoZeroStep > fAbandonThreshold_NoZeroSteps )
526  {
527  fParticleIsLooping = true;
528  ReportStuckParticle( fNoZeroStep, CurrentProposedStepLength, fFull_CurveLen_of_LastAttempt,
529  pPhysVol );
530  fNoZeroStep = 0;
531  }
532 
533  return TruePathLength;
534 }

626  G4cout << G4endl;
627  }

631 //
632 // Prints Step diagnostics
633 
634 void
635 G4PropagatorInField::PrintStepLengthDiagnostic(
636  G4double CurrentProposedStepLength,
637  G4double decreaseFactor,
638  G4double stepTrial,
