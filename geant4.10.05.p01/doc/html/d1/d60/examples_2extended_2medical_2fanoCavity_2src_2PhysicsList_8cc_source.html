126 void PhysicsList::AddStepMax()
127 {
128  // Step limitation seen as a process
129  StepMax* stepMaxProcess = new StepMax();
130 
131  auto particleIterator=GetParticleIterator();
132  particleIterator->reset();
133  while ((*particleIterator)()){
134  G4ParticleDefinition* particle = particleIterator->value();
135  G4ProcessManager* pmanager = particle->GetProcessManager();
136 
137  if (stepMaxProcess->IsApplicable(*particle) && !particle->IsShortLived())
138  {
139  pmanager ->AddDiscreteProcess(stepMaxProcess);
140  }
141  }
142 }
