113 void G4HadronPhysicsQGSP_BIC::Neutron()
114 {
115  auto neu = new G4NeutronBuilder;
116  AddBuilder(neu);
117  auto qgs = new G4QGSPNeutronBuilder(QuasiElasticQGS);
118  AddBuilder(qgs);
119  qgs->SetMinEnergy(minQGSP_neutron);
120  neu->RegisterMe(qgs);
121  auto ftf = new G4FTFPNeutronBuilder(QuasiElasticFTF);
122  AddBuilder(ftf);
123  ftf->SetMinEnergy(minFTFP_neutron);
124  ftf->SetMaxEnergy(maxFTFP_neutron);
125  neu->RegisterMe(ftf);
126  auto bic = new G4BinaryNeutronBuilder;
127  AddBuilder(bic);
128  bic->SetMaxEnergy(maxBIC_neutron);
129  neu->RegisterMe(bic);
130  neu->Build();
131 }

213 void G4HadronPhysicsQGSP_BIC::ExtraConfiguration()
214 {
215  // --- Kaons ---
216  auto xsk = new G4ComponentGGHadronNucleusXsc();
217  G4VCrossSectionDataSet * kaonxs = new G4CrossSectionInelastic(xsk);
218  G4PhysListUtil::FindInelasticProcess(G4KaonMinus::KaonMinus())->AddDataSet(kaonxs);
219  G4PhysListUtil::FindInelasticProcess(G4KaonPlus::KaonPlus())->AddDataSet(kaonxs);
220  G4PhysListUtil::FindInelasticProcess(G4KaonZeroShort::KaonZeroShort())->AddDataSet(kaonxs);
221  G4PhysListUtil::FindInelasticProcess(G4KaonZeroLong::KaonZeroLong())->AddDataSet(kaonxs);
222 
223  // --- Neutrons ---
224  const G4ParticleDefinition* neutron = G4Neutron::Neutron();
225  G4HadronicProcess* inel = G4PhysListUtil::FindInelasticProcess(neutron);
226  if(inel) { inel->AddDataSet(new G4NeutronInelasticXS()); }
227  G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
228  if (capture) {
229  capture->RegisterMe(new G4NeutronRadCapture());
230  }
231 }
