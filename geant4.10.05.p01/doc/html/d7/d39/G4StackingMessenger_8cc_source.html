46  clearCmd = new G4UIcmdWithAnInteger("/event/stack/clear",this);

75 void G4StackingMessenger::SetNewValue(G4UIcommand * command,G4String newValues)
76 {
77  if( command==statusCmd )
78  {
79  G4cout << "========================== Current status of the stack =====" << G4endl;
80  G4cout << " Number of tracks in the stack" << G4endl;
81  G4cout << " Urgent stack : " << fContainer->GetNUrgentTrack() << G4endl;
82  G4cout << " Waiting stack : " << fContainer->GetNWaitingTrack() << G4endl;
83  G4cout << " Postponed stack : " << fContainer->GetNPostponedTrack() << G4endl;
84  }
85  else if( command==clearCmd )
86  {
87  G4int vc = clearCmd->GetNewIntValue(newValues);
88  switch (vc)
89  {
90  case 2:
91  fContainer->ClearPostponeStack(); // fallthrough
92  /* no break */
93  case 1:
94  fContainer->ClearUrgentStack(); // fallthrough
95  /* no break */
96  case 0:
97  fContainer->ClearWaitingStack();
98  break;
99  case -1:
100  fContainer->ClearUrgentStack();
101  break;
102  case -2:
103  fContainer->ClearPostponeStack();
104  break;
105  }
106  }
107  else if( command==verboseCmd )
108  {
109  fContainer->SetVerboseLevel(verboseCmd->GetNewIntValue(newValues));
110  }
111 }
