300 inline
301 void G4OpBoundaryProcess::ChooseReflection()
302 {
303  G4double rand = G4UniformRand();
304  if ( rand >= 0.0 && rand < prob_ss ) {
305  theStatus = SpikeReflection;
306  theFacetNormal = theGlobalNormal;
307  }
308  else if ( rand >= prob_ss &&
309  rand <= prob_ss+prob_sl) {
310  theStatus = LobeReflection;
311  }
312  else if ( rand > prob_ss+prob_sl &&
313  rand < prob_ss+prob_sl+prob_bs ) {
314  theStatus = BackScattering;
315  }
316  else {
317  theStatus = LambertianReflection;
318  }
319 }

321 inline
322 void G4OpBoundaryProcess::DoAbsorption()
323 {
324  theStatus = Absorption;
325 
326  if ( G4BooleanRand(theEfficiency) ) {
327 
328  // EnergyDeposited =/= 0 means: photon has been detected
329  theStatus = Detection;
330  aParticleChange.ProposeLocalEnergyDeposit(thePhotonMomentum);
331  }
332  else {
333  aParticleChange.ProposeLocalEnergyDeposit(0.0);
334  }
335 
336  NewMomentum = OldMomentum;
337  NewPolarization = OldPolarization;
338 
339 // aParticleChange.ProposeEnergy(0.0);
340  aParticleChange.ProposeTrackStatus(fStopAndKill);
341 }

343 inline
344 void G4OpBoundaryProcess::DoReflection()
345 {
346  if ( theStatus == LambertianReflection ) {
347 
348  NewMomentum = G4LambertianRand(theGlobalNormal);
349  theFacetNormal = (NewMomentum - OldMomentum).unit();
350 
351  }
352  else if ( theFinish == ground ) {
353 
354  theStatus = LobeReflection;
355  if ( PropertyPointer1 && PropertyPointer2 ){
356  } else {
357  theFacetNormal =
358  GetFacetNormal(OldMomentum,theGlobalNormal);
359  }
360  G4double PdotN = OldMomentum * theFacetNormal;
361  NewMomentum = OldMomentum - (2.*PdotN)*theFacetNormal;
362 
363  }
364  else {
365 
366  theStatus = SpikeReflection;
367  theFacetNormal = theGlobalNormal;
368  G4double PdotN = OldMomentum * theFacetNormal;
369  NewMomentum = OldMomentum - (2.*PdotN)*theFacetNormal;
370 
371  }
372  G4double EdotN = OldPolarization * theFacetNormal;
373  NewPolarization = -OldPolarization + (2.*EdotN)*theFacetNormal;
374 }
