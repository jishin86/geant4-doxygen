166 void WLSPhysicsListMessenger::SetNewValue(G4UIcommand* command,
167  G4String newValue)
168 {
169  if( command == fSetAbsorptionCMD ) {
170  fPhysicsList->SetAbsorption(G4UIcmdWithABool::GetNewBoolValue(newValue));
171  }
172 
173  else if( command == fVerboseCmd ) {
174  fPhysicsList->SetVerbose(fVerboseCmd->GetNewIntValue(newValue));
175  }
176 
177  else if( command == fCerenkovCmd ) {
178  fPhysicsList->
179  SetNbOfPhotonsCerenkov(fCerenkovCmd->GetNewIntValue(newValue));
180  }
181 
182  else if (command == fPienuCMD) {
183  G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
184  G4ParticleDefinition* particleDef = particleTable->FindParticle("pi+");
185  G4VDecayChannel* mode =
186  new G4PhaseSpaceDecayChannel("pi+",1.0,2,"e+","nu_e");
187  G4DecayTable* table = new G4DecayTable();
188  table->Insert(mode);
189  // mode = new G4PionRadiativeDecayChannel("pi+",0.000017);
190  // table->Insert(mode);
191  particleDef->SetDecayTable(table);
192  }
193 
194  else if (command == fPimunuCMD) {
195  G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
196  G4ParticleDefinition* particleDef = particleTable->FindParticle("pi+");
197  G4VDecayChannel* mode =
198  new G4PhaseSpaceDecayChannel("pi+",1.000,2,"mu+","nu_mu");
199  G4DecayTable* table = new G4DecayTable();
200  table->Insert(mode);
201  particleDef->SetDecayTable(table);
202  }
203 
204  else if (command == fGammaCutCMD) {
205  fPhysicsList->SetCutForGamma(fGammaCutCMD
206  ->GetNewDoubleValue(newValue));
207  }
208  else if (command == fElectCutCMD) {
209  fPhysicsList->SetCutForElectron(fElectCutCMD
210  ->GetNewDoubleValue(newValue));
211  }
212  else if (command == fPosCutCMD) {
213  fPhysicsList->SetCutForPositron(fPosCutCMD
214  ->GetNewDoubleValue(newValue));
215  }
216  else if (command == fAllCutCMD) {
217  G4double cut = fAllCutCMD->GetNewDoubleValue(newValue);
218  fPhysicsList->SetCutForGamma(cut);
219  fPhysicsList->SetCutForElectron(cut);
220  fPhysicsList->SetCutForPositron(cut);
221  }
222  else if (command == fStepMaxCMD) {
223  fPhysicsList->SetStepMax(fStepMaxCMD
224  ->GetNewDoubleValue(newValue));
225  }
226  else if (command == fClearPhysicsCMD) {
227  fPhysicsList->ClearPhysics();
228  }
229  else if (command == fRemovePhysicsCMD) {
230  G4String name = newValue;
231  fPhysicsList->RemoveFromPhysicsList(name);
232  }
233 }
