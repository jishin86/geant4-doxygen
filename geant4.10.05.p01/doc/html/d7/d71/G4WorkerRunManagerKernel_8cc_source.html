47 void G4WorkerRunManagerKernel::SetupShadowProcess() const
48 {
49  //Master thread has created processes and setup a pointer
50  //to the master process, get it and copy it in this instance
51  G4ParticleTable* theParticleTable = G4ParticleTable::GetParticleTable();
52  G4ParticleTable::G4PTblDicIterator* theParticleIterator = theParticleTable->GetIterator();
53  theParticleIterator->reset();
54  //loop on particles and get process manager from there list of processes
55  while((*theParticleIterator)())
56  {
57  G4ParticleDefinition* pd = theParticleIterator->value();
58  G4ProcessManager* pm = pd->GetProcessManager();
59  G4ProcessManager* pmM= pd->GetMasterProcessManager();
60  if ( !pm || !pmM )
61  {
62  G4ExceptionDescription msg;
63  msg << "Process manager or process manager shadow to master are not set.\n";
64  msg << "Particle : "<<pd->GetParticleName()<<" ("<<pd<<"), proc-manager: "<<pm;
65  msg << " proc-manager-shadow: "<<pmM;
66  G4Exception("G4WorkerRunManagerKernel::SetupShadowProcess()","Run0116",FatalException,
67  msg);
68  return;
69  }
70  G4ProcessVector& procs = *(pm->GetProcessList());
71  G4ProcessVector& procsM= *(pmM->GetProcessList());
72  if( procs.size() != procsM.size() )
73  {
74  G4cout << "G4WorkerRunManagerKernel::SetupShadowProcess() for particle <"
75  << pd->GetParticleName() << ">" << G4endl;
76  G4cout << " ProcessManager : " << pm << " ProcessManagerShadow : " << pmM << G4endl;
77  for(G4int iv1=0;iv1<procs.size();iv1++)
78  { G4cout << " " << iv1 << " - " << procs[iv1]->GetProcessName() << G4endl; }
79  G4cout << "--------------------------------------------------------------" << G4endl;
80  for(G4int iv2=0;iv2<procsM.size();iv2++)
81  { G4cout << " " << iv2 << " - " << procsM[iv2]->GetProcessName() << G4endl; }
82  G4cout << "--------------------------------------------------------------" << G4endl;
83  G4ExceptionDescription msg;
84  msg<<" Size of G4ProcessVector is inconsistent between master and worker threads ";
85  msg<<" for the particle <"<<pd->GetParticleName()<<">. \n";
86  msg<<" size of G4ProcessVector for worker thread is "<<procs.size();
87  msg<<" while master thread is "<<procsM.size()<<".";
88  G4Exception("G4WorkerRunManagerKernel::SetupShadowProcess()","Run0117",FatalException,msg);
89  }
90  //To each process add the reference to the same
91  //process from master. Note that we rely on
92  //processes being in the correct order!
93  // We could use some checking using process name or type
94  for ( G4int idx = 0 ; idx < procs.size() ; ++idx )
95  {
96  procs[idx]->SetMasterProcess(procsM[idx]);
97  }
98  }
99 }
