80 EInside
81 G4ReplicaNavigation::Inside(const G4VPhysicalVolume *pVol,
82  const G4int replicaNo,
83  const G4ThreeVector &localPoint) const
84 {
85  EInside in = kOutside;
86 
87  // Replication data
88  //
89  EAxis axis;
90  G4int nReplicas;
91  G4double width, offset;
92  G4bool consuming;
93 
94  G4double coord, rad2, rmin, tolRMax2, rmax, tolRMin2;
95 
96  pVol->GetReplicationData(axis, nReplicas, width, offset, consuming);
97 
98  switch (axis)
99  {
100  case kXAxis:
101  case kYAxis:
102  case kZAxis:
103  coord = std::fabs(localPoint(axis))-width*0.5;
104  if ( coord<=-halfkCarTolerance )
105  {
106  in = kInside;
107  }
108  else if ( coord<=halfkCarTolerance )
109  {
110  in = kSurface;
111  }
112  break;
113  case kPhi:
114  if ( localPoint.y()||localPoint.x() )
115  {
116  coord = std::fabs(std::atan2(localPoint.y(),localPoint.x()))-width*0.5;
117  if ( coord<=-halfkAngTolerance )
118  {
119  in = kInside;
120  }
121  else if ( coord<=halfkAngTolerance )
122  {
123  in = kSurface;
124  }
125  }
126  else
127  {
128  in = kSurface;
129  }
130  break;
131  case kRho:
132  rad2 = localPoint.perp2();
133  rmax = (replicaNo+1)*width+offset;
134  tolRMax2 = rmax-halfkRadTolerance;
135  tolRMax2 *= tolRMax2;
136  if ( rad2>tolRMax2 )
137  {
138  tolRMax2 = rmax+halfkRadTolerance;
139  tolRMax2 *= tolRMax2;
140  if ( rad2<=tolRMax2 )
141  {
142  in = kSurface;
143  }
144  }
145  else
146  {
147  // Known to be inside outer radius
148  //
149  if ( replicaNo||offset )
150  {
151  rmin = rmax-width;
152  tolRMin2 = rmin-halfkRadTolerance;
153  tolRMin2 *= tolRMin2;
154  if ( rad2>tolRMin2 )
155  {
156  tolRMin2 = rmin+halfkRadTolerance;
157  tolRMin2 *= tolRMin2;
158  if ( rad2>=tolRMin2 )
159  {
160  in = kInside;
161  }
162  else
163  {
164  in = kSurface;
165  }
166  }
167  }
168  else
169  {
170  in = kInside;
171  }
172  }
173  break;
174  default:
175  G4Exception("G4ReplicaNavigation::Inside()", "GeomNav0002",
176  FatalException, "Unknown axis!");
177  break;
178  }
179  return in;
180 }

184 // ********************************************************************

186 G4double
187 G4ReplicaNavigation::DistanceToOut(const G4VPhysicalVolume *pVol,
188  const G4int replicaNo,
189  const G4ThreeVector &localPoint) const
190 {
191  // Replication data
192  //
193  EAxis axis;
194  G4int nReplicas;
195  G4double width,offset;
196  G4bool consuming;
197 
198  G4double safety=0.;
199  G4double safe1,safe2;
200  G4double coord, rho, rmin, rmax;
201 
202  pVol->GetReplicationData(axis, nReplicas, width, offset, consuming);
203  switch(axis)
204  {
205  case kXAxis:
206  case kYAxis:
207  case kZAxis:
208  coord = localPoint(axis);
209  safe1 = width*0.5-coord;
210  safe2 = width*0.5+coord;
211  safety = (safe1<=safe2) ? safe1 : safe2;
212  break;
213  case kPhi:
214  if ( localPoint.y()<=0 )
215  {
216  safety = localPoint.x()*std::sin(width*0.5)
217  + localPoint.y()*std::cos(width*0.5);
218  }
219  else
220  {
221  safety = localPoint.x()*std::sin(width*0.5)
222  - localPoint.y()*std::cos(width*0.5);
223  }
224  break;
225  case kRho:
226  rho = localPoint.perp();
227  rmax = width*(replicaNo+1)+offset;
228  if ( replicaNo||offset )
229  {
230  rmin = rmax-width;
231  safe1 = rho-rmin;
232  safe2 = rmax-rho;
233  safety = (safe1<=safe2) ? safe1 : safe2;
234  }
235  else
236  {
237  safety = rmax-rho;
238  }
239  break;
240  default:
241  G4Exception("G4ReplicaNavigation::DistanceToOut()", "GeomNav0002",
242  FatalException, "Unknown axis!");
243  break;
244  }
245  return (safety >= halfkCarTolerance) ? safety : 0;
246 }

249 // DistanceToOut

252 G4double
253 G4ReplicaNavigation::DistanceToOut(const G4VPhysicalVolume *pVol,
254  const G4int replicaNo,
255  const G4ThreeVector &localPoint,
256  const G4ThreeVector &localDirection,
257  G4ExitNormal& arExitNormal ) const
258 {
259  // Replication data
260  //
261  EAxis axis;
262  G4int nReplicas;
263  G4double width, offset;
264  G4bool consuming;
265 
266  G4double Dist=kInfinity;
267  G4double coord, Comp, lindist;
268  G4double signC = 0.0;
269  G4ExitNormal candidateNormal;
270 
271  pVol->GetReplicationData(axis, nReplicas, width, offset, consuming);
272  switch(axis)
273  {
274  case kXAxis:
275  case kYAxis:
276  case kZAxis:
277  coord = localPoint(axis);
278  Comp = localDirection(axis);
279  if ( Comp>0 )
280  {
281  lindist = width*0.5-coord;
282  Dist = (lindist>0) ? lindist/Comp : 0;
283  signC= 1.0;
284  }
285  else if ( Comp<0 )
286  {
287  lindist = width*0.5+coord;
288  Dist = (lindist>0) ? -lindist/Comp : 0;
289  signC= -1.0;
290  }
291  else
292  {
293  Dist = kInfinity;
294  }
295  // signC = sign<G4double>(Comp)
296  candidateNormal.exitNormal = ( signC * VecCartAxes[axis]);
297  candidateNormal.calculated = true;
298  candidateNormal.validConvex = true;
299  candidateNormal.exitSide =
300  (Comp>0) ? SideCartAxesPlus[axis] : SideCartAxesMinus[axis];
301  break;
302  case kPhi:
303  Dist = DistanceToOutPhi(localPoint,localDirection,width,candidateNormal);
304  // candidateNormal set in call
305  break;
306  case kRho:
307  Dist = DistanceToOutRad(localPoint,localDirection,width,offset,
308  replicaNo,candidateNormal);
309  // candidateNormal set in call
310  break;
311  default:
312  G4Exception("G4ReplicaNavigation::DistanceToOut()", "GeomNav0002",
313  FatalException, "Unknown axis!");
314  break;
315  }
316 
317  arExitNormal= candidateNormal; // .exitNormal;
318 
319  return Dist;
320 }

326 G4double
327 G4ReplicaNavigation::DistanceToOutPhi(const G4ThreeVector &localPoint,
328  const G4ThreeVector &localDirection,
329  const G4double width,
330  G4ExitNormal& foundNormal ) const
331 {
332  // Phi Intersection
333  // NOTE: width<=pi by definition
334  //
335  G4double sinSPhi= -2.0, cosSPhi= -2.0;
336  G4double pDistS, pDistE, compS, compE, Dist, dist2, yi;
337  G4ExitNormal::ESide sidePhi= G4ExitNormal::kNull;
338  G4ThreeVector candidateNormal;
339 
340  if ( (localPoint.x()!=0.0) || (localPoint.y()!=0.0) )
341  {
342  sinSPhi = std::sin(-width*0.5); // SIN of starting phi plane
343  cosSPhi = std::cos(width*0.5); // COS of starting phi plane
344 
345  // pDist -ve when inside
346  //
347  pDistS = localPoint.x()*sinSPhi-localPoint.y()*cosSPhi;
348  // Start plane at phi= -S
349  pDistE = localPoint.x()*sinSPhi+localPoint.y()*cosSPhi;
350  // End plane at phi= +S
351 
352  // Comp -ve when in direction of outwards normal
353  //
354  compS = -sinSPhi*localDirection.x()+cosSPhi*localDirection.y();
355  compE = -sinSPhi*localDirection.x()-cosSPhi*localDirection.y();
356 
357  if ( (pDistS<=halfkCarTolerance)&&(pDistE<=halfkCarTolerance) )
358  {
359  // Inside both phi *full* planes
360  //
361  if ( compS<0 )
362  {
363  dist2 = pDistS/compS;
364  yi = localPoint.y()+dist2*localDirection.y();
365 
366  // Check intersecting with correct half-plane (no -> no intersect)
367  //
368  if ( yi<=0 )
369  {
370  Dist = (pDistS<=-halfkCarTolerance) ? dist2 : 0;
371  sidePhi= G4ExitNormal::kSPhi; // tbc
372  }
373  else
374  {
375  Dist = kInfinity;
376  }
377  }
378  else
379  {
380  Dist = kInfinity;
381  }
382  if ( compE<0 )
383  {
384  dist2 = pDistE/compE;
385 
386  // Only check further if < starting phi intersection
387  //
388  if ( dist2<Dist )
389  {
390  yi = localPoint.y()+dist2*localDirection.y();
391 
392  // Check intersecting with correct half-plane
393  //
394  if ( yi>=0 )
395  {
396  // Leaving via ending phi
397  //
398  Dist = (pDistE<=-halfkCarTolerance) ? dist2 : 0;
399  sidePhi = G4ExitNormal::kEPhi;
400  }
401  }
402  }
403  }
404  else if ( (pDistS>halfkCarTolerance)&&(pDistE>halfkCarTolerance) )
405  {
406  // Outside both *full* phi planes
407  // if towards both >=0 then once inside will remain inside
408  //
409  Dist = ((compS>=0)&&(compE>=0)) ? kInfinity : 0;
410  }
411  else if ( (pDistS>halfkCarTolerance)&&(pDistE<=halfkCarTolerance) )
412  {
413  // Outside full starting plane, inside full ending plane
414  //
415  if ( compE<0 )
416  {
417  dist2 = pDistE/compE;
418  yi = localPoint.y()+dist2*localDirection.y();
419 
420  // Check intersection in correct half-plane
421  // (if not -> remain in extent)
422  //
423  Dist = (yi>0) ? dist2 : kInfinity;
424  if( yi> 0 ) { sidePhi = G4ExitNormal::kEPhi; }
425  }
426  else // Leaving immediately by starting phi
427  {
428  Dist = kInfinity;
429  }
430  }
431  else
432  {
433  // Must be (pDistS<=halfkCarTolerance)&&(pDistE>halfkCarTolerance)
434  // Inside full starting plane, outside full ending plane
435  //
436  if ( compE>=0 )
437  {
438  if ( compS<0 )
439  {
440  dist2 = pDistS/compS;
441  yi = localPoint.y()+dist2*localDirection.y();
442 
443  // Check intersection in correct half-plane
444  // (if not -> remain in extent)
445  //
446  Dist = (yi<0) ? dist2 : kInfinity;
447  if(yi<0) { sidePhi = G4ExitNormal::kSPhi; }
448  }
449  else
450  {
451  Dist = kInfinity;
452  }
453  }
454  else
455  {
456  // Leaving immediately by ending phi
457  //
458  Dist = 0;
459  sidePhi= G4ExitNormal::kEPhi;
460  }
461  }
462  }
463  else
464  {
465  // On z axis + travel not || to z axis -> use direction vector
466  //
467  if( (std::fabs(localDirection.phi())<=width*0.5) )
468  {
469  Dist= kInfinity;
470  }
471  else
472  {
473  Dist= 0;
474  sidePhi= G4ExitNormal::kMY;
475  }
476  }
477 
478  if(sidePhi == G4ExitNormal::kSPhi )
479  {
480  candidateNormal = G4ThreeVector(sinSPhi,-cosSPhi,0.) ;
481  }
482  else if (sidePhi == G4ExitNormal::kEPhi)
483  {
484  candidateNormal = G4ThreeVector(sinSPhi,cosSPhi,0.) ;
485  }
486  else if (sidePhi == G4ExitNormal::kMY )
487  {
488  candidateNormal = G4ThreeVector(0., -1.0, 0.); // Split -S and +S 'phi'
489  }
490  foundNormal.calculated= (sidePhi != G4ExitNormal::kNull );
491  foundNormal.exitNormal= candidateNormal;
492 
493  return Dist;
494 }

496 // ********************************************************************
497 // DistanceToOutRad
498 // ********************************************************************
499 //
500 G4double
501 G4ReplicaNavigation::DistanceToOutRad(const G4ThreeVector &localPoint,
502  const G4ThreeVector &localDirection,
503  const G4double width,
504  const G4double offset,
505  const G4int replicaNo,
506  G4ExitNormal& foundNormal ) const
507 {
508  G4double rmin, rmax, t1, t2, t3, deltaR;
509  G4double b, c, d2, srd;
510  G4ExitNormal::ESide sideR= G4ExitNormal::kNull;
511 
512  //
513  // Radial Intersections
514  //
515 
516  // Find intersction with cylinders at rmax/rmin
517  // Intersection point (xi,yi,zi) on line
518  // x=localPoint.x+t*localDirection.x etc.
519  //
520  // Intersects with x^2+y^2=R^2
521  //
522  // Hence (localDirection.x^2+localDirection.y^2)t^2+
523  // 2t(localPoint.x*localDirection.x+localPoint.y*localDirection.y)+
524  // localPoint.x^2+localPoint.y^2-R^2=0
525  //
526  // t1 t2 t3
527 
528  rmin = replicaNo*width+offset;
529  rmax = (replicaNo+1)*width+offset;
530 
531  t1 = 1.0-localDirection.z()*localDirection.z(); // since v normalised
532  t2 = localPoint.x()*localDirection.x()+localPoint.y()*localDirection.y();
533  t3 = localPoint.x()*localPoint.x()+localPoint.y()*localPoint.y();
534 
535  if ( t1>0 ) // Check not parallel
536  {
537  // Calculate srd, r exit distance
538  //
539  if ( t2>=0 )
540  {
541  // Delta r not negative => leaving via rmax
542  //
543  deltaR = t3-rmax*rmax;
544 
545  // NOTE: Should use
546  // rho-rmax<-halfkRadTolerance - [no sqrts for efficiency]
547  //
548  if ( deltaR<-halfkRadTolerance )
549  {
550  b = t2/t1;
551  c = deltaR/t1;
552  srd = -b+std::sqrt(b*b-c);
553  sideR= G4ExitNormal::kRMax;
554  }
555  else
556  {
557  // On tolerant boundary & heading outwards (or locally
558  // perpendicular to) outer radial surface -> leaving immediately
559  //
560  srd = 0;
561  sideR= G4ExitNormal::kRMax;
562  }
563  }
564  else
565  {
566  // Possible rmin intersection
567  //
568  if (rmin)
569  {
570  deltaR = t3-rmin*rmin;
571  b = t2/t1;
572  c = deltaR/t1;
573  d2 = b*b-c;
574  if ( d2>=0 )
575  {
576  // Leaving via rmin
577  // NOTE: Should use
578  // rho-rmin>halfkRadTolerance - [no sqrts for efficiency]
579  //
580  srd = (deltaR>halfkRadTolerance) ? -b-std::sqrt(d2) : 0.0;
581  // Is the following more accurate ?
582  // srd = (deltaR>halfkRadTolerance) ? c/( -b - std::sqrt(d2)) : 0.0;
583  sideR= G4ExitNormal::kRMin;
584  }
585  else
586  {
587  // No rmin intersect -> must be rmax intersect
588  //
589  deltaR = t3-rmax*rmax;
590  c = deltaR/t1;
591  d2 = b*b-c;
592  srd = (d2 < 0.) ? 0.0 : -b+std::sqrt(d2);
593  sideR= G4ExitNormal::kRMax;
594  }
595  }
596  else
597  {
598  // No rmin intersect -> must be rmax intersect
599  //
600  deltaR = t3-rmax*rmax;
601  b = t2/t1;
602  c = deltaR/t1;
603  d2 = b*b-c;
604  srd = (d2 < 0.) ? 0.0 : -b+std::sqrt(d2);
605  sideR= G4ExitNormal::kRMax;
606  }
607  }
608  }
609  else
610  {
611  srd=kInfinity;
612  sideR= G4ExitNormal::kNull;
613  }
614 
615  if( sideR != G4ExitNormal::kNull ) // if ((side == kRMax) || (side==kRMin))
616  {
617  // Note: returned vector not explicitly normalised
618  // (divided by fRMax for unit vector)
619 
620  G4double xi, yi;
621  xi = localPoint.x() + srd*localDirection.x();
622  yi = localPoint.y() + srd*localDirection.y();
623  G4ThreeVector normalR = G4ThreeVector(xi,yi,0.0);
624 
625  if( sideR == G4ExitNormal::kRMax )
626  {
627  normalR *= 1.0/rmax;
628  }
629  else
630  {
631  normalR *= (-1.0)/rmin;
632  }
633  foundNormal.exitNormal= normalR;
634  foundNormal.calculated= true;
635  foundNormal.validConvex = (sideR == G4ExitNormal::kRMax);
636  foundNormal.exitSide = sideR;
637  }
638  else
639  {
640  foundNormal.calculated= false;
641  }
642 
643  return srd;
644 }

646 // ********************************************************************
647 // ComputeTransformation
648 //
649 // Setup transformation and transform point into local system
650 // ********************************************************************
651 //
652 void
653 G4ReplicaNavigation::ComputeTransformation(const G4int replicaNo,
654  G4VPhysicalVolume* pVol,
655  G4ThreeVector& point) const
656 {
657  G4double val,cosv,sinv,tmpx,tmpy;
658 
659  // Replication data
660  //
661  EAxis axis;
662  G4int nReplicas;
663  G4double width,offset;
664  G4bool consuming;
665 
666  pVol->GetReplicationData(axis, nReplicas, width, offset, consuming);
667 
668  switch (axis)
669  {
670  case kXAxis:
671  val = -width*0.5*(nReplicas-1)+width*replicaNo;
672  pVol->SetTranslation(G4ThreeVector(val,0,0));
673  point.setX(point.x()-val);
674  break;
675  case kYAxis:
676  val = -width*0.5*(nReplicas-1)+width*replicaNo;
677  pVol->SetTranslation(G4ThreeVector(0,val,0));
678  point.setY(point.y()-val);
679  break;
680  case kZAxis:
681  val = -width*0.5*(nReplicas-1)+width*replicaNo;
682  pVol->SetTranslation(G4ThreeVector(0,0,val));
683  point.setZ(point.z()-val);
684  break;
685  case kPhi:
686  val = -(offset+width*(replicaNo+0.5));
687  SetPhiTransformation(val,pVol);
688  cosv = std::cos(val);
689  sinv = std::sin(val);
690  tmpx = point.x()*cosv-point.y()*sinv;
691  tmpy = point.x()*sinv+point.y()*cosv;
692  point.setY(tmpy);
693  point.setX(tmpx);
694  break;
695  case kRho:
696  // No setup required for radial case
697  default:
698  break;
699  }
700 }

702 // ********************************************************************
703 // ComputeTransformation
704 //
705 // Setup transformation into local system
706 // ********************************************************************
707 //
708 void
709 G4ReplicaNavigation::ComputeTransformation(const G4int replicaNo,
710  G4VPhysicalVolume* pVol) const
711 {
712  G4double val;
713 
714  // Replication data
715  //
716  EAxis axis;
717  G4int nReplicas;
718  G4double width, offset;
719  G4bool consuming;
720 
721  pVol->GetReplicationData(axis, nReplicas, width, offset, consuming);
722 
723  switch (axis)
724  {
725  case kXAxis:
726  val = -width*0.5*(nReplicas-1)+width*replicaNo;
727  pVol->SetTranslation(G4ThreeVector(val,0,0));
728  break;
729  case kYAxis:
730  val = -width*0.5*(nReplicas-1)+width*replicaNo;
731  pVol->SetTranslation(G4ThreeVector(0,val,0));
732  break;
