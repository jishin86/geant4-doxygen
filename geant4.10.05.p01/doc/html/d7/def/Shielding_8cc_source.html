68 Shielding::Shielding(G4int verbose, const G4String& n_model,
69  const G4String& HadrPhysVariant )
70 {
71  G4String LEN_model = n_model;
72  size_t find = LEN_model.find("LEND__");
73  G4String evaluation;
74  if ( find != G4String::npos )
75  {
76  evaluation=LEN_model;
77  evaluation.erase(0,find+6);
78  LEN_model="LEND";
79  }
80 
81  G4DataQuestionaire it(photon, neutron, radioactive);
82  G4cout << "<<< Geant4 Physics List simulation engine: Shielding"
83  << HadrPhysVariant << G4endl;
84  if ( LEN_model=="LEND" ) G4cout <<
85  "<<< LEND will be used for low energy neutron and gamma projectiles"
86  << G4endl;
87 
88  defaultCutValue = 0.7*CLHEP::mm;
89  SetCutValue(0, "proton");
90  SetVerboseLevel(verbose);
91 
92  // EM Physics
93  RegisterPhysics( new G4EmStandardPhysics(verbose));
94 
95  // Synchroton Radiation & GN Physics
96  G4EmExtraPhysics* emExtraPhysics = new G4EmExtraPhysics(verbose);
97  if ( LEN_model == "LEND" ) {
98  // Use LEND model for Gamma Nuclear
99  emExtraPhysics->LENDGammaNuclear(true);
100  }
101  RegisterPhysics( emExtraPhysics );
102 
103  // Decays
104  RegisterPhysics( new G4DecayPhysics(verbose) );
105  RegisterPhysics( new G4RadioactiveDecayPhysics(verbose) );
106 
107  // Hadron Elastic scattering
108  if ( LEN_model == "HP" )
109  {
110  RegisterPhysics( new G4HadronElasticPhysicsHP(verbose) );
111  }
112  else if ( LEN_model == "LEND" )
113  {
114  RegisterPhysics( new G4HadronElasticPhysicsLEND(verbose,evaluation));
115  G4DataQuestionaire itt(lend);
116  }
117  else
118  {
119  G4cout << "Shielding Physics List: Warning!" <<G4endl;
120  G4cout << "\"" << LEN_model
121  << "\" is not valid for the low energy neutorn model." <<G4endl;
122  G4cout << "Neutron HP package will be used." <<G4endl;
123  RegisterPhysics( new G4HadronElasticPhysicsHP(verbose) );
124  }
125 
126  G4VPhysicsConstructor* hpc;
127  // Hadron Physics HP or LEND
128  if (HadrPhysVariant == "M") {
129  hpc = new G4HadronPhysicsShielding("hInelastic Shielding", verbose,
130  9.5*CLHEP::GeV, 9.9*CLHEP::GeV);
131  } else {
132  hpc = new G4HadronPhysicsShielding("hInelastic Shielding", verbose,
133  4.0*CLHEP::GeV, 5.0*CLHEP::GeV);
134  }
135 
136  if ( LEN_model == "LEND" ) {
137  delete hpc;
138  if (HadrPhysVariant == "M") {
139  hpc = new G4HadronPhysicsShieldingLEND("hInelastic ShieldingLEND",
140  verbose, 9.5*CLHEP::GeV, 9.9*CLHEP::GeV);
141  } else {
142  hpc = new G4HadronPhysicsShieldingLEND("hInelastic ShieldingLEND",
143  verbose, 4.0*CLHEP::GeV, 5.0*CLHEP::GeV);
144  }
145  } else {
146  //G4cout << "Shielding Physics List: Warning." <<G4endl;
147  //G4cout << "Name of Low Energy Neutron model " << LEN_model
148  // << " is invalid." <<G4endl;
149  //G4cout << "Will use neutron HP package." <<G4endl;
150  }
151  RegisterPhysics( hpc );
152 
153  if ( LEN_model == "HP" ) {
154  //Activate prodcuton of fission fragments in neutronHP
155  G4ParticleHPManager::GetInstance()->SetProduceFissionFragments( true );
156  }
157 
158  // Stopping Physics
159  RegisterPhysics( new G4StoppingPhysics(verbose) );
160 
161  // Ion Physics
162  RegisterPhysics( new G4IonElasticPhysics(verbose));
163  RegisterPhysics( new G4IonQMDPhysics(verbose));
164 }
