54 void B4bSteppingAction::UserSteppingAction(const G4Step* step)
55 {
56 // Collect energy and track length step by step
57 
58  // get volume of the current step
59  auto volume = step->GetPreStepPoint()->GetTouchableHandle()->GetVolume();
60 
61  // energy deposit
62  auto edep = step->GetTotalEnergyDeposit();
63 
64  // step length
65  G4double stepLength = 0.;
66  if ( step->GetTrack()->GetDefinition()->GetPDGCharge() != 0. ) {
67  stepLength = step->GetStepLength();
68  }
69 
70  auto runData = static_cast<B4bRunData*>
71  (G4RunManager::GetRunManager()->GetNonConstCurrentRun());
72 
73  if ( volume == fDetConstruction->GetAbsorberPV() ) {
74  runData->Add(kAbs, edep, stepLength);
75  }
76 
77  if ( volume == fDetConstruction->GetGapPV() ) {
78  runData->Add(kGap, edep, stepLength);
79  }
80 }
