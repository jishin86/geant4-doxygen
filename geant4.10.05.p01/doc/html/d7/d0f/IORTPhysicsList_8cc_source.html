159 void IORTPhysicsList::AddStepMax()
160 {
161  // Step limitation seen as a process
162  stepMaxProcess = new IORTStepMax();
163 
164  auto particleIterator=GetParticleIterator();
165  particleIterator->reset();
166  while ((*particleIterator)()){
167  G4ParticleDefinition* particle = particleIterator->value();
168  G4ProcessManager* pmanager = particle->GetProcessManager();
169 
170  if (stepMaxProcess->IsApplicable(*particle) && pmanager)
171  {
172  pmanager ->AddDiscreteProcess(stepMaxProcess);
173  }
174  }
175 }

177 void IORTPhysicsList::SetCuts()
