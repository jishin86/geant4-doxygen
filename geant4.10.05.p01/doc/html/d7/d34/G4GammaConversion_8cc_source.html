73 void G4GammaConversion::InitialiseProcess(const G4ParticleDefinition*)
74 {
75  if(!isInitialised) {
76  isInitialised = true;
77  G4EmParameters* param = G4EmParameters::Instance();
78  G4double emin = std::max(param->MinKinEnergy(), 2*electron_mass_c2);
79  G4double emax = param->MaxKinEnergy();
80 
81  SetMinKinEnergy(emin);
82 
83  if(!EmModel(0)) { SetEmModel(new G4BetheHeitlerModel()); }
84  EmModel(0)->SetLowEnergyLimit(emin);
85  G4double energyLimit = std::min(EmModel(0)->HighEnergyLimit(), 80*GeV);
86  EmModel(0)->SetHighEnergyLimit(energyLimit);
87  AddEmModel(1, EmModel(0));
88 
89  if(emax > energyLimit) {
90  if(!EmModel(1)) { SetEmModel(new G4PairProductionRelModel()); }
91  EmModel(1)->SetLowEnergyLimit(energyLimit);
92  EmModel(1)->SetHighEnergyLimit(emax);
93  AddEmModel(1, EmModel(1));
94  }
95  }
96 }
