433 template<class OBJECT>
434  struct sortWatcher
435  {
436  bool operator()(const typename G4FastList<OBJECT>::Watcher* left,
437  const typename G4FastList<OBJECT>::Watcher* right) const
438  {
439  if(left && right)
440  {
441  if(left->GetPriority() != right->GetPriority())
442  {
443  return left->GetPriority() < right->GetPriority();
444  }
445  return left < right;
446  }
447  return false;
448  }
