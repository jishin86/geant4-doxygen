29 // class G4HelixImplicitEuler
30 //
31 // Class description:
32 //
33 // Helix Implicit Euler stepper for magnetic field:
34 // x_1 = x_0 + 1/2 * ( helix(h,t_0,x_0)
35 // + helix(h,t_0+h,x_0+helix(h,t0,x0) ) )
36 // Second order solver.
37 // Take the current derivative and add it to the current position.
38 // Take the output and its derivative. Add the mean of both derivatives
39 // to form the final output.
