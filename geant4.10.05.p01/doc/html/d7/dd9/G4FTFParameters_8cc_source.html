469 void G4FTFParameters::InitForInteraction( const G4ParticleDefinition* particle,
470  G4int theA, G4int theZ, G4double PlabPerParticle )
471 {
472  Reset();
473 
474  G4int ProjectilePDGcode = particle->GetPDGEncoding();
475  G4int ProjectileabsPDGcode = std::abs( ProjectilePDGcode );
476  G4double ProjectileMass = particle->GetPDGMass();
477  G4double ProjectileMass2 = ProjectileMass * ProjectileMass;
478 
479  G4int ProjectileBaryonNumber( 0 ), AbsProjectileBaryonNumber( 0 ), AbsProjectileCharge( 0 );
480  G4bool ProjectileIsNucleus = false;
481 
482  if ( std::abs( particle->GetBaryonNumber() ) > 1 ) { // The projectile is a nucleus
483  ProjectileIsNucleus = true;
484  ProjectileBaryonNumber = particle->GetBaryonNumber();
485  AbsProjectileBaryonNumber = std::abs( ProjectileBaryonNumber );
486  AbsProjectileCharge = G4int( particle->GetPDGCharge() );
487  if ( ProjectileBaryonNumber > 1 ) {
488  ProjectilePDGcode = 2212; ProjectileabsPDGcode = 2212; // Proton
489  } else {
490  ProjectilePDGcode = -2212; ProjectileabsPDGcode = 2212; // Anti-Proton
491  }
492  ProjectileMass = G4Proton::Proton()->GetPDGMass();
493  ProjectileMass2 = sqr( ProjectileMass );
