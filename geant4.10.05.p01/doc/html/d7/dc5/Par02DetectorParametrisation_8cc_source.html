44 G4double Par02DetectorParametrisation::GetResolution( Detector aDetector,
45  Parametrisation aParam,
46  G4double aMomentum ) {
47  aMomentum /= GeV; // To make sure momentum's unit is GeV
48  G4double res = 1.0;
49  if ( aParam == eCMS ) {
50  switch ( aDetector ) {
51  case Par02DetectorParametrisation::eTRACKER :
52  res = 0.013;
53  break;
54  case Par02DetectorParametrisation::eEMCAL :
55  res = std::sqrt( std::pow( 0.03 / std::sqrt( aMomentum ), 2 ) // stochastic
56  + std::pow( 0.12 / aMomentum, 2 ) // noise
57  + std::pow( 0.003, 2 ) ); // constant
58  break;
59  case Par02DetectorParametrisation::eHCAL :
60  res = std::sqrt( std::pow( 1.1 / std::sqrt( aMomentum ), 2 ) // stochastic
61  + std::pow( 0.09, 2 ) ); // constant
62  break;
63  }
64  } else if ( aParam == eATLAS ) {
65  switch ( aDetector ) {
66  case Par02DetectorParametrisation::eTRACKER :
67  res = 0.01;
68  break;
69  case Par02DetectorParametrisation::eEMCAL :
70  res = std::sqrt( std::pow( 0.1 / std::sqrt( aMomentum ), 2 ) // stochastic
71  + std::pow( 0.0017, 2 ) ); // constant
72  break;
73  case Par02DetectorParametrisation::eHCAL :
74  res = std::sqrt( std::pow( 0.55 / std::sqrt( aMomentum ), 2 ) // stochastic
75  + std::pow( 0.06, 2 ) ); // constant
76  break;
77  }
78  } else if ( aParam == eALEPH ) {
79  switch ( aDetector ) {
80  case Par02DetectorParametrisation::eTRACKER :
81  res = 0.01;
82  break;
83  case Par02DetectorParametrisation::eEMCAL :
84  res = std::sqrt( std::pow( 0.18 / std::sqrt( aMomentum ), 2 ) // stochastic
85  + std::pow( 0.009, 2 ) ); // constant
86  break;
87  case Par02DetectorParametrisation::eHCAL :
88  res = 0.85 / std::sqrt( aMomentum ); // stochastic
89  break;
90  }
91  }
92  return res;
93 }
