165 //
166 // Calculates total and inelastic Xsc, derives elastic as total - inelastic
167 // accordong to Glauber model with Gribov correction calculated in the dipole
168 // approximation on light cone. Gaussian density of point-like nucleons helps
169 // to calculate rest integrals of the model. [1] B.Z. Kopeliovich,
170 // nucl-th/0306044 + simplification above
171 
172 void G4ComponentGGNuclNuclXsc::ComputeCrossSections(
173  const G4ParticleDefinition* aParticle, G4double kinEnergy,
174  G4int Z, G4int A)
175 {
176  // check cache
177  if(aParticle == fParticle && fZ == Z && fA == A && kinEnergy == fEnergy)
178  { return; }
179  fParticle = aParticle;
180  fZ = Z;
181  fA = A;
182  fEnergy = kinEnergy;
183 
184  G4int pZ = G4lrint(aParticle->GetPDGCharge()*inve);
185  G4int pA = aParticle->GetBaryonNumber();
186 
187  // hydrogen
188  if(1 == Z && 1 == A) {
189  G4double e = kinEnergy*CLHEP::proton_mass_c2/aParticle->GetPDGMass();
190  fHadrNucl->ComputeCrossSections(theProton, e, pZ, pA);
191  fTotalXsc = fHadrNucl->GetTotalGlauberGribovXsc();
192  fElasticXsc = fHadrNucl->GetElasticGlauberGribovXsc();
193  fInelasticXsc = fHadrNucl->GetInelasticGlauberGribovXsc();
194  fProductionXsc = fHadrNucl->GetProductionGlauberGribovXsc();
195  fDiffractionXsc = fHadrNucl->GetDiffractionGlauberGribovXsc();
196  return;
197  }
198  static const G4double cofInelastic = 2.4;
199  static const G4double cofTotal = 2.0;
200 
201  G4double pTkin = kinEnergy/(G4double)pA;
202 
203  G4int pN = pA - pZ;
204  G4int tN = A - Z;
205 
206  G4double tR = GetNucleusRadius( Z, A);
207  G4double pR = GetNucleusRadius( pZ, pA);
208 
209  G4double cB = ComputeCoulombBarier(aParticle, kinEnergy, Z, A, pR, tR);
210 
211  if ( cB > 0. )
212  {
213  G4double sigma = (pZ*Z+pN*tN)*fHNXsc->HadronNucleonXscNS(theProton, theProton, pTkin);
214  G4double ppInXsc = fHNXsc->GetInelasticHadronNucleonXsc();
215 
216  sigma += (pZ*tN+pN*Z)*fHNXsc->HadronNucleonXscNS(theNeutron, theProton, pTkin);
217  G4double npInXsc = fHNXsc->GetInelasticHadronNucleonXsc();
218 
219  // G4cout<<"ppInXsc = "<<ppInXsc/millibarn<<"; npInXsc = "<<npInXsc/millibarn<<G4endl;
220  // G4cout<<"npTotXsc = "<<fHNXsc->GetTotalHadronNucleonXsc()/millibarn<<"; npElXsc = "
221  // <<fHNXsc->GetElasticHadronNucleonXsc()/millibarn<<G4endl;
222 
223  G4double nucleusSquare = cofTotal*CLHEP::pi*( pR*pR + tR*tR ); // basically 2piRR
224 
225  G4double ratio= sigma/nucleusSquare;
226  fTotalXsc = nucleusSquare*G4Log( 1. + ratio )*cB;
227  fInelasticXsc = nucleusSquare*G4Log( 1. + cofInelastic*ratio )*cB/cofInelastic;
228  fElasticXsc = std::max(fTotalXsc - fInelasticXsc, 0.0);
229 
230  G4double difratio = ratio/(1.+ratio);
231  fDiffractionXsc = 0.5*nucleusSquare*( difratio - G4Log( 1. + difratio ) );
232 
233  G4double xratio= ((pZ*Z+pN*tN)*ppInXsc + (pZ*tN+pN*Z)*npInXsc)/nucleusSquare;
234  fProductionXsc = nucleusSquare*G4Log( 1. + cofInelastic*xratio)*cB/cofInelastic;
235  fProductionXsc = std::min(fProductionXsc, fInelasticXsc);
236  }
237  else
238  {
239  fInelasticXsc = 0.;
240  fTotalXsc = 0.;
241  fElasticXsc = 0.;
242  fProductionXsc = 0.;
243  fDiffractionXsc= 0.;
244  }
245 }

249 G4double G4ComponentGGNuclNuclXsc::ComputeCoulombBarier(
250  const G4ParticleDefinition* aParticle,
251  G4double pTkin, G4int Z, G4int A,
252  G4double pR, G4double tR)
253 {
254  G4int pZ = aParticle->GetPDGCharge()*inve;
255  G4double pM = aParticle->GetPDGMass();
256  G4double tM = G4NucleiProperties::GetNuclearMass(A, Z);
257  G4double pElab = pTkin + pM;
258  G4double totEcm = std::sqrt(pM*pM + tM*tM + 2.*pElab*tM);
259  G4double totTcm = totEcm - pM -tM;
260 
261  static const G4double qfact = CLHEP::fine_structure_const*CLHEP::hbarc;
262  G4double bC = qfact*pZ*Z*0.5/(pR + tR);
263 
264  G4double ratio = (totTcm <= bC ) ? 0. : 1. - bC/totTcm;
265  // G4cout<<"G4ComponentGGNuclNuclXsc::ComputeCoulombBarier= "<<ratio
266  // <<"; pTkin(GeV)= " <<pTkin/GeV<<";
267  // " pPlab = "<<pPlab/GeV<<"; bC = "<<bC/GeV<<"; pTcm = "
268  // <<pTcm/GeV<<G4endl;
269  return ratio;
270 }

340 {

342  if(tParticle == theProton) {

351  }

390 G4double G4ComponentGGNuclNuclXsc::GetNucleusRadius(const G4DynamicParticle*,
391  const G4Element* anElement)
392 {
393  G4double At = anElement->GetN();
394  G4double R = fRadiusConst*fCalc->A13(At);
395 
396  static const G4double meanA = 21.;
397  static const G4double tauA1 = 40.;
398  static const G4double tauA2 = 10.;
399  static const G4double tauA3 = 5.;
400 
401  static const G4double a1 = 0.85;
402  static const G4double b1 = 1. - a1;
403 
404  static const G4double b2 = 0.3;
405  static const G4double b3 = 4.;
406 
407  if (At > 20.) // 20.
408  {
409  R *= ( a1 + b1*G4Exp( -(At - meanA)/tauA1) );
410  }
411  else if (At > 3.5)
412  {
413  R *= ( 1.0 + b2*( 1. - G4Exp( (At - meanA)/tauA2) ) );
414  }
415  else
416  {
417  R *= ( 1.0 + b3*( 1. - G4Exp( (At - meanA)/tauA3) ) );
418  }
419  return R;
420 }

449 G4double G4ComponentGGNuclNuclXsc::GetNucleusRadiusDE(G4int Z, G4int A)
450 {
451  // algorithm from diffuse-elastic
452  static const G4double a11 = 1.26; // 1.08, 1.16
453  static const G4double a12 = 1.19; // 1.08, 1.16
454  static const G4double a13 = 1.12; // 1.08, 1.16
455  static const G4double a2 = 1.1;
456  static const G4double a3 = 1.;
457 
458  G4double R = CLHEP::fermi;
459  // Special rms radii for light nucleii
460  if (A < 50)
461  {
462  if(A == 1) { return 0.89*R; }// p
463  else if(A == 2) { return 2.13*R; }// d
464  else if(Z == 1 && A == 3) { return 1.80*R; }// t
465  else if(Z == 2 && A == 3) { return 1.96*R; }// He3
466  else if(Z == 2 && A == 4) { return 1.68*R; }// He4
467  else if(Z == 3) { return 2.40*R; }// Li7
468  else if(Z == 4) { return 2.51*R; }// Be9
469  else if( 10 < A && A <= 15) { R *= a11*(1. - 1./fCalc->Z23(A)); }
470  else if( 15 < A && A <= 20) { R *= a12*(1. - 1./fCalc->Z23(A)); }
471  else if( 20 < A && A <= 30) { R *= a13*(1. - 1./fCalc->Z23(A)); }
472  else { R *= a2; }
473 
474  R *= fCalc->Z13(A);
475  }
476  else
477  {
478  R *= a3*fCalc->powZ(A, 0.27);
479  }
480  return R;
481 }
