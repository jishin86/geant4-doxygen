145 G4TwistTrapAlphaSide::GetNormal(const G4ThreeVector &tmpxx,
146  G4bool isGlobal)
147 {
148  // GetNormal returns a normal vector at a surface (or very close
149  // to surface) point at tmpxx.
150  // If isGlobal=true, it returns the normal in global coordinate.
151  //
152 
153  G4ThreeVector xx;
154  if (isGlobal)
155  {
156  xx = ComputeLocalPoint(tmpxx);
157  if ((xx - fCurrentNormal.p).mag() < 0.5 * kCarTolerance)
158  {
159  return ComputeGlobalDirection(fCurrentNormal.normal);
160  }
161  }
162  else
163  {
164  xx = tmpxx;
165  if (xx == fCurrentNormal.p)
166  {
167  return fCurrentNormal.normal;
168  }
169  }
170 
171  G4double phi ;
172  G4double u ;
173 
174  GetPhiUAtX(xx,phi,u) ; // phi,u for point xx close to surface
175 
176  G4ThreeVector normal = NormAng(phi,u) ; // the normal vector at phi,u
177 
178 #ifdef G4TWISTDEBUG
179  G4cout << "normal vector = " << normal << G4endl ;
180  G4cout << "phi = " << phi << " , u = " << u << G4endl ;
181 #endif
182 
183  if (isGlobal)
184  {
185  fCurrentNormal.normal = ComputeGlobalDirection(normal.unit());
186  }
187  else
188  {
189  fCurrentNormal.normal = normal.unit();
190  }
191 
192  return fCurrentNormal.normal;
193 }

321  c[4] = 4800*

324  + (-10*fdeltaX + fDx3minus1 + fDx4minus2)*v.z()

334  + fa1md1*(7*phixz + 6*fDz*v.y() - 3*fdeltaY*v.z()));

336  fDy1*(fa1md1*(-9*phiyz - 56*fDz*v.x() + 28*fdeltaX*v.z())

342  + 4*fdeltaX*v.z() - 4*fdeltaY*fTAlph*v.z()));

344 #ifdef G4TWISTDEBUG
345  G4cout << "coef = " << c[0] << " "
