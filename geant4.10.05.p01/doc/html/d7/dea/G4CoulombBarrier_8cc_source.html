38 G4CoulombBarrier::G4CoulombBarrier(G4int A, G4int Z)
39  : G4VCoulombBarrier(A, Z)
40 {
41  g4calc = G4Pow::GetInstance();
42  if(Z > 0) {
43  G4double rho = 1.2*CLHEP::fermi;
44  G4double r0 = 1.5*CLHEP::fermi;
45  if(1 == A) {
46  rho = 0.0;
47  } else if(A <= 3) {
48  rho = 0.8*CLHEP::fermi;
49  r0 = 1.7*CLHEP::fermi;
50  } else {
51  r0 = 1.7*CLHEP::fermi;
52  }
53  SetParameters(rho, r0);
54  }
55 }

67 G4double G4CoulombBarrier::BarrierPenetrationFactor(G4int aZ) const
68 {
69  G4double res = 1.0;
70  if(GetZ() == 1) {
71  res = (aZ >= 70) ? 0.80 :
72  (((0.2357e-5*aZ) - 0.42679e-3)*aZ + 0.27035e-1)*aZ + 0.19025;
73  res += 0.06*(GetA() - 1);
74 
75  } else if(GetZ() == 2 && GetA() <= 4) {
76  res = (aZ >= 70) ? 0.98 :
77  (((0.23684e-5*aZ) - 0.42143e-3)*aZ + 0.25222e-1)*aZ + 0.46699;
78  res += 0.12*(4 - GetA());
79  }
80  return res;
81 }
