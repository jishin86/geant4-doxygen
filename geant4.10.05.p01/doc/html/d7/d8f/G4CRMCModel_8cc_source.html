100 G4HadFinalState* G4CRMCModel::ApplyYourself( const G4HadProjectile &theProjectile,
101  G4Nucleus &theTarget ) {
102 
103  // Clean up data vectors
104  gCRMC_data.Clean();
105 
106  // The original track will always be discontinued and secondaries followed.
107  fFinalState.Clear();
108  fFinalState.SetStatusChange( stopAndKill );
109 
110  // Get relevant information about the projectile and target (A, Z, energy/nuc, momentum, etc)
111  const G4ParticleDefinition* definitionP = theProjectile.GetDefinition();
112  G4int AP = G4lrint( definitionP->GetBaryonNumber() );
113  G4int ZP = G4lrint( definitionP->GetPDGCharge() );
114  G4int AT = theTarget.GetA_asInt();
115  G4int ZT = theTarget.GetZ_asInt();
116  G4int idProj = definitionP->GetPDGEncoding();
117  G4int idTarg = ZT*10000 + AT*10;
118  G4ThreeVector pBefore = theProjectile.Get4Momentum().vect();
119  G4double E = theProjectile.GetTotalEnergy();
120  G4double totalEbefore = E*AP + theTarget.AtomicMass( AT, ZT ) + theTarget.GetEnergyDeposit();
121  G4double pProj = theProjectile.Get4Momentum().pz() / GeV; // Energy unit must be GeV
122  G4double pTarg = 0.0;
123 
124  fInterface->crmc_set( 1, // fNCollision,
125  pProj, // fCfg.fProjectileMomentum,
126  pTarg, // fCfg.fTargetMomentum,
127  idProj, // fCfg.fProjectileId,
128  idTarg ); // fCfg.fTargetId);
129 
130  // Sample 1 interaction
131  fInterface->crmc_generate( 0, // fCfg.fTypoaut,
132  1, // iColl+1,
133  gCRMC_data.fNParticles,
134  gCRMC_data.fImpactParameter,
135  gCRMC_data.fPartId[0],
136  gCRMC_data.fPartPx[0],
137  gCRMC_data.fPartPy[0],
138  gCRMC_data.fPartPz[0],
139  gCRMC_data.fPartEnergy[0],
140  gCRMC_data.fPartMass[0],
141  gCRMC_data.fPartStatus[0] );
142 
143  // Save secondary particles for output
144  for ( G4int i = 0; i < gCRMC_data.fNParticles; i++ ) {
145  // Keep only final state particles
146  // (-9 is the beam, 2 is a particle which decayed and 1 is final)
147  if ( gCRMC_data.fPartStatus[i] != 1 ) continue;
148  G4ParticleDefinition* pdef = GetParticleDefinition( gCRMC_data.fPartId[i] );
149  G4DynamicParticle* part = new G4DynamicParticle( pdef,
150  G4ThreeVector( gCRMC_data.fPartPx[i]*GeV,
151  gCRMC_data.fPartPy[i]*GeV,
152  gCRMC_data.fPartPz[i]*GeV ) );
153  fFinalState.AddSecondary( part );
154  }
155 
156  if ( verbose >= 3 ) {
157  G4double totalEafter = 0.0;
158  G4ThreeVector totalPafter;
159  G4double charge = 0.0;
160  G4int baryon = 0;
161  G4int nSecondaries = fFinalState.GetNumberOfSecondaries();
162  for ( G4int j = 0; j < nSecondaries; j++ ) {
163  totalEafter += fFinalState.GetSecondary(j)->GetParticle()->GetTotalEnergy();
164  totalPafter += fFinalState.GetSecondary(j)->GetParticle()->GetMomentum();
165  G4ParticleDefinition* pd = fFinalState.GetSecondary(j)->GetParticle()->GetDefinition();
166  charge += pd->GetPDGCharge();
167  baryon += pd->GetBaryonNumber();
168  }
169  G4cout << "----------------------------------------" << G4endl
170  << "Total energy before collision = " << totalEbefore << " MeV" << G4endl
171  << "Total energy after collision = " << totalEafter << " MeV" << G4endl
172  << "Total momentum before collision = " << pBefore << " MeV/c" << G4endl
173  << "Total momentum after collision = " << totalPafter << " MeV/c" << G4endl;
174  if ( verbose >= 4 ) {
175  G4cout << "Total charge before collision = " << (ZP + ZT)*eplus << G4endl
176  << "Total charge after collision = " << charge <<G4endl
177  << "Total baryon number before collision = " << (AP + AT) << G4endl
178  << "Total baryon number after collision = "<< baryon << G4endl;
179  }
180  G4cout << "----------------------------------------" << G4endl;
181  }
182 
183  return &fFinalState;
184 }

205 G4ParticleDefinition* G4CRMCModel::GetParticleDefinition( long particle_id ) {
206  // CRMC ion definition : id = crmc_ion_coef_0 + crmc_ion_coef_z*Z + crmc_ion_coef_a*A
207  const G4int crmc_ion_coef_0 = 1000000000;
208  const G4int crmc_ion_coef_z = 10000;
209  const G4int crmc_ion_coef_a = 10;
210  G4ParticleDefinition* pdef = fParticleTable->FindParticle( particle_id );
211  if ( ! pdef && particle_id > crmc_ion_coef_0 ) {
212  int Z = ( particle_id - crmc_ion_coef_0 ) / crmc_ion_coef_z;
213  int A = ( particle_id - crmc_ion_coef_0 - crmc_ion_coef_z*Z ) / crmc_ion_coef_a;
214  pdef = fIonTable->GetIon( Z, A );
215  }
216  return pdef;
217 }
