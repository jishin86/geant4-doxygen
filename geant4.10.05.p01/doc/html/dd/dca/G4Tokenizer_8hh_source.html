60

62  if(string2tokenize[actual]==str[i]) hasws=true;

66  hasws=false;
67  }

71

73  {
74  for(i=0;i<l;i++)
75  if(string2tokenize[j]==str[i]) break;
76  if(i<l) break;
77  }
78  if(j!=string2tokenize.size())
79  {
80  tmp=actual;
81  actual=j+1;
82  return string2tokenize(tmp,j-tmp);
83  }
84  else
85  {

87  actual=j;
88  return string2tokenize(tmp,j-tmp);
89  }
90  }
