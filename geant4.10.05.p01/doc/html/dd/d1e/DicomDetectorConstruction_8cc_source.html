95 G4VPhysicalVolume* DicomDetectorConstruction::Construct()
96 {
97  if(!fConstructed || fWorld_phys == 0) {
98  fConstructed = true;
99  InitialisationOfMaterials();
100 
101  //----- Build world
102  G4double worldXDimension = 1.*m;
103  G4double worldYDimension = 1.*m;
104  G4double worldZDimension = 1.*m;
105 
106  fWorld_solid = new G4Box( "WorldSolid",
107  worldXDimension,
108  worldYDimension,
109  worldZDimension );
110 
111  fWorld_logic = new G4LogicalVolume( fWorld_solid,
112  fAir,
113  "WorldLogical",
114  0, 0, 0 );
115 
116  fWorld_phys = new G4PVPlacement( 0,
117  G4ThreeVector(0,0,0),
118  "World",
119  fWorld_logic,
120  0,
121  false,
122  0 );
123 
124 #ifdef G4_DCMTK
125  ReadPhantomDataNew();
126  ConstructPhantomContainerNew();
127 #else
128  ReadPhantomData();
129  ConstructPhantomContainer();
130 #endif
131 
132  ConstructPhantom();
133  }
134  return fWorld_phys;
135 }

456 void DicomDetectorConstruction::ReadPhantomDataNew()
457 {
458 #ifdef G4_DCMTK
459  G4String fileName = DicomFileMgr::GetInstance()->GetFileOutName();
460 
461  std::ifstream fin(fileName);
462  std::vector<G4String> wl;
463  G4int nMaterials;
464  fin >> nMaterials;
465  G4String mateName;
466  G4int nmate;
467  for( G4int ii = 0; ii < nMaterials; ii++ ){
468  fin >> nmate;
469  fin >> mateName;
470  if( mateName[0] == '"' && mateName[mateName.length()-1] == '"' ) {
471  mateName = mateName.substr(1,mateName.length()-2);
472  }
473  G4cout << "GmReadPhantomG4Geometry::ReadPhantomData reading nmate "
474  << ii << " = " << nmate
475  << " mate " << mateName << G4endl;
476  if( ii != nmate )
477  G4Exception("GmReadPhantomG4Geometry::ReadPhantomData",
478  "Wrong argument", FatalErrorInArgument,
479  "Material number should be in increasing order:wrong material number");
480 
481  G4Material* mate = 0;
482  const G4MaterialTable* matTab = G4Material::GetMaterialTable();
483  std::vector<G4Material*>::const_iterator matite;
484  for( matite = matTab->begin(); matite != matTab->end(); ++matite ) {
485  if( (*matite)->GetName() == mateName ) {
486  mate = *matite;
487  }
488  }
489  if( mate == 0 ) {
490  mate = G4NistManager::Instance()->FindOrBuildMaterial(mateName);
491  }
492  if( !mate ) G4Exception("GmReadPhantomG4Geometry::ReadPhantomData",
493  "Wrong argument",
494  FatalErrorInArgument,
495  ("Material not found" + mateName).c_str());
496  thePhantomMaterialsOriginal[nmate] = mate;
497  }
498 
499  fin >> fNVoxelX >> fNVoxelY >> fNVoxelZ;
500  G4cout << "GmReadPhantomG4Geometry::ReadPhantomData fNVoxel X/Y/Z "
501  << fNVoxelX << " "
502  << fNVoxelY << " " << fNVoxelZ << G4endl;
503  fin >> fMinX >> fMaxX;
504  fin >> fMinY >> fMaxY;
505  fin >> fMinZ >> fMaxZ;
506  fVoxelHalfDimX = (fMaxX-fMinX)/fNVoxelX/2.;
507  fVoxelHalfDimY = (fMaxY-fMinY)/fNVoxelY/2.;
508  fVoxelHalfDimZ = (fMaxZ-fMinZ)/fNVoxelZ/2.;
509 #ifdef G4VERBOSE
510  G4cout << " Extension in X " << fMinX << " " << fMaxX << G4endl
511  << " Extension in Y " << fMinY << " " << fMaxY << G4endl
512  << " Extension in Z " << fMinZ << " " << fMaxZ << G4endl;
513 #endif
514 
515  fMateIDs = new size_t[fNVoxelX*fNVoxelY*fNVoxelZ];
516  for( G4int iz = 0; iz < fNVoxelZ; iz++ ) {
517  for( G4int iy = 0; iy < fNVoxelY; iy++ ) {
518  for( G4int ix = 0; ix < fNVoxelX; ix++ ) {
519  G4int mateID;
520  fin >> mateID;
521  G4int nnew = ix + (iy)*fNVoxelX + (iz)*fNVoxelX*fNVoxelY;
522  if( mateID < 0 || mateID >= nMaterials ) {
523  G4Exception("GmReadPhantomG4Geometry::ReadPhantomData",
524  "Wrong index in phantom file",
525  FatalException,
526  G4String("It should be between 0 and "
527  + G4UIcommand::ConvertToString(nMaterials-1)
528  + ", while it is "
529  + G4UIcommand::ConvertToString(mateID)).c_str());
530  }
531  fMateIDs[nnew] = mateID;
532  }
533  }
534  }
535 
536  ReadVoxelDensities( fin );
537 
538  fin.close();
539 #endif

553  if( part ) densityDiff = G4UIcommand::ConvertToDouble(part);
554

556  for( size_t ii = 0; ii < thePhantomMaterialsOriginal.size(); ii++ ){
557  densityDiffs[ii] = densityDiff; //currently all materials with same step
558  }

560

562  std::map< std::pair<G4Material*,G4int>, matInfo* > newMateDens;
563 
564  //---- Read the material densities

566  for( G4int iz = 0; iz < fNVoxelZ; iz++ ) {
567  for( G4int iy = 0; iy < fNVoxelY; iy++ ) {
568  for( G4int ix = 0; ix < fNVoxelX; ix++ ) {
569  fin >> dens;

571 
572  if( densityDiff != -1. ) continue;
573 
574  //--- store the minimum and maximum density for each material
575  mpite = densiMinMax.find( fMateIDs[copyNo] );
576  if( dens < (*mpite).second.first ) (*mpite).second.first = dens;
577  if( dens > (*mpite).second.second ) (*mpite).second.second = dens;
578  //--- Get material from original list of material in file
579  G4int mateID = fMateIDs[copyNo];
580  std::map<G4int,G4Material*>::const_iterator imite =
581  thePhantomMaterialsOriginal.find(mateID);
582

584  if(std::fabs(dens - (*imite).second->GetDensity()/CLHEP::g*CLHEP::cm3 )
585  < 1.e-9 ) continue;
586 
587  //--- Build material name with thePhantomMaterialsOriginal name+density
588  G4int densityBin = (G4int(dens/densityDiffs[mateID]));
589

591  + G4UIcommand::ConvertToString(densityBin);

594

596  newMateDens.find( matdens );

599  mi->fSumdens += dens;
600  mi->fNvoxels++;
601  fMateIDs[copyNo] = thePhantomMaterialsOriginal.size()-1 + mi->fId;
602  } else {
603  matInfo* mi = new matInfo;
604  mi->fSumdens = dens;
605  mi->fNvoxels = 1;
606  mi->fId = newMateDens.size()+1;
607  newMateDens[matdens] = mi;
608  fMateIDs[copyNo] = thePhantomMaterialsOriginal.size()-1 + mi->fId;
609  }
610  }
611  }
612  }
613 
614  if( densityDiff != -1. ) {
615  for( mpite = densiMinMax.begin(); mpite != densiMinMax.end(); mpite++ ){
616 #ifdef G4VERBOSE
617  G4cout << "DicomDetectorConstruction::ReadVoxelDensities"
618  << " ORIG MATERIALS DENSITY "
619  << (*mpite).first << " MIN " << (*mpite).second.first << " MAX "
620  << (*mpite).second.second << G4endl;
621 #endif
622  }
623  }
624 
625  //----- Build the list of phantom materials that go to Parameterisation
626  //--- Add original materials
627  std::map<G4int,G4Material*>::const_iterator mimite;
628  for( mimite = thePhantomMaterialsOriginal.begin();
629  mimite != thePhantomMaterialsOriginal.end(); mimite++ ){
630  fMaterials.push_back( (*mimite).second );
631  }
632  //
633  //---- Build and add new materials
634 std::map< std::pair<G4Material*,G4int>, matInfo* >::iterator mppite;
635  for( mppite= newMateDens.begin(); mppite != newMateDens.end(); mppite++ ){
636  G4double averdens = (*mppite).second->fSumdens/(*mppite).second->fNvoxels;

638 #ifndef G4VERBOSE

642  << G4int(1000*averdens)/1000
643  << " " << G4int(1000*averdens)/1000. << G4endl;
644 #endif
645 
646  G4String mateName = ((*mppite).first).first->GetName() + "_"
647  + G4UIcommand::ConvertToString(saverdens);

689 {

691  << fname << G4endl; //GDEB
692

694  G4cout << " DicomDetectorConstruction::ReadPhantomDataFile opening file "
695  << fname << G4endl;
696 #endif
697 
698  std::ifstream fin(fname.c_str(), std::ios_base::in);
699  if( !fin.is_open() ) {
700  G4Exception("DicomDetectorConstruction::ReadPhantomDataFile",
701  "",
702  FatalErrorInArgument,
703  G4String("File not found " + fname ).c_str());
704  }
705  //----- Define density differences (maximum density difference to create
706  // a new material)
707  char* part = getenv( "DICOM_CHANGE_MATERIAL_DENSITY" );
708  G4double densityDiff = -1.;
709  if( part ) densityDiff = G4UIcommand::ConvertToDouble(part);
710  if( densityDiff != -1. ) {
711  for( unsigned int ii = 0; ii < fOriginalMaterials.size(); ii++ ){
712  fDensityDiffs[ii] = densityDiff; //currently all materials with
713  // same difference
714  }
715  }else {
716  if( fMaterials.size() == 0 ) { // do it only for first slice
717  for( unsigned int ii = 0; ii < fOriginalMaterials.size(); ii++ ){
718  fMaterials.push_back( fOriginalMaterials[ii] );
719  }
720  }
721  }
722 
723  //----- Read data header
724  DicomPhantomZSliceHeader* sliceHeader = new DicomPhantomZSliceHeader( fin );
725  fZSliceHeaders.push_back( sliceHeader );
726 
727  //----- Read material indices
728  G4int nVoxels = sliceHeader->GetNoVoxels();
729 
730  //--- If first slice, initiliaze fMateIDs
731  if( fZSliceHeaders.size() == 1 ) {
732  //fMateIDs = new unsigned int[fNoFiles*nVoxels];
733  fMateIDs = new size_t[fNoFiles*nVoxels];
734 
735  }
736 
737  unsigned int mateID;
738  // number of voxels from previously read slices
739  G4int voxelCopyNo = (fZSliceHeaders.size()-1)*nVoxels;
740  for( G4int ii = 0; ii < nVoxels; ii++, voxelCopyNo++ ){
741  fin >> mateID;
742  fMateIDs[voxelCopyNo] = mateID;
743  }
744 
745  //----- Read material densities and build new materials if two voxels have
746  // same material but its density is in a different density interval
747  // (size of density intervals defined by densityDiff)
748  G4double density;
749  // number of voxels from previously read slices
750  voxelCopyNo = (fZSliceHeaders.size()-1)*nVoxels;
751  for( G4int ii = 0; ii < nVoxels; ii++, voxelCopyNo++ ){
752  fin >> density;
753 
754  //-- Get material from list of original materials
755  mateID = fMateIDs[voxelCopyNo];
756  G4Material* mateOrig = fOriginalMaterials[mateID];
757 
758  //-- Get density bin: middle point of the bin in which the current
759  // density is included
760  G4String newMateName = mateOrig->GetName();
761  float densityBin = 0.;
762  if( densityDiff != -1.) {
763  densityBin = fDensityDiffs[mateID] *
764  (G4int(density/fDensityDiffs[mateID])+0.5);

766  newMateName += G4UIcommand::ConvertToString(densityBin);
767  }
768 
769  //-- Look if a material with this name is already created
770  // (because a previous voxel was already in this density bin)
771  unsigned int im;
772  for( im = 0; im < fMaterials.size(); im++ ){
773  if( fMaterials[im]->GetName() == newMateName ) {
774  break;
775  }
776  }
777  //-- If material is already created use index of this material
778  if( im != fMaterials.size() ) {
779  fMateIDs[voxelCopyNo] = im;
780  //-- else, create the material
781  } else {
782  if( densityDiff != -1.) {
783  fMaterials.push_back( BuildMaterialWithChangingDensity( mateOrig,
784  densityBin, newMateName ) );
785  fMateIDs[voxelCopyNo] = fMaterials.size()-1;
786  } else {
787  G4cerr << " im " << im << " < " << fMaterials.size() << " name "

789  G4Exception("DicomDetectorConstruction::ReadPhantomDataFile",
790  "",
791  FatalErrorInArgument,
792  "Wrong index in material"); //it should never reach here
793  }
794  }
795  }
