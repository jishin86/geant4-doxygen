74 G4double G4GEMProbability::EmissionProbability(const G4Fragment & fragment,
75  G4double MaximalKineticEnergy)
76 {
77  G4double probability = 0.0;
78 
79  if (MaximalKineticEnergy > 0.0 && fragment.GetExcitationEnergy() > 0.0) {
80  G4double CoulombBarrier = GetCoulombBarrier(fragment);
81 
82  probability =
83  CalcProbability(fragment,MaximalKineticEnergy,CoulombBarrier);
84 
85  // Next there is a loop over excited states for this channel
86  // summing probabilities
87  size_t nn = ExcitEnergies.size();
88  if (0 < nn) {
89  G4double SavedSpin = Spin;
90  for (size_t i = 0; i <nn; ++i) {
91  Spin = ExcitSpins[i];
92  // substract excitation energies
93  G4double Tmax = MaximalKineticEnergy - ExcitEnergies[i];
94  if (Tmax > 0.0) {
95  G4double width = CalcProbability(fragment,Tmax,CoulombBarrier);
96  //JMQ April 2010 added condition to prevent reported crash
97  // update probability
98  if (width > 0. && fPlanck < width*ExcitLifetimes[i]) {
99  probability += width;
100  }
101  }
102  }
103  // Restore Spin
104  Spin = SavedSpin;
105  }
106  }
107  // Normalization = probability;
108  return probability;
109 }

111 G4double G4GEMProbability::CalcProbability(const G4Fragment & fragment,
112  G4double MaximalKineticEnergy,
113  G4double V)
114 
115 // Calculate integrated probability (width) for evaporation channel
116 {
117  G4int A = fragment.GetA_asInt();
118  G4int Z = fragment.GetZ_asInt();
119 
120  G4int ResidualA = A - theA;
121  G4int ResidualZ = Z - theZ;
122  G4double U = fragment.GetExcitationEnergy();
123 
124  G4double NuclearMass = fragment.ComputeGroundStateMass(theZ, theA);
125 
126  G4double Alpha = CalcAlphaParam(fragment);
127  G4double Beta = CalcBetaParam(fragment);
128 
129  // ***RESIDUAL***
130  //JMQ (September 2009) the following quantities refer to the RESIDUAL:
131 
132  G4double delta0 = fPairCorr->GetPairingCorrection(ResidualA, ResidualZ);
133 
134  G4double a = theEvapLDPptr->
135  LevelDensityParameter(ResidualA,ResidualZ,MaximalKineticEnergy+V-delta0);
136  G4double Ux = (2.5 + 150.0/G4double(ResidualA))*MeV;
137  G4double Ex = Ux + delta0;
138  G4double T = 1.0/(std::sqrt(a/Ux) - 1.5/Ux);
139  //JMQ fixed bug in units
140  G4double E0 = Ex - T*(G4Log(T/MeV) - G4Log(a*MeV)/4.0
141  - 1.25*G4Log(Ux/MeV) + 2.0*std::sqrt(a*Ux));
142  // ***end RESIDUAL ***
143  // ***PARENT***
144  //JMQ (September 2009) the following quantities refer to the PARENT:
145 
146  G4double deltaCN = fPairCorr->GetPairingCorrection(A, Z);
147  G4double aCN = theEvapLDPptr->LevelDensityParameter(A, Z, U-deltaCN);
148  G4double UxCN = (2.5 + 150.0/G4double(A))*MeV;
149  G4double ExCN = UxCN + deltaCN;
150  G4double TCN = 1.0/(std::sqrt(aCN/UxCN) - 1.5/UxCN);
151  // ***end PARENT***
152 
153  G4double Width;
154  G4double InitialLevelDensity;
155  G4double t = MaximalKineticEnergy/T;
156  if ( MaximalKineticEnergy < Ex ) {
157  //JMQ 190709 bug in I1 fixed (T was missing)
158  Width = (I1(t,t)*T + (Beta+V)*I0(t))/G4Exp(E0/T);
159  //JMQ 160909 fix: InitialLevelDensity has been taken away
160  //(different conditions for initial CN..)
161  } else {
162 
163  //VI minor speedup
164  G4double expE0T = G4Exp(E0/T);
165  static const G4double sqrt2 = std::sqrt(2.0);
166 
167  G4double tx = Ex/T;
168  G4double s0 = 2.0*std::sqrt(a*(MaximalKineticEnergy-delta0));
169  G4double sx = 2.0*std::sqrt(a*(Ex-delta0));
170  // VI: protection against FPE exception
171  if(s0 > 350.) { s0 = 350.; }
172  Width = I1(t,tx)*T/expE0T + I3(s0,sx)*G4Exp(s0)/(sqrt2*a);
173 
174  // VI this cannot happen!
175  // For charged particles (Beta+V) = 0 beacuse Beta = -V
176  //if (theZ == 0) {
177  // Width += (Beta+V)*(I0(tx)/expE0T + 2.0*sqrt2*I2(s0,sx)*G4Exp(s0));
178  //}
179  }
180 
181  //JMQ 14/07/2009 BIG BUG : NuclearMass is in MeV => hbarc instead of
182  // hbar_planck must be used
183  // G4double g = (2.0*Spin+1.0)*NuclearMass/(pi2* hbar_Planck*hbar_Planck);
184  G4double gg = (2.0*Spin+1.0)*NuclearMass/(pi2* hbarc*hbarc);
185 
186  //JMQ 190709 fix on Rb and geometrical cross sections according to
187  // Furihata's paper (JAERI-Data/Code 2001-105, p6)
188  G4double Rb = 0.0;
189  if (theA > 4)
190  {
191  G4double Ad = fG4pow->Z13(ResidualA);
192  G4double Aj = fG4pow->Z13(theA);
193  Rb = 1.12*(Aj + Ad) - 0.86*((Aj+Ad)/(Aj*Ad))+2.85;
194  Rb *= fermi;
195  }
196  else if (theA>1)
197  {
198  G4double Ad = fG4pow->Z13(ResidualA);
199  G4double Aj = fG4pow->Z13(theA);
200  Rb=1.5*(Aj+Ad)*fermi;
201  }
202  else
203  {
204  G4double Ad = fG4pow->Z13(ResidualA);
205  Rb = 1.5*Ad*fermi;
206  }
207  G4double GeometricalXS = pi*Rb*Rb;
208  //end of JMQ fix on Rb by 190709
209 
210  //JMQ 160909 fix: initial level density must be calculated according to the
211  // conditions at the initial compound nucleus
212  // (it has been removed from previous "if" for the residual)
213 
214  if ( U < ExCN )
215  {
216  //JMQ fixed bug in units
217  //VI moved the computation here
218  G4double E0CN = ExCN - TCN*(G4Log(TCN/MeV) - 0.25*G4Log(aCN*MeV)
219  - 1.25*G4Log(UxCN/MeV)
220  + 2.0*std::sqrt(aCN*UxCN));
221 
222  InitialLevelDensity = (pi/12.0)*G4Exp((U-E0CN)/TCN)/TCN;
223  }
224  else
225  {
226  //VI speedup
227  G4double x = U-deltaCN;
228  G4double x1 = std::sqrt(aCN*x);
229 
230  InitialLevelDensity = (pi/12.0)*G4Exp(2*x1)/(x*std::sqrt(x1));
231  }
232 
233  //JMQ 190709 BUG : pi instead of sqrt(pi) must be here according
234  // to Furihata's report:
235  Width *= pi*gg*GeometricalXS*Alpha/(12.0*InitialLevelDensity);
236 
237  return Width;
238 }

240 G4double G4GEMProbability::I3(G4double s0, G4double sx)
241 {
242  G4double s2 = s0*s0;
243  G4double sx2 = sx*sx;
244  G4double S = 1.0/std::sqrt(s0);
245  G4double S2 = S*S;
246  G4double Sx = 1.0/std::sqrt(sx);
247  G4double Sx2 = Sx*Sx;
248 
249  G4double p1 = S *(2.0 + S2 *( 4.0 + S2 *( 13.5 + S2 *( 60.0 + S2 * 325.125 ))));
250  G4double p2 = Sx*Sx2 *(
251  (s2-sx2) + Sx2 *(
252  (1.5*s2+0.5*sx2) + Sx2 *(
253  (3.75*s2+0.25*sx2) + Sx2 *(
254  (12.875*s2+0.625*sx2) + Sx2 *(
255  (59.0625*s2+0.9375*sx2) + Sx2 *(324.8*s2+3.28*sx2))))));
256 
257  p2 *= G4Exp(sx-s0);
258 
259  return p1-p2;
260 }
