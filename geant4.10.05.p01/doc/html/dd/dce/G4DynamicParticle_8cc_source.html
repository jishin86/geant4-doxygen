190 G4DynamicParticle::G4DynamicParticle(const G4ParticleDefinition * aParticleDefinition,
191  G4double totalEnergy,
192  const G4ThreeVector &aParticleMomentum):
193  theParticleDefinition(aParticleDefinition),
194  theKineticEnergy(0.0),
195  theProperTime(0.0),
196  theDynamicalMass(aParticleDefinition->GetPDGMass()),
197  theDynamicalCharge(aParticleDefinition->GetPDGCharge()),
198  theDynamicalSpin(aParticleDefinition->GetPDGSpin()),
199  theDynamicalMagneticMoment(aParticleDefinition->GetPDGMagneticMoment()),
200  theElectronOccupancy(nullptr),
201  thePreAssignedDecayProducts(nullptr),
202  thePreAssignedDecayTime(-1.0),
203  verboseLevel(1),
204  primaryParticle(nullptr),
205  thePDGcode(0)
206 {
207  // total energy and 3-dim momentum are given
208  G4double pModule2 = aParticleMomentum.mag2();
209  if (pModule2>0.0) {
210  G4double mass2 = totalEnergy*totalEnergy - pModule2;
211  G4double PDGmass2 = (aParticleDefinition->GetPDGMass())*(aParticleDefinition->GetPDGMass());
212  SetMomentumDirection(aParticleMomentum.unit());
213  if (mass2 < EnergyMomentumRelationAllowance*EnergyMomentumRelationAllowance) {
214  theDynamicalMass = 0.;
215  SetKineticEnergy(totalEnergy);
216  } else {
217  if (std::abs(PDGmass2-mass2)>EnergyMomentumRelationAllowance*EnergyMomentumRelationAllowance){
218  theDynamicalMass = std::sqrt(mass2);
219  SetKineticEnergy(totalEnergy-theDynamicalMass);
220  } else {
221  SetKineticEnergy(totalEnergy-theDynamicalMass);
222  }
223  }
224  } else {
225  SetMomentumDirection(1.0,0.0,0.0);
226  SetKineticEnergy(0.0);
227  }
228 }

373 void G4DynamicParticle::SetMomentum(const G4ThreeVector &momentum)
374 {
375  G4double pModule2 = momentum.mag2();
376  if (pModule2>0.0) {
377  G4double mass = theDynamicalMass;
378  SetMomentumDirection(momentum.unit());
379  SetKineticEnergy(std::sqrt(pModule2 + mass*mass)-mass);
380  } else {
381  SetMomentumDirection(1.0,0.0,0.0);
382  SetKineticEnergy(0.0);
383  }

387 void G4DynamicParticle::Set4Momentum(const G4LorentzVector &momentum )
388 {
389  G4double pModule2 = momentum.vect().mag2();
390  if (pModule2>0.0) {
391  SetMomentumDirection(momentum.vect().unit());
392  G4double totalenergy = momentum.t();
393  G4double mass2 = totalenergy*totalenergy - pModule2;
394  G4double PDGmass2 = (theParticleDefinition->GetPDGMass())*(theParticleDefinition->GetPDGMass());
395  if (mass2 < EnergyMomentumRelationAllowance*EnergyMomentumRelationAllowance) {
396  theDynamicalMass = 0.;
397  } else if (std::abs(PDGmass2-mass2)>EnergyMomentumRelationAllowance*EnergyMomentumRelationAllowance){
398  theDynamicalMass = std::sqrt(mass2);
399  }
400  SetKineticEnergy(totalenergy-theDynamicalMass);
401  } else {
402  SetMomentumDirection(1.0,0.0,0.0);
403  SetKineticEnergy(0.0);
