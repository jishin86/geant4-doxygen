50 G4bool WLSUserTrackInformation::AddStatusFlag(TrackStatus s)
51 // Try adding a status flag and return if it is successful or not
52 // Cannot Add Undefine or a flag that conflicts with another flag
53 // Return true if the addition of flag is successful, false otherwise
54 {
55  switch (s) {
56 
57  case left:
58  case right:
59 
60  // Allow the user to set left or right
61  // only if the track is undefined
62  if (fStatus == undefined) return fStatus |= s;
63 
64  return false;
65 
66  case EscapedFromSide:
67  case EscapedFromReadOut:
68 
69  // Allow the user to set escaped flag
70  // only if the photon hasn't exited the fiber yet
71 
72  if ((fStatus == undefined) || (fStatus & OutsideOfFiber)) return false;
73 
74  return fStatus |= s;
75 
76  case ReflectedAtMirror:
77  case ReflectedAtReadOut:
78  case murderee:
79 
80  return fStatus |= s;
81 
82  case InsideOfFiber:
83 
84  return ( fStatus =
85  (fStatus & ~(EscapedFromSide + EscapedFromReadOut + OutsideOfFiber)) | s);
86 
87  case OutsideOfFiber:
88 
89  return ( fStatus = (fStatus & ~InsideOfFiber) | s );
90 
91  default:
92
