91 G4double G4GammaConversionToMuons::GetMeanFreePath(const G4Track& aTrack,
92  G4double, G4ForceCondition*)
93 
94 // returns the photon mean free path in GEANT4 internal units
95 // (MeanFreePath is a private member of the class)
96 
97 {
98  const G4DynamicParticle* aDynamicGamma = aTrack.GetDynamicParticle();
99  G4double GammaEnergy = aDynamicGamma->GetKineticEnergy();
100  const G4Material* aMaterial = aTrack.GetMaterial();
101 
102  MeanFreePath = (GammaEnergy <= LowestEnergyLimit)
103  ? DBL_MAX : ComputeMeanFreePath(GammaEnergy,aMaterial);
104 
105  return MeanFreePath;
106 }

110 G4double
111 G4GammaConversionToMuons::ComputeMeanFreePath(G4double GammaEnergy,
112  const G4Material* aMaterial)
113 
114 // computes and returns the photon mean free path in GEANT4 internal units
115 {
116  const G4ElementVector* theElementVector = aMaterial->GetElementVector();
117  const G4double* NbOfAtomsPerVolume = aMaterial->GetVecNbOfAtomsPerVolume();
118 
119  G4double SIGMA = 0.0;
120  G4double fact = 1.0;
121  G4double e = GammaEnergy;
122  // low energy approximation as in Bethe-Heitler model
123  if(e < LimitEnergy) {
124  G4double y = (e - LowestEnergyLimit)/(LimitEnergy - LowestEnergyLimit);
125  fact = y*y;
126  e = LimitEnergy;
127  }
128 
129  for ( size_t i=0 ; i < aMaterial->GetNumberOfElements(); ++i)
130  {
131  SIGMA += NbOfAtomsPerVolume[i] * fact *
132  ComputeCrossSectionPerAtom(e, (*theElementVector)[i]->GetZasInt());
133  }
134  return (SIGMA > 0.0) ? 1./SIGMA : DBL_MAX;
135 }

151 G4double G4GammaConversionToMuons::ComputeCrossSectionPerAtom(
152  G4double Egam, G4int Z)
153 
154 // Calculates the microscopic cross section in GEANT4 internal units.
155 // Total cross section parametrisation from H.Burkhardt
156 // It gives a good description at any energy (from 0 to 10**21 eV)
157 {
158  if(Egam < LimitEnergy) return 0.0; // below threshold return 0
159 
160  G4double CrossSection = 0.0;
161  G4NistManager* nist = G4NistManager::Instance();
162 
163  G4double PowThres,Ecor,B,Dn,Zthird,Winfty,WMedAppr,
164  Wsatur,sigfac;
165 
166  if(Z==1) // special case of Hydrogen
167  { B=202.4;
168  Dn=1.49;
169  }
170  else
171  { B=183.;
172  Dn=1.54*nist->GetA27(Z);
173  }
174  Zthird=1./nist->GetZ13(Z); // Z**(-1/3)
175  Winfty=B*Zthird*Mmuon/(Dn*electron_mass_c2);
176  WMedAppr=1./(4.*Dn*sqrte*Mmuon);
177  Wsatur=Winfty/WMedAppr;
178  sigfac=4.*fine_structure_const*Z*Z*Rc*Rc;
179  PowThres=1.479+0.00799*Dn;
180  Ecor=-18.+4347./(B*Zthird);
181 
182  G4double CorFuc=1.+.04*G4Log(1.+Ecor/Egam);
183  //G4double Eg=pow(1.-4.*Mmuon/Egam,PowThres)*pow( pow(Wsatur,PowSat)+
184  // pow(Egam,PowSat),1./PowSat); // threshold and saturation
185  G4double Eg=G4Exp(G4Log(1.-4.*Mmuon/Egam)*PowThres)*
186  G4Exp(G4Log( G4Exp(G4Log(Wsatur)*PowSat)+G4Exp(G4Log(Egam)*PowSat))/PowSat);
187  CrossSection=7./9.*sigfac*G4Log(1.+WMedAppr*CorFuc*Eg);
188  CrossSection*=CrossSecFactor; // increase the CrossSection by (by default 1)
189  return CrossSection;
190 }

204 G4VParticleChange* G4GammaConversionToMuons::PostStepDoIt(
205  const G4Track& aTrack,
206  const G4Step& aStep)
207 //
208 // generation of gamma->mu+mu-
209 //
210 {
211  aParticleChange.Initialize(aTrack);
212  const G4Material* aMaterial = aTrack.GetMaterial();
213 
214  // current Gamma energy and direction, return if energy too low
215  const G4DynamicParticle *aDynamicGamma = aTrack.GetDynamicParticle();
216  G4double Egam = aDynamicGamma->GetKineticEnergy();
217  if (Egam <= LowestEnergyLimit) {
218  return G4VDiscreteProcess::PostStepDoIt(aTrack,aStep);
219  }
220  G4ParticleMomentum GammaDirection = aDynamicGamma->GetMomentumDirection();
221 
222  // select randomly one element constituting the material
223  const G4Element* anElement = SelectRandomAtom(aDynamicGamma, aMaterial);
224  G4int Z = anElement->GetZasInt();
225  G4NistManager* nist = G4NistManager::Instance();
226 
227  G4double B,Dn;
228  G4double A027 = nist->GetA27(Z);
229 
230  if(Z==1) // special case of Hydrogen
231  { B=202.4;
232  Dn=1.49;
233  }
234  else
235  { B=183.;
236  Dn=1.54*A027;
237  }
238  G4double Zthird=1./nist->GetZ13(Z); // Z**(-1/3)
239  G4double Winfty=B*Zthird*Mmuon/(Dn*electron_mass_c2);
240 
241  G4double C1Num=0.138*A027;
242  G4double C1Num2=C1Num*C1Num;
243  G4double C2Term2=electron_mass_c2/(183.*Zthird*Mmuon);
244 
245  G4double GammaMuonInv=Mmuon/Egam;
246 
247  // generate xPlus according to the differential cross section by rejection
248  G4double xmin=(Egam < LimitEnergy) ? GammaMuonInv : .5-sqrt(.25-GammaMuonInv);
249  G4double xmax=1.-xmin;
250 
251  G4double Ds2=(Dn*sqrte-2.);
252  G4double sBZ=sqrte*B*Zthird/electron_mass_c2;
253  G4double LogWmaxInv=1./G4Log(Winfty*(1.+2.*Ds2*GammaMuonInv)
254  /(1.+2.*sBZ*Mmuon*GammaMuonInv));
255  G4double xPlus,xMinus,xPM,result,W;
256  G4int nn = 0;
257  const G4int nmax = 1000;
258  do {
259  xPlus=xmin+G4UniformRand()*(xmax-xmin);
260  xMinus=1.-xPlus;
261  xPM=xPlus*xMinus;
262  G4double del=Mmuon*Mmuon/(2.*Egam*xPM);
263  W=Winfty*(1.+Ds2*del/Mmuon)/(1.+sBZ*del);
264  G4double xxp=1.-4./3.*xPM; // the main xPlus dependence
265  result=(xxp > 0.) ? xxp*G4Log(W)*LogWmaxInv : 0.0;
266  if(result>1.) {
267  G4cout << "G4GammaConversionToMuons::PostStepDoIt WARNING:"
268  << " in dSigxPlusGen, result=" << result << " > 1" << G4endl;
269  }
270  ++nn;
271  if(nn >= nmax) { break; }
272  }
273  // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
274  while (G4UniformRand() > result);
275 
276  // now generate the angular variables via the auxilary variables t,psi,rho
277  G4double t;
278  G4double psi;
279  G4double rho;
280 
281  G4double a3 = (GammaMuonInv/(2.*xPM));
282  G4double a33 = a3*a3;
283  G4double f1;
284  G4double b1 = 1./(4.*C1Num2);
285  G4double b3 = b1*b1*b1;
286  G4double a21 = a33 + b1;
287 
288  G4double f1_max=-(1.-xPM)*(2.*b1+(a21+a33)*G4Log(a33/a21))/(2*b3);
289 
290  G4double thetaPlus,thetaMinus,phiHalf; // final angular variables
291  nn = 0;
292  // t, psi, rho generation start (while angle < pi)
293  do {
294  //generate t by the rejection method
295  do {
296  ++nn;
297  t=G4UniformRand();
298  G4double a34=a33/(t*t);
299  G4double a22 = a34 + b1;
300  if(std::abs(b1)<0.0001*a34)
301  // special case of a34=a22 because of logarithm accuracy
302  {
303  f1=(1.-2.*xPM+4.*xPM*t*(1.-t))/(12.*a34*a34*a34*a34);
304  }
305  else
306  {
307  f1=-(1.-2.*xPM+4.*xPM*t*(1.-t))*(2.*b1+(a22+a34)*G4Log(a34/a22))/(2*b3);
308  }
309  if(f1<0.0 || f1> f1_max) // should never happend
310  {
311  G4cout << "G4GammaConversionToMuons::PostStepDoIt WARNING:"
312  << "outside allowed range f1=" << f1
313  << " is set to zero, a34 = "<< a34 << " a22 = "<<a22<<"."
314  << G4endl;
315  f1 = 0.0;
316  }
317  if(nn > nmax) { break; }
318  // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
319  } while ( G4UniformRand()*f1_max > f1);
320  // generate psi by the rejection method
321  G4double f2_max=1.-2.*xPM*(1.-4.*t*(1.-t));
322  // long version
323  G4double f2;
324  do {
325  ++nn;
326  psi=twopi*G4UniformRand();
327  f2=1.-2.*xPM+4.*xPM*t*(1.-t)*(1.+cos(2.*psi));
328  if(f2<0 || f2> f2_max) // should never happend
329  {
330  G4cout << "G4GammaConversionToMuons::PostStepDoIt WARNING:"
331  << "outside allowed range f2=" << f2 << " is set to zero"
332  << G4endl;
333  f2 = 0.0;
334  }
335  if(nn >= nmax) { break; }
336  // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
337  } while ( G4UniformRand()*f2_max > f2);
338 
339  // generate rho by direct transformation
340  G4double C2Term1=GammaMuonInv/(2.*xPM*t);
341  G4double C22 = C2Term1*C2Term1+C2Term2*C2Term2;
342  G4double C2=4.*C22*C22/sqrt(xPM);
343  G4double rhomax=(1./t-1.)*1.9/A027;
344  G4double beta=G4Log( (C2+rhomax*rhomax*rhomax*rhomax)/C2 );
345  rho=G4Exp(G4Log(C2 *( G4Exp(beta*G4UniformRand())-1. ))*0.25);
346 
347  //now get from t and psi the kinematical variables
348  G4double u=sqrt(1./t-1.);
349  G4double xiHalf=0.5*rho*cos(psi);
350  phiHalf=0.5*rho/u*sin(psi);
351 
352  thetaPlus =GammaMuonInv*(u+xiHalf)/xPlus;
353  thetaMinus=GammaMuonInv*(u-xiHalf)/xMinus;
354 
355  // protection against infinite loop
356  if(nn > nmax) {
357  if(std::abs(thetaPlus)>pi) { thetaPlus = 0.0; }
358  if(std::abs(thetaMinus)>pi) { thetaMinus = 0.0; }
359  }
360 
361  // Loop checking, 07-Aug-2015, Vladimir Ivanchenko
362  } while ( std::abs(thetaPlus)>pi || std::abs(thetaMinus) >pi);
363 
364  // now construct the vectors
365  // azimuthal symmetry, take phi0 at random between 0 and 2 pi
366  G4double phi0=twopi*G4UniformRand();
367  G4double EPlus=xPlus*Egam;
368  G4double EMinus=xMinus*Egam;
369 
370  // mu+ mu- directions for gamma in z-direction
371  G4ThreeVector MuPlusDirection ( sin(thetaPlus) *cos(phi0+phiHalf),
372  sin(thetaPlus) *sin(phi0+phiHalf), cos(thetaPlus) );
373  G4ThreeVector MuMinusDirection (-sin(thetaMinus)*cos(phi0-phiHalf),
374  -sin(thetaMinus) *sin(phi0-phiHalf), cos(thetaMinus) );
375  // rotate to actual gamma direction
376  MuPlusDirection.rotateUz(GammaDirection);
377  MuMinusDirection.rotateUz(GammaDirection);
378  aParticleChange.SetNumberOfSecondaries(2);
379  // create G4DynamicParticle object for the particle1
380  G4DynamicParticle* aParticle1= new G4DynamicParticle(
381  G4MuonPlus::MuonPlus(),MuPlusDirection,EPlus-Mmuon);
382  aParticleChange.AddSecondary(aParticle1);
383  // create G4DynamicParticle object for the particle2
384  G4DynamicParticle* aParticle2= new G4DynamicParticle(
385  G4MuonMinus::MuonMinus(),MuMinusDirection,EMinus-Mmuon);
386  aParticleChange.AddSecondary(aParticle2);
387  //
388  // Kill the incident photon
389  //
390  aParticleChange.ProposeMomentumDirection( 0., 0., 0. ) ;
391  aParticleChange.ProposeEnergy( 0. ) ;
392  aParticleChange.ProposeTrackStatus( fStopAndKill ) ;
393  // Reset NbOfInteractionLengthLeft and return aParticleChange
394  return G4VDiscreteProcess::PostStepDoIt( aTrack, aStep );
395 }

399 const G4Element* G4GammaConversionToMuons::SelectRandomAtom(
400  const G4DynamicParticle* aDynamicGamma,
401  const G4Material* aMaterial)
402 {
403  // select randomly 1 element within the material, invoked by PostStepDoIt
404 
405  const G4int NumberOfElements = aMaterial->GetNumberOfElements();
406  const G4ElementVector* theElementVector = aMaterial->GetElementVector();
407  const G4Element* elm = (*theElementVector)[0];
408 
409  if (NumberOfElements > 1) {
410  const G4double* NbOfAtomsPerVolume = aMaterial->GetVecNbOfAtomsPerVolume();
411 
412  G4double PartialSumSigma = 0.;
413  G4double rval = G4UniformRand()/MeanFreePath;
414 
415  for (G4int i=0; i<NumberOfElements; ++i)
416  {
417  elm = (*theElementVector)[i];
418  PartialSumSigma += NbOfAtomsPerVolume[i]
419  *GetCrossSectionPerAtom(aDynamicGamma, elm);
420  if (rval <= PartialSumSigma) { break; }
421  }
422  }
423  return elm;
424 }
