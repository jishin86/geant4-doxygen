72 void PrimaryGeneratorAction4::GeneratePrimaries(G4Event* anEvent)
73 {
74  //vertex position uniform in spherical shell
75  //
76  G4double cosTheta = 2*G4UniformRand() - 1; //cosTheta uniform in [0, pi]
77  G4double sinTheta = std::sqrt(1. - cosTheta*cosTheta);
78  G4double phi = twopi*G4UniformRand(); //phi uniform in [0, 2*pi]
79  G4ThreeVector ur(sinTheta*std::cos(phi),sinTheta*std::sin(phi),cosTheta);
80 
81  G4double R3 = fRmin3 + G4UniformRand()*(fRmax3 - fRmin3);
82  G4double R = std::pow(R3, 1./3);
83 
84  fParticleGun->SetParticlePosition(R*ur);
85 
86  //particle direction uniform around ur
87  //
88  //1- in World frame
89  //cosAlpha uniform in [cos(alphaMin), cos(alphaMax)]
90  G4double cosAlpha = fCosAlphaMin-G4UniformRand()*(fCosAlphaMin-fCosAlphaMax);
91  G4double sinAlpha = std::sqrt(1. - cosAlpha*cosAlpha);
92  G4double psi = twopi*G4UniformRand(); //psi uniform in (0,2*pi)
93  G4ThreeVector dir(sinAlpha*std::cos(psi),sinAlpha*std::sin(psi),cosAlpha);
94 
95  //2- rotate dir (rotateUz transforms uz to ur)
96  dir.rotateUz(ur);
97 
98  fParticleGun->SetParticleMomentumDirection(dir);
99 
100  //energy
101  //
102  fParticleGun->SetParticleEnergy(1*MeV);
103 
104  //create vertex
105  //
106  fParticleGun->GeneratePrimaryVertex(anEvent);
107 }
