98 void G4MonopolePhysics::ConstructProcess()
99 {
100  if(verboseLevel > 0) {
101  G4cout << "G4MonopolePhysics::ConstructProcess" << G4endl;
102  }
103 
104  G4PhysicsListHelper* ph = G4PhysicsListHelper::GetPhysicsListHelper();
105  G4ProcessManager* pmanager = fMpl->GetProcessManager();
106 
107  // defined monopole parameters and binning
108  G4double magn = fMpl->MagneticCharge();
109  G4double emin = std::min(fMonopoleMass/20000., CLHEP::keV);
110  G4double emax = std::max(10.*TeV, fMonopoleMass*100);
111  G4int nbin = G4lrint(10*std::log10(emax/emin));
112 
113  // dedicated trasporation
114  if(magn != 0.0) {
115  G4int idxt(0);
116  pmanager->RemoveProcess(idxt);
117  pmanager->AddProcess(new G4MonopoleTransportation(fMpl),-1, 0, 0);
118  }
119 
120  if(fMpl->GetPDGCharge() != 0.0) {
121  G4hIonisation* hhioni = new G4hIonisation();
122  hhioni->SetDEDXBinning(nbin);
123  hhioni->SetMinKinEnergy(emin);
124  hhioni->SetMaxKinEnergy(emax);
125  ph->RegisterProcess(hhioni, fMpl);
126  }
127  if(magn != 0.0) {
128  G4mplIonisation* mplioni = new G4mplIonisation(magn);
129  mplioni->SetDEDXBinning(nbin);
130  mplioni->SetMinKinEnergy(emin);
131  mplioni->SetMaxKinEnergy(emax);
132  ph->RegisterProcess(mplioni, fMpl);
133  }
134  ph->RegisterProcess(new G4StepLimiter(), fMpl);
135 }
