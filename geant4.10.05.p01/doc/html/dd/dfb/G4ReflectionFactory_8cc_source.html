29 // Class G4ReflectionFactory Implementation
30 //
31 // Decomposition of a general transformation
32 // that can include reflection in a "reflection-free" transformation:
33 //
34 // x(inM') = TG*x(inM) TG - general transformation
35 // = T*(R*x(inM)) T - "reflection-free" transformation
36 // = T* x(inReflM)
37 //
38 // Daughters transformation:
39 // When a volume V containing daughter D with transformation TD
40 // is placed in mother M with a general tranformation TGV,
41 // the TGV is decomposed,
42 // new reflected volume ReflV containing a new daughter ReflD
43 // with reflected transformation ReflTD is created:
44 //
45 // x(inV) = TD * x(inD);
46 // x(inM) = TGV * x(inV)
47 // = TV * R * x(inV)
48 // = TV * R * TD * x(inD)
49 // = TV * R*TD*R-1 * R*x(inD)
50 // = TV * ReflTD * x(inReflD)
51 
52 // Author: Ivana Hrivnacova, 16.10.2001 (Ivana.Hrivnacova@cern.ch)
53 // --------------------------------------------------------------------

107 G4PhysicalVolumesPair
108 G4ReflectionFactory::Place( const G4Transform3D& transform3D,
109  const G4String& name,
110  G4LogicalVolume* LV,
111  G4LogicalVolume* motherLV,
112  G4bool isMany,
113  G4int copyNo,
114  G4bool surfCheck)
115 {
116  // Evaluates the passed transformation; if it contains reflection
117  // it performs its decomposition, creates new reflected solid and
118  // logical volume (or retrieves them from a map if the reflected
119  // objects were already created), transforms the daughters (if present)
120  // and place it in the given mother.
121  // The result is a pair of physical volumes;
122  // the second physical volume is a placement in a reflected mother
123  // - or 0 if mother LV was not reflected.
124  // ---
125 
126  if (fVerboseLevel>0)
127  {
128  G4cout << "Place " << name << " lv " << LV << " "
129  << LV->GetName() << G4endl;
130  }
131 
132  // decompose transformation
133  G4Scale3D scale;
134  G4Rotate3D rotation;
135  G4Translate3D translation;
136 
137  transform3D.getDecomposition(scale, rotation, translation);
138  G4Transform3D pureTransform3D = translation * rotation;
139 
140  //PrintTransform(transform3D);
141  //PrintTransform(pureTransform3D);
142 
143  // check that scale correspond to fScale
144  //
145  CheckScale(scale);
146 
147  //
148  // reflection IS NOT present in transform3D
149  //
150 
151  if (! IsReflection(scale))
152  {
153  if (fVerboseLevel>0)
154  G4cout << "Scale positive" << G4endl;
155 
156  G4VPhysicalVolume* pv1
157  = new G4PVPlacement(pureTransform3D, LV, name,
158  motherLV, isMany, copyNo, surfCheck);
159 
160  G4VPhysicalVolume* pv2 = 0;
161  if (G4LogicalVolume* reflMotherLV = GetReflectedLV(motherLV))
162  {
163  // if mother was reflected
164  // reflect this LV and place it in reflected mother
165 
166  pv2 = new G4PVPlacement(fScale * (pureTransform3D * fScale.inverse()),
167  ReflectLV(LV, surfCheck), name, reflMotherLV,
168  isMany, copyNo, surfCheck);
169  }
170 
171  return G4PhysicalVolumesPair(pv1, pv2);
172  }
173 
174  //
175  // reflection IS present in transform3D
176  //
177 
178  if (fVerboseLevel>0)
179  G4cout << "scale negative" << G4endl;
180 
181  G4VPhysicalVolume* pv1
182  = new G4PVPlacement(pureTransform3D, ReflectLV(LV, surfCheck), name,
183  motherLV, isMany, copyNo, surfCheck);
184 
185  G4VPhysicalVolume* pv2 = 0;
186  if (G4LogicalVolume* reflMotherLV = GetReflectedLV(motherLV))
187  {
188 
189  // if mother was reflected
190  // place the refLV consituent in reflected mother
191 
192  pv2 = new G4PVPlacement(fScale * (pureTransform3D * fScale.inverse()),
193  LV, name, reflMotherLV, isMany, copyNo, surfCheck);
194  }
195 
196  return G4PhysicalVolumesPair(pv1, pv2);
197 }

369 G4LogicalVolume* G4ReflectionFactory::ReflectLV(G4LogicalVolume* LV,
370  G4bool surfCheck)
371 {
372  // Gets/creates the reflected solid and logical volume
373  // and copies + transforms LV daughters.
374  // ---
375 
376  G4LogicalVolume* refLV = GetReflectedLV(LV);
377 
378  if (!refLV)
379  {
380 
381  // create new (reflected) objects
382  //
383  refLV = CreateReflectedLV(LV);
384 
385  // process daughters
386  //
387  ReflectDaughters(LV, refLV, surfCheck);
388 
389  // check if to be set as root region
390  //
391  if (LV->IsRootRegion())
392  {
393  LV->GetRegion()->AddRootLogicalVolume(refLV);
394  }
395  }
396 
397  return refLV;
398 }

402 G4LogicalVolume* G4ReflectionFactory::CreateReflectedLV(G4LogicalVolume* LV)
403 {
404  // Creates the reflected solid and logical volume
405  // and add the logical volumes pair in the maps.
406  // ---
407 
408  // consistency check
409  //
410  if (fReflectedLVMap.find(LV) != fReflectedLVMap.end())
411  {
412  std::ostringstream message;
413  message << "Invalid reflection for volume: "
414  << LV->GetName() << G4endl
415  << "Cannot be applied to a volume already reflected !";
416  G4Exception("G4ReflectionFactory::CreateReflectedLV()",
417  "GeomVol0002", FatalException, message);
418  }
419 
420  G4VSolid* refSolid
421  = new G4ReflectedSolid(LV->GetSolid()->GetName() + fNameExtension,
422  LV->GetSolid(), fScale);
423 
424  G4LogicalVolume* refLV
425  = new G4LogicalVolume(refSolid,
426  LV->GetMaterial(),
427  LV->GetName() + fNameExtension,
428  LV->GetFieldManager(),
429  LV->GetSensitiveDetector(),
430  LV->GetUserLimits());
431  refLV->SetVisAttributes(LV->GetVisAttributes()); // vis-attributes
432  refLV->SetBiasWeight(LV->GetBiasWeight()); // biasing weight
433  if (LV->IsRegion())
434  {
435  refLV->SetRegion(LV->GetRegion()); // set a region in case
436  }
437 
438  fConstituentLVMap[LV] = refLV;
439  fReflectedLVMap[refLV] = LV;
440 
441  return refLV;
442 }

446 void G4ReflectionFactory::ReflectDaughters(G4LogicalVolume* LV,
447  G4LogicalVolume* refLV,
448  G4bool surfCheck)
449 {
450  // Reflects daughters recursively.
451  // ---
452 
453  if (fVerboseLevel>0)
454  {
455  G4cout << "G4ReflectionFactory::ReflectDaughters(): "
456  << LV->GetNoDaughters() << " of " << LV->GetName() << G4endl;
457  }
458 
459  for (G4int i=0; i<LV->GetNoDaughters(); i++)
460  {
461  G4VPhysicalVolume* dPV = LV->GetDaughter(i);
462 
463  if (! dPV->IsReplicated())
464  {
465  ReflectPVPlacement(dPV, refLV, surfCheck);
466  }
467  else if (! dPV->GetParameterisation())
468  {
469  ReflectPVReplica(dPV, refLV);
470  }
471  else if (G4VPVDivisionFactory::Instance() &&
472  G4VPVDivisionFactory::Instance()->IsPVDivision(dPV))
473  {
474  ReflectPVDivision(dPV, refLV);
475  }
476  else
477  {
478  ReflectPVParameterised(dPV, refLV, surfCheck);
479  }
480  }
481 }

485 void G4ReflectionFactory::ReflectPVPlacement(G4VPhysicalVolume* dPV,
486  G4LogicalVolume* refLV,
487  G4bool surfCheck)
488 {
489  // Copies and transforms daughter of PVPlacement type of
490  // a constituent volume into a reflected volume.
491  // ---
492 
493  G4LogicalVolume* dLV = dPV->GetLogicalVolume();
494 
495  // update daughter transformation
496  //
497  G4Transform3D dt(dPV->GetObjectRotationValue(), dPV->GetObjectTranslation());
498  dt = fScale * (dt * fScale.inverse());
499 
500  G4LogicalVolume* refDLV;
501 
502  if (fVerboseLevel>0)
503  G4cout << "Daughter: " << dPV << " " << dLV->GetName();
504 
505  if (!IsReflected(dLV))
506  {
507 
508  if (fVerboseLevel>0)
509  G4cout << " will be reflected." << G4endl;
510 
511  // get reflected volume if already created //
512  refDLV = GetReflectedLV(dLV);
513 
514  if (!refDLV)
515  {
516  // create new daughter solid and logical volume
517  //
518  refDLV = CreateReflectedLV(dLV);
519 
520  // recursive call
521  //
522  ReflectDaughters(dLV, refDLV, surfCheck);
523  }
524 
525  // create new daughter physical volume
526  // with updated transformation
527 
528  new G4PVPlacement(dt, refDLV, dPV->GetName(), refLV,
529  dPV->IsMany(), dPV->GetCopyNo(), surfCheck);
530 
531  }
532  else
533  {
534  if (fVerboseLevel>0)
535  G4cout << " will be reconstitued." << G4endl;
536 
537  refDLV = GetConstituentLV(dLV);
538 
539  new G4PVPlacement(dt, refDLV, dPV->GetName(), refLV,
540  dPV->IsMany(), dPV->GetCopyNo(), surfCheck);
541  }
542 }
