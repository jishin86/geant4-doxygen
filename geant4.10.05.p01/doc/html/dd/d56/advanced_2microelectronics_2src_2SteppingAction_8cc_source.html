55 void SteppingAction::UserSteppingAction(const G4Step* step)
56 {
57  G4double flagParticle=0.;
58  G4double flagProcess=0.;
59  G4double x,y,z,xp,yp,zp;
60 
61  if (step->GetTrack()->GetDynamicParticle()->GetDefinition() ->GetParticleName() == "e-") flagParticle = 1;
62  if (step->GetTrack()->GetDynamicParticle()->GetDefinition() ->GetParticleName() == "proton") flagParticle = 2;
63  if (step->GetTrack()->GetDynamicParticle()->GetDefinition() ->GetParticleName() == "GenericIon") flagParticle = 3;
64 
65  if (step->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()=="msc") flagProcess =10;
66  if (step->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()=="e-_G4MicroElecElastic") flagProcess =11;
67  if (step->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()=="e-_G4MicroElecInelastic") flagProcess =12;
68  if (step->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()=="eCapture") flagProcess =13;
69 
70  if (step->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()=="p_G4MicroElecInelastic") flagProcess =14;
71 
72  if (step->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()=="ion_G4MicroElecInelastic") flagProcess =15;
73 
74  if (step->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()=="hIoni") flagProcess =16;
75  if (step->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()=="eIoni") flagProcess =17;
76 
77  if (step->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()!="Transportation")
78  {
79  x=step->GetPreStepPoint()->GetPosition().x()/nanometer;
80  y=step->GetPreStepPoint()->GetPosition().y()/nanometer;
81  z=step->GetPreStepPoint()->GetPosition().z()/nanometer;
82  xp=step->GetPostStepPoint()->GetPosition().x()/nanometer;
83  yp=step->GetPostStepPoint()->GetPosition().y()/nanometer;
84  zp=step->GetPostStepPoint()->GetPosition().z()/nanometer;
85 
86  // get analysis manager
87  G4AnalysisManager* analysisManager = G4AnalysisManager::Instance();
88 
89  // fill ntuple
90  analysisManager->FillNtupleDColumn(0, flagParticle);
91  analysisManager->FillNtupleDColumn(1, flagProcess);
92  analysisManager->FillNtupleDColumn(2, x);
93  analysisManager->FillNtupleDColumn(3, y);
94  analysisManager->FillNtupleDColumn(4, z);
95  analysisManager->FillNtupleDColumn(5, step->GetTotalEnergyDeposit()/eV);
96  analysisManager->FillNtupleDColumn(6, std::sqrt((x-xp)*(x-xp)+(y-yp)*(y-yp)+(z-zp)*(z-zp))/nm);
97  analysisManager->FillNtupleDColumn(7, (step->GetPreStepPoint()->GetKineticEnergy() - step->GetPostStepPoint()->GetKineticEnergy())/eV );
98  analysisManager->AddNtupleRow();
99  }
100 }
