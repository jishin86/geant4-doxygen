127  G4int G4ExcitedSigmaConstructor::GetQuarkContents(G4int iQ, G4int iIso3)
128 {
129  G4int quark=0;
130  if ( iQ == 0 ){
131  // s-quark
132  quark = 3;
133  } else if ( iQ == 1 ){
134  if (iIso3 == -2) {
135  // d-quark
136  quark = 1;
137  } else {
138  // u-quark
139  quark = 2;
140  }
141  } else if ( iQ == 2 ){
142  if (iIso3 == +2) {
143  // u-quark
144  quark = 2;
145  } else {
146  // d-quark
147  quark = 1;
148  }
149  }
150  return quark;
151 }
