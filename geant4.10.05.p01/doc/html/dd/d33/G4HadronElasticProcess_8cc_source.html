69 G4VParticleChange*
70 G4HadronElasticProcess::PostStepDoIt(const G4Track& track,
71  const G4Step& /*step*/)
72 {
73  theTotalResult->Clear();
74  theTotalResult->Initialize(track);
75  G4double weight = track.GetWeight();
76  theTotalResult->ProposeWeight(weight);
77 
78  // For elastic scattering, _any_ result is considered an interaction
79  ClearNumberOfInteractionLengthLeft();
80 
81  G4double kineticEnergy = track.GetKineticEnergy();
82  const G4DynamicParticle* dynParticle = track.GetDynamicParticle();
83  const G4ParticleDefinition* part = dynParticle->GetDefinition();
84 
85  // NOTE: Very low energy scatters were causing numerical (FPE) errors
86  // in earlier releases; these limits have not been changed since.
87  if (kineticEnergy <= lowestEnergy) return theTotalResult;
88 
89  const G4Material* material = track.GetMaterial();
90  G4Nucleus* targNucleus = GetTargetNucleusPointer();
91 
92  // Select element
93  const G4Element* elm = nullptr;
94  try
95  {
96  elm = GetCrossSectionDataStore()->SampleZandA(dynParticle, material,
97  *targNucleus);
98  }
99  catch(G4HadronicException & aR)
100  {
101  G4ExceptionDescription ed;
102  aR.Report(ed);
103  DumpState(track,"SampleZandA",ed);
104  ed << " PostStepDoIt failed on element selection" << G4endl;
105  G4Exception("G4HadronElasticProcess::PostStepDoIt", "had003",
106  FatalException, ed);
107  }
108 
109  // Initialize the hadronic projectile from the track
110  // G4cout << "track " << track.GetDynamicParticle()->Get4Momentum()<<G4endl;
111  G4HadProjectile theProj(track);
112  G4HadronicInteraction* hadi = 0;
113  G4HadFinalState* result = 0;
114 
115  if(fDiffraction)
116  {
117  G4double ratio =
118  fDiffractionRatio->ComputeRatio(part, kineticEnergy,
119  targNucleus->GetZ_asInt(),
120  targNucleus->GetA_asInt());
121  // diffraction is chosen
122  if(ratio > 0.0 && G4UniformRand() < ratio)
123  {
124  try
125  {
126 // if(part->GetParticleName() == "pi-")
127 // G4cout<<part->GetParticleName()<<"; "<<kineticEnergy/CLHEP::GeV<<" GeV; r = "<<ratio<<G4endl;
128 
129  result = fDiffraction->ApplyYourself(theProj, *targNucleus);
130  }
131  catch(G4HadronicException & aR)
132  {
133  G4ExceptionDescription ed;
134  aR.Report(ed);
135  ed << "Call for " << fDiffraction->GetModelName() << G4endl;
136  ed << "Target element "<< elm->GetName()<<" Z= "
137  << targNucleus->GetZ_asInt()
138  << " A= " << targNucleus->GetA_asInt() << G4endl;
139  DumpState(track,"ApplyYourself",ed);
140  ed << " ApplyYourself failed" << G4endl;
141  G4Exception("G4HadronElasticProcess::PostStepDoIt", "had006",
142  FatalException, ed);
143  }
144  // Check the result for catastrophic energy non-conservation
145  result = CheckResult(theProj, *targNucleus, result);
146 
147  result->SetTrafoToLab(theProj.GetTrafoToLab());
148  ClearNumberOfInteractionLengthLeft();
149 
150  FillResult(result, track);
151 
152  if (epReportLevel != 0) {
153  CheckEnergyMomentumConservation(track, *targNucleus);
154  }
155  return theTotalResult;
156  }
157  }
158 
159  // ordinary elastic scattering
160  try
161  {
162  hadi = ChooseHadronicInteraction( theProj, *targNucleus, material, elm );
163  }
164  catch(G4HadronicException & aE)
165  {
166  G4ExceptionDescription ed;
167  aE.Report(ed);
168  ed << "Target element "<< elm->GetName()<<" Z= "
169  << targNucleus->GetZ_asInt() << " A= "
170  << targNucleus->GetA_asInt() << G4endl;
171  DumpState(track,"ChooseHadronicInteraction",ed);
172  ed << " No HadronicInteraction found out" << G4endl;
173  G4Exception("G4HadronElasticProcess::PostStepDoIt", "had005",
174  FatalException, ed);
175  }
176 
177  size_t idx = track.GetMaterialCutsCouple()->GetIndex();
178  G4double tcut = (*(G4ProductionCutsTable::GetProductionCutsTable()
179  ->GetEnergyCutsVector(3)))[idx];
180  hadi->SetRecoilEnergyThreshold(tcut);
181 
182  if(verboseLevel>1) {
183  G4cout << "G4HadronElasticProcess::PostStepDoIt for "
184  << part->GetParticleName()
185  << " in " << material->GetName()
186  << " Target Z= " << targNucleus->GetZ_asInt()
187  << " A= " << targNucleus->GetA_asInt() << G4endl;
188  }
189 
190  try
191  {
192  result = hadi->ApplyYourself( theProj, *targNucleus);
193  }
194  catch(G4HadronicException & aR)
195  {
196  G4ExceptionDescription ed;
197  aR.Report(ed);
198  ed << "Call for " << hadi->GetModelName() << G4endl;
199  ed << "Target element "<< elm->GetName()<<" Z= "
200  << targNucleus->GetZ_asInt()
201  << " A= " << targNucleus->GetA_asInt() << G4endl;
202  DumpState(track,"ApplyYourself",ed);
203  ed << " ApplyYourself failed" << G4endl;
204  G4Exception("G4HadronElasticProcess::PostStepDoIt", "had006",
205  FatalException, ed);
206  }
207 
208  // Check the result for catastrophic energy non-conservation
209  // cannot be applied because is not guranteed that recoil
210  // nucleus is created
211  // result = CheckResult(theProj, targNucleus, result);
212 
213  // directions
214  G4ThreeVector indir = track.GetMomentumDirection();
215  G4double phi = CLHEP::twopi*G4UniformRand();
216  G4ThreeVector it(0., 0., 1.);
217  G4ThreeVector outdir = result->GetMomentumChange();
218 
219  if(verboseLevel>1) {
220  G4cout << "Efin= " << result->GetEnergyChange()
221  << " de= " << result->GetLocalEnergyDeposit()
222  << " nsec= " << result->GetNumberOfSecondaries()
223  << " dir= " << outdir
224  << G4endl;
225  }
226 
227  // energies
228  G4double edep = result->GetLocalEnergyDeposit();
229  G4double efinal = result->GetEnergyChange();
230  if(efinal < 0.0) { efinal = 0.0; }
231  if(edep < 0.0) { edep = 0.0; }
232 
233  // NOTE: Very low energy scatters were causing numerical (FPE) errors
234  // in earlier releases; these limits have not been changed since.
235  if(efinal <= lowestEnergy) {
236  edep += efinal;
237  efinal = 0.0;
238  }
239 
240  // primary change
241  theTotalResult->ProposeEnergy(efinal);
242 
243  G4TrackStatus status = track.GetTrackStatus();
244  if(efinal > 0.0) {
245  outdir.rotate(phi, it);
246  outdir.rotateUz(indir);
247  theTotalResult->ProposeMomentumDirection(outdir);
248  } else {
249  if(part->GetProcessManager()->GetAtRestProcessVector()->size() > 0)
250  { status = fStopButAlive; }
251  else { status = fStopAndKill; }
252  theTotalResult->ProposeTrackStatus(status);
253  }
254 
255  //G4cout << "Efinal= " << efinal << " TrackStatus= " << status << G4endl;
256 
257  theTotalResult->SetNumberOfSecondaries(0);
258 
259  // recoil
260  if(result->GetNumberOfSecondaries() > 0) {
261  G4DynamicParticle* p = result->GetSecondary(0)->GetParticle();
262 
263  if(p->GetKineticEnergy() > tcut) {
264  theTotalResult->SetNumberOfSecondaries(1);
265  G4ThreeVector pdir = p->GetMomentumDirection();
266  // G4cout << "recoil " << pdir << G4endl;
268  pdir.rotate(phi, it);
269  pdir.rotateUz(indir);
270  // G4cout << "recoil rotated " << pdir << G4endl;
271  p->SetMomentumDirection(pdir);
272 
273  // in elastic scattering time and weight are not changed
274  G4Track* t = new G4Track(p, track.GetGlobalTime(),
275  track.GetPosition());
276  t->SetWeight(weight);
277  t->SetTouchableHandle(track.GetTouchableHandle());
278  theTotalResult->AddSecondary(t);
279 
280  } else {
281  edep += p->GetKineticEnergy();
282  delete p;
283  }
284  }
285  theTotalResult->ProposeLocalEnergyDeposit(edep);
286  theTotalResult->ProposeNonIonizingEnergyDeposit(edep);
287  result->Clear();
288 
289  return theTotalResult;
290 }
