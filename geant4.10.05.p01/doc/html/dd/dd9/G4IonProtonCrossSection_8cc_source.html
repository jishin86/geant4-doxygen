64 G4double
65 G4IonProtonCrossSection::GetElementCrossSection(
66  const G4DynamicParticle* dp,
67  G4int, const G4Material*)
68 {
69  const G4ParticleDefinition* p = dp->GetDefinition();
70  G4double e = dp->GetKineticEnergy()*CLHEP::proton_mass_c2/p->GetPDGMass();
71  G4int Z = p->GetAtomicNumber();
72  G4int A = p->GetAtomicMass();
73  return theForward->IsoCrossSection(e, Z, A);
74 }
