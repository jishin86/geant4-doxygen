35 // Given that it is an experimental, and perhaps temporary, new type of
36 // hadron physics, corresponding builders are not created and everything
37 // is implemented directly in this class.

178 void G4HadronPhysicsFTFQGSP_BERT::CreateModels()
179 {
180 
181  G4double minFTFP = 3.0 * GeV;
182  G4double maxBERT = 12.0 * GeV;
183  G4cout << " FTFQGSP_BERT : similar to FTFP_BERT but with" << G4endl
184  << " QGS string fragmentation (instead of Lund string fragmentation)." << G4endl;
185 
186  theStringModel = new G4FTFModel;
187 
188  theStringDecay = new G4ExcitedStringDecay( theQGSMFragmentation = new G4QGSMFragmentation );
189  theStringModel->SetFragmentationModel( theStringDecay );
190  thePreEquilib = new G4PreCompoundModel( theHandler = new G4ExcitationHandler );
191  theCascade = new G4GeneratorPrecompoundInterface( thePreEquilib );
192 
193  // FTF for neutrons, protons, pions, and kaons
194  theModel1 = new G4TheoFSGenerator( "FTFP" );
195  theModel1->SetMinEnergy( minFTFP );
196  theModel1->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
197  theModel1->SetTransport( theCascade );
198  theModel1->SetHighEnergyGenerator( theStringModel );
199 
200  // BERT for neutrons, protons, pions, and kaons
201  theBertini1 = new G4CascadeInterface;
202  theBertini1->SetMinEnergy( 0.0*GeV );
203  theBertini1->SetMaxEnergy( maxBERT );
204 
205  // FTF for hyperons
206  theModel2 = new G4TheoFSGenerator( "FTFP" );
207  theModel2->SetMinEnergy( 2.0*GeV );
208  theModel2->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
209  theModel2->SetTransport( theCascade );
210  theModel2->SetHighEnergyGenerator( theStringModel );
211 
212  // BERT for hyperons
213  theBertini2 = new G4CascadeInterface;
214  theBertini2->SetMinEnergy( 0.0*GeV );
215  theBertini2->SetMaxEnergy( 6.0*GeV );
216 
217  // FTF for Antibaryons
218  theModel3 = new G4TheoFSGenerator( "FTFP" );
219  theModel3->SetMinEnergy( 0.0*GeV );
220  theModel3->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
221  theModel3->SetTransport( theCascade );
222  theModel3->SetHighEnergyGenerator( theStringModel );
223 
224  // Neutron Capture
225  theNeutronCaptureModel = new G4NeutronRadCapture;
226  theNeutronCaptureModel->SetMinEnergy( 0.0 );
227  theNeutronCaptureModel->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
228 
229  // Cross sections
230  thePiXS = new G4BGGPionInelasticXS( G4PionMinus::Definition() );
231  theAntiNucleonXS = new G4CrossSectionInelastic( new G4ComponentAntiNuclNuclearXS );
232  theKaonXS = new G4CrossSectionInelastic( new G4ComponentGGHadronNucleusXsc );
233  theChipsHyperonInelasticXS = G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet( G4ChipsHyperonInelasticXS::Default_Name() );
234  theNeutronInelasticXS = G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet( G4NeutronInelasticXS::Default_Name() );
235  theNeutronCaptureXS = G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet( G4NeutronCaptureXS::Default_Name() );
236 }
