146 // Get bounding box
147 
148 void G4UParaboloid::BoundingLimits(G4ThreeVector& pMin,
149  G4ThreeVector& pMax) const
150 {
151  static G4bool checkBBox = true;
152 
153  G4double r2 = GetRadiusPlusZ();
154  G4double dz = GetZHalfLength();
155  pMin.set(-r2,-r2,-dz);
156  pMax.set( r2, r2, dz);
157 
158  // Check correctness of the bounding box
159  //
160  if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
161  {
162  std::ostringstream message;
163  message << "Bad bounding box (min >= max) for solid: "
164  << GetName() << " !"
165  << "\npMin = " << pMin
166  << "\npMax = " << pMax;
167  G4Exception("G4UParaboloid::BoundingLimits()", "GeomMgt0001",
168  JustWarning, message);
169  StreamInfo(G4cout);
170  }
171 
172  // Check consistency of bounding boxes
173  //
174  if (checkBBox)
175  {
176  U3Vector vmin, vmax;
177  Extent(vmin,vmax);
178  if (std::abs(pMin.x()-vmin.x()) > kCarTolerance ||
179  std::abs(pMin.y()-vmin.y()) > kCarTolerance ||
180  std::abs(pMin.z()-vmin.z()) > kCarTolerance ||
181  std::abs(pMax.x()-vmax.x()) > kCarTolerance ||
182  std::abs(pMax.y()-vmax.y()) > kCarTolerance ||
183  std::abs(pMax.z()-vmax.z()) > kCarTolerance)
184  {
185  std::ostringstream message;
186  message << "Inconsistency in bounding boxes for solid: "
187  << GetName() << " !"
188  << "\nBBox min: wrapper = " << pMin << " solid = " << vmin
189  << "\nBBox max: wrapper = " << pMax << " solid = " << vmax;
190  G4Exception("G4UParaboloid::BoundingLimits()", "GeomMgt0001",
191  JustWarning, message);
192  checkBBox = false;
193  }
194  }
195 }
