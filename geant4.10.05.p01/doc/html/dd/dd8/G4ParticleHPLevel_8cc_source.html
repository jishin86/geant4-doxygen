67  {

69  G4double sum = 0;
70  G4double * running = new G4double[nGammas];
71  running[0] = 0;
72  G4int i;

74  {
75  if(i!=0) running[i]=running[i-1];
76  running[i]+=theGammas[i]->GetWeight();
77  }
78  sum = running[nGammas-1];
79  G4int it(0);

82  {

85  }

87  theResult = theGammas[it]->GetDecayGammas();
