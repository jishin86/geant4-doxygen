32 G4double G4LineSection::Dist( G4ThreeVector OtherPnt ) const
33 {
34  G4double dist_sq;
35  G4ThreeVector VecAZ;
36  G4double sq_VecAZ, inner_prod, unit_projection ;
37 
38  VecAZ= OtherPnt - EndpointA;
39  sq_VecAZ = VecAZ.mag2();
40 
41  inner_prod= VecAtoB.dot( VecAZ );
42 
43  // Determine Projection(AZ on AB) / Length(AB)
44  //
45  if( fABdistanceSq != 0.0 )
46  {
47  // unit_projection= inner_prod * InvsqDistAB();
48  unit_projection = inner_prod/fABdistanceSq;
49 
50  if( (0. <= unit_projection ) && (unit_projection <= 1.0 ) )
51  {
52  dist_sq= sq_VecAZ - unit_projection * inner_prod ;
53  }
54  else
55  {
56  // The perpendicular from the point to the line AB meets the line
57  // in a point outside the line segment!
58 
59  if( unit_projection < 0. ) // A is the closest point
60  {
61  dist_sq= sq_VecAZ;
62  }
63  else // B is the closest point
64  {
65  G4ThreeVector EndpointB = EndpointA + VecAtoB;
66  G4ThreeVector VecBZ = OtherPnt - EndpointB;
67  dist_sq = VecBZ.mag2();
68  }
69  }
70  }
71  else
72  {
73  dist_sq = (OtherPnt - EndpointA).mag2() ;
74  }
75  if( dist_sq < 0.0 ) dist_sq = 0.0 ;
76 
77  return std::sqrt(dist_sq) ;
78 }
