126  G4EmParameters::Instance()->SetAugerCascade(ARMflag);
127  deex = new G4UAtomicDeexcitation();
128  deex->InitialiseAtomicDeexcitation();

130  }

135  ph->RegisterProcess(radioactiveDecay, G4GenericIon::GenericIon());

137  //printout
