168 void
169 G4VMultipleScattering::PreparePhysicsTable(const G4ParticleDefinition& part)
170 {
171  if(1 < verboseLevel) {
172  G4cout << "### G4VMultipleScattering::PrepearPhysicsTable() for "
173  << GetProcessName()
174  << " and particle " << part.GetParticleName()
175  << G4endl;
176  }
177  G4bool master = emManager->IsMaster();
178 
179  if(!firstParticle) { firstParticle = &part; }
180  if(part.GetParticleType() == "nucleus") {
181  stepLimit = fMinimal;
182  latDisplacement = false;
183  facrange = 0.2;
184  G4String pname = part.GetParticleName();
185  if(pname != "deuteron" && pname != "triton" &&
186  pname != "alpha+" && pname != "helium" &&
187  pname != "alpha" && pname != "He3" &&
188  pname != "hydrogen") {
189 
190  const G4ParticleDefinition* theGenericIon =
191  G4ParticleTable::GetParticleTable()->FindParticle("GenericIon");
192  if(&part == theGenericIon) { isIon = true; }
193 
194  if(theGenericIon && firstParticle != theGenericIon) {
195  G4ProcessManager* pm = theGenericIon->GetProcessManager();
196  G4ProcessVector* v = pm->GetAlongStepProcessVector();
197  size_t n = v->size();
198  for(size_t j=0; j<n; ++j) {
199  if((*v)[j] == this) {
200  firstParticle = theGenericIon;
201  isIon = true;
202  break;
203  }
204  }
205  }
206  }
207  }
208 
209  emManager->PreparePhysicsTable(&part, this, master);
210  currParticle = nullptr;
211 
212  if(1 < verboseLevel) {
213  G4cout << "### G4VMultipleScattering::PrepearPhysicsTable() for "
214  << GetProcessName()
215  << " and particle " << part.GetParticleName()
216  << " local particle " << firstParticle->GetParticleName()
217  << " isIon: " << isIon << " isMaster: " << master
218  << G4endl;
219  }
220 
221  if(firstParticle == &part) {
222 
223  // initialise process
224  InitialiseProcess(firstParticle);
225 
226  // heavy particles and not ions
227  if(!isIon) {
228  if(part.GetPDGMass() > MeV) {
229  stepLimit = theParameters->MscMuHadStepLimitType();
230  facrange = theParameters->MscMuHadRangeFactor();
231  latDisplacement = theParameters->MuHadLateralDisplacement();
232  } else {
233  stepLimit = theParameters->MscStepLimitType();
234  facrange = theParameters->MscRangeFactor();
235  latDisplacement = theParameters->LateralDisplacement();
236  }
237  if(latDisplacement) {
238  fDispBeyondSafety = theParameters->LatDisplacementBeyondSafety();
239  }
240  }
241  if(master) { SetVerboseLevel(theParameters->Verbose()); }
242  else { SetVerboseLevel(theParameters->WorkerVerbose()); }
243 
244  // initialisation of models
245  numberOfModels = modelManager->NumberOfModels();
246  /*
247  G4cout << "### G4VMultipleScattering::PreparePhysicsTable() for "
248  << GetProcessName()
249  << " and particle " << part.GetParticleName()
250  << " Nmod= " << numberOfModels << " " << this
251  << G4endl;
252  */
253  for(G4int i=0; i<numberOfModels; ++i) {
254  G4VMscModel* msc = static_cast<G4VMscModel*>(GetModelByIndex(i));
255  if(!msc) { continue; }
256  msc->SetIonisation(nullptr, firstParticle);
257  msc->SetMasterThread(master);
258  currentModel = msc;
259  msc->SetStepLimitType(stepLimit);
260  msc->SetLateralDisplasmentFlag(latDisplacement);
261  msc->SetSkin(theParameters->MscSkin());
262  msc->SetRangeFactor(facrange);
263  msc->SetGeomFactor(theParameters->MscGeomFactor());
264  msc->SetPolarAngleLimit(theParameters->MscThetaLimit());
265  G4double emax =
266  std::min(msc->HighEnergyLimit(),theParameters->MaxKinEnergy());
267  msc->SetHighEnergyLimit(emax);
268  }
269 
270  modelManager->Initialise(firstParticle, G4Electron::Electron(),
271  10.0, verboseLevel);
272 
273  if(!safetyHelper) {
274  safetyHelper = G4TransportationManager::GetTransportationManager()
275  ->GetSafetyHelper();
276  safetyHelper->InitialiseHelper();
277  }
278  }
279 }

283 void G4VMultipleScattering::BuildPhysicsTable(const G4ParticleDefinition& part)
284 {
285  G4String num = part.GetParticleName();
286  G4bool master = emManager->IsMaster();
287  if(1 < verboseLevel) {
288  G4cout << "### G4VMultipleScattering::BuildPhysicsTable() for "
289  << GetProcessName()
290  << " and particle " << num << " isIon: " << isIon
291  << " IsMaster: " << master << G4endl;
292  }
293  const G4VMultipleScattering* masterProcess =
294  static_cast<const G4VMultipleScattering*>(GetMasterProcess());
295 
296  if(firstParticle == &part) {
297  /*
298  G4cout << "### G4VMultipleScattering::BuildPhysicsTable() for "
299  << GetProcessName()
300  << " and particle " << num
301  << " IsMaster= " << G4LossTableManager::Instance()->IsMaster()
302  << " " << this
303  << G4endl;
304  */
305  emManager->BuildPhysicsTable(firstParticle);
306 
307  if(!master) {
308  // initialisation of models
309  /*
310  G4cout << "### G4VMultipleScattering::BuildPhysicsTable() for "
311  << GetProcessName()
312  << " and particle " << num
313  << " Nmod= " << numberOfModels << " " << this
314  << G4endl;
315  */
316  for(G4int i=0; i<numberOfModels; ++i) {
317  G4VMscModel* msc = static_cast<G4VMscModel*>(GetModelByIndex(i));
318  if(!msc) { continue; }
319  G4VMscModel* msc0=
320  static_cast<G4VMscModel*>(masterProcess->GetModelByIndex(i));
321  msc->SetCrossSectionTable(msc0->GetCrossSectionTable(), false);
322  msc->InitialiseLocal(firstParticle, msc0);
323  }
324  }
325  }
326 
327  // explicitly defined printout by particle name
328  if(1 < verboseLevel ||
329  (0 < verboseLevel && (num == "e-" ||
330  num == "e+" || num == "mu+" ||
331  num == "mu-" || num == "proton"||
332  num == "pi+" || num == "pi-" ||
333  num == "kaon+" || num == "kaon-" ||
334  num == "alpha" || num == "anti_proton" ||
335  num == "GenericIon" || num == "alpha+" ||
336  num == "alpha++" )))
337  {
338  StreamInfo(G4cout, part);
339  }
340 
341  if(1 < verboseLevel) {
342  G4cout << "### G4VMultipleScattering::BuildPhysicsTable() done for "
343  << GetProcessName()
344  << " and particle " << num
345  << G4endl;
346  }
347 }

398 G4double G4VMultipleScattering::AlongStepGetPhysicalInteractionLength(
399  const G4Track& track,
400  G4double,
401  G4double currentMinimalStep,
402  G4double&,
403  G4GPILSelection* selection)
404 {
405  // get Step limit proposed by the process
406  *selection = NotCandidateForSelection;
407  physStepLimit = gPathLength = tPathLength = currentMinimalStep;
408 
409  G4double ekin = track.GetKineticEnergy();
410  /*
411  G4cout << "MSC::AlongStepGPIL: Ekin= " << ekin
412  << " " << currParticle->GetParticleName()
413  << " currMod " << currentModel
414  << G4endl;
415  */
416  // isIon flag is used only to select a model
417  if(isIon) {
418  ekin *= proton_mass_c2/track.GetParticleDefinition()->GetPDGMass();
419  }
420 
421  // select new model
422  if(1 < numberOfModels) {
423  currentModel = static_cast<G4VMscModel*>(
424  SelectModel(ekin,track.GetMaterialCutsCouple()->GetIndex()));
425  }
426  // msc is active is model is active, energy above the limit,
427  // and step size is above the limit;
428  // if it is active msc may limit the step
429  if(currentModel->IsActive(ekin) && tPathLength > geomMin
430  && ekin >= lowestKinEnergy) {
431  isActive = true;
432  tPathLength =
433  currentModel->ComputeTruePathLengthLimit(track, gPathLength);
434  if (tPathLength < physStepLimit) {
435  *selection = CandidateForSelection;
436  }
437  } else { isActive = false; }
438 
439  //if(currParticle->GetPDGMass() > GeV)
440  /*
441  G4cout << "MSC::AlongStepGPIL: Ekin= " << ekin
442  << " " << currParticle->GetParticleName()
443  << " gPathLength= " << gPathLength
444  << " tPathLength= " << tPathLength
445  << " currentMinimalStep= " << currentMinimalStep
446  << " isActive " << isActive << G4endl;
447  */
448  return gPathLength;
449 }

463 G4VParticleChange*
464 G4VMultipleScattering::AlongStepDoIt(const G4Track& track, const G4Step& step)
465 {
466  fParticleChange.ProposeMomentumDirection(
467  step.GetPostStepPoint()->GetMomentumDirection());
468  fNewPosition = step.GetPostStepPoint()->GetPosition();
469  fParticleChange.ProposePosition(fNewPosition);
470  fPositionChanged = false;
471 
472  G4double geomLength = step.GetStepLength();
473 
474  // very small step - no msc
475  if(!isActive) {
476  tPathLength = geomLength;
477 
478  // sample msc
479  } else {
480  G4double range =
481  currentModel->GetRange(currParticle,track.GetKineticEnergy(),
482  track.GetMaterialCutsCouple());
483 
484  tPathLength = currentModel->ComputeTrueStepLength(geomLength);
485 
486  /*
487  if(currParticle->GetPDGMass() > 0.9*GeV)
488  G4cout << "G4VMsc::AlongStepDoIt: GeomLength= "
489  << geomLength
490  << " tPathLength= " << tPathLength
491  << " physStepLimit= " << physStepLimit
492  << " dr= " << range - tPathLength
493  << " ekin= " << track.GetKineticEnergy() << G4endl;
494  */
495  // protection against wrong t->g->t conversion
496  tPathLength = std::min(tPathLength, physStepLimit);
497 
498  // do not sample scattering at the last or at a small step
499  if(tPathLength < range && tPathLength > geomMin) {
500 
501  static const G4double minSafety = 1.20*CLHEP::nm;
502  static const G4double sFact = 0.99;
503 
504  G4ThreeVector displacement = currentModel->SampleScattering(
505  step.GetPostStepPoint()->GetMomentumDirection(),minSafety);
506 
507  G4double r2 = displacement.mag2();
508  //G4cout << " R= " << sqrt(r2) << " Rmin= " << sqrt(minDisplacement2)
509  // << " flag= " << fDispBeyondSafety << G4endl;
510  if(r2 > minDisplacement2) {
511 
512  fPositionChanged = true;
513  G4double dispR = std::sqrt(r2);
514  G4double postSafety =
515  sFact*safetyHelper->ComputeSafety(fNewPosition, dispR);
516  //G4cout<<" R= "<< dispR<<" postSafety= "<<postSafety<<G4endl;
517 
518  // far away from geometry boundary
519  if(postSafety > 0.0 && dispR <= postSafety) {
520  fNewPosition += displacement;
521 
522  //near the boundary
523  } else {
524  // displaced point is definitely within the volume
525  //G4cout<<" R= "<<dispR<<" postSafety= "<<postSafety<<G4endl;
526  if(dispR < postSafety) {
527  fNewPosition += displacement;
528 
529  // optional extra mechanism is applied only if a particle
530  // is stopped by the boundary
531  } else if(fDispBeyondSafety && 0.0 == postSafety) {
532  fNewPosition += displacement;
533  G4double maxshift =
534  std::min(2.0*dispR, geomLength*(physStepLimit/tPathLength - 1.0));
535  G4double dist = 0.0;
536  G4double safety = postSafety + dispR;
537  fNewDirection = *(fParticleChange.GetMomentumDirection());
538  /*
539  G4cout << "##MSC before Recheck maxshift= " << maxshift
540  << " postsafety= " << postSafety
541  << " Ekin= " << track.GetKineticEnergy()
542  << " " << track.GetDefinition()->GetParticleName()
543  << G4endl;
544  */
545  // check if it is possible to shift to the boundary
546  // and the shift is not large
547  if(safetyHelper->RecheckDistanceToCurrentBoundary(fNewPosition,
548  fNewDirection, maxshift, &dist, &safety)
549  && std::abs(dist) < maxshift) {
550  /*
551  G4cout << "##MSC after Recheck dist= " << dist
552  << " postsafety= " << postSafety
553  << " t= " << tPathLength
554  << " g= " << geomLength
555  << " p= " << physStepLimit
556  << G4endl;
557  */
558  // shift is positive
559  if(dist >= 0.0) {
560  tPathLength *= (1.0 + dist/geomLength);
561  fNewPosition += dist*fNewDirection;
562 
563  // shift is negative cannot be larger than geomLength
564  } else {
565  maxshift = std::min(maxshift, geomLength);
566  if(0.0 < maxshift + dist) {
567  const G4ThreeVector& postpoint =
568  step.GetPostStepPoint()->GetPosition();
569  G4ThreeVector point = fNewPosition + dist*fNewDirection;
570  G4double R2 = (postpoint - point).mag2();
571  G4double newdist = dist;
572  // check not more than 10 extra boundaries
573  for(G4int i=0; i<10; ++i) {
574  dist = 0.0;
575  if(safetyHelper->RecheckDistanceToCurrentBoundary(
576  point, fNewDirection, maxshift, &dist, &safety)
577  && std::abs(newdist + dist) < maxshift) {
578  point += dist*fNewDirection;
579  G4double R2new = (postpoint - point).mag2();
580  //G4cout << "Backward i= " << i << " dist= " << dist
581  // << " R2= " << R2new << G4endl;
582  if(dist >= 0.0 || R2new > R2) { break; }
583  R2 = R2new;
584  fNewPosition = point;
585  newdist += dist;
586  } else {
587  break;
588  }
589  }
590  tPathLength *= (1.0 + newdist/geomLength);
591  // shift on boundary is not possible for negative disp
592  } else {
593  fNewPosition += displacement*(postSafety/dispR - 1.0);
594  }
595  }
596  // shift on boundary is not possible for any disp
597  } else {
598  fNewPosition += displacement*(postSafety/dispR - 1.0);
599  }
600  // reduced displacement
601  } else if(postSafety > geomMin) {
602  fNewPosition += displacement*(postSafety/dispR);
603 
604  // very small postSafety
605  } else {
606  fPositionChanged = false;
607  }
608  }
