65 void PrimaryGeneratorAction0::GeneratePrimaries(G4Event* anEvent)
66 {
67  //vertex position fixed
68  //
69  G4double x0 = 0*mm, y0 = 0*mm, z0 = 0*mm;
70  fParticleGun->SetParticlePosition(G4ThreeVector(x0,y0,z0));
71 
72  //direction uniform in solid angle
73  //
74  G4double cosAlpha = fCosAlphaMin-G4UniformRand()*(fCosAlphaMin-fCosAlphaMax);
75  G4double sinAlpha = std::sqrt(1. - cosAlpha*cosAlpha);
76  G4double psi = fPsiMin + G4UniformRand()*(fPsiMax - fPsiMin);
77 
78  G4double ux = sinAlpha*std::cos(psi),
79  uy = sinAlpha*std::sin(psi),
80  uz = cosAlpha;
81 
82  fParticleGun->SetParticleMomentumDirection(G4ThreeVector(ux,uy,uz));
83 
84  fParticleGun->GeneratePrimaryVertex(anEvent);
85 }
