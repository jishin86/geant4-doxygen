35 G4GaussChebyshevQ::G4GaussChebyshevQ( function pFunction ,
36  G4int nChebyshev )
37  : G4VGaussianQuadrature(pFunction)
38 {
39  fNumber = nChebyshev ; // Try to reduce fNumber twice ??
40  G4double cof = pi/fNumber ;
41  fAbscissa = new G4double[fNumber] ;
42  fWeight = new G4double[fNumber] ;
43  for(G4int i=0;i<fNumber;i++)
44  {
45  fAbscissa[i] = std::cos(cof*(i + 0.5)) ;
46  fWeight[i] = cof*std::sqrt(1 - fAbscissa[i]*fAbscissa[i]) ;
47  }
48 }

62 G4double
63 G4GaussChebyshevQ::Integral(G4double a, G4double b) const
64 {
65  G4double xDiff=0.5*(b - a),
66  xMean=0.5*(a + b),
67  dx=0.0, integral=0.0 ;
68 
69  for(G4int i=0;i<fNumber;i++)
70  {
71  dx = xDiff*fAbscissa[i] ;
72  integral += fWeight[i]*fFunction(xMean + dx) ;
73  }
74  return integral *= xDiff ;
75 }
