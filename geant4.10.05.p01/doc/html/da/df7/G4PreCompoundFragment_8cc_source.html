85 G4double G4PreCompoundFragment::
86 IntegrateEmissionProbability(G4double low, G4double up,
87  const G4Fragment & aFragment)
88 {
89  static const G4double den = 1.0/CLHEP::MeV;
90  G4double del = (up - low);
91  G4int nbins = std::max(3,(G4int)(del*den));
92  del /= (G4double)nbins;
93  G4double e = low + 0.5*del;
94  probmax = ProbabilityDistributionFunction(e, aFragment);
95  //G4cout << " 0. e= " << e << " y= " << probmax << G4endl;
96 
97  G4double sum = probmax;
98  for (G4int i=1; i<nbins; ++i) {
99  e += del;
100 
101  G4double y = ProbabilityDistributionFunction(e, aFragment);
102  probmax = std::max(probmax, y);
103  sum += y;
104  if(y < sum*0.01) { break; }
105  //G4cout << " " << i << ". e= " << e << " y= " << y << " sum= " << sum << G4endl;
106  }
107  sum *= del;
108  //G4cout << "Evap prob: " << sum << " probmax= " << probmax << G4endl;
109  return sum;
110 }

112 G4double G4PreCompoundFragment::CrossSection(G4double ekin) const
113 {
114  G4double res;
115  if(OPTxs == 0 || (OPTxs == 4 && theMaxKinEnergy < 10.)) {
116  res = GetOpt0(ekin);
117 
118  } else if(OPTxs <= 2) {
119  res = G4ChatterjeeCrossSection::ComputeCrossSection(ekin, theCoulombBarrier,
120  theResA13, muu,
121  index, theZ, theResA);
122 
123  } else {
124  res = G4KalbachCrossSection::ComputeCrossSection(ekin, theCoulombBarrier,
125  theResA13, muu,
126  index, theZ, theA, theResA);
127  }
128  return res;
129 }
