130 G4double
131 G4NeutronCaptureXS::GetElementCrossSection(const G4DynamicParticle* aParticle,
132  G4int Z, const G4Material*)
133 {
134  G4double xs = 0.0;
135  G4double ekin = aParticle->GetKineticEnergy();
136  if(ekin > emax || Z < 1 || Z >= MAXZCAPTURE) { return xs; }
137  if(ekin < elimit) { ekin = elimit; }
138 
139  // element was not initialised
140  G4PhysicsVector* pv = data->GetElementData(Z);
141  if(!pv) { return xs; }
142 
143  G4double e1 = pv->Energy(0);
144  if(ekin < e1) { xs = (*pv)[0]*std::sqrt(e1/ekin); }
145  else if(ekin <= pv->GetMaxEnergy()) { xs = pv->Value(ekin); }
146 
147  if(verboseLevel > 0){
148  G4cout << "ekin= " << ekin << ", xs= " << xs << G4endl;
149  }
150  return xs;
151 }

162 G4double G4NeutronCaptureXS::IsoCrossSection(G4double ekin, G4int Z, G4int A)
163 {
164  G4double xs = 0.0;
165  if(ekin > emax || Z < 1 || Z >= MAXZCAPTURE) { return xs; }
166  if(ekin < elimit) { ekin = elimit; }
167 
168  G4PhysicsVector* pviso = data->GetComponentDataByID(Z, A - amin[Z]);
169  if(pviso) {
170  G4double e1 = pviso->Energy(1);
171  if(ekin < e1) { xs = (*pviso)[1]*std::sqrt(e1/ekin); }
172  else if(ekin <= pviso->GetMaxEnergy()) { xs = pviso->Value(ekin); }
173  if(verboseLevel > 0) {
174  G4cout << "G4NeutronCaptureXS::IsoXS: Ekin(MeV)= " << ekin/MeV
175  << " xs(b)= " << xs/barn
176  << " Z= " << Z << " A= " << A << G4endl;
177  }
178  return xs;
179  }
180  // isotope data are not available or applicable
181  G4PhysicsVector* pv = data->GetElementData(Z);
182  if(pv) {
183  G4double e1 = pv->Energy(1);
184  if(ekin < e1) { xs = (*pv)[1]*std::sqrt(e1/ekin); }
185  else if(ekin <= pv->GetMaxEnergy()) { xs = pv->Value(ekin); }
186  if(verboseLevel > 0) {
187  G4cout << "G4NeutronCaptureXS::IsoXS: Ekin(MeV)= " << ekin/MeV
188  << " xs(b)= " << xs/barn
189  << " Z= " << Z << " A= " << A << G4endl;
190  }
191  }
192  return xs;
193 }

195 const G4Isotope*
196 G4NeutronCaptureXS::SelectIsotope(const G4Element* anElement,
197  G4double kinEnergy)
198 {
199  size_t nIso = anElement->GetNumberOfIsotopes();
200  const G4Isotope* iso = anElement->GetIsotope(0);
201 
202  // more than 1 isotope
203  if(1 < nIso) {
204  G4int Z = anElement->GetZasInt();
205 
206  const G4double* abundVector = anElement->GetRelativeAbundanceVector();
207  G4double q = G4UniformRand();
208  G4double sum = 0.0;
209 
210  // is there isotope wise cross section?
211  size_t j;
212  if(0 == amin[Z] || Z >= MAXZCAPTURE) {
213  for (j = 0; j<nIso; ++j) {
214  sum += abundVector[j];
215  if(q <= sum) {
216  iso = anElement->GetIsotope(j);
217  break;
218  }
219  }
220  } else {
221  size_t nn = temp.size();
222  if(nn < nIso) { temp.resize(nIso, 0.); }
223 
224  for (j=0; j<nIso; ++j) {
225  sum += abundVector[j]*IsoCrossSection(kinEnergy, Z,
226  anElement->GetIsotope(j)->GetN());
227  temp[j] = sum;
228  }
229  sum *= q;
230  for (j = 0; j<nIso; ++j) {
231  if(temp[j] >= sum) {
232  iso = anElement->GetIsotope(j);
233  break;
234  }
235  }
236  }
237  }
238  return iso;
239 }

241 void
242 G4NeutronCaptureXS::BuildPhysicsTable(const G4ParticleDefinition& p)
243 {
244  if(verboseLevel > 0){
245  G4cout << "G4NeutronCaptureXS::BuildPhysicsTable for "
246  << p.GetParticleName() << G4endl;
247  }
248  if(p.GetParticleName() != "neutron") {
249  G4ExceptionDescription ed;
250  ed << p.GetParticleName() << " is a wrong particle type -"
251  << " only neutron is allowed";
252  G4Exception("G4NeutronCaptureXS::BuildPhysicsTable(..)","had012",
253  FatalException, ed, "");
254  return;
255  }
256 
257  if(!data) {
258 #ifdef G4MULTITHREADED
259  G4MUTEXLOCK(&neutronCaptureXSMutex);
260  if(!data) {
261 #endif
262  isMaster = true;
263  data = new G4ElementData();
264  data->SetName("NeutronCapture");
265  temp.resize(13,0.0);
266 #ifdef G4MULTITHREADED
267  }
268  G4MUTEXUNLOCK(&neutronCaptureXSMutex);
269 #endif
270  }
271 
272  // it is possible re-initialisation for the second run
273  if(isMaster) {
274 
275  // check environment variable
276  // Build the complete string identifying the file with the data set
277  char* path = getenv("G4PARTICLEXSDATA");
278 
279  // Access to elements
280  const G4ElementTable* theElmTable = G4Element::GetElementTable();
281  size_t numOfElm = G4Element::GetNumberOfElements();
282  for(size_t i=0; i<numOfElm; ++i) {
283  G4int Z = ((*theElmTable)[i])->GetZasInt();
284  if(Z >= MAXZCAPTURE) { Z = MAXZCAPTURE-1; }
285  //G4cout << "Z= " << Z << G4endl;
286  // Initialisation
287  if(!data->GetElementData(Z)) { Initialise(Z, path); }
288  }
289  }
290 }

292 void
293 G4NeutronCaptureXS::Initialise(G4int Z, const char* p)
294 {
295  if(data->GetElementData(Z) || Z < 1 || Z >= MAXZCAPTURE) { return; }
296  const char* path = p;
297 
298  // check environment variable
299  if(!p) {
300  path = getenv("G4PARTICLEXSDATA");
301  if (!path) {
302  G4Exception("G4NeutronCaptureXS::Initialise(..)","had013",FatalException,
303  "Environment variable G4PARTICLEXSDATA is not defined");
304  return;
305  }
306  }
307 
308  // upload element data
309  std::ostringstream ost;
310  ost << path << "/neutron/cap" << Z ;
311  G4PhysicsVector* v = RetrieveVector(ost, true);
312  data->InitialiseForElement(Z, v);
313 
314  // upload isotope data
315  if(amin[Z] > 0) {
316  size_t nmax = (size_t)(amax[Z]-amin[Z]+1);
317  data->InitialiseForComponent(Z, nmax);
318 
319  for(G4int A=amin[Z]; A<=amax[Z]; ++A) {
320  std::ostringstream ost1;
321  ost1 << path << "/neutron/cap" << Z << "_" << A;
322  v = RetrieveVector(ost1, false);
323  data->AddComponent(Z, A, v);
324  }
325  }
326 }
