68 void TrackingAction::PostUserTrackingAction(const G4Track* track)
69 {
70  // keep only outgoing particle
71  G4StepStatus status = track->GetStep()->GetPostStepPoint()->GetStepStatus();
72  if (status != fWorldBoundary) return;
73 
74  const G4ParticleDefinition* particle = track->GetParticleDefinition();
75  G4String name = particle->GetParticleName();
76  G4double energy = track->GetKineticEnergy();
77 
78  fEventAction->AddEflow(energy);
79 
80  Run* run = static_cast<Run*>(
81  G4RunManager::GetRunManager()->GetNonConstCurrentRun());
82  run->ParticleFlux(name,energy);
83 
84  // histograms: enery flow
85  //
86  G4AnalysisManager* analysis = G4AnalysisManager::Instance();
87 
88  G4int ih = 0;
89  G4String type = particle->GetParticleType();
90  G4double charge = particle->GetPDGCharge();
91  if (charge > 3.) ih = 10;
92  else if (particle == G4Gamma::Gamma()) ih = 4;
93  else if (particle == G4Electron::Electron()) ih = 5;
94  else if (particle == G4Positron::Positron()) ih = 5;
95  else if (particle == G4Neutron::Neutron()) ih = 6;
96  else if (particle == G4Proton::Proton()) ih = 7;
97  else if (particle == G4Deuteron::Deuteron()) ih = 8;
98  else if (particle == G4Alpha::Alpha()) ih = 9;
99  else if (type == "nucleus") ih = 10;
100  else if (type == "baryon") ih = 11;
101  else if (type == "meson") ih = 12;
102  else if (type == "lepton") ih = 13;
103  if (ih > 0) analysis->FillH1(ih,energy);
104 }
