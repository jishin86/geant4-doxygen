31 // Hadron physics for the new physics list FTFP_BERT_ATL.
32 // This is a modified version of the FTFP_BERT hadron physics for ATLAS.
33 // The hadron physics of FTFP_BERT_ATL has the transition between Bertini
34 // (BERT) intra-nuclear cascade model and Fritiof (FTF) string model in the
35 // energy region [9, 12] GeV (instead of [4, 5] GeV as in FTFP_BERT).

78 void G4HadronPhysicsFTFP_BERT_ATL::Pion()
79 {
80  auto pik = new G4PiKBuilder;
81  AddBuilder(pik);
82  auto ftfppik = new G4FTFPPiKBuilder(QuasiElastic);
83  AddBuilder(ftfppik);
84  ftfppik->SetMinEnergy(minFTFP_pion);
85  pik->RegisterMe(ftfppik);
86  auto bertpik = new G4BertiniPiKBuilder();
87  AddBuilder(bertpik);
88  bertpik->SetMaxEnergy(maxBERT_pion);
89  pik->RegisterMe(bertpik);
90  pik->Build();
91 }
