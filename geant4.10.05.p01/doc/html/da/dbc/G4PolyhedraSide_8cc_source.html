65 //
66 // Constructor
67 //
68 // Values for r1,z1 and r2,z2 should be specified in clockwise
69 // order in (r,z).
70 //
71 G4PolyhedraSide::G4PolyhedraSide( const G4PolyhedraSideRZ *prevRZ,
72  const G4PolyhedraSideRZ *tail,
73  const G4PolyhedraSideRZ *head,
74  const G4PolyhedraSideRZ *nextRZ,
75  G4int theNumSide,
76  G4double thePhiStart,
77  G4double thePhiTotal,
78  G4bool thePhiIsOpen,
79  G4bool isAllBehind )
80 {
81 
82  instanceID = subInstanceManager.CreateSubInstance();
83 
84  kCarTolerance = G4GeometryTolerance::GetInstance()->GetSurfaceTolerance();
85  fSurfaceArea=0.;
86  G4MT_phphi.first = G4ThreeVector(0,0,0);
87  G4MT_phphi.second= 0.0;
88 
89  //
90  // Record values
91  //
92  r[0] = tail->r; z[0] = tail->z;
93  r[1] = head->r; z[1] = head->z;
94 
95  G4double phiTotal;
96 
97  //
98  // Set phi to our convention
99  //
100  startPhi = thePhiStart;
101  while (startPhi < 0.0) // Loop checking, 13.08.2015, G.Cosmo
102  startPhi += twopi;
103 
104  phiIsOpen = thePhiIsOpen;
105  phiTotal = (phiIsOpen) ? thePhiTotal : twopi;
106 
107  allBehind = isAllBehind;
108 
109  //
110  // Make our intersecting cone
111  //
112  cone = new G4IntersectingCone( r, z );
113 
114  //
115  // Construct side plane vector set
116  //
117  numSide = theNumSide;
118  deltaPhi = phiTotal/theNumSide;
119  endPhi = startPhi+phiTotal;
120 
121  vecs = new G4PolyhedraSideVec[numSide];
122 
123  edges = new G4PolyhedraSideEdge[phiIsOpen ? numSide+1 : numSide];
124 
125  //
126  // ...this is where we start
127  //
128  G4double phi = startPhi;
129  G4ThreeVector a1( r[0]*std::cos(phi), r[0]*std::sin(phi), z[0] ),
130  b1( r[1]*std::cos(phi), r[1]*std::sin(phi), z[1] ),
131  c1( prevRZ->r*std::cos(phi), prevRZ->r*std::sin(phi), prevRZ->z ),
132  d1( nextRZ->r*std::cos(phi), nextRZ->r*std::sin(phi), nextRZ->z ),
133  a2, b2, c2, d2;
134  G4PolyhedraSideEdge *edge = edges;
135 
136  G4PolyhedraSideVec *vec = vecs;
137  do // Loop checking, 13.08.2015, G.Cosmo
138  {
139  //
140  // ...this is where we are going
141  //
142  phi += deltaPhi;
143  a2 = G4ThreeVector( r[0]*std::cos(phi), r[0]*std::sin(phi), z[0] );
144  b2 = G4ThreeVector( r[1]*std::cos(phi), r[1]*std::sin(phi), z[1] );
145  c2 = G4ThreeVector( prevRZ->r*std::cos(phi), prevRZ->r*std::sin(phi), prevRZ->z );
146  d2 = G4ThreeVector( nextRZ->r*std::cos(phi), nextRZ->r*std::sin(phi), nextRZ->z );
147 
148  G4ThreeVector tt;
149 
150  //
151  // ...build some relevant vectors.
152  // the point is to sacrifice a little memory with precalcs
153  // to gain speed
154  //
155  vec->center = 0.25*( a1 + a2 + b1 + b2 );
156 
157  tt = b2 + b1 - a2 - a1;
158  vec->surfRZ = tt.unit();
159  if (vec==vecs) lenRZ = 0.25*tt.mag();
160 
161  tt = b2 - b1 + a2 - a1;
162  vec->surfPhi = tt.unit();
163  if (vec==vecs)
164  {
165  lenPhi[0] = 0.25*tt.mag();
166  tt = b2 - b1;
167  lenPhi[1] = (0.5*tt.mag()-lenPhi[0])/lenRZ;
168  }
169 
170  tt = vec->surfPhi.cross(vec->surfRZ);
171  vec->normal = tt.unit();
172 
173  //
174  // ...edge normals are the average of the normals of
175  // the two faces they connect.
176  //
177  // ...edge normals are necessary if we are to accurately
178  // decide if a point is "inside" a face. For non-convex
179  // shapes, it is absolutely necessary to know information
180  // on adjacent faces to accurate determine this.
181  //
182  // ...we don't need them for the phi edges, since that
183  // information is taken care of internally. The r/z edges,
184  // however, depend on the adjacent G4PolyhedraSide.
185  //
186  G4ThreeVector a12, adj;
187 
188  a12 = a2-a1;
189 
190  adj = 0.5*(c1+c2-a1-a2);
191  adj = adj.cross(a12);
192  adj = adj.unit() + vec->normal;
193  vec->edgeNorm[0] = adj.unit();
194 
195  a12 = b1-b2;
196  adj = 0.5*(d1+d2-b1-b2);
197  adj = adj.cross(a12);
198  adj = adj.unit() + vec->normal;
199  vec->edgeNorm[1] = adj.unit();
200 
201  //
202  // ...the corners are crucial. It is important that
203  // they are calculated consistently for adjacent
204  // G4PolyhedraSides, to avoid gaps caused by roundoff.
205  //
206  vec->edges[0] = edge;
207  edge->corner[0] = a1;
208  edge->corner[1] = b1;
209  edge++;
210  vec->edges[1] = edge;
211 
212  a1 = a2;
213  b1 = b2;
214  c1 = c2;
215  d1 = d2;
216  } while( ++vec < vecs+numSide );
217 
218  //
219  // Clean up hanging edge
220  //
221  if (phiIsOpen)
222  {
223  edge->corner[0] = a2;
224  edge->corner[1] = b2;
225  }
226  else
227  {
228  vecs[numSide-1].edges[1] = edges;
229  }
230 
231  //
232  // Go back and fill in remaining fields in edges
233  //
234  vec = vecs;
235  G4PolyhedraSideVec *prev = vecs+numSide-1;
236  do // Loop checking, 13.08.2015, G.Cosmo
237  {
238  edge = vec->edges[0]; // The edge between prev and vec
239 
240  //
241  // Okay: edge normal is average of normals of adjacent faces
242  //
243  G4ThreeVector eNorm = vec->normal + prev->normal;
244  edge->normal = eNorm.unit();
245 
246  //
247  // Vertex normal is average of norms of adjacent surfaces (all four)
248  // However, vec->edgeNorm is unit vector in some direction
249  // as the sum of normals of adjacent PolyhedraSide with vec.
250  // The normalization used for this vector should be the same
251  // for vec and prev.
252  //
253  eNorm = vec->edgeNorm[0] + prev->edgeNorm[0];
254  edge->cornNorm[0] = eNorm.unit();
255 
256  eNorm = vec->edgeNorm[1] + prev->edgeNorm[1];
257  edge->cornNorm[1] = eNorm.unit();
258  } while( prev=vec, ++vec < vecs + numSide );
259 
260  if (phiIsOpen)
261  {
262  // G4double rFact = std::cos(0.5*deltaPhi);
263  //
264  // If phi is open, we need to patch up normals of the
265  // first and last edges and their corresponding
266  // vertices.
267  //
268  // We use vectors that are in the plane of the
269  // face. This should be safe.
270  //
271  vec = vecs;
272 
273  G4ThreeVector normvec = vec->edges[0]->corner[0]
274  - vec->edges[0]->corner[1];
275  normvec = normvec.cross(vec->normal);
276  if (normvec.dot(vec->surfPhi) > 0) normvec = -normvec;
277 
278  vec->edges[0]->normal = normvec.unit();
279 
280  vec->edges[0]->cornNorm[0] = (vec->edges[0]->corner[0]
281  - vec->center).unit();
282  vec->edges[0]->cornNorm[1] = (vec->edges[0]->corner[1]
283  - vec->center).unit();
284 
285  //
286  // Repeat for ending phi
287  //
288  vec = vecs + numSide - 1;
289 
290  normvec = vec->edges[1]->corner[0] - vec->edges[1]->corner[1];
291  normvec = normvec.cross(vec->normal);
292  if (normvec.dot(vec->surfPhi) < 0) normvec = -normvec;
293 
294  vec->edges[1]->normal = normvec.unit();
295 
296  vec->edges[1]->cornNorm[0] = (vec->edges[1]->corner[0]
297  - vec->center).unit();
298  vec->edges[1]->cornNorm[1] = (vec->edges[1]->corner[1]
299  - vec->center).unit();
300  }
301 
302  //
303  // edgeNorm is the factor one multiplies the distance along vector phi
304  // on the surface of one of our sides in order to calculate the distance
305  // from the edge. (see routine DistanceAway)
306  //
307  edgeNorm = 1.0/std::sqrt( 1.0 + lenPhi[1]*lenPhi[1] );
308 }

424 //
425 // Intersect
426 //
427 // Decide if a line intersects the face.
428 //
429 // Arguments:
430 // p = (in) starting point of line segment
431 // v = (in) direction of line segment (assumed a unit vector)
432 // A, B = (in) 2d transform variables (see note top of file)
433 // normSign = (in) desired sign for dot product with normal (see below)
434 // surfTolerance = (in) minimum distance from the surface
435 // vecs = (in) Vector set array
436 // distance = (out) distance to surface furfilling all requirements
437 // distFromSurface = (out) distance from the surface
438 // thisNormal = (out) normal vector of the intersecting surface
439 //
440 // Return value:
441 // true if an intersection is found. Otherwise, output parameters are
442 // undefined.
443 //
444 // Notes:
445 // * normSign: if we are "inside" the shape and only want to find out how far
446 // to leave the shape, we only want to consider intersections with surfaces in
447 // which the trajectory is leaving the shape. Since the normal vectors to the
448 // surface always point outwards from the inside, this means we want the dot
449 // product of the trajectory direction v and the normal of the side normals[i]
450 // to be positive. Thus, we should specify normSign as +1.0. Otherwise, if
451 // we are outside and want to go in, normSign should be set to -1.0.
452 // Don't set normSign to zero, or you will get no intersections!
453 //
454 // * surfTolerance: see notes on argument "surfTolerance" in routine
455 // "IntersectSidePlane".
456 // ----HOWEVER---- We should *not* apply this surface tolerance if the
457 // starting point is not within phi or z of the surface. Specifically,
458 // if the starting point p angle in x/y places it on a separate side from the
459 // intersection or if the starting point p is outside the z bounds of the
460 // segment, surfTolerance must be ignored or we should *always* accept the
461 // intersection!
462 // This is simply because the sides do not have infinite extent.
463 //
464 //
465 G4bool G4PolyhedraSide::Intersect( const G4ThreeVector &p,
466  const G4ThreeVector &v,
467  G4bool outgoing,
468  G4double surfTolerance,
469  G4double &distance,
470  G4double &distFromSurface,
471  G4ThreeVector &normal,
472  G4bool &isAllBehind )
473 {
474  G4double normSign = outgoing ? +1 : -1;
475 
476  //
477  // ------------------TO BE IMPLEMENTED---------------------
478  // Testing the intersection of individual phi faces is
479  // pretty straight forward. The simple thing therefore is to
480  // form a loop and check them all in sequence.
481  //
482  // But, I worry about one day someone making
483  // a polygon with a thousands sides. A linear search
484  // would not be ideal in such a case.
485  //
486  // So, it would be nice to be able to quickly decide
487  // which face would be intersected. One can make a very
488  // good guess by using the intersection with a cone.
489  // However, this is only reliable in 99% of the cases.
490  //
491  // My solution: make a decent guess as to the one or
492  // two potential faces might get intersected, and then
493  // test them. If we have the wrong face, use the test
494  // to make a better guess.
495  //
496  // Since we might have two guesses, form a queue of
497  // potential intersecting faces. Keep an array of
498  // already tested faces to avoid doing one more than
499  // once.
500  //
501  // Result: at worst, an iterative search. On average,
502  // a little more than two tests would be required.
503  //
504  G4ThreeVector q = p + v;
505 
506  G4int face = 0;
507  G4PolyhedraSideVec *vec = vecs;
508  do // Loop checking, 13.08.2015, G.Cosmo
509  {
510  //
511  // Correct normal?
512  //
513  G4double dotProd = normSign*v.dot(vec->normal);
514  if (dotProd <= 0) continue;
515 
516  //
517  // Is this face in front of the point along the trajectory?
518  //
519  G4ThreeVector delta = p - vec->center;
520  distFromSurface = -normSign*delta.dot(vec->normal);
521 
522  if (distFromSurface < -surfTolerance) continue;
523 
524  //
525  // phi
526  // c -------- d ^
527  // | | |
528  // a -------- b +---> r/z
529  //
530  //
531  // Do we remain on this particular segment?
532  //
533  G4ThreeVector qc = q - vec->edges[1]->corner[0];
534  G4ThreeVector qd = q - vec->edges[1]->corner[1];
535 
536  if (normSign*qc.cross(qd).dot(v) < 0) continue;
537 
538  G4ThreeVector qa = q - vec->edges[0]->corner[0];
539  G4ThreeVector qb = q - vec->edges[0]->corner[1];
540 
541  if (normSign*qa.cross(qb).dot(v) > 0) continue;
542 
543  //
544  // We found the one and only segment we might be intersecting.
545  // Do we remain within r/z bounds?
546  //
547 
548  if (r[0] > 1/kInfinity && normSign*qa.cross(qc).dot(v) < 0) return false;
549  if (r[1] > 1/kInfinity && normSign*qb.cross(qd).dot(v) > 0) return false;
550 
551  //
552  // We allow the face to be slightly behind the trajectory
553  // (surface tolerance) only if the point p is within
554  // the vicinity of the face
555  //
556  if (distFromSurface < 0)
557  {
558  G4ThreeVector ps = p - vec->center;
559 
560  G4double rz = ps.dot(vec->surfRZ);
561  if (std::fabs(rz) > lenRZ+surfTolerance) return false;
562 
563  G4double pp = ps.dot(vec->surfPhi);
564  if (std::fabs(pp) > lenPhi[0] + lenPhi[1]*rz + surfTolerance) return false;
565  }
566 
567 
568  //
569  // Intersection found. Return answer.
570  //
571  distance = distFromSurface/dotProd;
572  normal = vec->normal;
573  isAllBehind = allBehind;
574  return true;
575  } while( ++vec, ++face < numSide );
576 
577  //
578  // Oh well. Better luck next time.
579  //
580  return false;
581 }

584 G4double G4PolyhedraSide::Distance( const G4ThreeVector &p, G4bool outgoing )
585 {
586  G4double normSign = outgoing ? -1 : +1;
587 
588  //
589  // Try the closest phi segment first
590  //
591  G4int iPhi = ClosestPhiSegment( GetPhi(p) );
592 
593  G4ThreeVector pdotc = p - vecs[iPhi].center;
594  G4double normDist = pdotc.dot(vecs[iPhi].normal);
595 
596  if (normSign*normDist > -0.5*kCarTolerance)
597  {
598  return DistanceAway( p, vecs[iPhi], &normDist );
599  }
600 
601  //
602  // Now we have an interesting problem... do we try to find the
603  // closest facing side??
604  //
605  // Considered carefully, the answer is no. We know that if we
606  // are asking for the distance out, we are supposed to be inside,
607  // and vice versa.
608  //
609 
610  return kInfinity;
611 }

617 EInside G4PolyhedraSide::Inside( const G4ThreeVector &p,
618  G4double tolerance,
619  G4double *bestDistance )
620 {
621  //
622  // Which phi segment is closest to this point?
623  //
624  G4int iPhi = ClosestPhiSegment( GetPhi(p) );
625 
626  G4double norm;
627 
628  //
629  // Get distance to this segment
630  //
631  *bestDistance = DistanceToOneSide( p, vecs[iPhi], &norm );
632 
633  //
634  // Use distance along normal to decide return value
635  //
636  if ( (std::fabs(norm) < tolerance) && (*bestDistance < 2.0*tolerance) )
637  return kSurface;
638  else if (norm < 0)
639  return kInside;
640  else
641  return kOutside;
642 }

666 //
667 // Extent
668 //
669 G4double G4PolyhedraSide::Extent( const G4ThreeVector axis )
670 {
671  if (axis.perp2() < DBL_MIN)
672  {
673  //
674  // Special case
675  //
676  return axis.z() < 0 ? -cone->ZLo() : cone->ZHi();
677  }
678 
679  G4int iPhi, i1, i2;
680  G4double best;
681  G4ThreeVector *list[4];
682 
683  //
684  // Which phi segment, if any, does the axis belong to
685  //
686  iPhi = PhiSegment( GetPhi(axis) );
687 
688  if (iPhi < 0)
689  {
690  //
691  // No phi segment? Check front edge of first side and
692  // last edge of second side
693  //
694  i1 = 0; i2 = numSide-1;
695  }
696  else
697  {
698  //
699  // Check all corners of matching phi side
700  //
701  i1 = iPhi; i2 = iPhi;
702  }
703

705  list[1] = vecs[i1].edges[0]->corner+1;

773 //
774 // IntersectSidePlane
775 //
776 // Decide if a line correctly intersects one side plane of our segment.
777 // It is assumed that the correct side has been chosen, and thus only
778 // the z bounds (of the entire segment) are checked.
779 //
780 // normSign - To be multiplied against normal:
781 // = +1.0 normal is unchanged
782 // = -1.0 normal is reversed (now points inward)
783 //
784 // Arguments:
785 // p - (in) Point
786 // v - (in) Direction
787 // vec - (in) Description record of the side plane
788 // normSign - (in) Sign (+/- 1) to apply to normal
789 // surfTolerance - (in) Surface tolerance (generally > 0, see below)
790 // distance - (out) Distance along v to intersection
791 // distFromSurface - (out) Distance from surface normal
792 //
793 // Notes:
794 // surfTolerance - Used to decide if a point is behind the surface,
795 // a point is allow to be -surfTolerance behind the
796 // surface (as measured along the normal), but *only*
797 // if the point is within the r/z bounds + surfTolerance
798 // of the segment.
799 //
800 G4bool G4PolyhedraSide::IntersectSidePlane( const G4ThreeVector &p,
801  const G4ThreeVector &v,
802  const G4PolyhedraSideVec& vec,
803  G4double normSign,
804  G4double surfTolerance,
805  G4double &distance,
806  G4double &distFromSurface )
807 {
808  //
809  // Correct normal? Here we have straight sides, and can safely ignore
810  // intersections where the dot product with the normal is zero.
811  //
812  G4double dotProd = normSign*v.dot(vec.normal);
813 
814  if (dotProd <= 0) return false;
815 
816  //
817  // Calculate distance to surface. If the side is too far
818  // behind the point, we must reject it.
819  //
820  G4ThreeVector delta = p - vec.center;
821  distFromSurface = -normSign*delta.dot(vec.normal);
822 
823  if (distFromSurface < -surfTolerance) return false;
824 
825  //
826  // Calculate precise distance to intersection with the side
827  // (along the trajectory, not normal to the surface)
828  //
829  distance = distFromSurface/dotProd;
830 
831  //
832  // Do we fall off the r/z extent of the segment?
833  //
834  // Calculate this very, very carefully! Why?
835  // 1. If a RZ end is at R=0, you can't miss!
836  // 2. If you just fall off in RZ, the answer must
837  // be consistent with adjacent G4PolyhedraSide faces.
838  // (2) implies that only variables used by other G4PolyhedraSide
839  // faces may be used, which includes only: p, v, and the edge corners.
840  // It also means that one side is a ">" or "<", which the other
841  // must be ">=" or "<=". Fortunately, this isn't a new problem.
842  // The solution below I borrowed from Joseph O'Rourke,
843  // "Computational Geometry in C (Second Edition)"
844  // See: http://cs.smith.edu/~orourke/
845  //
846  G4ThreeVector ic = p + distance*v - vec.center;
847  G4double atRZ = vec.surfRZ.dot(ic);
848 
849  if (atRZ < 0)
850  {
851  if (r[0]==0) return true; // Can't miss!
852 
853  if (atRZ < -lenRZ*1.2) return false; // Forget it! Missed by a mile.
854 
855  G4ThreeVector q = p + v;
856  G4ThreeVector qa = q - vec.edges[0]->corner[0],
857  qb = q - vec.edges[1]->corner[0];
858  G4ThreeVector qacb = qa.cross(qb);
859  if (normSign*qacb.dot(v) < 0) return false;
860 
861  if (distFromSurface < 0)
862  {
863  if (atRZ < -lenRZ-surfTolerance) return false;
864  }
865  }
866  else if (atRZ > 0)
867  {
868  if (r[1]==0) return true; // Can't miss!
869 
870  if (atRZ > lenRZ*1.2) return false; // Missed by a mile
871 
872  G4ThreeVector q = p + v;
873  G4ThreeVector qa = q - vec.edges[0]->corner[1],
874  qb = q - vec.edges[1]->corner[1];
875  G4ThreeVector qacb = qa.cross(qb);
876  if (normSign*qacb.dot(v) >= 0) return false;
877 
878  if (distFromSurface < 0)
879  {
880  if (atRZ > lenRZ+surfTolerance) return false;
881  }
882  }
883 
884  return true;
885 }
