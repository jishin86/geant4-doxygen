80 {

98 
99  // Special treatment for water in gas state
100 
101  myFormula = G4String("H_2O") ;
102  const G4State theState = material->GetState() ;
103  if( theState == kStateGas && myFormula == chFormula) {
104  chFormula = G4String("H_2O-Gas");
