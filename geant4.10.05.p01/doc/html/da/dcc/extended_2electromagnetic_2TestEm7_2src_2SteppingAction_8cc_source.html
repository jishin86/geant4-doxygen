53 void SteppingAction::UserSteppingAction(const G4Step* step)
54 {
55  G4double edep = step->GetTotalEnergyDeposit();
56  if (edep <= 0.) return;
57 
58  G4StepPoint* prePoint = step->GetPreStepPoint();
59  G4StepPoint* postPoint = step->GetPostStepPoint();
60 
61  G4int copyNb = prePoint->GetTouchableHandle()->GetCopyNumber();
62  if (copyNb > 0) { fRunAction->FillTallyEdep(copyNb-1, edep); }
63 
64  G4double niel = step->GetNonIonizingEnergyDeposit();
65  fRunAction->FillEdep(edep, niel);
66 
67  if (step->GetTrack()->GetTrackID() == 1) {
68  fRunAction->AddPrimaryStep();
69  /*
70  G4cout << step->GetTrack()->GetMaterial()->GetName()
71  << " E1= " << step->GetPreStepPoint()->GetKineticEnergy()
72  << " E2= " << step->GetPostStepPoint()->GetKineticEnergy()
73  << " Edep= " << edep
74  << " Q= " << step->GetTrack()->GetDynamicParticle()->GetCharge()
75  << " Qp= " << step->GetPostStepPoint()->GetCharge()
76  << G4endl;
77  */
78  }
79 
80  //Bragg curve
81  //
82  G4double xmax = fDetector->GetAbsorSizeX();
83 
84  G4double x1 = prePoint->GetPosition().x() + xmax*0.5;
85  G4double x2 = postPoint->GetPosition().x() + xmax*0.5;
86  if(x1 >= 0.0 && x2 <= xmax) {
87  G4double x = x1 + G4UniformRand()*(x2-x1);
88  if (step->GetTrack()->GetDefinition()->GetPDGCharge() == 0.) x = x2;
89  G4AnalysisManager* analysisManager = G4AnalysisManager::Instance();
90  analysisManager->FillH1(1, x, edep);
91  analysisManager->FillH1(2, x, edep);
92  }
93 }
