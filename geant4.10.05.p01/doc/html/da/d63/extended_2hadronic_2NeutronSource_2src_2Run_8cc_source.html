84  fParticleDataMap1[name] = ParticleData(1, Ekin, Ekin, Ekin);

87  ParticleData& data = it->second;

135 void Run::Merge(const G4Run* run)
136 {
137  const Run* localRun = static_cast<const Run*>(run);
138 
139  //primary particle info
140  //
141  fParticle = localRun->fParticle;
142  fEkin = localRun->fEkin;
143 
144  // accumulate sums
145  //
146  fEnergyDeposit += localRun->fEnergyDeposit;
147  fEnergyDeposit2 += localRun->fEnergyDeposit2;
148  fEnergyFlow += localRun->fEnergyFlow;
149  fEnergyFlow2 += localRun->fEnergyFlow2;
150 
151  //map: processes count
152  std::map<G4String,G4int>::const_iterator itp;
153  for ( itp = localRun->fProcCounter.begin();
154  itp != localRun->fProcCounter.end(); ++itp ) {
155 
156  G4String procName = itp->first;
157  G4int localCount = itp->second;
158  if ( fProcCounter.find(procName) == fProcCounter.end()) {
159  fProcCounter[procName] = localCount;
160  }
161  else {
162  fProcCounter[procName] += localCount;
163  }
164  }
165 
166  //map: created particles count
167  std::map<G4String,ParticleData>::const_iterator itc;
168  for (itc = localRun->fParticleDataMap1.begin();
169  itc != localRun->fParticleDataMap1.end(); ++itc) {
170 
171  G4String name = itc->first;
172  const ParticleData& localData = itc->second;
173  if ( fParticleDataMap1.find(name) == fParticleDataMap1.end()) {
174  fParticleDataMap1[name]
175  = ParticleData(localData.fCount,
176  localData.fEmean,
177  localData.fEmin,
178  localData.fEmax);
179  }
180  else {
181  ParticleData& data = fParticleDataMap1[name];
182  data.fCount += localData.fCount;
183  data.fEmean += localData.fEmean;
184  G4double emin = localData.fEmin;
185  if (emin < data.fEmin) data.fEmin = emin;
186  G4double emax = localData.fEmax;
187  if (emax > data.fEmax) data.fEmax = emax;
188  }
189  }
190 
191  //map: particles flux count
192  std::map<G4String,ParticleData>::const_iterator itn;
193  for (itn = localRun->fParticleDataMap2.begin();
194  itn != localRun->fParticleDataMap2.end(); ++itn) {
195 
196  G4String name = itn->first;
197  const ParticleData& localData = itn->second;
198  if ( fParticleDataMap2.find(name) == fParticleDataMap2.end()) {
199  fParticleDataMap2[name]
200  = ParticleData(localData.fCount,
201  localData.fEmean,
202  localData.fEmin,
203  localData.fEmax);
204  }
205  else {
206  ParticleData& data = fParticleDataMap2[name];
207  data.fCount += localData.fCount;
208  data.fEmean += localData.fEmean;
209  G4double emin = localData.fEmin;
210  if (emin < data.fEmin) data.fEmin = emin;
211  G4double emax = localData.fEmax;
212  if (emax > data.fEmax) data.fEmax = emax;
213  }
214  }
215 
216  G4Run::Merge(run);
217 }

221 void Run::EndOfRun()
222 {
223  G4int prec = 5, wid = prec + 2;
224  G4int dfprec = G4cout.precision(prec);
225 
226  //run condition
227  //
228  G4Material* material = fDetector->GetAbsorMaterial();
229  G4String Particle = fParticle->GetParticleName();
230  G4cout << "\n The run is " << numberOfEvent << " "<< Particle << " of "
231  << G4BestUnit(fEkin,"Energy") << " within "
232  << material->GetName() << " (D = "
233  << G4BestUnit(2*(fDetector->GetAbsorRadius()),"Length") << " L = "
234  << G4BestUnit(fDetector->GetAbsorLength(),"Length") << ")" << G4endl;
235 
236  if (numberOfEvent == 0) { G4cout.precision(dfprec); return;}
237 
238  //frequency of processes
239  //
240  G4cout << "\n Process calls frequency :" << G4endl;
241  G4int index = 0;
242  std::map<G4String,G4int>::iterator it;
243  for (it = fProcCounter.begin(); it != fProcCounter.end(); it++) {
244  G4String procName = it->first;
245  G4int count = it->second;
246  G4String space = " "; if (++index%3 == 0) space = "\n";
247  G4cout << " " << std::setw(20) << procName << "="<< std::setw(7) << count
248  << space;
249  }
250  G4cout << G4endl;
251 
252  //particles count
253  //
254  G4cout << "\n List of generated particles:" << G4endl;
255 
256  std::map<G4String,ParticleData>::iterator itc;
257  for (itc = fParticleDataMap1.begin(); itc != fParticleDataMap1.end(); itc++) {
258  G4String name = itc->first;
259  ParticleData data = itc->second;
260  G4int count = data.fCount;
261  G4double eMean = data.fEmean/count;
262  G4double eMin = data.fEmin;
263  G4double eMax = data.fEmax;
264 
265  G4cout << " " << std::setw(13) << name << ": " << std::setw(7) << count
266  << " Emean = " << std::setw(wid) << G4BestUnit(eMean, "Energy")
267  << "\t( " << G4BestUnit(eMin, "Energy")
268  << " --> " << G4BestUnit(eMax, "Energy")
269  << ")" << G4endl;
270  }
271 
272  // compute mean Energy deposited and rms
273  //
274  G4int TotNbofEvents = numberOfEvent;
275  fEnergyDeposit /= TotNbofEvents; fEnergyDeposit2 /= TotNbofEvents;
276  G4double rmsEdep = fEnergyDeposit2 - fEnergyDeposit*fEnergyDeposit;
277  if (rmsEdep>0.) rmsEdep = std::sqrt(rmsEdep);
278  else rmsEdep = 0.;
279 
280  G4cout << "\n Mean energy deposit per event = "
281  << G4BestUnit(fEnergyDeposit,"Energy") << "; rms = "
282  << G4BestUnit(rmsEdep, "Energy")
283  << G4endl;
284 
285  // compute mean Energy flow and rms
286  //
287  fEnergyFlow /= TotNbofEvents; fEnergyFlow2 /= TotNbofEvents;
288  G4double rmsEflow = fEnergyFlow2 - fEnergyFlow*fEnergyFlow;
289  if (rmsEflow>0.) rmsEflow = std::sqrt(rmsEflow);
290  else rmsEflow = 0.;
291 
292  G4cout << " Mean energy flow per event = "
293  << G4BestUnit(fEnergyFlow,"Energy") << "; rms = "
294  << G4BestUnit(rmsEflow, "Energy")
295  << G4endl;
296 
297  //particles flux
298  //
299  G4cout << "\n List of particles emerging from the container :" << G4endl;
300 
301  std::map<G4String,ParticleData>::iterator itn;
302  for (itn = fParticleDataMap2.begin(); itn != fParticleDataMap2.end(); itn++) {
303  G4String name = itn->first;
304  ParticleData data = itn->second;
305  G4int count = data.fCount;
306  G4double eMean = data.fEmean/count;
307  G4double eMin = data.fEmin;
308  G4double eMax = data.fEmax;
309  G4double Eflow = data.fEmean/TotNbofEvents;
310 
311  G4cout << " " << std::setw(13) << name << ": " << std::setw(7) << count
312  << " Emean = " << std::setw(wid) << G4BestUnit(eMean, "Energy")
313  << "\t( " << G4BestUnit(eMin, "Energy")
314  << " --> " << G4BestUnit(eMax, "Energy")
315  << ") \tEflow/event = " << G4BestUnit(Eflow, "Energy") << G4endl;
316  }
317 
318  //remove all contents in fProcCounter, fCount
319  fProcCounter.clear();
320  fParticleDataMap2.clear();
321 
322  //restore default format
323  G4cout.precision(dfprec);
324 }
