70 G4ClassificationOfNewTrack
71 StackingAction::ClassifyNewTrack(const G4Track* track)
72 {
73  //get fDetector informations
74  if (first) {
75  fMatWall = fDetector->GetWallMaterial();
76  fZcav = 0.5*(fDetector->GetCavityThickness());
77  fEmCal = new G4EmCalculator();
78  first = false;
79  }
80 
81  G4ClassificationOfNewTrack status = fUrgent;
82 
83  //keep primary particle or neutral
84  //
85  G4ParticleDefinition* particle = track->GetDefinition();
86  G4bool neutral = (particle->GetPDGCharge() == 0.);
87  if ((track->GetParentID() == 0) || neutral) return status;
88 
89  Run* run = static_cast<Run*>(
90  G4RunManager::GetRunManager()->GetNonConstCurrentRun());
91 
92  //energy spectrum of charged secondaries
93  //
94  G4double energy = track->GetKineticEnergy();
95  run->SumEsecond(energy);
96 
97  // kill e- which cannot reach Cavity
98  //
99  G4double position = (track->GetPosition()).z();
100  G4double safe = std::abs(position) - fZcav;
101  G4double range = fEmCal->GetRangeFromRestricteDEDX(energy,particle,fMatWall);
102  if (fKillTrack) {
103  if (range < 0.8*safe) status = fKill;
104  }
105 
106  //histograms
107  //
108  G4AnalysisManager* analysisManager = G4AnalysisManager::Instance();
109  analysisManager->FillH1(1,position);
110  analysisManager->FillH1(2,energy);
111  G4ThreeVector direction = track->GetMomentumDirection();
112  analysisManager->FillH1(3,std::acos(direction.z()));
113 
114  return status;
115 }
