86

90  }
91 
92  RightHandSide(yTemp2,dydxTemp2);
93 
94  for( i = 0; i < fNumberOfVariables; i++ )
95  {
96  yOut[i] = yIn[i] + h * (0.25 * dydx[i] + 0.75 * dydxTemp2[i]);
97  }
98

100 }
