31 // G4SurfBits
32 //
33 // Class description:
34 //
35 // This class provides a simple container of bits, to be used for
36 // optimization of tessellated surfaces (G4TessellatedSolid).
37 // Each bit can be set and tested via the functions SetBitNumber and
38 // TestBitNumber.
39 // The default value of all bits is false.
40 // The size of the container is automatically extended when a bit
41 // number is either set or tested. To reduce the memory size of the
42 // container use the Compact function, this will discard the memory
43 // occupied by the upper bits that are 0.
44 //
45

48 // of Root's TBits class by P.Canal

122 inline void G4SurfBits::SetBitNumber(unsigned int bitnumber, G4bool value)
123 {
124  // set bit number 'bitnumber' to be value
125  if (bitnumber >= fNBits) {
126  unsigned int new_size = (bitnumber/8) + 1;
127  if (new_size > fNBytes) {
128  if (new_size < 100 * 1024 * 1024)
129  new_size *= 2;
130  unsigned char *old_location = fAllBits;
131  fAllBits = new unsigned char[new_size];
132  std::memcpy(fAllBits,old_location,fNBytes);
133  std::memset(fAllBits+fNBytes ,0, new_size-fNBytes);
134  fNBytes = new_size;
135  delete [] old_location;
136  }
137  fNBits = bitnumber+1;
138  }
139  unsigned int loc = bitnumber/8;
140  unsigned char bit = bitnumber%8;
141  if (value)
142  fAllBits[loc] |= (1<<bit);
143  else
144  fAllBits[loc] &= (0xFF ^ (1<<bit));
145 }

147 inline G4bool G4SurfBits::TestBitNumber(unsigned int bitnumber) const
148 {
149  // Return the current value of the bit
150 
151  if (bitnumber >= fNBits) return false;
152  unsigned int loc = bitnumber/8;
153  unsigned char value = fAllBits[loc];
154  unsigned char bit = bitnumber%8;
155  G4bool result = (value & (1<<bit)) != 0;
156  return result;
157  // short: return 0 != (fAllBits[bitnumber/8] & (1<< (bitnumber%8)));
158 }
