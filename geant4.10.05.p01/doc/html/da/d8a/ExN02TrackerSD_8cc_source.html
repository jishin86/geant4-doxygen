68 G4bool ExN02TrackerSD::ProcessHits(G4Step* aStep,G4TouchableHistory*)
69 {
70  G4double edep = aStep->GetTotalEnergyDeposit();
71 
72  if(edep==0.) return false;
73 
74  ExN02TrackerHit* newHit = new ExN02TrackerHit();
75  newHit->SetTrackID (aStep->GetTrack()->GetTrackID());
76  newHit->SetChamberNb(aStep->GetPreStepPoint()->GetTouchableHandle()
77  ->GetCopyNumber());
78  newHit->SetEdep (edep);
79  newHit->SetPos (aStep->GetPostStepPoint()->GetPosition());
80  trackerCollection->insert( newHit );
81 
82  //newHit->Print();
83  //newHit->Draw();
84 
85  return true;
