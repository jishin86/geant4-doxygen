68 G4double G4CrossSectionComposite::CrossSection(const G4KineticTrack& trk1,
69  const G4KineticTrack& trk2) const
70 {
71  // Cross section of composite is the sum of components cross sections
72 
73  G4double crossSection = 0.;
74  G4double ecm = (trk1.Get4Momentum() + trk2.Get4Momentum()).mag();
75 
76  const G4CrossSectionVector* components = GetComponents();
77  if (components != 0)
78  {
79  G4int nComponents = this->GetComponents()->size();
80 // G4int nValid = 0;
81 
82  G4int i;
83  for (i=0; i<nComponents; i++)
84  {
85  G4CrossSectionSourcePtr componentPtr = (*components)[i];
86  G4VCrossSectionSource* component = componentPtr();
87  if (component->IsValid(ecm))
88  {
89  crossSection += component->CrossSection(trk1,trk2);
90  }
91  }
92  }
93 
94  return crossSection;
95 }

98 G4bool G4CrossSectionComposite::IsValid(G4double e) const
99 {
100  // The composite is valid if any of its components are valid
101  G4bool answer = false;
102  const G4CrossSectionVector* components = GetComponents();
103  if (components != 0)
104  {
105  G4int n = components->size();
106  G4int i;
107  for (i=0; i<n; i++)
108  {
109  G4CrossSectionSourcePtr componentPtr = (*components)[i];
110  G4VCrossSectionSource* component = componentPtr();
111  if (component->IsValid(e))
112  {
113  answer = true;
114  break;
115  }
116  }
117  }
118  return answer;
119 }
