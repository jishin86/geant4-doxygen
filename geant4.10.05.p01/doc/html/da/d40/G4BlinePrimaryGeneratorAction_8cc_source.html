64 void G4BlinePrimaryGeneratorAction::GeneratePrimaries(G4Event* anEvent)
65 {
66  if (!fUserPrimaryAction)
67  {
68  G4Exception("G4BlinePrimaryGeneratorAction::GeneratePrimaries()",
69  "NullPointer", JustWarning,
70  "Primary generator action not defined !");
71  return;
72  }
73 
74  // For the first part of a bline the start position and time are defined
75  // by using the USER primary action while for the second part the previous
76  // values are taken.
77 
78  if (fFirstPartOfBline)
79  {
80  // set the position and time defined by using the USER primary action
81 
82  G4Event* tmpEvent = new G4Event();
83  fUserPrimaryAction->GeneratePrimaries(tmpEvent);
84  fBlineStartPosition = tmpEvent->GetPrimaryVertex()->GetPosition();
85  fT0 = tmpEvent->GetPrimaryVertex()->GetT0();
86  delete tmpEvent;
87  }
88  fFirstPartOfBline = false;
89 
90  G4PrimaryVertex* primary_vertex =
91  new G4PrimaryVertex(fBlineStartPosition, fT0);
92 
93  // Define the particle to be tracked as Charged Geantino
94 
95  G4ChargedGeantino* pdef = G4ChargedGeantino::ChargedGeantino();
96 
97  G4double mass = pdef->GetPDGMass();
98  G4double energy = 10000.*MeV + mass;
99  G4double pmom = std::sqrt(energy*energy-mass*mass);
100 
101  // The momentum direction and energy do not have an effect in tracing of
102  // bline but still need to be defined.
103 
104  G4double px = 0.;
105  G4double py = 0.;
106  G4double pz = pmom;
107 
108  G4PrimaryParticle* particle = new G4PrimaryParticle(pdef,px,py,pz);
109  particle->SetMass( mass );
110  particle->SetCharge(pdef->GetPDGCharge());
111  primary_vertex->SetPrimary( particle );
112 
113  anEvent->AddPrimaryVertex( primary_vertex );
114 }
