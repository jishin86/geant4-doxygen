61 // Constructor from a sequence of polygons
62 //
63 G4BoundingEnvelope::
64 G4BoundingEnvelope(const std::vector<const G4ThreeVectorList*>& polygons)
65  : fPolygons(&polygons)
66 {
67  // Check correctness of polygons
68  //
69  CheckBoundingPolygons();
70 
71  // Set bounding box
72  //
73  G4double xmin = kInfinity, ymin = kInfinity, zmin = kInfinity;
74  G4double xmax = -kInfinity, ymax = -kInfinity, zmax = -kInfinity;
75  std::vector<const G4ThreeVectorList*>::const_iterator ibase;
76  for (ibase = fPolygons->begin(); ibase != fPolygons->end(); ibase++)
77  {
78  std::vector<G4ThreeVector>::const_iterator ipoint;
79  for (ipoint = (*ibase)->begin(); ipoint != (*ibase)->end(); ipoint++)
80  {
81  G4double x = ipoint->x();
82  if (x < xmin) xmin = x;
83  if (x > xmax) xmax = x;
84  G4double y = ipoint->y();
85  if (y < ymin) ymin = y;
86  if (y > ymax) ymax = y;
87  G4double z = ipoint->z();
88  if (z < zmin) zmin = z;
89  if (z > zmax) zmax = z;
90  }
91  }
92  fMin.set(xmin,ymin,zmin);
93  fMax.set(xmax,ymax,zmax);
94 
95  // Check correctness of bounding box
96  //
97  CheckBoundingBox();
98 }

126 // Check correctness of the axis aligned bounding box
127 //
128 void G4BoundingEnvelope::CheckBoundingBox()
129 {
130  if (fMin.x() >= fMax.x() || fMin.y() >= fMax.y() || fMin.z() >= fMax.z())
131  {
132  std::ostringstream message;
133  message << "Badly defined bounding box (min >= max)!"
134  << "\npMin = " << fMin
135  << "\npMax = " << fMax;
136  G4Exception("G4BoundingEnvelope::CheckBoundingBox()",
137  "GeomMgt0001", JustWarning, message);
138  }
139 }

142 //
143 // Check correctness of the sequence of bounding polygons.
144 // Firsf and last polygons may consist of a single vertex
145 //
146 void G4BoundingEnvelope::CheckBoundingPolygons()
147 {
148  G4int nbases = fPolygons->size();
149  if (nbases < 2)
150  {
151  std::ostringstream message;
152  message << "Wrong number of polygons in the sequence: " << nbases
153  << "\nShould be at least two!";
154  G4Exception("G4BoundingEnvelope::CheckBoundingPolygons()",
155  "GeomMgt0001", FatalException, message);
156  return;
157  }
158 
159  G4int nsize = std::max((*fPolygons)[0]->size(),(*fPolygons)[1]->size());
160  if (nsize < 3)
161  {
162  std::ostringstream message;
163  message << "Badly constructed polygons!"
164  << "\nNumber of polygons: " << nbases
165  << "\nPolygon #0 size: " << (*fPolygons)[0]->size()
166  << "\nPolygon #1 size: " << (*fPolygons)[1]->size()
167  << "\n...";
168  G4Exception("G4BoundingEnvelope::CheckBoundingPolygons()",
169  "GeomMgt0001", FatalException, message);
170  return;
171  }
172 
173  for (G4int k=0; k<nbases; ++k)
174  {
175  G4int np = (*fPolygons)[k]->size();
176  if (np == nsize) continue;
177  if (np == 1 && k==0) continue;
178  if (np == 1 && k==nbases-1) continue;
179  std::ostringstream message;
180  message << "Badly constructed polygons!"
181  << "\nNumber of polygons: " << nbases
182  << "\nPolygon #" << k << " size: " << np
183  << "\nexpected size: " << nsize;
184  G4Exception("G4BoundingEnvelope::SetBoundingPolygons()",
185  "GeomMgt0001", FatalException, message);
186  return;
187  }
188 }

191 //
192 // Quick comparison: bounding box vs voxel, it return true if further
193 // calculations are not needed
194 //
195 G4bool
196 G4BoundingEnvelope::
197 BoundingBoxVsVoxelLimits(const EAxis pAxis,
198  const G4VoxelLimits& pVoxelLimits,
199  const G4Transform3D& pTransform3D,
200  G4double& pMin, G4double& pMax) const
201 {
202  pMin = kInfinity;
203  pMax = -kInfinity;
204  G4double xminlim = pVoxelLimits.GetMinXExtent();
205  G4double xmaxlim = pVoxelLimits.GetMaxXExtent();
206  G4double yminlim = pVoxelLimits.GetMinYExtent();
207  G4double ymaxlim = pVoxelLimits.GetMaxYExtent();
208  G4double zminlim = pVoxelLimits.GetMinZExtent();
209  G4double zmaxlim = pVoxelLimits.GetMaxZExtent();
210 
211  // Special case of pure translation
212  //
213  if (pTransform3D.xx()==1 && pTransform3D.yy()==1 && pTransform3D.zz()==1)
214  {
215  G4double xmin = fMin.x() + pTransform3D.dx();
216  G4double xmax = fMax.x() + pTransform3D.dx();
217  G4double ymin = fMin.y() + pTransform3D.dy();
218  G4double ymax = fMax.y() + pTransform3D.dy();
219  G4double zmin = fMin.z() + pTransform3D.dz();
220  G4double zmax = fMax.z() + pTransform3D.dz();
221 
222  if (xmin-kCarTolerance > xmaxlim) return true;
223  if (xmax+kCarTolerance < xminlim) return true;
224  if (ymin-kCarTolerance > ymaxlim) return true;
225  if (ymax+kCarTolerance < yminlim) return true;
226  if (zmin-kCarTolerance > zmaxlim) return true;
227  if (zmax+kCarTolerance < zminlim) return true;
228 
229  if (xmin >= xminlim && xmax <= xmaxlim &&
230  ymin >= yminlim && ymax <= ymaxlim &&
231  zmin >= zminlim && zmax <= zmaxlim)
232  {
233  if (pAxis == kXAxis)
234  {
235  pMin = (xmin-kCarTolerance < xminlim) ? xminlim : xmin;
236  pMax = (xmax+kCarTolerance > xmaxlim) ? xmaxlim : xmax;
237  }
238  else if (pAxis == kYAxis)
239  {
240  pMin = (ymin-kCarTolerance < yminlim) ? yminlim : ymin;
241  pMax = (ymax+kCarTolerance > ymaxlim) ? ymaxlim : ymax;
242  }
243  else if (pAxis == kZAxis)
244  {
245  pMin = (zmin-kCarTolerance < zminlim) ? zminlim : zmin;
246  pMax = (zmax+kCarTolerance > zmaxlim) ? zmaxlim : zmax;
247  }
248  pMin -= kCarTolerance;
249  pMax += kCarTolerance;
250  return true;
251  }
252  }
253 
254  // Find max scale factor of the transformation, set delta
255  // equal to kCarTolerance multiplied by the scale factor
256  //
257  G4double scale = FindScaleFactor(pTransform3D);
258  G4double delta = kCarTolerance*scale;
259 
260  // Set the sphere surrounding the bounding box
261  //
262  G4Point3D center = pTransform3D*G4Point3D(0.5*(fMin+fMax));
263  G4double radius = 0.5*scale*(fMax-fMin).mag() + delta;
264 
265  // Check if the sphere surrounding the bounding box is outside
266  // the voxel limits
267  //
268  if (center.x()-radius > xmaxlim) return true;
269  if (center.y()-radius > ymaxlim) return true;
270  if (center.z()-radius > zmaxlim) return true;
271  if (center.x()+radius < xminlim) return true;
272  if (center.y()+radius < yminlim) return true;
273  if (center.z()+radius < zminlim) return true;
274  return false;
275 }

278 //
279 // Calculate extent of the specified bounding envelope
280 //
281 G4bool
282 G4BoundingEnvelope::CalculateExtent(const EAxis pAxis,
283  const G4VoxelLimits& pVoxelLimits,
284  const G4Transform3D& pTransform3D,
285  G4double& pMin, G4double& pMax) const
286 {
287  pMin = kInfinity;
288  pMax = -kInfinity;
289  G4double xminlim = pVoxelLimits.GetMinXExtent();
290  G4double xmaxlim = pVoxelLimits.GetMaxXExtent();
291  G4double yminlim = pVoxelLimits.GetMinYExtent();
292  G4double ymaxlim = pVoxelLimits.GetMaxYExtent();
293  G4double zminlim = pVoxelLimits.GetMinZExtent();
294  G4double zmaxlim = pVoxelLimits.GetMaxZExtent();
295 
296  // Special case of pure translation
297  //
298  if (pTransform3D.xx()==1 && pTransform3D.yy()==1 && pTransform3D.zz()==1)
299  {
300  G4double xmin = fMin.x() + pTransform3D.dx();
301  G4double xmax = fMax.x() + pTransform3D.dx();
302  G4double ymin = fMin.y() + pTransform3D.dy();
303  G4double ymax = fMax.y() + pTransform3D.dy();
304  G4double zmin = fMin.z() + pTransform3D.dz();
305  G4double zmax = fMax.z() + pTransform3D.dz();
306 
307  if (xmin-kCarTolerance > xmaxlim) return false;
308  if (xmax+kCarTolerance < xminlim) return false;
309  if (ymin-kCarTolerance > ymaxlim) return false;
310  if (ymax+kCarTolerance < yminlim) return false;
311  if (zmin-kCarTolerance > zmaxlim) return false;
312  if (zmax+kCarTolerance < zminlim) return false;
313 
314  if (fPolygons == 0)
315  {
316  if (pAxis == kXAxis)
317  {
318  pMin = (xmin-kCarTolerance < xminlim) ? xminlim : xmin;
319  pMax = (xmax+kCarTolerance > xmaxlim) ? xmaxlim : xmax;
320  }
321  else if (pAxis == kYAxis)
322  {
323  pMin = (ymin-kCarTolerance < yminlim) ? yminlim : ymin;
324  pMax = (ymax+kCarTolerance > ymaxlim) ? ymaxlim : ymax;
325  }
326  else if (pAxis == kZAxis)
327  {
328  pMin = (zmin-kCarTolerance < zminlim) ? zminlim : zmin;
329  pMax = (zmax+kCarTolerance > zmaxlim) ? zmaxlim : zmax;
330  }
331  pMin -= kCarTolerance;
332  pMax += kCarTolerance;
333  return true;
334  }
335  }
336 
337  // Find max scale factor of the transformation, set delta
338  // equal to kCarTolerance multiplied by the scale factor
339  //
340  G4double scale = FindScaleFactor(pTransform3D);
341  G4double delta = kCarTolerance*scale;
342 
343  // Set the sphere surrounding the bounding box
344  //
345  G4Point3D center = pTransform3D*G4Point3D(0.5*(fMin+fMax));
346  G4double radius = 0.5*scale*(fMax-fMin).mag() + delta;
347 
348  // Check if the sphere surrounding the bounding box is within
349  // the voxel limits, if so then transform only one coordinate
350  //
351  if (center.x()-radius >= xminlim && center.x()+radius <= xmaxlim &&
352  center.y()-radius >= yminlim && center.y()+radius <= ymaxlim &&
353  center.z()-radius >= zminlim && center.z()+radius <= zmaxlim )
354  {
355  G4double cx, cy, cz, cd;
356  if (pAxis == kXAxis)
357  {
358  cx = pTransform3D.xx();
359  cy = pTransform3D.xy();
360  cz = pTransform3D.xz();
361  cd = pTransform3D.dx();
362  }
363  else if (pAxis == kYAxis)
364  {
365  cx = pTransform3D.yx();
366  cy = pTransform3D.yy();
367  cz = pTransform3D.yz();
368  cd = pTransform3D.dy();
369  }
370  else if (pAxis == kZAxis)
371  {
372  cx = pTransform3D.zx();
373  cy = pTransform3D.zy();
374  cz = pTransform3D.zz();
375  cd = pTransform3D.dz();
376  }
377  else
378  {
379  cx = cy = cz = cd = kInfinity;
380  }
381  G4double emin = kInfinity, emax = -kInfinity;
382  if (fPolygons == 0)
383  {
384  G4double coor;
385  coor = cx*fMin.x() + cy*fMin.y() + cz*fMin.z() + cd;
386  if (coor < emin) emin = coor;
387  if (coor > emax) emax = coor;
388  coor = cx*fMax.x() + cy*fMin.y() + cz*fMin.z() + cd;
389  if (coor < emin) emin = coor;
390  if (coor > emax) emax = coor;
391  coor = cx*fMax.x() + cy*fMax.y() + cz*fMin.z() + cd;
392  if (coor < emin) emin = coor;
393  if (coor > emax) emax = coor;
394  coor = cx*fMin.x() + cy*fMax.y() + cz*fMin.z() + cd;
395  if (coor < emin) emin = coor;
396  if (coor > emax) emax = coor;
397  coor = cx*fMin.x() + cy*fMin.y() + cz*fMax.z() + cd;
398  if (coor < emin) emin = coor;
399  if (coor > emax) emax = coor;
400  coor = cx*fMax.x() + cy*fMin.y() + cz*fMax.z() + cd;
401  if (coor < emin) emin = coor;
402  if (coor > emax) emax = coor;
403  coor = cx*fMax.x() + cy*fMax.y() + cz*fMax.z() + cd;
404  if (coor < emin) emin = coor;
405  if (coor > emax) emax = coor;
406  coor = cx*fMin.x() + cy*fMax.y() + cz*fMax.z() + cd;
407  if (coor < emin) emin = coor;
408  if (coor > emax) emax = coor;
409  }
410  else
411  {
412  std::vector<const G4ThreeVectorList*>::const_iterator ibase;
413  for (ibase = fPolygons->begin(); ibase != fPolygons->end(); ibase++)
414  {
415  G4ThreeVectorList::const_iterator ipoint;
416  for (ipoint = (*ibase)->begin(); ipoint != (*ibase)->end(); ipoint++)
417  {
418  G4double coor = ipoint->x()*cx + ipoint->y()*cy + ipoint->z()*cz + cd;
419  if (coor < emin) emin = coor;
420  if (coor > emax) emax = coor;
421  }
422  }
423  }
424  pMin = emin - delta;
425  pMax = emax + delta;
426  return true;
427  }
428 
429  // Check if the sphere surrounding the bounding box is outside
430  // the voxel limits
431  //
432  if (center.x()-radius > xmaxlim) return false;
433  if (center.y()-radius > ymaxlim) return false;
434  if (center.z()-radius > zmaxlim) return false;
435  if (center.x()+radius < xminlim) return false;
436  if (center.y()+radius < yminlim) return false;
437  if (center.z()+radius < zminlim) return false;
438 
439  // Allocate memory for transformed polygons
440  //
441  G4int nbases = (fPolygons == 0) ? 2 : fPolygons->size();
442  std::vector<G4Polygon3D*> bases(nbases);
443  if (fPolygons == 0)
444  {
445  bases[0] = new G4Polygon3D(4);
446  bases[1] = new G4Polygon3D(4);
447  }
448  else
449  {
450  for (G4int i=0; i<nbases; ++i)
451  {
452  bases[i] = new G4Polygon3D((*fPolygons)[i]->size());
453  }
454  }
455 
456  // Transform vertices
457  //
458  TransformVertices(pTransform3D, bases);
459 
460  // Create adjusted G4VoxelLimits box. New limits are extended by
461  // delta, kCarTolerance multiplied by max scale factor of
462  // the transformation
463  //
464  EAxis axis[] = { kXAxis,kYAxis,kZAxis };
465  G4VoxelLimits limits; // default is unlimited
466  for (G4int i=0; i<3; ++i)
467  {
468  if (pVoxelLimits.IsLimited(axis[i]))
469  {
470  G4double emin = pVoxelLimits.GetMinExtent(axis[i]) - delta;
471  G4double emax = pVoxelLimits.GetMaxExtent(axis[i]) + delta;
472  limits.AddLimit(axis[i], emin, emax);
473  }
474  }
475 
476  // Main loop along the set of prisms
477  //
478  G4Segment3D extent;
479  extent.first = G4Point3D( kInfinity, kInfinity, kInfinity);
480  extent.second = G4Point3D(-kInfinity,-kInfinity,-kInfinity);
481  for (G4int k=0; k<nbases-1; ++k)
482  {
483  // Find bounding box of current prism
484  G4Polygon3D* baseA = bases[k];
485  G4Polygon3D* baseB = bases[k+1];
486  G4Segment3D prismAABB;
487  GetPrismAABB(*baseA, *baseB, prismAABB);
488 
489  // Check if prismAABB is completely within the voxel limits
490  if (prismAABB.first.x() >= limits.GetMinXExtent() &&
491  prismAABB.first.y() >= limits.GetMinYExtent() &&
492  prismAABB.first.z() >= limits.GetMinZExtent() &&
493  prismAABB.second.x()<= limits.GetMaxXExtent() &&
494  prismAABB.second.y()<= limits.GetMaxYExtent() &&
495  prismAABB.second.z()<= limits.GetMaxZExtent())
496  {
497  if (extent.first.x() > prismAABB.first.x())
498  extent.first.setX( prismAABB.first.x() );
499  if (extent.first.y() > prismAABB.first.y())
500  extent.first.setY( prismAABB.first.y() );
501  if (extent.first.z() > prismAABB.first.z())
502  extent.first.setZ( prismAABB.first.z() );
503  if (extent.second.x() < prismAABB.second.x())
504  extent.second.setX(prismAABB.second.x());
505  if (extent.second.y() < prismAABB.second.y())
506  extent.second.setY(prismAABB.second.y());
507  if (extent.second.z() < prismAABB.second.z())
508  extent.second.setZ(prismAABB.second.z());
509  continue;
510  }
511 
512  // Check if prismAABB is outside the voxel limits
513  if (prismAABB.first.x() > limits.GetMaxXExtent()) continue;
514  if (prismAABB.first.y() > limits.GetMaxYExtent()) continue;
515  if (prismAABB.first.z() > limits.GetMaxZExtent()) continue;
516  if (prismAABB.second.x() < limits.GetMinXExtent()) continue;
517  if (prismAABB.second.y() < limits.GetMinYExtent()) continue;
518  if (prismAABB.second.z() < limits.GetMinZExtent()) continue;
519 
520  // Clip edges of the prism by adjusted G4VoxelLimits box
521  std::vector<G4Segment3D> vecEdges;
522  CreateListOfEdges(*baseA, *baseB, vecEdges);
523  if (ClipEdgesByVoxel(vecEdges, limits, extent)) continue;
524 
525  // Some edges of the prism are completely outside of the voxel
526  // limits, clip selected edges (see bits) of adjusted G4VoxelLimits
527  // by the prism
528  G4int bits = 0x000;
529  if (limits.GetMinXExtent() < prismAABB.first.x())
530  bits |= 0x988; // 1001 1000 1000
531  if (limits.GetMaxXExtent() > prismAABB.second.x())
532  bits |= 0x622; // 0110 0010 0010
533 
534  if (limits.GetMinYExtent() < prismAABB.first.y())
535  bits |= 0x311; // 0011 0001 0001
536  if (limits.GetMaxYExtent() > prismAABB.second.y())
537  bits |= 0xC44; // 1100 0100 0100
538 
539  if (limits.GetMinZExtent() < prismAABB.first.z())
540  bits |= 0x00F; // 0000 0000 1111
541  if (limits.GetMaxZExtent() > prismAABB.second.z())
542  bits |= 0x0F0; // 0000 1111 0000
543  if (bits == 0xFFF) continue;
544 
545  std::vector<G4Plane3D> vecPlanes;
546  CreateListOfPlanes(*baseA, *baseB, vecPlanes);
547  ClipVoxelByPlanes(bits, limits, vecPlanes, prismAABB, extent);
548  } // End of the main loop
549 
550  // Free memory
551  //
552  for (G4int i=0; i<nbases; ++i) { delete bases[i]; bases[i] = 0; }
553 
554  // Final adjustment of the extent
555  //
556  G4double emin = 0, emax = 0;
557  if (pAxis == kXAxis) { emin = extent.first.x(); emax = extent.second.x(); }
558  if (pAxis == kYAxis) { emin = extent.first.y(); emax = extent.second.y(); }
559  if (pAxis == kZAxis) { emin = extent.first.z(); emax = extent.second.z(); }
560 
561  if (emin > emax) return false;
562  emin -= delta;
563  emax += delta;
564  G4double minlim = pVoxelLimits.GetMinExtent(pAxis);
565  G4double maxlim = pVoxelLimits.GetMaxExtent(pAxis);
566  pMin = (emin < minlim) ? minlim-kCarTolerance : emin;
567  pMax = (emax > maxlim) ? maxlim+kCarTolerance : emax;
568  return true;
569 }

573 //
574 // Find max scale factor of the transformation
575 //
576 G4double
577 G4BoundingEnvelope::FindScaleFactor(const G4Transform3D& pTransform3D) const
578 {
579  if (pTransform3D.xx() == 1. &&

749 // Create list of planes bounding a prism
750 //
751 void
752 G4BoundingEnvelope::CreateListOfPlanes(const G4Polygon3D& baseA,
753  const G4Polygon3D& baseB,
754  std::vector<G4Plane3D>& pPlanes) const
755 {
756  // Find centers of the bases and internal point of the prism
757  //
758  G4int na = baseA.size();
759  G4int nb = baseB.size();

761  G4Normal3D norm;
762  for (G4int i=0; i<na; ++i) pa += baseA[i];
763  for (G4int i=0; i<nb; ++i) pb += baseB[i];
764  pa /= na; pb /= nb; p0 = (pa+pb)/2.;
765 
766  // Create list of planes
767  //
768  pPlanes.clear();

770  {
771  G4int k = na - 1;
772  for (G4int i=0; i<na; ++i)
773  {
774  norm = (baseB[k]-baseA[i]).cross(baseA[k]-baseB[i]);
775  if (norm.mag2() > kCarTolerance)
776  {
777  pPlanes.push_back(G4Plane3D(norm,baseA[i]));
778  }
779  k = i;
780  }
781  norm = (baseA[2]-baseA[0]).cross(baseA[1]-pa);
782  if (norm.mag2() > kCarTolerance)
783  {
784  pPlanes.push_back(G4Plane3D(norm,pa));
785  }
786  norm = (baseB[2]-baseB[0]).cross(baseB[1]-pb);
787  if (norm.mag2() > kCarTolerance)
788  {
789  pPlanes.push_back(G4Plane3D(norm,pb));
790  }
791  }
792  else if (nb == 1) // baseB has one vertex
793  {
794  G4int k = na - 1;
795  for (G4int i=0; i<na; ++i)
796  {
797  norm = (baseA[i]-baseB[0]).cross(baseA[k]-baseB[0]);
798  if (norm.mag2() > kCarTolerance)
799  {
800  pPlanes.push_back(G4Plane3D(norm,baseB[0]));
801  }
802  k = i;
803  }
804  norm = (baseA[2]-baseA[0]).cross(baseA[1]-pa);
805  if (norm.mag2() > kCarTolerance)
806  {
807  pPlanes.push_back(G4Plane3D(norm,pa));
808  }
809  }
810  else if (na == 1) // baseA has one vertex
811  {
812  G4int k = nb - 1;
813  for (G4int i=0; i<nb; ++i)
814  {
815  norm = (baseB[i]-baseA[0]).cross(baseB[k]-baseA[0]);
816  if (norm.mag2() > kCarTolerance)
817  {
818  pPlanes.push_back(G4Plane3D(norm,baseA[0]));
819  }
820  k = i;
821  }
822  norm = (baseB[2]-baseB[0]).cross(baseB[1]-pb);
823  if (norm.mag2() > kCarTolerance)
824  {
825  pPlanes.push_back(G4Plane3D(norm,pb));
826  }
827  }
828

830  //
831  G4int nplanes = pPlanes.size();
832  for (G4int i=0; i<nplanes; ++i)
833  {
834  pPlanes[i].normalize();
835  if (pPlanes[i].distance(p0) > 0)
836  {
837  pPlanes[i] = G4Plane3D(-pPlanes[i].a(),-pPlanes[i].b(),
838  -pPlanes[i].c(),-pPlanes[i].d());
839  }
840  }
841 }
