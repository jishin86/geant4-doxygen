49 G4AntiHe3* G4AntiHe3::Definition()
50 {
51  if (theInstance !=0) return theInstance;

53  // search in particle table]
54  G4ParticleTable* pTable = G4ParticleTable::GetParticleTable();
55  G4Ions* anInstance = reinterpret_cast<G4Ions*>(pTable->FindParticle(name));
56  if (anInstance ==0)
57  {
58  // create particle
59  //
60  // Arguments for constructor are as follows
61  // name mass width charge
62  // 2*spin parity C-conjugation
63  // 2*Isospin 2*Isospin3 G-parity
64  // type lepton number baryon number PDG encoding
65  // stable lifetime decay table
66  // shortlived subType anti_encoding
67  // excitation
68  anInstance = new G4Ions(
69  name, 2.808391*GeV, 0.0*MeV, -2.0*eplus,
70  1, +1, 0,
71  0, 0, 0,
72  "anti_nucleus", 0, -3, -1000020030,
73  true, -1.0, NULL,
74  false, "static", 1000020030,
75  0.0, 0
76  );
77 
78  // Magnetic Moment
79  G4double mN = eplus*hbar_Planck/2./(proton_mass_c2 /c_squared);
80  anInstance->SetPDGMagneticMoment( 2.12762485 * mN);
81 
82  }
83 
84  theInstance = reinterpret_cast<G4AntiHe3*>(anInstance);
85  return theInstance;
