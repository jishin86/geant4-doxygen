136 void G4WorkerRunManager::RunInitialization()
137 {
138  TIMEMORY_AUTO_TIMER("");
139 #ifdef G4MULTITHREADED
140  if(!visIsSetUp)
141  {
142  G4VVisManager* pVVis = G4VVisManager::GetConcreteInstance();
143  if(pVVis)
144  {
145  pVVis->SetUpForAThread();
146  visIsSetUp = true;
147  }
148  }
149 #endif
150 
151  if(!(kernel->RunInitialization(fakeRun))) return;
152 
153  //Signal this thread can start event loop.
154  //Note this will return only when all threads reach this point
155  G4MTRunManager::GetMasterRunManager()->ThisWorkerReady();
156  if(fakeRun) return;
157 
158  const G4UserWorkerInitialization* uwi
159  = G4MTRunManager::GetMasterRunManager()->GetUserWorkerInitialization();
160  CleanUpPreviousEvents();
161  if(currentRun) delete currentRun;
162  currentRun = 0;
163 
164  if(fGeometryHasBeenDestroyed) G4ParallelWorldProcessStore::GetInstance()->UpdateWorlds();
165  //Call a user hook: this is guaranteed all threads are "synchronized"
166  if(uwi) uwi->WorkerRunStart();
167 
168  if(userRunAction) currentRun = userRunAction->GenerateRun();
169  if(!currentRun) currentRun = new G4Run();
170 
171  currentRun->SetRunID(runIDCounter);
172  currentRun->SetNumberOfEventToBeProcessed(numberOfEventToBeProcessed);
173 
174  currentRun->SetDCtable(DCtable);
175  G4SDManager* fSDM = G4SDManager::GetSDMpointerIfExist();
176  if(fSDM)
177  { currentRun->SetHCtable(fSDM->GetHCtable()); }
178 
179  if ( G4VScoreNtupleWriter::Instance() )
180  {
181  auto hce = fSDM->PrepareNewEvent();
182  isScoreNtupleWriter = G4VScoreNtupleWriter::Instance()->Book(hce);
183  delete hce;
184  }
185 
186  std::ostringstream oss;
187  G4Random::saveFullState(oss);
188  randomNumberStatusForThisRun = oss.str();
189  currentRun->SetRandomNumberStatus(randomNumberStatusForThisRun);
190 
191  for(G4int i_prev=0;i_prev<n_perviousEventsToBeStored;i_prev++)
192  { previousEvents->push_back((G4Event*)0); }
193 
194  if(printModulo>0 || verboseLevel>0)
195  {
196  G4cout << "### Run " << currentRun->GetRunID() << " starts on worker thread "
197  << G4Threading::G4GetThreadId() << "." << G4endl;
198  }
199  if(userRunAction) userRunAction->BeginOfRunAction(currentRun);
200 
201  if (isScoreNtupleWriter) {
202  G4VScoreNtupleWriter::Instance()->OpenFile();
203  }
204 
205  if(storeRandomNumberStatus) {
206  G4String fileN = "currentRun";
207  if ( rngStatusEventsFlag ) {
208  std::ostringstream os;
209  os << "run" << currentRun->GetRunID();
210  fileN = os.str();
211  }
212  StoreRNGStatus(fileN);
213  }
214 
215  runAborted = false;
216  numberOfEventProcessed = 0;
217 }

219 void G4WorkerRunManager::DoEventLoop(G4int n_event, const char* macroFile , G4int n_select)
220 {
221  TIMEMORY_AUTO_TIMER("");
222  if(!userPrimaryGeneratorAction)
223  {
224  G4Exception("G4RunManager::GenerateEvent()", "Run0032", FatalException,
225  "G4VUserPrimaryGeneratorAction is not defined!");
226  }
227 
228  //This is the same as in the sequential case, just the for-loop indexes are
229  //different
230  InitializeEventLoop(n_event,macroFile,n_select);
231 
232  // Reset random number seeds queue
233  while(seedsQueue.size()>0)
234  { seedsQueue.pop(); }
235  // for each run, worker should receive at least one set of random number seeds.
236  runIsSeeded = false;
237 
238  // Event loop
239  eventLoopOnGoing = true;
241  G4int i_event = -1;
242  nevModulo = -1;
243  currEvID = -1;
244 
245  while(eventLoopOnGoing)
246  {
247  ProcessOneEvent(i_event);
248  if(eventLoopOnGoing)
249  {
250  TerminateOneEvent();
251  if(runAborted)
252  { eventLoopOnGoing = false; }
258  }
259  }
260 
261  TerminateEventLoop();
262 }

277 G4Event* G4WorkerRunManager::GenerateEvent(G4int i_event)
278 {
279  TIMEMORY_AUTO_TIMER("");
280  G4Event* anEvent = new G4Event(i_event);
281  long s1 = 0;
282  long s2 = 0;
283  long s3 = 0;
284  G4bool eventHasToBeSeeded = true;
285  if(G4MTRunManager::SeedOncePerCommunication()==1 && runIsSeeded)
286  { eventHasToBeSeeded = false; }
287 
288  if(i_event<0)
289  {
290  G4int nevM = G4MTRunManager::GetMasterRunManager()->GetEventModulo();
291  if(nevM==1)
292  {
293  eventLoopOnGoing = G4MTRunManager::GetMasterRunManager()
294  ->SetUpAnEvent(anEvent,s1,s2,s3,eventHasToBeSeeded);
295  runIsSeeded = true;
296  }
297  else
298  {
299  if(nevModulo<=0)
300  {
301  G4int nevToDo = G4MTRunManager::GetMasterRunManager()
302  ->SetUpNEvents(anEvent,&seedsQueue,eventHasToBeSeeded);
303  if(nevToDo==0)
304  { eventLoopOnGoing = false; }
305  else
306  {
307  currEvID = anEvent->GetEventID();
308  nevModulo = nevToDo - 1;
309  }
310  }
311  else
312  {
313  if(G4MTRunManager::SeedOncePerCommunication()>0) eventHasToBeSeeded = false;
314  anEvent->SetEventID(++currEvID);
315  nevModulo--;
316  }
317  if(eventLoopOnGoing && eventHasToBeSeeded)
318  {
319  s1 = seedsQueue.front(); seedsQueue.pop();
320  s2 = seedsQueue.front(); seedsQueue.pop();
321  }
322  }
323 
324  if(!eventLoopOnGoing)
325  {
326  delete anEvent;
327  return 0;
328  }
329  }
330  else if(eventHasToBeSeeded)
331  {
332  //Need to reseed random number generator
333  G4RNGHelper* helper = G4RNGHelper::GetInstance();
334  s1 = helper->GetSeed(i_event*2);
335  s2 = helper->GetSeed(i_event*2+1);
336  }
337 
338  if(eventHasToBeSeeded)
339  {
340  long seeds[3] = { s1, s2, 0 };
341  G4Random::setTheSeeds(seeds,-1);
342  runIsSeeded = true;
344  }
345 
346  //Read from file seed.
347  //Andrea Dotti 4 November 2015
348  //This is required for strong-reproducibility, in MT mode we have that each
349  //thread produces, for each event a status file, we want to do that.
350  //Search a random file with the format run{%d}evt{%d}.rndm
351 
352  //This is the filename base constructed from run and event
353  const auto filename = [&] {
354  std::ostringstream os;
355  os << "run"<<currentRun->GetRunID() << "evt" << anEvent->GetEventID();
356  return os.str();
357  };
358 
359  G4bool RNGstatusReadFromFile = false;
360  if ( readStatusFromFile ) {
361  //Build full path of RNG status file for this event
362  std::ostringstream os;
363  os << filename() << ".rndm";
364  const G4String& randomStatusFile = os.str();
365  std::ifstream ifile(randomStatusFile.c_str());
366  if ( ifile ) { //File valid and readable
367  RNGstatusReadFromFile = true;
368  G4Random::restoreEngineStatus(randomStatusFile.c_str());
369  }
370  }
371 
372 
373  if(storeRandomNumberStatusToG4Event==1 || storeRandomNumberStatusToG4Event==3)
374  {
375  std::ostringstream oss;
376  G4Random::saveFullState(oss);
377  randomNumberStatusForThisEvent = oss.str();
378  anEvent->SetRandomNumberStatus(randomNumberStatusForThisEvent);
379  }
380 
381  if(storeRandomNumberStatus && ! RNGstatusReadFromFile ) { //If reading from file, avoid to rewrite the same
382  G4String fileN = "currentEvent";
383  if ( rngStatusEventsFlag ) {
384  fileN = filename();
385  }
386  StoreRNGStatus(fileN);
387  }
388 
389  if(printModulo > 0 && anEvent->GetEventID()%printModulo == 0 )
390  {
391  G4cout << "--> Event " << anEvent->GetEventID() << " starts";
392  if(eventHasToBeSeeded)
393  { G4cout << " with initial seeds (" << s1 << "," << s2 << ")"; }
394  G4cout << "." << G4endl;
395  }
396  userPrimaryGeneratorAction->GeneratePrimaries(anEvent);
397  return anEvent;
398 }

461 void G4WorkerRunManager::ConstructScoringWorlds()
462 {
463  // Return if unnecessary
464  G4ScoringManager* ScM = G4ScoringManager::GetScoringManagerIfExist();
465  if(!ScM) return;
466  G4int nPar = ScM->GetNumberOfMesh();
467  if(nPar<1) return;
468 
469  // Update thread-local G4TransportationManager of all the world volumes
470  kernel->WorkerUpdateWorldVolume();
471 
472  G4ScoringManager* masterScM = G4MTRunManager::GetMasterScoringManager();
473  assert( masterScM != NULL );
474 
475  G4ParticleTable::G4PTblDicIterator* particleIterator
476  = G4ParticleTable::GetParticleTable()->GetIterator();
477 
478  for(G4int iw=0;iw<nPar;iw++)
479  {
480  G4VScoringMesh* mesh = ScM->GetMesh(iw);
481  if(fGeometryHasBeenDestroyed) mesh->GeometryHasBeenDestroyed();
482  G4VPhysicalVolume* pWorld
483  = G4TransportationManager::GetTransportationManager()
484  ->IsWorldExisting(ScM->GetWorldName(iw));
485  if(!pWorld)
486  {
487  G4ExceptionDescription ed;
488  ed<<"Mesh name <"<<ScM->GetWorldName(iw)<<"> is not found in the master thread.";
489  G4Exception("G4WorkerRunManager::ConstructScoringWorlds()","RUN79001",
490  FatalException,ed);
491  }
492  if(!(mesh->GetMeshElementLogical()))
493  {
494  G4AutoLock l(&ConstructScoringWorldsMutex);
495  G4VScoringMesh* masterMesh = masterScM->GetMesh(iw);
496  mesh->SetMeshElementLogical(masterMesh->GetMeshElementLogical());
497  l.unlock();
498 
499  G4ParallelWorldProcess* theParallelWorldProcess = mesh->GetParallelWorldProcess();
500  if(theParallelWorldProcess)
501  { theParallelWorldProcess->SetParallelWorld(ScM->GetWorldName(iw)); }
502  else
503  {
504  theParallelWorldProcess = new G4ParallelWorldProcess(ScM->GetWorldName(iw));
505  mesh->SetParallelWorldProcess(theParallelWorldProcess);
506  theParallelWorldProcess->SetParallelWorld(ScM->GetWorldName(iw));
507 
508  particleIterator->reset();
509  while( (*particleIterator)() ){
510  G4ParticleDefinition* particle = particleIterator->value();
511  G4ProcessManager* pmanager = particle->GetProcessManager();
512  if(pmanager)
513  {
514  pmanager->AddProcess(theParallelWorldProcess);
515  if(theParallelWorldProcess->IsAtRestRequired(particle))
516  { pmanager->SetProcessOrdering(theParallelWorldProcess, idxAtRest, 9900); }
517  pmanager->SetProcessOrderingToSecond(theParallelWorldProcess, idxAlongStep);
518  pmanager->SetProcessOrdering(theParallelWorldProcess, idxPostStep, 9900);
519  } //if(pmanager)
520  }//while
521  }
522  }
523  mesh->WorkerConstruct(pWorld);
524  }
525 }

606 void G4WorkerRunManager::DoWork()
607 {
608  TIMEMORY_AUTO_TIMER("");
609  G4MTRunManager* mrm = G4MTRunManager::GetMasterRunManager();
610  G4MTRunManager::WorkerActionRequest nextAction = mrm->ThisWorkerWaitForNextAction();
611  while( nextAction != G4MTRunManager::WorkerActionRequest::ENDWORKER )
612  {
613  if( nextAction == G4MTRunManager::WorkerActionRequest::NEXTITERATION ) // start the next run
614  {
615  //The following code deals with changing materials between runs
616  static G4ThreadLocal G4bool skipInitialization = true;
617  if(skipInitialization)
618  {
619  // re-initialization is not necessary for the first run
620  skipInitialization = false;
621  }
622  else
623  {
624 // ReinitializeGeometry();
625  workerContext->UpdateGeometryAndPhysicsVectorFromMaster();
626  }
627 
628  // Execute UI commands stored in the master UI manager
629  std::vector<G4String> cmds = mrm->GetCommandStack();
630  G4UImanager* uimgr = G4UImanager::GetUIpointer(); //TLS instance
631  std::vector<G4String>::const_iterator it = cmds.begin();
632  for(;it!=cmds.end();it++)
633  { uimgr->ApplyCommand(*it); }
634  //Start this run
635  G4int numevents = mrm->GetNumberOfEventsToBeProcessed();
636  G4String macroFile = mrm->GetSelectMacro();
637  G4int numSelect = mrm->GetNumberOfSelectEvents();
638  if ( macroFile == "" || macroFile == " " )
639  {
640  this->BeamOn(numevents);
641  }
642  else
643  {
644  this->BeamOn(numevents,macroFile,numSelect);
645  }
646  }
