67 G4Diproton* G4Diproton::Definition() {

71  G4ParticleTable* pTable = G4ParticleTable::GetParticleTable();
72  G4bool tableReady = pTable->GetReadiness();

74  pTable->Remove(theInstance); // Make invisible to GEANT4
