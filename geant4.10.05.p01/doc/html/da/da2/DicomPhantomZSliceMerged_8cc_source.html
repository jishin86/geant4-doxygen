60 void DicomPhantomZSliceMerged::CheckSlices()
61 {
62  G4cout << "\nDicomPhantomZSliceMerged::Checking "
63  << fSlices.size() << " fSlices..." << G4endl;
64 
65  G4bool uniformSliceThickness = true;
66 
67  if(fSlices.size() > 1) {
68  if(fSlices.size() == 2) {
69  DicomPhantomZSliceHeader* one = fSlices.begin()->second;
70  DicomPhantomZSliceHeader* two = fSlices.end()->second;
71 
72  G4double real_distance = (two->GetSliceLocation()-one->GetSliceLocation())/2.;
73 
74  if(one->GetMaxZ() != two->GetMinZ()) {
75  one->SetMaxZ(one->GetSliceLocation()+real_distance);
76  two->SetMinZ(two->GetSliceLocation()-real_distance);
77  //one->SetMinZ(one->GetSliceLocation()-real_distance);
78  //two->SetMaxZ(two->GetSliceLocation()+real_distance);
79  if(uniformSliceThickness) {
80  one->SetMinZ(one->GetSliceLocation()-real_distance);
81  two->SetMaxZ(two->GetSliceLocation()+real_distance);
82  }
83  }
84  } else {
85  std::map<G4double,DicomPhantomZSliceHeader*>::iterator ite0 = fSlices.begin();
86  std::map<G4double,DicomPhantomZSliceHeader*>::iterator ite1 = fSlices.begin();
87  std::map<G4double,DicomPhantomZSliceHeader*>::iterator ite2 = fSlices.begin();
88  ++ite1;
89  ++ite2; ++ite2;
90 
91  for(; ite2 != fSlices.end(); ++ite0, ++ite1, ++ite2) {
92  DicomPhantomZSliceHeader* prev = ite0->second;
93  DicomPhantomZSliceHeader* slice = ite1->second;
94  DicomPhantomZSliceHeader* next = ite2->second;
95  G4double real_max_distance = (next->GetSliceLocation() -
96  slice->GetSliceLocation())/2.;
97  G4double real_min_distance = (slice->GetSliceLocation() -
98  prev->GetSliceLocation())/2.;
99  G4double real_distance = real_max_distance + real_min_distance;
100  G4double stated_distance = slice->GetMaxZ()-slice->GetMinZ();
101  if(real_distance != stated_distance) {
102  unsigned int sliceNum = std::distance(fSlices.begin(),ite1);
103  G4cout << "\tDicomPhantomZSliceMerged::CheckSlices - \
104  Slice Distance Error in slice [" << sliceNum
105  << "]: Real Distance = "
106  << real_distance/mm
107  << " mm, Stated Distance = " << stated_distance/mm << G4endl;
108  slice->SetMinZ(slice->GetSliceLocation()-real_min_distance);
109  slice->SetMaxZ(slice->GetSliceLocation()+real_max_distance);
110 
111  if(ite0 == fSlices.begin()) {
112  prev->SetMaxZ(slice->GetMinZ());
113  // Using below would make all slice same thickness
114  //prev->SetMinZ(prev->GetSliceLocation()-real_min_distance);
115  if(uniformSliceThickness) {
116  prev->SetMinZ(prev->GetSliceLocation()-real_min_distance); }
117 
118  }
119  if(static_cast<unsigned int>(std::distance(fSlices.begin(),ite2)+1)==
120  fSlices.size()) {
121  next->SetMinZ(slice->GetMaxZ());
122  // Using below would make all slice same thickness
123  //next->SetMaxZ(next->GetSliceLocation()+real_max_distance);
124  if(uniformSliceThickness) {
125  next->SetMaxZ(next->GetSliceLocation()+real_max_distance); }
126  }
127  }
128  }
129  }
130  }
131  G4cout << G4endl;
132 
133  std::map<G4double,DicomPhantomZSliceHeader*>::iterator ite = fSlices.begin();
134  for(ite = fSlices.begin(); ite != fSlices.end(); ++ite) {
135  ite->second->DumpToFile();
136  }
137 
138 }
