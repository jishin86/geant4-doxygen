106 G4VITStepModel* G4ITModelManager::GetModel(const G4double globalTime)
107 {
108  if (!fModels.empty())
109  {
110  mapModels::reverse_iterator rit = fModels.rbegin();
111  if (rit != fModels.rend())
112  {
113  if (globalTime > rit->first)
114  {
115  return rit->second;
116  }
117  else
118  {
119  mapModels::iterator it = fModels.begin();
120 
121  if (globalTime < it->first)
122  {
123  G4ExceptionDescription exceptionDescription;
124  exceptionDescription << "No model was found at time ";
125  exceptionDescription << G4BestUnit(globalTime, "Time");
126  exceptionDescription << ". The first model is registered at time : ";
127  exceptionDescription << G4BestUnit(it->first, "Time") << ". ";
128  G4Exception("G4ITModelManager::GetModel", "ITModelManager003",
129  FatalErrorInArgument, exceptionDescription);
130  }
131 
132  it = fModels.lower_bound(globalTime);
133 
134  if (it != fModels.end()) return it->second;
135  }
136  }
137  }
138 
139  G4ExceptionDescription exceptionDescription;
140  exceptionDescription << "No model was found.";
141  G4Exception("G4ITModelManager::GetModel", "ITModelManager004",
142  FatalErrorInArgument, exceptionDescription);
