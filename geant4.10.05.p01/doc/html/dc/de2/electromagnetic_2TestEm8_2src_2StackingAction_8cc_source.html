62 StackingAction::ClassifyNewTrack(const G4Track* aTrack)
63 {
64  //stack or delete secondaries
65  G4ClassificationOfNewTrack status = fUrgent;
66 
67  //keep primary particle
68  if (aTrack->GetParentID() == 0 || !fKillSecondary) { return status; }
69 
70  Run* run
71  = static_cast<Run*>(G4RunManager::GetRunManager()->GetNonConstCurrentRun());
72 
73  // charged tracks are killed only inside sensitive volumes
74  if(aTrack->GetVolume()->GetLogicalVolume()->GetSensitiveDetector() &&
75  aTrack->GetDefinition()->GetPDGCharge() != 0.0)
76  {
77  run->AddEnergy(aTrack->GetKineticEnergy(), 0);
78  status = fKill;
79  }
80  return status;
81 }
