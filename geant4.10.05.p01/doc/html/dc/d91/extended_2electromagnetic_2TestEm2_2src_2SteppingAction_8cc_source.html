53 void SteppingAction::UserSteppingAction(const G4Step* step)
54 {
55  // energy deposit
56  //
57  G4double dEStep = step->GetTotalEnergyDeposit();
58  Run* run
59  = static_cast<Run*>(G4RunManager::GetRunManager()->GetNonConstCurrentRun());
60  run->AddStep(step->GetTrack()->GetDefinition()->GetPDGCharge());
61  if (dEStep > 0.) {
62  G4ThreeVector prePoint = step->GetPreStepPoint()->GetPosition();
63  G4ThreeVector delta = step->GetPostStepPoint()->GetPosition() - prePoint;
64  prePoint += G4UniformRand()*delta;
65  G4double x = prePoint.x(), y = prePoint.y(), z = prePoint.z();
66  G4double radius = std::sqrt(x*x + y*y);
67  G4double offset = 0.5*fDetector->GetfullLength();
68  G4int SlideNb = G4int((z + offset)/fDetector->GetdLlength());
69  G4int RingNb = G4int(radius/fDetector->GetdRlength());
70  run->FillPerStep(dEStep,SlideNb,RingNb);
71  }
72 }
