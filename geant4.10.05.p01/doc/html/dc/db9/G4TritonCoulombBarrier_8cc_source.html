37 {
38  // Data comes from
39  // Dostrovsky, Fraenkel and Friedlander
40  // Physical Review, vol 116, num. 3 1959
41  //
42  // const G4int size = 5;

44  // const G4double Kprot[size] = {0.42, 0.58, 0.68, 0.77, 0.80};
45  //
46  // K for Triton is K for protons + 0.12
47  G4double K = (aZ >= 70) ? 0.80 :
48  (((0.2357e-5*aZ) - 0.42679e-3)*aZ + 0.27035e-1)*aZ + 0.19025;
49 
50  return K+0.12;
