90 void RunAction::EndOfRunAction(const G4Run* aRun)
91 {
92  G4int NbOfEvents = aRun->GetNumberOfEvent();
93  if (NbOfEvents == 0) return;
94 
95  std::ios::fmtflags mode = G4cout.flags();
96  G4int prec = G4cout.precision(2);
97 
98  G4Material* material = fDetector->GetMaterial();
99  G4double length = fDetector->GetSize();
100  G4double density = material->GetDensity();
101 
102  G4String particle = fPrimary->GetParticleGun()->GetParticleDefinition()
103  ->GetParticleName();
104  G4double energy = fPrimary->GetParticleGun()->GetParticleEnergy();
105 
106  G4cout << "\n The run consists of " << NbOfEvents << " "<< particle << " of "
107  << G4BestUnit(energy,"Energy") << " through "
108  << G4BestUnit(length,"Length") << " of "
109  << material->GetName() << " (density: "
110  << G4BestUnit(density,"Volumic Mass") << ")" << G4endl;
111 
112  //total number of process calls
113  G4double countTot = 0.;
114  G4cout << "\n Number of process calls --->";
115  for (size_t i=0; i< fProcCounter->size();i++) {
116  G4String procName = (*fProcCounter)[i]->GetName();
117  if (procName != "Transportation") {
118  G4int count = (*fProcCounter)[i]->GetCounter();
119  G4cout << "\t" << procName << " : " << count;
120  countTot += count;
121  }
122  }
123  G4cout << G4endl;
124 
125  //compute totalCrossSection, meanFreePath and massicCrossSection
126  //
127  G4double totalCrossSection = countTot/(NbOfEvents*length);
128  G4double MeanFreePath = 1./totalCrossSection;
129  G4double massCrossSection =totalCrossSection/density;
130 
131  G4cout.precision(5);
132  G4cout << "\n Simulation: "
133  << "total CrossSection = " << totalCrossSection*cm << " /cm"
134  << "\t MeanFreePath = " << G4BestUnit(MeanFreePath,"Length")
135  << "\t massicCrossSection = " << massCrossSection*g/cm2 << " cm2/g"
136  << G4endl;
137 
138  //compute theoritical predictions
139  //
140  if(particle == "mu+" || particle == "mu-") {
141  totalCrossSection = 0.;
142  for (size_t i=0; i< fProcCounter->size();i++) {
143  G4String procName = (*fProcCounter)[i]->GetName();
144  if (procName != "Transportation") {
145  totalCrossSection += ComputeTheory(procName, NbOfEvents);
146  }
147  }
148 
149  MeanFreePath = 1./totalCrossSection;
150  massCrossSection = totalCrossSection/density;
151 
152  G4cout << " Theory: "
153  << "total CrossSection = " << totalCrossSection*cm << " /cm"
154  << "\t MeanFreePath = " << G4BestUnit(MeanFreePath,"Length")
155  << "\t massicCrossSection = " << massCrossSection*g/cm2 << " cm2/g"
156  << G4endl;
157  }
158 
159  G4cout.setf(mode,std::ios::floatfield);
160  G4cout.precision(prec);
161 
162  // delete and remove all contents in fProcCounter
163  while (fProcCounter->size()>0){
164  OneProcessCount* aProcCount=fProcCounter->back();
165  fProcCounter->pop_back();
166  delete aProcCount;
167  }
168  delete fProcCounter;
169 
170  fHistoManager->Save();
171 
172  // show Rndm status
173  CLHEP::HepRandom::showEngineStatus();
174 }

178 G4double RunAction::ComputeTheory(G4String process, G4int NbOfMu)
179 {
180  G4Material* material = fDetector->GetMaterial();
181  G4double ekin = fPrimary->GetParticleGun()->GetParticleEnergy();
182  MuCrossSections crossSections;
183 
184  G4int id = 0; G4double cut = 0.;
185  if (process == "muIoni") {id = 11; cut = GetEnergyCut(material,1);}
186  else if (process == "muPairProd") {id = 12; cut = 2*(GetEnergyCut(material,1)
187  + electron_mass_c2); }
188  else if (process == "muBrems") {id = 13; cut = GetEnergyCut(material,0);}
189  else if (process == "muonNuclear"){id = 14; }
190  if (id == 0) { return 0.; }
191 
192  G4int nbOfBins = 100;
193  G4double binMin = -10.;
194  G4double binMax = 0.;
195  G4double binWidth = (binMax-binMin)/G4double(nbOfBins);
196 
197  //create histo for theoritical crossSections, with same bining as simulation
198  //
199  G4AnalysisManager* analysisManager = G4AnalysisManager::Instance();
200 
201  G4AnaH1* histoTh = 0;
202  // G4AnaH1* histoMC = 0;
203  if (fHistoManager->HistoExist(id)) {
204  histoTh = analysisManager->GetH1(fHistoManager->GetHistoID(id));
205  //histoMC = analysisManager->GetH1(fHistoManager->GetHistoID(id-10));
206  nbOfBins = fHistoManager->GetNbins(id);
207  binMin = fHistoManager->GetVmin (id);
208  binMax = fHistoManager->GetVmax (id);
209  binWidth = fHistoManager->GetBinWidth(id);
210  }
211 
212  //compute and plot differential crossSection, as function of energy transfert.
213  //compute and return integrated crossSection for a given process.
214  //(note: to compare with simulation, the integrated crossSection is function
215  // of the energy cut.)
216  //
217  G4double lgeps, etransf, sigmaE, dsigma;
218  G4double sigmaTot = 0.;
219  const G4double ln10 = std::log(10.);
220  G4double length = fDetector->GetSize();
221 
222  for (G4int ibin=0; ibin<nbOfBins; ibin++) {
223  lgeps = binMin + (ibin+0.5)*binWidth;
224  etransf = ekin*std::pow(10.,lgeps);
225  sigmaE = crossSections.CR_Macroscopic(process,material,ekin,etransf);
226  dsigma = sigmaE*etransf*binWidth*ln10;
227  if (etransf > cut) sigmaTot += dsigma;
228  if (histoTh) {
229  G4double NbProcess = NbOfMu*length*dsigma;
230  histoTh->fill(lgeps, NbProcess);
231  }
232  }
233 
234  //return integrated crossSection
235  //
236  return sigmaTot;
237 }
