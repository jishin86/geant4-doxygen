80 void DicomIntersectVolume::SetNewValue(G4UIcommand * command,
81  G4String newValues)
82 {
83  G4AffineTransform theVolumeTransform;
84 
85  if (command == fUserVolumeCmd) {
86 
87  std::vector<G4String> params = GetWordsInString( newValues );
88  if( params.size() < 8 ) {
89  G4Exception("DicomIntersectVolume::SetNewValue",
90  " There must be at least 8 parameter: SOLID_TYPE POS_X POS_Y POS_Z "
91  "ANG_X ANG_Y ANG_Z SOLID_PARAM_1 (SOLID_PARAM_2 ...)",
92  FatalErrorInArgument,
93  G4String("Number of parameters given = " +
94  G4UIcommand::ConvertToString( G4int(params.size()) )).c_str());
95 
96  }
97 
98  //----- Build G4VSolid
99  BuildUserSolid(params);
100 
101  //----- Calculate volume inverse 3D transform
102  G4ThreeVector pos = G4ThreeVector( G4UIcommand::ConvertToDouble(params[0]),
103  G4UIcommand::ConvertToDouble(params[1]),
104  G4UIcommand::ConvertToDouble(params[2]) );
105  G4RotationMatrix* rotmat = new G4RotationMatrix;
106  std::vector<double> angles;
107  rotmat->rotateX( G4UIcommand::ConvertToDouble(params[3]) );
108  rotmat->rotateY( G4UIcommand::ConvertToDouble(params[4]) );
109  rotmat->rotateY( G4UIcommand::ConvertToDouble(params[5]) );
110  theVolumeTransform = G4AffineTransform( rotmat, pos ).Invert();
111 
112  } else if (command == fG4VolumeCmd) {
113  std::vector<G4String> params = GetWordsInString( newValues );
114  if( params.size() !=1 )
115  G4Exception("DicomIntersectVolume::SetNewValue",
116  "",
117  FatalErrorInArgument,
118  G4String("Command: "+ command->GetCommandPath() +
119  "/" + command->GetCommandName() +
120  " " + newValues +
121  " needs 1 argument: VOLUME_NAME").c_str());
122 
123  //----- Build G4VSolid
124  BuildG4Solid(params);
125 
126  //----- Calculate volume inverse 3D transform
127  G4VPhysicalVolume* pv = GetPhysicalVolumes( params[0], 1, 1)[0];
128 
129  theVolumeTransform =
130  G4AffineTransform( pv->GetFrameRotation(), pv->GetFrameTranslation() );
131  }
132 
133  //----- Calculate relative phantom - volume 3D transform
134  G4PhantomParameterisation* thePhantomParam = GetPhantomParam(true);
135 
136  G4RotationMatrix* rotph = new G4RotationMatrix();
137  // assumes the phantom mother is not rotated !!!
138  G4AffineTransform thePhantomTransform( rotph, G4ThreeVector() );
139  // assumes the phantom mother is not translated !!!
140 
141  G4AffineTransform theTransform = theVolumeTransform*thePhantomTransform;
142 
143  G4ThreeVector axisX( 1., 0., 0. );
144  G4ThreeVector axisY( 0., 1., 0. );
145  G4ThreeVector axisZ( 0., 0., 1. );
146  theTransform.ApplyAxisTransform(axisX);
147  theTransform.ApplyAxisTransform(axisY);
148  theTransform.ApplyAxisTransform(axisZ);
149 
150  //----- Write phantom header
151  G4String thePhantomFileName = "phantom.g4pdcm";
152  fout.open(thePhantomFileName);
153  std::vector<G4Material*> materials = thePhantomParam->GetMaterials();
154  fout << materials.size() << G4endl;
155  for( unsigned int ii = 0; ii < materials.size(); ii++ ) {
156  fout << ii << " " << materials[ii]->GetName() << G4endl;
157  }
158 
159  //----- Loop to pantom voxels
160  G4int nx = thePhantomParam->GetNoVoxelX();
161  G4int ny = thePhantomParam->GetNoVoxelY();
162  G4int nz = thePhantomParam->GetNoVoxelZ();
163  G4int nxy = nx*ny;
164  fVoxelIsInside = new G4bool[nx*ny*nz];
165  G4double voxelHalfWidthX = thePhantomParam->GetVoxelHalfX();
166  G4double voxelHalfWidthY = thePhantomParam->GetVoxelHalfY();
167  G4double voxelHalfWidthZ = thePhantomParam->GetVoxelHalfZ();
168 
169  //----- Write phantom dimensions and limits
170  fout << nx << " " << ny << " " << nz << G4endl;
171  fout << -voxelHalfWidthX*nx+thePhantomTransform.NetTranslation().x() << " "
172  << voxelHalfWidthX*nx+thePhantomTransform.NetTranslation().x() << G4endl;
173  fout << -voxelHalfWidthY*ny+thePhantomTransform.NetTranslation().y() << " "
174  << voxelHalfWidthY*ny+thePhantomTransform.NetTranslation().y() << G4endl;
175  fout << -voxelHalfWidthZ*nz+thePhantomTransform.NetTranslation().z() << " "
176  << voxelHalfWidthZ*nz+thePhantomTransform.NetTranslation().z() << G4endl;
177 
178  for( G4int iz = 0; iz < nz; iz++ ){
179  for( G4int iy = 0; iy < ny; iy++) {
180 
181  G4bool bPrevVoxelInside = true;
182  G4bool b1VoxelFoundInside = false;
183  G4int firstVoxel = -1;
184  G4int lastVoxel = -1;
185  for(G4int ix = 0; ix < nx; ix++ ){
186  G4ThreeVector voxelCentre( (-nx+ix*2+1)*voxelHalfWidthX,
187  (-ny+iy*2+1)*voxelHalfWidthY, (-nz+iz*2+1)*voxelHalfWidthZ);
188  theTransform.ApplyPointTransform(voxelCentre);
189  G4bool bVoxelIsInside = true;
190  for( G4int ivx = -1; ivx <= 1; ivx+=2 ) {
191  for( G4int ivy = -1; ivy <= 1; ivy+=2 ){
192  for( G4int ivz = -1; ivz <= 1; ivz+=2 ) {
193  G4ThreeVector voxelPoint = voxelCentre
194  + ivx*voxelHalfWidthX*axisX +
195  ivy*voxelHalfWidthY*axisY + ivz*voxelHalfWidthZ*axisZ;
196  if( fSolid->Inside( voxelPoint ) == kOutside ) {
197  bVoxelIsInside = false;
198  break;
199  } else {
200  }
201  }
202  if( !bVoxelIsInside ) break;
203  }
204  if( !bVoxelIsInside ) break;
205  }
206 
207  G4int copyNo = ix + nx*iy + nxy*iz;
208  if( bVoxelIsInside ) {
209  if( !bPrevVoxelInside ) {
210  G4Exception("DicomIntersectVolume::SetNewValue",
211  "",
212  FatalException,
213  "Volume cannot intersect phantom in discontiguous voxels, "
214  "please use other voxel");
215  }
216  if( !b1VoxelFoundInside ) {
217  firstVoxel = ix;
218  b1VoxelFoundInside = true;
219  }
220  lastVoxel = ix;
221  fVoxelIsInside[copyNo] = true;
222  } else {
223  fVoxelIsInside[copyNo] = false;
224  }
225 
226  }
227  fout << firstVoxel << " " << lastVoxel << G4endl;
228  }
229  }
230 
231  //----- Now write the materials
232  for( G4int iz = 0; iz < nz; iz++ ){
233  for( G4int iy = 0; iy < ny; iy++) {
234  G4bool b1xFound = false;
235  for(G4int ix = 0; ix < nx; ix++ ){
236  size_t copyNo = ix + ny*iy + nxy*iz;
237  // fout << " iz " << iz << " i " << iy << " ix " << ix << G4endl;
238  if( fVoxelIsInside[copyNo] ) {
239  fout << thePhantomParam->GetMaterialIndex(copyNo)<< " ";
240  b1xFound = true;
241  }
242  }
243  if(b1xFound ) fout << G4endl;
244  }
245  }
246 
247  // Now write densities
248  for( G4int iz = 0; iz < nz; iz++ ){
249  for( G4int iy = 0; iy < ny; iy++) {
250  G4bool b1xFound = false;
251  for(G4int ix = 0; ix < nx; ix++ ){
252  size_t copyNo = ix + ny*iy + nxy*iz;
253  if( fVoxelIsInside[copyNo] ) {
254  fout <<thePhantomParam->GetMaterial(copyNo)->GetDensity()/g*cm3<< " ";
255  b1xFound = true;
256  }
257  }
258  if(b1xFound ) fout << G4endl;
259  }
260  }

316 std::vector<G4VPhysicalVolume*> DicomIntersectVolume::GetPhysicalVolumes(
317  const G4String& name, bool exists, G4int nVols )
318 {
319  std::vector<G4VPhysicalVolume*> vvolu;
320  std::string::size_type ial = name.rfind(":");
321  G4String volname = "";
322  G4int volcopy = 0;
323  if( ial != std::string::npos ) {
324  std::string::size_type ial2 = name.rfind(":",ial-1);
325  if( ial2 != std::string::npos ) {
326  G4Exception("DicomIntersectVolume::GetPhysicalVolumes",
327  "",
328  FatalErrorInArgument,
329  G4String("Name corresponds to a touchable " + name).c_str());
330  }else {
331  volname = name.substr( 0, ial );
332  volcopy = G4UIcommand::ConvertToInt( name.substr( ial+1, name.length() ).c_str() );
333  }
334  } else {
335  volname = name;
336  volcopy = -1;
337  }
338 
339  G4PhysicalVolumeStore* pvs = G4PhysicalVolumeStore::GetInstance();
340  std::vector<G4VPhysicalVolume*>::iterator citepv;
341  for( citepv = pvs->begin(); citepv != pvs->end(); citepv++ ) {
342  if( volname == (*citepv)->GetName()
343  && ( (*citepv)->GetCopyNo() == volcopy || -1 == volcopy ) ){
344  vvolu.push_back( *citepv );
345  }
346  }
347 
348  //----- Check that at least one volume was found
349  if( vvolu.size() == 0 ) {
350  if(exists) {
351  G4Exception(" DicomIntersectVolume::GetPhysicalVolumes",
352  "",
353  FatalErrorInArgument,
354  G4String("No physical volume found with name " + name).c_str());
355  } else {
356  G4cerr << "!!WARNING: DicomIntersectVolume::GetPhysicalVolumes: "<<
357  " no physical volume found with name " << name << G4endl;
358  }
359  }
360 
361  if( nVols != -1 && G4int(vvolu.size()) != nVols ) {
362  G4Exception("DicomIntersectVolume::GetLogicalVolumes:",
363  "Wrong number of physical volumes found",
364  FatalErrorInArgument,
365  ("Number of physical volumes " +
366  G4UIcommand::ConvertToString(G4int(vvolu.size())) +
367  ", requesting " + G4UIcommand::ConvertToString(nVols)).c_str());
368  }
369 
370  return vvolu;
371 }
