105 void doiPETDetectorConstructionMessenger::SetNewValue(G4UIcommand* command,G4String newValue)
106 {
107  if ( command == ChoosePhantomCmd ) Detector->ChangePhantom(newValue);
108  else if (command == changeThePhantomPositionCmd )
109  {
110  G4ThreeVector size = changeThePhantomPositionCmd -> GetNew3VectorValue(newValue);
111  Detector -> SetPhantomPosition(size);
112  }
113  else if (command == changePhantomRadiusCmd)
114  {
115  Detector->SetPhantomRadius(changePhantomRadiusCmd->GetNewDoubleValue(newValue));
116  }
117  else if (command == changePhantomLengthCmd)
118  {
119  Detector->SetPhantomLength(changePhantomLengthCmd->GetNewDoubleValue(newValue));
120  }
121 }
