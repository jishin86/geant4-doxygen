88 void G4CascadeCheckBalance::collide(G4InuclParticle* bullet,
89  G4InuclParticle* target,
90  G4CollisionOutput& output) {
91  if (verboseLevel)
92  G4cout << " >>> G4CascadeCheckBalance(" << theName << ")::collide"
93  << G4endl;
94 
95  initial *= 0.; // Fast reset; some colliders only have one pointer
96  if (bullet) initial += bullet->getMomentum();
97  if (target) initial += target->getMomentum();
98 
99  // Baryon number, charge and strangeness must be computed "by hand"
100  initialCharge = 0;
101  if (bullet) initialCharge += G4int(bullet->getCharge());
102  if (target) initialCharge += G4int(target->getCharge());
103 
104  G4InuclElementaryParticle* pbullet =
105  dynamic_cast<G4InuclElementaryParticle*>(bullet);
106  G4InuclElementaryParticle* ptarget =
107  dynamic_cast<G4InuclElementaryParticle*>(target);
108 
109  G4InuclNuclei* nbullet = dynamic_cast<G4InuclNuclei*>(bullet);
110  G4InuclNuclei* ntarget = dynamic_cast<G4InuclNuclei*>(target);
111 
112  initialBaryon =
113  ((pbullet ? pbullet->baryon() : nbullet ? nbullet->getA() : 0) +
114  (ptarget ? ptarget->baryon() : ntarget ? ntarget->getA() : 0) );
115 
116  // NOTE: Currently we ignore possibility of hypernucleus target
117  initialStrange = 0;
118  if (pbullet) initialStrange += pbullet->getStrangeness();
119  if (ptarget) initialStrange += ptarget->getStrangeness();
120 
121  // Final state totals are computed for us
122  final = output.getTotalOutputMomentum();
123  finalBaryon = output.getTotalBaryonNumber();
124  finalCharge = output.getTotalCharge();
125  finalStrange = output.getTotalStrangeness();
126 
127  // Report results
128  if (verboseLevel) {
129  G4cout << " initial px " << initial.px() << " py " << initial.py()
130  << " pz " << initial.pz() << " E " << initial.e()
131  << " baryon " << initialBaryon << " charge " << initialCharge
132  << " strange " << initialStrange << G4endl
133  << " final px " << final.px() << " py " << final.py()
134  << " pz " << final.pz() << " E " << final.e()
135  << " baryon " << finalBaryon << " charge " << finalCharge
136  << " strange " << finalStrange << G4endl;
137  }
138 }

140 // For de-excitation, take G4Fragment as initial state
141 void G4CascadeCheckBalance::collide(const G4Fragment& fragment,
142  G4CollisionOutput& output) {
143  if (verboseLevel)
144  G4cout << " >>> G4CascadeCheckBalance(" << theName << ")::collide(<FRAG>)"
145  << G4endl;
146 
147  // Copy initial state directly from fragment (no bullet/target sums)
148  initial = fragment.GetMomentum()/GeV; // G4Fragment returns MeV
149  initialCharge = fragment.GetZ_asInt();
150  initialBaryon = fragment.GetA_asInt();
151  initialStrange = 0; // No hypernuclei at present
152 
153  // Final state totals are computed for us
154  final = output.getTotalOutputMomentum();
155 
156  // Remove electron masses when internal conversion occurs
157  G4double elMass = 0.;
158  std::vector<G4InuclElementaryParticle> outParts = output.getOutgoingParticles();
159  for (G4int i = 0; i < output.numberOfOutgoingParticles(); i++) {
160  if (outParts[i].getDefinition() == G4Electron::Electron() )
161  elMass += outParts[i].getDefinition()->GetPDGMass();
162  }
163 
164  final.setE(final.e() - elMass/GeV);
165  finalBaryon = output.getTotalBaryonNumber();
166  finalCharge = output.getTotalCharge();
167  finalStrange = output.getTotalStrangeness();
168 
169  // Report results
170  if (verboseLevel) {
171  G4cout << " initial px " << initial.px() << " py " << initial.py()
172  << " pz " << initial.pz() << " E " << initial.e()
173  << " baryon " << initialBaryon << " charge " << initialCharge
174  << " strange " << initialStrange << G4endl
175  << " final px " << final.px() << " py " << final.py()
176  << " pz " << final.pz() << " E " << final.e()
177  << " baryon " << finalBaryon << " charge " << finalCharge
178  << " strange " << finalStrange << G4endl;
179  }

253 G4bool G4CascadeCheckBalance::energyOkay() const {
254  G4bool relokay = (std::abs(relativeE()) < relativeLimit);
255  G4bool absokay = (std::abs(deltaE()) < absoluteLimit);
256 
257  if (verboseLevel && (!relokay || !absokay)) {
258  G4cerr << theName << ": Energy conservation: relative " << relativeE()
259  << (relokay ? " conserved" : " VIOLATED")
260  << " absolute " << deltaE()
261  << (absokay ? " conserved" : " VIOLATED") << G4endl;
262  } else if (verboseLevel > 1) {
263  G4cout << theName << ": Energy conservation: relative " << relativeE()
264  << " conserved absolute " << deltaE() << " conserved" << G4endl;
265  }
266 
267  return (relokay && absokay);
268 }
