54 void G4GDMLParser::Write(const G4String& filename,

56  G4bool refs,
57  const G4String& schemaLocation)

63 
64  if (!pvol)

67  GetNavigatorForTracking()->GetWorldVolume()->GetLogicalVolume();

69  else
70  {
71  lvol = pvol->GetLogicalVolume();
72  }
73 
74  if (rexp) { ExportRegions(refs); }
75  writer->Write(filename,lvol,schemaLocation,depth,refs);
