47 G4EnclosingCylinder::G4EnclosingCylinder( const G4ReduciblePolygon *rz,
48  G4bool thePhiIsOpen,
49  G4double theStartPhi,
50  G4double theTotalPhi )
51  : startPhi(theStartPhi), totalPhi(theTotalPhi),
52  rx1(0.), ry1(0.), dx1(0.), dy1(0.),
53  rx2(0.), ry2(0.), dx2(0.), dy2(0.),
54  concave(theTotalPhi > pi)
55 {
56  //
57  // Obtain largest r and smallest and largest z
58  //
59  radius = rz->Amax();
60  zHi = rz->Bmax();
61  zLo = rz->Bmin();
62 
63  G4double kCarTolerance = G4GeometryTolerance::GetInstance()
64  ->GetSurfaceTolerance();
65  //
66  // Save phi info
67  //
68  phiIsOpen = thePhiIsOpen;
69  if ( phiIsOpen )
70  {
71  rx1 = std::cos(startPhi);
72  ry1 = std::sin(startPhi);
73  dx1 = +ry1*10*kCarTolerance;
74  dy1 = -rx1*10*kCarTolerance;
75 
76  rx2 = std::cos(startPhi+totalPhi);
77  ry2 = std::sin(startPhi+totalPhi);
78  dx2 = -ry2*10*kCarTolerance;
79  dy2 = +rx2*10*kCarTolerance;
80  }
81 
82  //
83  // Add safety
84  //
85  radius += 10*kCarTolerance;
86  zLo -= 10*kCarTolerance;
87  zHi += 10*kCarTolerance;
88 }

110 // Outside
111 //
112 // Decide very rapidly if the point is outside the cylinder
113 //
114 // If one is not certain, return false
115 //
116 G4bool G4EnclosingCylinder::MustBeOutside( const G4ThreeVector &p ) const
117 {
118  if (p.perp() > radius) return true;
119  if (p.z() < zLo) return true;
120  if (p.z() > zHi) return true;
121 
122  if (phiIsOpen)
123  {
124  if (concave)
125  {
126  if ( ((p.x()-dx1)*ry1 - (p.y()-dy1)*rx1) < 0) return false;
127  if ( ((p.x()-dx2)*ry2 - (p.y()-dy2)*rx2) > 0) return false;
128  }
129  else
130  {
131  if ( ((p.x()-dx1)*ry1 - (p.y()-dy1)*rx1) > 0) return true;
132  if ( ((p.x()-dx2)*ry2 - (p.y()-dy2)*rx2) < 0) return true;
133  }
134  }
135 
136  return false;
137 }
138 
139 
140 //
141 // Misses
142 //
143 // Decide very rapidly if the trajectory is going to miss the cylinder
144 //
145 // If one is not sure, return false
146 //
147 G4bool G4EnclosingCylinder::ShouldMiss( const G4ThreeVector &p,
148  const G4ThreeVector &v ) const
149 {
150  if (!MustBeOutside(p)) return false;
151 
152  G4double cross = p.x()*v.y() - p.y()*v.x();
153  if (cross > radius) return true;
154 
155  if (p.perp() > radius)
156  {
157  G4double dot = p.x()*v.x() + p.y()*v.y();
158  if (dot > 0) return true;
159  }
160 
161  return false;
162 }
