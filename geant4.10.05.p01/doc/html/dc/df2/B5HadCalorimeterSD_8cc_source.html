57 void B5HadCalorimeterSD::Initialize(G4HCofThisEvent* hce)
58 {
59  fHitsCollection
60  = new B5HadCalorimeterHitsCollection(SensitiveDetectorName,collectionName[0]);
61  if (fHCID<0) {
62  fHCID = G4SDManager::GetSDMpointer()->GetCollectionID(fHitsCollection);
63  }
64  hce->AddHitsCollection(fHCID,fHitsCollection);
65 
66  // fill calorimeter hits with zero energy deposition
67  for (auto column=0;column<kNofHadColumns;column++) {
68  for (auto row=0;row<kNofHadRows;row++) {
69  fHitsCollection->insert(new B5HadCalorimeterHit());
70  }
71  }
72 }

76 G4bool B5HadCalorimeterSD::ProcessHits(G4Step* step, G4TouchableHistory*)
77 {
78  auto edep = step->GetTotalEnergyDeposit();
79  if (edep==0.) return true;
80 
81  auto touchable = step->GetPreStepPoint()->GetTouchable();
82  auto rowNo = touchable->GetCopyNumber(2);
83  auto columnNo = touchable->GetCopyNumber(3);
84  auto hitID = kNofHadRows*columnNo+rowNo;
85  auto hit = (*fHitsCollection)[hitID];
86 
87  // check if it is first touch
88  if (hit->GetColumnID()<0) {
89  hit->SetColumnID(columnNo);
90  hit->SetRowID(rowNo);
91  auto depth = touchable->GetHistory()->GetDepth();
92  auto transform = touchable->GetHistory()->GetTransform(depth-2);
93  transform.Invert();
94  hit->SetRot(transform.NetRotation());
95  hit->SetPos(transform.NetTranslation());
96  }
97  // add energy deposition
98  hit->AddEdep(edep);
99 
100  return true;
101 }
