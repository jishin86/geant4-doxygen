162 inline size_t G4LevelManager::NearestLowEdgeLevelIndex(G4double energy) const
163 {
164  size_t idx = nTransitions;
165  if(energy < fLevelEnergy[nTransitions]) {
166  idx = std::lower_bound(fLevelEnergy.begin(), fLevelEnergy.end(), energy)
167  - fLevelEnergy.begin() - 1;
168  }
169  return idx;
170 }
