64 G4VParticleChange* SplitProcess::PostStepDoIt(const G4Track& track, const G4Step& step) {
65  G4VParticleChange* particleChange(0);
66 
67  if ( step.GetPreStepPoint()->GetTouchableHandle()->GetVolume()->GetLogicalVolume()
68  ->GetRegion() != fRegion ) {
69  particleChange = pRegProcess->PostStepDoIt(track, step);
70  assert ( 0 != particleChange);
71  return particleChange;
72  }
73 
74  if ( fNSplit == 1 ) {
75  particleChange = pRegProcess->PostStepDoIt(track, step);
76  assert ( 0 != particleChange);
77  return particleChange;
78  }
79 
80  UserTrackInformation* parentInformation =
81  (UserTrackInformation*)(step.GetTrack()->GetUserInformation());
82  G4int initialSplitTrackID = parentInformation->GetSplitTrackID();
83 
84  if ( initialSplitTrackID > 1 ) {
85  particleChange = pRegProcess->PostStepDoIt(track, step);
86  assert ( 0 != particleChange);
87  return particleChange;
88  }
89 
90  G4double weight = track.GetWeight()/fNSplit;
91  G4int splitTrackID = 3;
92 
93  std::vector<G4Track*> secondaries;
94  std::vector<G4int> vSplitTrack;
95 
96  for ( int i = 0; i < fNSplit; i++ ) {
97  particleChange = pRegProcess->PostStepDoIt(track, step);
98  assert( 0 != particleChange);
99  particleChange->SetVerboseLevel(0);
100  G4Track* newTrack = new G4Track(*(particleChange->GetSecondary(0)));
101 
102  secondaries.push_back( newTrack );
103  vSplitTrack.push_back( splitTrackID );
104 
105  splitTrackID++;
106  }
107 
108  parentInformation->SetSplitTrackID(2);
109 
110  particleChange->SetNumberOfSecondaries(secondaries.size());
111  particleChange->SetSecondaryWeightByProcess(true);
112 
113  std::vector<G4Track*>::iterator iter = secondaries.begin();
114  G4int i = 0;
115  while( iter != secondaries.end() ) {
116  G4Track* newTrack = *iter;
117  newTrack->SetWeight(weight);
118 
119  UserTrackInformation* secondaryInformation = new UserTrackInformation();
120  secondaryInformation->SetSplitTrackID(vSplitTrack[i]);
121  newTrack->SetUserInformation(secondaryInformation);
122 
123  particleChange->AddSecondary(newTrack);
124 
125  iter++;
126  i++;
127  }
128 
129  return particleChange;
130 }
