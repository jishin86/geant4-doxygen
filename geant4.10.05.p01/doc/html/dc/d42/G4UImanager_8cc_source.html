410 G4String G4UImanager::SolveAlias(const char* aCmd)
411 {
412  G4String aCommand = aCmd;
413  G4int ia = aCommand.index("{");
414  G4int iz = aCommand.index("#");
415  while((ia != G4int(std::string::npos))&&((iz==G4int(std::string::npos))||(ia<iz)))
416  {
417  G4int ibx = -1;
418  while(ibx<0)
419  {
420  G4int ib = aCommand.index("}");
421  if( ib == G4int(std::string::npos) )
422  {
423  G4cerr << aCommand << G4endl;
424  for(G4int i=0;i<ia;i++) G4cerr << " ";
425  G4cerr << "^" << G4endl;
426  G4cerr << "Unmatched alias parenthesis -- command ignored" << G4endl;
427  G4String nullStr;
428  return nullStr;
429  }
430  G4String ps = aCommand(ia+1,aCommand.length()-(ia+1));
431  G4int ic = ps.index("{");
432  G4int id = ps.index("}");
433  if(ic!=G4int(std::string::npos) && ic < id)
434  { ia+=ic+1; }
435  else
436  { ibx = ib; }
437  }
438  //--- Here ia represents the position of innermost "{"
439  //--- and ibx represents corresponding "}"
440  G4String subs;
441  if(ia>0) subs = aCommand(0,ia);
442  G4String alis = aCommand(ia+1,ibx-ia-1);
443  G4String rems = aCommand(ibx+1,aCommand.length()-ibx);
444  // G4cout << "<" << subs << "> <" << alis << "> <" << rems << ">" << G4endl;
445  G4String* alVal = aliasList->FindAlias(alis);
446  if(!alVal)
447  {
448  G4cerr << "Alias <" << alis << "> not found -- command ignored" << G4endl;
449  G4String nullStr;
450  return nullStr;
451  }
452  aCommand = subs+(*alVal)+rems;
453  ia = aCommand.index("{");
454  }
455  return aCommand;
456 }

465 G4int G4UImanager::ApplyCommand(const char * aCmd)
466 {
467  G4String aCommand = SolveAlias(aCmd);
468  if(aCommand.isNull()) return fAliasNotFound;
469  if(verboseLevel) G4cout << aCommand << G4endl;
470  G4String commandString;
471  G4String commandParameter;
472 
473  G4int i = aCommand.index(" ");
474  if( i != G4int(std::string::npos) )
475  {
476  commandString = aCommand(0,i);
477  commandParameter = aCommand(i+1,aCommand.length()-(i+1));
478  }
479  else
480  {
481  commandString = aCommand;
482  }
483 
484  // remove doubled slash
485  G4int len = commandString.length();
486  G4int ll = 0;
487  G4String a1;
488  G4String a2;
489  while(ll<len-1)
490  {
491  if(commandString(ll,2)=="//")
492  {
493  if(ll==0)
494  { commandString.remove(ll,1); }
495  else
496  {
497  a1 = commandString(0,ll);
498  a2 = commandString(ll+1,len-ll-1);
499  commandString = a1+a2;
500  }
501  len--;
502  }
503  else
504  { ll++; }
505  }
506 
507  if(isMaster&&bridges)
508  {
509  std::vector<G4UIbridge*>::iterator itr = bridges->begin();
510  for(;itr!=bridges->end();itr++)
511  {
512  G4int leng = (*itr)->DirLength();
513  if(commandString(0,leng)==(*itr)->DirName())
514  { return (*itr)->LocalUI()->ApplyCommand(commandString+" "+commandParameter); }
515  }
516  }
517 
518  G4UIcommand * targetCommand = treeTop->FindPath( commandString );
519  if( targetCommand == NULL )
520  {
521  if(ignoreCmdNotFound)
522  {
523  if(stackCommandsForBroadcast)
524  { commandStack->push_back(commandString+" "+commandParameter); }
525  return fCommandSucceeded;
526  }
527  else
528  { return fCommandNotFound; }
529  }
530 
531  if(stackCommandsForBroadcast && targetCommand->ToBeBroadcasted())
532  { commandStack->push_back(commandString+" "+commandParameter); }
533 
534  if(!(targetCommand->IsAvailable()))
535  { return fIllegalApplicationState; }
536 
537  if(saveHistory) historyFile << aCommand << G4endl;
538  if( G4int(histVec.size()) >= maxHistSize )
539  { histVec.erase(histVec.begin()); }
540  histVec.push_back(aCommand);
541 
542  targetCommand->ResetFailure();
543  G4int commandFailureCode = targetCommand->DoIt( commandParameter );
544  if(commandFailureCode==0)
545  {
546  G4int additionalFailureCode = targetCommand->IfCommandFailed();
547  if(additionalFailureCode > 0)
548  {
549  G4ExceptionDescription msg;
550  msg << targetCommand->GetFailureDescription() << "\n"
551  << "Error code : " << additionalFailureCode;
552  G4Exception("G4UImanager::ApplyCommand","UIMAN0123",
553  JustWarning,msg);
554  commandFailureCode += additionalFailureCode;
555  }
556  }
557  return commandFailureCode;
558 }

594 G4UIcommandTree* G4UImanager::FindDirectory(const char* dirName)
595 {
596  G4String aDirName = dirName;
597  G4String targetDir = aDirName.strip(G4String::both);
598  if( targetDir( targetDir.length()-1 ) != '/' )
599  { targetDir += "/"; }
600  G4UIcommandTree* comTree = treeTop;
601  if( targetDir == "/" )
602  { return comTree; }
603  G4int idx = 1;
604  while( idx < G4int(targetDir.length())-1 )
605  {
606  G4int i = targetDir.index("/",idx);
607  G4String targetDirString = targetDir(0,i+1);
608  comTree = comTree->GetTree(targetDirString);
609  if( comTree == NULL )
610  { return NULL; }
611  idx = i+1;
612  }
613  return comTree;
614 }

698 void G4UImanager::ParseMacroSearchPath()
699 {
700  searchDirs.clear();
701 
702  size_t idxfirst = 0;
703  size_t idxend = 0;
