48 G4GlobalMagFieldMessenger::G4GlobalMagFieldMessenger(const G4ThreeVector& value)
49  : G4UImessenger(),
50  fMagField(0),
51  fVerboseLevel(0),
52  fDirectory(0),
53  fSetValueCmd(0),
54  fSetVerboseCmd(0)
55 {
56  fDirectory = new G4UIdirectory("/globalField/");
57  fDirectory->SetGuidance("Global uniform magnetic field UI commands");
58 
59  fSetValueCmd = new G4UIcmdWith3VectorAndUnit("/globalField/setValue",this);
60  fSetValueCmd->SetGuidance("Set uniform magnetic field value.");
61  fSetValueCmd->SetParameterName("Bx", "By", "By", false);
62  fSetValueCmd->SetUnitCategory("Magnetic flux density");
63  fSetValueCmd->AvailableForStates(G4State_PreInit,G4State_Idle);
64 
65  fSetVerboseCmd = new G4UIcmdWithAnInteger("/globalField/verbose",this);
66  fSetVerboseCmd->SetGuidance("Set verbose level: ");
67  fSetVerboseCmd->SetGuidance(" 0: no output");
68  fSetVerboseCmd->SetGuidance(" 1: printing new field value");
69  fSetVerboseCmd->SetParameterName("globalFieldVerbose", false);
70  fSetVerboseCmd->SetRange("globalFieldVerbose>=0");
71  fSetVerboseCmd->AvailableForStates(G4State_PreInit,G4State_Idle);
72 
73  // Create field
74  fMagField = new G4UniformMagField(value);
75 
76  // Set field value (the field is not activated if value is zero)
77  SetField(value, "G4GlobalMagFieldMessenger::G4GlobalMagFieldMessenger");
78 }

92 void G4GlobalMagFieldMessenger::SetField(const G4ThreeVector& value,
93  const G4String& /*inFunction*/)
94 {
95  // Get field manager (or create it if it does not yet exist)
96  G4FieldManager* fieldManager
97  = G4TransportationManager::GetTransportationManager()->GetFieldManager();
98 
99  // Inactivate field if its value is zero
100  if ( value == G4ThreeVector() )
101  {
102  fieldManager->SetDetectorField(0);
103  fieldManager->CreateChordFinder(0);
104 
105  if ( fVerboseLevel > 0 )
106  {
107  G4cout << "Magnetic field is inactive, fieldValue = (0,0,0)." << G4endl;
108  }
109  }
110  else
111  {
112  fMagField->SetFieldValue(value);
113  fieldManager->SetDetectorField(fMagField);
114  fieldManager->CreateChordFinder(fMagField);
115 
116  if ( fVerboseLevel > 0 )
117  {
118  G4cout << "Magnetic field is active, fieldValue = ("
119  << G4BestUnit(value, "Magnetic flux density")
120  << ")." << G4endl;
121  }
122  }
123 }
