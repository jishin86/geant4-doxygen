95 void G4HadronInelasticQBBC::ConstructProcess()
96 {
97  if(verbose > 1) {
98  G4cout << "### HadronInelasticQBBC Construct Process " << G4endl;
99  }
100 
101  G4double emax = G4HadronicParameters::Instance()->GetMaxEnergy();
102 
103  //G4cout << "G4HadronInelasticQBBC::ConstructProcess new PRECO"<< G4endl;
104 
105  // PreCompound and Evaporation models are instantiated here
106  G4PreCompoundModel* thePreCompound = nullptr;
107  G4HadronicInteraction* p =
108  G4HadronicInteractionRegistry::Instance()->FindModel("PRECO");
109  thePreCompound = static_cast<G4PreCompoundModel*>(p);
110  if(!thePreCompound) { thePreCompound = new G4PreCompoundModel(); }
111 
112  // configure models
113  //G4HadronicInteraction* theQGSP =
114  // BuildModel(new G4QGSBuilder("QGSP",thePreCompound,true,false),12.5*GeV,emax);
115  G4HadronicInteraction* theFTFP =
116  BuildModel(new G4FTFBuilder("FTFP",thePreCompound),3.0*GeV,emax);
117  G4HadronicInteraction* theFTFP1 =
118  BuildModel(new G4FTFBuilder("FTFP",thePreCompound),3.0*GeV,emax);
119  G4HadronicInteraction* theFTFP2 =
120  BuildModel(new G4FTFBuilder("FTFP",thePreCompound),0.0,emax);
121 
122  G4CascadeInterface* casc = new G4CascadeInterface();
123  casc->usePreCompoundDeexcitation();
124  G4HadronicInteraction* theBERT = NewModel(casc,1.0*GeV,5.0*GeV);
125 
126  casc = new G4CascadeInterface();
127  casc->usePreCompoundDeexcitation();
128  G4HadronicInteraction* theBERT1 = NewModel(casc,0.0*GeV,5.0*GeV);
129 
130  //G4cout << "G4HadronInelasticQBBC::ConstructProcess new Binary"<< G4endl;
131  G4BinaryCascade* bic = new G4BinaryCascade(thePreCompound);
132  G4HadronicInteraction* theBIC = NewModel(bic,0.0,1.5*GeV);
133 
134  // cross sections
135  G4CrossSectionInelastic* anucxs = InelasticXS("AntiAGlauber");
136  if(!anucxs) {
137  anucxs = new G4CrossSectionInelastic(new G4ComponentAntiNuclNuclearXS());
138  }
139  G4CrossSectionInelastic* kaonxs =
140  InelasticXS(G4ComponentGGHadronNucleusXsc::Default_Name());
141  if(!kaonxs) {
142  kaonxs = new G4CrossSectionInelastic(new G4ComponentGGHadronNucleusXsc());
143  }
144  G4HadronicProcess* hp;
145  // loop over particles
146  auto myParticleIterator=GetParticleIterator();
147  myParticleIterator->reset();
148  while( (*myParticleIterator)() ) {
149  G4ParticleDefinition* particle = myParticleIterator->value();
150  G4String pname = particle->GetParticleName();
151  if(verbose > 1) {
152  G4cout << "### HadronInelasticQBBC: " << pname << G4endl;
153  }
154 
155  //
156  // model and X-section configuration per particle type
157  //
158  if(pname == "proton") {
159  hp = FindInelasticProcess(particle);
160  //hp->AddDataSet(new G4ParticleInelasticXS(particle));
161  hp->AddDataSet(new G4BGGNucleonInelasticXS(particle));
162 
163  hp->RegisterMe(theFTFP);
164  hp->RegisterMe(theBERT);
165  hp->RegisterMe(theBIC);
166 
167  } else if(pname == "neutron") {
168  hp = FindInelasticProcess(particle);
169  hp->AddDataSet(new G4NeutronInelasticXS());
170  hp->RegisterMe(theFTFP);
171 
172  G4HadronicProcess* capture = FindCaptureProcess();
173  capture->RegisterMe(new G4NeutronRadCapture());
174  hp->RegisterMe(theBERT);
175  hp->RegisterMe(theBIC);
176 
177  } else if(pname == "pi-" || pname == "pi+") {
178  hp = FindInelasticProcess(particle);
179  hp->AddDataSet(new G4BGGPionInelasticXS(particle));
180  hp->RegisterMe(theFTFP);
181  hp->RegisterMe(theBERT1);
182 
183  } else if(pname == "kaon-" ) {
184  hp = FindInelasticProcess(particle);
185  hp->RegisterMe(theFTFP1);
186  hp->RegisterMe(theBERT1);
187  hp->AddDataSet(kaonxs);
188 
189  } else if(pname == "kaon+" ) {
190  hp = FindInelasticProcess(particle);
191  hp->RegisterMe(theFTFP1);
192  hp->RegisterMe(theBERT1);
193  hp->AddDataSet(kaonxs);
194 
195  } else if(pname == "kaon0S" ||
196  pname == "kaon0L") {
197  hp = FindInelasticProcess(particle);
198  hp->RegisterMe(theFTFP1);
199  hp->RegisterMe(theBERT1);
200  hp->AddDataSet(kaonxs);
201 
202  } else if(pname == "lambda" ||
203  pname == "omega-" ||
204  pname == "sigma-" ||
205  pname == "sigma+" ||
206  pname == "sigma0" ||
207  pname == "xi-" ||
208  pname == "xi0") {
209  hp = FindInelasticProcess(particle);
210  hp->RegisterMe(theFTFP1);
211  hp->RegisterMe(theBERT1);
212  hp->AddDataSet(G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4ChipsHyperonInelasticXS::Default_Name()));
213 
214  } else if(pname == "anti_alpha" ||
215  pname == "anti_deuteron"||
216  pname == "anti_He3" ||
217  pname == "anti_proton" ||
218  pname == "anti_triton" ||
219  pname == "anti_lambda" ||
220  pname == "anti_neutron" ||
221  pname == "anti_omega-" ||
222  pname == "anti_sigma-" ||
223  pname == "anti_sigma+" ||
224  pname == "anti_xi-" ||
225  pname == "anti_xi0"
226  ) {
227  hp = FindInelasticProcess(particle);
228  hp->RegisterMe(theFTFP2);
229  hp->AddDataSet(anucxs);
230  }
231  }
232 }
