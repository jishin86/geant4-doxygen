32 std::ostream& operator<<( std::ostream& os, const G4FieldTrack& SixVec)
33 {
34  const G4double *SixV = SixVec.SixVector;
35  const int precPos= 9; // For position
36  const int precEp= 9; // For Energy / momentum
37  const int precLen= 12; // For Length along track
38  const int precSpin= 9; // For polarisation
39  const int precTime= 6; // For time of flight
40  const int oldpr= os.precision(precPos);
41  os << " ( ";
42  os << " X= " << SixV[0] << " " << SixV[1] << " "
43  << SixV[2] << " "; // Position
44  os.precision(precEp);
45  os << " P= " << SixV[3] << " " << SixV[4] << " "
46  << SixV[5] << " "; // Momentum
47  os << " Pmag= "
48  << G4ThreeVector(SixV[3], SixV[4], SixV[5]).mag(); // mom magnitude
49  os << " Ekin= " << SixVec.fKineticEnergy ;
50  os.precision(precLen);
51  os << " l= " << SixVec.GetCurveLength();
52  os.precision(6);
53  os << " m0= " << SixVec.fRestMass_c2;
54  os << " (Pdir-1)= " << SixVec.fMomentumDir.mag()-1.0;
55  if( SixVec.fLabTimeOfFlight > 0.0 ) os.precision(precTime);
56  else os.precision(3);
57  os << " t_lab= " << SixVec.fLabTimeOfFlight;
58  os << " t_proper= " << SixVec.fProperTimeOfFlight ;
59  G4ThreeVector pol= SixVec.GetPolarization();
60  if( pol.mag2() > 0.0 ){
61  os.precision(precSpin);
62  os << " PolV= " << pol; // SixVec.GetPolarization();
63  }else{
64  os << " PolV= (0,0,0) ";
65  }
66  os << " ) ";
67  os.precision(oldpr);
68  return os;
69 }

165 void G4FieldTrack::LoadFromArray(const G4double valArrIn[ncompSVEC],
166  G4int noVarsIntegrated)
167 {
168  G4int i;
169 
170  // Fill the variables not integrated with zero -- so it's clear !!
171  G4double valArr[ncompSVEC];
172  for( i=0; i<noVarsIntegrated; i++){
173  valArr[i]= valArrIn[i];
174  }
175  for( i=noVarsIntegrated; i<ncompSVEC; i++) {
176  valArr[i]= 0.0;
177  }
178 
179  SixVector[0]=valArr[0];
180  SixVector[1]=valArr[1];
181  SixVector[2]=valArr[2];
182  SixVector[3]=valArr[3];
183  SixVector[4]=valArr[4];
184  SixVector[5]=valArr[5];
185 
186  G4ThreeVector Momentum(valArr[3],valArr[4],valArr[5]);
187 
188  G4double momentum_square= Momentum.mag2();
189  fMomentumDir= Momentum.unit();
190 
191  fKineticEnergy = momentum_square /
192  (std::sqrt(momentum_square+fRestMass_c2*fRestMass_c2)
193  + fRestMass_c2 );
194  // The above equation is stable for small and large momenta
195 
196  // The following components may or may not be
197  // integrated over -- integration is optional
198  // fKineticEnergy= valArr[6];
199 
200  fLabTimeOfFlight=valArr[7];
201  fProperTimeOfFlight=valArr[8];
202  G4ThreeVector vecPolarization= G4ThreeVector(valArr[9],valArr[10],valArr[11]);
203  SetPolarization( vecPolarization );
204 
205  // fMomentumDir=G4ThreeVector(valArr[13],valArr[14],valArr[15]);
206  // fDistanceAlongCurve= valArr[];
207 }
