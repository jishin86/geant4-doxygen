151 inline
152 G4double G4CompetitiveFission::Ratio(G4double A, G4double A11,
153  G4double B1, G4double A00)
154 {
155  G4double res;
156  if (A11 >= A*0.5 && A11 <= (A00+10.0)) {
157  G4double x = (A11-A00)/A;
158  res = 1.0 - B1*x*x;
159  } else {
160  G4double x = 10.0/A;
161  res = 1.0 - B1*x*x - 2.0*x*B1*(A11-A00-10.0)/A;
162  }
163  return res;
164 }
