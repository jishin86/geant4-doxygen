163  fin >> nMaterials;
164  G4String stemp;
165  G4int nmate;

168  G4cout << "DicomPartialDetectorConstruction::ReadPhantomData reading nmate "

170  if( ii != nmate )
171  G4Exception("DicomPartialDetectorConstruction::ReadPhantomData",
172  "",

174  "Material number should be in increasing order: \
175  wrong material number ");
176 
177  }
178 
179  fin >> fNVoxelX >> fNVoxelY >> fNVoxelZ;
180  G4cout << "DicomPartialDetectorConstruction::ReadPhantomData nVoxel X/Y/Z "
181  << fNVoxelX << " " << fNVoxelY << " " << fNVoxelZ << G4endl;

183  fDimX = (fDimX - fOffsetX)/fNVoxelX;
184  fin >> fOffsetY >> fDimY;

186  fin >> fOffsetZ >> fDimZ;
187  fDimZ = (fDimZ - fOffsetZ)/fNVoxelZ;
188  G4cout << "DicomPartialDetectorConstruction::ReadPhantomData voxelDimX "
189  << fDimX << " fOffsetX " << fOffsetX << G4endl;
190  G4cout << "DicomPartialDetectorConstruction::ReadPhantomData voxelDimY "
191  << fDimY << " fOffsetY " << fOffsetY << G4endl;
192  G4cout << "DicomPartialDetectorConstruction::ReadPhantomData voxelDimZ "
193  << fDimZ << " fOffsetZ " << fOffsetZ << G4endl;
194 
195  //--- Read voxels that are filled
196  fNVoxels = 0;
197  // G4bool* isFilled = new G4bool[fNVoxelX*fNVoxelY*fNVoxelZ];
198  // fFilledIDs = new size_t[fNVoxelZ*fNVoxelY+1];
199  //? fFilledIDs.insert(0);
200  G4int ifxmin1, ifxmax1;
201  for( G4int iz = 0; iz < fNVoxelZ; iz++ ) {
202  std::map< G4int, G4int > ifmin, ifmax;
203  for( G4int iy = 0; iy < fNVoxelY; iy++ ) {
204  fin >> ifxmin1 >> ifxmax1;
205  // check coherence ...
206 
207  ifmin[iy] = ifxmin1;
208  ifmax[iy] = ifxmax1;
209  G4int ifxdiff = ifxmax1-ifxmin1+1;
210  if( ifxmax1 == -1 && ifxmin1 == -1 ) ifxdiff = 0;
211  fFilledIDs.insert(std::pair<G4int,G4int>(ifxdiff+fNVoxels-1, ifxmin1));
212  G4cout << "DicomPartialDetectorConstruction::ReadPhantomData insert "
213  << " FilledIDs " << ifxdiff+fNVoxels-1 << " min " << ifxmin1
214  << " N= " << fFilledIDs.size() << G4endl;
215  //filledIDs[iz*fNVoxelY+iy+1] = ifxmax1-ifxmin1+1;
216  for( G4int ix = 0; ix < fNVoxelX; ix++ ) {
217  if( ix >= G4int(ifxmin1) && ix <= G4int(ifxmax1) ) {
218  fNVoxels++;
219  } else {
220  }
221  }
222  }
223  fFilledMins[iz] = ifmin;
224  fFilledMaxs[iz] = ifmax;
225  }
226 
227  //--- Read material IDs
228  G4int mateID1;
229  fMateIDs = new size_t[fNVoxelX*fNVoxelY*fNVoxelZ];
230  G4int copyNo = 0;
231  for( G4int iz = 0; iz < fNVoxelZ; iz++ ) {
232  std::map< G4int, G4int > ifmin = fFilledMins[iz];
233  std::map< G4int, G4int > ifmax = fFilledMaxs[iz];
234  for( G4int iy = 0; iy < fNVoxelY; iy++ ) {
235  ifxmin1 = ifmin[iy];
236  ifxmax1 = ifmax[iy];
237  for( G4int ix = 0; ix < fNVoxelX; ix++ ) {
238  if( ix >= G4int(ifxmin1) && ix <= G4int(ifxmax1) ) {
239  fin >> mateID1;
240  if( mateID1 < 0 || mateID1 >= nMaterials ) {
241  G4Exception("DicomPartialDetectorConstruction::ReadPhantomData",
242  "",
243  FatalException,
244  G4String("Wrong index in phantom file: It should be between 0 and "
245  + G4UIcommand::ConvertToString(nMaterials-1)
246  + ", while it is "
247  + G4UIcommand::ConvertToString(mateID1)).c_str());
248  }
249  fMateIDs[copyNo] = mateID1;
250  copyNo++;
251  }
252  }
253  }
254  }
255 
256  ReadVoxelDensitiesPartial( fin );
257 
258  fin.close();
259 }

262 void DicomPartialDetectorConstruction::
263 ReadVoxelDensitiesPartial( std::ifstream& fin )
264 {
265  std::map<G4int, std::pair<G4double,G4double> > densiMinMax;
266  std::map<G4int, std::pair<G4double,G4double> >::iterator mpite;
267  for( size_t ii = 0; ii < fOriginalMaterials.size(); ii++ ){
268  densiMinMax[ii] = std::pair<G4double,G4double>(DBL_MAX,-DBL_MAX);
269  }
270 
271  //----- Define density differences (maximum density difference to create
272  // a new material)
273  char* part = getenv( "DICOM_CHANGE_MATERIAL_DENSITY" );
274  G4double densityDiff = -1.;
275  if( part ) densityDiff = G4UIcommand::ConvertToDouble(part);
276  std::map<G4int,G4double> densityDiffs;
277  if( densityDiff != -1. ) {
278  for( size_t ii = 0; ii < fOriginalMaterials.size(); ii++ ){
279  densityDiffs[ii] = densityDiff; //currently all materials
280  //with same step
281  }
282  }else {
283  if( fMaterials.size() == 0 ) { // do it only for first slice
284  for( unsigned int ii = 0; ii < fOriginalMaterials.size(); ii++ ){
285  fMaterials.push_back( fOriginalMaterials[ii] );
286  }
287  }
288  }
289  // densityDiffs[0] = 0.0001; //fAir
290 
291  //--- Calculate the average material density for each material/density bin
292  std::map< std::pair<G4Material*,G4int>, matInfo* > newMateDens;
293 
294  G4double dens1;
295  G4int ifxmin1, ifxmax1;
296 
297  //---- Read the material densities
298  G4int copyNo = 0;
299  for( G4int iz = 0; iz < fNVoxelZ; iz++ ) {
300  std::map< G4int, G4int > ifmin = fFilledMins[iz];
301  std::map< G4int, G4int > ifmax = fFilledMaxs[iz];
302  for( G4int iy = 0; iy < fNVoxelY; iy++ ) {
303  ifxmin1 = ifmin[iy];
304  ifxmax1 = ifmax[iy];
305  for( G4int ix = 0; ix < fNVoxelX; ix++ ) {
306  if( ix >= G4int(ifxmin1) && ix <= G4int(ifxmax1) ) {
307  fin >> dens1;
308  //--- store the minimum and maximum density for each material
309  //(just for printing)
310  mpite = densiMinMax.find( fMateIDs[copyNo] );
311  if( dens1 < (*mpite).second.first ) (*mpite).second.first = dens1;
312  if( dens1 > (*mpite).second.second ) (*mpite).second.second = dens1;
313 
314  //--- Get material from original list of material in file
315  G4int mateID = fMateIDs[copyNo];
316  G4Material* mate = fOriginalMaterials[mateID];
317  // G4cout << copyNo << " mateID " << mateID << G4endl;
318  //--- Check if density is equal to the original material density
319  if( std::fabs(dens1 - mate->GetDensity()/g*cm3 ) < 1.e-9 ) continue;
320 
321  //--- Build material name with fOriginalMaterials name + density
322  // float densityBin = densityDiffs[mateID] *
323  // (G4int(dens1/densityDiffs[mateID])+0.5);
324  G4String newMateName = mate->GetName();
325 
326  G4int densityBin = 0;
327  // G4cout << " densityBin " << densityBin << " "
328  // << dens1 << " " <<densityDiffs[mateID] << G4endl;
329 
330  if( densityDiff != -1.) {
331  densityBin = (G4int(dens1/densityDiffs[mateID]));
332  newMateName =
333  mate->GetName()+G4UIcommand::ConvertToString(densityBin);
334  }
335 
336  //--- Look if it is the first voxel with this material/densityBin
337  std::pair<G4Material*,G4int> matdens(mate, densityBin );
338 
339  std::map< std::pair<G4Material*,G4int>, matInfo* >::iterator mppite
340  = newMateDens.find( matdens );
341  if( mppite != newMateDens.end() ){
342  matInfo* mi = (*mppite).second;
343  mi->fSumdens += dens1;
344  mi->fNvoxels++;
345  fMateIDs[copyNo] = fOriginalMaterials.size()-1 + mi->fId;
346  // G4cout << copyNo << " mat new again "
347  //<< fOriginalMaterials.size()-1 + mi->id << " " << mi->id << G4endl;
348  } else {
349  matInfo* mi = new matInfo;
350  mi->fSumdens = dens1;
351  mi->fNvoxels = 1;
352  mi->fId = newMateDens.size()+1;
353  newMateDens[matdens] = mi;
354  fMateIDs[copyNo] = fOriginalMaterials.size()-1 + mi->fId;
355  // G4cout << copyNo << " mat new first "
356  // << fOriginalMaterials.size()-1 + mi->fId << G4endl;
357  }
358  copyNo++;
359  // G4cout << ix << " " << iy << " " << iz
360  // << " filling fMateIDs " << copyNo << " = " << atoi(cid)-1 << G4endl;
361  // fMateIDs[copyNo] = atoi(cid)-1;
362  }
363  }
364  }
365  }
366 
367  //----- Build the list of phantom materials that go to Parameterisation
368  //--- Add original materials
369  std::vector<G4Material*>::const_iterator mimite;
370  for( mimite = fOriginalMaterials.begin(); mimite != fOriginalMaterials.end();
371  mimite++ ){
372  fPhantomMaterials.push_back( (*mimite) );
373  }
374  //
375  //---- Build and add new materials
376  std::map< std::pair<G4Material*,G4int>, matInfo* >::iterator mppite;
377  for( mppite= newMateDens.begin(); mppite != newMateDens.end(); mppite++ ){
378  G4double averdens = (*mppite).second->fSumdens/(*mppite).second->fNvoxels;
379  G4double saverdens = G4int(1000.001*averdens)/1000.;
380  G4cout << "GmReadPhantomGeometry::ReadVoxelDensities AVER DENS "
381  << averdens << " -> " << saverdens << " -> "
382  << G4int(1000*averdens) << " " << G4int(1000*averdens)/1000 << " "
383  << G4int(1000*averdens)/1000. << G4endl;
384 
385  G4cout << "GmReadPhantomGeometry::ReadVoxelDensities AVER DENS "
386  << averdens << " -> " << saverdens << " -> "
387  << G4UIcommand::ConvertToString(saverdens) << " Nvoxels "
388  <<(*mppite).second->fNvoxels << G4endl;
389  G4String mateName = ((*mppite).first).first->GetName() + "_" +
390  G4UIcommand::ConvertToString(saverdens);
391  fPhantomMaterials.push_back( BuildMaterialWithChangingDensity(
392  (*mppite).first.first,
393  averdens, mateName ) );
394  }
