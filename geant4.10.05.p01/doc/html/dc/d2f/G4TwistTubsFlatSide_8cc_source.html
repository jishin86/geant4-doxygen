81 G4TwistTubsFlatSide::G4TwistTubsFlatSide( const G4String &name,
82  G4double EndInnerRadius[2],
83  G4double EndOuterRadius[2],
84  G4double DPhi,
85  G4double EndPhi[2],
86  G4double EndZ[2],
87  G4int handedness )
88  : G4VTwistSurface(name)
89 {
90  fHandedness = handedness; // +z = +ve, -z = -ve
91  fAxis[0] = kRho; // in local coordinate system
92  fAxis[1] = kPhi;
93  G4int i = (handedness < 0 ? 0 : 1);
94  fAxisMin[0] = EndInnerRadius[i]; // Inner-hype radius at z=0
95  fAxisMax[0] = EndOuterRadius[i]; // Outer-hype radius at z=0
96  fAxisMin[1] = -0.5*DPhi;
97  fAxisMax[1] = -fAxisMin[1];
98  fCurrentNormal.normal.set(0, 0, (fHandedness < 0 ? -1 : 1));
99  // Unit vector, in local coordinate system
100  fRot.rotateZ(EndPhi[i]);
101  fTrans.set(0, 0, EndZ[i]);
102  fIsValidNorm = true;
103 
104  SetCorners();
105  SetBoundaries();
106 
107  fSurfaceArea = 0.5*DPhi * (EndOuterRadius[i]*EndOuterRadius[i]
108  - EndInnerRadius[i]*EndInnerRadius[i] ) ;
109 
110 }

145 G4int G4TwistTubsFlatSide::DistanceToSurface(const G4ThreeVector &gp,
146  const G4ThreeVector &gv,
147  G4ThreeVector gxx[],
148  G4double distance[],
149  G4int areacode[],
150  G4bool isvalid[],
151  EValidate validate)
152 {
153  fCurStatWithV.ResetfDone(validate, &gp, &gv);
154 
155  if (fCurStatWithV.IsDone()) {
156  G4int i;
157  for (i=0; i<fCurStatWithV.GetNXX(); i++) {
158  gxx[i] = fCurStatWithV.GetXX(i);
159  distance[i] = fCurStatWithV.GetDistance(i);
160  areacode[i] = fCurStatWithV.GetAreacode(i);
161  isvalid[i] = fCurStatWithV.IsValid(i);
162  }
163  return fCurStatWithV.GetNXX();
164  } else {
165  // initialize
166  G4int i;
167  for (i=0; i<2; i++) {
168  distance[i] = kInfinity;
169  areacode[i] = sOutside;
170  isvalid[i] = false;
171  gxx[i].set(kInfinity, kInfinity, kInfinity);
172  }
173  }
174 
175  G4ThreeVector p = ComputeLocalPoint(gp);
176  G4ThreeVector v = ComputeLocalDirection(gv);
177 
178  //
179  // special case!
180  // if p is on surface, distance = 0.
181  //
182 
183  if (std::fabs(p.z()) == 0.) { // if p is on the plane
184  distance[0] = 0;
185  G4ThreeVector xx = p;
186  gxx[0] = ComputeGlobalPoint(xx);
187 
188  if (validate == kValidateWithTol) {
189  areacode[0] = GetAreaCode(xx);
190  if (!IsOutside(areacode[0])) {
191  isvalid[0] = true;
192  }
193  } else if (validate == kValidateWithoutTol) {
194  areacode[0] = GetAreaCode(xx, false);
195  if (IsInside(areacode[0])) {
196  isvalid[0] = true;
197  }
198  } else { // kDontValidate
199  areacode[0] = sInside;
200  isvalid[0] = true;
201  }
202 
203  return 1;
204  }
205  //
206  // special case end
207  //
208 
209  if (v.z() == 0) {
210 
211  fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
212  isvalid[0], 0, validate, &gp, &gv);
213  return 0;
214  }
215 
216  distance[0] = - (p.z() / v.z());
217 
218  G4ThreeVector xx = p + distance[0]*v;
219  gxx[0] = ComputeGlobalPoint(xx);
220 
221  if (validate == kValidateWithTol) {
222  areacode[0] = GetAreaCode(xx);
223  if (!IsOutside(areacode[0])) {
224  if (distance[0] >= 0) isvalid[0] = true;
225  }
226  } else if (validate == kValidateWithoutTol) {
227  areacode[0] = GetAreaCode(xx, false);
228  if (IsInside(areacode[0])) {
229  if (distance[0] >= 0) isvalid[0] = true;
230  }
231  } else { // kDontValidate
232  areacode[0] = sInside;
233  if (distance[0] >= 0) isvalid[0] = true;
234  }
235 
236  fCurStatWithV.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
237  isvalid[0], 1, validate, &gp, &gv);
238 
239 #ifdef G4TWISTDEBUG
240  G4cerr << "ERROR - G4TwistTubsFlatSide::DistanceToSurface(p,v)" << G4endl;
241  G4cerr << " Name : " << GetName() << G4endl;
242  G4cerr << " xx : " << xx << G4endl;
243  G4cerr << " gxx[0] : " << gxx[0] << G4endl;
244  G4cerr << " dist[0] : " << distance[0] << G4endl;
245  G4cerr << " areacode[0] : " << areacode[0] << G4endl;
246  G4cerr << " isvalid[0] : " << isvalid[0] << G4endl;
247 #endif
248  return 1;
249 }

254 G4int G4TwistTubsFlatSide::DistanceToSurface(const G4ThreeVector &gp,
255  G4ThreeVector gxx[],
256  G4double distance[],
257  G4int areacode[])
258 {
259  // Calculate distance to plane in local coordinate,
260  // then return distance and global intersection points.
261  //
262 
263  fCurStat.ResetfDone(kDontValidate, &gp);
264 
265  if (fCurStat.IsDone()) {
266  G4int i;
267  for (i=0; i<fCurStat.GetNXX(); i++) {
268  gxx[i] = fCurStat.GetXX(i);
269  distance[i] = fCurStat.GetDistance(i);
270  areacode[i] = fCurStat.GetAreacode(i);
271  }
272  return fCurStat.GetNXX();
273  } else {
274  // initialize
275  G4int i;
276  for (i=0; i<2; i++) {
277  distance[i] = kInfinity;
278  areacode[i] = sOutside;
279  gxx[i].set(kInfinity, kInfinity, kInfinity);
280  }
281  }
282 
283  G4ThreeVector p = ComputeLocalPoint(gp);
284  G4ThreeVector xx;
285 
286  // The plane is placed on origin with making its normal
287  // parallel to z-axis.
288  if (std::fabs(p.z()) <= 0.5 * kCarTolerance) { // if p is on the plane, return 1
289  distance[0] = 0;
290  xx = p;
291  } else {
292  distance[0] = std::fabs(p.z());
293  xx.set(p.x(), p.y(), 0);
294  }
295 
296  gxx[0] = ComputeGlobalPoint(xx);
297  areacode[0] = sInside;
298  G4bool isvalid = true;
299  fCurStat.SetCurrentStatus(0, gxx[0], distance[0], areacode[0],
300  isvalid, 1, kDontValidate, &gp);
301  return 1;
302

308 G4int G4TwistTubsFlatSide::GetAreaCode(const G4ThreeVector &xx,
309  G4bool withTol)
310 {
311  const G4double rtol
312  = 0.5*G4GeometryTolerance::GetInstance()->GetRadialTolerance();
313 
314  G4int areacode = sInside;
315 
316  if (fAxis[0] == kRho && fAxis[1] == kPhi) {
317  G4int rhoaxis = 0;
318  // G4int phiaxis = 0;
319 
320  G4ThreeVector dphimin; // direction of phi-minimum boundary
321  G4ThreeVector dphimax; // direction of phi-maximum boundary
322  dphimin = GetCorner(sC0Max1Min);
323  dphimax = GetCorner(sC0Max1Max);
324 
325  if (withTol) {
326 
327  G4bool isoutside = false;
328 
329  // test boundary of rho-axis
330 
331  if (xx.getRho() <= fAxisMin[rhoaxis] + rtol) {
332 
333  areacode |= (sAxis0 & (sAxisRho | sAxisMin)) | sBoundary; // rho-min
334  if (xx.getRho() < fAxisMin[rhoaxis] - rtol) isoutside = true;
335 
336  } else if (xx.getRho() >= fAxisMax[rhoaxis] - rtol) {
337 
338  areacode |= (sAxis0 & (sAxisRho | sAxisMax)) | sBoundary; // rho-max
339  if (xx.getRho() > fAxisMax[rhoaxis] + rtol) isoutside = true;
340 
341  }
342 
343  // test boundary of phi-axis
344 
345  if (AmIOnLeftSide(xx, dphimin) >= 0) { // xx is on dphimin
346 
347  areacode |= (sAxis1 & (sAxisPhi | sAxisMin));
348  if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
349  else areacode |= sBoundary;
350 
351  if (AmIOnLeftSide(xx, dphimin) > 0) isoutside = true;
352 
353  } else if (AmIOnLeftSide(xx, dphimax) <= 0) { // xx is on dphimax
354 
355  areacode |= (sAxis1 & (sAxisPhi | sAxisMax));
356  if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
357  else areacode |= sBoundary;
358 
359  if (AmIOnLeftSide(xx, dphimax) < 0) isoutside = true;
360 
361  }
362 
363  // if isoutside = true, clear inside bit.
364  // if not on boundary, add axis information.
365 
366  if (isoutside) {
367  G4int tmpareacode = areacode & (~sInside);
368  areacode = tmpareacode;
369  } else if ((areacode & sBoundary) != sBoundary) {
370  areacode |= (sAxis0 & sAxisRho) | (sAxis1 & sAxisPhi);
371  }
372 
373  } else {
374 
375  // out of boundary of rho-axis
376 
377  if (xx.getRho() < fAxisMin[rhoaxis]) {
378  areacode |= (sAxis0 & (sAxisRho | sAxisMin)) | sBoundary;
379  } else if (xx.getRho() > fAxisMax[rhoaxis]) {
380  areacode |= (sAxis0 & (sAxisRho | sAxisMax)) | sBoundary;
381  }
382 
383  // out of boundary of phi-axis
384 
385  if (AmIOnLeftSide(xx, dphimin, false) >= 0) { // xx is leftside or
386  areacode |= (sAxis1 & (sAxisPhi | sAxisMin)) ; // boundary of dphimin
387  if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
388  else areacode |= sBoundary;
389 
390  } else if (AmIOnLeftSide(xx, dphimax, false) <= 0) { // xx is rightside or
391  areacode |= (sAxis1 & (sAxisPhi | sAxisMax)) ; // boundary of dphimax
392  if (areacode & sBoundary) areacode |= sCorner; // xx is on the corner.
393  else areacode |= sBoundary;
394 
395  }
396 
397  if ((areacode & sBoundary) != sBoundary) {
398  areacode |= (sAxis0 & sAxisRho) | (sAxis1 & sAxisPhi);
399  }
400 
401  }
402  return areacode;
403  } else {
404 
405  std::ostringstream message;
406  message << "Feature NOT implemented !" << G4endl
407  << " fAxis[0] = " << fAxis[0] << G4endl
408  << " fAxis[1] = " << fAxis[1];
409  G4Exception("G4TwistTubsFlatSide::GetAreaCode()", "GeomSolids0001",
410  FatalException, message);
411  }
412  return areacode;
413 }

419 void G4TwistTubsFlatSide::SetCorners()
420 {
421  // Set Corner points in local coodinate.
422 
423  if (fAxis[0] == kRho && fAxis[1] == kPhi) {
424 
425  G4int rhoaxis = 0; // kRho
426  G4int phiaxis = 1; // kPhi
427 
428  G4double x, y, z;
429  // corner of Axis0min and Axis1min
430  x = fAxisMin[rhoaxis]*std::cos(fAxisMin[phiaxis]);
431  y = fAxisMin[rhoaxis]*std::sin(fAxisMin[phiaxis]);
432  z = 0;
433  SetCorner(sC0Min1Min, x, y, z);
434  // corner of Axis0max and Axis1min
435  x = fAxisMax[rhoaxis]*std::cos(fAxisMin[phiaxis]);
436  y = fAxisMax[rhoaxis]*std::sin(fAxisMin[phiaxis]);
437  z = 0;
438  SetCorner(sC0Max1Min, x, y, z);
439  // corner of Axis0max and Axis1max
440  x = fAxisMax[rhoaxis]*std::cos(fAxisMax[phiaxis]);
441  y = fAxisMax[rhoaxis]*std::sin(fAxisMax[phiaxis]);
442  z = 0;
443  SetCorner(sC0Max1Max, x, y, z);
444  // corner of Axis0min and Axis1max
445  x = fAxisMin[rhoaxis]*std::cos(fAxisMax[phiaxis]);
446  y = fAxisMin[rhoaxis]*std::sin(fAxisMax[phiaxis]);
447  z = 0;
448  SetCorner(sC0Min1Max, x, y, z);
449 
450  } else {
451  std::ostringstream message;

508  G4int faces[][4], G4int iside )

518

528  phimax = GetBoundaryMax(r) ;

531  {

533

535  p = SurfacePoint(phi,r,true) ; // surface point in global coord.system
536 
537  xyz[nnode][0] = p.x() ;
538  xyz[nnode][1] = p.y() ;
539  xyz[nnode][2] = p.z() ;
540 
541  if ( i<n-1 && j<k-1 ) { // conterclock wise filling
542

544
