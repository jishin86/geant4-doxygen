75 G4double G4HadronicInteraction::GetMinEnergy(
76  const G4Material *aMaterial, const G4Element *anElement ) const
77 {
78  if( IsBlocked(aMaterial) ) { return 0.0; }
79  if( IsBlocked(anElement) ) { return 0.0; }
80  size_t length = theMinEnergyListElements.size();
81  if(0 < length) {
82  for(size_t i=0; i<length; ++i ) {
83  if( anElement == theMinEnergyListElements[i].second )
84  { return theMinEnergyListElements[i].first; }
85  }
86  }
87  length = theMinEnergyList.size();
88  if(0 < length) {
89  for(size_t i=0; i<length; ++i ) {
90  if( aMaterial == theMinEnergyList[i].second )
91  { return theMinEnergyList[i].first; }
92  }
93  }
94  if(IsBlocked()) { return 0.0; }
95  if( verboseLevel > 1 ) {
96  G4cout << "*** Warning from HadronicInteraction::GetMinEnergy" << G4endl
97  << " material " << aMaterial->GetName()
98  << " not found in min energy List" << G4endl;
99  }
100  return theMinEnergy;
101 }
