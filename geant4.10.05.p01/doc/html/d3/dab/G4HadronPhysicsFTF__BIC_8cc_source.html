101 void G4HadronPhysicsFTF_BIC::Neutron()
102 {
103  //General schema:
104  // 1) Create a builder
105  // 2) Call AddBuilder
106  // 3) Configure the builder, possibly with sub-builders
107  // 4) Call builder->Build()
108  auto neu = new G4NeutronBuilder;
109  AddBuilder(neu);
110  auto ftfn = new G4FTFBinaryNeutronBuilder(QuasiElastic);
111  AddBuilder( ftfn );
112  neu->RegisterMe(ftfn);
113  auto bicn = new G4BinaryNeutronBuilder;
114  AddBuilder(bicn);
115  neu->RegisterMe(bicn);
116  bicn->SetMinEnergy(0.*GeV);
117  bicn->SetMaxEnergy(maxBIC_neutron);
118  neu->Build();
119 }

200 void G4HadronPhysicsFTF_BIC::ExtraConfiguration()
201 {
202  // --- Kaons ---
203  auto xsk = new G4ComponentGGHadronNucleusXsc();
204  G4VCrossSectionDataSet * kaonxs = new G4CrossSectionInelastic(xsk);
205  G4PhysListUtil::FindInelasticProcess(G4KaonMinus::KaonMinus())->AddDataSet(kaonxs);
206  G4PhysListUtil::FindInelasticProcess(G4KaonPlus::KaonPlus())->AddDataSet(kaonxs);
207  G4PhysListUtil::FindInelasticProcess(G4KaonZeroShort::KaonZeroShort())->AddDataSet(kaonxs);
208  G4PhysListUtil::FindInelasticProcess(G4KaonZeroLong::KaonZeroLong())->AddDataSet(kaonxs);
209 
210  // --- Neutrons ---
211  const G4ParticleDefinition* neutron = G4Neutron::Neutron();
212  G4HadronicProcess* inel = G4PhysListUtil::FindInelasticProcess(neutron);
213  if(inel) { inel->AddDataSet(new G4NeutronInelasticXS()); }
214  G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
215  if (capture) {
216  capture->RegisterMe(new G4NeutronRadCapture());
217  }
218 }
