72 void G4MagneticFieldModel::DescribeYourselfTo (G4VGraphicsScene& sceneHandler)
73 {
74 // G4cout << "G4MagneticFieldModel::DescribeYourselfTo" << G4endl;
75 
76  const G4VisExtent& extent = sceneHandler.GetExtent();
77  const G4double xMin = extent.GetXmin();
78  const G4double yMin = extent.GetYmin();
79  const G4double zMin = extent.GetZmin();
80  const G4double xMax = extent.GetXmax();
81  const G4double yMax = extent.GetYmax();
82  const G4double zMax = extent.GetZmax();
83  const G4double xHalfScene = 0.5 * (xMax - xMin);
84  const G4double yHalfScene = 0.5 * (yMax - yMin);
85  const G4double zHalfScene = 0.5 * (zMax - zMin);
86  const G4double xSceneCentre = 0.5 * (xMax + xMin);
87  const G4double ySceneCentre = 0.5 * (yMax + yMin);
88  const G4double zSceneCentre = 0.5 * (zMax + zMin);
89  const G4double maxHalfScene =
90  std::max(xHalfScene,std::max(yHalfScene,zHalfScene));
91  if (maxHalfScene <= 0.) {
92  G4cout
93  << "Extent non-positive."
94  << G4endl;
95  return;
96  }
97 
98  G4TransportationManager* tMgr =
99  G4TransportationManager::GetTransportationManager();
100  assert(tMgr);
101  G4Navigator* navigator = tMgr->GetNavigatorForTracking();
102  assert(navigator);
103 
104  G4FieldManager* globalFieldMgr = tMgr->GetFieldManager();
105  const G4Field* globalField = 0;
106  const G4String intro = "G4MagneticFieldModel::DescribeYourselfTo: ";
107  if (globalFieldMgr) {
108  if (globalFieldMgr->DoesFieldExist()) {
109  globalField = globalFieldMgr->GetDetectorField();
110  if (!globalField) {
111  static G4bool warned = false;
112  if (!warned) {
113  G4cout << intro
114  << "Null global field pointer."
115  << G4endl;
116  warned = true;
117  }
118  }
119  }
120  } else {
121  static G4bool warned = false;
122  if (!warned) {
123  G4cout << intro
124  << "No global field manager."
125  << G4endl;
126  warned = true;
127  }
128  }
129 
130  // Constants
131  const G4double interval = maxHalfScene / fNDataPointsPerMaxHalfScene;
132  const G4int nDataPointsPerXHalfScene = G4int(xHalfScene / interval);
133  const G4int nDataPointsPerYHalfScene = G4int(yHalfScene / interval);
134  const G4int nDataPointsPerZHalfScene = G4int(zHalfScene / interval);
135  const G4int nXSamples = 2 * nDataPointsPerXHalfScene + 1;
136  const G4int nYSamples = 2 * nDataPointsPerYHalfScene + 1;
137  const G4int nZSamples = 2 * nDataPointsPerZHalfScene + 1;
138  const G4int nSamples = nXSamples * nYSamples * nZSamples;
139  const G4int nSamples3 = nSamples * 3;
140  const G4double arrowLengthMax = 0.8 * interval;
141  const G4int nResults = 6; // 3 B-field + 3 E-field.
142 
143  // Working space for GetFieldValue.
144  double position_time[4] = {0,0,0,0};
145  double result[nResults];
146 
147  // Working vectors for field values, etc.
148  std::vector<G4double> BField(nSamples3); // Initialises to zero.
149  std::vector<G4double> BFieldMagnitude(nSamples); // Initialises to zero.
150  std::vector<G4double> xyz(nSamples3); // Initialises to zero.
151 
152  // Get field values and ascertain maximum field.
153  G4double BFieldMagnitudeMax = -std::numeric_limits<G4double>::max();
154  for (G4int i = 0; i < nXSamples; i++) {
155  G4double x = xSceneCentre + (i - nDataPointsPerXHalfScene) * interval;
156  position_time[0] = x;
157  for (G4int j = 0; j < nYSamples; j++) {
158  G4double y = ySceneCentre + (j - nDataPointsPerYHalfScene) * interval;
159  position_time[1] = y;
160  for (G4int k = 0; k < nZSamples; k++) {
161  G4double z = zSceneCentre + (k - nDataPointsPerZHalfScene) * interval;
162  position_time[2] = z;
163  // Calculate indices into working vectors
164  const G4int ijk = i * nYSamples * nZSamples + j * nZSamples + k;
165  const G4int ijk3 = ijk * 3;
166  // Find volume at this location.
167  G4ThreeVector pos(x,y,z);
168  const G4VPhysicalVolume* pPV =
169  navigator->LocateGlobalPointAndSetup(pos,0,false,true);
170  const G4Field* field = globalField;
171  if (pPV) {
172  // Get logical volume.
173  const G4LogicalVolume* pLV = pPV->GetLogicalVolume();
174  if (pLV) {
175  // Value for Region, if any, overrides
176  G4Region* pRegion = pLV->GetRegion();
177  if (pRegion) {
178  G4FieldManager* pRegionFieldMgr = pRegion->GetFieldManager();
179  if (pRegionFieldMgr) {
180  field = pRegionFieldMgr->GetDetectorField();
181  // G4cout << "Region with field" << G4endl;
182  }
183  }
184  // 'Local' value from logical volume, if any, overrides
185  G4FieldManager* pLVFieldMgr = pLV->GetFieldManager();
186  if (pLVFieldMgr) {
187  field = pLVFieldMgr->GetDetectorField();
188  // G4cout << "Logical volume with field" << G4endl;
189  }
190  }
191  }
192  // If field found, get values and store in working vectors.
193  if (field) {
194  // Get field values in result array.
195  field->GetFieldValue(position_time,result);
196  // G4cout
197  // << "BField/T:"
198  // << " " << result[0]/tesla
199  // << " " << result[1]/tesla
200  // << " " << result[2]/tesla
201  // << G4endl;
202  // Store B-field components.
203  for (G4int l = 0; l < 3; l++) {
204  BField[ijk3 + l] = result[l];
205  }
206  // Calculate magnitude and store.
207  G4double mag = sqrt
208  (result[0]*result[0]+result[1]*result[1]+result[2]*result[2]);
209  BFieldMagnitude[ijk] = mag;
210  // Store position.
211  xyz[ijk3] = x;
212  xyz[ijk3 + 1] = y;
213  xyz[ijk3 + 2] = z;
214  // Find maximum field magnitude.
215  if (mag > BFieldMagnitudeMax) {
216  BFieldMagnitudeMax = mag;
217  }
218  }
219  }
220  }
221  }
222 
223  if (BFieldMagnitudeMax <= 0) {
224  G4cout
225  << "No field in this scene."
226  << G4endl;
227  return;
228  }
229 
230  if (fRepresentation == Representation::lightArrow) sceneHandler.BeginPrimitives();
231  for (G4int i = 0; i < nSamples; i++) {
232  if (BFieldMagnitude[i] > 0) {
233  const G4int i3 = i * 3;
234  // Field (Bx,By,Bz) at (x,y,z).
235  const G4double Bx = BField[i3];
236  const G4double By = BField[i3 + 1];
237  const G4double Bz = BField[i3 + 2];
238  const G4double x = xyz[i3];
239  const G4double y = xyz[i3 + 1];
240  const G4double z = xyz[i3 + 2];
241  const G4double B = BFieldMagnitude[i];
242 // G4cout
243 // << "Position/mm, BField/T unpacked:"
244 // << ' ' << x/mm
245 // << ' ' << y/mm
246 // << ' ' << z/mm
247 // << " " << Bx/tesla
248 // << " " << By/tesla
249 // << " " << Bz/tesla
250 // << G4endl;
251  if (B > 0.) {
252  const G4double f = B / BFieldMagnitudeMax;
253  G4double red = 0., green = 0., blue = 0., alpha = 1.;
254  if (f < 0.5) { // Linear colour scale: 0->0.5->1 is blue->green->red.
255  green = 2. * f;
256  blue = 2. * (0.5 - f);
257  } else {
258  red = 2. * (f - 0.5);
259  green = 2. * (1.0 - f);
260  }
261  const G4Colour arrowColour(red,green,blue,alpha);
262  const G4double arrowLength = arrowLengthMax * f;
263  // Base of arrow is at (x,y,z).
264  const G4double& x1 = x;
265  const G4double& y1 = y;
266  const G4double& z1 = z;
267  // Head of arrow depends on field direction and strength.
268  const G4double x2 = x1 + arrowLength * Bx / B;
269  const G4double y2 = y1 + arrowLength * By / B;
270  const G4double z2 = z1 + arrowLength * Bz / B;
271  if (fRepresentation == Representation::fullArrow) {
272  G4ArrowModel BArrow(x1,y1,z1,x2,y2,z2,arrowLength/5,arrowColour,
273  "BField",
274  fArrow3DLineSegmentsPerCircle);
275  BArrow.DescribeYourselfTo(sceneHandler);
276  } else if (fRepresentation == Representation::lightArrow) {
277  G4Polyline BArrowLite;
278  G4VisAttributes va(arrowColour);
279  BArrowLite.SetVisAttributes(va);
280  BArrowLite.push_back(G4Point3D(x1,y1,z1));
281  BArrowLite.push_back(G4Point3D(x2,y2,z2));
282  sceneHandler.AddPrimitive(BArrowLite);
283  }
284  }
285  }
286  }
287  if (fRepresentation == Representation::lightArrow) sceneHandler.EndPrimitives();
288 }
