107 void G4GDMLRead::StripNames() const
108 {
109  // Strips off names of volumes, solids elements and materials from possible
110  // reference pointers or IDs attached to their original identifiers.
111 
112  G4PhysicalVolumeStore* pvols = G4PhysicalVolumeStore::GetInstance();
113  G4LogicalVolumeStore* lvols = G4LogicalVolumeStore::GetInstance();
114  G4SolidStore* solids = G4SolidStore::GetInstance();
115  const G4ElementTable* elements = G4Element::GetElementTable();
116  const G4MaterialTable* materials = G4Material::GetMaterialTable();
117 
118  G4cout << "Stripping off GDML names of materials, solids and volumes ..."
119  << G4endl;
120 
121  G4String sname;
122  size_t i;
123 
124  // Solids...
125  //
126  for (i=0; i<solids->size(); ++i)
127  {
128  G4VSolid* psol = (*solids)[i];
129  sname = psol->GetName();
130  StripName(sname);
131  psol->SetName(sname);
132  }
133 
134  // Logical volumes...
135  //
136  for (i=0; i<lvols->size(); ++i)
137  {
138  G4LogicalVolume* lvol = (*lvols)[i];
139  sname = lvol->GetName();
140  StripName(sname);
141  lvol->SetName(sname);
142  }
143 
144  // Physical volumes...
145  //
146  for (i=0; i<pvols->size(); ++i)
147  {
148  G4VPhysicalVolume* pvol = (*pvols)[i];
149  sname = pvol->GetName();
150  StripName(sname);
151  pvol->SetName(sname);
152  }
153 
154  // Materials...
155  //
156  for (i=0; i<materials->size(); ++i)
157  {
158  G4Material* pmat = (*materials)[i];
159  sname = pmat->GetName();
160  StripName(sname);
161  pmat->SetName(sname);
162  }
163 
164  // Elements...
165  //
166  for (i=0; i<elements->size(); ++i)
167  {
168  G4Element* pelm = (*elements)[i];
169  sname = pelm->GetName();
170  StripName(sname);
171  pelm->SetName(sname);
172  }
173 }

175 void G4GDMLRead::LoopRead(const xercesc::DOMElement* const element,
176  void(G4GDMLRead::*func)(const xercesc::DOMElement* const))
177 {
178  G4String var;
179  G4String from;
180  G4String to;
181  G4String step;
182 
183  const xercesc::DOMNamedNodeMap* const attributes = element->getAttributes();
184  XMLSize_t attributeCount = attributes->getLength();
185 
186  for (XMLSize_t attribute_index=0;
187  attribute_index<attributeCount;attribute_index++)
188  {
189  xercesc::DOMNode* attribute_node = attributes->item(attribute_index);
190 
191  if (attribute_node->getNodeType() != xercesc::DOMNode::ATTRIBUTE_NODE)
192  { continue; }
193 
194  const xercesc::DOMAttr* const attribute
195  = dynamic_cast<xercesc::DOMAttr*>(attribute_node);
196  if (!attribute)
197  {
198  G4Exception("G4GDMLRead::LoopRead()", "InvalidRead",
199  FatalException, "No attribute found!");
200  return;
201  }
202  const G4String attribute_name = Transcode(attribute->getName());
203  const G4String attribute_value = Transcode(attribute->getValue());
204 
205  if (attribute_name=="for") { var = attribute_value; } else
206  if (attribute_name=="from") { from = attribute_value; } else
207  if (attribute_name=="to") { to = attribute_value; } else
208  if (attribute_name=="step") { step = attribute_value; }
209  }
210 
211  if (var.empty())
212  {
213  G4Exception("G4GDMLRead::loopRead()", "InvalidRead",
214  FatalException, "No variable is determined for loop!");
215  }
216 
217  if (!eval.IsVariable(var))
218  {
219  G4Exception("G4GDMLRead::loopRead()", "InvalidRead",
220  FatalException, "Variable is not defined in loop!");
221  }
222 
223  G4int _var = eval.EvaluateInteger(var);
224  G4int _from = eval.EvaluateInteger(from);
225  G4int _to = eval.EvaluateInteger(to);
226  G4int _step = eval.EvaluateInteger(step);
227 
228  if (!from.empty()) { _var = _from; }
229 
230  if (_from == _to)
231  {
232  G4Exception("G4GDMLRead::loopRead()", "InvalidRead",
233  FatalException, "Empty loop!");
234  }
235  if ((_from < _to) && (_step <= 0))
236  {
237  G4Exception("G4GDMLRead::loopRead()", "InvalidRead",
238  FatalException, "Infinite loop!");
239  }
240  if ((_from > _to) && (_step >= 0))
241  {
242  G4Exception("G4GDMLRead::loopRead()", "InvalidRead",
243  FatalException, "Infinite loop!");
244  }
245 
246  inLoop++;
247 
248  while (_var <= _to)
249  {
250  eval.SetVariable(var,_var);
251  (this->*func)(element);
252  _var += _step;
253  loopCount++;
254  }
255 
256  inLoop--;
257  if (!inLoop) { loopCount = 0; }
258 }

260 G4GDMLAuxStructType G4GDMLRead::
261 AuxiliaryRead(const xercesc::DOMElement* const auxiliaryElement)
262 {
263  G4GDMLAuxStructType auxstruct = {"","","",0};
264  G4GDMLAuxListType* auxList=0;
265 
266  const xercesc::DOMNamedNodeMap* const attributes
267  = auxiliaryElement->getAttributes();
268  XMLSize_t attributeCount = attributes->getLength();
269 
270  for (XMLSize_t attribute_index=0;
271  attribute_index<attributeCount; attribute_index++)
272  {
273  xercesc::DOMNode* attribute_node = attributes->item(attribute_index);
274 
275  if (attribute_node->getNodeType() != xercesc::DOMNode::ATTRIBUTE_NODE)
276  { continue; }
277 
278  const xercesc::DOMAttr* const attribute
279  = dynamic_cast<xercesc::DOMAttr*>(attribute_node);
280  if (!attribute)
281  {
282  G4Exception("G4GDMLRead::AuxiliaryRead()",
283  "InvalidRead", FatalException, "No attribute found!");
284  return auxstruct;
285  }
286  const G4String attName = Transcode(attribute->getName());
287  const G4String attValue = Transcode(attribute->getValue());
288 
289  if (attName=="auxtype") { auxstruct.type = attValue; } else
290  if (attName=="auxvalue") { auxstruct.value = attValue; } else
291  if (attName=="auxunit") { auxstruct.unit = attValue; }
292  }
293 
294  for (xercesc::DOMNode* iter = auxiliaryElement->getFirstChild();
295  iter != 0; iter = iter->getNextSibling())
296  {
297  if (iter->getNodeType() != xercesc::DOMNode::ELEMENT_NODE) { continue; }
298 
299  const xercesc::DOMElement* const child
300  = dynamic_cast<xercesc::DOMElement*>(iter);
301  if (!child)
302  {
303  G4Exception("G4GDMLRead::AuxiliaryRead()",
304  "InvalidRead", FatalException, "No child found!");
305  break;
306  }
307  const G4String tag = Transcode(child->getTagName());
308 
309  if (tag=="auxiliary")
310  {
311  if(!auxList) { auxList = new G4GDMLAuxListType; }
312  auxList->push_back(AuxiliaryRead(child));
313  }
314  }
315 
316  if (auxList) { auxstruct.auxList = auxList; }
317 
318  return auxstruct;
319 }

361 void G4GDMLRead::Read(const G4String& fileName,
362  G4bool validation,
363  G4bool isModule,
364  G4bool strip)
365 {
366  dostrip = strip;
367 #ifdef G4VERBOSE
368  if (isModule)
369  {
370  G4cout << "G4GDML: Reading module '" << fileName << "'..." << G4endl;
371  }
372  else
373  {
374  G4cout << "G4GDML: Reading '" << fileName << "'..." << G4endl;
375  }
376 #endif
377  inLoop = 0;
378  validate = validation;
379 
380  xercesc::ErrorHandler* handler = new G4GDMLErrorHandler(!validate);
381  xercesc::XercesDOMParser* parser = new xercesc::XercesDOMParser;
382 
383  if (validate)
384  {
385  parser->setValidationScheme(xercesc::XercesDOMParser::Val_Always);
386  }
387  parser->setValidationSchemaFullChecking(validate);
388  parser->setCreateEntityReferenceNodes(false);
389  // Entities will be automatically resolved by Xerces
390 
391  parser->setDoNamespaces(true);
392  parser->setDoSchema(validate);
393  parser->setErrorHandler(handler);
394 
395  try { parser->parse(fileName.c_str()); }
396  catch (const xercesc::XMLException &e)
397  { G4cout << "G4GDML: " << Transcode(e.getMessage()) << G4endl; }
398  catch (const xercesc::DOMException &e)
399  { G4cout << "G4GDML: " << Transcode(e.getMessage()) << G4endl; }
400 
401  xercesc::DOMDocument* doc = parser->getDocument();
402 
403  if (!doc)
404  {
405  G4String error_msg = "Unable to open document: " + fileName;
406  G4Exception("G4GDMLRead::Read()", "InvalidRead",
407  FatalException, error_msg);
408  return;
409  }
410  xercesc::DOMElement* element = doc->getDocumentElement();
411 
412  if (!element)
413  {
414  std::ostringstream message;
415  message << "ERROR - Empty document or unable to validate schema!" << G4endl
416  << " Check Internet connection is ON in case of schema"
417  << G4endl
418  << " validation enabled and location defined as URL in"
419  << G4endl
420  << " the GDML file - " << fileName << " - being imported!"
421  << G4endl
422  << " Otherwise, verify GDML schema server is reachable!";
423  G4Exception("G4GDMLRead::Read()", "InvalidRead", FatalException, message);
424  return;
425  }
426 
427  for (xercesc::DOMNode* iter = element->getFirstChild();
428  iter != 0; iter = iter->getNextSibling())
429  {
430  if (iter->getNodeType() != xercesc::DOMNode::ELEMENT_NODE) { continue; }
431 
432  const xercesc::DOMElement* const child
433  = dynamic_cast<xercesc::DOMElement*>(iter);
434  if (!child)
435  {
436  G4Exception("G4GDMLRead::Read()", "InvalidRead",
437  FatalException, "No child found!");
438  return;
439  }
440  const G4String tag = Transcode(child->getTagName());
441 
442  if (tag=="define") { DefineRead(child); } else
443  if (tag=="materials") { MaterialsRead(child); } else
444  if (tag=="solids") { SolidsRead(child); } else
445  if (tag=="setup") { SetupRead(child); } else
446  if (tag=="structure") { StructureRead(child); } else
447  if (tag=="userinfo") { UserinfoRead(child); } else
448  if (tag=="extension") { ExtensionRead(child); }
449  else
450  {
451  G4String error_msg = "Unknown tag in gdml: " + tag;
452  G4Exception("G4GDMLRead::Read()", "InvalidRead",
453  FatalException, error_msg);
454  }
455  }
456 
457  delete parser;
458  delete handler;
459 
460  if (isModule)
461  {
462 #ifdef G4VERBOSE
463  G4cout << "G4GDML: Reading module '" << fileName << "' done!" << G4endl;
464 #endif
465  }
466  else
467  {
468  G4cout << "G4GDML: Reading '" << fileName << "' done!" << G4endl;
469  if (strip) { StripNames(); }
470  }
471 }
