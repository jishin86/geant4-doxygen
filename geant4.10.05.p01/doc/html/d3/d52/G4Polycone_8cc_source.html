62 G4Polycone::G4Polycone( const G4String& name,
63  G4double phiStart,
64  G4double phiTotal,
65  G4int numZPlanes,
66  const G4double zPlane[],
67  const G4double rInner[],
68  const G4double rOuter[] )
69  : G4VCSGfaceted( name )
70 {
71  //
72  // Some historical ugliness
73  //
74  original_parameters = new G4PolyconeHistorical();
75 
76  original_parameters->Start_angle = phiStart;
77  original_parameters->Opening_angle = phiTotal;
78  original_parameters->Num_z_planes = numZPlanes;
79  original_parameters->Z_values = new G4double[numZPlanes];
80  original_parameters->Rmin = new G4double[numZPlanes];
81  original_parameters->Rmax = new G4double[numZPlanes];
82 
83  G4int i;
84  for (i=0; i<numZPlanes; i++)
85  {
86  if(rInner[i]>rOuter[i])
87  {
88  DumpInfo();
89  std::ostringstream message;
90  message << "Cannot create a Polycone with rInner > rOuter for the same Z"
91  << G4endl
92  << " rInner > rOuter for the same Z !" << G4endl
93  << " rMin[" << i << "] = " << rInner[i]
94  << " -- rMax[" << i << "] = " << rOuter[i];
95  G4Exception("G4Polycone::G4Polycone()", "GeomSolids0002",
96  FatalErrorInArgument, message);
97  }
98  if (( i < numZPlanes-1) && ( zPlane[i] == zPlane[i+1] ))
99  {
100  if( (rInner[i] > rOuter[i+1])
101  ||(rInner[i+1] > rOuter[i]) )
102  {
103  DumpInfo();
104  std::ostringstream message;
105  message << "Cannot create a Polycone with no contiguous segments."
106  << G4endl
107  << " Segments are not contiguous !" << G4endl
108  << " rMin[" << i << "] = " << rInner[i]
109  << " -- rMax[" << i+1 << "] = " << rOuter[i+1] << G4endl
110  << " rMin[" << i+1 << "] = " << rInner[i+1]
111  << " -- rMax[" << i << "] = " << rOuter[i];
112  G4Exception("G4Polycone::G4Polycone()", "GeomSolids0002",
113  FatalErrorInArgument, message);
114  }
115  }
116  original_parameters->Z_values[i] = zPlane[i];
117  original_parameters->Rmin[i] = rInner[i];
118  original_parameters->Rmax[i] = rOuter[i];
119  }
120 
121  //
122  // Build RZ polygon using special PCON/PGON GEANT3 constructor
123  //
124  G4ReduciblePolygon *rz =
125  new G4ReduciblePolygon( rInner, rOuter, zPlane, numZPlanes );
126 
127  //
128  // Do the real work
129  //
130  Create( phiStart, phiTotal, rz );
131 
132  delete rz;
133 }

175 // Create
176 //
177 // Generic create routine, called by each constructor after
178 // conversion of arguments
179 //
180 void G4Polycone::Create( G4double phiStart,
181  G4double phiTotal,
182  G4ReduciblePolygon *rz )
183 {
184  //
185  // Perform checks of rz values
186  //
187  if (rz->Amin() < 0.0)
188  {
189  std::ostringstream message;
190  message << "Illegal input parameters - " << GetName() << G4endl
191  << " All R values must be >= 0 !";
192  G4Exception("G4Polycone::Create()", "GeomSolids0002",
193  FatalErrorInArgument, message);
194  }
195 
196  G4double rzArea = rz->Area();
197  if (rzArea < -kCarTolerance)
198  {
199  rz->ReverseOrder();
200  }
201  else if (rzArea < kCarTolerance)
202  {
203  std::ostringstream message;
204  message << "Illegal input parameters - " << GetName() << G4endl
205  << " R/Z cross section is zero or near zero: " << rzArea;
206  G4Exception("G4Polycone::Create()", "GeomSolids0002",
207  FatalErrorInArgument, message);
208  }
209 
210  if ( (!rz->RemoveDuplicateVertices( kCarTolerance ))
211  || (!rz->RemoveRedundantVertices( kCarTolerance )) )
212  {
213  std::ostringstream message;
214  message << "Illegal input parameters - " << GetName() << G4endl
215  << " Too few unique R/Z values !";
216  G4Exception("G4Polycone::Create()", "GeomSolids0002",
217  FatalErrorInArgument, message);
218  }
219 
220  if (rz->CrossesItself(1/kInfinity))
221  {
222  std::ostringstream message;
223  message << "Illegal input parameters - " << GetName() << G4endl
224  << " R/Z segments cross !";
225  G4Exception("G4Polycone::Create()", "GeomSolids0002",
226  FatalErrorInArgument, message);
227  }
228 
229  numCorner = rz->NumVertices();
230 
231  //
232  // Phi opening? Account for some possible roundoff, and interpret
233  // nonsense value as representing no phi opening
234  //
235  if (phiTotal <= 0 || phiTotal > twopi-1E-10)
236  {
237  phiIsOpen = false;
238  startPhi = 0;
239  endPhi = twopi;
240  }
241  else
242  {
243  phiIsOpen = true;
244 
245  //
246  // Convert phi into our convention
247  //
248  startPhi = phiStart;
249  while( startPhi < 0 ) // Loop checking, 13.08.2015, G.Cosmo
250  startPhi += twopi;
251 
252  endPhi = phiStart+phiTotal;
253  while( endPhi < startPhi ) // Loop checking, 13.08.2015, G.Cosmo
254  endPhi += twopi;
255  }
256 
257  //
258  // Allocate corner array.
259  //
260  corners = new G4PolyconeSideRZ[numCorner];
261 
262  //
263  // Copy corners
264  //
265  G4ReduciblePolygonIterator iterRZ(rz);
266 
267  G4PolyconeSideRZ *next = corners;
268  iterRZ.Begin();
269  do // Loop checking, 13.08.2015, G.Cosmo
270  {
271  next->r = iterRZ.GetA();
272  next->z = iterRZ.GetB();
273  } while( ++next, iterRZ.Next() );
274 
275  //
276  // Allocate face pointer array
277  //
278  numFace = phiIsOpen ? numCorner+2 : numCorner;
279  faces = new G4VCSGface*[numFace];
280 
281  //
282  // Construct conical faces
283  //
284  // But! Don't construct a face if both points are at zero radius!
285  //
286  G4PolyconeSideRZ *corner = corners,
287  *prev = corners + numCorner-1,
288  *nextNext;
289  G4VCSGface **face = faces;
290  do // Loop checking, 13.08.2015, G.Cosmo
291  {
292  next = corner+1;
293  if (next >= corners+numCorner) next = corners;
294  nextNext = next+1;
295  if (nextNext >= corners+numCorner) nextNext = corners;
296 
297  if (corner->r < 1/kInfinity && next->r < 1/kInfinity) continue;
298 
299  //
300  // We must decide here if we can dare declare one of our faces
301  // as having a "valid" normal (i.e. allBehind = true). This
302  // is never possible if the face faces "inward" in r.
303  //
304  G4bool allBehind;
305  if (corner->z > next->z)
306  {
307  allBehind = false;
308  }
309  else
310  {
311  //
312  // Otherwise, it is only true if the line passing
313  // through the two points of the segment do not
314  // split the r/z cross section
315  //
316  allBehind = !rz->BisectedBy( corner->r, corner->z,
317  next->r, next->z, kCarTolerance );
318  }
319 
320  *face++ = new G4PolyconeSide( prev, corner, next, nextNext,
321  startPhi, endPhi-startPhi, phiIsOpen, allBehind );
322  } while( prev=corner, corner=next, corner > corners );
323 
324  if (phiIsOpen)
325  {
326  //
327  // Construct phi open edges
328  //
329  *face++ = new G4PolyPhiFace( rz, startPhi, 0, endPhi );
330  *face++ = new G4PolyPhiFace( rz, endPhi, 0, startPhi );
331  }
332 
333  //
334  // We might have dropped a face or two: recalculate numFace
335  //
336  numFace = face-faces;
337 
338  //
339  // Make enclosingCylinder
340  //
341  enclosingCylinder =
342  new G4EnclosingCylinder( rz, phiIsOpen, phiStart, phiTotal );
343 }

525 void G4Polycone::BoundingLimits(G4ThreeVector& pMin,
526  G4ThreeVector& pMax) const
527 {
528  G4double rmin = kInfinity, rmax = -kInfinity;
529  G4double zmin = kInfinity, zmax = -kInfinity;
530 
531  for (G4int i=0; i<GetNumRZCorner(); ++i)
532  {
533  G4PolyconeSideRZ corner = GetCorner(i);
534  if (corner.r < rmin) rmin = corner.r;
535  if (corner.r > rmax) rmax = corner.r;
536  if (corner.z < zmin) zmin = corner.z;
537  if (corner.z > zmax) zmax = corner.z;
538  }
539 
540  if (IsOpen())
541  {
542  G4TwoVector vmin,vmax;
543  G4GeomTools::DiskExtent(rmin,rmax,
544  GetSinStartPhi(),GetCosStartPhi(),
545  GetSinEndPhi(),GetCosEndPhi(),
546  vmin,vmax);
547  pMin.set(vmin.x(),vmin.y(),zmin);
548  pMax.set(vmax.x(),vmax.y(),zmax);
549  }
550  else
551  {
552  pMin.set(-rmax,-rmax, zmin);
553  pMax.set( rmax, rmax, zmax);
554  }
555 
556  // Check correctness of the bounding box
557  //
558  if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
559  {
560  std::ostringstream message;
561  message << "Bad bounding box (min >= max) for solid: "
562  << GetName() << " !"
563  << "\npMin = " << pMin
564  << "\npMax = " << pMax;
565  G4Exception("G4Polycone::BoundingLimits()", "GeomMgt0001",
566  JustWarning, message);
567  DumpInfo();
568  }
569 }

572 //
573 // Calculate extent under transform and specified limit
574 
575 G4bool G4Polycone::CalculateExtent(const EAxis pAxis,
576  const G4VoxelLimits& pVoxelLimit,
577  const G4AffineTransform& pTransform,
578  G4double& pMin, G4double& pMax) const
579 {
580  G4ThreeVector bmin, bmax;
581  G4bool exist;
582 
583  // Check bounding box (bbox)
584  //
585  BoundingLimits(bmin,bmax);
586  G4BoundingEnvelope bbox(bmin,bmax);
587 #ifdef G4BBOX_EXTENT
588  if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
589 #endif
590  if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
591  {
592  return exist = (pMin < pMax) ? true : false;
593  }
594 
595  // To find the extent, RZ contour of the polycone is subdivided
596  // in triangles. The extent is calculated as cumulative extent of
597  // all sub-polycones formed by rotation of triangles around Z
598  //
599  G4TwoVectorList contourRZ;
600  G4TwoVectorList triangles;
601  std::vector<G4int> iout;
602  G4double eminlim = pVoxelLimit.GetMinExtent(pAxis);
603  G4double emaxlim = pVoxelLimit.GetMaxExtent(pAxis);
604 
605  // get RZ contour, ensure anticlockwise order of corners
606  for (G4int i=0; i<GetNumRZCorner(); ++i)
607  {
608  G4PolyconeSideRZ corner = GetCorner(i);
609  contourRZ.push_back(G4TwoVector(corner.r,corner.z));
610  }
611  G4GeomTools::RemoveRedundantVertices(contourRZ,iout,2*kCarTolerance);
612  G4double area = G4GeomTools::PolygonArea(contourRZ);
613  if (area < 0.) std::reverse(contourRZ.begin(),contourRZ.end());
614 
615  // triangulate RZ countour
616  if (!G4GeomTools::TriangulatePolygon(contourRZ,triangles))
617  {
618  std::ostringstream message;
619  message << "Triangulation of RZ contour has failed for solid: "
620  << GetName() << " !"
621  << "\nExtent has been calculated using boundary box";
622  G4Exception("G4Polycone::CalculateExtent()",
623  "GeomMgt1002", JustWarning, message);
624  return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
625  }
626 
627  // set trigonometric values
628  const G4int NSTEPS = 24; // number of steps for whole circle
629  G4double astep = twopi/NSTEPS; // max angle for one step
630 
631  G4double sphi = GetStartPhi();
632  G4double ephi = GetEndPhi();
633  G4double dphi = IsOpen() ? ephi-sphi : twopi;
634  G4int ksteps = (dphi <= astep) ? 1 : (G4int)((dphi-deg)/astep) + 1;
635  G4double ang = dphi/ksteps;
636 
637  G4double sinHalf = std::sin(0.5*ang);
638  G4double cosHalf = std::cos(0.5*ang);
639  G4double sinStep = 2.*sinHalf*cosHalf;
640  G4double cosStep = 1. - 2.*sinHalf*sinHalf;
641 
642  G4double sinStart = GetSinStartPhi();
643  G4double cosStart = GetCosStartPhi();
644  G4double sinEnd = GetSinEndPhi();
645  G4double cosEnd = GetCosEndPhi();
646 
647  // define vectors and arrays
648  std::vector<const G4ThreeVectorList *> polygons;
649  polygons.resize(ksteps+2);
650  G4ThreeVectorList pols[NSTEPS+2];
651  for (G4int k=0; k<ksteps+2; ++k) pols[k].resize(6);
652  for (G4int k=0; k<ksteps+2; ++k) polygons[k] = &pols[k];
653  G4double r0[6],z0[6]; // contour with original edges of triangle
654  G4double r1[6]; // shifted radii of external edges of triangle
655 
656  // main loop along triangles
657  pMin = kInfinity;
658  pMax =-kInfinity;
659  G4int ntria = triangles.size()/3;
660  for (G4int i=0; i<ntria; ++i)
661  {
662  G4int i3 = i*3;
663  for (G4int k=0; k<3; ++k)
664  {
665  G4int e0 = i3+k, e1 = (k<2) ? e0+1 : i3;
666  G4int k2 = k*2;
667  // set contour with original edges of triangle
668  r0[k2+0] = triangles[e0].x(); z0[k2+0] = triangles[e0].y();
669  r0[k2+1] = triangles[e1].x(); z0[k2+1] = triangles[e1].y();
670  // set shifted radii
671  r1[k2+0] = r0[k2+0];
672  r1[k2+1] = r0[k2+1];
673  if (z0[k2+1] - z0[k2+0] <= 0) continue;
674  r1[k2+0] /= cosHalf;
675  r1[k2+1] /= cosHalf;
676  }
677 
678  // rotate countour, set sequence of 6-sided polygons
679  G4double sinCur = sinStart*cosHalf + cosStart*sinHalf;
680  G4double cosCur = cosStart*cosHalf - sinStart*sinHalf;
681  for (G4int j=0; j<6; ++j) pols[0][j].set(r0[j]*cosStart,r0[j]*sinStart,z0[j]);
682  for (G4int k=1; k<ksteps+1; ++k)
683  {
684  for (G4int j=0; j<6; ++j) pols[k][j].set(r1[j]*cosCur,r1[j]*sinCur,z0[j]);
685  G4double sinTmp = sinCur;
686  sinCur = sinCur*cosStep + cosCur*sinStep;
687  cosCur = cosCur*cosStep - sinTmp*sinStep;
688  }
689  for (G4int j=0; j<6; ++j) pols[ksteps+1][j].set(r0[j]*cosEnd,r0[j]*sinEnd,z0[j]);
690
