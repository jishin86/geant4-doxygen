51 void TrackingAction::PreUserTrackingAction(const G4Track* track )
52 {
53  //get Run
54  Run* run = static_cast<Run*>(
55  G4RunManager::GetRunManager()->GetNonConstCurrentRun());
56 
57  // Energy flow initialisation for primary particle
58  //
59  if (track->GetTrackID() == 1) {
60  G4int Idnow = 1;
61  if (track->GetVolume() != fDetector->GetphysiWorld()) {
62  // unique identificator of layer+absorber
63  const G4VTouchable* touchable = track->GetTouchable();
64  G4int absorNum = touchable->GetCopyNumber();
65  G4int layerNum = touchable->GetReplicaNumber(1);
66  Idnow = (fDetector->GetNbOfAbsor())*layerNum + absorNum;
67  }
68 
69  G4double Eflow = track->GetKineticEnergy();
70  if (track->GetDefinition() == G4Positron::Positron()) {
71  Eflow += 2*electron_mass_c2;
72  }
73 
74  //flux artefact, if primary vertex is inside the calorimeter
75  for (G4int pl=1; pl<=Idnow; ++pl) {run->SumEnergyFlow(pl, Eflow);}
76  } else {
77  run->AddSecondaryTrack(track);
78  }
79 }
