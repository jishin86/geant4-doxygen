41 GB03DetectorMessenger::GB03DetectorMessenger(GB03DetectorConstruction* det)
42  : G4UImessenger(), fDetector(det)
43 {
44  fDirectory = new G4UIdirectory("/GB03/");
45  fDirectory->SetGuidance("UI commands of this example");
46 
47  G4String matList;
48  const G4MaterialTable* matTbl = G4Material::GetMaterialTable();
49  for(size_t i=0;i<G4Material::GetNumberOfMaterials();i++)
50  {
51  matList += (*matTbl)[i]->GetName();
52  matList += " ";
53  }
54 
55  fAbsMaterialCmd = new G4UIcmdWithAString("/GB03/setAbsMat",this);
56  fAbsMaterialCmd->SetGuidance("Select Material of the Absorber.");
57  fAbsMaterialCmd->SetParameterName("choice",false);
58  fAbsMaterialCmd->AvailableForStates(G4State_Idle);
59  fAbsMaterialCmd->SetCandidates(matList);
60 
61  fGapMaterialCmd = new G4UIcmdWithAString("/GB03/setGapMat",this);
62  fGapMaterialCmd->SetGuidance("Select Material of the Gap.");
63  fGapMaterialCmd->SetParameterName("choice",false);
64  fGapMaterialCmd->AvailableForStates(G4State_Idle);
65  fGapMaterialCmd->SetCandidates(matList);
66 
67  fNumLayerCmd = new G4UIcmdWithAnInteger("/GB03/numberOfLayers",this);
68  fNumLayerCmd->SetGuidance("Set number of layers.");
69  fNumLayerCmd->SetParameterName("nl",false);
70  fNumLayerCmd->AvailableForStates(G4State_Idle);
71  fNumLayerCmd->SetRange("nl>0");
72 
73  fVerboseCmd = new G4UIcmdWithAnInteger("/GB03/verbose",this);
74  fVerboseCmd->SetGuidance("Set verbosity level");
75  fVerboseCmd->SetParameterName("verbose",false);
76  fVerboseCmd->AvailableForStates(G4State_Idle);
77  fVerboseCmd->SetRange("verbose>=0");

79 }
