40 bool
41 G4TrajectoryParticleFilter::Evaluate(const G4VTrajectory& traj) const
42 {
43  G4String particle = traj.GetParticleName();
44 
45  if (GetVerbose()) G4cout<<"G4TrajectoryParticleFilter processing trajectory with particle type: "<<particle<<G4endl;
46 
47  std::vector<G4String>::const_iterator iter = std::find(fParticles.begin(), fParticles.end(), particle);
48 
49  // Fail if particle type not found in particle list
50  if (iter == fParticles.end()) return false;
51 
52  return true;
53 }
