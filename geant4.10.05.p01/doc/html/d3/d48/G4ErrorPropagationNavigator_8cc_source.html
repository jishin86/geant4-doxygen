59 G4double G4ErrorPropagationNavigator::
60 ComputeStep ( const G4ThreeVector &pGlobalPoint,
61  const G4ThreeVector &pDirection,
62  const G4double pCurrentProposedStepLength,
63  G4double &pNewSafety )
64 {
65  G4double safetyGeom= DBL_MAX;
66 
67  G4double Step = G4Navigator::ComputeStep(pGlobalPoint, pDirection,
68  pCurrentProposedStepLength,
69  safetyGeom);
70 
71  G4ErrorPropagatorData * g4edata
72  = G4ErrorPropagatorData::GetErrorPropagatorData();
73 
74  if (g4edata !=0)
75  {
76  const G4ErrorTarget* target = g4edata->GetTarget();
77  if( target != 0 )
78  {
79  G4double StepPlane= target->GetDistanceFromPoint(pGlobalPoint,pDirection);
80 
81  if( StepPlane < 0. ) // Negative means target is crossed, will not be found
82  {
83  StepPlane = DBL_MAX;
84  }
85 #ifdef G4VERBOSE
86  if( G4ErrorPropagatorData::verbose() >= 4 )
87  {
88  G4cout << "G4ErrorPropagationNavigator::ComputeStep()" << G4endl
89  << " Target step: " << StepPlane
90  << ", Transportation step: " << Step << G4endl;
91  target->Dump( "G4ErrorPropagationNavigator::ComputeStep Target " );
92  }
93 #endif
94 
95  if(StepPlane<Step)
96  {
97 #ifdef G4VERBOSE
98  if( G4ErrorPropagatorData::verbose() >= 2 )
99  {
100  G4cout << "G4ErrorPropagationNavigator::ComputeStep()" << G4endl
101  << " TargetCloserThanBoundary: " << StepPlane << " < "
102  << Step << G4endl;
103  }
104 #endif
105  Step = StepPlane;
106  g4edata->SetState(G4ErrorState_TargetCloserThanBoundary);
107  }
108  else
109  {
110  g4edata->SetState(G4ErrorState_Propagating);
111  }
112  }
113  }
114  G4double safetyTarget = TargetSafetyFromPoint(pGlobalPoint);
115  // Avoid call to G4Navigator::ComputeSafety - which could have side effects
116  pNewSafety= std::min(safetyGeom, safetyTarget);
117 
118 #ifdef G4VERBOSE
119  if( G4ErrorPropagatorData::verbose() >= 3 )
120  {
121  G4cout << "G4ErrorPropagationNavigator::ComputeStep()" << G4endl
122  << " Step: " << Step << ", ComputeSafety: " << pNewSafety
123  << G4endl;
124  }
125 #endif
126 
127  return Step;
128 }

168 G4ThreeVector G4ErrorPropagationNavigator::
169 GetGlobalExitNormal(const G4ThreeVector& point, G4bool* valid)
170 {
171  G4ErrorPropagatorData *g4edata
172  = G4ErrorPropagatorData::GetErrorPropagatorData();
173  const G4ErrorTarget* target = 0;
174 
175  G4ThreeVector normal(0.0, 0.0, 0.0);
176  G4double distance= 0;
177 
178  // Determine which 'geometry' limited the step
179  if (g4edata)
180  {
181  target = g4edata->GetTarget();
182  if(target)
183  {
184  distance = target->GetDistanceFromPoint(point);
185  }
186  }
187 
188  if( distance > kCarTolerance // Not reached the target.
189  || (!target) )
190  // If a target does not exist, this seems the best we can do
191  {
192  normal= G4Navigator::GetGlobalExitNormal(point, valid);
193  }
194  else
195  {
196  switch( target->GetType() )
197  {
198  case G4ErrorTarget_GeomVolume:
199  // The volume is in the 'real' mass geometry
200  normal= G4Navigator::GetGlobalExitNormal(point, valid);
201  break;
202  case G4ErrorTarget_TrkL:
203  normal= G4ThreeVector( 0.0, 0.0, 0.0);
204  *valid= false;
205  G4Exception("G4ErrorPropagationNavigator::GetGlobalExitNormal",
206  "Geometry1003",
207  JustWarning, "Unexpected value of Target type");
208  break;
209  case G4ErrorTarget_PlaneSurface:
210  case G4ErrorTarget_CylindricalSurface:
211  const G4ErrorSurfaceTarget* surfaceTarget=
212  static_cast<const G4ErrorSurfaceTarget*>(target);
213  normal= surfaceTarget->GetTangentPlane(point).normal().unit();
214  *valid= true;
215  break;
216 
217 // default:
218 // normal= G4ThreeVector( 0.0, 0.0, 0.0);
219 // *valid= false;
220 // G4Exception("G4ErrorPropagationNavigator::GetGlobalExitNormal",
221 // "Geometry:003",
222 // FatalException, "Impossible value of Target type");
223 // exit(1);
224 // break;
225  }
226  }
227  return normal;
228 }
