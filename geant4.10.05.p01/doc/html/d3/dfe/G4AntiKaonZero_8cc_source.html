52 G4AntiKaonZero* G4AntiKaonZero::Definition()
53 {
54  if (theInstance !=0) return theInstance;
55  const G4String name = "anti_kaon0";
56  // search in particle table]
57  G4ParticleTable* pTable = G4ParticleTable::GetParticleTable();
58  G4ParticleDefinition* anInstance = pTable->FindParticle(name);
59  if (anInstance ==0)
60  {
61  // create particle
62  //
63  // Arguments for constructor are as follows
64  // name mass width charge
65  // 2*spin parity C-conjugation
66  // 2*Isospin 2*Isospin3 G-parity
67  // type lepton number baryon number PDG encoding
68  // stable lifetime decay table
69  // shortlived subType anti_encoding
70 
71  anInstance = new G4ParticleDefinition(
72  name, 0.497614*GeV, 0.0*MeV, 0.0,
73  0, -1, 0,
74  1, 1, 0,
75  "meson", 0, 0, -311,
76  false, 0.0, NULL,
77  false, "kaon");
78  //create Decay Table
79  G4DecayTable* table = new G4DecayTable();
80 
81  // create decay channels
82  G4VDecayChannel** mode = new G4VDecayChannel*[2];
83  // kaon0 -> Kaon0L
84  mode[0] = new G4PhaseSpaceDecayChannel("anti_kaon0",0.500,1,"kaon0L");
85  // kaon0 -> Kaon0S
86  mode[1] = new G4PhaseSpaceDecayChannel("anti_kaon0",0.500,1,"kaon0S");
87 
88  for (G4int index=0; index <2; index++ ) table->Insert(mode[index]);
89  delete [] mode;
90 
91  anInstance->SetDecayTable(table);
92  }
93  theInstance = reinterpret_cast<G4AntiKaonZero*>(anInstance);
94  return theInstance;
95 }
