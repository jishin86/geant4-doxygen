85 void G4HadronPhysicsQGSP_BIC_HP::CreateModels()
86 {
87  G4bool quasiElasticFTF= false; // Use built-in quasi-elastic (not add-on)
88  G4bool quasiElasticQGS= true; // For QGS, it must use it.
89 
90  const G4double maxFTFP = 25.0*GeV;
91  const G4double minFTFP = 9.5*GeV;
92  const G4double maxBIC = 9.9*GeV;
93  const G4double maxBERT = 5.0*GeV;
94  const G4double maxHP = 19.9*MeV;
95 
96  tpdata->theNeutrons=new G4NeutronBuilder( true ); // Fission on
97  tpdata->theNeutrons->RegisterMe(tpdata->theQGSPNeutron=new G4QGSPNeutronBuilder(quasiElasticQGS));
98  tpdata->theNeutrons->RegisterMe(tpdata->theFTFPNeutron=new G4FTFPNeutronBuilder(quasiElasticFTF));
99  tpdata->theFTFPNeutron->SetMinEnergy(minFTFP);
100  tpdata->theFTFPNeutron->SetMaxEnergy(maxFTFP);
101 
102  tpdata->theNeutrons->RegisterMe(tpdata->theBinaryNeutron=new G4BinaryNeutronBuilder);
103  tpdata->theBinaryNeutron->SetMinEnergy(maxHP);
104  tpdata->theBinaryNeutron->SetMaxEnergy(maxBIC);
105 
106  tpdata->theNeutrons->RegisterMe(tpdata->theHPNeutron=new G4NeutronPHPBuilder);
107 
108  tpdata->thePro=new G4ProtonBuilder;
109  tpdata->thePro->RegisterMe(tpdata->theQGSPPro=new G4QGSPProtonBuilder(quasiElasticQGS));
110  tpdata->thePro->RegisterMe(tpdata->theFTFPPro=new G4FTFPProtonBuilder(quasiElasticFTF));
111  tpdata->theFTFPPro->SetMinEnergy(minFTFP);
112  tpdata->theFTFPPro->SetMaxEnergy(maxFTFP);
113  tpdata->thePro->RegisterMe(tpdata->theBinaryPro=new G4BinaryProtonBuilder);
114  tpdata->theBinaryPro->SetMaxEnergy(maxBIC);
115 
116  tpdata->thePiK=new G4PiKBuilder;
117  tpdata->thePiK->RegisterMe(tpdata->theQGSPPiK=new G4QGSPPiKBuilder(quasiElasticQGS));
118  tpdata->thePiK->RegisterMe(tpdata->theFTFPPiK=new G4FTFPPiKBuilder(quasiElasticFTF));
119  tpdata->theFTFPPiK->SetMaxEnergy(maxFTFP);
120  tpdata->thePiK->RegisterMe(tpdata->theBertiniPiK=new G4BertiniPiKBuilder);
121  tpdata->theBertiniPiK->SetMaxEnergy(maxBERT);
122 
123  tpdata->theHyperon=new G4HyperonFTFPBuilder;
124 
125  tpdata->theAntiBaryon=new G4AntiBarionBuilder;
126  tpdata->theAntiBaryon->RegisterMe(tpdata->theFTFPAntiBaryon=new G4FTFPAntiBarionBuilder(quasiElasticFTF));
127 }

167 void G4HadronPhysicsQGSP_BIC_HP::ConstructProcess()
168 {
169  if ( !tpdata ) tpdata = new ThreadPrivate;
170  CreateModels();
171  tpdata->theNeutrons->Build();
172  tpdata->thePro->Build();
173  tpdata->thePiK->Build();
174 
175  // --- Kaons ---
176  G4VCrossSectionDataSet * kaonxs =
177  new G4CrossSectionInelastic(new G4ComponentGGHadronNucleusXsc());
178  G4PhysListUtil::FindInelasticProcess(G4KaonMinus::KaonMinus())->AddDataSet(kaonxs);
179  G4PhysListUtil::FindInelasticProcess(G4KaonPlus::KaonPlus())->AddDataSet(kaonxs);
180  G4PhysListUtil::FindInelasticProcess(G4KaonZeroShort::KaonZeroShort())->AddDataSet(kaonxs);
181  G4PhysListUtil::FindInelasticProcess(G4KaonZeroLong::KaonZeroLong())->AddDataSet(kaonxs);
182 
183  tpdata->theHyperon->Build();
184  tpdata->theAntiBaryon->Build();
185 
186  // --- Neutrons ---
187  const G4ParticleDefinition* neutron = G4Neutron::Neutron();
188  G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
189  if (capture) {
190  G4NeutronRadCapture* theNeutronRadCapture = new G4NeutronRadCapture();
191  theNeutronRadCapture->SetMinEnergy( 19.9*MeV );
192  capture->RegisterMe( theNeutronRadCapture );
193  }
194  G4HadronicProcess* fission = G4PhysListUtil::FindFissionProcess(neutron);
195  if (fission) {
196  G4LFission* theNeutronLEPFission = new G4LFission();
197  theNeutronLEPFission->SetMinEnergy( 19.9*MeV );
198  theNeutronLEPFission->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
199  fission->RegisterMe( theNeutronLEPFission );
200  }
201 }
