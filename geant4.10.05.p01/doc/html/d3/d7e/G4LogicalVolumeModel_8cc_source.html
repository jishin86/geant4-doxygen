47 G4LogicalVolumeModel::G4LogicalVolumeModel
48 (G4LogicalVolume* pLV,
49  G4int soughtDepth,
50  G4bool booleans,
51  G4bool voxels,
52  G4bool readout,
53  G4bool checkOverlaps,
54  const G4Transform3D& modelTransformation,
55  const G4ModelingParameters* pMP):
56  // Instantiate a G4PhysicalVolumeModel with a G4PVPlacement to
57  // represent this logical volume. It has no rotation and a null
58  // translation so that the logical volume will be seen in its own
59  // reference system. It will be added to the physical volume store
60  // but it will not be part of the normal geometry heirarchy so it
61  // has no mother.
62  G4PhysicalVolumeModel
63 (new G4PVPlacement (0, // No rotation.
64  G4ThreeVector(), // Null traslation.
65  "PhysVol representation of LogVol " + pLV -> GetName (),
66  pLV,
67  0, // No mother.
68  false, // Not "MANY".
69  0), // Copy number.
70  soughtDepth,
71  modelTransformation,
72  pMP,
73  true), // Use full extent.
74  fpLV (pLV),
75  fBooleans (booleans),
76  fVoxels (voxels),
77  fReadout (readout),
78  fCheckOverlaps(checkOverlaps),
79  fOverlapsPrinted(false)
80 {
81  fType = "G4LogicalVolumeModel";
82  fGlobalTag = fpLV -> GetName ();
83  fGlobalDescription = "G4LogicalVolumeModel " + fGlobalTag;
84 }

89  // Keep a vector of solid-transform pairs to avoid duplication.
90  typedef std::pair<G4VSolid*,G4Transform3D> solidTransformPair;
91  std::vector<solidTransformPair> solidTransformVector;
92  void drawSolidsAndPoint
93  (G4VGraphicsScene& sceneHandler,
94  const G4ThreeVector& point,
95  G4VSolid* sol1, const G4Transform3D& t1,
96  G4VSolid* sol2, const G4Transform3D& t2)
97  {
98  const G4Colour highlightSolidColour(1.0,0.8,0.8);
99  const G4double highlightSolidLineWidth(10./*pixels*/);
100  const G4Colour highlightPointColour(0.5,0.5,1.0);
101  const G4double highlightPointDiameter(20./*pixels*/);
102 
103  // Draw first solid. Avoid duplication.
104  std::pair<G4VSolid*,G4Transform3D> pair1(sol1,t1);
105  auto iter1 = solidTransformVector.begin();
106  for ( ; iter1 != solidTransformVector.end(); ++iter1) {
107  if (iter1->first == pair1.first &&
108  iter1->second == pair1.second) break;
109  }
110  if (iter1 == solidTransformVector.end()) {
111  solidTransformVector.push_back(pair1);
112  G4VisAttributes highlightSolidVisAtts(highlightSolidColour);
113  highlightSolidVisAtts.SetLineWidth(highlightSolidLineWidth);
114  sceneHandler.PreAddSolid(t1,highlightSolidVisAtts);
115  sceneHandler.AddSolid(*sol1);
116  sceneHandler.PostAddSolid();
117  }
118 
119  // Draw second solid. Avoid duplication.
120  std::pair<G4VSolid*,G4Transform3D> pair2(sol2,t2);
121  auto iter2 = solidTransformVector.begin();
122  for ( ; iter2 != solidTransformVector.end(); ++iter2) {
123  if (iter2->first == pair2.first &&
124  iter2->second == pair2.second) break;
125  }
126  if (iter2 == solidTransformVector.end()) {
127  solidTransformVector.push_back(pair2);
128  G4VisAttributes highlightSolidVisAtts(highlightSolidColour);
129  highlightSolidVisAtts.SetLineWidth(highlightSolidLineWidth);
130  sceneHandler.PreAddSolid(t2,highlightSolidVisAtts);
131  sceneHandler.AddSolid(*sol2);
132  sceneHandler.PostAddSolid();
133  }
134 
135  // Draw points. Draw them all.
136  G4VisAttributes highlightPointVisAtts(highlightPointColour);
137  G4Circle overlapPoint;
138  overlapPoint.SetVisAttributes(highlightPointVisAtts);
139  overlapPoint.SetPosition(point);
140  overlapPoint.SetDiameter(G4VMarker::SizeType::screen,highlightPointDiameter);
141  overlapPoint.SetFillStyle(G4VMarker::FillStyle::filled);
142  sceneHandler.BeginPrimitives();
143  sceneHandler.AddPrimitive(overlapPoint);
144  sceneHandler.EndPrimitives();
145  }

148 void G4LogicalVolumeModel::DescribeYourselfTo
149 (G4VGraphicsScene& sceneHandler) {
150 
151  // Store current modeling parameters and ensure nothing is culled.
152  const G4ModelingParameters* tmpMP = fpMP;
153  G4ModelingParameters nonCulledMP;
154  if (fpMP) nonCulledMP = *fpMP;
155  nonCulledMP.SetCulling (false);
156  fpMP = &nonCulledMP;
157  G4PhysicalVolumeModel::DescribeYourselfTo (sceneHandler);
158  fpMP = tmpMP;
159 
160  if (fVoxels) {
161  if (fpTopPV->GetLogicalVolume()->GetVoxelHeader()) {
162  // Add Voxels.
163  G4DrawVoxels dv;
164  G4PlacedPolyhedronList* pPPL =
165  dv.CreatePlacedPolyhedra (fpTopPV -> GetLogicalVolume ());
166  for (size_t i = 0; i < pPPL -> size (); i++) {
167  const G4Transform3D& transform = (*pPPL)[i].GetTransform ();
168  const G4Polyhedron& polyhedron = (*pPPL)[i].GetPolyhedron ();
169  sceneHandler.BeginPrimitives (transform);
170  sceneHandler.AddPrimitive (polyhedron);
171  sceneHandler.EndPrimitives ();
172  }
173  delete pPPL;
174  }
175  }
176 
177  if (fReadout) {
178  // Draw readout geometry...
179  G4VSensitiveDetector* sd = fpLV->GetSensitiveDetector();
180  if (sd) {
181  G4VReadOutGeometry* roGeom = sd->GetROgeometry();
182  if (roGeom) {
183  G4VPhysicalVolume* roWorld = roGeom->GetROWorld();
184  G4cout << "Readout geometry \"" << roGeom->GetName()
185  << "\" with top physical volume \""
186  << roWorld->GetName()
187  << "\"" << G4endl;
188  G4PhysicalVolumeModel pvModel(roWorld);
189  pvModel.SetModelingParameters(fpMP);
190  pvModel.DescribeYourselfTo(sceneHandler);
191  }
192  }
193  }
194 
195  if (fCheckOverlaps) {
196  G4LogicalVolume* motherLog = fpTopPV->GetLogicalVolume();
197  G4VSolid* motherSolid = motherLog->GetSolid();
198  G4int nDaughters = motherLog->GetNoDaughters();
199 
200  // Models are called repeatedly by the scene handler so be careful...
201  // Print overlaps - but only the first time for a given instantiation of G4LogicalVolume
202  if (!fOverlapsPrinted) {
203  for (G4int iDaughter = 0; iDaughter < nDaughters; ++iDaughter) {
204  G4VPhysicalVolume* daughterPhys = motherLog->GetDaughter(iDaughter);
205  daughterPhys->CheckOverlaps();
206  }
207  fOverlapsPrinted = true;
208  }
209 
210  // Draw overlaps. This algorithm is based on G4PVPlacement::CheckOverlaps.
211  solidTransformVector.clear();
212  for (G4int iDaughter = 0; iDaughter < nDaughters; ++iDaughter) {
213  G4VPhysicalVolume* daughterPhys = motherLog->GetDaughter(iDaughter);
214  // Replicas and paramaterisations not presently processed
215  if (!dynamic_cast<G4PVPlacement*>(daughterPhys)) continue;
216  G4AffineTransform tDaughter(daughterPhys->GetRotation(),daughterPhys->GetTranslation());
217  G4VSolid* daughterSolid = daughterPhys->GetLogicalVolume()->GetSolid();
218  const G4int nTrials = 1000;
219  for (G4int i = 0; i < nTrials; ++i) {
220  G4ThreeVector p = daughterSolid->GetPointOnSurface();
221  // Transform to mother's coordinate system
222  G4ThreeVector pMother = tDaughter.TransformPoint(p);
223  // Check overlaps with the mother volume
224  if (motherSolid->Inside(pMother)==kOutside) {
225  // Draw mother and daughter and point
226  drawSolidsAndPoint
227  (sceneHandler,pMother,motherSolid,G4Transform3D(),daughterSolid,tDaughter);
228  }
229  // Check other daughters
230  for (G4int iSister = 0; iSister < nDaughters; ++iSister) {
231  if (iSister == iDaughter) continue;
232  G4VPhysicalVolume* sisterPhys = motherLog->GetDaughter(iSister);
233  G4AffineTransform tSister(sisterPhys->GetRotation(),sisterPhys->GetTranslation());
234  // Transform to sister's coordinate system
235  G4ThreeVector pSister = tSister.InverseTransformPoint(pMother);
236  G4LogicalVolume* sisterLog = sisterPhys->GetLogicalVolume();
237  G4VSolid* sisterSolid = sisterLog->GetSolid();
238  if (sisterSolid->Inside(pSister)==kInside) {
239  // Draw daughter and sister and point
240  drawSolidsAndPoint
241  (sceneHandler,pMother,daughterSolid,tDaughter,sisterSolid,tSister);
242  }
243  }
244  }
245  }
246  }

251 void G4LogicalVolumeModel::DescribeSolid
252 (const G4Transform3D& theAT,
253  G4VSolid* pSol,
254  const G4VisAttributes* pVisAttribs,
255  G4VGraphicsScene& sceneHandler) {
256 
257  if (fBooleans) {
258  // Look for "constituents". Could be a Boolean solid.
259  G4VSolid* pSol0 = pSol -> GetConstituentSolid (0);
260  if (pSol0) { // Composite solid...
261  G4VSolid* pSol1 = pSol -> GetConstituentSolid (1);
262  if (!pSol1) {
263  G4Exception
264  ("G4PhysicalVolumeModel::DescribeSolid",
265  "modeling0001", FatalException,
266  "2nd component solid in Boolean is missing.");
267  }
268  // Draw these constituents white and "forced wireframe"...
269  G4VisAttributes constituentAttributes;
270  constituentAttributes.SetForceWireframe(true);
271  DescribeSolid (theAT, pSol0, &constituentAttributes, sceneHandler);
272  DescribeSolid (theAT, pSol1, &constituentAttributes, sceneHandler);
273  }
274  }
275 
276  // In any case draw the original/resultant solid...
277  sceneHandler.PreAddSolid (theAT, *pVisAttribs);
278  pSol -> DescribeYourselfTo (sceneHandler);
279  sceneHandler.PostAddSolid ();
280 }
