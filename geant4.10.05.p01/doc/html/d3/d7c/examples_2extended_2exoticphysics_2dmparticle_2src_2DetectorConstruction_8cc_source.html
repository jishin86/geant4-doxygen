105  G4Box * sAbsor = new G4Box("Absorber",
106  fAbsorSizeXY / 2, fAbsorSizeXY / 2, fAbsorSizeZ / 2);
107 
108  fLogAbsor = new G4LogicalVolume(sAbsor,
109  fAbsorMaterial,
110  "Absorber");
111 
112  new G4PVPlacement(0, //no rotation
113  G4ThreeVector(), //at (0,0,0)
114  fLogAbsor, //logical volume
115  "Absorber", //name
116  lWorld, //mother volume
117  false, //no boolean operation
118  0); //copy number

120  PrintParameters();
121 
122  /************ always return the World volume *****************/

160 void DetectorConstruction::SetMaterial(const G4String& namemat)
161 {

163 
164  G4Material* mat = G4NistManager::Instance()->FindOrBuildMaterial(namemat);

168  return;

175  }
