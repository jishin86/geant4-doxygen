116 void G4HadronPhysicsQGSP_BERT::Neutron()
117 {
118  //General schema:
119  // 1) Create a builder
120  // 2) Call AddBuilder
121  // 3) Configure the builder, possibly with sub-builders
122  // 4) Call builder->Build()
123  auto neu = new G4NeutronBuilder;
124  AddBuilder(neu);
125  auto qgs = new G4QGSPNeutronBuilder(QuasiElasticQGS);
126  AddBuilder(qgs);
127  qgs->SetMinEnergy(minQGSP_neutron);
128  neu->RegisterMe(qgs);
129  auto ftf = new G4FTFPNeutronBuilder(QuasiElasticFTF);
130  AddBuilder(ftf);
131  ftf->SetMinEnergy(minFTFP_neutron);
132  ftf->SetMaxEnergy(maxFTFP_neutron);
133  neu->RegisterMe(ftf);
134  auto bert = new G4BertiniNeutronBuilder;
135  AddBuilder(bert);
136  bert->SetMinEnergy(minBERT_neutron);
137  bert->SetMaxEnergy(maxBERT_neutron);
138  neu->RegisterMe(bert);
139  neu->Build();
140 }

216 void G4HadronPhysicsQGSP_BERT::ExtraConfiguration()
217 {
218  //Modify XS for kaons
219  auto xsk = new G4ComponentGGHadronNucleusXsc();
220  G4VCrossSectionDataSet * kaonxs = new G4CrossSectionInelastic(xsk);
221  G4PhysListUtil::FindInelasticProcess(G4KaonMinus::KaonMinus())->AddDataSet(kaonxs);
222  G4PhysListUtil::FindInelasticProcess(G4KaonPlus::KaonPlus())->AddDataSet(kaonxs);
223  G4PhysListUtil::FindInelasticProcess(G4KaonZeroShort::KaonZeroShort())->AddDataSet(kaonxs);
224  G4PhysListUtil::FindInelasticProcess(G4KaonZeroLong::KaonZeroLong())->AddDataSet(kaonxs);
225 
226  //Modify Neutrons
227  const G4ParticleDefinition* neutron = G4Neutron::Neutron();
228  G4HadronicProcess* inel = G4PhysListUtil::FindInelasticProcess(neutron);
229  if(inel) { inel->AddDataSet(new G4NeutronInelasticXS()); }
230  G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
231  if (capture) {
232  capture->RegisterMe(new G4NeutronRadCapture());
233  }
234 }
