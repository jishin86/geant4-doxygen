75

77  fHistoManager->ScoreNewTrack(aTrack);

82  if(fHistoManager->GetVerbose() > 1 ) {
83  G4cout << "Track #"

85  << " E(MeV)= " << aTrack->GetKineticEnergy()/MeV
86  << " produced by Track ID= " << aTrack->GetParentID()
87  << G4endl;
88  }
89 
90  //stack or delete secondaries
91  if(aTrack->GetTrackID() > 1) {
92  if (fKillSecondary || fParticle == part) { status = fKill; }
93  }
94  return status;
95 }
