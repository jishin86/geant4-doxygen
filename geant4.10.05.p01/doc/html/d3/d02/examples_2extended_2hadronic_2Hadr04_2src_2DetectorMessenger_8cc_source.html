114 void DetectorMessenger::SetNewValue(G4UIcommand* command,G4String newValue)
115 {
116  if( command == fMaterCmd )
117  { fDetector->SetMaterial(newValue);}
118 
119  if( command == fSizeCmd )
120  { fDetector->SetSize(fSizeCmd->GetNewDoubleValue(newValue));}
121 
122  if (command == fIsotopeCmd)
123  {
124  G4int Z; G4int A; G4double dens;
125  G4String name, unt;
126  std::istringstream is(newValue);
127  is >> name >> Z >> A >> dens >> unt;
128  dens *= G4UIcommand::ValueOf(unt);
129  fDetector->MaterialWithSingleIsotope (name,name,dens,Z,A);
130  fDetector->SetMaterial(name);
131  }
132 }
