80 void PrimaryGeneratorAction::GeneratePrimaries(G4Event* anEvent)
81 {
82  //this function is called at the begining of event
83 
84  //randomize the beam, if requested.
85  //
86  if (fRndmBeam > 0.)
87  {
88  G4ThreeVector oldPosition = fParticleGun->GetParticlePosition();
89  if (fRndmBeam > fDetector->GetAbsorSizeYZ())
90  fRndmBeam = fDetector->GetAbsorSizeYZ();
91  G4double rbeam = 0.5*fRndmBeam;
92  G4double x0 = oldPosition.x();
93  G4double y0 = oldPosition.y() + (2*G4UniformRand()-1.)*rbeam;
94  G4double z0 = oldPosition.z() + (2*G4UniformRand()-1.)*rbeam;
95  fParticleGun->SetParticlePosition(G4ThreeVector(x0,y0,z0));
96  fParticleGun->GeneratePrimaryVertex(anEvent);
97  fParticleGun->SetParticlePosition(oldPosition);
98  }
99 
100  else fParticleGun->GeneratePrimaryVertex(anEvent);
101 }
