90 void G4HadronPhysicsQGSP_BERT_HP::Neutron()
91 {
92  auto neu = new G4NeutronBuilder( true ); // Fission on
93  AddBuilder(neu);
94  auto qgs = new G4QGSPNeutronBuilder(QuasiElasticQGS);
95  AddBuilder(qgs);
96  qgs->SetMinEnergy(minQGSP_neutron);
97  neu->RegisterMe(qgs);
98  auto ftf = new G4FTFPNeutronBuilder(QuasiElasticFTF);
99  AddBuilder(ftf);
100  ftf->SetMinEnergy(minFTFP_neutron);
101  ftf->SetMaxEnergy(maxFTFP_neutron);
102  neu->RegisterMe(ftf);
103  auto bert = new G4BertiniNeutronBuilder;
104  AddBuilder(bert);
105  bert->SetMinEnergy(minBERT_neutron);
106  bert->SetMaxEnergy(maxBERT_neutron);
107  neu->RegisterMe(bert);
108  auto hp = new G4NeutronPHPBuilder;
109  AddBuilder(hp);
110  neu->RegisterMe(hp);
111  neu->Build();
112 }

114 void G4HadronPhysicsQGSP_BERT_HP::ExtraConfiguration()
115 {
116  //Modify XS for kaons
117  auto xsk = new G4ComponentGGHadronNucleusXsc();
118  G4VCrossSectionDataSet * kaonxs = new G4CrossSectionInelastic(xsk);
119  G4PhysListUtil::FindInelasticProcess(G4KaonMinus::KaonMinus())->AddDataSet(kaonxs);
120  G4PhysListUtil::FindInelasticProcess(G4KaonPlus::KaonPlus())->AddDataSet(kaonxs);
121  G4PhysListUtil::FindInelasticProcess(G4KaonZeroShort::KaonZeroShort())->AddDataSet(kaonxs);
122  G4PhysListUtil::FindInelasticProcess(G4KaonZeroLong::KaonZeroLong())->AddDataSet(kaonxs);
123 
124  // --- Neutrons ---
125  const G4ParticleDefinition* neutron = G4Neutron::Neutron();
126  G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
127  if (capture) {
128  G4NeutronRadCapture* theNeutronRadCapture = new G4NeutronRadCapture();
129  theNeutronRadCapture->SetMinEnergy( minBERT_neutron );
130  capture->RegisterMe( theNeutronRadCapture );
131  }
132  G4HadronicProcess* fission = G4PhysListUtil::FindFissionProcess(neutron);
133  if (fission) {
134  G4LFission* theNeutronLEPFission = new G4LFission();
135  theNeutronLEPFission->SetMinEnergy( minBERT_neutron );
136  theNeutronLEPFission->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
137  fission->RegisterMe( theNeutronLEPFission );
138  }
139 }
