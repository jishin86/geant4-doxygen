42 void StandaloneVisAction::Draw() {
43  G4VVisManager* pVisManager = G4VVisManager::GetConcreteInstance();
44  if (pVisManager) {
45 
46  // Simple box...
47  pVisManager->Draw(G4Box("box",2*m,2*m,2*m),
48  G4VisAttributes(G4Colour(1,1,0)));
49 
50  // Boolean solid...
51  G4Box boxA("boxA",3*m,3*m,3*m);
52  G4Box boxB("boxB",1*m,1*m,1*m);
53  G4SubtractionSolid subtracted("subtracted_boxes",&boxA,&boxB,
54  G4Translate3D(3*m,3*m,3*m));
55  pVisManager->Draw(subtracted,
56  G4VisAttributes(G4Colour(0,1,1)),
57  G4Translate3D(-6*m,-6*m,-6*m));
58 
59  // Same, but explicit polyhedron...
60  G4Polyhedron* pA = G4Box("boxA",3*m,3*m,3*m).CreatePolyhedron();
61  G4Polyhedron* pB = G4Box("boxB",1*m,1*m,1*m).CreatePolyhedron();
62  pB->Transform(G4Translate3D(3*m,3*m,3*m));
63  G4Polyhedron* pSubtracted = new G4Polyhedron(pA->subtract(*pB));
64  G4VisAttributes subVisAtts(G4Colour(0,1,1));
65  pSubtracted->SetVisAttributes(&subVisAtts);
66  pVisManager->Draw(*pSubtracted,G4Translate3D(6*m,6*m,6*m));
67  delete pA;
68  delete pB;
69  delete pSubtracted;
70  }
71 }
