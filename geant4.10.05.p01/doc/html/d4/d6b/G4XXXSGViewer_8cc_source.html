59 void G4XXXSGViewer::DrawView() {
60  //#ifdef G4XXXSGDEBUG
61  G4cout << "G4XXXSGViewer::DrawView() called." << G4endl;
62  //#endif
63 
64  // First, a view should decide when to re-visit the G4 kernel.
65  // Sometimes it might not be necessary, e.g., if the scene is stored
66  // in a graphical database (e.g., OpenGL's display lists) and only
67  // the viewing angle has changed. But graphics systems without a
68  // graphical database will always need to visit the G4 kernel.
69 
70  // The fNeedKernelVisit flag might have been set by the user in
71  // /vis/viewer/rebuild, but if not, make decision and set flag only
72  // if necessary...
73  if (!fNeedKernelVisit) KernelVisitDecision();
74  G4bool kernelVisitWasNeeded = fNeedKernelVisit; // Keep (ProcessView resets).
75 
76  ProcessView (); // Clears store and processes scene only if necessary.
77 
78  if (kernelVisitWasNeeded) {
79  // Some systems, notably OpenGL, can draw while re-building, so
80  // there might not be a need to draw from store again here. But
81  // in this case...
82  DrawFromStore("G4XXXSGViewer::DrawView");
83  } else {
84  DrawFromStore("G4XXXSGViewer::DrawView");
85  }
86 
87  // ...before finally...
88  FinishView (); // Flush streams and/or swap buffers.
89 }

99 void G4XXXSGViewer::KernelVisitDecision () {
100 
101  // If there's a significant difference with the last view parameters
102  // of either the scene handler or this viewer, trigger a rebuild.
103 
104  SceneGraph& sceneGraph =
105  static_cast<G4XXXSGSceneHandler&>(fSceneHandler).fSceneGraph;
106  if (sceneGraph.fDaughters.size() == 3 // I.e., only the root nodes.
107  // (The above needs re-thinking.)
108  || CompareForKernelVisit(fLastVP)) {
109  NeedKernelVisit (); // Sets fNeedKernelVisit.
110  }
111  fLastVP = fVP;
112 }

114 G4bool G4XXXSGViewer::CompareForKernelVisit(G4ViewParameters& lastVP)
115 {
116  // Typical comparison. Taken from OpenGL.
117  if (
118  (lastVP.GetDrawingStyle () != fVP.GetDrawingStyle ()) ||
119  (lastVP.IsAuxEdgeVisible () != fVP.IsAuxEdgeVisible ()) ||
120  (lastVP.IsCulling () != fVP.IsCulling ()) ||
121  (lastVP.IsCullingInvisible () != fVP.IsCullingInvisible ()) ||
122  (lastVP.IsDensityCulling () != fVP.IsDensityCulling ()) ||
123  (lastVP.IsCullingCovered () != fVP.IsCullingCovered ()) ||
124  (lastVP.GetCBDAlgorithmNumber() !=
125  fVP.GetCBDAlgorithmNumber()) ||
126  // No need to visit kernel if section plane changes.
127  // No need to visit kernel if cutaway planes change.
128  (lastVP.IsExplode () != fVP.IsExplode ()) ||
129  (lastVP.GetNoOfSides () != fVP.GetNoOfSides ()) ||
130  (lastVP.IsMarkerNotHidden () != fVP.IsMarkerNotHidden ()) ||
131  (lastVP.GetDefaultVisAttributes()->GetColour() !=
132  fVP.GetDefaultVisAttributes()->GetColour()) ||
133  (lastVP.GetDefaultTextVisAttributes()->GetColour() !=
134  fVP.GetDefaultTextVisAttributes()->GetColour()) ||
135  (lastVP.GetBackgroundColour ()!= fVP.GetBackgroundColour ())||
136  (lastVP.GetVisAttributesModifiers() !=
137  fVP.GetVisAttributesModifiers())
138  ) {
139  return true;
140  }
141 
142  if (lastVP.IsDensityCulling () &&
143  (lastVP.GetVisibleDensity () != fVP.GetVisibleDensity ()))
144  return true;
145 
146  if (lastVP.GetCBDAlgorithmNumber() > 0) {
147  if (lastVP.GetCBDParameters().size() != fVP.GetCBDParameters().size()) return true;
148  else if (lastVP.GetCBDParameters() != fVP.GetCBDParameters()) return true;
149  }
150 
151  if (lastVP.IsExplode () &&
152  (lastVP.GetExplodeFactor () != fVP.GetExplodeFactor ()))
153  return true;
154 
155  return false;
156 }
