129 void
130 G4ReplicatedSlice::CheckAndSetParameters( const EAxis pAxis,
131  const G4int nDivs,
132  const G4double width,
133  const G4double half_gap,
134  const G4double offset,
135  DivisionType divType,
136  G4LogicalVolume* pMotherLogical,
137  const G4LogicalVolume* pLogical )
138 {
139  if(!pMotherLogical)
140  {
141  std::ostringstream message;
142  message << "Invalid setup." << G4endl
143  << "NULL pointer specified as mother! Volume: " << GetName();
144  G4Exception("G4ReplicatedSlice::CheckAndSetParameters()", "GeomDiv0002",
145  FatalException, message);
146  }
147  if(pLogical == pMotherLogical)
148  {
149  std::ostringstream message;
150  message << "Invalid setup." << G4endl
151  << "Cannot place a volume inside itself! Volume: " << GetName();
152  G4Exception("G4ReplicatedSlice::CheckAndSetParameters()", "GeomDiv0002",
153  FatalException, message);
154  }
155 
156  //----- Check that mother solid is of the same type as
157  // daughter solid (otherwise, the corresponding
158  // Parameterisation::ComputeDimension() will not be called)
159  //
160  G4String msolType = pMotherLogical->GetSolid()->GetEntityType();
161  G4String dsolType = pLogical->GetSolid()->GetEntityType();
162  if( msolType != dsolType && ( msolType != "G4Trd" || dsolType != "G4Trap" ) )
163  {
164  std::ostringstream message;
165  message << "Invalid setup." << G4endl
166  << "Incorrect solid type for division of volume: "
167  << GetName() << G4endl
168  << " It is: " << msolType
169  << ", while it should be: " << dsolType;
170  G4Exception("G4ReplicatedSlice::CheckAndSetParameters()",
171  "GeomDiv0002", FatalException, message);
172  }
173 
174  pMotherLogical->AddDaughter(this);
175  SetMotherLogical(pMotherLogical);
176  SetParameterisation(pMotherLogical, pAxis, nDivs,
177  width, half_gap, offset, divType);
178 
179  if( divType == DivWIDTH )
180  {
181  fnReplicas = fparam->GetNoDiv();
182  }
183  else
184  {
185  fnReplicas = nDivs;
186  }
187  if (fnReplicas < 1 )
188  {
189  G4Exception("G4ReplicatedSlice::CheckAndSetParameters()", "GeomDiv0002",
190  FatalException, "Illegal number of replicas!");
191  }
192  if( divType != DivNDIV)
193  {
194  fwidth = fparam->GetWidth();
195  }
196  else
197  {
198  fwidth = width;
199  }
200  if( fwidth < 0 )
201  {
202  G4Exception("G4ReplicatedSlice::CheckAndSetParameters()", "GeomDiv0002",
203  FatalException, "Width must be positive!");
204  }
205  if( fwidth < 2.*half_gap )
206  {
207  G4Exception("G4ReplicatedSlice::CheckAndSetParameters()", "GeomDiv0002",
208  FatalException, "Half_gap is too large!");
209  }
210 
211  foffset = offset;
212  fdivAxis = pAxis;
213 
215  //
216  if( pAxis == kRho || pAxis == kRadial3D || pAxis == kPhi )
217  {
218  faxis = kZAxis;
219  }
220  else
221  {
222  faxis = pAxis;
223  }
224 
225  switch (faxis)
226  {
227  case kPhi:
228  case kRho:
229  case kXAxis:
230  case kYAxis:
231  case kZAxis:
232  break;
233  default:
234  G4Exception("G4ReplicatedSlice::CheckAndSetParameters()", "GeomDiv0002",
235  FatalException, "Unknown axis of replication.");
236  break;
237  }
238 
239  // Create rotation matrix: for phi axis it will be changed
240  // in G4VPVParameterisation::ComputeTransformation, for others
241  // it will stay the unity
242  //
243  G4RotationMatrix *pRMat = new G4RotationMatrix();
244  SetRotation(pRMat);
245 }

311 void G4ReplicatedSlice::SetParameterisation( G4LogicalVolume* motherLogical,
312  const EAxis axis,
313  const G4int nDivs,
314  const G4double width,
315  const G4double half_gap,
316  const G4double offset,
317  DivisionType divType )
318 {
319  G4VSolid* mSolid = motherLogical->GetSolid();
320  G4String mSolidType = mSolid->GetEntityType();
321  fparam = 0;
322 
323  // If the solid is a reflected one, update type to its
324  // real constituent solid.
325  //
326  if (mSolidType == "G4ReflectedSolid")
327  {
328  mSolidType = ((G4ReflectedSolid*)mSolid)->GetConstituentMovedSolid()
329  ->GetEntityType();
330  }
331 
332  // Parameterisation type depend of mother solid type and axis of division
333  //
334  if( mSolidType == "G4Box" )
335  {
336  switch( axis )
337  {
338  case kXAxis:
339  fparam = new G4ParameterisationBoxX( axis, nDivs, width,
340  offset, mSolid, divType );
341  break;
342  case kYAxis:
343  fparam = new G4ParameterisationBoxY( axis, nDivs, width,
344  offset, mSolid, divType );
345  break;
346  case kZAxis:
347  fparam = new G4ParameterisationBoxZ( axis, nDivs, width,
348  offset, mSolid, divType );
349  break;
350  default:
351  ErrorInAxis( axis, mSolid );
352  break;
353  }
354  }
355  else if( mSolidType == "G4Tubs" )
356  {
357  switch( axis )
358  {
359  case kRho:
360  fparam = new G4ParameterisationTubsRho( axis, nDivs, width,
361  offset, mSolid, divType );
362  break;
363  case kPhi:
