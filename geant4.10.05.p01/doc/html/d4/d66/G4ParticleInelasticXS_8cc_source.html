156 G4double G4ParticleInelasticXS::GetElementCrossSection(
157  const G4DynamicParticle* aParticle,
158  G4int ZZ, const G4Material*)
159 {
160  G4double xs = 0.0;
161  G4double ekin = aParticle->GetKineticEnergy();
162 
163  G4int Z = (ZZ >= MAXZINELP) ? MAXZINELP - 1 : ZZ;
164 
165  G4PhysicsVector* pv = data->GetElementData(Z);
166  // G4cout << "G4ParticleInelasticXS::GetCrossSection e= " << ekin
167  // << " Z= " << Z << G4endl;
168 
169  // element was not initialised or below threshold
170  if(!pv || ekin <= pv->Energy(0)) { return xs; }
171 
172  if(ekin <= pv->GetMaxEnergy()) {
173  xs = pv->Value(ekin);
174  } else if(1 == Z) {
175  if(fNucleon) {
176  fNucleon->GetHadronNucleonXscNS(aParticle, proton);
177  xs = coeff[1]*fNucleon->GetInelasticHadronNucleonXsc();
178  } else {
179  nnXsection->GetZandACrossSection(aParticle, 1, 1);
180  xs = coeff[1]*nnXsection->GetInelasticGlauberGribovXsc();
181  }
182  } else {
183  G4int Amean = G4lrint(fNist->GetAtomicMassAmu(Z));
184  if(ggXsection) {
185  ggXsection->GetIsoCrossSection(aParticle, Z, Amean);
186  xs = coeff[Z]*ggXsection->GetInelasticGlauberGribovXsc();
187  } else {
188  nnXsection->GetZandACrossSection(aParticle, Z, Amean);
189  xs = coeff[Z]*nnXsection->GetInelasticGlauberGribovXsc();
190  }
191  }
192 
193  if(verboseLevel > 1) {
194  G4cout << "ElmXS: Z= " << Z << " Ekin(MeV)= " << ekin/CLHEP::MeV
195  << " xs(bn)= " << xs/CLHEP::barn << " element data for "
196  << particleName << G4endl;
197  }
198  return xs;
199 }

210 G4double
211 G4ParticleInelasticXS::IsoCrossSection(G4double ekin, G4int ZZ, G4int A)
212 {
213  G4double xs = 0.0;
214  G4int Z = (ZZ >= MAXZINELP) ? MAXZINELP - 1 : ZZ;
215  /*
216  G4cout << "IsoCrossSection Z= " << Z << " A= " << A
217  << " Amin= " << amin[Z] << " Amax= " << amax[Z]
218  << " E(MeV)= " << ekin << G4endl;
219  */
220  // first compute isotope cross section
221  if(ekin <=emax && amin[Z]>0 && A >= amin[Z] && A <= amax[Z]) {
222  G4PhysicsVector* pviso = data->GetComponentDataByIndex(Z, A - amin[Z]);
223  if(pviso) {
224  xs = pviso->Value(ekin);
225  if(verboseLevel > 0){
226  G4cout << "IsoXS for " << particleName
227  << " Target Z= " << Z << " A= " << A
228  << " Ekin(MeV)= " << ekin/CLHEP::MeV
229  << " xs(bn)= " << xs/CLHEP::barn << G4endl;
230  }
231  return xs;
232  }
233  }
234  // isotope data are not available or applicable
235  G4PhysicsVector* pv = data->GetElementData(Z);
236  if(pv) { xs = pv->Value(ekin); }
237  if(verboseLevel > 0) {
238  G4cout << "IsoXS for " << particleName
239  << " Target Z= " << Z << " A= " << A
240  << " Ekin(MeV)= " << ekin/CLHEP::MeV
241  << " xs(bn)= " << xs/CLHEP::barn << G4endl;
242  }
243  return xs;
244 }

246 const G4Isotope* G4ParticleInelasticXS::SelectIsotope(
247  const G4Element* anElement, G4double kinEnergy)
248 {
249  size_t nIso = anElement->GetNumberOfIsotopes();
250  const G4Isotope* iso = anElement->GetIsotope(0);
251 
252  //G4cout << "SelectIsotope NIso= " << nIso << G4endl;
253  if(1 == nIso) { return iso; }
254 
255  // more than 1 isotope
256  G4int Z = anElement->GetZasInt();
257  //G4cout << "SelectIsotope Z= " << Z << G4endl;
258 
259  const G4double* abundVector = anElement->GetRelativeAbundanceVector();
260  G4double q = G4UniformRand();
261  G4double sum = 0.0;
262 
263  // is there isotope wise cross section?
264  size_t j;
265  if(kinEnergy > emax || 0 == amin[Z] || Z >= MAXZINELP) {
266  for (j = 0; j<nIso; ++j) {
267  sum += abundVector[j];
268  if(q <= sum) {
269  iso = anElement->GetIsotope(j);
270  break;
271  }
272  }
273  return iso;
274  }
275 
276  size_t nn = temp.size();
277  if(nn < nIso) { temp.resize(nIso, 0.); }
278 
279  for (j=0; j<nIso; ++j) {
280  //G4cout << j << "-th isotope " << (*isoVector)[j]->GetN()
281  // << " abund= " << abundVector[j] << G4endl;
282  sum += abundVector[j]*IsoCrossSection(kinEnergy, Z,
283  anElement->GetIsotope(j)->GetN());
284  temp[j] = sum;
285  }
286  sum *= q;
287  for (j = 0; j<nIso; ++j) {
288  if(temp[j] >= sum) {
289  iso = anElement->GetIsotope(j);
290  break;
291  }
292  }
293  return iso;
294 }

297 G4ParticleInelasticXS::BuildPhysicsTable(const G4ParticleDefinition& p)
298 {
299  if(verboseLevel > 0){
300  G4cout << "G4ParticleInelasticXS::BuildPhysicsTable for "
301  << p.GetParticleName() << G4endl;
302  }
303  if(&p != particle) {
304  G4ExceptionDescription ed;
305  ed << p.GetParticleName() << " is a wrong particle type -"
306  << particleName << " is expected";
307  G4Exception("G4ParticleInelasticXS::BuildPhysicsTable(..)","had012",
308  FatalException, ed, "");
309  return;
310  }
311 
312  if(!data) {
313 #ifdef G4MULTITHREADED
314  G4MUTEXLOCK(&particleInelasticXSMutex);
315  if(!data) {
316 #endif
317  isMaster = true;
318  data = new G4ElementData();
319  data->SetName(particleName + "Inelastic");
320  temp.resize(13,0.0);
321 #ifdef G4MULTITHREADED
322  }
323  G4MUTEXUNLOCK(&particleInelasticXSMutex);
324 #endif
325  }
326 
327  // it is possible re-initialisation for the new run
328  if(isMaster) {
329 
330  // check environment variable
331  // Build the complete string identifying the file with the data set
332  char* path = getenv("G4PARTICLEXSDATA");
333 
334  G4DynamicParticle* dynParticle =
335  new G4DynamicParticle(particle,G4ThreeVector(1,0,0),1);
336 
337  // Access to elements
338  const G4ElementTable* theElmTable = G4Element::GetElementTable();
339  size_t numOfElm = G4Element::GetNumberOfElements();
340  for(size_t i=0; i<numOfElm; ++i) {
341  G4int Z = ((*theElmTable)[i])->GetZasInt();
342  if(Z >= MAXZINELP) { Z = MAXZINELP-1; }
343  //G4cout << "Z= " << Z << G4endl;
344  // Initialisation
345  if(!(data->GetElementData(Z))) {
346  Initialise(Z, dynParticle, path);
347  }
348  }
349  delete dynParticle;
350  }

353 void G4ParticleInelasticXS::Initialise(G4int Z, G4DynamicParticle* dp,
354  const char* p)
355 {
356  if(data->GetElementData(Z)) { return; }
357  const char* path = p;
358  if(!p) {
359  // check environment variable
360  // Build the complete string identifying the file with the data set
361  path = getenv("G4PARTICLEXSDATA");
362  if (!path) {
363  G4Exception("G4ParticleInelasticXS::Initialise(..)","had013",
364  FatalException,
365  "Environment variable G4PARTICLEXSDATA is not defined");
366  return;
367  }
368  }
369 
370  // upload element data
371  std::ostringstream ost;
372  ost << path << "/" << particleName << "/inel" << Z ;
373  G4PhysicsVector* v = RetrieveVector(ost, true);
374  data->InitialiseForElement(Z, v);
375  /*
376  G4cout << "G4ParticleInelasticXS::Initialise for Z= " << Z
377  << " A= " << Amean << " Amin= " << amin[Z]
378  << " Amax= " << amax[Z] << G4endl;
379  */
380  // upload isotope data
381  if(amin[Z] > 0) {
382  size_t nmax = (size_t)(amax[Z]-amin[Z]+1);
383  data->InitialiseForComponent(Z, nmax);
384 
385  for(G4int A=amin[Z]; A<=amax[Z]; ++A) {
386  std::ostringstream ost1;
387  ost1 << path << "/" << particleName << "/inel" << Z << "_" << A;
388  G4PhysicsVector* v1 = RetrieveVector(ost1, false);
389  data->AddComponent(Z, A, v1);
390  }
391  }
392 
393  // smooth transition
394  G4double sig1 = (*v)[v->GetVectorLength()-1];
395  dp->SetKineticEnergy(v->GetMaxEnergy());
396  G4double sig2 = 0.0;
397  if(1 == Z) {
398  if(fNucleon) {
399  fNucleon->GetHadronNucleonXscNS(dp, proton);
400  sig2 = fNucleon->GetInelasticHadronNucleonXsc();
401  } else {
402  nnXsection->GetZandACrossSection(dp, 1, 1);
403  sig2 = nnXsection->GetInelasticGlauberGribovXsc();
404  }
405  } else {
406  G4int Amean = G4lrint(fNist->GetAtomicMassAmu(Z));
407  if(ggXsection) {
408  ggXsection->GetIsoCrossSection(dp, Z, Amean);
409  sig2 = ggXsection->GetInelasticGlauberGribovXsc();
410  } else {
411  nnXsection->GetZandACrossSection(dp, Z, Amean);
412  sig2 = nnXsection->GetInelasticGlauberGribovXsc();
413  }
414  }
415  if(sig2 > 0.) { coeff[Z] = sig1/sig2; }
416 }
