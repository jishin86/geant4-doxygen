264 void PhysicsList::AddStepMax()
265 {
266  //Step limitation seen as a process
267 
268  auto particleIterator=GetParticleIterator();
269  particleIterator->reset();
270  while ((*particleIterator)())
271  {
272  G4ParticleDefinition* particle = particleIterator->value();
273  G4ProcessManager* pmanager = particle->GetProcessManager();
274 
275  if (fStepMaxProcess->IsApplicable(*particle))
276  {
277  pmanager ->AddDiscreteProcess(fStepMaxProcess);
278  }
279  }
280 }
