67 G4double G4WeightWindowStore::GetLowerWeight(const G4GeometryCell &gCell,
68  G4double partEnergy) const
69 {
70  SetInternalIterator(gCell);
71  G4GeometryCellWeight::const_iterator gCellIterator = fCurrentIterator;
72  if (gCellIterator == fCellToUpEnBoundLoWePairsMap.end()) {
73  Error("GetLowerWitgh() - Cell does not exist!");
74  return 0.;
75  }
76  G4UpperEnergyToLowerWeightMap upEnLoWeiPairs =
77  fCurrentIterator->second;
78  G4double lowerWeight = -1;
79  G4bool found = false;
80  for (G4UpperEnergyToLowerWeightMap::iterator it =
81  upEnLoWeiPairs.begin(); it != upEnLoWeiPairs.end(); it++) {
82  if (partEnergy < it->first) {
83  lowerWeight = it->second;
84  found = true;
85  break;
86  }
87  }
88  if (!found) {
89  std::ostringstream err_mess;
90  err_mess << "GetLowerWitgh() - Couldn't find lower weight bound." << G4endl
91  << "Energy: " << partEnergy << ".";
92  Error(err_mess.str());
93  }
94  return lowerWeight;
95

97 }

159 void G4WeightWindowStore::
160 AddLowerWeights(const G4GeometryCell & gCell,
161  const std::vector<G4double> &lowerWeights)
162 {
163  if (fGeneralUpperEnergyBounds.empty()) {
164  Error("AddLowerWeights() - No general upper energy limits set!");
165  }
166  if (IsKnown(gCell)) {
167  Error("AddLowerWeights() - Cell already in the store.");
168  }
169  if (lowerWeights.size() != fGeneralUpperEnergyBounds.size()) {
170  std::ostringstream err_mess;
171  err_mess << "AddLowerWeights() - Mismatch between "
172  << "number of lower weights (" << lowerWeights.size()
173  << ") and energy bounds (" << fGeneralUpperEnergyBounds.size()
174  << ")!";
175  Error(err_mess.str());
176  }
177  G4UpperEnergyToLowerWeightMap map;
178  G4int i = 0;
179  for (std::set<G4double, std::less<G4double> >::iterator it =
180  fGeneralUpperEnergyBounds.begin();
181  it != fGeneralUpperEnergyBounds.end();
182  it++) {
183  map[*it] = lowerWeights[i];
184  i++;
185  }
186  fCellToUpEnBoundLoWePairsMap[gCell] = map;
187 }
