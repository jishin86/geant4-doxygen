226 G4double G4InterpolationDriver<T>::BinsearchChord(State& y,
227  G4double hstart,
228  G4double hmaximum,
229  G4double chordDistance)
230 {
231  const G4double interval = fIntegrationInterval.second - fIntegrationInterval.first;
232  const G4double tauStart = (hstart - fIntegrationInterval.first) / interval;
233 
234  const G4ThreeVector start = field_utils::makeVector(y, field_utils::Value3D::Position);
235  auto calcChordDistance = [&](G4double hstep)
236  {
237  using namespace field_utils;
238 
239  const G4double deltaTau = hstep / interval;
240  const G4double tau = tauStart + deltaTau;
241  const G4double tauMid = tauStart + 0.5 * deltaTau;
242 
243  assert(tau > tauStart && tau <= 1);
244  //G4cout << "tau: " << tau << " ";
245 
246  Base::GetStepper()->Interpolate(tauMid, y);
247  const G4ThreeVector mid = makeVector(y, Value3D::Position);
248  //G4cout << "mid: " << mid << " ";
249 
250  Base::GetStepper()->Interpolate(tau, y);
251  const G4ThreeVector end = makeVector(y, Value3D::Position);
252  //G4cout << "end: " << end << G4endl;
253 
254  return G4LineSection::Distline(mid, start, end);
255  };
256 
257  if (calcChordDistance(hmaximum) < chordDistance)
258  {
259  fChordStepEstimate = std::max(hmaximum, fChordStepEstimate);
260  return hmaximum;
261  }
262 
263  G4double hmax = hmaximum;
264  G4double hmin = 0;
265  G4double hstep = fChordStepEstimate ?
266  std::min(fChordStepEstimate, hmax) : 0.5 * (hmax + hmin);
267 
268  G4double distance;
269 
270  for (G4int i = 1; i < 100; ++i)
271  {
272  distance = calcChordDistance(hstep);
273 
274  //G4cout << "i " << i << " hmin " << hmin << " hstep " << hstep << " hmax " << hmax <<" ";
275  //G4cout << "disntace " << distance << " chordDistance " << chordDistance << G4endl;
276 
277  if (distance <= chordDistance && distance > 0.9 * chordDistance)
278  {
279  fChordStepEstimate = hstep;
280  fMaxTrials = std::max(fMaxTrials, i);
281  return hstep;
282  }
283 
284  if (distance < chordDistance)
285  {
286  hmin = hstep;
287  } else // distance > chordDistance
288  {
289  hmax = hstep;
290  }
291 
292  hstep = 0.5 * (hmax + hmin);
293  //hstep = 0.5 * (hmax + hmin);//std::min(hstep * std::sqrt(chordDistance / distance), hmax);
294  }
295 
296  G4cout << "distance " << distance << " requested " << chordDistance << " "
297  << "step " << hstep << G4endl;
298 
299  G4Exception("G4InterpolationDriver::FindNextChord()",
300  "GeomField1001", FatalException, "cannot converge");
301 
302  return hstep;
303 }

305 // Runge-Kutta driver with adaptive stepsize control. Integrate starting
306 // values at y_current over hstep x2 with accuracy eps.
307 // On output ystart is replaced by values at the end of the integration
308 // interval. RightHandSide is the right-hand side of ODE system.
309 // The source is similar to odeint routine from NRC p.721-722 .
310 template <class T>
311 G4bool G4InterpolationDriver<T>::
312 AccurateAdvance(G4FieldTrack& track, G4double hstep,
313  G4double /*eps*/, G4double /*hinitial*/)
314 {
315  //G4cout << "AA hstep " << hstep << G4endl;
316  ++fNoAccurateAdvanceCalls;
317 
318  if (hstep == 0.0)
319  {
320  std::ostringstream message;
321  message << "Proposed step is zero; hstep = " << hstep << " !";
322  G4Exception("G4InterpolationDriver::AccurateAdvance()",
323  "GeomField1001", JustWarning, message);
324  return true;
325  }
326 
327  if (hstep < 0)
328  {
329  std::ostringstream message;
330  message << "Invalid run condition." << G4endl
331  << "Proposed step is negative; hstep = " << hstep << "." << G4endl
332  << "Requested step cannot be negative! Aborting event.";
333  G4Exception("G4InterpolationDriver::AccurateAdvance()",
334  "GeomField0003", EventMustBeAborted, message);
335  return false;
336  }
337 
338  const G4double curveLength = track.GetCurveLength();
339  const G4double curveLengthEnd = curveLength + hstep;
340  assert(curveLength >= fIntegrationInterval.first && curveLengthEnd <= fIntegrationInterval.second);
341 
342  State y;
343  const G4double tau = (curveLengthEnd - fIntegrationInterval.first) / (fIntegrationInterval.second - fIntegrationInterval.first);
344  Base::GetStepper()->Interpolate(tau, y);
345 
346  track.LoadFromArray(y, Base::GetStepper()->GetNumberOfVariables());
347  track.SetCurveLength(curveLengthEnd);
348 
349  return true;
350 }

352 // Driver for one Runge-Kutta Step with monitoring of local truncation error
353 // to ensure accuracy and adjust stepsize. Input are dependent variable
354 // array y[0,...,5] and its derivative dydx[0,...,5] at the
355 // starting value of the independent variable x . Also input are stepsize
356 // to be attempted htry, and the required accuracy eps. On output y and x
357 // are replaced by their new values, hdid is the stepsize that was actually
358 // accomplished, and hnext is the estimated next stepsize.
359 // This is similar to the function rkqs from the book:
360 // Numerical Recipes in C: The Art of Scientific Computing (NRC), Second
361 // Edition, by William H. Press, Saul A. Teukolsky, William T.
362 // Vetterling, and Brian P. Flannery (Cambridge University Press 1992),
363 // 16.2 Adaptive StepSize Control for Runge-Kutta, p. 719
364 template <class T>
365 void G4InterpolationDriver<T>::OneGoodStep(const G4double y[], // InOut
366  const G4double dydx[],
367  G4double htry,
368  G4double eps_rel_max,
369  G4double& hdid, // Out
370  G4double& hnext) // Out
371 
372 {
373  G4double error2 = DBL_MAX;
374 
375  G4double yerr[G4FieldTrack::ncompSVEC], ytemp[G4FieldTrack::ncompSVEC];
376 
377  G4double h = htry;
378  //G4cout << "htry: " << htry << G4endl;
379 
380  static G4ThreadLocal G4int tot_no_trials = 0;
381  const G4int max_trials = 100;
382 
383  for (G4int iter = 0; iter < max_trials; ++iter)
384  {
385  tot_no_trials++;
386 
387  Base::GetStepper()->Stepper(y, dydx, h, ytemp, yerr);
388  error2 = field_utils::relativeError2(y, yerr, std::max(h, fMinimumStep), eps_rel_max);
389 
390  if (error2 <= 1.0)
391  {
392  break;
393  }
394 
395  h = Base::ShrinkStepSize2(h, error2);
396  }
397 
398  hnext = Base::GrowStepSize2(h, error2);
399  hdid = h;
400  //G4cout << "hdid: " << hdid << G4endl;
401 }

403 template <class T>
404 void G4InterpolationDriver<T>::
405 CheckStep( const G4ThreeVector& posIn,
406  const G4ThreeVector& posOut, G4double hdid)
407 {
408  const G4double endPointDist = (posOut - posIn).mag();
409  if (endPointDist >= hdid * (1. + CLHEP::perMillion))
410  {
411  ++fNoAccurateAdvanceBadSteps;
412 #ifdef G4DEBUG_FIELD
413  // Issue a warning only for gross differences -
414  // we understand how small difference occur.
415  if (endPointDist >= hdid * (1. + perThousand))
416  {
417  G4Exception("G4InterpolationDriver::CheckStep()",
418  "GeomField1002", JustWarning,
419  "endPointDist >= hdid!");
420  }
421 #endif
422  }
423  else
424  {
425  ++fNoAccurateAdvanceGoodSteps;
426  }
427 }
