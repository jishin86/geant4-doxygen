233 // CheckOverlaps
234 //
235 G4bool G4PVPlacement::CheckOverlaps(G4int res, G4double tol,
236  G4bool verbose, G4int maxErr)
237 {
238  if (res<=0) { return false; }
239 
240  G4VSolid* solid = GetLogicalVolume()->GetSolid();
241  G4LogicalVolume* motherLog = GetMotherLogical();
242  if (!motherLog) { return false; }
243 
244  G4int trials = 0;
245  G4bool retval = false;
246 
247  G4VSolid* motherSolid = motherLog->GetSolid();
248 
249  if (verbose)
250  {
251  G4cout << "Checking overlaps for volume " << GetName() << " ... ";
252  }
253 
254  // Create the transformation from daughter to mother
255  //
256  G4AffineTransform Tm( GetRotation(), GetTranslation() );
257 
258  for (G4int n=0; n<res; n++)
259  {
260  // Generate a random point on the solid's surface
261  //
262  G4ThreeVector point = solid->GetPointOnSurface();
263 
264  // Transform the generated point to the mother's coordinate system
265  //
266  G4ThreeVector mp = Tm.TransformPoint(point);
267 
268  // Checking overlaps with the mother volume
269  //
270  if (motherSolid->Inside(mp)==kOutside)
271  {
272  G4double distin = motherSolid->DistanceToIn(mp);
273  if (distin > tol)
274  {
275  trials++; retval = true;
276  std::ostringstream message;
277  message << "Overlap with mother volume !" << G4endl
278  << " Overlap is detected for volume "
279  << GetName() << ':' << GetCopyNo() << G4endl
280  << " with its mother volume "
281  << motherLog->GetName() << G4endl
282  << " at mother local point " << mp << ", "
283  << "overlapping by at least: "
284  << G4BestUnit(distin, "Length");
285  if (trials>=maxErr)
286  {
287  message << G4endl
288  << "NOTE: Reached maximum fixed number -" << maxErr
289  << "- of overlaps reports for this volume !";
290  }
291  G4Exception("G4PVPlacement::CheckOverlaps()",
292  "GeomVol1002", JustWarning, message);
293  if (trials>=maxErr) { return true; }
294  }
295  }
296 
297  // Checking overlaps with each 'sister' volume
298  //
299  for (G4int i=0; i<motherLog->GetNoDaughters(); i++)
300  {
301  G4VPhysicalVolume* daughter = motherLog->GetDaughter(i);
302 
303  if (daughter == this) { continue; }
304 
305  // Create the transformation for daughter volume and transform point
306  //
307  G4AffineTransform Td( daughter->GetRotation(),
308  daughter->GetTranslation() );
309  G4ThreeVector md = Td.InverseTransformPoint(mp);
310 
311  G4VSolid* daughterSolid = daughter->GetLogicalVolume()->GetSolid();
312  if (daughterSolid->Inside(md)==kInside)
313  {
314  G4double distout = daughterSolid->DistanceToOut(md);
315  if (distout > tol)
316  {
317  trials++; retval = true;
318  std::ostringstream message;
319  message << "Overlap with volume already placed !" << G4endl
320  << " Overlap is detected for volume "
321  << GetName() << ':' << GetCopyNo() << G4endl
322  << " with " << daughter->GetName()
323  << ':' << daughter->GetCopyNo() << " volume's"
324  << G4endl
325  << " local point " << md << ", "
326  << "overlapping by at least: "
327  << G4BestUnit(distout,"Length");
328  if (trials>=maxErr)
329  {
330  message << G4endl
331  << "NOTE: Reached maximum fixed number -" << maxErr
332  << "- of overlaps reports for this volume !";
333  }
334  G4Exception("G4PVPlacement::CheckOverlaps()",
335  "GeomVol1002", JustWarning, message);
336  if (trials>=maxErr) { return true; }
337  }
338  }
339 
340  // Now checking that 'sister' volume is not totally included and
341  // overlapping. Do it only once, for the first point generated
342  //
343  if (n==0)
344  {
345  // Generate a single point on the surface of the 'sister' volume
346  // and verify that the point is NOT inside the current volume
347 
348  G4ThreeVector dPoint = daughterSolid->GetPointOnSurface();
349 
350  // Transform the generated point to the mother's coordinate system
351  // and finally to current volume's coordinate system
352  //
353  G4ThreeVector mp2 = Td.TransformPoint(dPoint);
354  G4ThreeVector msi = Tm.InverseTransformPoint(mp2);
355 
356  if (solid->Inside(msi)==kInside)
357  {
358  trials++; retval = true;
359  std::ostringstream message;
360  message << "Overlap with volume already placed !" << G4endl
361  << " Overlap is detected for volume "
362  << GetName() << ':' << GetCopyNo() << G4endl
363  << " apparently fully encapsulating volume "
364  << daughter->GetName() << ':' << daughter->GetCopyNo()
365  << " at the same level !";
366  G4Exception("G4PVPlacement::CheckOverlaps()",
367  "GeomVol1002", JustWarning, message);
368  if (trials>=maxErr) { return true; }
369  }
370  }
371  }
372  }
373 
374  if (verbose)
375  {
376  G4cout << "OK! " << G4endl;
377  }
378 
379  return retval;
380 }
