98

100  G4double local[4], global[4];
101 
102  G4ThreeVector globalPosition;
103  local[3] = 0.0;
104  for (int i=0; i<2; ++i) {
105  local[0] = (i==0 ? -1.0 : 1.0) * GetWidth()/2.;
106  for (int j=0; j<2; ++j) {
107  local[1] = (j==0 ? -1.0 : 1.0) * GetHeight()/2.;
108  for (int k=0; k<2; ++k) {
109  local[2] = (k==0 ? -1.0 : 1.0) * GetLength()/2.;
110  G4ThreeVector localPosition(local[0],local[1],local[2]);
111  globalPosition =
112  fGlobal2local.Inverse().TransformPoint(localPosition);
113  global[0] = globalPosition.x();
114  global[1] = globalPosition.y();
115  global[2] = globalPosition.z();
