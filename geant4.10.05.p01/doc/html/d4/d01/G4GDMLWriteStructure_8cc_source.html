74 {

110 void G4GDMLWriteStructure::PhysvolWrite(xercesc::DOMElement* volumeElement,
111  const G4VPhysicalVolume* const physvol,
112  const G4Transform3D& T,
113  const G4String& ModuleName)
114 {
115  HepGeom::Scale3D scale;
116  HepGeom::Rotate3D rotate;
117  HepGeom::Translate3D translate;
118 
119  T.getDecomposition(scale,rotate,translate);
120 
121  const G4ThreeVector scl(scale(0,0),scale(1,1),scale(2,2));
122  const G4ThreeVector rot = GetAngles(rotate.getRotation());
123  const G4ThreeVector pos = T.getTranslation();
124 
125  const G4String name = GenerateName(physvol->GetName(),physvol);
126  const G4int copynumber = physvol->GetCopyNo();
127 
128  xercesc::DOMElement* physvolElement = NewElement("physvol");
129  physvolElement->setAttributeNode(NewAttribute("name",name));
130  if (copynumber) physvolElement->setAttributeNode(NewAttribute("copynumber", copynumber));
131 
132  volumeElement->appendChild(physvolElement);
133 
134  G4LogicalVolume* lv;
135  // Is it reflected?
136  if (reflFactory->IsReflected(physvol->GetLogicalVolume()))
137  {
138  lv = reflFactory->GetConstituentLV(physvol->GetLogicalVolume());
139  }
140  else
141  {
142  lv = physvol->GetLogicalVolume();
143  }
144 
145  const G4String volumeref = GenerateName(lv->GetName(), lv);
146 
147  if (ModuleName.empty())
148  {
149  xercesc::DOMElement* volumerefElement = NewElement("volumeref");
150  volumerefElement->setAttributeNode(NewAttribute("ref",volumeref));
151  physvolElement->appendChild(volumerefElement);
152  }
153  else
154  {
155  xercesc::DOMElement* fileElement = NewElement("file");
156  fileElement->setAttributeNode(NewAttribute("name",ModuleName));
157  fileElement->setAttributeNode(NewAttribute("volname",volumeref));
158  physvolElement->appendChild(fileElement);
159  }
160 
161  if (std::fabs(pos.x()) > kLinearPrecision
162  || std::fabs(pos.y()) > kLinearPrecision
163  || std::fabs(pos.z()) > kLinearPrecision)
164  {
165  PositionWrite(physvolElement,name+"_pos",pos);
166  }
167  if (std::fabs(rot.x()) > kAngularPrecision
168  || std::fabs(rot.y()) > kAngularPrecision
169  || std::fabs(rot.z()) > kAngularPrecision)
170  {
171  RotationWrite(physvolElement,name+"_rot",rot);
172  }
173  if (std::fabs(scl.x()-1.0) > kRelativePrecision
174  || std::fabs(scl.y()-1.0) > kRelativePrecision
175  || std::fabs(scl.z()-1.0) > kRelativePrecision)
176  {
177  ScaleWrite(physvolElement,name+"_scl",scl);
178  }
179 }

390 G4Transform3D G4GDMLWriteStructure::
391 TraverseVolumeTree(const G4LogicalVolume* const volumePtr, const G4int depth)
392 {
393  if (VolumeMap().find(volumePtr) != VolumeMap().end())
394  {
395  return VolumeMap()[volumePtr]; // Volume is already processed
396  }
397 
398  G4VSolid* solidPtr = volumePtr->GetSolid();
399  G4Transform3D R,invR;
400  G4int trans=0;
401 
402  std::map<const G4LogicalVolume*, G4GDMLAuxListType>::iterator auxiter;
403 
404  levelNo++;
405 
406  while (true) // Solve possible displacement/reflection
407  { // of the referenced solid!
408  if (trans>maxTransforms)
409  {
410  G4String ErrorMessage = "Referenced solid in volume '"
411  + volumePtr->GetName()
412  + "' was displaced/reflected too many times!";
413  G4Exception("G4GDMLWriteStructure::TraverseVolumeTree()",
414  "InvalidSetup", FatalException, ErrorMessage);
415  }
416 
417  if (G4ReflectedSolid* refl = dynamic_cast<G4ReflectedSolid*>(solidPtr))
418  {
419  R = R*refl->GetTransform3D();
420  solidPtr = refl->GetConstituentMovedSolid();
421  trans++;
422  continue;
423  }
424 
425  if (G4DisplacedSolid* disp = dynamic_cast<G4DisplacedSolid*>(solidPtr))
426  {
427  R = R*G4Transform3D(disp->GetObjectRotation(),
428  disp->GetObjectTranslation());
429  solidPtr = disp->GetConstituentMovedSolid();
430  trans++;
431  continue;
432  }
433 
434  break;
435  }
436 
437  //check if it is a reflected volume
438  G4LogicalVolume* tmplv = const_cast<G4LogicalVolume*>(volumePtr);
439 
440  if (reflFactory->IsReflected(tmplv))
441  {
442  tmplv = reflFactory->GetConstituentLV(tmplv);
443  if (VolumeMap().find(tmplv) != VolumeMap().end())
444  {
445  return R; // Volume is already processed
446  }
447  }
448 
449  // Only compute the inverse when necessary!
450  //
451  if (trans>0) { invR = R.inverse(); }
452 
453  const G4String name
454  = GenerateName(tmplv->GetName(), tmplv);
455 
456  G4String materialref = "NULL";
457 
458  if(volumePtr->GetMaterial())
459  {
460  materialref = GenerateName(volumePtr->GetMaterial()->GetName(),
461  volumePtr->GetMaterial());
462  }
463 
464  const G4String solidref
465  = GenerateName(solidPtr->GetName(),solidPtr);
466 
467  xercesc::DOMElement* volumeElement = NewElement("volume");
468  volumeElement->setAttributeNode(NewAttribute("name",name));
469  xercesc::DOMElement* materialrefElement = NewElement("materialref");
470  materialrefElement->setAttributeNode(NewAttribute("ref",materialref));
471  volumeElement->appendChild(materialrefElement);
472  xercesc::DOMElement* solidrefElement = NewElement("solidref");
473  solidrefElement->setAttributeNode(NewAttribute("ref",solidref));
474  volumeElement->appendChild(solidrefElement);
475 
476  G4int daughterCount = volumePtr->GetNoDaughters();
477 
478  if (levelNo == maxLevel) // Stop exporting if reached levels limit
479  {
480  daughterCount = 0;
481  }
482 
483  for (G4int i=0;i<daughterCount;i++) // Traverse all the children!
484  {
485  const G4VPhysicalVolume* const physvol = volumePtr->GetDaughter(i);
486  const G4String ModuleName = Modularize(physvol,depth);
487 
488  G4Transform3D daughterR;
489 
490  if (ModuleName.empty()) // Check if subtree requested to be
491  { // a separate module!
492  daughterR = TraverseVolumeTree(physvol->GetLogicalVolume(),depth+1);
493  }
494  else
495  {
496  G4GDMLWriteStructure writer;
497  daughterR = writer.Write(ModuleName,physvol->GetLogicalVolume(),
498  SchemaLocation,depth+1);
499  }
500 
501  if (const G4PVDivision* const divisionvol
502  = dynamic_cast<const G4PVDivision*>(physvol)) // Is it division?
503  {
504  if (!G4Transform3D::Identity.isNear(invR*daughterR,kRelativePrecision))
505  {
506  G4String ErrorMessage = "Division volume in '" + name
507  + "' can not be related to reflected solid!";
508  G4Exception("G4GDMLWriteStructure::TraverseVolumeTree()",
509  "InvalidSetup", FatalException, ErrorMessage);
510  }
511  DivisionvolWrite(volumeElement,divisionvol);
512  } else
513  if (physvol->IsParameterised()) // Is it a paramvol?
514  {
515  if (!G4Transform3D::Identity.isNear(invR*daughterR,kRelativePrecision))
516  {
517  G4String ErrorMessage = "Parameterised volume in '" + name
518  + "' can not be related to reflected solid!";
519  G4Exception("G4GDMLWriteStructure::TraverseVolumeTree()",
520  "InvalidSetup", FatalException, ErrorMessage);
521  }
522  ParamvolWrite(volumeElement,physvol);
523  } else
524  if (physvol->IsReplicated()) // Is it a replicavol?
525  {
526  if (!G4Transform3D::Identity.isNear(invR*daughterR,kRelativePrecision))
527  {
528  G4String ErrorMessage = "Replica volume in '" + name
529  + "' can not be related to reflected solid!";
530  G4Exception("G4GDMLWriteStructure::TraverseVolumeTree()",
531  "InvalidSetup", FatalException, ErrorMessage);
532  }
533  ReplicavolWrite(volumeElement,physvol);
534  }
535  else // Is it a physvol?
536  {
537  G4RotationMatrix rot;
538  if (physvol->GetFrameRotation() != 0)
539  {
540  rot = *(physvol->GetFrameRotation());
541  }
542  G4Transform3D P(rot,physvol->GetObjectTranslation());
543 
544  PhysvolWrite(volumeElement,physvol,invR*P*daughterR,ModuleName);
545  }
546  // BorderSurfaceCache(GetBorderSurface(physvol));
547  GetBorderSurface(physvol);
548  }
549 
550  if (cexport) { ExportEnergyCuts(volumePtr); }
551  // Add optional energy cuts
552 
553  if (sdexport) { ExportSD(volumePtr); }
554  // Add optional SDs
555 
556  // Here write the auxiliary info
557  //
558  auxiter = auxmap.find(volumePtr);
559  if (auxiter != auxmap.end())
560  {
561  AddAuxInfo(&(auxiter->second), volumeElement);
562  }
563 
564  structureElement->appendChild(volumeElement);
565  // Append the volume AFTER traversing the children so that
566  // the order of volumes will be correct!
567 
568  VolumeMap()[tmplv] = R;
569 
570  AddExtension(volumeElement, volumePtr);
571  // Add any possible user defined extension attached to a volume
572 
573  AddMaterial(volumePtr->GetMaterial());
574  // Add the involved materials and solids!
575 
576  AddSolid(solidPtr);
577 
578  SkinSurfaceCache(GetSkinSurface(volumePtr));
579 
580  return R;
581 }

601 void
602 G4GDMLWriteStructure::ExportEnergyCuts(const G4LogicalVolume* const lvol)
603 {
604  G4GDMLEvaluator eval;
605  G4ProductionCuts* pcuts = lvol->GetRegion()->GetProductionCuts();
606  G4ProductionCutsTable* ctab = G4ProductionCutsTable::GetProductionCutsTable();
607  G4Gamma* gamma = G4Gamma::Gamma();
608  G4Electron* eminus = G4Electron::Electron();
609  G4Positron* eplus = G4Positron::Positron();
610  G4Proton* proton = G4Proton::Proton();
611 
612  G4double gamma_cut = ctab->ConvertRangeToEnergy(gamma, lvol->GetMaterial(),
613  pcuts->GetProductionCut("gamma"));
614  G4double eminus_cut = ctab->ConvertRangeToEnergy(eminus, lvol->GetMaterial(),
615  pcuts->GetProductionCut("e-"));
616  G4double eplus_cut = ctab->ConvertRangeToEnergy(eplus, lvol->GetMaterial(),
617  pcuts->GetProductionCut("e+"));
618  G4double proton_cut = ctab->ConvertRangeToEnergy(proton, lvol->GetMaterial(),
619  pcuts->GetProductionCut("proton"));
620 
621  G4GDMLAuxStructType gammainfo = {"gammaECut",
622  eval.ConvertToString(gamma_cut), "MeV", 0};
623  G4GDMLAuxStructType eminusinfo = {"electronECut",
624  eval.ConvertToString(eminus_cut), "MeV", 0};
625  G4GDMLAuxStructType eplusinfo = {"positronECut",
626  eval.ConvertToString(eplus_cut), "MeV", 0};
627  G4GDMLAuxStructType protinfo = {"protonECut",
628  eval.ConvertToString(proton_cut), "MeV", 0};
629 
630  AddVolumeAuxiliary(gammainfo, lvol);
