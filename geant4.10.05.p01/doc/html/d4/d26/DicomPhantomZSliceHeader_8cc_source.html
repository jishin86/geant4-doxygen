78 {
79  //----- Read material indices and names

83  fin >> nmate;
84 #ifdef G4VERBOSE
85  G4cout << " DicomPhantomZSliceHeader reading number of materials "
86  << nmate << G4endl;
87 #endif
88 
89  for( G4int im = 0; im < nmate; im++ ){
90  fin >> mateindex >> matename;
91 #ifdef G4VERBOSE
92  //G4cout << " DicomPhantomZSliceHeader reading material "
93  // << im << " : "<< mateindex << " " << matename << G4endl;
94 #endif
95 
96  if( ! CheckMaterialExists( matename ) ) {
97  G4Exception("DicomPhantomZSliceHeader::DicomPhantomZSliceHeader",
98  "A material is found in file that is not built in the C++ code",
99  FatalErrorInArgument, matename.c_str());
100  }
101 
102  fMaterialNames.push_back(matename);
103  }
104 
105  //----- Read number of voxels
106  fin >> fNoVoxelX >> fNoVoxelY >> fNoVoxelZ;
107 #ifdef G4VERBOSE
108  G4cout << " Number of voxels " << fNoVoxelX << " " << fNoVoxelY
109  << " " << fNoVoxelZ << G4endl;
110 #endif
111 
112  //----- Read minimal and maximal extensions (= walls of phantom)
113  fin >> fMinX >> fMaxX;
114  fin >> fMinY >> fMaxY;
115  fin >> fMinZ >> fMaxZ;
116 #ifdef G4VERBOSE
117  G4cout << " Extension in X " << fMinX << " " << fMaxX << G4endl

119  << " Extension in Z " << fMinZ << " " << fMaxZ << G4endl;
120 #endif
121 
122  fSliceLocation = 0.5*(fMinZ + fMaxZ);

150 DicomPhantomZSliceHeader DicomPhantomZSliceHeader::operator+(
151  const DicomPhantomZSliceHeader& rhs )
152 {
153  //----- Check that both slices has the same dimensions
154  if( fNoVoxelX != rhs.GetNoVoxelX()
155  || fNoVoxelY != rhs.GetNoVoxelY() ) {
156  G4cerr << "DicomPhantomZSliceHeader error adding two slice headers:\
157  !!! Different number of voxels: "
158  << " X= " << fNoVoxelX << " =? " << rhs.GetNoVoxelX()
159  << " Y= " << fNoVoxelY << " =? " << rhs.GetNoVoxelY()
160  << " Z= " << fNoVoxelZ << " =? " << rhs.GetNoVoxelZ()
161  << G4endl;
162  G4Exception("DicomPhantomZSliceHeader::DicomPhantomZSliceHeader",
163  "",FatalErrorInArgument,"");
164  }
165  //----- Check that both slices has the same extensions
166  if( fMinX != rhs.GetMinX() || fMaxX != rhs.GetMaxX()
167  || fMinY != rhs.GetMinY() || fMaxY != rhs.GetMaxY() ) {
168  G4cerr << "DicomPhantomZSliceHeader error adding two slice headers:\
169  !!! Different extensions: "
170  << " Xmin= " << fMinX << " =? " << rhs.GetMinX()
171  << " Xmax= " << fMaxX << " =? " << rhs.GetMaxX()
172  << " Ymin= " << fMinY << " =? " << rhs.GetMinY()
173  << " Ymax= " << fMaxY << " =? " << rhs.GetMaxY()
174  << G4endl;
175  G4Exception("DicomPhantomZSliceHeader::operator+","",
176  FatalErrorInArgument,"");
177  }
178 
179  //----- Check that both slices have the same materials
180  std::vector<G4String> fMaterialNames2 = rhs.GetMaterialNames();
181  if( fMaterialNames.size() != fMaterialNames2.size() ) {
182  G4cerr << "DicomPhantomZSliceHeader error adding two slice headers:\
183  !!! Different number of materials: " << fMaterialNames.size() << " =? "
184  << fMaterialNames2.size() << G4endl;
185  G4Exception("DicomPhantomZSliceHeader::operator+","",
186  FatalErrorInArgument,"");
187  }
188  for( unsigned int ii = 0; ii < fMaterialNames.size(); ii++ ) {
189  if( fMaterialNames[ii] != fMaterialNames2[ii] ) {
190  G4cerr << "DicomPhantomZSliceHeader error adding two slice headers:\
191  !!! Different material number " << ii << " : "
192  << fMaterialNames[ii] << " =? "
193  << fMaterialNames2[ii] << G4endl;
194  G4Exception("DicomPhantomZSliceHeader::operator+","",
195  FatalErrorInArgument,"");
196  }
197  }
198 
199  //----- Check that the slices are contiguous in Z
200  if( std::fabs( fMinZ - rhs.GetMaxZ() ) >
201  G4GeometryTolerance::GetInstance()->GetRadialTolerance() &&
202  std::fabs( fMaxZ - rhs.GetMinZ() ) >
203  G4GeometryTolerance::GetInstance()->GetRadialTolerance() ){
204  G4cerr << "DicomPhantomZSliceHeader error adding two slice headers:!!!\
205  Slices are not contiguous in Z "
206  << " Zmin= " << fMinZ << " & " << rhs.GetMinZ()
207  << " Zmax= " << fMaxZ << " & " << rhs.GetMaxZ()
208  << G4endl;
209  G4Exception("DicomPhantomZSliceHeader::operator+","",
210  FatalErrorInArgument,"");
211  }
212 
213  //----- Build slice header copying first one
214  DicomPhantomZSliceHeader temp( *this );
215 
216  //----- Add data from second slice header
217  temp.SetMinZ( std::min( fMinZ, rhs.GetMinZ() ) );
218  temp.SetMaxZ( std::max( fMaxZ, rhs.GetMaxZ() ) );
219  temp.SetNoVoxelZ( fNoVoxelZ + rhs.GetNoVoxelZ() );
220 
221  return temp;
222 }

267 void DicomPhantomZSliceHeader::ReadDataFromFile()
268 {
269  std::ifstream in;
270  in.open(fFilename.c_str());
271 
272  if(!in) {
273  G4String descript = "DicomPhantomZSliceHeader::DumpToFile: could not open "
274  +fFilename;
275  G4Exception(descript.c_str(),"", FatalException, "");
276  }
277 
278  G4int nMaterials;
279  in >> nMaterials;
280 
281  fMaterialNames.resize(nMaterials,"");
282  for(G4int i = 0; i < nMaterials; ++i) {
283  G4String str1, str2;
284  in >> str1 >> str2;
285  if(!IsInteger(str1)) {
286  G4String descript = "String : " + str1 + " supposed to be integer";
287  G4Exception("DicomPhantomZSliceHeader::ReadDataFromFile - error in \
288  formatting: missing material index","", FatalException,descript.c_str());
289  }
290  G4int index = G4s2n<G4int>(str1);
291  if(index > nMaterials || index < 0) {
292  G4String descript = "Index : " + str1;
293  G4Exception("DicomPhantomZSliceHeader::ReadDataFromFile - error:\
294  bad material index","", FatalException,descript.c_str());
295  }
296  fMaterialNames[index] = str2;
297  }
298 
299  in >> fNoVoxelX >> fNoVoxelY >> fNoVoxelZ;
300 
301  G4double tmpMinX, tmpMinY, tmpMinZ;
302  G4double tmpMaxX, tmpMaxY, tmpMaxZ;
303 
304  in >> tmpMinX >> tmpMaxX;
305  in >> tmpMinY >> tmpMaxY;
306  in >> tmpMinZ >> tmpMaxZ;
307 
308  fMinX = (CheckConsistency(tmpMinX,fMinX,"Min X value")) ?
309  fMinX : ((fMinX == 0) ? tmpMinX : fMinX);
310  fMaxX = (CheckConsistency(tmpMaxX,fMaxX,"Max X value")) ?
311  fMaxX : ((fMaxX == 0) ? tmpMaxX : fMaxX);
312 
313  fMinY = (CheckConsistency(tmpMinY,fMinY,"Min Y value")) ?
314  fMinY : ((fMinY == 0) ? tmpMinY : fMinY);
315  fMaxY = (CheckConsistency(tmpMaxY,fMaxY,"Max Y value")) ?
316  fMaxY : ((fMaxY == 0) ? tmpMaxY : fMaxY);
317 
318  fMinZ = (CheckConsistency(tmpMinZ,fMinZ,"Min Z value")) ?
319  fMinZ : ((fMinZ == 0) ? tmpMinZ : fMinZ);
320  fMaxZ = (CheckConsistency(tmpMaxZ,fMaxZ,"Max Z value")) ?
321  fMaxZ : ((fMaxZ == 0) ? tmpMaxZ : fMaxZ);
322 
323  fMateIDs.clear();
324  fValues.clear();
325  fMateIDs.resize(fNoVoxelY*fNoVoxelZ,std::vector<G4int>(fNoVoxelX,0));
326  fValues.resize(fNoVoxelY*fNoVoxelZ,std::vector<G4double>(fNoVoxelX,0.));
327  for(G4int k = 0; k < fNoVoxelZ; ++k) {
328  for(G4int j = 0; j < fNoVoxelY; ++j) {
329  for(G4int i = 0; i < fNoVoxelX; ++i) {
330  G4int tmpMateID;
331  in >> tmpMateID;
332  G4int row = j*(k+1);
333  fMateIDs[row][i] = tmpMateID;
334  }
335  }
336  }
337 
338  for(G4int k = 0; k < fNoVoxelZ; ++k) {
339  for(G4int j = 0; j < fNoVoxelY; ++j) {
340  for(G4int i = 0; i < fNoVoxelX; ++i) {
341  G4double tmpValue;
342  in >> tmpValue;
343  G4int row = j*(k+1);
344  fValues[row][i] = tmpValue;
345  }
346  }
347  }
348 
349  in.close();
350 }
