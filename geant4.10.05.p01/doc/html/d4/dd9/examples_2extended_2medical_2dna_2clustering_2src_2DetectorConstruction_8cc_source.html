82 G4VPhysicalVolume* DetectorConstruction::ConstructVolumes()
83 {
84  // World
85  G4double worldSize = 2*micrometer;
86 
87  G4Box* solidWorld = new G4Box("World",
88  worldSize/2.,
89  worldSize/2.,
90  worldSize/2.);
91  G4LogicalVolume* logicWorld = new G4LogicalVolume(solidWorld,
92  fpDefaultMaterial,
93  "World");
94  G4VPhysicalVolume* physiWorld = new G4PVPlacement(0,
95  G4ThreeVector(),
96  "World",
97  logicWorld,
98  0,
99  false,
100  0);
101  // Target Box
102  // from Francis et al. Comput. Meth. Programs. Biomed. 2011 101(3)
103  G4double targetSizeXY=1*micrometer;
104  G4double targetSizeZ=0.5*micrometer;
105 
106  G4Box* solidTarget
107  = new G4Box("Target", targetSizeXY/2.,targetSizeXY/2.,targetSizeZ/2.);
108  G4LogicalVolume* logicTarget
109  = new G4LogicalVolume(solidTarget, fpWaterMaterial, "Target");
110  new G4PVPlacement(0,
111  G4ThreeVector(),
112  "Target",
113  logicTarget,
114  physiWorld,
115  false,
116  0);
117 
118  // Visualization attributes
119  G4VisAttributes* worldVisAtt= new G4VisAttributes(G4Colour(1.0,1.0,1.0));
120  worldVisAtt->SetVisibility(true);
121  logicWorld->SetVisAttributes(worldVisAtt);
122 
123  G4VisAttributes* targetVisAtt = new G4VisAttributes(G4Colour(0.0,1.0,0.0));
124  targetVisAtt->SetVisibility(true);
125  targetVisAtt->SetForceAuxEdgeVisible(true);
126  logicTarget->SetVisAttributes(targetVisAtt);
127 
128  //always return the world volume
129  return physiWorld;
130 }
