70 G4int RE02PSPassageCellFlux::GetIndex(G4Step* aStep)
71 {

73  G4int ix = touchable->GetReplicaNumber(1);

76 
77  G4int tmp = fNy;
78  if (tmp) return iy*fNx*fNz+ix*fNz+iz;
