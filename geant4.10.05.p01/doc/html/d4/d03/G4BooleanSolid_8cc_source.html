252 // Creates list of constituent primitives of and their placements
253 
254 void G4BooleanSolid::GetListOfPrimitives(
255  std::vector<std::pair<G4VSolid *,G4Transform3D>>& primitives,
256  const G4Transform3D& curPlacement) const
257 {
258  G4Transform3D transform;
259  G4VSolid* solid;
260  G4String type;
261 
262  // Repeat two times, first time for fPtrSolidA and then for fPtrSolidB
263  //
264  for (G4int i=0; i<2; i++)
265  {
266  transform = curPlacement;
267  solid = (i == 0) ? fPtrSolidA : fPtrSolidB;
268  type = solid->GetEntityType();
269 
270  // While current solid is a trasformed solid just modify transform
271  //
272  while (type == "G4DisplacedSolid" ||
273  type == "G4ReflectedSolid" ||
274  type == "G4ScaledSolid")
275  {
276  if (type == "G4DisplacedSolid")
277  {
278  transform = transform * G4Transform3D(
279  ((G4DisplacedSolid*)solid)->GetObjectRotation(),
280  ((G4DisplacedSolid*)solid)->GetObjectTranslation());
281  solid = ((G4DisplacedSolid*)solid)->GetConstituentMovedSolid();
282  }
283  else if (type == "G4ReflectedSolid")
284  {
285  transform= transform*((G4ReflectedSolid*)solid)->GetDirectTransform3D();
286  solid = ((G4ReflectedSolid*)solid)->GetConstituentMovedSolid();
287  }
288  else if (type == "G4ScaledSolid")
289  {
290  transform = transform * ((G4ScaledSolid*)solid)->GetScaleTransform();
291  solid = ((G4ScaledSolid*)solid)->GetUnscaledSolid();
292  }
293  type = solid->GetEntityType();
294  }
295 
296  // If current solid is a Boolean solid then continue recursion,
297  // otherwise add it to the list of primitives
298  //
299  if (type == "G4UnionSolid" ||
300  type == "G4SubtractionSolid" ||
301  type == "G4IntersectionSolid" ||
302  type == "G4BooleanSolid")
303  {
304  ((G4BooleanSolid *)solid)->GetListOfPrimitives(primitives,transform);
305  }
306  else
307  {
308  primitives.push_back(std::pair<G4VSolid*,G4Transform3D>(solid,transform));
309  }
310  }
311 }

315 // Returns a point (G4ThreeVector) randomly and uniformly selected
316 // on the surface of the solid
317 
318 G4ThreeVector G4BooleanSolid::GetPointOnSurface() const
319 {
320  G4int nprims = fPrimitives.size();
321  std::pair<G4VSolid *, G4Transform3D> prim;
322 
323  // Get list of primitives and find the total area of their surfaces
324  //
325  if (nprims == 0)
326  {
327  GetListOfPrimitives(fPrimitives, G4Transform3D());
328  nprims = fPrimitives.size();
329  fPrimitivesSurfaceArea = 0.;
330  for (G4int i=0; i<nprims; i++)
331  {
332  fPrimitivesSurfaceArea += fPrimitives[i].first->GetSurfaceArea();
333  }
334  }
335 
336  // Select random primitive, get random point on its surface and
337  // check that the point belongs to the surface of the solid
338  //
339  G4ThreeVector p;
340  for (G4int k=0; k<1000000; k++) // try 1000000 times
341  {
342  G4double rand = fPrimitivesSurfaceArea * G4UniformRand();
343  G4double area = 0.;
344  for (G4int i=0; i<nprims; i++)
345  {
346  prim = fPrimitives[i];
347  area += prim.first->GetSurfaceArea();
348  if (rand < area) break;
349  }
350  p = prim.first->GetPointOnSurface();
351  p = prim.second * G4Point3D(p);
352  if (Inside(p) == kSurface) return p;
353  }
354  std::ostringstream message;
355  message << "Solid - " << GetName() << "\n"
356  << "All attempts to generate a point on the surface have failed!\n"
357  << "The solid created may be an invalid Boolean construct!";
358  G4Exception("G4BooleanSolid::GetPointOnSurface()",
359  "GeomSolids1001", JustWarning, message);
360  return p;
361 }

385 // Stacks polyhedra for processing. Returns top polyhedron.
386 
387 G4Polyhedron*
388 G4BooleanSolid::StackPolyhedron(HepPolyhedronProcessor& processor,
389  const G4VSolid* solid) const
390 {
391  HepPolyhedronProcessor::Operation operation;
392  const G4String& type = solid->GetEntityType();
393  if (type == "G4UnionSolid")
394  { operation = HepPolyhedronProcessor::UNION; }
395  else if (type == "G4IntersectionSolid")
396  { operation = HepPolyhedronProcessor::INTERSECTION; }
397  else if (type == "G4SubtractionSolid")
398  { operation = HepPolyhedronProcessor::SUBTRACTION; }
399  else
400  {
401  std::ostringstream message;
402  message << "Solid - " << solid->GetName()
403  << " - Unrecognised composite solid" << G4endl
404  << " Returning NULL !";
405  G4Exception("StackPolyhedron()", "GeomSolids1001", JustWarning, message);
406  return 0;
407  }
408 
409  G4Polyhedron* top = 0;
410  const G4VSolid* solidA = solid->GetConstituentSolid(0);
411  const G4VSolid* solidB = solid->GetConstituentSolid(1);
412 
413  if (solidA->GetConstituentSolid(0))
414  {
415  top = StackPolyhedron(processor, solidA);
416  }
417  else
418  {
419  top = solidA->GetPolyhedron();
420  }
421  G4Polyhedron* operand = solidB->GetPolyhedron();
422  if (operand)
423  {
424  processor.push_back (operation, *operand);
425  }
426  else
427  {
428  std::ostringstream message;
429  message << "Solid - " << solid->GetName()
430  << " - No G4Polyhedron for Boolean component";
431  G4Exception("G4BooleanSolid::StackPolyhedron()",
432  "GeomSolids2001", JustWarning, message);
433  }
434 
435  return top;
436 }
