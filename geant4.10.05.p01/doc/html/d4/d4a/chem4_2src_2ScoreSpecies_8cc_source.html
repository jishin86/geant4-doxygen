88 G4bool ScoreSpecies::ProcessHits(G4Step* aStep,G4TouchableHistory*)
89 {
90  G4double edep = aStep->GetTotalEnergyDeposit();
91 
92  if ( edep == 0. ) return FALSE;
93 
94  edep *= aStep->GetPreStepPoint()->GetWeight(); // (Particle Weight)
95  G4int index = GetIndex(aStep);
96  fEvtMap->add(index,edep);
97  fEdep+=edep;
98 
99  return TRUE;
100 }

119 void ScoreSpecies::EndOfEvent(G4HCofThisEvent*)
120 {
121  if(G4EventManager::GetEventManager()->GetConstCurrentEvent()->IsAborted())
122  {
123  fEdep = 0.;
124  G4MoleculeCounter::Instance()->ResetCounter();
125  return;
126  }
127 
128  auto species = G4MoleculeCounter::Instance()->GetRecordedMolecules();
129 
130  if(species.get() == 0 || species->size() == 0)
131  {
132  G4cout << "No molecule recorded, energy deposited= "
133  << G4BestUnit(fEdep, "Energy") << G4endl;
134  ++fNEvent;
135  fEdep = 0.;
136  G4MoleculeCounter::Instance()->ResetCounter();
137  return;
138  }
139 
140  // G4cout << "ScoreSpecies::EndOfEvent"<<G4endl;
141 // G4cout << "End of event, deposited energy: "
142 // << G4BestUnit(fEdep, "Energy") << G4endl;
143 
144 #ifdef _ScoreSpecies_FOR_ALL_EVENTS
145  int eventID=
146  G4EventManager::GetEventManager()->GetConstCurrentEvent()->GetEventID();
147 #endif
148 
149  for(auto molecule: *species)
150  {
151  for(auto time_mol: fTimeToRecord)
152  {
153  double n_mol =
154  G4MoleculeCounter::Instance()->GetNMoleculesAtTime(molecule,
155  time_mol);
156 
157  if(n_mol < 0)
158  {
159  G4cerr << "N molecules not valid < 0 " << G4endl;
160  G4Exception("","N<0",FatalException,"");
161  }
162 
163  SpeciesInfo& molInfo =
164  fSpeciesInfoPerTime[time_mol][molecule];
165  molInfo.fNumber += n_mol;
166  double gValue = (n_mol/(fEdep/eV)) * 100.;
167  molInfo.fG += gValue;
168  molInfo.fG2 += gValue*gValue;
169 
170 #ifdef _ScoreSpecies_FOR_ALL_EVENTS
171  SpeciesInfoSOA& molInfoPerEvent =
172  fSpeciesInfoPerEvent[time_mol][molecule];
173  molInfoPerEvent.fNumber.push_back(n_mol);
174  molInfoPerEvent.fG.push_back(gValue);
175  molInfoPerEvent.fG2.push_back(gValue*gValue);
176  molInfoPerEvent.fEventID.push_back(eventID);
177 #endif
178  // G4cout << "In Save molucule: fNumber " << molInfo.fNumber
179  // << " fG " << molInfo.fG
180  // << " fEdep " << fEdep/eV
181  // << G4endl;
182  }
183  }
184 
185  ++fNEvent;
186 
187 // G4cout << "End of event " << fNEvent
188 // << ", energy deposited=" << G4BestUnit(fEdep, "Energy") << G4endl;
189 
190  fEdep = 0.;
191  G4MoleculeCounter::Instance()->ResetCounter();
192 }

196 void
197 ScoreSpecies::AbsorbResultsFromWorkerScorer(G4VPrimitiveScorer* workerScorer)
198 {
199  ScoreSpecies* right =
200  dynamic_cast<ScoreSpecies*>(dynamic_cast<G4VPrimitiveScorer*>(workerScorer));
201 
202  if(right == 0)
203  {
204  return;
205  }
206  if(right == this)
207  {
208  return;
209  }
210 
211  // G4cout<<"ScoreSpecies::AbsorbResultsFromWorkerScorer"<<G4endl;
212  {
213  SpeciesMap::iterator it_map1 = right->fSpeciesInfoPerTime.begin();
214  SpeciesMap::iterator end_map1 = right->fSpeciesInfoPerTime.end();
215 
216  for(; it_map1 != end_map1; ++it_map1)
217  {
218  InnerSpeciesMap& map2 = it_map1->second;
219  InnerSpeciesMap::iterator it_map2 = map2.begin();
220  InnerSpeciesMap::iterator end_map2 = map2.end();
221 
222  for(; it_map2 != end_map2; ++it_map2)
223  {
224  SpeciesInfo& molInfo =
225  fSpeciesInfoPerTime[it_map1->first][it_map2->first] ;
226  molInfo.fNumber += it_map2->second.fNumber;
227  molInfo.fG += it_map2->second.fG;
228  molInfo.fG2 += it_map2->second.fG2;
229 
230  // G4cout << "In AbsorbeResultsFromWorkerScorer: fNumber "
231  // << molInfo.fNumber
232  // << " fG "
233  // << molInfo.fG
234  // << G4endl;
235  }
236  }
237  }
238  //---------------------------------------------------------
239 #ifdef _ScoreSpecies_FOR_ALL_EVENTS
240  {
241  SpeciesMapPerEvent::iterator it_map1 = right->fSpeciesInfoPerEvent.begin();
242  SpeciesMapPerEvent::iterator end_map1 = right->fSpeciesInfoPerEvent.end();
243 
244  for(; it_map1 != end_map1; ++it_map1)
245  {
246  auto& map2 = it_map1->second;
247  InnerSpeciesMapPerEvent::iterator it_map2 = map2.begin();
248  InnerSpeciesMapPerEvent::iterator end_map2 = map2.end();
249 
250  for(; it_map2 != end_map2; ++it_map2)
251  {
252  SpeciesInfoSOA& molInfo =
253  fSpeciesInfoPerEvent[it_map1->first][it_map2->first] ;
254  molInfo.fNumber.insert(molInfo.fNumber.end(),
255  it_map2->second.fNumber.begin(),
256  it_map2->second.fNumber.end());
257  molInfo.fG.insert(molInfo.fG.end(),
258  it_map2->second.fG.begin(),
259  it_map2->second.fG.end());
260  molInfo.fG2.insert(molInfo.fG2.end(),
261  it_map2->second.fG2.begin(),
262  it_map2->second.fG2.end());
263  molInfo.fEventID.insert(molInfo.fEventID.end(),
264  it_map2->second.fEventID.begin(),
265  it_map2->second.fEventID.end());
266  // G4cout << "In AbsorbeResultsFromWorkerScorer: fNumber "
267  // << molInfo.fNumber
268  // << " fG "
269  // << molInfo.fG
270  // << G4endl;
271  }
272  }
273  right->fSpeciesInfoPerEvent.clear();
274  }
275 #endif
276 
277  fNEvent += right->fNEvent;
278  right->fNEvent = 0;
279  right->fEdep = 0.;
280 }
