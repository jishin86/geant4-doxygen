112 void HadrontherapyRunAction::EndOfRunAction(const G4Run*)
113 {
114  auto analysisManager = G4AnalysisManager::Instance();
115 
116  //G4cout << " Summary of Run " << aRun -> GetRunID() <<" :"<< G4endl;
117  //G4cout << "Number of electromagnetic processes of primary particles in the phantom:"
118  // << electromagnetic << G4endl;
119  //G4cout << "Number of hadronic processes of primary particles in the phantom:"
120  // << hadronic << G4endl;
121  G4AccumulableManager* accumulableManager = G4AccumulableManager::Instance();
122  accumulableManager->Merge();
123 
124  // Tell the RBE class what we have accumulated...
125  HadrontherapyRBE *rbe = HadrontherapyRBE::GetInstance();
126  if (rbe->IsCalculationEnabled() && IsMaster())
127  {
128  if (rbe->IsAccumulationEnabled())
129  {
130  rbe->AddAlphaNumerator(fRBEAccumulable.GetAlphaNumerator());
131  rbe->AddBetaNumerator(fRBEAccumulable.GetBetaNumerator());
132  rbe->AddDenominator(fRBEAccumulable.GetDenominator());
133  rbe->AddEnergyDeposit(fRBEAccumulable.GetEnergyDeposit());
134  }
135  else
136  {
137  rbe->SetAlphaNumerator(fRBEAccumulable.GetAlphaNumerator());
138  rbe->SetBetaNumerator(fRBEAccumulable.GetBetaNumerator());
139  rbe->SetDenominator(fRBEAccumulable.GetDenominator());
140  rbe->SetEnergyDeposit(fRBEAccumulable.GetEnergyDeposit());
141  }
142 
143  rbe->StoreAlphaAndBeta();
144  rbe->StoreRBE();
145  }
146 
147  analysisManager->Write();
148  analysisManager->CloseFile();
149 
150 }
