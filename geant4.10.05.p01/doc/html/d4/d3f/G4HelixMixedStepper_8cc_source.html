92 void G4HelixMixedStepper::Stepper( const G4double yInput[7],
93  const G4double dydx[7],
94  G4double Step,
95  G4double yOut[7],
96  G4double yErr[])
97 {
98  //Estimation of the Stepping Angle
99  G4ThreeVector Bfld;
100  MagFieldEvaluate(yInput, Bfld);
101 
102  G4double Bmag = Bfld.mag();
103  const G4double *pIn = yInput+3;
104  G4ThreeVector initVelocity= G4ThreeVector( pIn[0], pIn[1], pIn[2]);
105  G4double velocityVal = initVelocity.mag();
106 
107  const G4double R_1=std::abs(GetInverseCurve(velocityVal,Bmag)); // curv= inverse Radius
108  G4double Ang_curve= R_1 * Step;
109  // SetAngCurve(Ang_curve);
110  // SetCurve(std::abs(1/R_1)); // Move below, to avoid un-needed division if RK used
111 
112  if(Ang_curve< fAngle_threshold)
113  {
114  fNumCallsRK4++;
115  fRK4Stepper->Stepper(yInput,dydx,Step,yOut,yErr);
116  }
117  else
118  {
119  constexpr G4int nvar = 6 ;
120  constexpr G4int nvarMax = 8 ;
121  G4double yTemp[nvarMax], yIn[nvarMax], yTemp2[nvarMax];
122  G4ThreeVector Bfld_midpoint;
123 
124  SetAngCurve(Ang_curve);
125  SetCurve(std::abs(1.0/R_1));
126  fNumCallsHelix++;
127 
128  // Saving yInput because yInput and yOut can be aliases for same array
129  for(G4int i=0;i<nvar;i++) yIn[i]=yInput[i];
130 
131  G4double halfS = Step * 0.5;
132  // 1. Do first half step and full step
133  AdvanceHelix(yIn, Bfld, halfS, yTemp, yTemp2); // yTemp2 for s=2*h (halfS)
134  //**********
135 
136  MagFieldEvaluate(yTemp, Bfld_midpoint) ;
137 
138  // 2. Do second half step - with revised field
139  // NOTE: Could avoid this call if 'Bfld_midpoint == Bfld'
140  // or diff 'almost' zero
141  AdvanceHelix(yTemp, Bfld_midpoint, halfS, yOut);
142  // Not requesting y at s=2*h (halfS)
143  //**********
144 
145  // 3. Estimate the integration error
146  // should be (nearly) zero if Bfield= constant
147  for(G4int i=0;i<nvar;i++) {
148  yErr[i] = yOut[i] - yTemp2[i] ;
149  }
150  }
151 }

162 G4double G4HelixMixedStepper::DistChord() const
163 {
164  // Implementation : must check whether h/R > 2 pi !!
165  // If( h/R < pi) use G4LineSection::DistLine
166  // Else DistChord=R_helix
167  //
168  G4double distChord;
169  G4double Ang_curve=GetAngCurve();
170 
171  if(Ang_curve<=pi){
172  distChord=GetRadHelix()*(1-std::cos(0.5*Ang_curve));
173  }
174  else
175  {
176  if(Ang_curve<twopi){
177  distChord=GetRadHelix()*(1+std::cos(0.5*(twopi-Ang_curve)));
178  }
179  else{
180  distChord=2.*GetRadHelix();
181  }
182  }
183 
184  return distChord;
185 }

269  // ii) segmented uniform field (maybe)
