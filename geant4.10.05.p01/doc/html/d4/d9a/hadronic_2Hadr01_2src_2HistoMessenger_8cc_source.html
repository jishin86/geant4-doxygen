94 void HistoMessenger::SetNewValue(G4UIcommand* command, G4String newValues)
95 {
96  if (command == fFactoryCmd) { fHisto->SetFileName(newValues); }
97 
98  if (command == fHistoCmd) {
99  G4int ih,nbBins;
100  G4double vmin,vmax;
101  std::istringstream is(newValues);
102  G4String unts;
103  is >> ih >> nbBins >> vmin >> vmax >> unts;
104  G4String unit = unts;
105  G4double vUnit = 1. ;
106  if(unit != "none") { vUnit = G4UIcommand::ValueOf(unit); }
107  if(vUnit <= 0.0) { vUnit = 1.; }
108  fHisto->SetHisto1D(ih,nbBins,vmin,vmax,vUnit);
109  }
110 }
