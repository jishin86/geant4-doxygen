65 {
66  //stack or delete secondaries
67  G4ClassificationOfNewTrack status = fUrgent;
68 
69  //keep primary particle
70  if(aTrack->GetParentID() == 0) { return status; }
71 
72  const G4ParticleDefinition* part = aTrack->GetDefinition();
73  if(part == G4LDMPhoton::LDMPhoton() ||
74  part == G4LDMHi::LDMHi() ||
75  part == G4LDMHiBar::LDMHiBar()) {
76  G4cout << "### New exotic particle produced: "
77  << part->GetParticleName()
78  << " Ekin(GeV)= " << aTrack->GetKineticEnergy()/CLHEP::GeV
79  << " Mass(GeV)= " << part->GetPDGMass()/CLHEP::GeV
80  << " TrackId= " << aTrack->GetTrackID()
81  << G4endl;
82  return status;
83  }
84 
85  if(fKillSecondary) { status = fKill; }
86  return status;
