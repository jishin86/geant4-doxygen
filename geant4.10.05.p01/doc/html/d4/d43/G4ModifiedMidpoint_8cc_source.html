51 void G4ModifiedMidpoint::DoStep( const G4double yIn[], const G4double dydyIn[],
52  G4double yOut[], G4double hstep) const
53 {
54  G4double y0[G4FieldTrack::ncompSVEC];
55  G4double y1[G4FieldTrack::ncompSVEC];
56  G4double yTemp[G4FieldTrack::ncompSVEC];
57  setValue(yIn, Value1D::LabTime, y0, y1, yTemp, yOut);
58 
59  G4double dydx[G4FieldTrack::ncompSVEC];
60 
61  const G4double h = hstep / fsteps;
62  const G4double h2 = 2 * h;
63 
64  // y1 = yIn + h * dydx
65  for (G4int i = 0; i < fnvar; ++i)
66  {
67  y1[i] = yIn[i] + h * dydyIn[i];
68  }
69 
70  fEquation->RightHandSide(y1, dydx);
71 
72  copy(y0, yIn);
73 
74  // general step
75  // yTemp = y1; y1 = y0 + h2 * dydx; y0 = yTemp
76  for (G4int i = 1; i < fsteps; ++i)
77  {
78  copy(yTemp, y1);
79  for (G4int j = 0; j < fnvar; ++j)
80  {
81  y1[j] = y0[j] + h2 * dydx[j];
82  }
83  copy(y0, yTemp);
84 
85  fEquation->RightHandSide(y1, dydx);
86  }
87 
88  // last step
89  // yOut = 0.5 * (y0 + y1 + h * dydx)
90  for (G4int i = 0; i < fnvar; ++i)
91  {
92  yOut[i] = 0.5 * (y0[i] + y1[i] + h * dydx[i]);
93  }
94 }

96 void G4ModifiedMidpoint::DoStep( const G4double yIn[], const G4double dydxIn[],
97  G4double yOut[], G4double hstep, G4double yMid[],
98  G4double derivs[][G4FieldTrack::ncompSVEC]) const
99 {
100  G4double y0[G4FieldTrack::ncompSVEC];
101  G4double y1[G4FieldTrack::ncompSVEC];
102  G4double yTemp[G4FieldTrack::ncompSVEC];
103  setValue(yIn, Value1D::LabTime, y0, y1, yTemp, yMid, yOut);
104 
105  const G4double h = hstep / fsteps;
106  const G4double h2 = 2 * h;
107 
108  // y0 = yIn
109  copy(y0, yIn);
110 
111  // y1 = y0 + h * dydx
112  for (G4int i = 0; i < fnvar; ++i)
113  {
114  y1[i] = y0[i] + h * dydxIn[i];
115  }
116 
117  // result of first step already gives approximation
118  // at the center of the interval
119  if(fsteps == 2)
120  {
121  copy(yMid, y1);
122  }
123 
124  fEquation->RightHandSide(y1, derivs[0]);
125 
126  // general step
127  // yTemp = y1; y1 = y0 + h2 * dydx; y0 = yTemp
128  for (G4int i = 1; i < fsteps; ++i)
129  {
130  copy(yTemp, y1);
131  for (G4int j = 0; j < fnvar; ++j)
132  {
133  y1[j] = y0[j] + h2 * derivs[i-1][j];
134  }
135  copy(y0, yTemp);
136 
137  // save approximation at the center of the interval
138  if(i == fsteps / 2 - 1 )
139  {
140  copy(yMid, y1);
141  }
142 
143  fEquation->RightHandSide(y1, derivs[i]);
144  }
145 
146  // last step
147  // yOut = 0.5 * (y0 + y1 + h * dydx)
148  for (G4int i = 0; i < fnvar; ++i)
149  {
150  yOut[i] = 0.5 * (y0[i] + y1[i] + h * derivs[fsteps-1][i]);
151  }
152 }
