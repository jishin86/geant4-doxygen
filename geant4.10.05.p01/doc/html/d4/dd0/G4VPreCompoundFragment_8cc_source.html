78 void
79 G4VPreCompoundFragment::Initialize(const G4Fragment & aFragment)
80 {
81  theFragA = aFragment.GetA_asInt();
82  theFragZ = aFragment.GetZ_asInt();
83  theResA = theFragA - theA;
84  theResZ = theFragZ - theZ;
85 
86  theMinKinEnergy = theMaxKinEnergy = theCoulombBarrier = 0.0;
87  if ((theResA < theResZ) || (theResA < theA) || (theResZ < theZ)) {
88  return;
89  }
90 
91  theResA13 = g4calc->Z13(theResA);
92  theCoulombBarrier = theCoulombBarrierPtr->
93  GetCoulombBarrier(theResA,theResZ,aFragment.GetExcitationEnergy());
94 
95  theMinKinEnergy = (0 == OPTxs) ? theCoulombBarrier : theCoulombBarrier*0.7;
96 
97  // Calculate masses
98  theResMass = G4NucleiProperties::GetNuclearMass(theResA, theResZ);
99  theReducedMass = theResMass*theMass/(theResMass + theMass);
100 
101  // Compute Binding Energies for fragments
102  // needed to separate a fragment from the nucleus
103  theBindingEnergy = theResMass + theMass - aFragment.GetGroundStateMass();
104 
105  // Compute Maximal Kinetic Energy which can be carried by fragments
106  // after separation - the true assimptotic value
107  G4double Ecm = aFragment.GetMomentum().m();
108  theMaxKinEnergy = ((Ecm-theResMass)*(Ecm+theResMass) + theMass*theMass)
109  /(2.0*Ecm) - theMass;
110 }
