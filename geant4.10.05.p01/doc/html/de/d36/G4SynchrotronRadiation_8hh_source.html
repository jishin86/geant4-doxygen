111 inline G4double
112 G4SynchrotronRadiation::Chebyshev(G4double a, G4double b, const G4double c[],
113  G4int n, G4double x)
114 {
115  G4double y;
116  G4double y2=2.0*(y=(2.0*x-a-b)/(b-a)); // Change of variable.
117  G4double d=0.,dd=0.;
118  for (G4int j=n-1;j>=1;--j) // Clenshaw's recurrence.
119  { G4double sv=d;
120  d=y2*d-dd+c[j];
121  dd=sv;
122  }
123  return y*d-dd+0.5*c[0];
124 }
