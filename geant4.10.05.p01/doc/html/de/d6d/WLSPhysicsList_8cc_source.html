153 void WLSPhysicsList::ConstructProcess()
154 {
155  G4VModularPhysicsList::ConstructProcess();
156 
157  SetVerbose(0);
158 
159  G4DecayWithSpin* decayWithSpin = new G4DecayWithSpin();
160 
161  G4ProcessTable* processTable = G4ProcessTable::GetProcessTable();
162 
163  G4VProcess* decay;
164  decay = processTable->FindProcess("Decay",G4MuonPlus::MuonPlus());
165 
166  G4ProcessManager* pManager;
167  pManager = G4MuonPlus::MuonPlus()->GetProcessManager();
168 
169  if (pManager) {
170  if (decay) pManager->RemoveProcess(decay);
171  pManager->AddProcess(decayWithSpin);
172  // set ordering for PostStepDoIt and AtRestDoIt
173  pManager ->SetProcessOrdering(decayWithSpin, idxPostStep);
174  pManager ->SetProcessOrdering(decayWithSpin, idxAtRest);
175  }
176 
177  decay = processTable->FindProcess("Decay",G4MuonMinus::MuonMinus());
178 
179  pManager = G4MuonMinus::MuonMinus()->GetProcessManager();
180 
181  if (pManager) {
182  if (decay) pManager->RemoveProcess(decay);
183  pManager->AddProcess(decayWithSpin);
184  // set ordering for PostStepDoIt and AtRestDoIt
185  pManager ->SetProcessOrdering(decayWithSpin, idxPostStep);
186  pManager ->SetProcessOrdering(decayWithSpin, idxAtRest);
187  }
188 
189  G4PionDecayMakeSpin* poldecay = new G4PionDecayMakeSpin();
190 
191  decay = processTable->FindProcess("Decay",G4PionPlus::PionPlus());
192 
193  pManager = G4PionPlus::PionPlus()->GetProcessManager();
194 
195  if (pManager) {
196  if (decay) pManager->RemoveProcess(decay);
197  pManager->AddProcess(poldecay);
198  // set ordering for PostStepDoIt and AtRestDoIt
199  pManager ->SetProcessOrdering(poldecay, idxPostStep);
200  pManager ->SetProcessOrdering(poldecay, idxAtRest);
201  }
202 
203  decay = processTable->FindProcess("Decay",G4PionMinus::PionMinus());
204 
205  pManager = G4PionMinus::PionMinus()->GetProcessManager();
206 
207  if (pManager) {
208  if (decay) pManager->RemoveProcess(decay);
209  pManager->AddProcess(poldecay);
210  // set ordering for PostStepDoIt and AtRestDoIt
211  pManager ->SetProcessOrdering(poldecay, idxPostStep);
212  pManager ->SetProcessOrdering(poldecay, idxAtRest);
213  }
214 
215  AddStepMax();
216 
217 }

221 void WLSPhysicsList::RemoveFromPhysicsList(const G4String& name)
222 {
223  G4bool success = false;
224  for (G4PhysConstVector::iterator p = fPhysicsVector->begin();
225  p != fPhysicsVector->end(); ++p) {
226  G4VPhysicsConstructor* e = (*p);
227  if (e->GetPhysicsName() == name) {
228  fPhysicsVector->erase(p);
229  success = true;
230  break;
231  }
232  }
233  if (!success) {
234  G4ExceptionDescription message;
235  message << "PhysicsList::RemoveFromEMPhysicsList "<< name << "not found";
236  G4Exception("example WLSPhysicsList::RemoveFromPhysicsList()",
237  "ExamWLSPhysicsList01",FatalException,message);
238  }
239 }

311 void WLSPhysicsList::AddStepMax()
312 {
313  // Step limitation seen as a process
314 
315  auto particleIterator=GetParticleIterator();
316  particleIterator->reset();
317  while ((*particleIterator)()){
318  G4ParticleDefinition* particle = particleIterator->value();
319  G4ProcessManager* pmanager = particle->GetProcessManager();
320 
321  if (fStepMaxProcess->IsApplicable(*particle) && !particle->IsShortLived())
322  {
323  if (pmanager) pmanager ->AddDiscreteProcess(fStepMaxProcess);
324  }
325  }
326 }
