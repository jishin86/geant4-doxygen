76 G4double G4GEMChannel::GetEmissionProbability(G4Fragment* fragment)
77 {
78  G4int anA = fragment->GetA_asInt();
79  G4int aZ = fragment->GetZ_asInt();
80  ResidualA = anA - A;
81  ResidualZ = aZ - Z;
82  /*
83  G4cout << "G4GEMChannel: Z= " << Z << " A= " << A
84  << " FragmentZ= " << aZ << " FragmentA= " << anA
85  << " Zres= " << ResidualZ << " Ares= " << ResidualA
86  << G4endl;
87  */
88  // We only take into account channels which are physically allowed
89  EmissionProbability = 0.0;
90 
91  // Only channels which are physically allowed are taken into account
92  if (ResidualA >= ResidualZ && ResidualZ > 0 && ResidualA >= A) {
93 
94  //Effective excitation energy
95  G4double ExEnergy = fragment->GetExcitationEnergy()
96  - pairingCorrection->GetPairingCorrection(anA, aZ);
97  if(ExEnergy > 0.0) {
98  ResidualMass = G4NucleiProperties::GetNuclearMass(ResidualA, ResidualZ);
99  G4double FragmentMass = fragment->GetGroundStateMass();
100  G4double Etot = FragmentMass + ExEnergy;
101  // Coulomb Barrier calculation
102  CoulombBarrier =
103  theCoulombBarrierPtr->GetCoulombBarrier(ResidualA,ResidualZ,ExEnergy);
104  /*
105  G4cout << "Eexc(MeV)= " << ExEnergy/MeV
106  << " CoulBarrier(MeV)= " << CoulombBarrier/MeV << G4endl;
107  */
108  if(Etot > ResidualMass + EvaporatedMass + CoulombBarrier) {
109 
110  // Maximal Kinetic Energy
111  MaximalKineticEnergy = ((Etot-ResidualMass)*(Etot+ResidualMass)
112  + EvaporatedMass*EvaporatedMass)/(2.0*Etot)
113  - EvaporatedMass - CoulombBarrier;
114 
115  //G4cout << "CBarrier(MeV)= " << CoulombBarrier/MeV << G4endl;
116 
117  if (MaximalKineticEnergy > 0.0) {
118  // Total emission probability for this channel
119  EmissionProbability = theEvaporationProbabilityPtr->
120  EmissionProbability(*fragment, MaximalKineticEnergy);
121  }
122  }
123  }
124  }
125  //G4cout << "Prob= " << EmissionProbability << G4endl;
126  return EmissionProbability;
127 }

129 G4Fragment* G4GEMChannel::EmittedFragment(G4Fragment* theNucleus)
130 {
131  G4Fragment* evFragment = 0;
132  G4double evEnergy = SampleKineticEnergy(*theNucleus) + EvaporatedMass;
133 
134  G4ThreeVector momentum = G4RandomDirection()*
135  std::sqrt((evEnergy - EvaporatedMass)*(evEnergy + EvaporatedMass));
136 
137  G4LorentzVector EvaporatedMomentum(momentum, evEnergy);
138  G4LorentzVector ResidualMomentum = theNucleus->GetMomentum();
139  EvaporatedMomentum.boost(ResidualMomentum.boostVector());
140 
141  evFragment = new G4Fragment(A, Z, EvaporatedMomentum);
142  ResidualMomentum -= EvaporatedMomentum;
143  theNucleus->SetZandA_asInt(ResidualZ, ResidualA);
144  theNucleus->SetMomentum(ResidualMomentum);
145 
146  return evFragment;
147 }

149 G4double G4GEMChannel::SampleKineticEnergy(const G4Fragment & fragment)
150 // Samples fragment kinetic energy.
151 {
152  G4double U = fragment.GetExcitationEnergy();
153 
154  G4double Alpha = theEvaporationProbabilityPtr->CalcAlphaParam(fragment);
155  G4double Beta = theEvaporationProbabilityPtr->CalcBetaParam(fragment);
156 
157  // ***RESIDUAL***
158  //JMQ (September 2009) the following quantities refer to the RESIDUAL:
159  G4double delta0 = pairingCorrection->GetPairingCorrection(ResidualA,ResidualZ);
160  G4double Ux = (2.5 + 150.0/ResidualA)*MeV;
161  G4double Ex = Ux + delta0;
162  G4double InitialLevelDensity;
163  // ***end RESIDUAL ***
164 
165  // ***PARENT***
166  //JMQ (September 2009) the following quantities refer to the PARENT:
167 
168  G4double deltaCN = pairingCorrection->GetPairingCorrection(fragment.GetA_asInt(),
169  fragment.GetZ_asInt());
170  G4double aCN = theLevelDensityPtr->LevelDensityParameter(fragment.GetA_asInt(),
171  fragment.GetZ_asInt(),
172  U-deltaCN);
173  G4double UxCN = (2.5 + 150.0/G4double(fragment.GetA_asInt()))*MeV;
174  G4double ExCN = UxCN + deltaCN;
175  G4double TCN = 1.0/(std::sqrt(aCN/UxCN) - 1.5/UxCN);
176  // ***end PARENT***
177 
178  //JMQ quantities calculated for CN in InitialLevelDensity
179  if ( U < ExCN )
180  {
181  G4double E0CN = ExCN - TCN*(G4Log(TCN/MeV) - G4Log(aCN*MeV)/4.0
182  - 1.25*G4Log(UxCN/MeV) + 2.0*std::sqrt(aCN*UxCN));
183  InitialLevelDensity = (pi/12.0)*G4Exp((U-E0CN)/TCN)/TCN;
184  }
185  else
186  {
187  G4double x = U-deltaCN;
188  G4double x1 = std::sqrt(aCN*x);
189  InitialLevelDensity = (pi/12.0)*G4Exp(2*x1)/(x*std::sqrt(x1));
190  }
191 
192  G4double Spin = theEvaporationProbabilityPtr->GetSpin();
193  //JMQ BIG BUG fixed: hbarc instead of hbar_Planck !!!!
194  // it was fixed in total probability (for this channel) but remained still here!!
195  // G4double g = (2.0*Spin+1.0)*NuclearMass/(pi2* hbar_Planck*hbar_Planck);
196  G4double gg = (2.0*Spin+1.0)*EvaporatedMass/(pi2* hbarc*hbarc);
197  //
198  //JMQ fix on Rb and geometrical cross sections according to Furihata's paper
199  // (JAERI-Data/Code 2001-105, p6)
200  G4double Rb = 0.0;
201  if (A > 4)
202  {
203  G4double Ad = fG4pow->Z13(ResidualA);
204  G4double Aj = fG4pow->Z13(A);
205  Rb = (1.12*(Aj + Ad) - 0.86*((Aj+Ad)/(Aj*Ad))+2.85)*fermi;
206  }
207  else if (A>1)
208  {
209  G4double Ad = fG4pow->Z13(ResidualA);
210  G4double Aj = fG4pow->Z13(A);
211  Rb=1.5*(Aj+Ad)*fermi;
212  }
213  else
214  {
215  G4double Ad = fG4pow->Z13(ResidualA);
216  Rb = 1.5*Ad*fermi;
217  }
218  G4double GeometricalXS = pi*Rb*Rb;
219 
220  G4double ConstantFactor = gg*GeometricalXS*Alpha*pi/(InitialLevelDensity*12);
221  //JMQ : this is the assimptotic maximal kinetic energy of the ejectile
222  // (obtained by energy-momentum conservation).
223  // In general smaller than U-theSeparationEnergy
224  G4double theEnergy = MaximalKineticEnergy + CoulombBarrier;
225  G4double KineticEnergy;
226  G4double Probability;
227 
228  for(G4int i=0; i<100; ++i) {
229  KineticEnergy = CoulombBarrier + G4UniformRand()*(MaximalKineticEnergy);
230  G4double edelta = theEnergy-KineticEnergy-delta0;
231  Probability = ConstantFactor*(KineticEnergy + Beta);
232  G4double a =
233  theLevelDensityPtr->LevelDensityParameter(ResidualA,ResidualZ,edelta);
234  G4double T = 1.0/(std::sqrt(a/Ux) - 1.5/Ux);
235  //JMQ fix in units
236 
237  if (theEnergy - KineticEnergy < Ex) {
238  G4double E0 = Ex - T*(G4Log(T) - G4Log(a)*0.25
239  - 1.25*G4Log(Ux) + 2.0*std::sqrt(a*Ux));
240  Probability *= G4Exp((theEnergy-KineticEnergy-E0)/T)/T;
241  } else {
242  G4double e2 = edelta*edelta;
243  Probability *=
244  G4Exp(2*std::sqrt(a*edelta) - 0.25*G4Log(a*edelta*e2*e2));
245  }
246  if(EmissionProbability*G4UniformRand() <= Probability) { break; }
247  }
248 
249  return KineticEnergy;
250 }
