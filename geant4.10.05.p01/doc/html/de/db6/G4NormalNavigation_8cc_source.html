59 // ComputeStep
60 // ********************************************************************
61 //
62 // On entry
63 // exitNormal, validExitNormal: for previous exited volume (daughter)
64 //
65 // On exit
66 // exitNormal, validExitNormal: for mother, if exiting it (else unchanged)
67 G4double
68 G4NormalNavigation::ComputeStep(const G4ThreeVector &localPoint,
69  const G4ThreeVector &localDirection,
70  const G4double currentProposedStepLength,
71  G4double &newSafety,
72  G4NavigationHistory &history,
73  G4bool &validExitNormal,
74  G4ThreeVector &exitNormal,
75  G4bool &exiting,
76  G4bool &entering,
77  G4VPhysicalVolume *(*pBlockedPhysical),
78  G4int &blockedReplicaNo)
79 {
80  G4VPhysicalVolume *motherPhysical, *samplePhysical, *blockedExitedVol=0;
81  G4LogicalVolume *motherLogical;
82  G4VSolid *motherSolid;
83  G4ThreeVector sampleDirection;
84  G4double ourStep=currentProposedStepLength, ourSafety;
85  G4double motherSafety, motherStep=DBL_MAX;
86  G4int localNoDaughters, sampleNo;
87  G4bool motherValidExitNormal=false;
88  G4ThreeVector motherExitNormal;
89 
90  motherPhysical = history.GetTopVolume();
91  motherLogical = motherPhysical->GetLogicalVolume();
92  motherSolid = motherLogical->GetSolid();
93 
94  // Compute mother safety
95  //
96  motherSafety = motherSolid->DistanceToOut(localPoint);
97  ourSafety = motherSafety; // Working isotropic safety
98 
99  localNoDaughters = motherLogical->GetNoDaughters();
100 
101 #ifdef G4VERBOSE
102  if ( fCheck && ( (localNoDaughters>0) || (ourStep < motherSafety) ) )
103  {
104  fLogger->PreComputeStepLog(motherPhysical, motherSafety, localPoint);
105  }
106 #endif
107  // Compute daughter safeties & intersections
108  //
109 
110  // Exiting normal optimisation
111  //
112  if ( exiting&&validExitNormal )
113  {
114  if ( localDirection.dot(exitNormal)>=kMinExitingNormalCosine )
115  {
116  // Block exited daughter volume
117  //
118  blockedExitedVol = (*pBlockedPhysical);
119  ourSafety = 0;
120  }
121  }
122  exiting = false;
123  entering = false;
124 
125 #ifdef G4VERBOSE
126  if ( fCheck )
127  {
128  // Compute early:
129  // a) to check whether point is (wrongly) outside
130  // (signaled if step < 0 or step == kInfinity )
131  // b) to check value against answer of daughters!
132 
133  motherStep = motherSolid->DistanceToOut(localPoint,
134  localDirection,
135  true,
136  &motherValidExitNormal,
137  &motherExitNormal);
138 
139  if( (motherStep >= kInfinity) || (motherStep < 0.0) )
140  {
141  // Error - indication of being outside solid !!
142  fLogger->ReportOutsideMother(localPoint, localDirection, motherPhysical);
143 
144  ourStep = motherStep = 0.0;
145 
146  exiting= true;
147  entering= false;
148 
149  // If we are outside the solid does the normal make sense?
150  validExitNormal= motherValidExitNormal;
151  exitNormal= motherExitNormal;
152 
153  *pBlockedPhysical= 0; // or motherPhysical ?
154  blockedReplicaNo= 0; // or motherReplicaNumber ?
155 
156  newSafety= 0.0;
157  return ourStep;
158  }
159  }
160 #endif
161 
162  for ( sampleNo=localNoDaughters-1; sampleNo>=0; sampleNo--)
163  {
164  samplePhysical = motherLogical->GetDaughter(sampleNo);
165  if ( samplePhysical!=blockedExitedVol )
166  {
167  G4AffineTransform sampleTf(samplePhysical->GetRotation(),
168  samplePhysical->GetTranslation());
169  sampleTf.Invert();
170  const G4ThreeVector samplePoint = sampleTf.TransformPoint(localPoint);
171  const G4VSolid *sampleSolid =
172  samplePhysical->GetLogicalVolume()->GetSolid();
173  const G4double sampleSafety =
174  sampleSolid->DistanceToIn(samplePoint);
175 
176  if ( sampleSafety<ourSafety )
177  {
178  ourSafety=sampleSafety;
179  }
180 
181  if ( sampleSafety<=ourStep )
182  {
183  sampleDirection = sampleTf.TransformAxis(localDirection);
184  const G4double sampleStep =
185  sampleSolid->DistanceToIn(samplePoint,sampleDirection);
186 #ifdef G4VERBOSE
187  if( fCheck )
188  {
189  fLogger->PrintDaughterLog(sampleSolid, samplePoint,
190  sampleSafety, true,
191  sampleDirection, sampleStep);
192  }
193 #endif
194  if ( sampleStep<=ourStep )
195  {
196  ourStep = sampleStep;
197  entering = true;
198  exiting = false;
199  *pBlockedPhysical = samplePhysical;
200  blockedReplicaNo = -1;
201 #ifdef G4VERBOSE
202  if( fCheck )
203  {
204  fLogger->AlongComputeStepLog(sampleSolid, samplePoint,
205  sampleDirection, localDirection, sampleSafety, sampleStep);
206  }
207 #endif
208  }
209 
210 #ifdef G4VERBOSE
211  if( fCheck && (sampleStep < kInfinity) && (sampleStep >= motherStep) )
212  {
213  // The intersection point with the daughter is at or after the exit
214  // point from the mother volume. Double check!
215  fLogger->CheckDaughterEntryPoint(sampleSolid,
216  samplePoint, sampleDirection,
217  motherSolid,
218  localPoint, localDirection,
219  motherStep, sampleStep);
220  }
221 #endif
222  } // end of if ( sampleSafety <= ourStep )
223 #ifdef G4VERBOSE
224  else if( fCheck )
225  {
226  fLogger->PrintDaughterLog(sampleSolid, samplePoint,
227  sampleSafety, false,
228  G4ThreeVector(0.,0.,0.), -1.0 );
229  }
230 #endif
231  }
232  }
233  if ( currentProposedStepLength<ourSafety )
234  {
235  // Guaranteed physics limited
236  //
237  entering = false;
238  exiting = false;
239  *pBlockedPhysical = 0;
240  ourStep = kInfinity;
241  }
242  else
243  {
244  // Consider intersection with mother solid
245  //
246  if ( motherSafety<=ourStep )
247  {
248  if ( !fCheck ) // The call is moved above when running in check_mode
249  {
250  motherStep = motherSolid->DistanceToOut(localPoint,
251  localDirection,
252  true,
253  &motherValidExitNormal,
254  &motherExitNormal);
255  }
256 #ifdef G4VERBOSE
257  else // check_mode
258  {
259  fLogger->PostComputeStepLog(motherSolid, localPoint, localDirection,
260  motherStep, motherSafety);
261  if( motherValidExitNormal )
262  {
263  fLogger->CheckAndReportBadNormal(motherExitNormal,
264  localPoint,
265  localDirection,
266  motherStep,
267  motherSolid,
268  "From motherSolid::DistanceToOut" );
269  }
270  }
271 #endif
272 
273  if( (motherStep >= kInfinity) || (motherStep < 0.0) )
274  {
275 #ifdef G4VERBOSE
276  if( fCheck ) // Clearly outside the mother solid!
277  {
278  fLogger->ReportOutsideMother(localPoint, localDirection,
279  motherPhysical);
280  }
281 #endif
282  ourStep = motherStep = 0.0;
283  exiting = true;
284  entering = false;
285  // validExitNormal= motherValidExitNormal;
286  // exitNormal= motherExitNormal;
287  // The normal could be useful - but only if near the mother
288  // But it could be unreliable!
289  validExitNormal = false;
290  *pBlockedPhysical= 0; // or motherPhysical ?
291  blockedReplicaNo= 0; // or motherReplicaNumber ?
292  newSafety= 0.0;
293  return ourStep;
294  }
295 
296  if ( motherStep<=ourStep )
297  {
298  ourStep = motherStep;
299  exiting = true;
300  entering = false;
301  validExitNormal= motherValidExitNormal;
302  exitNormal= motherExitNormal;
303 
304  if ( motherValidExitNormal )
305  {
306  const G4RotationMatrix *rot = motherPhysical->GetRotation();
307  if (rot)
308  {
309  exitNormal *= rot->inverse();
310 #ifdef G4VERBOSE
311  if( fCheck )
312  fLogger->CheckAndReportBadNormal(exitNormal, // rotated
313  motherExitNormal, // original
314  *rot,
315  "From RotationMatrix" );
316 #endif
317  }
318  }
319  }
320  else
321  {
322  validExitNormal = false;
323  }
324  }
325  }
326  newSafety = ourSafety;
327  return ourStep;
328 }

334 G4double G4NormalNavigation::ComputeSafety(const G4ThreeVector &localPoint,
335  const G4NavigationHistory &history,
336  const G4double)
337 {
338  G4VPhysicalVolume *motherPhysical, *samplePhysical;
339  G4LogicalVolume *motherLogical;
340  G4VSolid *motherSolid;
341  G4double motherSafety, ourSafety;
342  G4int localNoDaughters, sampleNo;
343 
344  motherPhysical = history.GetTopVolume();
345  motherLogical = motherPhysical->GetLogicalVolume();
346  motherSolid = motherLogical->GetSolid();
347 
348  // Compute mother safety
349  //
350  motherSafety = motherSolid->DistanceToOut(localPoint);
351  ourSafety = motherSafety; // Working isotropic safety
352 
353 #ifdef G4VERBOSE
354  if( fCheck )
355  {
356  fLogger->ComputeSafetyLog(motherSolid,localPoint,motherSafety,true,true);
357  }
358 #endif
359 
360  // Compute daughter safeties
361  //
362  localNoDaughters = motherLogical->GetNoDaughters();
363  for ( sampleNo=localNoDaughters-1; sampleNo>=0; sampleNo-- )
364  {
365  samplePhysical = motherLogical->GetDaughter(sampleNo);
366  G4AffineTransform sampleTf(samplePhysical->GetRotation(),
367  samplePhysical->GetTranslation());
368  sampleTf.Invert();
369  const G4ThreeVector samplePoint =
370  sampleTf.TransformPoint(localPoint);
371  const G4VSolid *sampleSolid =
372  samplePhysical->GetLogicalVolume()->GetSolid();
373  const G4double sampleSafety =
374  sampleSolid->DistanceToIn(samplePoint);
375  if ( sampleSafety<ourSafety )
376  {
377  ourSafety = sampleSafety;
378  }
379 #ifdef G4VERBOSE
380  if(fCheck)
381  {
382  fLogger->ComputeSafetyLog(sampleSolid,samplePoint,
383  sampleSafety,false,false);
384  // Not mother, no banner
385  }
386 #endif
387  }
388  return ourSafety;
389 }
