174 inline G4int G4PAIModel::FindCoupleIndex(const G4MaterialCutsCouple* couple)
175 {
176  G4int idx = -1;
177  size_t jMatMax = fMaterialCutsCoupleVector.size();
178  for(size_t jMat = 0;jMat < jMatMax; ++jMat) {
179  if(couple == fMaterialCutsCoupleVector[jMat]) {
180  idx = jMat;
181  break;
182  }
183  }
184  return idx;
185 }
