144 G4double G4MonopoleTransportation::
145 AlongStepGetPhysicalInteractionLength( const G4Track& track,
146  G4double, // previousStepSize
147  G4double currentMinimumStep,
148  G4double& currentSafety,
149  G4GPILSelection* selection )
150 {
151  fMagSetup->SetStepperAndChordFinder(1);
152  // change to monopole equation
153 
154  G4double geometryStepLength, newSafety ;
155  fParticleIsLooping = false ;
156 
157  // Initial actions moved to StartTrack()
158  // --------------------------------------
159  // Note: in case another process changes touchable handle
160  // it will be necessary to add here (for all steps)
161  // fCurrentTouchableHandle = aTrack->GetTouchableHandle();
162 
163  // GPILSelection is set to defaule value of CandidateForSelection
164  // It is a return value
165  //
166  *selection = CandidateForSelection ;
167 
168  // Get initial Energy/Momentum of the track
169  //
170  const G4DynamicParticle* pParticle = track.GetDynamicParticle() ;
171  G4ThreeVector startMomentumDir = pParticle->GetMomentumDirection() ;
172  G4ThreeVector startPosition = track.GetPosition() ;
173 
174  // G4double theTime = track.GetGlobalTime() ;
175 
176  // The Step Point safety can be limited by other geometries and/or the
177  // assumptions of any process - it's not always the geometrical safety.
178  // We calculate the starting point's isotropic safety here.
179  //
180  G4ThreeVector OriginShift = startPosition - fPreviousSftOrigin ;
181  G4double MagSqShift = OriginShift.mag2() ;
182  if( MagSqShift >= sqr(fPreviousSafety) )
183  {
184  currentSafety = 0.0 ;
185  }
186  else
187  {
188  currentSafety = fPreviousSafety - std::sqrt(MagSqShift) ;
189  }
190 
191  // Is the monopole charged ?
192  //
193  G4double particleMagneticCharge = fParticleDef->MagneticCharge() ;
194  G4double particleElectricCharge = pParticle->GetCharge();
195 
196  fGeometryLimitedStep = false ;
197  // fEndGlobalTimeComputed = false ;
198 
199  // There is no need to locate the current volume. It is Done elsewhere:
200  // On track construction
201  // By the tracking, after all AlongStepDoIts, in "Relocation"
202 
203  // Check whether the particle have an (EM) field force exerting upon it
204  //
205  G4FieldManager* fieldMgr=0;
206  G4bool fieldExertsForce = false ;
207 
208  if( (particleMagneticCharge != 0.0) )
209  {
210  fieldMgr= fFieldPropagator->FindAndSetFieldManager( track.GetVolume() );
211  if (fieldMgr != 0) {
212  // Message the field Manager, to configure it for this track
213  fieldMgr->ConfigureForTrack( &track );
214  // Moved here, in order to allow a transition
215  // from a zero-field status (with fieldMgr->(field)0
216  // to a finite field status
217 
218  // If the field manager has no field, there is no field !
219  fieldExertsForce = (fieldMgr->GetDetectorField() != 0);
220  }
221  }
222 
223  // G4cout << " G4Transport: field exerts force= " << fieldExertsForce
224  // << " fieldMgr= " << fieldMgr << G4endl;
225 
226  // Choose the calculation of the transportation: Field or not
227  //
228  if( !fieldExertsForce )
229  {
230  G4double linearStepLength ;
231  if( fShortStepOptimisation && (currentMinimumStep <= currentSafety) )
232  {
233  // The Step is guaranteed to be taken
234  //
235  geometryStepLength = currentMinimumStep ;
236  fGeometryLimitedStep = false ;
237  }
238  else
239  {
240  // Find whether the straight path intersects a volume
241  //
242  linearStepLength = fLinearNavigator->ComputeStep( startPosition,
243  startMomentumDir,
244  currentMinimumStep,
245  newSafety) ;
246  // Remember last safety origin & value.
247  //
248  fPreviousSftOrigin = startPosition ;
249  fPreviousSafety = newSafety ;
250  // fpSafetyHelper->SetCurrentSafety( newSafety, startPosition);
251 
252  // The safety at the initial point has been re-calculated:
253  //
254  currentSafety = newSafety ;
255 
256  fGeometryLimitedStep= (linearStepLength <= currentMinimumStep);
257  if( fGeometryLimitedStep )
258  {
259  // The geometry limits the Step size (an intersection was found.)
260  geometryStepLength = linearStepLength ;
261  }
262  else
263  {
264  // The full Step is taken.
265  geometryStepLength = currentMinimumStep ;
266  }
267  }
268  endpointDistance = geometryStepLength ;
269 
270  // Calculate final position
271  //
272  fTransportEndPosition = startPosition+geometryStepLength*startMomentumDir ;
273 
274  // Momentum direction, energy and polarisation are unchanged by transport
275  //
276  fTransportEndMomentumDir = startMomentumDir ;
277  fTransportEndKineticEnergy = track.GetKineticEnergy() ;
278  fTransportEndSpin = track.GetPolarization();
279  fParticleIsLooping = false ;
280  fMomentumChanged = false ;
281  fEndGlobalTimeComputed = false ;
282  }
283  else // A field exerts force
284  {
285  G4double momentumMagnitude = pParticle->GetTotalMomentum() ;
286  G4ThreeVector EndUnitMomentum ;
287  G4double lengthAlongCurve ;
288  G4double restMass = fParticleDef->GetPDGMass() ;
289 
290  G4ChargeState chargeState(particleElectricCharge, // The charge can change
291  fParticleDef->GetPDGSpin(),
292  0, // Magnetic moment:
293  0, // Electric Dipole moment
294  particleMagneticCharge ); // in Mev/c
295 
296  G4EquationOfMotion* equationOfMotion = fFieldPropagator->
297  GetChordFinder()->GetIntegrationDriver()->GetEquationOfMotion();
298 
299  equationOfMotion->SetChargeMomentumMass( chargeState,
300  momentumMagnitude,
301  restMass );
302  // SetChargeMomentumMass now passes both the electric and magnetic
303  // charge in chargeState
304 
305  G4ThreeVector spin = track.GetPolarization() ;
306  G4FieldTrack aFieldTrack = G4FieldTrack( startPosition,
307  track.GetMomentumDirection(),
308  0.0,
309  track.GetKineticEnergy(),
310  restMass,
311  track.GetVelocity(),
312  track.GetGlobalTime(), // Lab.
313  track.GetProperTime(), // Part.
314  &spin ) ;
315  if( currentMinimumStep > 0 )
316  {
317  // Do the Transport in the field (non recti-linear)
318  //
319  lengthAlongCurve = fFieldPropagator->ComputeStep( aFieldTrack,
320  currentMinimumStep,
321  currentSafety,
322  track.GetVolume() ) ;
323  fGeometryLimitedStep= lengthAlongCurve < currentMinimumStep;
324  if( fGeometryLimitedStep ) {
325  geometryStepLength = lengthAlongCurve ;
326  } else {
327  geometryStepLength = currentMinimumStep ;
328  }
329  }
330  else
331  {
332  geometryStepLength = lengthAlongCurve= 0.0 ;
333  fGeometryLimitedStep = false ;
334  }
335 
336  // Remember last safety origin & value.
337  //
338  fPreviousSftOrigin = startPosition ;
339  fPreviousSafety = currentSafety ;
340  // fpSafetyHelper->SetCurrentSafety( newSafety, startPosition);
341 
342  // Get the End-Position and End-Momentum (Dir-ection)
343  //
344  fTransportEndPosition = aFieldTrack.GetPosition() ;
345 
346  // Momentum: Magnitude and direction can be changed too now ...
347  //
348  fMomentumChanged = true ;
349  fTransportEndMomentumDir = aFieldTrack.GetMomentumDir() ;
350 
351  fTransportEndKineticEnergy = aFieldTrack.GetKineticEnergy() ;
352 
353  fCandidateEndGlobalTime = aFieldTrack.GetLabTimeOfFlight();
354  fEndGlobalTimeComputed = true;
355 
356  fTransportEndSpin = aFieldTrack.GetSpin();
357  fParticleIsLooping = fFieldPropagator->IsParticleLooping() ;
358  endpointDistance = (fTransportEndPosition - startPosition).mag() ;
359  }
360 
361  // If we are asked to go a step length of 0, and we are on a boundary
362  // then a boundary will also limit the step -> we must flag this.
363  //
364  if( currentMinimumStep == 0.0 )
365  {
366  if( currentSafety == 0.0 ) fGeometryLimitedStep = true ;
367  }
368 
369  // Update the safety starting from the end-point,
370  // if it will become negative at the end-point.
371  //
372  if( currentSafety < endpointDistance )
373  {
374  // if( particleMagneticCharge == 0.0 )
375  //G4cout << " Avoiding call to ComputeSafety : charge = 0.0 " << G4endl;
376 
377  if( particleMagneticCharge != 0.0 ) {
378 
379  G4double endSafety =
380  fLinearNavigator->ComputeSafety( fTransportEndPosition) ;
381  currentSafety = endSafety ;
382  fPreviousSftOrigin = fTransportEndPosition ;
383  fPreviousSafety = currentSafety ;
384  fpSafetyHelper->SetCurrentSafety(currentSafety, fTransportEndPosition);
385 
386  // Because the Stepping Manager assumes it is from the start point,
387  // add the StepLength
388  //
389  currentSafety += endpointDistance ;
390 
391 #ifdef G4DEBUG_TRANSPORT
392  G4cout.precision(12) ;
393  G4cout << "***G4MonopoleTransportation::AlongStepGPIL ** " << G4endl;
394  G4cout << " Called Navigator->ComputeSafety at "
395  << fTransportEndPosition
396  << " and it returned safety= " << endSafety << G4endl;
397  G4cout << " Adding endpoint distance " << endpointDistance
398  << " to obtain pseudo-safety= " << currentSafety << G4endl;
399 #endif
400  }
401  }
402 
403  fParticleChange.ProposeTrueStepLength(geometryStepLength) ;
404 
405  fMagSetup->SetStepperAndChordFinder(0);
406  // change back to usual equation
407 
408  return geometryStepLength ;
409 }

415 
416 G4VParticleChange* G4MonopoleTransportation::AlongStepDoIt(
417  const G4Track& track, const G4Step& stepData )
418 {
419  const G4ParticleDefinition* fOpticalPhoton =
420  G4ParticleTable::GetParticleTable()->FindParticle("opticalphoton");
421 
422  ++noCalls;
423 
424  fParticleChange.Initialize(track) ;
425 
426  // Code for specific process
427  //
428  fParticleChange.ProposePosition(fTransportEndPosition) ;
429  fParticleChange.ProposeMomentumDirection(fTransportEndMomentumDir) ;
430  fParticleChange.ProposeEnergy(fTransportEndKineticEnergy) ;
431  fParticleChange.SetMomentumChanged(fMomentumChanged) ;
432 
433  fParticleChange.ProposePolarization(fTransportEndSpin);
434 
435  G4double deltaTime = 0.0 ;
436 
437  // Calculate Lab Time of Flight (ONLY if field Equations used it!)
438 
439  G4double startTime = track.GetGlobalTime() ;
440 
441  if (!fEndGlobalTimeComputed)
442  {
443  // The time was not integrated .. make the best estimate possible
444  //
445  G4double finalVelocity = track.GetVelocity() ;
446  G4double initialVelocity = stepData.GetPreStepPoint()->GetVelocity() ;
447  G4double stepLength = track.GetStepLength() ;
448 
449  deltaTime= 0.0; // in case initialVelocity = 0
450  const G4DynamicParticle* fpDynamicParticle = track.GetDynamicParticle();
451  if (fpDynamicParticle->GetDefinition()== fOpticalPhoton)
452  {
453  // A photon is in the medium of the final point
454  // during the step, so it has the final velocity.
455  deltaTime = stepLength/finalVelocity ;
456  }
457  else if (finalVelocity > 0.0)
458  {
459  G4double meanInverseVelocity ;
460  // deltaTime = stepLength/finalVelocity ;
461  meanInverseVelocity = 0.5
462  * ( 1.0 / initialVelocity + 1.0 / finalVelocity );
463  deltaTime = stepLength * meanInverseVelocity ;
464  }
465  else if( initialVelocity > 0.0 )
466  {
467  deltaTime = stepLength/initialVelocity ;
468  }
469  fCandidateEndGlobalTime = startTime + deltaTime ;
470  }
471  else
472  {
473  deltaTime = fCandidateEndGlobalTime - startTime ;
474  }
475 
476  fParticleChange.ProposeGlobalTime( fCandidateEndGlobalTime ) ;
477 
478  // Now Correct by Lorentz factor to get "proper" deltaTime
479 
480  G4double restMass = track.GetDynamicParticle()->GetMass();
481  G4double deltaProperTime = deltaTime*( restMass/track.GetTotalEnergy() );
482 
483  fParticleChange.ProposeProperTime(track.GetProperTime() + deltaProperTime);
484 
485  // If the particle is caught looping or is stuck (in very difficult
486  // boundaries) in a magnetic field (doing many steps)
487  // THEN this kills it ...
488  //
489  if ( fParticleIsLooping )
490  {
491  G4double endEnergy= fTransportEndKineticEnergy;
492 
493  if( (endEnergy < fThreshold_Important_Energy)
494  || (fNoLooperTrials >= fThresholdTrials ) ){
495  // Kill the looping particle
496  //
497  fParticleChange.ProposeTrackStatus( fStopAndKill ) ;
498 
499  // 'Bare' statistics
500  fSumEnergyKilled += endEnergy;
501  if( endEnergy > fMaxEnergyKilled) { fMaxEnergyKilled= endEnergy; }
502 
503 #ifdef G4VERBOSE
504  if( (verboseLevel > 1) ||
505  ( endEnergy > fThreshold_Warning_Energy ) ) {
506  G4cout << " G4MonopoleTransportation is killing track "
507  << "that is looping or stuck " << G4endl
508  << " This track has " << track.GetKineticEnergy() / MeV
509  << " MeV energy." << G4endl;
510  G4cout << " Number of trials = " << fNoLooperTrials
511  << " No of calls to AlongStepDoIt = " << noCalls
512  << G4endl;
513  }
514 #endif
515  fNoLooperTrials=0;
516  }
517  else{
518  fNoLooperTrials ++;
519 #ifdef G4VERBOSE
520  if( (verboseLevel > 2) ){
521  G4cout << " G4MonopoleTransportation::AlongStepDoIt(): "
522  << "Particle looping - "
523  << " Number of trials = " << fNoLooperTrials
524  << " No of calls to = " << noCalls
525  << G4endl;
526  }
527 #endif
528  }
529  }else{
530  fNoLooperTrials=0;
531  }
532 
533  // Another (sometimes better way) is to use a user-limit maximum Step size
534  // to alleviate this problem ..
535 
536  // Introduce smooth curved trajectories to particle-change
537  //
538  fParticleChange.SetPointerToVectorOfAuxiliaryPoints
539  (fFieldPropagator->GimmeTrajectoryVectorAndForgetIt() );
540 
541  return &fParticleChange ;
542 }

561 G4VParticleChange* G4MonopoleTransportation::PostStepDoIt( const G4Track& track,
562  const G4Step& )
563 {
564  G4TouchableHandle retCurrentTouchable ; // The one to return
565 
566  // Initialize ParticleChange (by setting all its members equal
567  // to corresponding members in G4Track)
568  // fParticleChange.Initialize(track) ; // To initialise TouchableChange
569 
570  fParticleChange.ProposeTrackStatus(track.GetTrackStatus()) ;
571 
572  // If the Step was determined by the volume boundary,
573  // logically relocate the particle
574 
575  if(fGeometryLimitedStep)
576  {
577  // fCurrentTouchable will now become the previous touchable,
578  // and what was the previous will be freed.
579  // (Needed because the preStepPoint can point to the previous touchable)
580 
581  fLinearNavigator->SetGeometricallyLimitedStep() ;
582  fLinearNavigator->
583  LocateGlobalPointAndUpdateTouchableHandle( track.GetPosition(),
584  track.GetMomentumDirection(),
585  fCurrentTouchableHandle,
586  true ) ;
587  // Check whether the particle is out of the world volume
588  // If so it has exited and must be killed.
589  //
590  if( fCurrentTouchableHandle->GetVolume() == 0 )
591  {
592  fParticleChange.ProposeTrackStatus( fStopAndKill ) ;
593  }
594  retCurrentTouchable = fCurrentTouchableHandle ;
595  fParticleChange.SetTouchableHandle( fCurrentTouchableHandle ) ;
596  }
597  else // fGeometryLimitedStep is false
598  {
599  // This serves only to move the Navigator's location
600  //
601  fLinearNavigator->LocateGlobalPointWithinVolume( track.GetPosition() ) ;
602 
603  // The value of the track's current Touchable is retained.
604  // (and it must be correct because we must use it below to
605  // overwrite the (unset) one in particle change)
606  // It must be fCurrentTouchable too ??
607  //
608  fParticleChange.SetTouchableHandle( track.GetTouchableHandle() ) ;
609  retCurrentTouchable = track.GetTouchableHandle() ;
610  } // endif ( fGeometryLimitedStep )
611 
612  const G4VPhysicalVolume* pNewVol = retCurrentTouchable->GetVolume() ;
613  const G4Material* pNewMaterial = 0 ;
614  const G4VSensitiveDetector* pNewSensitiveDetector = 0 ;
615  if( pNewVol != 0 )
616  {
617  pNewMaterial= pNewVol->GetLogicalVolume()->GetMaterial();
618  pNewSensitiveDetector= pNewVol->GetLogicalVolume()->GetSensitiveDetector();
619  }
620 
621  // ( <const_cast> pNewMaterial ) ;
622  // ( <const_cast> pNewSensitiveDetector) ;
623 
624  fParticleChange.SetMaterialInTouchable(
625  (G4Material *) pNewMaterial ) ;
626  fParticleChange.SetSensitiveDetectorInTouchable(
627  (G4VSensitiveDetector *) pNewSensitiveDetector ) ;
628 
629  const G4MaterialCutsCouple* pNewMaterialCutsCouple = 0;
630  if( pNewVol != 0 )
631  {
632  pNewMaterialCutsCouple=pNewVol->GetLogicalVolume()->GetMaterialCutsCouple();
633  }
634 
635  if( pNewVol!=0 && pNewMaterialCutsCouple!=0 &&
636  pNewMaterialCutsCouple->GetMaterial()!=pNewMaterial )
637  {
638  // for parametrized volume
639  //
640  pNewMaterialCutsCouple =
641  G4ProductionCutsTable::GetProductionCutsTable()
642  ->GetMaterialCutsCouple(pNewMaterial,
643  pNewMaterialCutsCouple->GetProductionCuts());
644  }
645  fParticleChange.SetMaterialCutsCoupleInTouchable( pNewMaterialCutsCouple );
646 
647  // temporarily until Get/Set Material of ParticleChange,
648  // and StepPoint can be made const.
649  // Set the touchable in ParticleChange
650  // this must always be done because the particle change always
651  // uses this value to overwrite the current touchable pointer.
652  //
653  fParticleChange.SetTouchableHandle(retCurrentTouchable) ;
654 
655  return &fParticleChange ;
656 }
