60 ScoreSpecies::ScoreSpecies(G4String name, G4int depth)
61 : G4VPrimitiveScorer(name,depth),
62  fEdep(0),
63  fHCID(-1),
64  fEvtMap(0)
65 {
66  fNEvent = 0;
67  G4double tMin = 1.0 * CLHEP::picosecond;
68  G4double tMax = 999999 * CLHEP::picosecond;
69  G4double tLogMin = std::log10(tMin);
70  G4double tLogMax = std::log10(tMax);
71  G4int tBins = 50;
72  for ( int i = 0; i <= tBins; i++ )
73  AddTimeToRecord(std::pow(10., tLogMin + i*(tLogMax-tLogMin)/tBins));
74 
75  fEdep = 0;
76 }

85 G4bool ScoreSpecies::ProcessHits(G4Step* aStep,G4TouchableHistory*)
86 {
87  G4double edep = aStep->GetTotalEnergyDeposit();
88 
89  if ( edep == 0. ) return FALSE;
90 
91  edep *= aStep->GetPreStepPoint()->GetWeight();
92  G4int index = GetIndex(aStep);
93  fEvtMap->add(index,edep);
94  fEdep+=edep;
95 
96  return TRUE;
97 }

116 void ScoreSpecies::EndOfEvent(G4HCofThisEvent*)
117 {
118  if(G4EventManager::GetEventManager()->GetConstCurrentEvent()->IsAborted())
119  {
120  fEdep = 0.;
121  G4MoleculeCounter::Instance()->ResetCounter();
122  return;
123  }
124 
125  auto species = G4MoleculeCounter::Instance()->GetRecordedMolecules();
126 
127  if(species.get() == 0 || species->size() == 0)
128  {
129  G4cout << "No molecule recorded, energy deposited= "
130  << G4BestUnit(fEdep, "Energy") << G4endl;
131  ++fNEvent;
132  fEdep = 0.;
133  G4MoleculeCounter::Instance()->ResetCounter();
134  return;
135  }
136 
137  for(auto molecule: *species)
138  {
139  for(auto time_mol: fTimeToRecord)
140  {
141  double n_mol =
142  G4MoleculeCounter::Instance()->GetNMoleculesAtTime(molecule,
143  time_mol);
144 
145  if(n_mol < 0)
146  {
147  G4cerr << "N molecules not valid < 0 " << G4endl;
148  G4Exception("","N<0",FatalException,"");
149  }
150 
151  SpeciesInfo& molInfo =
152  fSpeciesInfoPerTime[time_mol][molecule];
153  molInfo.fNumber += n_mol;
154  double gValue = (n_mol/(fEdep/eV)) * 100.;
155  molInfo.fG += gValue;
156  molInfo.fG2 += gValue*gValue;
157  }
158  }
159  ++fNEvent;
160 
161  fEdep = 0.;
162  G4MoleculeCounter::Instance()->ResetCounter();
163 }

167 void
168 ScoreSpecies::AbsorbResultsFromWorkerScorer(G4VPrimitiveScorer* workerScorer)
169 {
170  ScoreSpecies* right =
171  dynamic_cast<ScoreSpecies*>(dynamic_cast<G4VPrimitiveScorer*>(workerScorer));
172 
173  if(right == 0)
174  {
175  return;
176  }
177  if(right == this)
178  {
179  return;
180  }
181 
182  SpeciesMap::iterator it_map1 = right->fSpeciesInfoPerTime.begin();
183  SpeciesMap::iterator end_map1 = right->fSpeciesInfoPerTime.end();
184 
185  for(; it_map1 != end_map1; ++it_map1)
186  {
187  InnerSpeciesMap& map2 = it_map1->second;
188  InnerSpeciesMap::iterator it_map2 = map2.begin();
189  InnerSpeciesMap::iterator end_map2 = map2.end();
190 
191  for(; it_map2 != end_map2; ++it_map2)
192  {
193  SpeciesInfo& molInfo =
194  fSpeciesInfoPerTime[it_map1->first][it_map2->first] ;
195  molInfo.fNumber += it_map2->second.fNumber;
196  molInfo.fG += it_map2->second.fG;
197  molInfo.fG2 += it_map2->second.fG2;
198  }
199  }
200 
201  fNEvent += right->fNEvent;
202  right->fNEvent = 0;
203  right->fEdep = 0.;
204 }

242 void ScoreSpecies::ASCII()
243 {
244  std::ofstream out("Species.txt");
245  if(!out) return;
246 
247  out << "# Time [ps] G-value (/100 eV) RMS Molecule" << G4endl;
248 
249  std::map<G4String, std::map<G4double, std::pair<G4double,G4double>>> mol;
250 
251  for(auto it_map1: fSpeciesInfoPerTime)
252  {
253  InnerSpeciesMap& map2 = it_map1.second;
254  G4double time = it_map1.first/ps;
255  for(auto it_map2: map2)
256  {
257  G4double G = it_map2.second.fG;
258  G4double G2 = it_map2.second.fG2;
259  G4double N = fNEvent;
260  G /= N;
261  G2 = std::sqrt( N/(N-1) * ( G2/N - G*G) );
262  mol[it_map2.first->GetName()][time]=std::make_pair(G,G2);
263  }
264  }
265 
266  for ( auto it1 : mol )
267  for ( auto it2 : it1.second )
268  out << std::setw(12) << it2.first << std::setw(12) << it2.second.first
269  << std::setw(12) << it2.second.second << std::setw(12)
270  << std::setw(12) << it1.first << G4endl;
271 
272  out.close();
273 }
