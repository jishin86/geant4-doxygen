68 G4bool G4UnstableFragmentBreakUp::BreakUpChain(G4FragmentVector* results,
69  G4Fragment* nucleus)
70 {
71  //G4cout << "G4UnstableFragmentBreakUp::EmittedFragment" << G4endl;
72  G4Fragment* frag = nullptr;
73 
74  G4int Z = nucleus->GetZ_asInt();
75  G4int A = nucleus->GetA_asInt();
76 
77  // if the isotope is in the database it is not exotic
78  // so, cannot be handled by this class
79  if(fLevelData->GetLevelManager(Z, A)) { return false; }
80 
81  G4LorentzVector lv = nucleus->GetMomentum();
82  G4double time = nucleus->GetCreationTime();
83 
84  G4double mass1(0.0), mass2(0.0);
85 
86  G4double mass = lv.mag();
87  G4bool done = false;
88 
89  G4int Amax = A;
90  G4int Zres = Z;
91  G4int Ares = A;
92 
93  for(G4int k=0; k<Amax; ++k) {
94  // look for the decay channel with normal masses
95  // without Coulomb barrier and paring corrections
96 
97  // 1 - recoil, 2 - emitted light ion
98 
99  //G4cout << "Unstable decay #" << k << " Z= " << Z << " A= " << A << G4endl;
100 
101  G4bool isChannel = false;
102  G4double ekin = 0.0;
103  G4int i;
104  for(i=0; i<6; ++i) {
105  Zres = Z - Zfr[i];
106  Ares = A - Afr[i];
107  for(G4int j=0; j<6; ++j) {
108  if(Zres == Zfr[j] && Ares == Afr[j]) {
109  /*
110  G4cout << "i= " << i << " j= " << j << " Zres= " << Zres
111  << " Ares= " << Ares << " dm= " << mass - masses[i] - masses[j]
112  << G4endl;
113  */
114  if(mass >= masses[i] + masses[j]) {
115  mass2 = masses[i];
116  mass1 = masses[j];
117  ekin = 0.0;
118  done = true;
119  isChannel = true;
120  break;
121  }
122  }
123  }
124  if(isChannel) { break; }
125  if(Zres >= 0 && Ares >= Zres && Ares > 0) {
126  G4double mres = G4NucleiProperties::GetNuclearMass(Ares, Zres);
127  G4double e = mass - mres - masses[i];
128  // select the most excited state
129  const G4LevelManager* lman = fLevelData->GetLevelManager(Z, A);
130  if(lman && e >= ekin) {
131  mass2 = masses[i];
132  mass1 = mres;
133  ekin = lman->NearestLowEdgeLevelEnergy(e);
134  isChannel = true;
135  done = true;
136  break;
137  }
138  }
139  }
140 
141  // no decay channel - assume that primary mass is biased
142  // only energy will be conserved
143  if(!isChannel) {
144  for(i=0; i<6; ++i) {
145  Zres = Z - Zfr[i];
146  Ares = A - Afr[i];
147  if(Zres >= 0 && Ares >= Zres && Ares > 0) {
148 
149  G4double mres = proton_mass_c2*Zres + neutron_mass_c2*(Ares - Zres);
150  G4double e = mass - mres - masses[i];
151  if(e >= ekin) {
152  mass2 = masses[i];
153  mass1 = mres;
154  ekin = e;
155  isChannel = true;
156  break;
157  }
158  }
159  }
160  }
161  if(!isChannel) {
162  // further decay impossible
163  // if no one decay sampled do not update primary
164  if(0 == k) { return false; }
165  else { break; }
166  }
167 
168  // useful to left max excitation for the residual
169  mass1 += ekin;
170 
171  // compute energy of light fragment
172  G4double e2 = 0.5*((mass - mass1)*(mass + mass1) + mass2*mass2)/mass;
173  e2 = std::max(e2, mass2);
174  G4double mom = std::sqrt((e2 - mass2)*(e2 + mass2));
175 
176  // sample decay
177  G4ThreeVector bst = lv.boostVector();
178  G4ThreeVector v = G4RandomDirection();
179  G4LorentzVector mom2 = G4LorentzVector(v*mom, e2);
180  mom2.boost(bst);
181  frag = new G4Fragment(Afr[i], Zfr[i], mom2);
182  frag->SetCreationTime(time);
183  results->push_back(frag);
184 
185  // residual
186  lv -= mom2;
187  Z -= Zfr[i];
188  A -= Afr[i];
189 
190  mass = lv.mag();
191 
192  if(done) { break; }
193  }
194 
195  // updated primary
196  nucleus->SetZandA_asInt(Z, A);
197  nucleus->SetMomentum(lv);
198  return true;
199 }
