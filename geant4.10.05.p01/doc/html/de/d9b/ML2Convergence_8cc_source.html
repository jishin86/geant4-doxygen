113 G4bool CML2Convergence::convergenceCriteria()
114 {
115  G4bool bStopRun=true;
116  G4int nEventsAccumulated=0;
117  if (bCompareExp)
118  {
119  nEventsAccumulated=ML2ExpVoxels->getMaxNumberOfEvents();
120  // It checks if the maximum number of events is reached at least in one voxel. Having more rotations the limits is incremented each rotation
121  if (ML2ExpVoxels->getMaxNumberOfEvents()>= maxNumberOfEvents)
122  {bStopRun = true; ML2ExpVoxels->resetNEventsInVoxels();}
123  else
124  {bStopRun = false;}
125  }
126  G4cout <<"\n ++++++++++++++++++++ " << G4endl;
127  G4cout <<"current geometry: " << nGeometry;
128  G4cout << "\nNumber of events accumulated in the current geometry:"<<
129  nEventsAccumulated<<"\nNumber of events to be accumulated:" <<
130  maxNumberOfEvents<< "\n -------------------------\n" << G4endl;
131  return bStopRun;
