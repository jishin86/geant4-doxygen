114 void G4eCoulombScatteringModel::Initialise(const G4ParticleDefinition* part,
115  const G4DataVector& cuts)
116 {
117  if(!wokvi) { wokvi = new G4WentzelOKandVIxSection(); }
118 
119  SetupParticle(part);
120  currentCouple = nullptr;
121 
122  // defined theta limit between single and multiple scattering
123  if(isCombined) {
124  cosThetaMin = 1.0;
125  G4double tet = PolarAngleLimit();
126  if(tet >= pi) { cosThetaMin = -1.0; }
127  else if(tet > 0.0) { cosThetaMin = cos(tet); }
128  }
129 
130  wokvi->Initialise(part, cosThetaMin);
131  /*
132  G4cout << "G4eCoulombScatteringModel: " << particle->GetParticleName()
133  << " 1-cos(ThetaLimit)= " << 1 - cosThetaMin
134  << " cos(thetaMax)= " << cosThetaMax
135  << G4endl;
136  */
137  pCuts = &cuts;
138  //G4ProductionCutsTable::GetProductionCutsTable()->GetEnergyCutsVector(3);
139  /*
140  G4cout << "!!! G4eCoulombScatteringModel::Initialise for "
141  << part->GetParticleName() << " cos(TetMin)= " << cosThetaMin
142  << " cos(TetMax)= " << cosThetaMax <<G4endl;
143  G4cout << "cut= " << (*pCuts)[0] << " cut1= " << (*pCuts)[1] << G4endl;
144  */
145  if(!fParticleChange) {
146  fParticleChange = GetParticleChangeForGamma();
147  }
148  if(IsMaster() && mass < GeV && part->GetParticleName() != "GenericIon") {
149  InitialiseElementSelectors(part, cuts);
150  }
151 }

163 G4double
164 G4eCoulombScatteringModel::MinPrimaryEnergy(const G4Material* material,
165  const G4ParticleDefinition* part,
166  G4double)
167 {
168  SetupParticle(part);
169 
170  // define cut using cuts for proton
171  G4double cut =
172  std::max(recoilThreshold, (*pCuts)[CurrentCouple()->GetIndex()]);
173 
174  // find out lightest element
175  const G4ElementVector* theElementVector = material->GetElementVector();
176  G4int nelm = material->GetNumberOfElements();
177 
178  // select lightest element
179  G4int Z = 300;
180  for (G4int j=0; j<nelm; ++j) {
181  Z = std::min(Z,(*theElementVector)[j]->GetZasInt());
182  }
183  G4int A = G4lrint(fNistManager->GetAtomicMassAmu(Z));
184  G4double targetMass = G4NucleiProperties::GetNuclearMass(A, Z);
185  G4double t = std::max(cut, 0.5*(cut + sqrt(2*cut*targetMass)));
186 
187  return t;
188 }

192 G4double G4eCoulombScatteringModel::ComputeCrossSectionPerAtom(
193  const G4ParticleDefinition* p,
194  G4double kinEnergy,
195  G4double Z, G4double,
196  G4double cutEnergy, G4double)
197 {
198  //G4cout << "### G4eCoulombScatteringModel::ComputeCrossSectionPerAtom for "
199  //<< p->GetParticleName()<<" Z= "<<Z<<" e(MeV)= "<< kinEnergy/MeV << G4endl;
200  G4double cross = 0.0;
201  elecRatio = 0.0;
202  if(p != particle) { SetupParticle(p); }
203 
204  // cross section is set to zero to avoid problems in sample secondary
205  if(kinEnergy <= 0.0) { return cross; }
206  DefineMaterial(CurrentCouple());
207  G4double costmin = wokvi->SetupKinematic(kinEnergy, currentMaterial);
208 
209  if(cosThetaMax < costmin) {
210  G4int iz = G4lrint(Z);
211  G4double cut = (0.0 < fixedCut) ? fixedCut : cutEnergy;
212  costmin = wokvi->SetupTarget(iz, cut);
213  G4double costmax = (1 == iz && particle == theProton && cosThetaMax < 0.0)
214  ? 0.0 : cosThetaMax;
215  if(costmin > costmax) {
216  cross = wokvi->ComputeNuclearCrossSection(costmin, costmax)
217  + wokvi->ComputeElectronCrossSection(costmin, costmax);
218  }
219  /*
220  if(p->GetParticleName() == "e-")
221  G4cout << "Z= " << Z << " e(MeV)= " << kinEnergy/MeV
222  << " cross(b)= " << cross/barn << " 1-costmin= " << 1-costmin
223  << " 1-costmax= " << 1-costmax
224  << " 1-cosThetaMax= " << 1-cosThetaMax
225  << " " << currentMaterial->GetName()
226  << G4endl;
227  */
228  }
229  return cross;
230 }

234 void G4eCoulombScatteringModel::SampleSecondaries(
235  std::vector<G4DynamicParticle*>* fvect,
236  const G4MaterialCutsCouple* couple,
237  const G4DynamicParticle* dp,
238  G4double cutEnergy,
239  G4double)
240 {
241  G4double kinEnergy = dp->GetKineticEnergy();
242  SetupParticle(dp->GetDefinition());
243  DefineMaterial(couple);
244  /*
245  G4cout << "G4eCoulombScatteringModel::SampleSecondaries e(MeV)= "
246  << kinEnergy << " " << particle->GetParticleName()
247  << " cut= " << cutEnergy<< G4endl;
248  */
249  // Choose nucleus
250  G4double cut = (0.0 < fixedCut) ? fixedCut : cutEnergy;
251 
252  wokvi->SetupKinematic(kinEnergy, currentMaterial);
253 
254  const G4Element* currentElement =
255  SelectRandomAtom(couple,particle,kinEnergy,cut,kinEnergy);
256 
257  G4int iz = currentElement->GetZasInt();
258 
259  G4double costmin = wokvi->SetupTarget(iz, cut);
260  G4double costmax = (1 == iz && particle == theProton && cosThetaMax < 0.0)
261  ? 0.0 : cosThetaMax;
262  if(costmin <= costmax) { return; }
263 
264  G4double cross = wokvi->ComputeNuclearCrossSection(costmin, costmax);
265  G4double ecross = wokvi->ComputeElectronCrossSection(costmin, costmax);
266  G4double ratio = ecross/(cross + ecross);
267 
268  G4int ia = SelectIsotopeNumber(currentElement);
269  G4double targetMass = G4NucleiProperties::GetNuclearMass(ia, iz);
270  wokvi->SetTargetMass(targetMass);
271 
272  G4ThreeVector newDirection =
273  wokvi->SampleSingleScattering(costmin, costmax, ratio);
274  G4double cost = newDirection.z();
275  /*
276  G4cout << "SampleSec: e(MeV)= " << kinEnergy/MeV
277  << " 1-costmin= " << 1-costmin
278  << " 1-costmax= " << 1-costmax
279  << " 1-cost= " << 1-cost
280  << " ratio= " << ratio
281  << G4endl;
282  */
283  G4ThreeVector direction = dp->GetMomentumDirection();
284  newDirection.rotateUz(direction);
285 
286  fParticleChange->ProposeMomentumDirection(newDirection);
287 
288  // recoil sampling assuming a small recoil
289  // and first order correction to primary 4-momentum
290  G4double mom2 = wokvi->GetMomentumSquare();
291  G4double trec = mom2*(1.0 - cost)
292  /(targetMass + (mass + kinEnergy)*(1.0 - cost));
293 
294  // the check likely not needed
295  if(trec > kinEnergy) { trec = kinEnergy; }
296  G4double finalT = kinEnergy - trec;
297  G4double edep = 0.0;
298  /*
299  G4cout<<"G4eCoulombScatteringModel: finalT= "<<finalT<<" Trec= "
300  <<trec << " Z= " << iz << " A= " << ia
301  << " tcut(keV)= " << (*pCuts)[currentMaterialIndex]/keV << G4endl;
302  */
303  G4double tcut = recoilThreshold;
304  if(pCuts) { tcut= std::max(tcut,(*pCuts)[currentMaterialIndex]); }
305 
306  if(trec > tcut) {
307  G4ParticleDefinition* ion = theIonTable->GetIon(iz, ia, 0);
308  G4ThreeVector dir = (direction*sqrt(mom2) -
309  newDirection*sqrt(finalT*(2*mass + finalT))).unit();
310  G4DynamicParticle* newdp = new G4DynamicParticle(ion, dir, trec);
311  fvect->push_back(newdp);
312  } else {
313  edep = trec;
314  fParticleChange->ProposeNonIonizingEnergyDeposit(edep);
315  }
316 
317  // finelize primary energy and energy balance
318  // this threshold may be applied only because for low-enegry
319  // e+e- msc model is applied
320  if(finalT < 0.0) {
321  edep += finalT;
322  finalT = 0.0;
323  }
324  edep = std::max(edep, 0.0);
325  fParticleChange->SetProposedKineticEnergy(finalT);
326  fParticleChange->ProposeLocalEnergyDeposit(edep);
327 }
