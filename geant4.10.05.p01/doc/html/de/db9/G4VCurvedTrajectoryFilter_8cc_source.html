40 std::vector<G4ThreeVector>*
41 G4VCurvedTrajectoryFilter::GimmeThePointsAndForgetThem()
42 {
43  std::vector<G4ThreeVector>* tmp = fpFilteredPoints;
44  // ParticleChangeForTransport invokes this method (via
45  // PropagatorInField) at every Step, even if the step did not
46  // involve PropagatorInField. Must, therefore, ensure that points
47  // submitted by previous invocations of PIF are not
48  // copied. Therefore the points must be cleared. (Note that the
49  // responsibility for deleting the vector lies with the
50  // SmoothTrajctoryPoint, which is the vector's final destination.)
51  // (jacek 08/11/2002)
52  fpFilteredPoints = 0;
53  return tmp;
54 }
