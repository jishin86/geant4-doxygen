61 void G4EnergyRangeManager::RegisterMe(G4HadronicInteraction* a)
62 {
63  if(!a) { return; }
64  if(0 < theHadronicInteractionCounter) {
65  for(G4int i=0; i<theHadronicInteractionCounter; ++i) {
66  if(a == theHadronicInteraction[i]) { return; }
67  }
68  }
69  theHadronicInteraction.push_back(a);
70  ++theHadronicInteractionCounter;
71 }

74 G4HadronicInteraction*
75 G4EnergyRangeManager::GetHadronicInteraction(const G4HadProjectile & aHadProjectile,
76  G4Nucleus & aTargetNucleus,
77  const G4Material* aMaterial,
78  const G4Element* anElement) const
79 {
80  if(0 == theHadronicInteractionCounter) {
81  throw G4HadronicException(__FILE__, __LINE__,
82  "GetHadronicInteraction: NO MODELS STORED");
83  }
84 
85  G4double kineticEnergy = aHadProjectile.GetKineticEnergy();
86  // For ions, get kinetic energy per nucleon
87  if ( aHadProjectile.GetDefinition()->GetBaryonNumber() > 1.5 ) {
88  kineticEnergy /= aHadProjectile.GetDefinition()->GetBaryonNumber();
89  }
90 
91  G4int cou = 0, memory = 0, memor2 = 0;
92  G4double emi1 = 0.0, ema1 = 0.0, emi2 = 0.0, ema2 = 0.0;
93 
94  for (G4int i = 0; i<theHadronicInteractionCounter; ++i) {
95  if ( theHadronicInteraction[i]->IsApplicable( aHadProjectile, aTargetNucleus ) ) {
96  G4double low = theHadronicInteraction[i]->GetMinEnergy( aMaterial, anElement );
97  // Work-around for particles with 0 kinetic energy, which still
98  // require a model to return a ParticleChange
99  //if (low == 0.) low = -DBL_MIN;
100  G4double high = theHadronicInteraction[i]->GetMaxEnergy( aMaterial, anElement );
101  if (low <= kineticEnergy && high > kineticEnergy) {
102  ++cou;
103  emi2 = emi1;
104  ema2 = ema1;
105  emi1 = low;
106  ema1 = high;
107  memor2 = memory;
108  memory = i;
109  }
110  }
111  }
112 
113  G4int mem = -1;
114  G4double rand;
115  switch (cou) {
116  case 0:
117  G4cout<<"G4EnergyRangeManager:GetHadronicInteraction: counter="
118  <<theHadronicInteractionCounter<<", Ek="
119  <<kineticEnergy<<", Material = "<<aMaterial->GetName()
120  <<", Element = "
121  <<anElement->GetName()<<G4endl;
122  for( G4int j=0; j<theHadronicInteractionCounter; ++j)
123  {
124  G4HadronicInteraction* HInt=theHadronicInteraction[j];
125  G4cout<<"*"<<j<<"* low=" <<HInt->GetMinEnergy(aMaterial,anElement)
126  <<", high="<<HInt->GetMaxEnergy(aMaterial,anElement)<<G4endl;
127  }
128  throw G4HadronicException(__FILE__, __LINE__,
129  "GetHadronicInteraction: No Model found");
130  return 0;
131  case 1:
132  mem = memory;
133  break;
134  case 2:
135  if( (emi2<=emi1 && ema2>=ema1) || (emi2>=emi1 && ema2<=ema1) )
136  {
137  G4cout<<"G4EnergyRangeManager:GetHadronicInteraction: counter="
138  <<theHadronicInteractionCounter<<", Ek="
139  <<kineticEnergy<<", Material = "<<aMaterial->GetName()
140  <<", Element = "
141  <<anElement->GetName()<<G4endl;
142  for( G4int j=0; j<theHadronicInteractionCounter; ++j)
143  {
144  G4HadronicInteraction* HInt=theHadronicInteraction[j];
145  G4cout<<"*"<<j<<"* low=" <<HInt->GetMinEnergy(aMaterial,anElement)
146  <<", high="<<HInt->GetMaxEnergy(aMaterial,anElement)<<G4endl;
147  }
148  throw G4HadronicException(__FILE__, __LINE__,
149  "GetHadronicInteraction: Energy ranges of two models fully overlapping");
150  }
151  rand = G4UniformRand();
152  if( emi1 < emi2 )
153  {
154  if( (ema1-kineticEnergy) < rand*(ema1-emi2) ) {
155  mem = memor2;
156  } else {
157  mem = memory;
158  }
159  } else {
160  if( (ema2-kineticEnergy) < rand*(ema2-emi1) ) {
161  mem = memory;
162  } else {
163  mem = memor2;
164  }
165  }
166  break;
167  default:
168  throw G4HadronicException(__FILE__, __LINE__,
169  "GetHadronicInteraction: More than two competing models in this energy range");
170  }
171 
172  return theHadronicInteraction[mem];
173 }
