284  template<class T> void TLBE<T>::ConstructEM() {
285 
286  // models & processes:
287  // Use Livermore models up to 20 MeV, and standard
288  // models for higher energy
289  G4double LivermoreHighEnergyLimit = 20*CLHEP::MeV;
290  //
291  auto myParticleIterator=G4ParticleTable::GetParticleTable()->GetIterator();
292  myParticleIterator->reset();
293  while( (*(myParticleIterator))() ){
294  G4ParticleDefinition* particle = myParticleIterator->value();
295  G4ProcessManager* pmanager = particle->GetProcessManager();
296  G4String particleName = particle->GetParticleName();
297  G4String particleType = particle->GetParticleType();
298  G4double charge = particle->GetPDGCharge();
299 
300  if (particleName == "gamma")
301  {
302  G4PhotoElectricEffect* thePhotoElectricEffect = new G4PhotoElectricEffect();
303  G4LivermorePhotoElectricModel* theLivermorePhotoElectricModel =
304  new G4LivermorePhotoElectricModel();
305  theLivermorePhotoElectricModel->SetHighEnergyLimit(LivermoreHighEnergyLimit);
306  thePhotoElectricEffect->AddEmModel(0, theLivermorePhotoElectricModel);
307  pmanager->AddDiscreteProcess(thePhotoElectricEffect);
308 
309  G4ComptonScattering* theComptonScattering = new G4ComptonScattering();
310  G4LivermoreComptonModel* theLivermoreComptonModel =
311  new G4LivermoreComptonModel();
312  theLivermoreComptonModel->SetHighEnergyLimit(LivermoreHighEnergyLimit);
313  theComptonScattering->AddEmModel(0, theLivermoreComptonModel);
314  pmanager->AddDiscreteProcess(theComptonScattering);
315 
316  G4GammaConversion* theGammaConversion = new G4GammaConversion();
317  G4LivermoreGammaConversionModel* theLivermoreGammaConversionModel =
318  new G4LivermoreGammaConversionModel();
319  theLivermoreGammaConversionModel->SetHighEnergyLimit(LivermoreHighEnergyLimit);
320  theGammaConversion->AddEmModel(0, theLivermoreGammaConversionModel);
321  pmanager->AddDiscreteProcess(theGammaConversion);
322 
323  G4RayleighScattering* theRayleigh = new G4RayleighScattering();
324  G4LivermoreRayleighModel* theRayleighModel = new G4LivermoreRayleighModel();
325  theRayleighModel->SetHighEnergyLimit(LivermoreHighEnergyLimit);
326  theRayleigh->AddEmModel(0, theRayleighModel);
327  pmanager->AddDiscreteProcess(theRayleigh);
328 
329  }
330  else if (particleName == "e-")
331  {
332  //electron
333  // process ordering: AddProcess(name, at rest, along step, post step)
334  // -1 = not implemented, then ordering
335  G4eMultipleScattering* msc = new G4eMultipleScattering();
336  //msc->AddEmModel(0, new G4UrbanMscModel());
337  msc->SetStepLimitType(fUseDistanceToBoundary);
338  pmanager->AddProcess(msc, -1, 1, 1);
339 
340  // Ionisation
341  G4eIonisation* eIoni = new G4eIonisation();
342  G4LivermoreIonisationModel* theIoniLivermore = new
343  G4LivermoreIonisationModel();
344  theIoniLivermore->SetHighEnergyLimit(1*CLHEP::MeV);
345  eIoni->AddEmModel(0, theIoniLivermore, new G4UniversalFluctuation() );
346  eIoni->SetStepFunction(0.2, 100*CLHEP::um); //
347  pmanager->AddProcess(eIoni, -1, 2, 2);
348 
349  // Bremsstrahlung
350  G4eBremsstrahlung* eBrem = new G4eBremsstrahlung();
351  G4LivermoreBremsstrahlungModel* theBremLivermore = new
352  G4LivermoreBremsstrahlungModel();
353  theBremLivermore->SetHighEnergyLimit(LivermoreHighEnergyLimit);
354  eBrem->AddEmModel(0, theBremLivermore);
355  pmanager->AddProcess(eBrem, -1,-3, 3);
356  }
357  else if (particleName == "e+")
358  {
359  //positron
360  G4eMultipleScattering* msc = new G4eMultipleScattering();
361  //msc->AddEmModel(0, new G4UrbanMscModel());
362  msc->SetStepLimitType(fUseDistanceToBoundary);
363  pmanager->AddProcess(msc, -1, 1, 1);
364  G4eIonisation* eIoni = new G4eIonisation();
365  eIoni->SetStepFunction(0.2, 100*CLHEP::um);
366  pmanager->AddProcess(eIoni, -1, 2, 2);
367  pmanager->AddProcess(new G4eBremsstrahlung, -1,-3, 3);
368  pmanager->AddProcess(new G4eplusAnnihilation,0,-1, 4);
369  }
370  else if( particleName == "mu+" ||
371  particleName == "mu-" )
372  {
373  //muon
374  G4MuMultipleScattering* aMultipleScattering = new G4MuMultipleScattering();
375  pmanager->AddProcess(aMultipleScattering, -1, 1, 1);
376  pmanager->AddProcess(new G4MuIonisation(), -1, 2, 2);
377  pmanager->AddProcess(new G4MuBremsstrahlung(), -1,-1, 3);
378  pmanager->AddProcess(new G4MuPairProduction(), -1,-1, 4);
379  if( particleName == "mu-" )
380  pmanager->AddProcess(new G4MuonMinusCapture(), 0,-1,-1);
381  }
382  else if (particleName == "GenericIon")
383  {
384  pmanager->AddProcess(new G4hMultipleScattering, -1, 1, 1);
385  G4ionIonisation* ionIoni = new G4ionIonisation();
386  ionIoni->SetEmModel(new G4IonParametrisedLossModel());
387  ionIoni->SetStepFunction(0.1, 10*CLHEP::um);
388  pmanager->AddProcess(ionIoni, -1, 2, 2);
389  pmanager->AddProcess(new G4NuclearStopping(), -1, 3,-1);
390  }
391  else if (particleName == "alpha" || particleName == "He3")
392  {
393  //MSC, ion-Ionisation, Nuclear Stopping
394  pmanager->AddProcess(new G4hMultipleScattering, -1, 1, 1);
395 
396  G4ionIonisation* ionIoni = new G4ionIonisation();
397  ionIoni->SetStepFunction(0.1, 20*CLHEP::um);
398  pmanager->AddProcess(ionIoni, -1, 2, 2);
399  pmanager->AddProcess(new G4NuclearStopping(), -1, 3,-1);
400  }
401  else if (particleName == "proton" ||
402  particleName == "deuteron" ||
403  particleName == "triton" ||
404  particleName == "pi+" ||
405  particleName == "pi-" ||
406  particleName == "kaon+" ||
407  particleName == "kaon-")
408  {
409  //MSC, h-ionisation, bremsstrahlung
410  pmanager->AddProcess(new G4hMultipleScattering, -1, 1, 1);
411  G4hIonisation* hIoni = new G4hIonisation();
412  hIoni->SetStepFunction(0.2, 50*CLHEP::um);
413  pmanager->AddProcess(hIoni, -1, 2, 2);
414  pmanager->AddProcess(new G4hBremsstrahlung, -1,-3, 3);
415  }
416  else if ((!particle->IsShortLived()) &&
417  (charge != 0.0) &&
418  (particle->GetParticleName() != "chargedgeantino"))
419  {
420  //all others charged particles except geantino
421  pmanager->AddProcess(new G4hMultipleScattering, -1, 1, 1);
422  pmanager->AddProcess(new G4hIonisation, -1, 2, 2);
423  }
424 
425  }
426 }

435  template<class T> void TLBE<T>::ConstructOp()
436 {
437  // default scintillation process
438  //Coverity report: check that the process is actually used, if not must delete
439  G4bool theScintProcessDefNeverUsed = true;
440  G4Scintillation* theScintProcessDef = new G4Scintillation("Scintillation");
441  // theScintProcessDef->DumpPhysicsTable();
442  theScintProcessDef->SetTrackSecondariesFirst(true);
443  theScintProcessDef->SetScintillationYieldFactor(1.0); //
444  theScintProcessDef->SetScintillationExcitationRatio(0.0); //
445  theScintProcessDef->SetVerboseLevel(OpVerbLevel);
446 
447  // scintillation process for alpha:
448  G4bool theScintProcessAlphaNeverUsed = true;
449  G4Scintillation* theScintProcessAlpha = new G4Scintillation("Scintillation");
450  // theScintProcessNuc->DumpPhysicsTable();
451  theScintProcessAlpha->SetTrackSecondariesFirst(true);
452  theScintProcessAlpha->SetScintillationYieldFactor(1.1);
453  theScintProcessAlpha->SetScintillationExcitationRatio(1.0);
454  theScintProcessAlpha->SetVerboseLevel(OpVerbLevel);
455 
456  // scintillation process for heavy nuclei
457  G4bool theScintProcessNucNeverUsed = true;
458  G4Scintillation* theScintProcessNuc = new G4Scintillation("Scintillation");
459  // theScintProcessNuc->DumpPhysicsTable();
460  theScintProcessNuc->SetTrackSecondariesFirst(true);
461  theScintProcessNuc->SetScintillationYieldFactor(0.2);
462  theScintProcessNuc->SetScintillationExcitationRatio(1.0);
463  theScintProcessNuc->SetVerboseLevel(OpVerbLevel);
464 
465  // optical processes
466  G4bool theAbsorptionProcessNeverUsed = true;
467  G4OpAbsorption* theAbsorptionProcess = new G4OpAbsorption();
468  // G4OpRayleigh* theRayleighScatteringProcess = new G4OpRayleigh();
469  G4bool theBoundaryProcessNeverUsed = true;
470  G4OpBoundaryProcess* theBoundaryProcess = new G4OpBoundaryProcess();
471  // theAbsorptionProcess->DumpPhysicsTable();
472  // theRayleighScatteringProcess->DumpPhysicsTable();
473  theAbsorptionProcess->SetVerboseLevel(OpVerbLevel);
474  // theRayleighScatteringProcess->SetVerboseLevel(OpVerbLevel);
475  theBoundaryProcess->SetVerboseLevel(OpVerbLevel);
476 
477  auto myParticleIterator=G4ParticleTable::GetParticleTable()->GetIterator();
478  myParticleIterator->reset();
479  while( (*(myParticleIterator))() )
480  {
481  G4ParticleDefinition* particle = myParticleIterator->value();
482  G4ProcessManager* pmanager = particle->GetProcessManager();
483  G4String particleName = particle->GetParticleName();
484  if (theScintProcessDef->IsApplicable(*particle)) {
485  // if(particle->GetPDGMass() > 5.0*CLHEP::GeV)
486  if(particle->GetParticleName() == "GenericIon") {
487  pmanager->AddProcess(theScintProcessNuc); // AtRestDiscrete
488  pmanager->SetProcessOrderingToLast(theScintProcessNuc,idxAtRest);
489  pmanager->SetProcessOrderingToLast(theScintProcessNuc,idxPostStep);
490  theScintProcessNucNeverUsed = false;
491  }
492  else if(particle->GetParticleName() == "alpha") {
493  pmanager->AddProcess(theScintProcessAlpha);
494  pmanager->SetProcessOrderingToLast(theScintProcessAlpha,idxAtRest);
495  pmanager->SetProcessOrderingToLast(theScintProcessAlpha,idxPostStep);
496  theScintProcessAlphaNeverUsed = false;
497  }
498  else {
499  pmanager->AddProcess(theScintProcessDef);
500  pmanager->SetProcessOrderingToLast(theScintProcessDef,idxAtRest);
501  pmanager->SetProcessOrderingToLast(theScintProcessDef,idxPostStep);
502  theScintProcessDefNeverUsed = false;
503  }
504  }
505 
506  if (particleName == "opticalphoton") {
507  pmanager->AddDiscreteProcess(theAbsorptionProcess);
508  theAbsorptionProcessNeverUsed = false;
509  // pmanager->AddDiscreteProcess(theRayleighScatteringProcess);
510  theBoundaryProcessNeverUsed = false;
511  pmanager->AddDiscreteProcess(theBoundaryProcess);
512  }
513  }
514  if ( theScintProcessDefNeverUsed ) delete theScintProcessDef;
515  if ( theScintProcessAlphaNeverUsed ) delete theScintProcessAlpha;
516  if ( theScintProcessNucNeverUsed ) delete theScintProcessNuc;
517  if ( theBoundaryProcessNeverUsed ) delete theBoundaryProcess;
518  if ( theAbsorptionProcessNeverUsed ) delete theAbsorptionProcess;
519 }

588 // to those particles with GHEISHA interactions (INTRC > 0).

896  template<class T> void TLBE<T>::ConstructGeneral() {
897 
898  // Add Decay Process
899  G4Decay* theDecayProcess = new G4Decay();
900  G4bool theDecayProcessNeverUsed = true; //Check if theDecayProcess will be used
901  auto myParticleIterator=G4ParticleTable::GetParticleTable()->GetIterator();
902  myParticleIterator->reset();
903  while( (*(myParticleIterator))() )
904  {
905  G4ParticleDefinition* particle = myParticleIterator->value();
906  G4ProcessManager* pmanager = particle->GetProcessManager();
907 
908  if (theDecayProcess->IsApplicable(*particle) && !particle->IsShortLived())
909  {
910  theDecayProcessNeverUsed = false;
911  pmanager ->AddProcess(theDecayProcess);
912  // set ordering for PostStepDoIt and AtRestDoIt
913  pmanager ->SetProcessOrdering(theDecayProcess, idxPostStep);
914  pmanager ->SetProcessOrdering(theDecayProcess, idxAtRest);
915  }
916  }
917 
918  // Declare radioactive decay to the GenericIon in the IonTable.
919  const G4IonTable *theIonTable =
920  G4ParticleTable::GetParticleTable()->GetIonTable();
921  G4RadioactiveDecay *theRadioactiveDecay = new G4RadioactiveDecay();
922 
923  //Fix for activation of RadioactiveDecay, based on G4RadioactiveDecayPhysics
924  G4EmParameters* param = G4EmParameters::Instance();
925  param->SetAugerCascade(true);
926  param->AddPhysics("world","G4RadioactiveDecay");
927 
928  G4DeexPrecoParameters* deex = G4NuclearLevelData::GetInstance()->GetParameters();
929  deex->SetStoreAllLevels(true);
930  deex->SetMaxLifeTime(G4NuclideTable::GetInstance()->GetThresholdOfHalfLife()
931  /std::log(2.));
932 
933  G4LossTableManager* man = G4LossTableManager::Instance();
934  G4VAtomDeexcitation* ad = man->AtomDeexcitation();
935  if(!ad) {
936  ad = new G4UAtomicDeexcitation();
937  man->SetAtomDeexcitation(ad);
938  ad->InitialiseAtomicDeexcitation();
939  }
940 
941  for (G4int i=0; i<theIonTable->Entries(); i++)
942  {
943  G4String particleName = theIonTable->GetParticle(i)->GetParticleName();
944  G4String particleType = theIonTable->GetParticle(i)->GetParticleType();
945 
946  if (particleName == "GenericIon")
947  {
948  G4ProcessManager* pmanager =
949  theIonTable->GetParticle(i)->GetProcessManager();
950  pmanager->SetVerboseLevel(VerboseLevel);
951  pmanager ->AddProcess(theRadioactiveDecay);
952  pmanager ->SetProcessOrdering(theRadioactiveDecay, idxPostStep);
953  pmanager ->SetProcessOrdering(theRadioactiveDecay, idxAtRest);
954  }
955  }
956  //If we actually never used the process, delete it
