74 void SteppingAction::UserSteppingAction(const G4Step* theStep)
75 {
76  if(theStep->GetPostStepPoint()->GetProcessDefinedStep()->GetProcessName()!=
77  "Transportation")
78  {
79  // Get position and edep of current step
80  //
81  G4double x = theStep->GetPreStepPoint()->GetPosition().x()/nanometer;
82  G4double y = theStep->GetPreStepPoint()->GetPosition().y()/nanometer;
83  G4double z = theStep->GetPreStepPoint()->GetPosition().z()/nanometer;
84  G4double edepStep = theStep->GetTotalEnergyDeposit()/eV;
85 
86  G4LogicalVolume* targetVolume =
87  G4LogicalVolumeStore::GetInstance()->GetVolume("BoundingLV");
88  G4LogicalVolume* theVolume =
89  theStep->GetPreStepPoint()->GetPhysicalVolume()->GetLogicalVolume();
90 
91  if ((edepStep > 0.) && (theVolume==targetVolume))
92  {
93  // Add edep to this event
94  //
95  fpEventAction->AddEdepEvent(edepStep);
96  if (fpDetector->GetBarycenterList()==NULL)
97  {
98  G4cout << "Barycenter list is null!!!" << G4endl;
99  }
100  else
101  {
102  CheckAndProcessDNAHit(x,y,z,edepStep);
103  }
104  }
105  }
106 }

110 G4bool SteppingAction::CheckAndProcessDNAHit(G4double x,G4double y, G4double z,
111  G4double edepStep)
112 {
113  int numStrand=0;
114  int numNucl=0;
115  int intResidue=-1; // 0 for Phospat, 1 for Sugar, 2 for Base
116  unsigned short int hit = (fpDetector->GetPDBlib()).ComputeMatchEdepDNA(
117  fpDetector->GetBarycenterList(),
118  fpDetector->GetMoleculeList(),
119  x*10., y*10., z*10.,// x10 => angstrom<->nm
120  numStrand, numNucl, intResidue);
121 
122  if (hit==1)
123  {
124  if ((intResidue==0)||(intResidue==1)) //Edep in Phosphate or Sugar
125  {
126  fpEventAction->AddEdepToNucleotide(numStrand,numNucl,edepStep);
127  return true;
128  }
129  else
130  {
131  return false;
132  }
133  }
134  else
135  {
136  return false;
137  }
