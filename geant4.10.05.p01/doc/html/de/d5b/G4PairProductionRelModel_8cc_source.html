167 G4double G4PairProductionRelModel::ComputeXSectionPerAtom(G4double gammaEnergy,
168  G4double Z)
169 {
170  G4double xSection = 0.0;
171  // check if LPM suppression needs to be used
172  const G4bool isLPM = (fIsUseLPMCorrection && gammaEnergy>gEgLPMActivation);
173  // determine the kinematical limits (taken into account the correction due to
174  // the way in which the Coulomb correction is applied i.e. avoid negative DCS)
175  const G4int iz = std::min(gMaxZet, G4lrint(Z));
176  const G4double eps0 = CLHEP::electron_mass_c2/gammaEnergy;
177  const G4double dmax = gElementData[iz]->fDeltaMax;
178  const G4double dmin = 4.*eps0*gElementData[iz]->fDeltaFactor;
179  const G4double eps1 = 0.5 - 0.5*std::sqrt(1.-dmin/dmax);
180  const G4double epsMin = std::max(eps0, eps1);
181  const G4double epsMax = 0.5; // DCS is symmetric around eps=0.5
182  // let Et be the total energy transferred to the e- or to the e+
183  // the [Et-min, Et-max] interval will be divided into i=1,2,..,n subintervals
184  // with width of dInterv = (Et-max - Et-min)/n and numerical integration will
185  // be done in each sub-inteval using the xi = (Et - Et_i-min)/dInterv variable
186  // that is in [0,1]. The 8-point GL q. is used for the integration on [0,1].
187  const G4int numSub = 2;
188  const G4double dInterv= (epsMax - epsMin)*gammaEnergy/G4double(numSub);
189  G4double minEti = epsMin*gammaEnergy; // Et-min i.e. Et_0-min
190  for (G4int i = 0; i < numSub; ++i) {
191  for (G4int ngl = 0; ngl < 8; ++ngl) {
192  const G4double Et = (minEti + gXGL[ngl]*dInterv);
193  const G4double xs = isLPM ? ComputeRelDXSectionPerAtom(Et, gammaEnergy, Z)
194  : ComputeDXSectionPerAtom(Et, gammaEnergy, Z);
195  xSection += gWGL[ngl]*xs;
196  }
197  // update minimum Et of the sub-inteval
198  minEti += dInterv;
199  }
200  // apply corrections of variable transformation and half interval integration
201  xSection = std::max(2.*xSection*dInterv, 0.);
202  return xSection;
203 }

205 // DCS WITHOUT LPM SUPPRESSION
206 // Computes DCS value for a given target element (Z), initial gamma energy (Eg),
207 // total energy transferred to one of the e-/e+ pair(Et) WITHOUT LPM suppression
208 // The constant factor 4 \alpha r_0^2 Z (Z +\eta(Z)) is not included here and
209 // the returned value will be differential in total energy transfer instead of
210 // the eps=Et/Eg. The computed part of the DCS
211 // NORMAL CASE: DEFAULT STTING (i.e. fIsUseCompleteScreening = FALSE)
212 // ds/deps(Et,Eg,Z) = ds/deps(eps,Z) = (eps^2+(1-eps)^2)*[phi1(d)/4-ln(Z)/3-fc]
213 // + 2*eps(1-eps)*[phi2(d)/4-ln(Z)/3-fc]/3 where the universal (in the TF model)
214 // screening variable d=d(eps)=136Z^(-1/3)eps0/[eps*(1-eps)] with eps0=mc^2/Eg.
215 // COMPLETE SCREENING (when d(eps) approx-equal-to 0) : NEED TO BE SET BY USER
216 // ds/deps(Et,Eg,Z) = ds/deps(eps,Z) = (eps^2+(1-eps)^2+eps*(1-eps)/3)*[Lel-fc]
217 // -eps(1-eps)/9 where Lel=phi1(0)/4-ln(Z)/3 is the elastic(coherent) radiation
218 // logarithm, fc is the Coulomb correction and the relation phi2(0)/4-ln(Z)/3 =
219 // phi1(0)/4-1/6-ln(Z)/3 = Lel-1/6 (due to phi2(0)=phi1(0)-2/3) was used.
220 G4double G4PairProductionRelModel::ComputeDXSectionPerAtom(G4double pEnergy,
221  G4double gammaEnergy,
222  G4double Z)
223 {
224  G4double xSection = 0.;
225  const G4int iz = std::min(gMaxZet, G4lrint(Z));
226  const G4double eps = pEnergy/gammaEnergy;
227  const G4double epsm = 1.-eps;
228  const G4double dum = eps*epsm;
229  if (fIsUseCompleteScreening) {
230  // complete screening:
231  const G4double Lel = gElementData[iz]->fLradEl;
232  const G4double fc = gElementData[iz]->fCoulomb;
233  xSection = (eps*eps + epsm*epsm + 2.*dum/3.)*(Lel-fc) - dum/9.;
234  } else {
235  // normal case:
236  const G4double eps0 = CLHEP::electron_mass_c2/gammaEnergy;
237  const G4double fc = gElementData[iz]->fCoulomb;
238  const G4double lnZ13 = gElementData[iz]->fLogZ13;
239  const G4double delta = gElementData[iz]->fDeltaFactor*eps0/dum;
240  G4double phi1, phi2;
241  ComputePhi12(delta, phi1, phi2);
242  xSection = (eps*eps + epsm*epsm)*(0.25*phi1-lnZ13-fc)
243  + 2.*dum*(0.25*phi2-lnZ13-fc)/3.;
244  }
245  // non-const. part of the DCS differential in total energy transfer not in eps
246  // ds/dEt=ds/deps deps/dEt with deps/dEt=1/Eg
247  return std::max(xSection, 0.0)/gammaEnergy;
248 }

250 // DCS WITH POSSIBLE LPM SUPPRESSION
251 // Computes DCS value for a given target element (Z), initial gamma energy (Eg),
252 // total energy transferred to one of the e-/e+ pair(Et) WITH LPM suppression.
253 // For a given Z, the LPM suppression will depend on the material through the
254 // LMP-Energy. This will determine the suppression variable s and the LPM sup-
255 // pression functions xi(s), fi(s) and G(s).
256 // The constant factor 4 \alpha r_0^2 Z (Z +\eta(Z)) is not included here and
257 // the returned value will be differential in total energy transfer instead of
258 // the eps=Et/Eg. The computed part of the DCS
259 // NORMAL CASE: DEFAULT STTING (i.e. fIsUseCompleteScreening = FALSE)
260 // ds/deps(Et,Eg,Z)=ds/deps(eps,Z) = xi(s)*{ (eps^2+(1-eps)^2)*[2fi(s)/3+G(s)/3]
261 // *[phi1(d)/4-ln(Z)/3-fc] + 2*eps(1-eps)*G(s)*[phi2(d)/4-ln(Z)/3-fc]/3 } where
262 // the universal (in the TF model) screening variable d=d(eps)=136Z^(-1/3)eps0
263 // /[eps*(1-eps)] with eps0=mc^2/Eg.
264 // COMPLETE SCREENING (when d(eps) approx-equal-to 0) : NEED TO BE SET BY USER
265 // ds/deps(Et,Eg,Z) = ds/deps(eps,Z) = xi(s)*{ [Lel-fc]*[ (eps^2+(1-eps)^2+eps
266 // *(1-eps)/3)*2fi(s)/3 + G(s)/3] - eps(1-eps)*G(s)/9 }
267 // Note, that when the LPM suppression is absent i.e. xi(s)=fi(s)=G(s)=1, both
268 // the normal and the complete screening DCS give back the NO-LMP case above.
269 G4double G4PairProductionRelModel::ComputeRelDXSectionPerAtom(G4double pEnergy,
270  G4double gammaEnergy,
271  G4double Z)
272 {
273  G4double xSection = 0.;
274  const G4int iz = std::min(gMaxZet, G4lrint(Z));
275  const G4double eps = pEnergy/gammaEnergy;
276  const G4double epsm = 1.-eps;
277  const G4double dum = eps*epsm;
278  // evaluate LPM suppression functions
279  G4double fXiS, fGS, fPhiS;
280  ComputeLPMfunctions(fXiS, fGS, fPhiS, eps, gammaEnergy, iz);
281  if (fIsUseCompleteScreening) {
282  // complete screening:
283  const G4double Lel = gElementData[iz]->fLradEl;
284  const G4double fc = gElementData[iz]->fCoulomb;
285  xSection = (Lel-fc)*((eps*eps+epsm*epsm)*2.*fPhiS + fGS)/3. - dum*fGS/9.;
286  } else {
287  // normal case:
288  const G4double eps0 = CLHEP::electron_mass_c2/gammaEnergy;
289  const G4double fc = gElementData[iz]->fCoulomb;
290  const G4double lnZ13 = gElementData[iz]->fLogZ13;
291  const G4double delta = gElementData[iz]->fDeltaFactor*eps0/dum;
292  G4double phi1, phi2;
293  ComputePhi12(delta, phi1, phi2);
294  xSection = (eps*eps + epsm*epsm)*(2.*fPhiS+fGS)*(0.25*phi1-lnZ13-fc)/3.
295  + 2.*dum*fGS*(0.25*phi2-lnZ13-fc)/3.;
296  }
297  // non-const. part of the DCS differential in total energy transfer not in eps
298  // ds/dEt=ds/deps deps/dEt with deps/dEt=1/Eg
299  return std::max(fXiS*xSection, 0.0)/gammaEnergy;
300 }

302 G4double
303 G4PairProductionRelModel::ComputeCrossSectionPerAtom(const G4ParticleDefinition*,
304  G4double gammaEnergy, G4double Z, G4double, G4double, G4double)
305 {
306  G4double crossSection = 0.0 ;
307  // check kinematical limit
308  if ( gammaEnergy <= 2.0*electron_mass_c2 ) { return crossSection; }
309  // Computes the cross section with or without LPM suppression depending on
310  // settings (by default with if the gamma energy is above a given threshold)
311  // and using or not using complete sreening approximation (by default not).
312  // Only the dependent part is computed in the numerical integration of the DCS
313  // i.e. the result must be multiplied here with 4 \alpha r_0^2 Z(Z+\eta(Z))
314  crossSection = ComputeXSectionPerAtom(gammaEnergy, Z);
315  // apply the constant factors:
316  // - eta(Z) is a correction to account interaction in the field of e-
317  // - gXSecFactor = 4 \alpha r_0^2
318  const G4int iz = std::min(gMaxZet, G4lrint(Z));
319  const G4double eta = gElementData[iz]->fEtaValue;
320  crossSection *= gXSecFactor*Z*(Z+eta);
321  // final protection
322  return std::max(crossSection, 0.);
323 }

331 void
332 G4PairProductionRelModel::SampleSecondaries(std::vector<G4DynamicParticle*>* fvect,
333  const G4MaterialCutsCouple* couple,
334  const G4DynamicParticle* aDynamicGamma,
335  G4double,
336  G4double)
337 // The secondaries e+e- energies are sampled using the Bethe - Heitler
338 // cross sections with Coulomb correction.
339 // A modified version of the random number techniques of Butcher & Messel
340 // is used (Nuc Phys 20(1960),15).
341 //
342 // GEANT4 internal units.
343 //
344 // Note 1 : Effects due to the breakdown of the Born approximation at
345 // low energy are ignored.
346 // Note 2 : The differential cross section implicitly takes account of
347 // pair creation in both nuclear and atomic electron fields.
348 // However triplet prodution is not generated.
349 {
350  const G4Material* mat = couple->GetMaterial();
351  const G4double gammaEnergy = aDynamicGamma->GetKineticEnergy();
352  const G4double eps0 = CLHEP::electron_mass_c2/gammaEnergy ;
353  //
354  // check kinematical limit: gamma energy(Eg) must be at least 2 e- rest mass
355  // (but the model should be used at higher energies above 100 MeV)
356  if (eps0 > 0.5) { return; }
357  //
358  // select target atom of the material
359  const G4Element* anElement = SelectRandomAtom(couple, fTheGamma, gammaEnergy);
360  CLHEP::HepRandomEngine* rndmEngine = G4Random::getTheEngine();
361  //
362  // 'eps' is the total energy transferred to one of the e-/e+ pair in initial
363  // gamma energy units Eg. Since the corresponding DCS is symmetric on eps=0.5,
364  // the kinematical limits for eps0=mc^2/Eg <= eps <= 0.5
365  //
366  // The Coulomb factor for the target element (Z) (Eg>50 MeV is assumed)
367  // F(Z) = 8*ln(Z)/3 + 8*fc(Z)
368  //
369  // The screening variable 'delta(eps)' = 136*Z^{-1/3}*eps0/[eps(1-eps)]
370  // Due to the Coulomb correction, the DCS can go below zero even at
371  // kinematicaly allowed eps > eps0 values. In order to exclude this eps
372  // range with negative DCS, the minimum eps value will be set to eps_min =
373  // max[eps0, epsp] with epsp is the solution of SF(delta(epsp)) - F(Z)/2 = 0
374  // with SF being the screening function (SF1=SF2 at high value of delta).
375  // The solution is epsp = 0.5 - 0.5*sqrt[ 1 - 4*136*Z^{-1/3}eps0/deltap]
376  // with deltap = Exp[(42.038-F(Z))/8.29]-0.958. So the limits are:
377  // - when eps=eps_max = 0.5 => delta_min = 136*Z^{-1/3}*eps0/4
378  // - epsp = 0.5 - 0.5*sqrt[ 1 - delta_min/deltap]
379  // - and eps_min = max[eps0, epsp]
380  const G4int iZet = std::min(gMaxZet, anElement->GetZasInt());
381  const G4double deltaFactor = gElementData[iZet]->fDeltaFactor*eps0;
382  const G4double deltaMin = 4.*deltaFactor;
383  const G4double deltaMax = gElementData[iZet]->fDeltaMax;
384  // compute the limits of eps
385  const G4double epsp = 0.5 - 0.5*std::sqrt(1. - deltaMin/deltaMax) ;
386  const G4double epsMin = std::max(eps0,epsp);
387  const G4double epsRange = 0.5 - epsMin;
388  const G4double FZ = 8.*(gElementData[iZet]->fLogZ13 +
389  gElementData[iZet]->fCoulomb);
390  //
391  // sample the energy rate (eps) of the created electron (or positron)
392  G4double F10, F20;
393  ScreenFunction12(deltaMin, F10, F20);
394  F10 -= FZ;
395  F20 -= FZ;
396  const G4double NormF1 = std::max(F10 * epsRange * epsRange, 0.);
397  const G4double NormF2 = std::max(1.5 * F20 , 0.);
398  const G4double NormCond = NormF1/(NormF1 + NormF2);
399  // check if LPM correction is active
400  const G4bool isLPM = (fIsUseLPMCorrection && gammaEnergy>gEgLPMActivation);
401  fLPMEnergy = mat->GetRadlen()*gLPMconstant;
402  // we will need 3 uniform random number for each trial of sampling
403  G4double rndmv[3];
404  G4double greject = 0.;
405  G4double eps;
406  do {
407  rndmEngine->flatArray(3, rndmv);
408  if (NormCond > rndmv[0]) {
409  eps = 0.5 - epsRange * fG4Calc->A13(rndmv[1]);
410  const G4double delta = deltaFactor/(eps*(1.-eps));
411  if (isLPM) {
412  G4double lpmXiS, lpmGS, lpmPhiS, phi1, phi2;
413  ComputePhi12(delta, phi1, phi2);
414  ComputeLPMfunctions(lpmXiS, lpmGS, lpmPhiS, eps, gammaEnergy, iZet);
415  greject = lpmXiS*((2.*lpmPhiS+lpmGS)*phi1-lpmGS*phi2-lpmPhiS*FZ)/F10;
416  } else {
417  greject = (ScreenFunction1(delta)-FZ)/F10;
418  }
419  } else {
420  eps = epsMin + epsRange*rndmv[1];
421  const G4double delta = deltaFactor/(eps*(1.-eps));
422  if (isLPM) {
423  G4double lpmXiS, lpmGS, lpmPhiS, phi1, phi2;
424  ComputePhi12(delta, phi1, phi2);
425  ComputeLPMfunctions(lpmXiS, lpmGS, lpmPhiS, eps, gammaEnergy, iZet);
426  greject = lpmXiS*( (lpmPhiS+0.5*lpmGS)*phi1 + 0.5*lpmGS*phi2
427  -0.5*(lpmGS+lpmPhiS)*FZ )/F20;
428  } else {
429  greject = (ScreenFunction2(delta)-FZ)/F20;
430  }
431  }
432  // Loop checking, 03-Aug-2015, Vladimir Ivanchenko
433  } while (greject < rndmv[2]);
434  // end of eps sampling
435  //
436  // select charges randomly
437  G4double eTotEnergy, pTotEnergy;
438  if (rndmEngine->flat() > 0.5) {
439  eTotEnergy = (1.-eps)*gammaEnergy;
440  pTotEnergy = eps*gammaEnergy;
441  } else {
442  pTotEnergy = (1.-eps)*gammaEnergy;
443  eTotEnergy = eps*gammaEnergy;
444  }
445  //
446  // sample pair kinematics
447  //
448  const G4double eKinEnergy = std::max(0.,eTotEnergy - CLHEP::electron_mass_c2);
449  const G4double pKinEnergy = std::max(0.,pTotEnergy - CLHEP::electron_mass_c2);
450  //
451  G4ThreeVector eDirection, pDirection;
452  //
453  GetAngularDistribution()->SamplePairDirections(aDynamicGamma,
454  eKinEnergy, pKinEnergy,
455  eDirection, pDirection);
456  // create G4DynamicParticle object for the particle1
457  G4DynamicParticle* aParticle1= new G4DynamicParticle(
458  fTheElectron,eDirection,eKinEnergy);
459 
460  // create G4DynamicParticle object for the particle2
461  G4DynamicParticle* aParticle2= new G4DynamicParticle(

471 // should be called only by the master and at initialisation
472 void G4PairProductionRelModel::InitialiseElementData()
473 {
474  G4int size = gElementData.size();
475  if (size < gMaxZet+1) {
476  gElementData.resize(gMaxZet+1, nullptr);
477  }
478  // create for all elements that are in the detector
479  const G4ElementTable* elemTable = G4Element::GetElementTable();
480  size_t numElems = (*elemTable).size();
481  for (size_t ie = 0; ie < numElems; ++ie) {
482  const G4Element* elem = (*elemTable)[ie];
483  const G4int iz = std::min(gMaxZet, elem->GetZasInt());
484  if (!gElementData[iz]) { // create it if doesn't exist yet
485  const G4double logZ13 = elem->GetIonisation()->GetlogZ3();
486  const G4double Z13 = elem->GetIonisation()->GetZ3();
487  const G4double fc = elem->GetfCoulomb();
488  const G4double FZ = 8.*(logZ13 + fc);
489  G4double Fel;
490  G4double Finel;
491  if (iz<5) { // use data from Dirac-Fock atomic model
492  Fel = gFelLowZet[iz];
493  Finel = gFinelLowZet[iz];
494  } else { // use the results of the Thomas-Fermi-Moliere model
495  Fel = G4Log(184.) - logZ13;
496  Finel = G4Log(1194.) - 2.*logZ13;
497  }
498  ElementData* elD = new ElementData();
499  elD->fLogZ13 = logZ13;
500  elD->fCoulomb = fc;
501  elD->fLradEl = Fel;
502  elD->fDeltaFactor = 136./Z13;
503  elD->fDeltaMax = G4Exp((42.038 - FZ)/8.29) - 0.958;
504  elD->fEtaValue = Finel/(Fel-fc);
505  elD->fLPMVarS1Cond = std::sqrt(2.)*Z13*Z13/(184.*184.);
506  elD->fLPMILVarS1Cond = 1./G4Log(elD->fLPMVarS1Cond);
507  gElementData[iz] = elD;
508  }
509  }

526 // used only at initialisation time
527 void G4PairProductionRelModel::ComputeLPMGsPhis(G4double &funcGS, G4double &funcPhiS, const G4double varShat) {
528  if (varShat < 0.01) {
529  funcPhiS = 6.0*varShat*(1.0-CLHEP::pi*varShat);
530  funcGS = 12.0*varShat-2.0*funcPhiS;
531  } else {
532  const G4double varShat2 = varShat*varShat;
533  const G4double varShat3 = varShat*varShat2;
534  const G4double varShat4 = varShat2*varShat2;
535  if (varShat < 0.415827397755) { // Stanev ap.: for \psi(s) and compute G(s)
536  funcPhiS = 1.0-G4Exp( -6.0*varShat*(1.0+varShat*(3.0-CLHEP::pi))
537  + varShat3/(0.623+0.796*varShat+0.658*varShat2));
538  // 1-\exp \left\{-4s-\frac{8s^2}{1+3.936s+4.97s^2-0.05s^3+7.5s^4} \right\}
539  const G4double funcPsiS = 1.0-G4Exp( -4.0*varShat - 8.0*varShat2/(1.0
540  + 3.936*varShat+4.97*varShat2-0.05*varShat3+7.5*varShat4));
541  // G(s) = 3 \psi(s) - 2 \phi(s)
542  funcGS = 3.0*funcPsiS - 2.0*funcPhiS;
543  } else if (varShat < 1.55) {
544  funcPhiS = 1.0-G4Exp( -6.0*varShat*(1.0+varShat*(3.0-CLHEP::pi))
545  + varShat3/(0.623+0.796*varShat+0.658*varShat2));
546  const G4double dum0 = -0.16072300849123999+3.7550300067531581*varShat
547  -1.7981383069010097 *varShat2
548  +0.67282686077812381*varShat3
549  -0.1207722909879257 *varShat4;
550  funcGS = std::tanh(dum0);
551  } else {
552  funcPhiS = 1.0-0.01190476/varShat4;
553  if (varShat < 1.9156) {
554  const G4double dum0 = -0.16072300849123999+3.7550300067531581*varShat
555  -1.7981383069010097 *varShat2
556  +0.67282686077812381*varShat3
557  -0.1207722909879257 *varShat4;
558  funcGS = std::tanh(dum0);
559  } else {
560  funcGS = 1.0-0.0230655/varShat4;
561  }
562  }
563  }
564 }

566 // used at run-time to get some pre-computed LPM function values
567 void G4PairProductionRelModel::GetLPMFunctions(G4double &lpmGs,
568  G4double &lpmPhis,
569  const G4double sval) {
570  if (sval < gLPMFuncs.fSLimit) {
571  G4double val = sval*gLPMFuncs.fISDelta;
572  const G4int ilow = (G4int)val;
573  val -= ilow;
574  lpmGs = (gLPMFuncs.fLPMFuncG[ilow+1]-gLPMFuncs.fLPMFuncG[ilow])*val
575  + gLPMFuncs.fLPMFuncG[ilow];
576  lpmPhis = (gLPMFuncs.fLPMFuncPhi[ilow+1]-gLPMFuncs.fLPMFuncPhi[ilow])*val
577  + gLPMFuncs.fLPMFuncPhi[ilow];
578  } else {
579  G4double ss = sval*sval;
580  ss *= ss;
581  lpmPhis = 1.0-0.01190476/ss;
582  lpmGs = 1.0-0.0230655/ss;
583  }
584 }

586 void G4PairProductionRelModel::ComputeLPMfunctions(G4double &funcXiS,
587  G4double &funcGS, G4double &funcPhiS, const G4double eps,
588  const G4double egamma, const G4int izet)
589 {
590  // 1. y = E_+/E_{\gamma} with E_+ being the total energy transfered
591  // to one of the e-/e+ pair
592  // s' = \sqrt{ \frac{1}{8} \frac{1}{y(1-y)} \frac{E^{KL}_{LPM}}{E_{\gamma}} }
593  const G4double varSprime = std::sqrt(0.125*fLPMEnergy/(eps*egamma*(1.0-eps)));
594  const G4double condition = gElementData[izet]->fLPMVarS1Cond;
595  funcXiS = 2.0;
596  if (varSprime > 1.0) {
597  funcXiS = 1.0;
598  } else if (varSprime > condition) {
599  const G4double dum = gElementData[izet]->fLPMILVarS1Cond;
600  const G4double funcHSprime = G4Log(varSprime)*dum;
601  funcXiS = 1.0 + funcHSprime
602  - 0.08*(1.0-funcHSprime)*funcHSprime*(2.0-funcHSprime)*dum;
603  }
604  // 2. s=\frac{s'}{\sqrt{\xi(s')}}
605  const G4double varShat = varSprime / std::sqrt(funcXiS);
606  GetLPMFunctions(funcGS, funcPhiS, varShat);
607  // MAKE SURE SUPPRESSION IS SMALLER THAN 1: due to Migdal's approximation on xi
608  if (funcXiS * funcPhiS > 1. || varShat > 0.57) {
609  funcXiS = 1. / funcPhiS;
610  }
611 }
