174 void DetectorMessenger::SetNewValue(G4UIcommand* command,G4String newValue)
175 {
176 
177  if( command == fAbsMaterCmd )
178  { fDetector->SetTarget1Material(newValue);}
179 
180  if( command == fWorldMaterCmd )
181  { fDetector->SetTarget2Material(newValue);}
182 
183  if( command == fAbsThickCmd )
184  { fDetector->SetMylarZ(fAbsThickCmd->GetNewDoubleValue(newValue));}
185 
186  if( command == fAbsGapCmd )
187  { fDetector->SetGap(fAbsGapCmd->GetNewDoubleValue(newValue));}
188 
189  if( command == fAbsSizYZCmd )
190  { fDetector->SetTarget1Z(fAbsSizYZCmd->GetNewDoubleValue(newValue));}
191 
192  if( command == fWorldXCmd )
193  { fDetector->SetTarget2Z(fWorldXCmd->GetNewDoubleValue(newValue));}
194 
195  if( command == fXMagFieldCmd )
196  { fDetector->SetCheckShiftZ(fXMagFieldCmd->GetNewDoubleValue(newValue));}
197 
198  if( command == fYMagFieldCmd )
199  { G4double x = fYMagFieldCmd->GetNewDoubleValue(newValue);
200  fDetector->SetAbsorberZ(x);
201  }
202 
203  if( command == fZMagFieldCmd )
204  { fDetector->SetAbsorberShiftZ(fZMagFieldCmd->GetNewDoubleValue(newValue));}
205 
206  if( command == fNumOfAbsCmd )
207  {
208  fDetector->SetNumberDivR(fNumOfAbsCmd->GetNewIntValue(newValue));
209  }
210 
211  if( command == fNumOfEvt )
212  { fDetector->SetNumberDivZ(fNumOfEvt->GetNewIntValue(newValue));}
213 
214  if( command == fIntCmd )
215  { fDetector->SetNumberDivE(fIntCmd->GetNewIntValue(newValue));}
216  if( command == fDeltaECmd )
217  { fDetector->SetMaxEnergy(fDeltaECmd->GetNewDoubleValue(newValue));}
218 
219  if( command == fVerbCmd ){
220  G4int ver = fVerbCmd->GetNewIntValue(newValue);
221  fDetector->SetVerbose(ver);
222  }
223 
224 }
