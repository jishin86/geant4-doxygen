60 void PhysListEmPolarized::ConstructProcess()
61 {
62  // Add standard EM Processes
63 
64  auto particleIterator=GetParticleIterator();
65  particleIterator->reset();
66  while( (*particleIterator)() ){
67  G4ParticleDefinition* particle = particleIterator->value();
68  G4ProcessManager* pmanager = particle->GetProcessManager();
69  G4String particleName = particle->GetParticleName();
70 
71  if (particleName == "gamma") {
72  pmanager->AddDiscreteProcess(new G4PolarizedPhotoElectricEffect);
73  pmanager->AddDiscreteProcess(new G4PolarizedCompton);
74  pmanager->AddDiscreteProcess(new G4PolarizedGammaConversion);
75 
76  } else if (particleName == "e-") {
77  pmanager->AddProcess(new G4eMultipleScattering, -1,1,1);
78  pmanager->AddProcess(new G4ePolarizedIonisation, -1,2,2);
79  pmanager->AddProcess(new G4ePolarizedBremsstrahlung, -1,3,3);
80 
81  } else if (particleName == "e+") {
82  pmanager->AddProcess(new G4eMultipleScattering, -1, 1,1);
83  pmanager->AddProcess(new G4ePolarizedIonisation, -1, 2,2);
84  pmanager->AddProcess(new G4ePolarizedBremsstrahlung, -1, 3,3);
85  pmanager->AddProcess(new G4eplusPolarizedAnnihilation, 0,-1,4);
86  }
87  }
88 }
