122 G4VParticleChange*
123 G4OpRayleigh::PostStepDoIt(const G4Track& aTrack, const G4Step& aStep)
124 {
125  aParticleChange.Initialize(aTrack);
126 
127  const G4DynamicParticle* aParticle = aTrack.GetDynamicParticle();
128 
129  if (verboseLevel>0) {
130  G4cout << "Scattering Photon!" << G4endl;
131  G4cout << "Old Momentum Direction: "
132  << aParticle->GetMomentumDirection() << G4endl;
133  G4cout << "Old Polarization: "
134  << aParticle->GetPolarization() << G4endl;
135  }
136 
137  G4double cosTheta;
138  G4ThreeVector OldMomentumDirection, NewMomentumDirection;
139  G4ThreeVector OldPolarization, NewPolarization;
140 
141  G4double rand, constant;
142  G4double CosTheta, SinTheta, SinPhi, CosPhi, unit_x, unit_y, unit_z;
143 
144  do {
145  // Try to simulate the scattered photon momentum direction
146  // w.r.t. the initial photon momentum direction
147 
148  CosTheta = G4UniformRand();
149  SinTheta = std::sqrt(1.-CosTheta*CosTheta);
150  // consider for the angle 90-180 degrees
151  if (G4UniformRand() < 0.5) CosTheta = -CosTheta;
152 
153  // simulate the phi angle
154  rand = twopi*G4UniformRand();
155  SinPhi = std::sin(rand);
156  CosPhi = std::cos(rand);
157 
158  // start constructing the new momentum direction
159  unit_x = SinTheta * CosPhi;
160  unit_y = SinTheta * SinPhi;
161  unit_z = CosTheta;
162  NewMomentumDirection.set (unit_x,unit_y,unit_z);
163 
164  // Rotate the new momentum direction into global reference system
165  OldMomentumDirection = aParticle->GetMomentumDirection();
166  OldMomentumDirection = OldMomentumDirection.unit();
167  NewMomentumDirection.rotateUz(OldMomentumDirection);
168  NewMomentumDirection = NewMomentumDirection.unit();
169 
170  // calculate the new polarization direction
171  // The new polarization needs to be in the same plane as the new
172  // momentum direction and the old polarization direction
173  OldPolarization = aParticle->GetPolarization();
174  constant = -NewMomentumDirection.dot(OldPolarization);
175 
176  NewPolarization = OldPolarization + constant*NewMomentumDirection;
177  NewPolarization = NewPolarization.unit();
178 
179  // There is a corner case, where the Newmomentum direction
180  // is the same as oldpolariztion direction:
181  // random generate the azimuthal angle w.r.t. Newmomentum direction
182  if (NewPolarization.mag() == 0.) {
183  rand = G4UniformRand()*twopi;
184  NewPolarization.set(std::cos(rand),std::sin(rand),0.);
185  NewPolarization.rotateUz(NewMomentumDirection);
186  } else {
187  // There are two directions which are perpendicular
188  // to the new momentum direction
189  if (G4UniformRand() < 0.5) NewPolarization = -NewPolarization;
190  }
191 
192  // simulate according to the distribution cos^2(theta)
193  cosTheta = NewPolarization.dot(OldPolarization);
194  // Loop checking, 13-Aug-2015, Peter Gumplinger
195  } while (std::pow(cosTheta,2) < G4UniformRand());
196 
197  aParticleChange.ProposePolarization(NewPolarization);
198  aParticleChange.ProposeMomentumDirection(NewMomentumDirection);
199 
200  if (verboseLevel>0) {
201  G4cout << "New Polarization: "
202  << NewPolarization << G4endl;
203  G4cout << "Polarization Change: "
204  << *(aParticleChange.GetPolarization()) << G4endl;
205  G4cout << "New Momentum Direction: "
206  << NewMomentumDirection << G4endl;
207  G4cout << "Momentum Change: "
208  << *(aParticleChange.GetMomentumDirection()) << G4endl;
209  }
210 
211  return G4VDiscreteProcess::PostStepDoIt(aTrack, aStep);
212 }

214 // BuildPhysicsTable for the Rayleigh Scattering process
215 // --------------------------------------------------------
216 void G4OpRayleigh::BuildPhysicsTable(const G4ParticleDefinition&)
217 {
218  if (thePhysicsTable) {
219  thePhysicsTable->clearAndDestroy();
220  delete thePhysicsTable;
221  thePhysicsTable = NULL;
222  }
223 
224  const G4MaterialTable* theMaterialTable = G4Material::GetMaterialTable();
225  const G4int numOfMaterials = G4Material::GetNumberOfMaterials();
226 
227  thePhysicsTable = new G4PhysicsTable( numOfMaterials );
228 
229  for( G4int iMaterial = 0; iMaterial < numOfMaterials; iMaterial++ )
230  {
231  G4Material* material = (*theMaterialTable)[iMaterial];
232  G4MaterialPropertiesTable* materialProperties =
233  material->GetMaterialPropertiesTable();
234  G4PhysicsOrderedFreeVector* rayleigh = NULL;
235  if ( materialProperties != NULL ) {
236  rayleigh = materialProperties->GetProperty( kRAYLEIGH );
237  if ( rayleigh == NULL ) rayleigh =
238  CalculateRayleighMeanFreePaths( material );
239  }
240  thePhysicsTable->insertAt( iMaterial, rayleigh );
241  }
242 }

264 // CalculateRayleighMeanFreePaths()
265 // --------------------------------
266 // Private method to compute Rayleigh Scattering Lengths
267 G4PhysicsOrderedFreeVector*
268 G4OpRayleigh::CalculateRayleighMeanFreePaths( const G4Material* material ) const
269 {
270  G4MaterialPropertiesTable* materialProperties =
271  material->GetMaterialPropertiesTable();
272 
273  // Retrieve the beta_T or isothermal compressibility value. For backwards
274  // compatibility use a constant if the material is "Water". If the material
275  // doesn't have an ISOTHERMAL_COMPRESSIBILITY constant then return
276  G4double betat;
277  if ( material->GetName() == "Water" )
278  betat = 7.658e-23*m3/MeV;
279  else if(materialProperties->ConstPropertyExists("ISOTHERMAL_COMPRESSIBILITY"))
280  betat = materialProperties->GetConstProperty(kISOTHERMAL_COMPRESSIBILITY);
281  else
282  return NULL;
283 
284  // If the material doesn't have a RINDEX property vector then return
285  G4MaterialPropertyVector* rIndex = materialProperties->GetProperty(kRINDEX);
286  if ( rIndex == NULL ) return NULL;
287 
288  // Retrieve the optional scale factor, (this just scales the scattering length
289  G4double scaleFactor = 1.0;
290  if( materialProperties->ConstPropertyExists( "RS_SCALE_FACTOR" ) )
291  scaleFactor= materialProperties->GetConstProperty(kRS_SCALE_FACTOR );
292 
293  // Retrieve the material temperature. For backwards compatibility use a
294  // constant if the material is "Water"
295  G4double temperature;
296  if( material->GetName() == "Water" )
297  temperature = 283.15*kelvin; // Temperature of water is 10 degrees celsius
298  else
299  temperature = material->GetTemperature();
300 
301  G4PhysicsOrderedFreeVector* rayleighMeanFreePaths =
302  new G4PhysicsOrderedFreeVector();
303  // This calculates the meanFreePath via the Einstein-Smoluchowski formula
304  const G4double c1 = scaleFactor * betat * temperature * k_Boltzmann /
305  ( 6.0 * pi );
306 
307  for( size_t uRIndex = 0; uRIndex < rIndex->GetVectorLength(); uRIndex++ )
308  {
309  const G4double energy = rIndex->Energy( uRIndex );
310  const G4double rIndexSquared = (*rIndex)[uRIndex] * (*rIndex)[uRIndex];
311  const G4double xlambda = h_Planck * c_light / energy;
312  const G4double c2 = std::pow(twopi/xlambda,4);
313  const G4double c3 =
314  std::pow(((rIndexSquared-1.0)*(rIndexSquared+2.0 )/3.0),2);
315 
316  const G4double meanFreePath = 1.0 / ( c1 * c2 * c3 );
317 
318  if( verboseLevel>0 )
319  G4cout << energy << "MeV\t" << meanFreePath << "mm" << G4endl;
320 
321  rayleighMeanFreePaths->InsertValues( energy, meanFreePath );
322  }
323 
324  return rayleighMeanFreePaths;
325 }
