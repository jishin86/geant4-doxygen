46 void
47 G4TrajectoryDrawByEncounteredVolume::Draw(const G4VTrajectory& traj, const G4bool& visible) const
48 {
49  try
50  {
51  const G4RichTrajectory& richTrajectory = dynamic_cast<const G4RichTrajectory&>(traj);
52 
53  G4Colour colour(fDefault);
54  G4String soughtPVName("none");
55 
56  for (const auto& item: fMap.GetBasicMap()) {
57  soughtPVName = item.first;
58  for (G4int iPoint = 0; iPoint < richTrajectory.GetPointEntries(); iPoint++) {
59  G4VTrajectoryPoint* point = richTrajectory.GetPoint(iPoint);
60  if (!point) continue;
61  std::vector<G4AttValue>* attValues = point->CreateAttValues();
62  std::vector<G4AttValue>::const_iterator iAtt;
63  for (iAtt = attValues->begin(); iAtt != attValues->end(); ++iAtt) {
64  if (iAtt->GetName() == "PostVPath" &&
65  iAtt->GetValue().contains(soughtPVName)) break;
66  }
67  if (iAtt != attValues->end()) { // Required value found
68  fMap.GetColour(soughtPVName, colour);
69  break; // First found pvname determines colour.
70  }
71  }
72  }
73 
74  G4VisTrajContext myContext(GetContext());
75 
76  myContext.SetLineColour(colour);
77  myContext.SetVisible(visible);
78 
79  if (GetVerbose()) {
80  G4cout
81  << "G4TrajectoryDrawByEncounteredVolume drawer named " << Name()
82  << ", drawing trajectory touching physical volume " << soughtPVName
83  << ", with configuration:" << G4endl;
84  myContext.Print(G4cout);
85  }
86 
87  G4TrajectoryDrawerUtils::DrawLineAndPoints(richTrajectory, myContext);
88 
89  }
90 
91  catch (const std::bad_cast&)
92  {
93  G4ExceptionDescription ed;
94  ed << "Requires G4RichTrajectory - \"/vis/scene/add/trajectories rich\"";
95  G4Exception
96  ("G4TrajectoryDrawByEncounteredVolume::Draw(const G4VTrajectory& traj,...",
97  "modeling0125",
98  JustWarning, ed);
99  return;
100  }
101 }
