91  G4int l,
92  G4double rot)
93 {
94  // Align Miller normal vector (hkl) with +Z axis, and rotation about axis
95 
96  if (verboseLevel)
97  {
98  G4cout << "G4LatticePhysical::SetMillerOrientation(" << h << " "
99  << k << " " << l << ", " << rot/CLHEP::deg << " deg)" << G4endl;
100  }
101 
102  hMiller = h;
103  kMiller = k;
104  lMiller = l;
105  fRot = rot;
106 
107  G4ThreeVector norm = (h*GetBasis(0)+k*GetBasis(1)+l*GetBasis(2)).unit();
108 
109  if (verboseLevel>1) G4cout << " norm = " << norm << G4endl;
110 
111  // Aligns geometry +Z axis with lattice (hkl) normal
112  fOrient = G4RotationMatrix::IDENTITY;
113  fOrient.rotateZ(rot).rotateY(norm.theta()).rotateZ(norm.phi());
114  fInverse = fOrient.inverse();
115 
116  if (verboseLevel>1) G4cout << " fOrient = " << fOrient << G4endl;
117 
118  // FIXME: Is this equivalent to (phi,theta,rot) Euler angles???
119 }
