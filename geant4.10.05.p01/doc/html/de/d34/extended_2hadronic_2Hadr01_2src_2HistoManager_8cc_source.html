205 {

207  G4cout << "HistoManager: End of run actions are started" << G4endl;
208 
209  // Average values
210  G4cout<<"========================================================"<<G4endl;
211 
212  G4double x = (G4double)fNevt;
213  if(fNevt > 0) { x = 1.0/x; }

215  G4double xe = x*(G4double)fNelec;

323 void HistoManager::ScoreNewTrack(const G4Track* track)
324 {
325  const G4ParticleDefinition* pd = track->GetDefinition();
326  G4String name = pd->GetParticleName();
327  G4double e = track->GetKineticEnergy();
328 
329  // Primary track
330  if(0 == track->GetParentID()) {
331 
332  fNevt++;
333  fPrimaryKineticEnergy = e;
334  fPrimaryDef = pd;
335  G4ThreeVector dir = track->GetMomentumDirection();
336  if(1 < fVerbose)
337  G4cout << "### Primary " << name
338  << " kinE(MeV)= " << e/MeV
339  << "; m(MeV)= " << pd->GetPDGMass()/MeV
340  << "; pos(mm)= " << track->GetPosition()/mm
341  << "; dir= " << track->GetMomentumDirection()
342  << G4endl;
343 
344  // Secondary track
345  } else {
346  if(1 < fVerbose)
347  G4cout << "=== Secondary " << name
348  << " kinE(MeV)= " << e/MeV
349  << "; m(MeV)= " << pd->GetPDGMass()/MeV
350  << "; pos(mm)= " << track->GetPosition()/mm
351  << "; dir= " << track->GetMomentumDirection()
352  << G4endl;
353  e = std::log10(e/MeV);
354  if(pd == G4Gamma::Gamma()) {
355  fNgam++;
356  fHisto->Fill(1,e,1.0);
357  } else if ( pd == G4Electron::Electron()) {
358  fNelec++;
359  fHisto->Fill(2,e,1.0);
360  } else if ( pd == G4Positron::Positron()) {
361  fNposit++;
362  fHisto->Fill(3,e,1.0);
363  } else if ( pd == G4Proton::Proton()) {
364  fNproton++;
365  fHisto->Fill(4,e,1.0);
366  } else if ( pd == fNeutron) {
367  fNneutron++;
368  fHisto->Fill(5,e,1.0);
369  } else if ( pd == G4AntiProton::AntiProton()) {
370  fNaproton++;
371  } else if ( pd == G4PionPlus::PionPlus() ) {
372  fNcpions++;
373  fHisto->Fill(6,e,1.0);
374  fHisto->Fill(19,e,1.0);
375 
376  } else if ( pd == G4PionMinus::PionMinus()) {
377  fNcpions++;
378  fHisto->Fill(6,e,1.0);
379  fHisto->Fill(20,e,1.0);
380 
381  } else if ( pd == G4PionZero::PionZero()) {
382  fNpi0++;
383  fHisto->Fill(7,e,1.0);
384  } else if ( pd == G4KaonPlus::KaonPlus() ||
385  pd == G4KaonMinus::KaonMinus()) {
386  fNkaons++;
387  fHisto->Fill(8,e,1.0);
388  } else if ( pd == G4KaonZeroShort::KaonZeroShort() ||
389  pd == G4KaonZeroLong::KaonZeroLong()) {
390  fNkaons++;
391  fHisto->Fill(9,e,1.0);
392  } else if ( pd == G4Deuteron::Deuteron() || pd == G4Triton::Triton()) {
393  fNdeut++;
394  fHisto->Fill(10,e,1.0);
395  } else if ( pd == G4He3::He3() || pd == G4Alpha::Alpha()) {
396  fNalpha++;
397  fHisto->Fill(11,e,1.0);
398  } else if ( pd->GetParticleType() == "nucleus") {
399  fNions++;
400  fHisto->Fill(12,e,1.0);
401  } else if ( pd == G4MuonPlus::MuonPlus() ||
402  pd == G4MuonMinus::MuonMinus()) {
403  fNmuons++;
404  fHisto->Fill(13,e,1.0);
405  }
406  }

411 void HistoManager::AddTargetStep(const G4Step* step)
412 {
413  fNstep++;
414  G4double fEdep = step->GetTotalEnergyDeposit();
415  if(1 < fVerbose) {
416  G4cout << "TargetSD::ProcessHits: beta1= "
417  << step->GetPreStepPoint()->GetVelocity()/c_light
418  << " beta2= " << step->GetPostStepPoint()->GetVelocity()/c_light
419  << " weight= " << step->GetTrack()->GetWeight()
420  << G4endl;
421  }
422  if(fEdep >= DBL_MIN) {
423  const G4Track* track = step->GetTrack();
424 
425  G4ThreeVector pos =
426  (step->GetPreStepPoint()->GetPosition() +
427  step->GetPostStepPoint()->GetPosition())*0.5;
428 
429  G4double z = pos.z() - fAbsZ0;
430 
431  // scoring
432  fEdepEvt += fEdep;
433  fHisto->Fill(0,z,fEdep);
434  const G4ParticleDefinition* pd = track->GetDefinition();
435 
436  if(pd == G4Gamma::Gamma() || pd == G4Electron::Electron()
437  || pd == G4Positron::Positron()) {
438  fEdepEM += fEdep;
439  } else if ( pd == G4PionPlus::PionPlus() ||
440  pd == G4PionMinus::PionMinus()) {
441  fEdepPI += fEdep;
442  } else if ( pd == G4Proton::Proton() ||
443  pd == G4AntiProton::AntiProton()) {
444  fEdepP += fEdep;
445  }
446 
447  if(1 < fVerbose) {
448  G4cout << "HistoManager::AddEnergy: e(keV)= " << fEdep/keV
449  << "; z(mm)= " << z/mm
450  << "; step(mm)= " << step->GetStepLength()/mm
451  << " by " << pd->GetParticleName()
452  << " E(MeV)= " << track->GetKineticEnergy()/MeV
453  << G4endl;
454  }
455  }
456 }

460 void HistoManager::AddLeakingParticle(const G4Track* track)
461 {
462  const G4ParticleDefinition* pd = track->GetDefinition();
463  G4double e = std::log10(track->GetKineticEnergy()/MeV);
464 
465  G4ThreeVector pos = track->GetPosition();
466  G4ThreeVector dir = track->GetMomentumDirection();
467  G4double x = pos.x();
468  G4double y = pos.y();
469  G4double z = pos.z();
470 
471  G4bool isLeaking = false;
472 
473  // Forward
474  if(z > -fAbsZ0 && dir.z() > 0.0) {
475  isLeaking = true;
476  if(pd == fNeutron) {
477  ++fNneu_forw;
478  fHisto->Fill(15,e,1.0);
479  } else isLeaking = true;
480 
481  // Backward
482  } else if (z < fAbsZ0 && dir.z() < 0.0) {
483  isLeaking = true;
484  if(pd == fNeutron) {
485  ++fNneu_back;
486  fHisto->Fill(16,e,1.0);
487  } else isLeaking = true;
488 
489  // Side
490  } else if (std::abs(z) <= -fAbsZ0 && x*dir.x() + y*dir.y() > 0.0) {
491  isLeaking = true;
492  if(pd == fNeutron) {
493  ++fNneu_leak;
494  fHisto->Fill(14,e,1.0);
495  } else isLeaking = true;
496  }
497 
498  // protons and pions
499  if(isLeaking) {
500  if(pd == G4Proton::Proton()) {
501  fHisto->Fill(17,e,1.0);
502  ++fNprot_leak;
503  } else if (pd == G4PionPlus::PionPlus() ||
504  pd == G4PionMinus::PionMinus()) {
505  fHisto->Fill(18,e,1.0);
506  ++fNpiofNleak;
507  }
508  }
509 }
