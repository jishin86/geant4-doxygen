52 G4tgrVolumeDivision::G4tgrVolumeDivision( const std::vector<G4String>& wl )
53 {
54  // wl: NAME PARENT MATERIAL AXIS STEP/NDIV OFFSET
55 
56  G4tgrUtils::CheckWLsize( wl, 6, WLSIZE_GE,
57  "G4tgrVolumeDivision::G4tgrVolumeDivision" );
58  G4tgrUtils::CheckWLsize( wl, 8, WLSIZE_LE,
59  "G4tgrVolumeDivision::G4tgrVolumeDivision" );
60 
61  theType = "VOLDivision";
62 
63  // :DIV NAME PARENT MATERIAL AXIS STEP/NDIV OFFSET
64 
65  //---------- set name
66  theName = G4tgrUtils::GetString( wl[1] );
67 
68  //---------- set the pointer to the parent DU
69  G4String parentName = G4tgrUtils::GetString(wl[2]);
70  G4tgrVolumeMgr::GetInstance()->FindVolume( parentName, 1); // check existance
71 
72  //---------- initialize G4tgrPlace
73  thePlaceDiv = new G4tgrPlaceDivRep();
74  thePlaceDiv->SetParentName( parentName );
75  thePlaceDiv->SetType("PlaceDivision");
76  thePlaceDiv->SetVolume( this );
77 
78  //---------- set material name
79  theMaterialName = G4tgrUtils::GetString( wl[3] );
80 
81  //----- set axis of replica
82  thePlaceDiv->SetAxis( thePlaceDiv->BuildAxis(G4tgrUtils::GetString(wl[4])) );
83 
84  //------ register parent - child
85  G4tgrVolumeMgr::GetInstance()->RegisterParentChild( parentName, thePlaceDiv );
86 #ifdef G4VERBOSE
87  if( G4tgrMessenger::GetVerboseLevel() >= 3 )
88  {
89  G4cout << " G4tgrVolumeDivision::G4tgrVolumeDivision() -"
90  << " Replica register parent - child " << G4endl;
91  }
92 #endif
93 
94  //---------- set if division is given by number of divisions of by width
95  G4String wl0 = wl[0];
96  for( size_t ii = 0; ii < wl0.length(); ii++ )
97  {
98  wl0[ii] = toupper( wl0[ii] );
99  }
100 
101  if( wl0 == ":DIV_NDIV" )
102  {
103  thePlaceDiv->SetDivType( DivByNdiv );
104  thePlaceDiv->SetNDiv( G4tgrUtils::GetInt( wl[5] ) );
105  if( wl.size() == 7 )
106  {
107  thePlaceDiv->SetOffset( G4tgrUtils::GetDouble( wl[6] )*mm );
108  }
109  }
110  else if( wl0 == ":DIV_WIDTH" )
111  {
112  thePlaceDiv->SetDivType( DivByWidth );
113  thePlaceDiv->SetWidth( G4tgrUtils::GetDouble( wl[5] )*mm );
114  if( wl.size() == 7 )
115  {
116  thePlaceDiv->SetOffset( G4tgrUtils::GetDouble( wl[6] )*mm );
117  }
118  }
119  else if( wl0 == ":DIV_NDIV_WIDTH" )
120  {
121  thePlaceDiv->SetDivType( DivByNdivAndWidth );
122  thePlaceDiv->SetNDiv( G4tgrUtils::GetInt( wl[5] ) );
123  thePlaceDiv->SetWidth( G4tgrUtils::GetDouble( wl[6] )*mm );
124  if( wl.size() == 8 )
125  {
126  thePlaceDiv->SetOffset( G4tgrUtils::GetDouble( wl[7] )*mm );
127  }
128  }
129  else
130  {
131  G4String ErrMessage = "Division type not supported, sorry... " + wl[0];
132  G4Exception("G4tgrVolumeDivision::G4tgrVolumeDivision()",
133  "NotImplemented", FatalException, ErrMessage);
134  }
135 
136  theVisibility = 1;
137  theRGBColour = new G4double[3];
138  for(size_t ii=0; ii<3; ii++) { theRGBColour[ii] = -1.; }
139 
140 #ifdef G4VERBOSE
141  if( G4tgrMessenger::GetVerboseLevel() >= 1 )
142  {
143  G4cout << " Created " << *this << G4endl;
144  }
145 #endif
146 
147  theSolid = 0;
148 }
