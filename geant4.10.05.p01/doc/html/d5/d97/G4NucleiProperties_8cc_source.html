75 G4double G4NucleiProperties::GetNuclearMass(const G4int A, const G4int Z)
76 {
77  if (mass_proton <= 0.0 ) {
78  const G4ParticleDefinition * nucleus = nullptr;
79  nucleus = G4ParticleTable::GetParticleTable()->FindParticle("proton"); // proton
80  if (nucleus!=nullptr) mass_proton = nucleus->GetPDGMass();
81 
82  nucleus = G4ParticleTable::GetParticleTable()->FindParticle("neutron"); // neutron
83  if (nucleus!=nullptr) mass_neutron = nucleus->GetPDGMass();
84 
85  nucleus = G4ParticleTable::GetParticleTable()->FindParticle("deuteron"); // deuteron
86  if (nucleus!=nullptr) mass_deuteron = nucleus->GetPDGMass();
87 
88  nucleus = G4ParticleTable::GetParticleTable()->FindParticle("triton"); // triton
89  if (nucleus!=nullptr) mass_triton = nucleus->GetPDGMass();
90 
91  nucleus = G4ParticleTable::GetParticleTable()->FindParticle("alpha"); // alpha
92  if (nucleus!=nullptr) mass_alpha = nucleus->GetPDGMass();
93 
94  nucleus = G4ParticleTable::GetParticleTable()->FindParticle("He3"); // He3
95  if (nucleus!=nullptr) mass_He3 = nucleus->GetPDGMass();
96 
97  }
98 
99  if (A < 1 || Z < 0 || Z > A) {
100 #ifdef G4VERBOSE
101  if (G4ParticleTable::GetParticleTable()->GetVerboseLevel()>0) {
102  G4cout << "G4NucleiProperties::GetNuclearMass: Wrong values for A = " << A
103  << " and Z = " << Z << G4endl;
104  }
105 #endif
106  return 0.0;
107  }
108 
109  G4double mass= -1.;
110  if ( (Z<=2) ) {
111  // light nuclei
112  if ( (Z==1)&&(A==1) ) {
113  mass = mass_proton;
114  } else if ( (Z==0)&&(A==1) ) {
115  mass = mass_neutron;
116  } else if ( (Z==1)&&(A==2) ) {
117  mass = mass_deuteron;
118  } else if ( (Z==1)&&(A==3) ) {
119  mass = mass_triton;
120  } else if ( (Z==2)&&(A==4) ) {
121  mass = mass_alpha;
122  } else if ( (Z==2)&&(A==3) ) {
123  mass = mass_He3;
124  }
125  }
126 
127  if (mass < 0.) {
128  if ( G4NucleiPropertiesTableAME12::IsInTable(Z,A) ) {
129  // AME table
130  mass = G4NucleiPropertiesTableAME12::GetNuclearMass(Z,A);
131  } else if (G4NucleiPropertiesTheoreticalTable::IsInTable(Z,A)){
132  // Theoretical table
133  mass = G4NucleiPropertiesTheoreticalTable::GetNuclearMass(Z,A);
134  } else {
135  mass = NuclearMass(G4double(A),G4double(Z));
136  }
137  }
138 
139  if (mass < 0.) mass = 0.0;
140  return mass;
141 }

260 G4double G4NucleiProperties::AtomicMass(G4double A, G4double Z)
261 {
262  G4double hydrogen_mass_excess;
263  G4double neutron_mass_excess;
264  hydrogen_mass_excess = G4NucleiPropertiesTableAME12::GetMassExcess(1,1);
265  neutron_mass_excess = G4NucleiPropertiesTableAME12::GetMassExcess(0,1);
266  G4double mass =
267  (A-Z)*neutron_mass_excess + Z*hydrogen_mass_excess - BindingEnergy(A,Z) + A*amu_c2;
268 
269  return mass;
270 }

272 G4double G4NucleiProperties::NuclearMass(G4double A, G4double Z)
273 {
274  if (A < 1 || Z < 0 || Z > A) {
275 #ifdef G4VERBOSE
276  if (G4ParticleTable::GetParticleTable()->GetVerboseLevel()>0) {
277  G4cout << "G4NucleiProperties::NuclearMass: Wrong values for A = "
278  << A << " and Z = " << Z << G4endl;
279  }
280 #endif
281  return 0.0;
282  }
283 
284  G4double mass = AtomicMass(A,Z);
285  // atomic mass is converted to nuclear mass according formula in AME03 and 12
286  mass -= Z*electron_mass_c2;
287  mass += ( 14.4381*std::pow ( Z , 2.39 ) + 1.55468*1e-6*std::pow ( Z , 5.35 ) )*eV;
288 
289  return mass;
290 }

292 G4double G4NucleiProperties::BindingEnergy(G4double A, G4double Z)
293 {
294  //
295  // Weitzsaecker's Mass formula
296  //
297  G4int Npairing = G4int(A-Z)%2; // pairing
298  G4int Zpairing = G4int(Z)%2;
299  G4double binding =
300  - 15.67*A // nuclear volume
301  + 17.23*std::pow(A,2./3.) // surface energy
302  + 93.15*((A/2.-Z)*(A/2.-Z))/A // asymmetry
303  + 0.6984523*Z*Z*std::pow(A,-1./3.); // coulomb
304  if( Npairing == Zpairing ) binding += (Npairing+Zpairing-1) * 12.0 / std::sqrt(A); // pairing
305 
306  return -binding*MeV;
307 }
