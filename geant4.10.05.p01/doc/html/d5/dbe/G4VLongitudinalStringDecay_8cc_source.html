64 G4VLongitudinalStringDecay::G4VLongitudinalStringDecay()
65 {
66  MassCut = 0.35*GeV;
67  ClusterMass = 0.15*GeV;
68 
69  SmoothParam = 0.9;
70  StringLoopInterrupt = 1000;
71  ClusterLoopInterrupt = 500;
72 
73  // Changable Parameters below.
74  SigmaQT = 0.5 * GeV;
75 
76  StrangeSuppress = 0.44; // 27 % strange quarks produced, ie. u:d:s=1:1:0.27
77  DiquarkSuppress = 0.07;
78  DiquarkBreakProb = 0.1;
79 
80  //... pspin_meson is probability to create pseudo-scalar meson
81  pspin_meson = 0.5;
82 
83  //... pspin_barion is probability to create 1/2 barion
84  pspin_barion = 0.5;
85 
86  //... vectorMesonMix[] is quark mixing parameters for vector mesons (Variable spin = 3)
87  vectorMesonMix.resize(6);
88  vectorMesonMix[0] = 0.;
89  vectorMesonMix[1] = 0.375;
90  vectorMesonMix[2] = 0.0;
91  vectorMesonMix[3] = 0.375;
92  vectorMesonMix[4] = 1.0;
93  vectorMesonMix[5] = 1.0;
94 
95  //... scalarMesonMix[] is quark mixing parameters for scalar mesons (Variable spin=1)
96  scalarMesonMix.resize(6);
97  scalarMesonMix[0] = 0.5;
98  scalarMesonMix[1] = 0.25;
99  scalarMesonMix[2] = 0.5;
100  scalarMesonMix[3] = 0.25;
101  scalarMesonMix[4] = 1.0;
102  scalarMesonMix[5] = 0.5;
103 
104  // Parameters may be changed until the first fragmentation starts
105  PastInitPhase=false;
106  hadronizer = new G4HadronBuilder(pspin_meson,pspin_barion,
107  scalarMesonMix,vectorMesonMix);
108  Kappa = 1.0 * GeV/fermi;
109 }

146 G4KineticTrackVector* G4VLongitudinalStringDecay::LightFragmentationTest(const G4ExcitedString * const string)
147 {
148  // Check string decay threshold
149  G4KineticTrackVector * result=0; // return 0 when string exceeds the mass cut
150 
151  pDefPair hadrons((G4ParticleDefinition *)0,(G4ParticleDefinition *)0);
152 
153  G4FragmentingString aString(*string);
154  if ( sqr(FragmentationMass(&aString,0,&hadrons)+MassCut) < aString.Mass2()) {
155  return 0;
156  }
157 
158  // The string mass is very low ---------------------------
159 
160  result=new G4KineticTrackVector;
161 
162  if ( hadrons.second ==0 )
163  {
164  // Substitute string by light hadron, Note that Energy is not conserved here!
165 
166  #ifdef debug_VStringDecay
167  G4cout << "VlongSF Warning replacing string by single hadron (G4VLongitudinalStringDecay)" <<G4endl;
168  G4cout << hadrons.first->GetParticleName()<<G4endl
169  << "string .. " << string->Get4Momentum() << " "
170  << string->Get4Momentum().m() << G4endl;
171  #endif
172 
173  G4ThreeVector Mom3 = string->Get4Momentum().vect();
174  G4LorentzVector Mom( Mom3, std::sqrt( Mom3.mag2() + sqr(hadrons.first->GetPDGMass())) );
175  result->push_back( new G4KineticTrack( hadrons.first, 0, string->GetPosition(), Mom ) );
176  } else
177  {
178  //... string was qq--qqbar type: Build two stable hadrons,
179 
180  #ifdef debug_VStringDecay
181  G4cout << "VlongSF Warning replacing qq-qqbar string by TWO hadrons (G4VLongitudinalStringDecay)"
182  << hadrons.first->GetParticleName() << " / "
183  << hadrons.second->GetParticleName()
184  << "string .. " << string->Get4Momentum() << " "
185  << string->Get4Momentum().m() << G4endl;
186  #endif
187 
188  G4LorentzVector Mom1, Mom2;
189  Sample4Momentum(&Mom1, hadrons.first->GetPDGMass(),
190  &Mom2,hadrons.second->GetPDGMass(),
191  string->Get4Momentum().mag());
192 
193  result->push_back( new G4KineticTrack( hadrons.first, 0, string->GetPosition(), Mom1 ) );
194  result->push_back( new G4KineticTrack( hadrons.second, 0, string->GetPosition(), Mom2) );
195 
196  G4ThreeVector Velocity = string->Get4Momentum().boostVector();
197  result->Boost(Velocity);
198  }
199 
200  return result;
201 
202 }

206 G4double G4VLongitudinalStringDecay::FragmentationMass( const G4FragmentingString * const string,
207  Pcreate build, pDefPair * pdefs )
208 {
209  G4double mass;
210 
211  if ( build==0 ) build=&G4HadronBuilder::BuildLowSpin;
212 
213  G4ParticleDefinition *Hadron1, *Hadron2=0;
214 
215  if (!string->FourQuarkString() )
216  {
217  // spin 0 meson or spin 1/2 barion will be built
218 
219  Hadron1 = (hadronizer->*build)(string->GetLeftParton(), string->GetRightParton());
220  #ifdef debug_VStringDecay
221  G4cout<<"Quarks at the string ends "<<string->GetLeftParton()->GetParticleName()<<" "<<string->GetRightParton()->GetParticleName()<<G4endl;
222  G4cout<<"(G4VLongitudinalStringDecay) Hadron "<<Hadron1->GetParticleName()<<" "<<Hadron1->GetPDGMass()<<G4endl;
223  #endif
224  mass= (Hadron1)->GetPDGMass();
225  } else
226  {
227  //... string is qq--qqbar: Build two stable hadrons,
228  //... with extra uubar or ddbar quark pair
229  G4int iflc = (G4UniformRand() < 0.5)? 1 : 2;
230  if (string->GetLeftParton()->GetPDGEncoding() < 0) iflc = -iflc;
231 
232  //... theSpin = 4; spin 3/2 baryons will be built
233  Hadron1 = (hadronizer->*build)(string->GetLeftParton(), FindParticle(iflc));
234  Hadron2 = (hadronizer->*build)(string->GetRightParton(), FindParticle(-iflc));
235  mass = (Hadron1)->GetPDGMass() + (Hadron2)->GetPDGMass();
236  }
237 
238  if ( pdefs != 0 )
239  { // need to return hadrons as well....
240  pdefs->first = Hadron1;
241  pdefs->second = Hadron2;
242  }
243 
244  return mass;
245 }

304 G4VLongitudinalStringDecay::pDefPair G4VLongitudinalStringDecay::CreatePartonPair(G4int NeedParticle,G4bool AllowDiquarks)
305 {
306  // NeedParticle = +1 for Particle, -1 for Antiparticle
307  if ( AllowDiquarks && G4UniformRand() < DiquarkSuppress )
308  {
309  // Create a Diquark - AntiDiquark pair , first in pair is anti to IsParticle
310  G4int q1 = SampleQuarkFlavor();
311  G4int q2 = SampleQuarkFlavor();
312 
313  G4int spin = (q1 != q2 && G4UniformRand() <= 0.5)? 1 : 3;
314  // convention: quark with higher PDG number is first
315  G4int PDGcode = (std::max(q1,q2) * 1000 + std::min(q1,q2) * 100 + spin) * NeedParticle;
316  return pDefPair (FindParticle(-PDGcode),FindParticle(PDGcode));
317 
318  } else {
319  // Create a Quark - AntiQuark pair, first in pair IsParticle
320  G4int PDGcode=SampleQuarkFlavor()*NeedParticle;
321  return pDefPair (FindParticle(PDGcode),FindParticle(-PDGcode));
322  }
323 }

325 //-----------------------------------------------------------------------------
326 G4ThreeVector G4VLongitudinalStringDecay::SampleQuarkPt(G4double ptMax)
327 {
328  G4double Pt;
329  if ( ptMax < 0 ) {
330  // sample full gaussian
331  Pt = -G4Log(G4UniformRand());
332  } else {
333  // sample in limited range
334  Pt = -G4Log(G4RandFlat::shoot(G4Exp(-sqr(ptMax)/sqr(SigmaQT)), 1.));
335  }
336  Pt = SigmaQT * std::sqrt(Pt);
337  G4double phi = 2.*pi*G4UniformRand();
338  return G4ThreeVector(Pt * std::cos(phi),Pt * std::sin(phi),0);
339 }

343 void G4VLongitudinalStringDecay::CalculateHadronTimePosition(G4double theInitialStringMass, G4KineticTrackVector* Hadrons)
344 {
345  // `yo-yo` formation time
346  // const G4double kappa = 1.0 * GeV/fermi/4.;
347  G4double kappa = GetStringTensionParameter();
348  for(size_t c1 = 0; c1 < Hadrons->size(); c1++)
349  {
350  G4double SumPz = 0;
351  G4double SumE = 0;
352  for(size_t c2 = 0; c2 < c1; c2++)
353  {
354  SumPz += Hadrons->operator[](c2)->Get4Momentum().pz();
355  SumE += Hadrons->operator[](c2)->Get4Momentum().e();
356  }
357  G4double HadronE = Hadrons->operator[](c1)->Get4Momentum().e();
358  G4double HadronPz = Hadrons->operator[](c1)->Get4Momentum().pz();
359  Hadrons->operator[](c1)->SetFormationTime(
360  (theInitialStringMass - 2.*SumPz + HadronE - HadronPz ) / (2.*kappa) / c_light );
361  G4ThreeVector aPosition( 0, 0,
362  (theInitialStringMass - 2.*SumE - HadronE + HadronPz) / (2.*kappa) );
363  Hadrons->operator[](c1)->SetPosition(aPosition);
364  }
365 }
