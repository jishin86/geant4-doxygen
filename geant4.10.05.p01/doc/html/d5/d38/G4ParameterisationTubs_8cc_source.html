154 {

156

158  G4double pRMax = msol->GetInnerRadius() + foffset + fwidth * (copyNo+1) - fhgap;
159  G4double pDz = msol->GetZHalfLength();
160  //- already rotated G4double pSR = foffset + copyNo*fwidth;
161  G4double pSPhi = msol->GetStartPhiAngle();
162  G4double pDPhi = msol->GetDeltaPhiAngle();;
163 
164  tubs.SetInnerRadius( pRMin );
165  tubs.SetOuterRadius( pRMax );
166  tubs.SetZHalfLength( pDz );
167  tubs.SetStartPhiAngle( pSPhi, false );
