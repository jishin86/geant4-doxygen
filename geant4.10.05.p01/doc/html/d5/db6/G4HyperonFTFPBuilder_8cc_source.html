47 G4HyperonFTFPBuilder::G4HyperonFTFPBuilder():
48  theLambdaInelastic(0),
49  theAntiLambdaInelastic(0),
50  theSigmaMinusInelastic(0),
51  theAntiSigmaMinusInelastic(0),
52  theSigmaPlusInelastic(0),
53  theAntiSigmaPlusInelastic(0),
54  theXiZeroInelastic(0),
55  theAntiXiZeroInelastic(0),
56  theXiMinusInelastic(0),
57  theAntiXiMinusInelastic(0),
58  theOmegaMinusInelastic(0),
59  theAntiOmegaMinusInelastic(0),
60  wasActivated(false)
61 {
62 
63 // Hyperon : Bertini at low energies, then FTFP
64 
65  HyperonFTFP = new G4TheoFSGenerator("FTFP");
66 
67  HyperonFTFP->SetMinEnergy( 2.*GeV );
68  HyperonFTFP->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
69 
70  theStringModel = new G4FTFModel;
71  theStringDecay = new G4ExcitedStringDecay(theLund = new G4LundStringFragmentation);
72  theStringModel->SetFragmentationModel(theStringDecay);
73 
74  theCascade = new G4GeneratorPrecompoundInterface;
75 
76  HyperonFTFP->SetTransport(theCascade);
77  HyperonFTFP->SetHighEnergyGenerator(theStringModel);
78 
79  theBertini = new G4CascadeInterface;
80  theBertini->SetMinEnergy( 0.*GeV );
81  theBertini->SetMaxEnergy( 6.*GeV );
82 
83 // AntiHyperons: Use FTFP for full energy range, starting at 0.
84 
85  AntiHyperonFTFP = new G4TheoFSGenerator("FTFP");
86  AntiHyperonFTFP->SetMinEnergy( 0.*GeV );
87  AntiHyperonFTFP->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
88  AntiHyperonFTFP->SetTransport(theCascade);
89  AntiHyperonFTFP->SetHighEnergyGenerator(theStringModel);
90 
91 // use CHIPS cross sections
92  theCHIPSInelastic = G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4ChipsHyperonInelasticXS::Default_Name());
93 }
