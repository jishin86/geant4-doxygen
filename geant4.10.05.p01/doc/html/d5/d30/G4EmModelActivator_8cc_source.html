425 void G4EmModelActivator::ActivatePAI()
426 {
427  const std::vector<G4String> regnamesPAI = theParameters->RegionsPAI();
428  G4int nreg = regnamesPAI.size();
429  if(0 == nreg) { return; }
430  G4int verbose = theParameters->Verbose() - 1;
431  if(verbose > 0) {
432  G4cout << "### G4EmModelActivator::ActivatePAI for " << nreg << " regions"
433  << G4endl;
434  }
435  const std::vector<G4String> particlesPAI = theParameters->ParticlesPAI();
436  const std::vector<G4String> typesPAI = theParameters->TypesPAI();
437 
438  const std::vector<G4VEnergyLossProcess*>& v = G4LossTableManager::Instance()
439  ->GetEnergyLossProcessVector();
440 
441  G4RegionStore* regionStore = G4RegionStore::GetInstance();
442 
443  const G4ParticleDefinition* elec = G4Electron::Electron();
444  const G4ParticleDefinition* posi = G4Positron::Positron();
445  const G4ParticleDefinition* mupl = G4MuonPlus::MuonPlus();
446  const G4ParticleDefinition* mumi = G4MuonMinus::MuonMinus();
447  const G4ParticleDefinition* gion = G4GenericIon::GenericIon();
448 
449  for(G4int i = 0; i < nreg; ++i) {
450  const G4ParticleDefinition* p = nullptr;
451  if(particlesPAI[i] != "all") {
452  p = G4ParticleTable::GetParticleTable()->FindParticle(particlesPAI[i]);
453  if(!p) {
454  G4cout << "### WARNING: ActivatePAI::FindParticle fails to find "
455  << particlesPAI[i] << G4endl;
456  continue;
457  }
458  }
459  const G4Region* r = regionStore->GetRegion(regnamesPAI[i], false);
460  if(!r) {
461  G4cout << "### WARNING: ActivatePAI::GetRegion fails to find "
462  << regnamesPAI[i] << G4endl;
463  continue;
464  }
465 
466  G4String name = "hIoni";
467  if(p == elec || p == posi)
468  { name = "eIoni"; }
469  else if (p == mupl || p == mumi)
470  { name = "muIoni"; }
471  else if (p == gion)
472  { name = "ionIoni"; }
473 
474  for(auto proc : v) {
475 
476  if(!proc->IsIonisationProcess()) { continue; }
477 
478  G4String namep = proc->GetProcessName();
479  if(p) {
480  if(name != namep) { continue; }
481  } else {
482  if(namep != "hIoni" && namep != "muIoni" &&
483  namep != "eIoni" && namep != "ionIoni")
484  { continue; }
485  }
486  G4VEmModel* em = nullptr;
487  G4VEmFluctuationModel* fm = nullptr;
488  if(typesPAI[i] == "PAIphoton" || typesPAI[i] == "pai_photon") {
489  G4PAIPhotModel* mod = new G4PAIPhotModel(p,"PAIPhotModel");
490  em = mod;
491  fm = mod;
492  } else {
493  G4PAIModel* mod = new G4PAIModel(p,"PAIModel");
494  em = mod;
495  fm = mod;
496  }
497  proc->AddEmModel(0, em, fm, r);
498  if(verbose > 0) {
499  G4cout << "### G4EmModelActivator: add <" << typesPAI[i]
500  << "> model for " << particlesPAI[i]
501  << " in the " << regnamesPAI[i] << G4endl;
502  }
503  }
504  }
505 }

509 void G4EmModelActivator::ActivateMicroElec()
510 {
511  const std::vector<G4String> regnamesME = theParameters->RegionsMicroElec();
512  G4int nreg = regnamesME.size();
513  if(0 == nreg)
514  {
515  return;
516  }
517  G4int verbose = theParameters->Verbose() - 1;
518  if(verbose > 0)
519  {
520  G4cout << "### G4EmModelActivator::ActivateMicroElec for " << nreg
521  << " regions" << G4endl;
522  }
523  G4LossTableManager* man = G4LossTableManager::Instance();
524 
525  const G4ParticleDefinition* elec = G4Electron::Electron();
526  const G4ParticleDefinition* prot = G4Proton::Proton();
527  const G4ParticleDefinition* gion = G4GenericIon::GenericIon();
528  G4ProcessManager* eman = elec->GetProcessManager();
529  G4ProcessManager* pman = prot->GetProcessManager();
530  G4ProcessManager* iman = gion->GetProcessManager();
531 
532  G4bool emsc = HasMsc(eman);
533 
534  // MicroElec elastic is not active in the world
535  G4MicroElecElastic* eElasProc =
536  new G4MicroElecElastic("e-G4MicroElecElastic");
537  eman->AddDiscreteProcess(eElasProc);
538 
539  G4MicroElecInelastic* eInelProc =
540  new G4MicroElecInelastic("e-G4MicroElecInelastic");
541  eman->AddDiscreteProcess(eInelProc);
