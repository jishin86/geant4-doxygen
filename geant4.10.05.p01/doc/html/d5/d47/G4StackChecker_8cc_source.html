42 G4ClassificationOfNewTrack G4StackChecker::ClassifyNewTrack
43 (const G4Track* track)
44 {
45  G4ClassificationOfNewTrack result = fUrgent;
46  G4double e = track->GetKineticEnergy();
47  if ( (!(e < 0.0) && !(e > 0.0) && !(e == 0.0)) ||
48  track->GetMomentumDirection() == nullDirection)
49  {
50  result = fKill;
51  G4String nam = track->GetDefinition()->GetParticleName();
