55 void SteppingAction::UserSteppingAction(const G4Step* step)
56 {
57  // energy continuously deposited along trajectory
58  //
59  G4int trackID = step->GetTrack()->GetTrackID();
60  G4double Edep = step->GetTotalEnergyDeposit();
61  if (Edep > 0.) fEventaction->SumEnergyDeposited(trackID, Edep);
62 
63  // the rest for primary track only
64  if (trackID > 1) return;
65 
66  // count processes
67  //
68  const G4StepPoint* endPoint = step->GetPostStepPoint();
69  const G4VProcess* process = endPoint->GetProcessDefinedStep();
70  G4String procName = process->GetProcessName();
71  G4int subtype = process-> GetProcessSubType();
72  G4int nbsec = step->GetNumberOfSecondariesInCurrentStep();
73  if ((subtype == 2)&&(nbsec == 0)) procName = "Edep alone";
74  fRunaction->CountProcesses(procName);
75 
76  // step size and track length
77  //
78  G4double stepSize = step->GetStepLength();
79  fRunaction->TrackLength(stepSize);
80  G4AnalysisManager::Instance()->FillH1(1,stepSize);
81 
82  if (nbsec == 0) return; // no secondary particles
83 
84  // energy transfered to secondary particles
85  //
86  const std::vector<const G4Track*>* secondaries
87  = step->GetSecondaryInCurrentStep();
88  G4double Etransfer = 0.;
89  for (G4int itr=0; itr<nbsec; itr++) {
90  const G4Track* trk = (*secondaries)[itr];
91  const G4ParticleDefinition* particle = trk->GetParticleDefinition();
92  G4String name = particle->GetParticleName();
93  G4double energy = trk->GetKineticEnergy();
94  fRunaction->EnergySpectrumOfSecondaries(name,energy);
95  G4int ih = 0;
96  if (particle == G4Gamma::Gamma()) ih = 11;
97  else if (particle == G4Electron::Electron()) ih = 12;
98  else if (particle == G4Positron::Positron()) ih = 13;
99  if (ih > 0) G4AnalysisManager::Instance()->FillH1(ih,energy);
100  if (subtype == 4) energy = trk->GetTotalEnergy(); //(e+,e-) production
101  Etransfer += energy;
102  }
103  fEventaction->SumEnergyTransfered(process, Etransfer);
104 }
