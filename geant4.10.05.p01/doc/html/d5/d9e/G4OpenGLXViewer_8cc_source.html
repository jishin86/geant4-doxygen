170 void G4OpenGLXViewer::CreateGLXContext (XVisualInfo* v) {
171 
172  vi = v;
173 // get window's attributes
174  if (!XGetWindowAttributes(dpy, XRootWindow (dpy, vi -> screen), &xwa)) {
175  fViewId = -1; // This flags an error.
176  G4cerr << "G4OpenGLXViewer::G4OpenGLXViewer couldn't return window attributes"
177  << G4endl;
178  return;
179  }
180 
181 // create the master GLX context
182  cxMaster = glXCreateContext (dpy, vi, 0, true);
183  if (!cxMaster) {
184  fViewId = -1; // This flags an error.
185  G4cerr << "G4OpenGLXViewer::G4OpenGLXViewer couldn't create context."
186  << G4endl;
187  return;
188  }
189 
190 // New stab at getting a colormap
191 
192  Status status;
193  XStandardColormap *standardCmaps = XAllocStandardColormap ();
194  int i, numCmaps;
195 
196  status = XmuLookupStandardColormap (dpy,
197  vi -> screen,
198  vi -> visualid,
199  vi -> depth,
200  XA_RGB_BEST_MAP,
201  False,
202  True);
203 
204  if (status == 1) {
205  cmap = 0;
206  status = XGetRGBColormaps (dpy,
207  XRootWindow (dpy, vi -> screen),
208  &standardCmaps,
209  &numCmaps,
210  XA_RGB_BEST_MAP);
211  if (status == 1)
212  for (i = 0; i < numCmaps; i++) {
213  if (standardCmaps[i].visualid == vi -> visualid) {
214  cmap = standardCmaps[i].colormap;
215  XFree (standardCmaps);
216  break;
217  }
218  }
219  if (!cmap) {
220  fViewId = -1; // This flags an error.
221  if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
222  G4cerr <<
223  "G4OpenGLXViewer::G4OpenGLXViewer failed to allocate a standard colormap."
224  << G4endl;
225  return;
226  }
227  if (G4VisManager::GetVerbosity() >= G4VisManager::confirmations)
228  G4cout << "Got standard cmap" << G4endl;
229  } else {
230  cmap = XCreateColormap (dpy,
231  XRootWindow(dpy, vi -> screen),
232  vi -> visual,
233  AllocNone);
234  if (G4VisManager::GetVerbosity() >= G4VisManager::confirmations)
235  G4cout << "Created own cmap" << G4endl;
236  }
237 
238  if (!cmap) {
239  fViewId = -1; // This flags an error.
240  if (G4VisManager::GetVerbosity() >= G4VisManager::errors)
241  G4cout << "G4OpenGLXViewer::G4OpenGLXViewer failed to allocate a Colormap."
242  << G4endl;
243  return;
244  }

248 void G4OpenGLXViewer::CreateMainWindow () {
249 
250 // create a window
251  swa.colormap = cmap;
252  swa.border_pixel = 0;
253  swa.event_mask = ExposureMask | ButtonPressMask | StructureNotifyMask;
254  swa.backing_store = WhenMapped;
255 
256  // Window size and position...
257  size_hints = XAllocSizeHints();
258 
259  ResizeWindow(fVP.GetWindowSizeHintX(),fVP.GetWindowSizeHintY());
260 
261  G4int x_origin = fVP.GetWindowAbsoluteLocationHintX(DisplayWidth(dpy, vi -> screen));
262 
263  // FIXME, screen size != window size on MAC, but I don't know have to get the menuBar
264  // size on MAC. L.Garnier 01/2009
265  G4int y_origin = fVP.GetWindowAbsoluteLocationHintY(DisplayHeight(dpy, vi -> screen));
266 
267  size_hints->base_width = getWinWidth();
268  size_hints->base_height = getWinHeight();
269  size_hints->x = x_origin;
270  size_hints->y = y_origin;
271  if (fVP.IsWindowSizeHintX () && fVP.IsWindowLocationHintX () && fVP.IsWindowLocationHintY ()) {
272  size_hints->flags |= PSize | PPosition;
273  } else if (fVP.IsWindowSizeHintX () && !(fVP.IsWindowLocationHintX () || fVP.IsWindowLocationHintY ())) {
274  size_hints->flags |= PSize;
275  } else if ((!fVP.IsWindowSizeHintX ()) && fVP.IsWindowLocationHintX () && fVP.IsWindowLocationHintY ()) {
276  size_hints->flags |= PPosition;
277  }
278  if (G4VisManager::GetVerbosity() >= G4VisManager::confirmations)
279  G4cout << "Window name: " << fName << G4endl;
280  strncpy (charViewName, fName, 99); charViewName[99] = '\0';
281  char *window_name = charViewName;
282  char *icon_name = charViewName;
283  //char tmpatom[] = "XA_WM_NORMAL_HINTS";
284  wm_hints = XAllocWMHints();
285  class_hints = XAllocClassHint();
286 
287  XStringListToTextProperty (&window_name, 1, &windowName);
288  XStringListToTextProperty (&icon_name, 1, &iconName);
289 
290  wm_hints -> initial_state = NormalState;
291  wm_hints -> input = True;
292  wm_hints -> icon_pixmap = icon_pixmap;
293  wm_hints -> flags = StateHint | IconPixmapHint | InputHint;
294 
295  class_hints -> res_name = NewString("G4OpenGL");
296  class_hints -> res_class = NewString("G4OpenGL");
297 
298  win = XCreateWindow (dpy, XRootWindow (dpy, vi -> screen), x_origin,
299  y_origin, getWinWidth(), getWinHeight(), 0, vi -> depth,
300  InputOutput, vi -> visual,
301  CWBorderPixel | CWColormap |
302  CWEventMask | CWBackingStore,
303  &swa);
304 
305  XSetWMProperties (dpy, win, &windowName, &iconName, 0, 0,
306  size_hints, wm_hints, class_hints);
307 
308 // request X to Draw window on screen.
309  XMapWindow (dpy, win);
310 
311 // Wait for window to appear (wait for an "expose" event).
312  XIfEvent (dpy, &event, G4OpenGLXViewerWaitForNotify, (char*) win);
313 
314 // connect the context to a window
315  Bool success = glXMakeCurrent (dpy, win, cxMaster);
316  if (!success) {
317  fViewId = -1; // This flags an error.
318  G4cerr << "G4OpenGLXViewer::G4OpenGLXViewer failed to attach a GLX context."
319  << G4endl;
320  GLint error = GL_NO_ERROR;
321  while ((error = glGetError()) != GL_NO_ERROR) {
322  switch (error) {
323  case GL_INVALID_ENUM :
324  G4cout << "GL Error: GL_INVALID_ENUM" << G4endl;break;
325  case GL_INVALID_VALUE :
326  G4cout << "GL Error: GL_INVALID_VALUE" << G4endl;break;
327  case GL_INVALID_OPERATION :
328  G4cout << "GL Error: GL_INVALID_OPERATION" << G4endl;break;
329  case GL_OUT_OF_MEMORY :
330  G4cout << "GL Error: GL_OUT_OF_MEMORY" << G4endl;break;
331  case GL_STACK_UNDERFLOW :
332  G4cout << "GL Error: GL_STACK_UNDERFLOW" << G4endl;break;
333  case GL_STACK_OVERFLOW :
334  G4cout << "GL Error: GL_STACK_OVERFLOW" << G4endl;break;
335  default :
336  G4cout << "GL Error: " << error << G4endl;break;
337  }
338  }
339  return;
340  }

368  if (!font_base) {
369  G4cerr <<

371  << G4endl;
372  continue;
373  }

375  G4int last = font_info->max_char_or_byte2;

377  G4int width = font_info->max_bounds.width;
378  G4OpenGLFontBaseStore::AddFontBase
379  (this, font_base, i->first, i->second, width);

383 void G4OpenGLXViewer::DrawText(const G4Text& g4text)
384 {
385  if (isGl2psWriting()) {
386 
387  G4OpenGLViewer::DrawText(g4text);
388 
389  } else {
390 
391  G4VSceneHandler::MarkerSizeType sizeType;
392  G4double size = fSceneHandler.GetMarkerSize(g4text,sizeType);
393 
394  const G4OpenGLFontBaseStore::FontInfo& fontInfo =
395  G4OpenGLFontBaseStore::GetFontInfo(this,(int)size);
396  if (fontInfo.fFontBase < 0) {
397  static G4int callCount = 0;
398  ++callCount;
399  //if (callCount <= 10 || callCount%100 == 0) {
400  if (callCount <= 1) {
401  G4cout <<
402  "G4OpenGLXViewer::DrawText: No fonts available for \""
403  << fName <<
404  "\"\n Called with "
405  << g4text
406  << G4endl;
407  }
408  return;
409  }
410 
411  const G4Colour& c = fSceneHandler.GetTextColour(g4text);
412  glColor4d(c.GetRed(),c.GetGreen(),c.GetBlue(),c.GetAlpha());
413 
414  G4Point3D position = g4text.GetPosition();
415 
416  G4String textString = g4text.GetText();
417  const char* textCString = textString.c_str();
418 
419  // Set position for raster-style drawers (X, Xm)
420  glRasterPos3d(position.x(),position.y(),position.z());
421 
422  glPushAttrib(GL_LIST_BIT);
423 
424  // Calculate move for centre and right adjustment
425  G4double span = textString.size() * fontInfo.fWidth;
426  G4double xmove = 0., ymove = 0.;
427  switch (g4text.GetLayout()) {
428  case G4Text::left: break;
429  case G4Text::centre: xmove -= span / 2.; break;
430  case G4Text::right: xmove -= span;
431  }
432 
433  //Add offsets
434  xmove += g4text.GetXOffset();
435  ymove += g4text.GetYOffset();
436 
437  // Do move
438  glBitmap(0,0,0,0,xmove,ymove,0);
439 
440  // Write characters
441  glListBase(fontInfo.fFontBase);
442  glCallLists(strlen(textCString),GL_UNSIGNED_BYTE,(GLubyte*)textCString);
443  glPopAttrib();
444  }
445 }
