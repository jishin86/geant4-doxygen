66 G4HadronicInteraction* G4QGSBuilder::BuildModel()
67 {
68  G4TheoFSGenerator* theQGSModel = new G4TheoFSGenerator(GetName());
69  theQGStringModel = new G4QGSModel< G4QGSParticipants >;
70  theQGSM = new G4QGSMFragmentation();
71  theQGStringDecay = new G4ExcitedStringDecay(theQGSM);
72  theQGStringModel->SetFragmentationModel(theQGStringDecay);
73  theQGSModel->SetHighEnergyGenerator(theQGStringModel);
74 
75  if(quasielFlag) {
76  theQuasiElastic = new G4QuasiElasticChannel();
77  theQGSModel->SetQuasiElasticChannel(theQuasiElastic);
78  }
79 
80  if(!thePreCompound) {
81  thePreCompound = new G4PreCompoundModel(new G4ExcitationHandler());
82  }
83 
84  if(GetName() == "QGSB") {
85  G4BinaryCascade* bic = new G4BinaryCascade();
86  bic->SetDeExcitation(thePreCompound);
87  theQGSModel->SetTransport(bic);
88 
89  } else {
90  G4GeneratorPrecompoundInterface* pint = new G4GeneratorPrecompoundInterface();
91  pint->SetDeExcitation(thePreCompound);
92  theQGSModel->SetTransport(pint);
93  }
94 
95  return theQGSModel;
