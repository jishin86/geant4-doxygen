134 G4AffineTransform::operator *= (const G4AffineTransform& tf)

137  //
138  G4double nrxx=rxx*tf.rxx+rxy*tf.ryx+rxz*tf.rzx;
139  G4double nrxy=rxx*tf.rxy+rxy*tf.ryy+rxz*tf.rzy;
140  G4double nrxz=rxx*tf.rxz+rxy*tf.ryz+rxz*tf.rzz;
141 
142  G4double nryx=ryx*tf.rxx+ryy*tf.ryx+ryz*tf.rzx;
143  G4double nryy=ryx*tf.rxy+ryy*tf.ryy+ryz*tf.rzy;
144  G4double nryz=ryx*tf.rxz+ryy*tf.ryz+ryz*tf.rzz;
145 
146  G4double nrzx=rzx*tf.rxx+rzy*tf.ryx+rzz*tf.rzx;
147  G4double nrzy=rzx*tf.rxy+rzy*tf.ryy+rzz*tf.rzy;
148  G4double nrzz=rzx*tf.rxz+rzy*tf.ryz+rzz*tf.rzz;
149 
150  G4double ntx=tx*tf.rxx+ty*tf.ryx+tz*tf.rzx+tf.tx;
151  G4double nty=tx*tf.rxy+ty*tf.ryy+tz*tf.rzy+tf.ty;
152  G4double ntz=tx*tf.rxz+ty*tf.ryz+tz*tf.rzz+tf.tz;
153 
154  tx=ntx; ty=nty; tz=ntz;
155  rxx=nrxx; rxy=nrxy; rxz=nrxz;
156  ryx=nryx; ryy=nryy; ryz=nryz;
157  rzx=nrzx; rzy=nrzy; rzz=nrzz;
158

171  ryy = tf1.ryx*tf2.rxy + tf1.ryy*tf2.ryy + tf1.ryz*tf2.rzy;
172  ryz = tf1.ryx*tf2.rxz + tf1.ryy*tf2.ryz + tf1.ryz*tf2.rzz;
173 
174  rzx = tf1.rzx*tf2.rxx + tf1.rzy*tf2.ryx + tf1.rzz*tf2.rzx;
175  rzy = tf1.rzx*tf2.rxy + tf1.rzy*tf2.ryy + tf1.rzz*tf2.rzy;
176  rzz = tf1.rzx*tf2.rxz + tf1.rzy*tf2.ryz + tf1.rzz*tf2.rzz;
177 
178  tx = tf1.tx*tf2.rxx + tf1.ty*tf2.ryx + tf1.tz*tf2.rzx + tf2.tx;
179  ty = tf1.tx*tf2.rxy + tf1.ty*tf2.ryy + tf1.tz*tf2.rzy + tf2.ty;
180  tz = tf1.tx*tf2.rxz + tf1.ty*tf2.ryz + tf1.tz*tf2.rzz + tf2.tz;

185 inline G4AffineTransform&
186 G4AffineTransform::InverseProduct( const G4AffineTransform& tf1,
187  const G4AffineTransform& tf2 )
188 {
189  if ((tf2.rxx + tf2.ryy + tf2.rzz) == 3.)
190  {
191  rxx = tf1.rxx;
192  rxy = tf1.rxy;
193  rxz = tf1.rxz;
194 
195  ryx = tf1.ryx;
196  ryy = tf1.ryy;
197  ryz = tf1.ryz;
198 
199  rzx = tf1.rzx;
200  rzy = tf1.rzy;
201  rzz = tf1.rzz;
202 
203  tx = tf1.tx - tf2.tx;
204  ty = tf1.ty - tf2.ty;
205  tz = tf1.tz - tf2.tz;
206  }
207  else
208  {
209  G4double tf1rxx = tf1.rxx, tf1rxy = tf1.rxy, tf1rxz = tf1.rxz;
210  rxx = tf1rxx*tf2.rxx + tf1rxy*tf2.rxy + tf1rxz*tf2.rxz;
211  rxy = tf1rxx*tf2.ryx + tf1rxy*tf2.ryy + tf1rxz*tf2.ryz;
212  rxz = tf1rxx*tf2.rzx + tf1rxy*tf2.rzy + tf1rxz*tf2.rzz;
213 
214  G4double tf1ryx = tf1.ryx, tf1ryy = tf1.ryy, tf1ryz = tf1.ryz;
215  ryx = tf1ryx*tf2.rxx + tf1ryy*tf2.rxy + tf1ryz*tf2.rxz;
216  ryy = tf1ryx*tf2.ryx + tf1ryy*tf2.ryy + tf1ryz*tf2.ryz;
217  ryz = tf1ryx*tf2.rzx + tf1ryy*tf2.rzy + tf1ryz*tf2.rzz;
218 
219  G4double tf1rzx = tf1.rzx, tf1rzy = tf1.rzy, tf1rzz = tf1.rzz;
220  rzx = tf1rzx*tf2.rxx + tf1rzy*tf2.rxy + tf1rzz*tf2.rxz;
221  rzy = tf1rzx*tf2.ryx + tf1rzy*tf2.ryy + tf1rzz*tf2.ryz;
222  rzz = tf1rzx*tf2.rzx + tf1rzy*tf2.rzy + tf1rzz*tf2.rzz;
223 
224  G4double tf1_2tx = tf1.tx - tf2.tx;
225  G4double tf1_2ty = tf1.ty - tf2.ty;
226  G4double tf1_2tz = tf1.tz - tf2.tz;
227  tx = tf1_2tx*tf2.rxx + tf1_2ty*tf2.rxy + tf1_2tz*tf2.rxz;
228  ty = tf1_2tx*tf2.ryx + tf1_2ty*tf2.ryy + tf1_2tz*tf2.ryz;
229  tz = tf1_2tx*tf2.rzx + tf1_2ty*tf2.rzy + tf1_2tz*tf2.rzz;
230  }
231  return *this;
232 }
