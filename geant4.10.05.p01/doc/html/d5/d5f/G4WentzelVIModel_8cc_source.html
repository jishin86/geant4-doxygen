122 void G4WentzelVIModel::Initialise(const G4ParticleDefinition* p,
123  const G4DataVector& cuts)
124 {
125  // reset parameters
126  SetupParticle(p);
127  currentRange = 0.0;
128 
129  if(isCombined) {
130  G4double tet = PolarAngleLimit();
131  if(tet <= 0.0) { cosThetaMax = 1.0; }
132  else if(tet < CLHEP::pi) { cosThetaMax = cos(tet); }
133  }
134  //G4cout << "G4WentzelVIModel::Initialise " << p->GetParticleName()
135  // << " " << this << " " << wokvi << G4endl;
136 
137  wokvi->Initialise(p, cosThetaMax);
138  /*
139  G4cout << "G4WentzelVIModel: " << particle->GetParticleName()
140  << " 1-cos(ThetaLimit)= " << 1 - cosThetaMax
141  << " SingScatFactor= " << ssFactor
142  << G4endl;
143  */
144  currentCuts = &cuts;
145 
146  // set values of some data members
147  fParticleChange = GetParticleChangeForMSC(p);
148 
149  // build second moment table only if transport table is build
150  G4PhysicsTable* table = GetCrossSectionTable();
151  if(useSecondMoment && IsMaster() && table) {
152 
153  //G4cout << "### G4WentzelVIModel::Initialise: build 2nd moment table "
154  // << table << G4endl;
155  fSecondMoments =
156  G4PhysicsTableHelper::PreparePhysicsTable(fSecondMoments);
157  // Access to materials
158  const G4ProductionCutsTable* theCoupleTable =
159  G4ProductionCutsTable::GetProductionCutsTable();
160  size_t numOfCouples = theCoupleTable->GetTableSize();
161 
162  G4bool splineFlag = true;
163  G4PhysicsVector* aVector = nullptr;
164  G4PhysicsVector* bVector = nullptr;
165  G4double emin = std::max(LowEnergyLimit(), LowEnergyActivationLimit());
166  G4double emax = std::min(HighEnergyLimit(), HighEnergyActivationLimit());
167  if(emin < emax) {
168  size_t n = G4EmParameters::Instance()->NumberOfBinsPerDecade()
169  *G4lrint(std::log10(emax/emin));
170  if(n < 3) { n = 3; }
171 
172  for(size_t i=0; i<numOfCouples; ++i) {
173 
174  //G4cout<< "i= " << i << " Flag= " << fSecondMoments->GetFlag(i)
175  // << G4endl;
176  if(fSecondMoments->GetFlag(i)) {
177  DefineMaterial(theCoupleTable->GetMaterialCutsCouple(i));
178 
179  delete (*fSecondMoments)[i];
180  if(!aVector) {
181  aVector = new G4PhysicsLogVector(emin, emax, n);
182  bVector = aVector;
183  } else {
184  bVector = new G4PhysicsVector(*aVector);
185  }
186  for(size_t j=0; j<n; ++j) {
187  G4double e = bVector->Energy(j);
188  bVector->PutValue(j, ComputeSecondMoment(p, e)*e*e);
189  }
190  if(splineFlag) { bVector->FillSecondDerivatives(); }
191  (*fSecondMoments)[i] = bVector;
192  }
193  }
194  }
195  //G4cout << *fSecondMoments << G4endl;
196  }
197 }

227 {

241  cross = wokvi->ComputeTransportCrossSectionPerAtom(cost);

244  G4cout << "G4WentzelVIModel::CS: Z= " << G4int(Z) << " e(MeV)= "<<kinEnergy

246  << " " << particle->GetParticleName() << G4endl;
247  */
248  }
249  return cross;
250 }

266 G4double G4WentzelVIModel::ComputeTruePathLengthLimit(
267  const G4Track& track,
268  G4double& currentMinimalStep)
269 {
270  G4double tlimit = currentMinimalStep;
271  const G4DynamicParticle* dp = track.GetDynamicParticle();
272  G4StepPoint* sp = track.GetStep()->GetPreStepPoint();
273  G4StepStatus stepStatus = sp->GetStepStatus();
274  singleScatteringMode = false;
275 
276  //G4cout << "G4WentzelVIModel::ComputeTruePathLengthLimit stepStatus= "
277  // << stepStatus << " " << track.GetDefinition()->GetParticleName()
278  // << G4endl;
279 
280  // initialisation for each step, lambda may be computed from scratch
281  preKinEnergy = dp->GetKineticEnergy();
282  effKinEnergy = preKinEnergy;
283  DefineMaterial(track.GetMaterialCutsCouple());
284  lambdaeff = GetTransportMeanFreePath(particle,preKinEnergy);
285  currentRange = GetRange(particle,preKinEnergy,currentCouple);
286  cosTetMaxNuc = wokvi->SetupKinematic(preKinEnergy, currentMaterial);
287 
288  //G4cout << "lambdaeff= " << lambdaeff << " Range= " << currentRange
289  // << " tlimit= " << tlimit << " 1-cost= " << 1 - cosTetMaxNuc << G4endl;
290 
291  // extra check for abnormal situation
292  // this check needed to run MSC with eIoni and eBrem inactivated
293  if(tlimit > currentRange) { tlimit = currentRange; }
294 
295  // stop here if small range particle
296  if(tlimit < tlimitminfix) {
297  return ConvertTrueToGeom(tlimit, currentMinimalStep);
298  }
299 
300  // pre step
301  G4double presafety = sp->GetSafety();
302  // far from geometry boundary
303  if(currentRange < presafety) {
304  return ConvertTrueToGeom(tlimit, currentMinimalStep);
305  }
306 
307  // compute presafety again if presafety <= 0 and no boundary
308  // i.e. when it is needed for optimization purposes
309  if(stepStatus != fGeomBoundary && presafety < tlimitminfix) {
310  presafety = ComputeSafety(sp->GetPosition(), tlimit);
311  if(currentRange < presafety) {
312  return ConvertTrueToGeom(tlimit, currentMinimalStep);
313  }
314  }
315  /*
316  G4cout << "e(MeV)= " << preKinEnergy/MeV
317  << " " << particle->GetParticleName()
318  << " CurLimit(mm)= " << tlimit/mm <<" safety(mm)= " << presafety/mm
319  << " R(mm)= " <<currentRange/mm
320  << " L0(mm^-1)= " << lambdaeff*mm
321  << G4endl;
322  */
323  // natural limit for high energy
324  G4double rlimit = std::max(facrange*currentRange,
325  (1.0 - cosTetMaxNuc)*lambdaeff*invssFactor);
326 
327  // low-energy e-
328  if(cosThetaMax > cosTetMaxNuc) {
329  rlimit = std::min(rlimit, facsafety*presafety);
330  }
331 
332  // cut correction
333  G4double rcut = currentCouple->GetProductionCuts()->GetProductionCut(1);
334  //G4cout << "rcut= " << rcut << " rlimit= " << rlimit << " presafety= "
335  // << presafety << " 1-cosThetaMax= " <<1-cosThetaMax
336  //<< " 1-cosTetMaxNuc= " << 1-cosTetMaxNuc << G4endl;
337  if(rcut > rlimit) { rlimit = std::min(rlimit, rcut*sqrt(rlimit/rcut)); }
338 
339  tlimit = std::min(tlimit, rlimit);
340  tlimit = std::max(tlimit, tlimitminfix);
341 
342  // step limit in infinite media
343  tlimit = std::min(tlimit, 50*currentMaterial->GetRadlen()/facgeom);
344 
345  //compute geomlimit and force few steps within a volume
346  if (steppingAlgorithm == fUseDistanceToBoundary
347  && stepStatus == fGeomBoundary) {
348 
349  G4double geomlimit = ComputeGeomLimit(track, presafety, currentRange);
350  tlimit = std::min(tlimit, geomlimit/facgeom);
351  }
352  /*
353  G4cout << particle->GetParticleName() << " e= " << preKinEnergy
354  << " L0= " << lambdaeff << " R= " << currentRange
355  << " tlimit= " << tlimit
356  << " currentMinimalStep= " << currentMinimalStep << G4endl;
357  */
358  return ConvertTrueToGeom(tlimit, currentMinimalStep);
359 }

363 G4double G4WentzelVIModel::ComputeGeomPathLength(G4double truelength)
364 {
365  zPathLength = tPathLength = truelength;
366 
367  // small step use only single scattering
368  cosThetaMin = 1.0;
369  ComputeTransportXSectionPerVolume(cosThetaMin);
370  //G4cout << "xtsec= " << xtsec << " Nav= "
371  // << zPathLength*xtsec << G4endl;
372  if(0.0 >= lambdaeff || G4int(zPathLength*xtsec) < minNCollisions) {
373  singleScatteringMode = true;
374  lambdaeff = DBL_MAX;
375 
376  } else {
377  //G4cout << "ComputeGeomPathLength: tLength= " << tPathLength
378  // << " Leff= " << lambdaeff << G4endl;
379  // small step
380  if(tPathLength < numlimit*lambdaeff) {
381  G4double tau = tPathLength/lambdaeff;
382  zPathLength *= (1.0 - 0.5*tau + tau*tau/6.0);
383 
384  // medium step
385  } else {
386  G4double e1 = 0.0;
387  if(currentRange > tPathLength) {
388  e1 = GetEnergy(particle,currentRange-tPathLength,currentCouple);
389  }
390  effKinEnergy = 0.5*(e1 + preKinEnergy);
391  cosTetMaxNuc = wokvi->SetupKinematic(effKinEnergy, currentMaterial);
392  lambdaeff = GetTransportMeanFreePath(particle, effKinEnergy);
393  //G4cout << " tLength= "<< tPathLength<< " Leff= " << lambdaeff << G4endl;
394  zPathLength = lambdaeff;
395  if(tPathLength*numlimit < lambdaeff) {
396  zPathLength *= (1.0 - G4Exp(-tPathLength/lambdaeff));
397  }
398  }
399  }
400  //G4cout << "Comp.geom: zLength= "<<zPathLength<<" tLength= "
401  // << tPathLength<< " Leff= " << lambdaeff << G4endl;
402  return zPathLength;
403 }

407 G4double G4WentzelVIModel::ComputeTrueStepLength(G4double geomStepLength)
408 {
409  // initialisation of single scattering x-section
410  /*
411  G4cout << "ComputeTrueStepLength: Step= " << geomStepLength
412  << " geomL= " << zPathLength
413  << " Lambda= " << lambdaeff
414  << " 1-cosThetaMaxNuc= " << 1 - cosTetMaxNuc << G4endl;
415  */
416  if(singleScatteringMode) {
417  zPathLength = tPathLength = geomStepLength;
418 
419  } else {
420 
421  // step defined by transportation
422  // change both geom and true step lengths
423  if(geomStepLength < zPathLength) {
424 
425  // single scattering
426  if(G4int(geomStepLength*xtsec) < minNCollisions) {
427  zPathLength = tPathLength = geomStepLength;
428  lambdaeff = DBL_MAX;
429  singleScatteringMode = true;
430 
431  // multiple scattering
432  } else {
433  // small step
434  if(geomStepLength < numlimit*lambdaeff) {
435  G4double tau = geomStepLength/lambdaeff;
436  tPathLength = geomStepLength*(1.0 + 0.5*tau + tau*tau/3.0);
437 
438  // energy correction for a big step
439  } else {
440  tPathLength *= geomStepLength/zPathLength;
441  G4double e1 = 0.0;
442  if(currentRange > tPathLength) {
443  e1 = GetEnergy(particle,currentRange-tPathLength,currentCouple);
444  }
445  effKinEnergy = 0.5*(e1 + preKinEnergy);
446  cosTetMaxNuc = wokvi->SetupKinematic(effKinEnergy, currentMaterial);
447  lambdaeff = GetTransportMeanFreePath(particle, effKinEnergy);
448  G4double tau = geomStepLength/lambdaeff;
449 
450  if(tau < 0.999999) { tPathLength = -lambdaeff*G4Log(1.0 - tau); }
451  else { tPathLength = currentRange; }
452  }
453  zPathLength = geomStepLength;
454  }
455  }
456  }
457  // check of step length
458  // define threshold angle between single and multiple scattering
459  if(!singleScatteringMode) {
460  cosThetaMin -= ssFactor*tPathLength/lambdaeff;
461  xtsec = 0.0;
462 
463  // recompute transport cross section - do not change energy
464  // anymore - cannot be applied for big steps
465  if(cosThetaMin > cosTetMaxNuc) {
466  // new computation
467  G4double cross = ComputeTransportXSectionPerVolume(cosThetaMin);
468  //G4cout << "%%%% cross= " << cross << " xtsec= " << xtsec
469  // << " 1-cosTMin= " << 1.0 - cosThetaMin << G4endl;
470  if(cross <= 0.0) {
471  singleScatteringMode = true;
472  tPathLength = zPathLength;
473  lambdaeff = DBL_MAX;
474  cosThetaMin = 1.0;
475  } else if(xtsec > 0.0) {
476 
477  lambdaeff = 1./cross;
478  G4double tau = zPathLength*cross;
479  if(tau < numlimit) {
480  tPathLength = zPathLength*(1.0 + 0.5*tau + tau*tau/3.0);
481  } else if(tau < 0.999999) {
482  tPathLength = -lambdaeff*G4Log(1.0 - tau);
483  } else {
484  tPathLength = currentRange;
485  }
486  }
487  }
488  }
489  tPathLength = std::min(tPathLength, currentRange);
490  /*
491  G4cout <<"Comp.true: zLength= "<<zPathLength<<" tLength= "<<tPathLength
492  <<" Leff(mm)= "<<lambdaeff/mm<<" sig0(1/mm)= " << xtsec <<G4endl;
493  G4cout << particle->GetParticleName() << " 1-cosThetaMin= " << 1-cosThetaMin
494  << " 1-cosTetMaxNuc= " << 1-cosTetMaxNuc
495  << " e(MeV)= " << preKinEnergy/MeV << " "
496  << " SSmode= " << singleScatteringMode << G4endl;
497  */
498  return tPathLength;
499 }

503 G4ThreeVector&
504 G4WentzelVIModel::SampleScattering(const G4ThreeVector& oldDirection,
505  G4double /*safety*/)
506 {
507  fDisplacement.set(0.0,0.0,0.0);
508  //G4cout << "!##! G4WentzelVIModel::SampleScattering for "
509  // << particle->GetParticleName() << G4endl;
510 
511  // ignore scattering for zero step length and energy below the limit
512  if(preKinEnergy < lowEnergyLimit || tPathLength <= 0.0)
513  { return fDisplacement; }
514 
515  G4double invlambda = 0.0;
516  if(lambdaeff < DBL_MAX) { invlambda = 0.5/lambdaeff; }
517 
518  // use average kinetic energy over the step
519  G4double cut = (*currentCuts)[currentMaterialIndex];
520  if(fixedCut > 0.0) { cut = fixedCut; }
521  /*
522  G4cout <<"SampleScat: E0(MeV)= "<< preKinEnergy/MeV
523  << " Leff= " << lambdaeff <<" sig0(1/mm)= " << xtsec
524  << " xmsc= " << tPathLength*invlambda
525  << " safety= " << safety << G4endl;
526  */
527  // step limit due msc
528  G4int nMscSteps = 1;
529  G4double x0 = tPathLength;
530  G4double z0 = x0*invlambda;
531  //G4double zzz = 0.0;
532  G4double prob2 = 0.0;
533 
534  CLHEP::HepRandomEngine* rndmEngine = G4Random::getTheEngine();
535 
536  // large scattering angle case - two step approach
537  if(!singleScatteringMode) {
538  static const G4double zzmin = 0.05;
539  if(useSecondMoment) {
540  G4double z1 = invlambda*invlambda;
541  G4double z2 = SecondMoment(particle, currentCouple, effKinEnergy);
542  prob2 = (z2 - z1)/(1.5*z1 - z2);
543  }
544  // if(z0 > zzmin && safety > tlimitminfix) {
545  if(z0 > zzmin) {
546  x0 *= 0.5;
547  z0 *= 0.5;
548  nMscSteps = 2;
549  }
550  //if(z0 > zzmin) { zzz = G4Exp(-1.0/z0); }
551  G4double zzz = 0.0;
552  if(z0 > zzmin) {
553  zzz = G4Exp(-1.0/z0);
554  z0 += zzz;
555  prob2 *= (1 + zzz);
556  }
557  prob2 /= (1 + prob2);
558  }
559 
560  // step limit due to single scattering
561  G4double x1 = 2*tPathLength;
562  if(0.0 < xtsec) { x1 = -G4Log(rndmEngine->flat())/xtsec; }
563 
564  // no scattering case
565  if(singleScatteringMode && x1 > tPathLength)
566  { return fDisplacement; }
567 
568  const G4ElementVector* theElementVector =
569  currentMaterial->GetElementVector();
570  G4int nelm = currentMaterial->GetNumberOfElements();
571 
572  // geometry
573  G4double sint, cost, phi;
574  G4ThreeVector temp(0.0,0.0,1.0);
575 
576  // current position and direction relative to the end point
577  // because of magnetic field geometry is computed relatively to the
578  // end point of the step
579  G4ThreeVector dir(0.0,0.0,1.0);
580  fDisplacement.set(0.0,0.0,-zPathLength);
581 
582  G4double mscfac = zPathLength/tPathLength;
583 
584  // start a loop
585  G4double x2 = x0;
586  G4double step, z;
587  G4bool singleScat;
588  /*
589  G4cout << "Start of the loop x1(mm)= " << x1 << " x2(mm)= " << x2
590  << " 1-cost1= " << 1 - cosThetaMin << " SSmode= " << singleScatteringMode
591  << " xtsec= " << xtsec << " Nst= " << nMscSteps << G4endl;
592  */
593  do {
594 
595  //G4cout << "# x1(mm)= "<< x1<< " x2(mm)= "<< x2 << G4endl;
596  // single scattering case

598  fDisplacement += x2*mscfac*dir;
599  break;
600  }
601

603  if(x1 <= x2) {
604  step = x1;
605  singleScat = true;
606  } else {
607  step = x2;
608  singleScat = false;
609  }
610 
611  //G4cout << "# step(mm)= "<< step<< " singlScat= "<< singleScat << G4endl;
612 
613  // new position
614  fDisplacement += step*mscfac*dir;
615 
616  if(singleScat) {
617 
618  // select element
619  G4int i = 0;
620  if(nelm > 1) {
621  G4double qsec = rndmEngine->flat()*xtsec;
622  for (; i<nelm; ++i) { if(xsecn[i] >= qsec) { break; } }
623  }

625  wokvi->SetupTarget((*theElementVector)[i]->GetZasInt(), cut);

631  temp.rotateUz(dir);
632  dir = temp;
633  //G4cout << dir << G4endl;
634

638 
639  // multiple scattering
640  } else {
641  --nMscSteps;
642  x1 -= step;
643  x2 = x0;
644

650  if(isFirst) { z = -G4Log(rndmEngine->flat()); }
651  else { z = G4RandGamma::shoot(rndmEngine, 2.0, 2.0); }
652  z *= z0;
653  // Loop checking, 03-Aug-2015, Vladimir Ivanchenko
654  } while(z > 1.0);
655 
656  cost = 1.0 - 2.0*z/*factCM*/;
657  if(cost > 1.0) { cost = 1.0; }
658  else if(cost < -1.0) { cost =-1.0; }
659  sint = sqrt((1.0 - cost)*(1.0 + cost));
660  phi = twopi*rndmEngine->flat();
661  G4double vx1 = sint*cos(phi);
662  G4double vy1 = sint*sin(phi);
663 
664  // lateral displacement
665  if (latDisplasment) {
666  G4double rms = invsqrt12*sqrt(2*z0);
667  G4double r = x0*mscfac;
668  G4double dx = r*(0.5*vx1 + rms*G4RandGauss::shoot(rndmEngine,0.0,1.0));
669  G4double dy = r*(0.5*vy1 + rms*G4RandGauss::shoot(rndmEngine,0.0,1.0));
670  G4double d = r*r - dx*dx - dy*dy;
671

673  if(d >= 0.0) {

675  temp.rotateUz(dir);
676  fDisplacement += temp;
677  }
678  }
679  // change direction
680  temp.set(vx1,vy1,cost);
681  temp.rotateUz(dir);
682  dir = temp;
683  }
684  // Loop checking, 03-Aug-2015, Vladimir Ivanchenko
685  } while (0 < nMscSteps);
686 
687  dir.rotateUz(oldDirection);
688 
689  //G4cout<<"G4WentzelVIModel sampling is done 1-cost= "<< 1.-dir.z()<<G4endl;
690  // end of sampling -------------------------------
691 
692  fParticleChange->ProposeMomentumDirection(dir);
693 
694  // lateral displacement
695  fDisplacement.rotateUz(oldDirection);
696 
697  /*
698  G4cout << " r(mm)= " << fDisplacement.mag()
699  << " safety= " << safety
700  << " trueStep(mm)= " << tPathLength
701  << " geomStep(mm)= " << zPathLength
702  << " x= " << fDisplacement.x()
703  << " y= " << fDisplacement.y()
704  << " z= " << fDisplacement.z()
705  << G4endl;
706  */
707 
708  //G4cout<< "G4WentzelVIModel::SampleScattering end NewDir= " << dir<< G4endl;
709  return fDisplacement;
710 }

714 G4double G4WentzelVIModel::ComputeTransportXSectionPerVolume(G4double cosTheta)
715 {
716  // prepare recomputation of x-sections
717  const G4ElementVector* theElementVector = currentMaterial->GetElementVector();
718  const G4double* theAtomNumDensityVector =
719  currentMaterial->GetVecNbOfAtomsPerVolume();
720  G4int nelm = currentMaterial->GetNumberOfElements();
721  if(nelm > nelments) {
722  nelments = nelm;
723  xsecn.resize(nelm);
724  prob.resize(nelm);
725  }
726 
727  // check consistency
728  xtsec = 0.0;
729  if(cosTetMaxNuc >= cosTheta) { return 0.0; }
730 
731  G4double cut = (*currentCuts)[currentMaterialIndex];
732  if(fixedCut > 0.0) { cut = fixedCut; }
733 
734  // loop over elements
735  G4double xs = 0.0;
736  for (G4int i=0; i<nelm; ++i) {
737  G4double costm =
738  wokvi->SetupTarget((*theElementVector)[i]->GetZasInt(), cut);
739  G4double density = theAtomNumDensityVector[i];
740 
741  G4double esec = 0.0;
742  if(costm < cosTheta) {
743 
744  // recompute the transport x-section
745  if(1.0 > cosTheta) {
746  xs += density*wokvi->ComputeTransportCrossSectionPerAtom(cosTheta);
747  }
748  // recompute the total x-section
749  G4double nucsec = wokvi->ComputeNuclearCrossSection(cosTheta, costm);
750  esec = wokvi->ComputeElectronCrossSection(cosTheta, costm);
751  nucsec += esec;
752  if(nucsec > 0.0) { esec /= nucsec; }
753  xtsec += nucsec*density;
754  }
755  xsecn[i] = xtsec;
756  prob[i] = esec;
757  //G4cout << i << " xs= " << xs << " xtsec= " << xtsec
758  // << " 1-cosTheta= " << 1-cosTheta
759  // << " 1-cosTetMaxNuc2= " <<1-cosTetMaxNuc2<< G4endl;
760  }
761 
762  //G4cout << "ComputeXS result: xsec(1/mm)= " << xs
763  // << " txsec(1/mm)= " << xtsec <<G4endl;
764  return xs;
765 }
