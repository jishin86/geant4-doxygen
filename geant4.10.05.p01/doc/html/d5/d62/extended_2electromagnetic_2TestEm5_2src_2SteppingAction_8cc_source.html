56 void SteppingAction::UserSteppingAction(const G4Step* aStep)
57 {
58  if (aStep->GetPreStepPoint()->GetTouchableHandle()->GetVolume()
59  != fDetector->GetAbsorber()) return;
60 
61  fEventAction->AddEnergy (aStep->GetTotalEnergyDeposit());
62 
63  G4double charge = aStep->GetTrack()->GetDefinition()->GetPDGCharge();
64  if (charge != 0.) {
65  fEventAction->AddTrakLenCharg(aStep->GetStepLength());
66  fEventAction->CountStepsCharg();
67  } else {
68  fEventAction->AddTrakLenNeutr(aStep->GetStepLength());
69  fEventAction->CountStepsNeutr();
70  }
71 }
