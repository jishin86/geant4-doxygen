60 //
61 // Given values for the variables y[0,..,n-1] and their derivatives
62 // dydx[0,...,n-1] known at x, use the classical 4th Runge-Kutta
63 // method to advance the solution over an interval h and return the
64 // incremented variables as yout[0,...,n-1], which not be a distinct
65 // array from y. The user supplies the routine RightHandSide(x,y,dydx),
66 // which returns derivatives dydx at x. The source is routine rk4 from
67 // NRC p. 712-713 .
68 
69 void
70 G4ClassicalRK4::DumbStepper( const G4double yIn[],
71  const G4double dydx[],
72  G4double h,
73  G4double yOut[])
74 {
75  const G4int nvar = this->GetNumberOfVariables(); // fNumberOfVariables();
76  G4int i;
77  G4double hh = h*0.5 , h6 = h/6.0 ;
78 
79  // Initialise time to t0, needed when it is not updated by the integration.
80  // [ Note: Only for time dependent fields (usually electric)
81  // is it neccessary to integrate the time.]
82  yt[7] = yIn[7];
83  yOut[7] = yIn[7];
84 
85  for(i=0;i<nvar;i++)
86  {
87  yt[i] = yIn[i] + hh*dydx[i] ; // 1st Step K1=h*dydx
88  }
89  RightHandSide(yt,dydxt) ; // 2nd Step K2=h*dydxt
90 
91  for(i=0;i<nvar;i++)
92  {
93  yt[i] = yIn[i] + hh*dydxt[i] ;
94  }
95  RightHandSide(yt,dydxm) ; // 3rd Step K3=h*dydxm
96 
97  for(i=0;i<nvar;i++)
98  {
99  yt[i] = yIn[i] + h*dydxm[i] ;
100  dydxm[i] += dydxt[i] ; // now dydxm=(K2+K3)/h
101  }
102  RightHandSide(yt,dydxt) ; // 4th Step K4=h*dydxt
103 
104  for(i=0;i<nvar;i++) // Final RK4 output
105  {
106  yOut[i] = yIn[i]+h6*(dydx[i]+dydxt[i]+2.0*dydxm[i]); //+K1/6+K4/6+(K2+K3)/3
107  }
108  if ( nvar == 12 ) { NormalisePolarizationVector ( yOut ); }
109 
110 } // end of DumbStepper ....................................................
