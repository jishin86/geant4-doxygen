36 G4INCLXXProtonBuilder::
37 G4INCLXXProtonBuilder()
38 {
39  thePreCompoundMin = 0;
40  thePreCompoundMax = 2*MeV;
41  theMin = 1.0*MeV;
42  theMax = 3.0*GeV;
43  G4HadronicInteraction* p =
44  G4HadronicInteractionRegistry::Instance()->FindModel("PRECO");
45  thePreCompoundModel = static_cast<G4VPreCompoundModel*>(p);
46  if(!thePreCompoundModel) { thePreCompoundModel = new G4PreCompoundModel(); }
47  theModel = new G4INCLXXInterface(thePreCompoundModel);
48 }

50 void G4INCLXXProtonBuilder::
51 Build(G4ProtonInelasticProcess * aP)
52 {
53  thePreCompoundModel->SetMinEnergy(thePreCompoundMin);
54  thePreCompoundModel->SetMaxEnergy(thePreCompoundMax);
55  aP->RegisterMe(thePreCompoundModel);
56  theModel->SetMinEnergy(theMin);
57  theModel->SetMaxEnergy(theMax);
58  aP->RegisterMe(theModel);
59  aP->AddDataSet(new G4BGGNucleonInelasticXS(G4Proton::Proton()));
60 }
