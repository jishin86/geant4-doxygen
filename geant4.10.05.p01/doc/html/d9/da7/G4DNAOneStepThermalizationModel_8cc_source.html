88 double Meesungnoen2002::GetRmean(double k){
89  G4double k_eV = k/eV;
90 
91  if(k_eV>0.1){ // data until 0.2 eV
92  G4double r_mean = 0;
93  for(int8_t i=12; i!=-1 ; --i){
94  r_mean+=gCoeff[12-i]*std::pow(k_eV,i);
95  }
96  r_mean*=CLHEP::nanometer;
97  return r_mean;
98  }
99  return 0;
100 }

102 void GetGaussianPenetrationFromRmean3D(G4double r_mean,
103  G4ThreeVector& displacement)
104 {
105  if(r_mean == 0)
106  {
107  // rare events:
108  // prevent H2O and secondary electron from being placed at the same position
109  displacement = G4RandomDirection() * (1e-3*CLHEP::nanometer);
110  return;
111  }
112 
113  static constexpr double convertRmean3DToSigma1D = 0.62665706865775006;
114  // = sqrt(CLHEP::pi)/pow(2,3./2.)
115 
116  // Use r_mean to build a 3D gaussian
117  const double sigma1D = r_mean * convertRmean3DToSigma1D;
118  displacement = G4ThreeVector(G4RandGauss::shoot(0, sigma1D),
119  G4RandGauss::shoot(0, sigma1D),
120  G4RandGauss::shoot(0, sigma1D));
121 }

140 double Terrisol1990::Get3DStdDeviation(double energy){
141  G4double k_eV = energy/eV;
142  if(k_eV < 0.2){
143  // rare events:
144  // prevent H2O and secondary electron to be at the spot
145  return 1e-3*CLHEP::nanometer;
146  }
147  else if(k_eV == 9.){
148  return gStdDev_T1990[10];
149  }
150  else if(k_eV > 9.){
151  G4ExceptionDescription description;
152  description << "Terrisol1990 is not tabulated for energies greater than 9eV";
153  G4Exception("Terrisol1990::Get3DStdDeviation",
154  "INVALID_ARGUMENT",
155  FatalErrorInArgument,
156  description);
157  }
158 
159  size_t lowBin, upBin;
160 
161  if(k_eV >= 1.){
162  lowBin=std::floor(k_eV)+1;
163  upBin=std::min(lowBin+1, size_t(10));
164  }
165  else{
166  auto it=std::lower_bound(&gEnergies_T1990[0],
167  &gEnergies_T1990[2],
168  k_eV);
169  lowBin = it-&gEnergies_T1990[0];
170  upBin = lowBin+1;
171  }
172 
173  double lowE = gEnergies_T1990[lowBin];
174  double upE = gEnergies_T1990[upBin];
175 
176  double lowS = gStdDev_T1990[lowBin];
177  double upS = gStdDev_T1990[upBin];
178 
179  double tanA = (lowS-upS)/(lowE-upE);
180  double sigma3D = lowS + (k_eV-lowE)*tanA;
181  return sigma3D;
182 }

184 double Terrisol1990::GetRmean(double energy){
185  double sigma3D=Get3DStdDeviation(energy);

187  static constexpr double s2r=1.595769121605731; // = pow(2,3./2.)/sqrt(CLHEP::pi)

193 void Terrisol1990::GetPenetration(G4double energy,
194  G4ThreeVector& displacement){

196 
197  static constexpr double factor = 2.20496999539; // = 1./(3. - 8./CLHEP::pi);

211 G4VEmModel* G4DNASolvationModelFactory::Create(const G4String& penetrationModel)
212 {
213  G4String modelNamePrefix("DNAOneStepThermalizationModel_");
214 
215  if(penetrationModel == "Terrisol1990")
216  {
217  return new G4TDNAOneStepThermalizationModel<DNA::Penetration::Terrisol1990>(G4Electron::Definition(), modelNamePrefix + penetrationModel);
218  }
219  else if(penetrationModel == "Meesungnoen2002")
220  {
221  return new G4TDNAOneStepThermalizationModel<DNA::Penetration::Meesungnoen2002>(G4Electron::Definition(), modelNamePrefix + penetrationModel);
222  }
223  else if(penetrationModel == "Ritchie1994")
224  {
225  return new G4TDNAOneStepThermalizationModel<DNA::Penetration::Ritchie1994>(G4Electron::Definition(), modelNamePrefix + penetrationModel);
226  }
227  else
228  {
229  G4ExceptionDescription description;
230  description << penetrationModel + " is not a valid model name.";
231  G4Exception("G4DNASolvationModelFactory::Create",
232  "INVALID_ARGUMENT",
233  FatalErrorInArgument,
234  description,
235  "Options are: Terrisol1990, Meesungnoen2002, Ritchie1994.");
236  }
237  return nullptr;
238 }
