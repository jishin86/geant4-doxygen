33 // G4PolyhedraSide
34 //
35 // Class description:
36 //
37 // Class implementing a face that represents one segmented side
38 // of a polyhedra:
39 //
40 // G4PolyhedraSide( const G4PolyhedraSideRZ *prevRZ,
41 // const G4PolyhedraSideRZ *tail,
42 // const G4PolyhedraSideRZ *head,
43 // const G4PolyhedraSideRZ *nextRZ,
44 // G4int numSide,
45 // G4double phiStart, G4double phiTotal,
46 // G4bool phiIsOpen, G4bool isAllBehind=false )
47 //
48 // Values for r1,z1 and r2,z2 should be specified in clockwise
49 // order in (r,z).
50 
51 // Author:
