73 void EventAction::SumDeStep(G4int iModule, G4int iLayer, G4int iFiber,
74  G4double deStep )
75 {
76  if (iModule > 0) EtotCalor += deStep;
77 
78  G4int kLayer = 0; G4int kFiber = 0;
79  if (iLayer > 0) {
80  kLayer = (iModule-1)*nbOfLayers + iLayer;
81  EtotLayer[kLayer] += deStep;
82  }
83 
84  if (iFiber > 0) {
85  EvisLayer[kLayer] += deStep;
86  EvisCalor += deStep;
87  kFiber = 1000*kLayer + iFiber;
88  EvisFiber[kFiber] += deStep;
89  }
90 }
