71 void PrimaryGeneratorAction::GeneratePrimaries(G4Event* anEvent)
72 {
73  //particle type
74  if (fParticleGun->GetParticleDefinition() == G4Geantino::Geantino()) {
75  G4int Z = 95, A = 241;
76  G4double ionCharge = 0.*eplus;
77  G4double excitEnergy = 0.*keV;
78 
79  G4ParticleDefinition* ion
80  = G4IonTable::GetIonTable()->GetIon(Z,A,excitEnergy);
81  fParticleGun->SetParticleDefinition(ion);
82  fParticleGun->SetParticleCharge(ionCharge);
83  }
84 
85  //vertex position uniform within the absorber
86  //
87  G4double Rmax = fDetector->GetAbsorRadius();
88  G4double Rmax2 = Rmax*Rmax;
89  G4double R2 = G4UniformRand()*Rmax2;
90  G4double R = std::sqrt(R2);
91 
92  G4double phi = twopi*G4UniformRand();
93  G4double ux = R*std::cos(phi),
94  uy = R*std::sin(phi);
95  G4double uz = (fDetector->GetAbsorLength())*(G4UniformRand() - 0.5);
96 
97  fParticleGun->SetParticlePosition(G4ThreeVector(ux,uy,uz));
98 
99  //distribution uniform in solid angle
100  //
101  G4double cosTheta = 2*G4UniformRand() - 1.;
102  G4double sinTheta = std::sqrt(1. - cosTheta*cosTheta);
103  phi = twopi*G4UniformRand();
104  G4double vx = sinTheta*std::cos(phi),
105  vy = sinTheta*std::sin(phi),
106  vz = cosTheta;
107 
108  fParticleGun->SetParticleMomentumDirection(G4ThreeVector(vx,vy,vz));
109 
110  fParticleGun->GeneratePrimaryVertex(anEvent);
111 }
