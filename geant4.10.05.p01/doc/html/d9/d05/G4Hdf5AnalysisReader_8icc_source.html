46 template <typename T>
47 T* G4Hdf5AnalysisReader::ReadHnImpl(
48  const G4String& htName, const G4String& fileName,
49  const G4String& dirName)
50 {
51  // Get directory
52  auto directory = fFileManager->GetHistoRDirectory(fileName, dirName, false);
53  if ( directory < 0 ) return nullptr;
54 
55  // Read hn
56  T* ht;
57  auto result = tools::hdf5::read_histo(G4cout, directory, htName, ht);
58  if ( ! result ) return nullptr;
59 
60  if ( ! ht ) {
61  G4ExceptionDescription description;
62  description
63  << " "
64  << "Streaming " << htName << " from file " << fileName << " failed.";
65  G4Exception("G4Hdf5AnalysisReader::ReadHnImpl",
66  "Analysis_WR011", JustWarning, description);
67 
68  // when failure, directory and file should be closed
69  // ::H5Gclose(histos);
70  // ::H5Fclose(file);
71 
72  return nullptr;
73  }
74 
75  return ht;
76 }
