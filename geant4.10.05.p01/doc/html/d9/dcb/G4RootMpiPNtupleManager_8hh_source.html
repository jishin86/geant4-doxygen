168 template <typename T>
169 G4int G4RootMpiPNtupleManager::CreateNtupleTColumn(
170  G4int ntupleId, const G4String& name, std::vector<T>* vector)
171 {
172 #ifdef G4VERBOSE
173  if ( fState.GetVerboseL4() ) {
174  G4ExceptionDescription description;
175  description << name << " ntupleId " << ntupleId;
176  fState.GetVerboseL4()->Message("create", "pntuple T column", description);
177  }
178 #endif
179 
180  auto ntupleDescription = GetNtupleDescriptionInFunction(ntupleId, "CreateNtupleTColumn");
181  if ( ! ntupleDescription ) return G4Analysis::kInvalidId;
182 
183  // Save column info in booking
184  auto& ntupleBooking = ntupleDescription->fNtupleBooking;
185  auto index = ntupleBooking.columns().size();
186  if ( ! vector )
187  ntupleBooking.template add_column<T>(name);
188  else
189  ntupleBooking.template add_column<T>(name, *vector);
190 
191  fLockFirstNtupleColumnId = true;
192 
193 #ifdef G4VERBOSE
194  if ( fState.GetVerboseL2() ) {
195  G4ExceptionDescription description;
196  description << name << " ntupleId " << ntupleId;
197  fState.GetVerboseL2()->Message("create", "pntuple T column", description);
198  }
199 #endif
200 
201  return index + fFirstNtupleColumnId;
202 
203 }

206 template <>
207 inline G4bool G4RootMpiPNtupleManager::FillNtupleTColumn(
208  G4int ntupleId, G4int columnId, const std::string& value)
209 {
210  if ( fState.GetIsActivation() && ( ! GetActivation(ntupleId) ) ) {
211  G4cout << "Skipping FillNtupleIColumn for " << ntupleId << G4endl;
212  return false;
213  }
214 
215 #ifdef G4VERBOSE
216  if ( fState.GetVerboseL4() ) {
217  G4ExceptionDescription description;
218  description << " ntupleId " << ntupleId
219  << " columnId " << columnId << " value " << value;
220  fState.GetVerboseL4()->Message("fill", "pntuple T column", description);
221  }
222 #endif
223 
224  auto ntuple = GetNtupleInFunction(ntupleId, "FillNtupleTColumn");
225  if ( ! ntuple ) return false;
226 
227  auto index = columnId - fFirstNtupleColumnId;
228  if ( index < 0 || index >= G4int(ntuple->columns().size()) ) {
229  G4ExceptionDescription description;
230  description << " " << "ntupleId " << ntupleId
231  << " columnId " << columnId << " does not exist.";
232  G4Exception("G4RootNtupleManager::FillNtupleTColumn()",
233  "Analysis_W011", JustWarning, description);
234  return false;
235  }
236 
237  auto icolumn = ntuple->columns()[index];
238  auto column = dynamic_cast<tools::wroot::base_pntuple::column_string* >(icolumn);
239  if ( ! column ) {
240  G4ExceptionDescription description;
241  description << " Column type does not match: "
242  << " ntupleId " << ntupleId
243  << " columnId " << columnId << " value " << value;
244  G4Exception("G4RootNtupleManager:FillNtupleColumn",
245  "Analysis_W011", JustWarning, description);
246  return false;
247  }
248 
249  column->fill(value);
250 
251 #ifdef G4VERBOSE
252  if ( fState.GetVerboseL4() ) {
253  G4ExceptionDescription description;
254  description << " ntupleId " << ntupleId
255  << " columnId " << columnId << " value " << value;
256  fState.GetVerboseL4()->Message("done fill", "pntuple T column", description);
257  }
258 #endif
259  return true;
260 }
