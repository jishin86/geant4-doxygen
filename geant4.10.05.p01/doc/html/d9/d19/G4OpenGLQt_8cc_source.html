49 G4bool G4OpenGLQt::IsUISessionCompatible () const
50 {
51  G4bool isCompatible = false;
52  G4UImanager* ui = G4UImanager::GetUIpointer();
53  G4UIsession* session = ui->GetSession();
54 
55  // If session is a batch session, it may be:
56  // a) this is a batch job (the user has not instantiated any UI session);
57  // b) we are currently processing a UI command, in which case the UI
58  // manager creates a temporary batch session and to find out if there is
59  // a genuine UI session that the user has instantiated we must drill
60  // down through previous sessions to a possible non-batch session.
61  while (G4UIbatch* batch = dynamic_cast<G4UIbatch*>(session)) {
62  session = batch->GetPreviousSession();
63  }
64 
65  // Qt windows are only appropriate in a Qt session.
66  if (session) {
67  // If non-zero, this is the originating non-batch session
68  // The user has instantiated a UI session...
69  if (dynamic_cast<G4UIQt*>(session)) {
70  // ...and it's a G4UIQt session, which is OK.
71  isCompatible = true;
72  }
73  }
74  return isCompatible;
75 }
