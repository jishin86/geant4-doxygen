52 G4QGSBinaryPiKBuilder::
53 G4QGSBinaryPiKBuilder(G4bool quasiElastic)
54 {
55  theMin = 12*GeV;
56  theModel = new G4TheoFSGenerator("QGSB");
57 
58  theStringModel = new G4QGSModel< G4QGSParticipants >;
59  theStringDecay = new G4ExcitedStringDecay(new G4QGSMFragmentation);
60  theStringModel->SetFragmentationModel(theStringDecay);
61 
62  theCascade = new G4BinaryCascade;
63 
64  theModel->SetHighEnergyGenerator(theStringModel);
65  if (quasiElastic)
66  {
67  theQuasiElastic=new G4QuasiElasticChannel;
68  theModel->SetQuasiElasticChannel(theQuasiElastic);
69  } else
70  { theQuasiElastic=0;}
71  theModel->SetTransport(theCascade);
72 }

74 G4QGSBinaryPiKBuilder::
75 ~G4QGSBinaryPiKBuilder()
76 {
77  if ( theQuasiElastic ) delete theQuasiElastic;
78  delete theStringDecay;
79  delete theStringModel;
80 }
