58 G4double G4MinEkineCuts::PostStepGetPhysicalInteractionLength(
59  const G4Track& aTrack,
60  G4double ,
61  G4ForceCondition* condition
62  )
63 {
64  // condition is set to "Not Forced"
65  *condition = NotForced;
66 
67  G4double proposedStep = DBL_MAX;
68  // get the pointer to UserLimits
69  G4UserLimits* pUserLimits = aTrack.GetVolume()->GetLogicalVolume()->GetUserLimits();
70  const G4DynamicParticle* aParticle = aTrack.GetDynamicParticle();
71  G4ParticleDefinition* aParticleDef = aTrack.GetDefinition();
72 
73  if (pUserLimits && aParticleDef->GetPDGCharge() != 0.0) {
74  //min kinetic energy
75  G4double temp = DBL_MAX;
76  G4double eKine = aParticle->GetKineticEnergy();
77  const G4MaterialCutsCouple* couple = aTrack.GetMaterialCutsCouple();
78  G4double eMin = pUserLimits->GetUserMinEkine(aTrack);
79 
80  G4double rangeNow = DBL_MAX;
81 
82  rangeNow = G4EnergyLossTables::GetRange(aParticleDef,eKine,couple);
83 
84  if (eKine < eMin ) {
85  proposedStep = 0.;
86  } else {
87  // charged particles only
88  G4double rangeMin = G4EnergyLossTables::GetRange(aParticleDef,eMin,couple);
89  temp = rangeNow - rangeMin;
90  if (proposedStep > temp) proposedStep = temp;
91  }
92  }
93  return proposedStep;
94 }
