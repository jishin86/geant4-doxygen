92 G4Monopole* G4Monopole::MonopoleDefinition(G4double mass, G4double mCharge,
93  G4double eCharge)
94 {
95  if(!theMonopole) {
96  magCharge = eplus * mCharge / fine_structure_const * 0.5;
97  theMonopole = new G4Monopole(
98  "monopole", mass, 0.0*MeV, eplus*eCharge,
99  0, 0, 0,
100  0, 0, 0,
101  "boson", 0, 0, 0,

105  G4cout << "Monopole is created: m(GeV)= " << theMonopole->GetPDGMass()/GeV
106  << " Qel= " << theMonopole->GetPDGCharge()/eplus
107  << " Qmag= " << magCharge/eplus
108  << G4endl;
