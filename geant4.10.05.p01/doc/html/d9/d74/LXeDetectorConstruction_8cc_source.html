100 
101  G4int polyeth = 1;
102  G4int nC_eth = 2*polyeth;
103  G4int nH_eth = 4*polyeth;
104 
105  //***Elements
106  fH = new G4Element("H", "H", z=1., a=1.01*g/mole);
107  fC = new G4Element("C", "C", z=6., a=12.01*g/mole);
108  fN = new G4Element("N", "N", z=7., a= 14.01*g/mole);

110 
111  //***Materials
112  //Liquid Xenon
113  fLXe = new G4Material("LXe",z=54.,a=131.29*g/mole,density=3.020*g/cm3);
114  //Aluminum

116  //Vacuum
117  fVacuum = new G4Material("Vacuum",z=1.,a=1.01*g/mole,
118  density=universe_mean_density,kStateGas,0.1*kelvin,
119  1.e-19*pascal);
120  //Air
121  fAir = new G4Material("Air", density= 1.29*mg/cm3, 2);
122  fAir->AddElement(fN, 70*perCent);
123  fAir->AddElement(fO, 30*perCent);

126  fGlass->AddElement(fC,91.533*perCent);

135  fPMMA->AddElement(fC,nC_PMMA);

154  assert(sizeof(lxe_RIND) == sizeof(lxe_Energy));
155  G4double lxe_ABSL[] = { 35.*cm, 35.*cm, 35.*cm};
156  assert(sizeof(lxe_ABSL) == sizeof(lxe_Energy));

161  fLXe_mt->AddProperty("ABSLENGTH", lxe_Energy, lxe_ABSL, lxenum);
162  fLXe_mt->AddConstProperty("SCINTILLATIONYIELD",12000./MeV);
163  fLXe_mt->AddConstProperty("RESOLUTIONSCALE",1.0);
164  fLXe_mt->AddConstProperty("FASTTIMECONSTANT",20.*ns);

172 
173  G4double glass_RIND[]={1.49,1.49,1.49};

243 G4VPhysicalVolume* LXeDetectorConstruction::Construct(){
244 
245  if (fExperimentalHall_phys) { return fExperimentalHall_phys; }
246 
247  //The experimental hall walls are all 1m away from housing walls
248  G4double expHall_x = fScint_x+fD_mtl+1.*m;
249  G4double expHall_y = fScint_y+fD_mtl+1.*m;
250  G4double expHall_z = fScint_z+fD_mtl+1.*m;
251 
252  //Create experimental hall
253  fExperimentalHall_box
254  = new G4Box("expHall_box",expHall_x,expHall_y,expHall_z);
255  fExperimentalHall_log = new G4LogicalVolume(fExperimentalHall_box,
256  fVacuum,"expHall_log",0,0,0);
257  fExperimentalHall_phys = new G4PVPlacement(0,G4ThreeVector(),
258  fExperimentalHall_log,"expHall",0,false,0);
259 
260  fExperimentalHall_log->SetVisAttributes(G4VisAttributes::GetInvisible());
261 
262  //Place the main volume
263  if(fMainVolumeOn){
264  fMainVolume
265  = new LXeMainVolume(0,G4ThreeVector(),fExperimentalHall_log,false,0,this);
266  }
267 
268  //Place the WLS slab
269  if(fWLSslab){
270  G4VPhysicalVolume* slab = new LXeWLSSlab(0,G4ThreeVector(0.,0.,
271  -fScint_z/2.-fSlab_z-1.*cm),
272  fExperimentalHall_log,false,0,
273  this);
274 
275  //Surface properties for the WLS slab
276  G4OpticalSurface* scintWrap = new G4OpticalSurface("ScintWrap");
277 
278  new G4LogicalBorderSurface("ScintWrap", slab,
279  fExperimentalHall_phys,
280  scintWrap);
281 
282  scintWrap->SetType(dielectric_metal);
283  scintWrap->SetFinish(polished);
284  scintWrap->SetModel(glisur);
285 
286  G4double pp[] = {2.0*eV, 3.5*eV};
287  const G4int num = sizeof(pp)/sizeof(G4double);
288  G4double reflectivity[] = {1., 1.};
289  assert(sizeof(reflectivity) == sizeof(pp));
290  G4double efficiency[] = {0.0, 0.0};
291  assert(sizeof(efficiency) == sizeof(pp));
292 
293  G4MaterialPropertiesTable* scintWrapProperty
294  = new G4MaterialPropertiesTable();
295 
296  scintWrapProperty->AddProperty("REFLECTIVITY",pp,reflectivity,num);
297  scintWrapProperty->AddProperty("EFFICIENCY",pp,efficiency,num);
298  scintWrap->SetMaterialPropertiesTable(scintWrapProperty);
299  }
300 
301  return fExperimentalHall_phys;
302 }

306 void LXeDetectorConstruction::ConstructSDandField() {
307 
308  if (!fMainVolume) return;
309 
310  // PMT SD
311 
312  if (!fPmt_SD.Get()) {
313  //Created here so it exists as pmts are being placed
314  G4cout << "Construction /LXeDet/pmtSD" << G4endl;
315  LXePMTSD* pmt_SD = new LXePMTSD("/LXeDet/pmtSD");
316  fPmt_SD.Put(pmt_SD);
317 
318  pmt_SD->InitPMTs((fNx*fNy+fNx*fNz+fNy*fNz)*2); //let pmtSD know # of pmts
319  pmt_SD->SetPmtPositions(fMainVolume->GetPmtPositions());
320  }
321  G4SDManager::GetSDMpointer()->AddNewDetector(fPmt_SD.Get());
322  //sensitive detector is not actually on the photocathode.
323  //processHits gets done manually by the stepping action.
324  //It is used to detect when photons hit and get absorbed&detected at the
325  //boundary to the photocathode (which doesnt get done by attaching it to a
326  //logical volume.
327  //It does however need to be attached to something or else it doesnt get
328  //reset at the begining of events
329 
330  SetSensitiveDetector(fMainVolume->GetLogPhotoCath(), fPmt_SD.Get());
331 
332  // Scint SD
333 
334  if (!fScint_SD.Get()) {
335  G4cout << "Construction /LXeDet/scintSD" << G4endl;
336  LXeScintSD* scint_SD = new LXeScintSD("/LXeDet/scintSD");
337  fScint_SD.Put(scint_SD);
338  }
339  G4SDManager::GetSDMpointer()->AddNewDetector(fScint_SD.Get());
340  SetSensitiveDetector(fMainVolume->GetLogScint(), fScint_SD.Get());
341 }
