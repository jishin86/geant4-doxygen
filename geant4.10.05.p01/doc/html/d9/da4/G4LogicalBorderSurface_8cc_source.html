139 G4LogicalBorderSurface*
140 G4LogicalBorderSurface::GetSurface(const G4VPhysicalVolume* vol1,
141  const G4VPhysicalVolume* vol2)
142 {
143  if (theBorderSurfaceTable)
144  {
145  for (size_t i=0; i<theBorderSurfaceTable->size(); i++)
146  {
147  if( ((*theBorderSurfaceTable)[i]->GetVolume1() == vol1) &&
148  ((*theBorderSurfaceTable)[i]->GetVolume2() == vol2) )
149  return (*theBorderSurfaceTable)[i];
150  }
151  }
152  return 0;
