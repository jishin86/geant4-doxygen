117 G4VPhysicalVolume*
118 G4PVDivisionFactory::CreatePVDivision(const G4String& pName,
119  G4LogicalVolume* pLogical,
120  G4LogicalVolume* pMotherLogical,
121  const G4VPVParameterisation* param)
122 {
123  // Create division - with parameterisation
124  // ---
125 
126  // Get parameterisation data
127  //
128  const G4VDivisionParameterisation* divParam
129  = dynamic_cast<const G4VDivisionParameterisation*>(param);
130 
131  if (!divParam)
132  {
133  G4Exception("G4PVDivisionFactory::CreatePVDivision()",
134  "GeomDiv0001", FatalException,
135  "Unexpected parameterisation type!");
136  return 0;
137  }
138  else
139  {
140  EAxis axis = divParam->GetAxis();
141  G4int nofDivisions = divParam->GetNoDiv();
142  G4double width = divParam->GetWidth();
143  G4double offset = divParam->GetOffset();
144 
145  return new G4PVDivision(pName, pLogical, pMotherLogical,
146  axis, nofDivisions, width, offset);
147  }
148 }
