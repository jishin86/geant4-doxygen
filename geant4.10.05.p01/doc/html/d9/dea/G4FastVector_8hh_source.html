62  void Initialize(G4int items)
63  // Normally the pointer ptr points to the stack-array
64  // theArray; only when the number of items is greater
65  // than N, memory is allocated dynamically.
66  {
67  if (ptr != &theArray[0])
68  delete [] ptr;

73  }
