276 void G4VisCommandReviewKeptEvents::SetNewValue (G4UIcommand*, G4String newValue)
277 {
278  static bool reviewing = false;
279  if (reviewing) {
280  G4cout <<
281  "\"/vis/reviewKeptEvents\" not allowed within an already started review."
282  "\n No action taken."
283  << G4endl;
284  return;
285  }
286 
287  G4String& macroFileName = newValue;
288  G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
289 
290  G4RunManager* runManager = G4RunManager::GetRunManager();
291 #ifdef G4MULTITHREADED
292  if(G4Threading::IsMultithreadedApplication())
293  { runManager = G4MTRunManager::GetMasterRunManager(); }
294 #endif
295  const G4Run* run = runManager? runManager->GetCurrentRun(): 0;
296  const std::vector<const G4Event*>* events = run? run->GetEventVector(): 0;
297  size_t nKeptEvents = events? events->size(): 0;
298 
299  if (!nKeptEvents) {
300  if (verbosity >= G4VisManager::errors) {
301  G4cerr <<
302  "ERROR: G4VisCommandReviewKeptEvents::SetNewValue: No kept events,"
303  "\n or kept events not accessible."
304  << G4endl;
305  }
306  return;
307  }
308 
309  G4VViewer* viewer = fpVisManager->GetCurrentViewer();
310  if (!viewer) {
311  if (verbosity >= G4VisManager::errors) {
312  G4cerr <<
313  "ERROR: No current viewer - \"/vis/viewer/list\" to see possibilities."
314  << G4endl;
315  }
316  return;
317  }
318 
319  G4Scene* pScene = fpVisManager->GetCurrentScene();
320  if (!pScene) {
321  if (verbosity >= G4VisManager::errors) {
322  G4cerr << "ERROR: No current scene. Please create one." << G4endl;
323  }
324  return;
325  }
326 
327  G4UImanager* UImanager = G4UImanager::GetUIpointer();
328  G4int keepVerbose = UImanager->GetVerboseLevel();
329  G4int newVerbose(0);
330  if (keepVerbose >= 2 || verbosity >= G4VisManager::confirmations)
331  newVerbose = 2;
332  UImanager->SetVerboseLevel(newVerbose);
333 
334  G4VVisManager* keepConcreteInstance = fpVisManager->GetConcreteInstance();
335  fpVisManager->Enable();
336 
337  // Event by event refreshing...
338  reviewing = true;
339  G4bool currentRefreshAtEndOfEvent = pScene->GetRefreshAtEndOfEvent();
340  pScene->SetRefreshAtEndOfEvent(true);
341  if (macroFileName.empty()) {
342 
343  // Draw to viewer and pause session...
344  G4UIsession* session = UImanager->GetSession();
345  for (size_t i = 0; i < nKeptEvents; ++i) {
346  const G4Event* event = (*events)[i];
347  if (verbosity >= G4VisManager::warnings) {
348  G4cout << "Drawing event : " << event->GetEventID() <<
349  ". At EndOfEvent, enter any command, then \"cont[inue]\"..."
350  << G4endl;
351  static G4bool first = true;
352  if (first) {
353  first = false;
354  G4cout <<
355  " Useful commands might be:"
356  "\n \"/vis/scene/add/trajectories\" if not already added."
357  "\n \"/vis/viewer/...\" to change the view (zoom, set/viewpoint,...)."
358  "\n \"/vis/oglx/printEPS\" to get hard copy."
359  "\n \"/vis/open\" to get alternative viewer."
360  "\n \"/vis/abortReviewKeptEvents\", then \"cont[inue]\", to abort."
361  << G4endl;
362  }
363  }
364  fpVisManager->SetRequestedEvent(event);
365  UImanager->ApplyCommand("/vis/viewer/rebuild");
366  /* The above command forces a rebuild of the scene, including
367  the detector. This is fine for "immediate" viewers - a
368  refresh requires a rebuild anyway. But for "stored mode"
369  viewers, you could, in principle, avoid a rebuild of the
370  detector with something like the following:
371  sceneHandler->ClearTransientStore();
372  viewer->DrawView();
373  sceneHandler->DrawEvent(event);
374  but this causes mayhem for "immediate" viewers because
375  ClearTransientStore issues a DrawView and some curious sort
376  of recursion takes place. For "stored" viewers, the event
377  gets drawn but not the eventID, so something odd is happening
378  there too. This needs further investigation - enhanced
379  features or a complete re-think.
380  */
381  UImanager->ApplyCommand("/vis/viewer/flush");
382  session->PauseSessionStart("EndOfEvent");
383  fpVisManager->SetRequestedEvent(0);
384  if (fpVisManager->GetAbortReviewKeptEvents()) break;
385  }
386  fpVisManager->SetAbortReviewKeptEvents(false);
387 
388  } else {
389 
390  // Execute macro file...
391  for (size_t i = 0; i < nKeptEvents; ++i) {
392  const G4Event* event = (*events)[i];
393  if (verbosity >= G4VisManager::warnings) {
394  G4cout << "Drawing event : " << event->GetEventID()
395  << " with macro file \"" << macroFileName << G4endl;
396  }
397  fpVisManager->SetRequestedEvent(event);
398  UImanager->ApplyCommand("/control/execute " + macroFileName);
399  fpVisManager->SetRequestedEvent(0);
400  }
401  }
402  pScene->SetRefreshAtEndOfEvent(currentRefreshAtEndOfEvent);
403  reviewing = false;
404 
405  if (keepConcreteInstance) fpVisManager->Enable();
406  else fpVisManager->Disable();
407  UImanager->SetVerboseLevel(keepVerbose);
408 }
