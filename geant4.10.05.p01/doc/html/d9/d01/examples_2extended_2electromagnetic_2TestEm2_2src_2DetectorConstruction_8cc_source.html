124 void DetectorConstruction::UpdateParameters()
125 {
126  G4double Radl = fMaterial->GetRadlen();
127  fDLlength = fDLradl*Radl; fDRlength = fDRradl*Radl;
128  fEcalLength = fNLtot*fDLlength; fEcalRadius = fNRtot*fDRlength;
129  if(fSolidEcal) {
130  fSolidEcal->SetOuterRadius(fEcalRadius);
131  fSolidEcal->SetZHalfLength(0.5*fEcalLength);
132  }
133 }
