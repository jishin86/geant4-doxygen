63 void BrachyUserScoreWriter::DumpQuantityToFile(const G4String & psName,
64  const G4String & fileName,
65  const G4String & option)
66 {
67 if(verboseLevel > 0)
68  {G4cout << "BrachyUserScorer-defined DumpQuantityToFile() method is invoked."
69  << G4endl;
70  }
71 
72 // change the option string into lowercase to the case-insensitive.
73 G4String opt = option;
74 std::transform(opt.begin(), opt.end(), opt.begin(), (int (*)(int))(tolower));
75 
76 // confirm the option
77 if(opt.size() == 0) opt = "csv";
78 
79 // open the file
80 std::ofstream ofile(fileName);
81 
82 if(!ofile)
83 {
84  G4cerr << "ERROR : DumpToFile : File open error -> " << fileName << G4endl;
85  return;
86 }
87  ofile << "# mesh name: " << fScoringMesh->GetWorldName() << G4endl;
88 
89 // retrieve the map
90 MeshScoreMap fSMap = fScoringMesh -> GetScoreMap();
91 
92 MeshScoreMap::const_iterator msMapItr = fSMap.find(psName);
93 
94 if(msMapItr == fSMap.end())
95  {
96  G4cerr << "ERROR : DumpToFile : Unknown quantity, \""<< psName
97  << "\"." << G4endl;
98  return;
99  }
100 
101 std::map<G4int, G4StatDouble*> * score = msMapItr -> second-> GetMap();
102 
103 ofile << "# primitive scorer name: " << msMapItr -> first << G4endl;
104 //
105 // Write quantity in the ASCII output file and in brachytherapy.root
106 //
107 ofile << std::setprecision(16); // for double value with 8 bytes
108 
109 for(int x = 0; x < fNMeshSegments[0]; x++) {
110  for(int y = 0; y < fNMeshSegments[1]; y++) {
111  for(int z = 0; z < fNMeshSegments[2]; z++){
112  G4int numberOfVoxel_x = fNMeshSegments[0];
113  G4int numberOfVoxel_y = fNMeshSegments[1];
114  G4int numberOfVoxel_z =fNMeshSegments[2];
115  // If the voxel width is changed in the macro file,
116  // the voxel width variable must be updated
117  G4double voxelWidth = 0.25 *CLHEP::mm;
118  //
119  G4double xx = ( - numberOfVoxel_x + 1+ 2*x )* voxelWidth/2;
120  G4double yy = ( - numberOfVoxel_y + 1+ 2*y )* voxelWidth/2;
121  G4double zz = ( - numberOfVoxel_z + 1+ 2*z )* voxelWidth/2;
122  G4int idx = GetIndex(x, y, z);
123  std::map<G4int, G4StatDouble*>::iterator value = score -> find(idx);
124 
125  if (value != score -> end())
126  {
127  // Print in the ASCII output file the information
128 
129  ofile << xx << " " << yy << " " << zz <<" "
130  <<(value->second->sum_wx())/CLHEP::keV << G4endl;
131 
132 #ifdef ANALYSIS_USE
133  // Save the same information in the output analysis file
134  BrachyAnalysisManager* analysis = BrachyAnalysisManager::GetInstance();
135 
136  if(zz> -0.125 *CLHEP::mm && zz < 0.125*CLHEP::mm)
137  analysis -> FillH2WithEnergyDeposition(xx,yy,
138  (value->second->sum_wx())/CLHEP::keV);
139 #endif
140 }}}}
141 
142 ofile << std::setprecision(6);
143 
144 // Close the output ASCII file
145 ofile.close();
146 }
