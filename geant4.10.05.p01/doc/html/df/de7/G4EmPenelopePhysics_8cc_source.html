137 G4EmPenelopePhysics::G4EmPenelopePhysics(G4int ver, const G4String&)
138  : G4VPhysicsConstructor("G4EmPenelope"), verbose(ver)
139 {
140  G4EmParameters* param = G4EmParameters::Instance();
141  param->SetDefaults();
142  param->SetVerbose(verbose);
143  param->SetMinEnergy(100*eV);
144  param->SetLowestElectronEnergy(100*eV);
145  param->SetNumberOfBinsPerDecade(20);
146  param->SetStepFunction(0.2, 10*um);
147  param->SetStepFunctionMuHad(0.2, 50*um);
148  param->SetUseMottCorrection(true);
149  param->SetMscStepLimitType(fUseSafetyPlus);
150  param->SetMscSkin(3);
151  param->SetMscRangeFactor(0.08);
152  param->SetMuHadLateralDisplacement(true);
153  param->SetFluo(true);
154  //param->SetAugerCascade(true);
155  param->SetPIXEElectronCrossSectionModel("Penelope");
156  SetPhysicsType(bElectromagnetic);
157 }

197 void G4EmPenelopePhysics::ConstructProcess()
198 {
199  if(verbose > 1) {
200  G4cout << "### " << GetPhysicsName() << " Construct Processes " << G4endl;
201  }
202  G4PhysicsListHelper* ph = G4PhysicsListHelper::GetPhysicsListHelper();
203  G4LossTableManager* man = G4LossTableManager::Instance();
204 
205  // muon & hadron bremsstrahlung and pair production
206  G4MuBremsstrahlung* mub = new G4MuBremsstrahlung();
207  G4MuPairProduction* mup = new G4MuPairProduction();
208  G4hBremsstrahlung* pib = new G4hBremsstrahlung();
209  G4hPairProduction* pip = new G4hPairProduction();
210  G4hBremsstrahlung* kb = new G4hBremsstrahlung();
211  G4hPairProduction* kp = new G4hPairProduction();
212  G4hBremsstrahlung* pb = new G4hBremsstrahlung();
213  G4hPairProduction* pp = new G4hPairProduction();
214  G4ePairProduction* ee = new G4ePairProduction();
215 
216  // muon & hadron multiple scattering
217  G4MuMultipleScattering* mumsc = new G4MuMultipleScattering();
218  mumsc->SetEmModel(new G4WentzelVIModel());
219  G4CoulombScattering* muss = new G4CoulombScattering();
220 
221  G4hMultipleScattering* pimsc = new G4hMultipleScattering();
222  pimsc->SetEmModel(new G4WentzelVIModel());
223  G4CoulombScattering* piss = new G4CoulombScattering();
224 
225  G4hMultipleScattering* kmsc = new G4hMultipleScattering();
226  kmsc->SetEmModel(new G4WentzelVIModel());
227  G4CoulombScattering* kss = new G4CoulombScattering();
228 
229  G4hMultipleScattering* hmsc = new G4hMultipleScattering("ionmsc");
230 
231  // high energy limit for e+- scattering models
232  G4double highEnergyLimit = G4EmParameters::Instance()->MscEnergyLimit();
233 
234  // nuclear stopping
235  G4NuclearStopping* pnuc = new G4NuclearStopping();
236 
237  //Applicability range for Penelope models
238  //for higher energies, the Standard models are used
239  G4double PenelopeHighEnergyLimit = 1.0*GeV;
240 
241  // Add Penelope EM Processes
242  G4ParticleTable* table = G4ParticleTable::GetParticleTable();
243  for(const auto& particleName : partList.PartNames()) {
244  G4ParticleDefinition* particle = table->FindParticle(particleName);
245  if (!particle) { continue; }
246  if (particleName == "gamma") {
247 
248  //Photo-electric effect
249  G4PhotoElectricEffect* thePhotoElectricEffect = new G4PhotoElectricEffect();
250  thePhotoElectricEffect->SetEmModel(new G4PEEffectFluoModel());
251  G4PenelopePhotoElectricModel* thePEPenelopeModel = new
252  G4PenelopePhotoElectricModel();
253  thePEPenelopeModel->SetHighEnergyLimit(PenelopeHighEnergyLimit);
254  thePhotoElectricEffect->AddEmModel(0, thePEPenelopeModel);
255 
256  //Compton scattering
257  G4ComptonScattering* theComptonScattering = new G4ComptonScattering();
258  G4PenelopeComptonModel* theComptonPenelopeModel =
259  new G4PenelopeComptonModel();
260  theComptonScattering->SetEmModel(new G4KleinNishinaModel());
261  theComptonPenelopeModel->SetHighEnergyLimit(PenelopeHighEnergyLimit);
262  theComptonScattering->AddEmModel(0, theComptonPenelopeModel);
263 
264  //Gamma conversion
265  G4GammaConversion* theGammaConversion = new G4GammaConversion();
266  G4PenelopeGammaConversionModel* theGCPenelopeModel =
267  new G4PenelopeGammaConversionModel();
268  theGCPenelopeModel->SetHighEnergyLimit(PenelopeHighEnergyLimit);
269  theGammaConversion->AddEmModel(0, theGCPenelopeModel);
270 
271  //Rayleigh scattering
272  G4RayleighScattering* theRayleigh = new G4RayleighScattering();
273  G4PenelopeRayleighModel* theRayleighPenelopeModel =
274  new G4PenelopeRayleighModel();
275  theRayleigh->SetEmModel(theRayleighPenelopeModel);
276 
277  ph->RegisterProcess(thePhotoElectricEffect, particle);
278  ph->RegisterProcess(theComptonScattering, particle);
279  ph->RegisterProcess(theGammaConversion, particle);
280  ph->RegisterProcess(theRayleigh, particle);
281 
282  } else if (particleName == "e-") {
283 
284  // multiple scattering
285  G4eMultipleScattering* msc = new G4eMultipleScattering;
286  G4GoudsmitSaundersonMscModel* msc1 = new G4GoudsmitSaundersonMscModel();
287  G4WentzelVIModel* msc2 = new G4WentzelVIModel();
288  msc1->SetHighEnergyLimit(highEnergyLimit);
289  msc2->SetLowEnergyLimit(highEnergyLimit);
290  msc->SetEmModel(msc1);
291  msc->SetEmModel(msc2);
292 
293  G4eCoulombScatteringModel* ssm = new G4eCoulombScatteringModel();
294  G4CoulombScattering* ss = new G4CoulombScattering();
295  ss->SetEmModel(ssm);
296  ss->SetMinKinEnergy(highEnergyLimit);
297  ssm->SetLowEnergyLimit(highEnergyLimit);
298  ssm->SetActivationLowEnergyLimit(highEnergyLimit);
299 
300  //Ionisation
301  G4eIonisation* eioni = new G4eIonisation();
302  G4PenelopeIonisationModel* theIoniPenelope =
303  new G4PenelopeIonisationModel();
304  theIoniPenelope->SetHighEnergyLimit(PenelopeHighEnergyLimit);
305  eioni->AddEmModel(0,theIoniPenelope,new G4UniversalFluctuation());
306 
307  //Bremsstrahlung
308  G4eBremsstrahlung* brem = new G4eBremsstrahlung();
309  G4PenelopeBremsstrahlungModel* theBremPenelope = new
310  G4PenelopeBremsstrahlungModel();
311  theBremPenelope->SetHighEnergyLimit(PenelopeHighEnergyLimit);
312  brem->SetEmModel(theBremPenelope);
313 
314  // register processes
315  ph->RegisterProcess(msc, particle);
316  ph->RegisterProcess(eioni, particle);
317  ph->RegisterProcess(brem, particle);
318  ph->RegisterProcess(ee, particle);
319  ph->RegisterProcess(ss, particle);
320 
321  } else if (particleName == "e+") {
322 
323  // multiple scattering
324  G4eMultipleScattering* msc = new G4eMultipleScattering;
325  G4GoudsmitSaundersonMscModel* msc1 = new G4GoudsmitSaundersonMscModel();
326  G4WentzelVIModel* msc2 = new G4WentzelVIModel();
327  msc1->SetHighEnergyLimit(highEnergyLimit);
328  msc2->SetLowEnergyLimit(highEnergyLimit);
329  msc->SetEmModel(msc1);
330  msc->SetEmModel(msc2);
331 
332  G4eCoulombScatteringModel* ssm = new G4eCoulombScatteringModel();
333  G4CoulombScattering* ss = new G4CoulombScattering();
334  ss->SetEmModel(ssm);
335  ss->SetMinKinEnergy(highEnergyLimit);
336  ssm->SetLowEnergyLimit(highEnergyLimit);
337  ssm->SetActivationLowEnergyLimit(highEnergyLimit);
338 
339  //Ionisation
340  G4eIonisation* eioni = new G4eIonisation();
341  G4PenelopeIonisationModel* theIoniPenelope =
342  new G4PenelopeIonisationModel();
343  theIoniPenelope->SetHighEnergyLimit(PenelopeHighEnergyLimit);
344  eioni->AddEmModel(0,theIoniPenelope,new G4UniversalFluctuation());
345 
346  //Bremsstrahlung
347  G4eBremsstrahlung* brem = new G4eBremsstrahlung();
348  G4PenelopeBremsstrahlungModel* theBremPenelope = new
349  G4PenelopeBremsstrahlungModel();
350  theBremPenelope->SetHighEnergyLimit(PenelopeHighEnergyLimit);
351  brem->SetEmModel(theBremPenelope);
352 
353  //Annihilation
354  G4eplusAnnihilation* anni = new G4eplusAnnihilation();
355  G4PenelopeAnnihilationModel* theAnnPenelope = new
356  G4PenelopeAnnihilationModel();
357  theAnnPenelope->SetHighEnergyLimit(PenelopeHighEnergyLimit);
358  anni->AddEmModel(0, theAnnPenelope);
359 
360  // register processes
361  ph->RegisterProcess(msc, particle);
362  ph->RegisterProcess(eioni, particle);
363  ph->RegisterProcess(brem, particle);
364  ph->RegisterProcess(ee, particle);
365  ph->RegisterProcess(anni, particle);
366  ph->RegisterProcess(ss, particle);
367 
368  } else if (particleName == "mu+" ||
369  particleName == "mu-" ) {
370 
371  G4MuIonisation* muIoni = new G4MuIonisation();
372 
373  ph->RegisterProcess(mumsc, particle);
374  ph->RegisterProcess(muIoni, particle);
375  ph->RegisterProcess(mub, particle);
376  ph->RegisterProcess(mup, particle);
377  ph->RegisterProcess(muss, particle);
378 
379  } else if (particleName == "alpha" ||
380  particleName == "He3" ) {
381 
382  G4hMultipleScattering* msc = new G4hMultipleScattering();
383  G4ionIonisation* ionIoni = new G4ionIonisation();
384 
385  ph->RegisterProcess(msc, particle);
386  ph->RegisterProcess(ionIoni, particle);
387  ph->RegisterProcess(pnuc, particle);
388 
389  } else if (particleName == "GenericIon") {
390 
391  G4ionIonisation* ionIoni = new G4ionIonisation();
392  ionIoni->SetEmModel(new G4IonParametrisedLossModel());
393  ionIoni->SetStepFunction(0.1, 1*um);
394 
395  ph->RegisterProcess(hmsc, particle);
396  ph->RegisterProcess(ionIoni, particle);
397  ph->RegisterProcess(pnuc, particle);
398 
399  } else if (particleName == "pi+" ||
400  particleName == "pi-" ) {
401 
402  G4hIonisation* hIoni = new G4hIonisation();
403 
404  ph->RegisterProcess(pimsc, particle);
405  ph->RegisterProcess(hIoni, particle);
406  ph->RegisterProcess(pib, particle);
407  ph->RegisterProcess(pip, particle);
408  ph->RegisterProcess(piss, particle);
409 
410  } else if (particleName == "kaon+" ||
411  particleName == "kaon-" ) {
412 
413  G4hIonisation* hIoni = new G4hIonisation();
414 
415  ph->RegisterProcess(kmsc, particle);
416  ph->RegisterProcess(hIoni, particle);
417  ph->RegisterProcess(kb, particle);
418  ph->RegisterProcess(kp, particle);
419  ph->RegisterProcess(kss, particle);
420 
421  } else if (particleName == "proton" ||
422  particleName == "anti_proton") {
423 
424  G4hMultipleScattering* pmsc = new G4hMultipleScattering();
425  pmsc->SetEmModel(new G4WentzelVIModel());
426  G4hIonisation* hIoni = new G4hIonisation();
427 
428  ph->RegisterProcess(pmsc, particle);
429  ph->RegisterProcess(hIoni, particle);
430  ph->RegisterProcess(pb, particle);
431  ph->RegisterProcess(pp, particle);
432  ph->RegisterProcess(new G4CoulombScattering(), particle);
433 
434  } else if (particleName == "B+" ||
435  particleName == "B-" ||
436  particleName == "D+" ||
437  particleName == "D-" ||
438  particleName == "Ds+" ||
439  particleName == "Ds-" ||
440  particleName == "anti_He3" ||
441  particleName == "anti_alpha" ||
442  particleName == "anti_deuteron" ||
443  particleName == "anti_lambda_c+" ||
444  particleName == "anti_omega-" ||
445  particleName == "anti_sigma_c+" ||
446  particleName == "anti_sigma_c++" ||
447  particleName == "anti_sigma+" ||
448  particleName == "anti_sigma-" ||
449  particleName == "anti_triton" ||
450  particleName == "anti_xi_c+" ||
451  particleName == "anti_xi-" ||
452  particleName == "deuteron" ||
453  particleName == "lambda_c+" ||
454  particleName == "omega-" ||
455  particleName == "sigma_c+" ||
456  particleName == "sigma_c++" ||
457  particleName == "sigma+" ||
458  particleName == "sigma-" ||
459  particleName == "tau+" ||
460  particleName == "tau-" ||
461  particleName == "triton" ||
462  particleName == "xi_c+" ||
463  particleName == "xi-" ) {
464 
465  ph->RegisterProcess(hmsc, particle);
466  ph->RegisterProcess(new G4hIonisation(), particle);
467  ph->RegisterProcess(pnuc, particle);
468  }
469  }
470 
471  // Nuclear stopping
472  pnuc->SetMaxKinEnergy(MeV);
473 
474  // Deexcitation
475  //
476  G4VAtomDeexcitation* deexcitation = new G4UAtomicDeexcitation();
477  man->SetAtomDeexcitation(deexcitation);
478 
479  G4EmModelActivator mact(GetPhysicsName());
480 }
