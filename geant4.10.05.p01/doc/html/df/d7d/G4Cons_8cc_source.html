207 EInside G4Cons::Inside(const G4ThreeVector& p) const
208 {
209  G4double r2, rl, rh, pPhi, tolRMin, tolRMax; // rh2, rl2 ;
210  EInside in;
211 
212  if (std::fabs(p.z()) > fDz + halfCarTolerance ) { return in = kOutside; }
213  else if(std::fabs(p.z()) >= fDz - halfCarTolerance ) { in = kSurface; }
214  else { in = kInside; }
215 
216  r2 = p.x()*p.x() + p.y()*p.y() ;
217  rl = 0.5*(fRmin2*(p.z() + fDz) + fRmin1*(fDz - p.z()))/fDz ;
218  rh = 0.5*(fRmax2*(p.z()+fDz)+fRmax1*(fDz-p.z()))/fDz;
219 
220  // rh2 = rh*rh;
221 
222  tolRMin = rl - halfRadTolerance;
223  if ( tolRMin < 0 ) { tolRMin = 0; }
224  tolRMax = rh + halfRadTolerance;
225 
226  if ( (r2<tolRMin*tolRMin) || (r2>tolRMax*tolRMax) ) { return in = kOutside; }
227 
228  if (rl) { tolRMin = rl + halfRadTolerance; }
229  else { tolRMin = 0.0; }
230  tolRMax = rh - halfRadTolerance;
231 
232  if (in == kInside) // else it's kSurface already
233  {
234  if ( (r2 < tolRMin*tolRMin) || (r2 >= tolRMax*tolRMax) ) { in = kSurface; }
235  }
236  if ( !fPhiFullCone && ((p.x() != 0.0) || (p.y() != 0.0)) )
237  {
238  pPhi = std::atan2(p.y(),p.x()) ;
239 
240  if ( pPhi < fSPhi - halfAngTolerance ) { pPhi += twopi; }
241  else if ( pPhi > fSPhi + fDPhi + halfAngTolerance ) { pPhi -= twopi; }
242 
243  if ( (pPhi < fSPhi - halfAngTolerance) ||
244  (pPhi > fSPhi + fDPhi + halfAngTolerance) ) { return in = kOutside; }
245 
246  else if (in == kInside) // else it's kSurface anyway already
247  {
248  if ( (pPhi < fSPhi + halfAngTolerance) ||
249  (pPhi > fSPhi + fDPhi - halfAngTolerance) ) { in = kSurface; }
250  }
251  }
252  else if ( !fPhiFullCone ) { in = kSurface; }
253 
254  return in ;
255 }

273 void G4Cons::BoundingLimits(G4ThreeVector& pMin, G4ThreeVector& pMax) const
274 {
275  G4double rmin = std::min(GetInnerRadiusMinusZ(),GetInnerRadiusPlusZ());
276  G4double rmax = std::max(GetOuterRadiusMinusZ(),GetOuterRadiusPlusZ());
277  G4double dz = GetZHalfLength();
278 
279  // Find bounding box
280  //
281  if (GetDeltaPhiAngle() < twopi)
282  {
283  G4TwoVector vmin,vmax;
284  G4GeomTools::DiskExtent(rmin,rmax,
285  GetSinStartPhi(),GetCosStartPhi(),
286  GetSinEndPhi(),GetCosEndPhi(),
287  vmin,vmax);
288  pMin.set(vmin.x(),vmin.y(),-dz);
289  pMax.set(vmax.x(),vmax.y(), dz);
290  }
291  else
292  {
293  pMin.set(-rmax,-rmax,-dz);
294  pMax.set( rmax, rmax, dz);
295  }
296 
297  // Check correctness of the bounding box
298  //
299  if (pMin.x() >= pMax.x() || pMin.y() >= pMax.y() || pMin.z() >= pMax.z())
300  {
301  std::ostringstream message;
302  message << "Bad bounding box (min >= max) for solid: "
303  << GetName() << " !"
304  << "\npMin = " << pMin
305  << "\npMax = " << pMax;
306  G4Exception("G4Cons::BoundingLimits()", "GeomMgt0001",
307  JustWarning, message);
308  DumpInfo();

316 G4bool G4Cons::CalculateExtent( const EAxis pAxis,
317  const G4VoxelLimits& pVoxelLimit,
318  const G4AffineTransform& pTransform,
319  G4double& pMin,
320  G4double& pMax ) const
321 {
322  G4ThreeVector bmin, bmax;
323  G4bool exist;
324 
325  // Get bounding box
326  BoundingLimits(bmin,bmax);
327 
328  // Check bounding box
329  G4BoundingEnvelope bbox(bmin,bmax);
330 #ifdef G4BBOX_EXTENT
331  if (true) return bbox.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
332 #endif
333  if (bbox.BoundingBoxVsVoxelLimits(pAxis,pVoxelLimit,pTransform,pMin,pMax))
334  {
335  return exist = (pMin < pMax) ? true : false;
336  }
337 
338  // Get parameters of the solid
339  G4double rmin1 = GetInnerRadiusMinusZ();
340  G4double rmax1 = GetOuterRadiusMinusZ();
341  G4double rmin2 = GetInnerRadiusPlusZ();
342  G4double rmax2 = GetOuterRadiusPlusZ();
343  G4double dz = GetZHalfLength();
344  G4double dphi = GetDeltaPhiAngle();
345 
346  // Find bounding envelope and calculate extent
347  //
348  const G4int NSTEPS = 24; // number of steps for whole circle
349  G4double astep = twopi/NSTEPS; // max angle for one step
350  G4int ksteps = (dphi <= astep) ? 1 : (G4int)((dphi-deg)/astep) + 1;
351  G4double ang = dphi/ksteps;
352 
353  G4double sinHalf = std::sin(0.5*ang);
354  G4double cosHalf = std::cos(0.5*ang);
355  G4double sinStep = 2.*sinHalf*cosHalf;
356  G4double cosStep = 1. - 2.*sinHalf*sinHalf;
357  G4double rext1 = rmax1/cosHalf;
358  G4double rext2 = rmax2/cosHalf;
359 
360  // bounding envelope for full cone without hole consists of two polygons,
361  // in other cases it is a sequence of quadrilaterals
362  if (rmin1 == 0 && rmin2 == 0 && dphi == twopi)
363  {
364  G4double sinCur = sinHalf;
365  G4double cosCur = cosHalf;
366 
367  G4ThreeVectorList baseA(NSTEPS),baseB(NSTEPS);
368  for (G4int k=0; k<NSTEPS; ++k)
369  {
370  baseA[k].set(rext1*cosCur,rext1*sinCur,-dz);
371  baseB[k].set(rext2*cosCur,rext2*sinCur, dz);
372 
373  G4double sinTmp = sinCur;
374  sinCur = sinCur*cosStep + cosCur*sinStep;
375  cosCur = cosCur*cosStep - sinTmp*sinStep;
376  }
377  std::vector<const G4ThreeVectorList *> polygons(2);
378  polygons[0] = &baseA;
379  polygons[1] = &baseB;
380  G4BoundingEnvelope benv(bmin,bmax,polygons);
381  exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
382  }
383  else
384  {
385  G4double sinStart = GetSinStartPhi();
386  G4double cosStart = GetCosStartPhi();
387  G4double sinEnd = GetSinEndPhi();
388  G4double cosEnd = GetCosEndPhi();
389  G4double sinCur = sinStart*cosHalf + cosStart*sinHalf;
390  G4double cosCur = cosStart*cosHalf - sinStart*sinHalf;
391 
392  // set quadrilaterals
393  G4ThreeVectorList pols[NSTEPS+2];
394  for (G4int k=0; k<ksteps+2; ++k) pols[k].resize(4);
395  pols[0][0].set(rmin2*cosStart,rmin2*sinStart, dz);
396  pols[0][1].set(rmin1*cosStart,rmin1*sinStart,-dz);
397  pols[0][2].set(rmax1*cosStart,rmax1*sinStart,-dz);
398  pols[0][3].set(rmax2*cosStart,rmax2*sinStart, dz);
399  for (G4int k=1; k<ksteps+1; ++k)
400  {
401  pols[k][0].set(rmin2*cosCur,rmin2*sinCur, dz);
402  pols[k][1].set(rmin1*cosCur,rmin1*sinCur,-dz);
403  pols[k][2].set(rext1*cosCur,rext1*sinCur,-dz);
404  pols[k][3].set(rext2*cosCur,rext2*sinCur, dz);
405 
406  G4double sinTmp = sinCur;
407  sinCur = sinCur*cosStep + cosCur*sinStep;
408  cosCur = cosCur*cosStep - sinTmp*sinStep;
409  }
410  pols[ksteps+1][0].set(rmin2*cosEnd,rmin2*sinEnd, dz);
411  pols[ksteps+1][1].set(rmin1*cosEnd,rmin1*sinEnd,-dz);
412  pols[ksteps+1][2].set(rmax1*cosEnd,rmax1*sinEnd,-dz);
413  pols[ksteps+1][3].set(rmax2*cosEnd,rmax2*sinEnd, dz);
414 
415  // set envelope and calculate extent
416  std::vector<const G4ThreeVectorList *> polygons;
417  polygons.resize(ksteps+2);
418  for (G4int k=0; k<ksteps+2; ++k) polygons[k] = &pols[k];
419  G4BoundingEnvelope benv(bmin,bmax,polygons);
420  exist = benv.CalculateExtent(pAxis,pVoxelLimit,pTransform,pMin,pMax);
421  }
422  return exist;
423 }

427 // Return unit normal of surface closest to p
428 // - note if point on z axis, ignore phi divided sides
429 // - unsafe if point close to z axis a rmin=0 - no explicit checks
430 
431 G4ThreeVector G4Cons::SurfaceNormal( const G4ThreeVector& p) const
432 {
433  G4int noSurfaces = 0;
434  G4double rho, pPhi;
435  G4double distZ, distRMin, distRMax;
436  G4double distSPhi = kInfinity, distEPhi = kInfinity;
437  G4double tanRMin, secRMin, pRMin, widRMin;
438  G4double tanRMax, secRMax, pRMax, widRMax;
439 
440  G4ThreeVector norm, sumnorm(0.,0.,0.), nZ = G4ThreeVector(0.,0.,1.);
441  G4ThreeVector nR, nr(0.,0.,0.), nPs, nPe;
442 
443  distZ = std::fabs(std::fabs(p.z()) - fDz);
444  rho = std::sqrt(p.x()*p.x() + p.y()*p.y());
445 
446  tanRMin = (fRmin2 - fRmin1)*0.5/fDz;
447  secRMin = std::sqrt(1 + tanRMin*tanRMin);
448  pRMin = rho - p.z()*tanRMin;
449  widRMin = fRmin2 - fDz*tanRMin;
450  distRMin = std::fabs(pRMin - widRMin)/secRMin;
451 
452  tanRMax = (fRmax2 - fRmax1)*0.5/fDz;
453  secRMax = std::sqrt(1+tanRMax*tanRMax);
454  pRMax = rho - p.z()*tanRMax;
455  widRMax = fRmax2 - fDz*tanRMax;
456  distRMax = std::fabs(pRMax - widRMax)/secRMax;
457 
458  if (!fPhiFullCone) // Protected against (0,0,z)
459  {
460  if ( rho )
461  {
462  pPhi = std::atan2(p.y(),p.x());
463 
464  if (pPhi < fSPhi-halfCarTolerance) { pPhi += twopi; }
465  else if (pPhi > fSPhi+fDPhi+halfCarTolerance) { pPhi -= twopi; }
466 
467  distSPhi = std::fabs( pPhi - fSPhi );
468  distEPhi = std::fabs( pPhi - fSPhi - fDPhi );
469  }
470  else if( !(fRmin1) || !(fRmin2) )
471  {
472  distSPhi = 0.;
473  distEPhi = 0.;
474  }
475  nPs = G4ThreeVector(std::sin(fSPhi), -std::cos(fSPhi), 0);
476  nPe = G4ThreeVector(-std::sin(fSPhi+fDPhi), std::cos(fSPhi+fDPhi), 0);
477  }
478  if ( rho > halfCarTolerance )
479  {
480  nR = G4ThreeVector(p.x()/rho/secRMax, p.y()/rho/secRMax, -tanRMax/secRMax);
481  if (fRmin1 || fRmin2)
482  {
483  nr = G4ThreeVector(-p.x()/rho/secRMin,-p.y()/rho/secRMin,tanRMin/secRMin);
484  }
485  }
486 
487  if( distRMax <= halfCarTolerance )
488  {
489  noSurfaces ++;
490  sumnorm += nR;
491  }
492  if( (fRmin1 || fRmin2) && (distRMin <= halfCarTolerance) )
493  {
494  noSurfaces ++;
495  sumnorm += nr;
496  }
497  if( !fPhiFullCone )
498  {
499  if (distSPhi <= halfAngTolerance)
500  {
501  noSurfaces ++;
502  sumnorm += nPs;
503  }
504  if (distEPhi <= halfAngTolerance)
505  {
506  noSurfaces ++;
507  sumnorm += nPe;
508  }
509  }
510  if (distZ <= halfCarTolerance)
511  {
512  noSurfaces ++;
513  if ( p.z() >= 0.) { sumnorm += nZ; }
514  else { sumnorm -= nZ; }
515  }
516  if ( noSurfaces == 0 )
517  {
518 #ifdef G4CSGDEBUG
519  G4Exception("G4Cons::SurfaceNormal(p)", "GeomSolids1002",
520  JustWarning, "Point p is not on surface !?" );
521 #endif
522  norm = ApproxSurfaceNormal(p);
523  }
524  else if ( noSurfaces == 1 ) { norm = sumnorm; }
525  else { norm = sumnorm.unit(); }
526 
527  return norm ;
528 }

535 G4ThreeVector G4Cons::ApproxSurfaceNormal( const G4ThreeVector& p ) const
536 {
537  ENorm side ;
538  G4ThreeVector norm ;

540  G4double distZ, distRMin, distRMax, distSPhi, distEPhi, distMin ;
541  G4double tanRMin, secRMin, pRMin, widRMin ;
542  G4double tanRMax, secRMax, pRMax, widRMax ;
543 
544  distZ = std::fabs(std::fabs(p.z()) - fDz) ;
545  rho = std::sqrt(p.x()*p.x() + p.y()*p.y()) ;
546

548  secRMin = std::sqrt(1 + tanRMin*tanRMin) ;
549  pRMin = rho - p.z()*tanRMin ;

552

554  secRMax = std::sqrt(1+tanRMax*tanRMax) ;
555  pRMax = rho - p.z()*tanRMax ;
556  widRMax = fRmax2 - fDz*tanRMax ;
557  distRMax = std::fabs(pRMax - widRMax)/secRMax ;
558

560  {

562  {

564  side = kNZ ;
565  }
566  else

569  side = kNRMin ;
570  }
571  }
572  else
