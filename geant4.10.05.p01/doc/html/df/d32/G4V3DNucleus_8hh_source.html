86 ChooseImpactXandY(G4double maxImpact)
87 {
88  G4double x,y;
89  do
90  {
91  x = 2*G4UniformRand() - 1;
92  y = 2*G4UniformRand() - 1;
93  }
94  while(x*x + y*y > 1); /* Loop checking, 30-Oct-2015, G.Folger */
95 
96  G4double impactX = x*(maxImpact);
97  G4double impactY = y*(maxImpact);
