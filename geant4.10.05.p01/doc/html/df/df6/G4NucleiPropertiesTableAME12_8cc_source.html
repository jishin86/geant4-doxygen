103 G4double G4NucleiPropertiesTableAME12::GetNuclearMass(G4int Z, G4int A)
104 {
105  if (!isIntialized) {
106  // calculate electron mass in orbit with binding energy
107  isIntialized = true;
108  for (int iz=1; iz<ZMax; iz+=1){
109  electronMass[iz] = iz*electron_mass_c2
110  - ( 14.4381 * std::pow( G4double(iz) , 2.39 )) *eV
111  - ( 1.55468*1e-6 * std::pow( G4double(iz) , 5.35 ) ) *eV;
112  }
113  }
114 
115  G4double nuclearMass = GetAtomicMass(Z,A) - electronMass[Z];
116 
117  if (nuclearMass <0.0) nuclearMass = 0.0;
118 
119  return nuclearMass;
120 
121 }
