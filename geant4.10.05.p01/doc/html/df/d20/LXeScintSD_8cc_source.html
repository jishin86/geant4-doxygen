71 G4bool LXeScintSD::ProcessHits(G4Step* aStep,G4TouchableHistory* ){
72  G4double edep = aStep->GetTotalEnergyDeposit();
73  if(edep==0.) return false; //No edep so dont count as hit
74 
75  G4StepPoint* thePrePoint = aStep->GetPreStepPoint();
76  G4TouchableHistory* theTouchable =
77  (G4TouchableHistory*)(aStep->GetPreStepPoint()->GetTouchable());
78  G4VPhysicalVolume* thePrePV = theTouchable->GetVolume();
79 
80  G4StepPoint* thePostPoint = aStep->GetPostStepPoint();
81 
82  //Get the average position of the hit
83  G4ThreeVector pos = thePrePoint->GetPosition() + thePostPoint->GetPosition();
84  pos/=2.;
85 
86  LXeScintHit* scintHit = new LXeScintHit(thePrePV);
87 
88  scintHit->SetEdep(edep);
89  scintHit->SetPos(pos);
90 
91  fScintCollection->insert(scintHit);
92 
93  return true;
94 }
