78 void G4ScoreNtupleWriter::CreateAnalysisManager()
79 {
80  if ( fAnalysisManager ) return;
81 
82  switch ( GetOutputType() ) {
83  case G4AnalysisOutput::kCsv:
84  // create Csv analysis manager
85  if ( ! G4CsvAnalysisManager::IsInstance() ) fHasAnalysisManager = true;
86  fAnalysisManager = G4CsvAnalysisManager::Instance();
87  break;
88  case G4AnalysisOutput::kRoot:
89  // create Root analysis manager
90  if ( ! G4RootAnalysisManager::IsInstance() ) fHasAnalysisManager = true;
91  fAnalysisManager = G4RootAnalysisManager::Instance();
92  break;
93  case G4AnalysisOutput::kXml:
94  // create Xml analysis manager
95  if ( ! G4XmlAnalysisManager::IsInstance() ) fHasAnalysisManager = true;
96  fAnalysisManager = G4XmlAnalysisManager::Instance();
97  break;
98  case G4AnalysisOutput::kNone:
99  // do nothing
100  break;
101  }
102 
103  fAnalysisManager->SetVerboseLevel(fVerboseLevel);
104 }

126 G4bool G4ScoreNtupleWriter::Book(G4HCofThisEvent* hce)
127 {
128 #ifdef G4VERBOSE
129  if ( fVerboseLevel > 1 ) {
130  G4cout << "--- G4ScoreNtupleWriter::Book" << G4endl;
131  }
132 #endif
133  if ( ! hce) return false;
134 
135  // book collection ID if DoubleHitsMap
136  if (fHCIds.size() == 0) {
137 #ifdef G4VERBOSE
138  if ( fVerboseLevel > 1 ) {
139  G4cout << "--- going to fill fHCIds " << G4endl;
140  }
141 #endif
142  for (G4int i=0; i<hce->GetNumberOfCollections(); ++i) {
143  auto hitsMap
144  = dynamic_cast<G4THitsMap<G4double>*>(hce->GetHC(i));
145 
146  if ( hitsMap ) {
147 #ifdef G4VERBOSE
148  if ( fVerboseLevel > 1 ) {
149  G4cout << "--- adding hcID = " << i << G4endl;
150  }
151 #endif
152  fHCIds.push_back(i);
153  }
154  }
155  }
156 
157  // Create analysis manager
158  if ( fHCIds.size() ) {
159  CreateAnalysisManager();
160  }
161 
162  // Create ntuples (only once)
163  if ( ! fIsInitialized ) {
164 #ifdef G4VERBOSE
165  if ( fVerboseLevel > 1 ) {
166  G4cout << "-- going to create ntuples " << G4endl;
167  }
168 #endif
169  auto first = true;
170  for (auto id : fHCIds) {
171  auto hitsMap
172  = static_cast<G4THitsMap<G4double>*>(hce->GetHC(id));
173 
174  // Create ntuple for this primitive
175  G4String ntupleName(hitsMap->GetSDname());
176  ntupleName.append("_");
177  ntupleName.append(hitsMap->GetName());
178 
179  G4int ntupleId
180  = fAnalysisManager->CreateNtuple(ntupleName, ntupleName);
181  if ( first ) {
182  fFirstNtupleId = ntupleId;
183 #ifdef G4VERBOSE
184  if ( fVerboseLevel > 1 ) {
185  G4cout << "-- set first ntuple Id " << fFirstNtupleId << G4endl;
186  }
187 #endif
188  first = false;
189  }
190 
191  G4String colName(ntupleName);
192  colName.append("_eventId");
193  fAnalysisManager->CreateNtupleIColumn(colName);
194  colName = ntupleName;
195  colName.append("_cell");
196  fAnalysisManager->CreateNtupleIColumn(colName);
197  colName = ntupleName;
198  colName.append("_score");
199  fAnalysisManager->CreateNtupleDColumn(colName);
200  fAnalysisManager->FinishNtuple();
201  fIsBooked = true;
202  }
203  fIsInitialized = true;
204  }
205 
206  return fIsBooked;
207 }

242 void G4ScoreNtupleWriter::Fill(G4HCofThisEvent* hce, G4int eventNumber)
243 {
244 
245 #ifdef G4VERBOSE
246  if ( fVerboseLevel > 1 ) {
247  G4cout << "--- start G4ScoreNtupleWriter::Fill" << G4endl;
248  }
249 #endif
250 
251  auto counter = 0;
252  for (auto id : fHCIds) {
253 #ifdef G4VERBOSE
254  if ( fVerboseLevel > 1 ) {
255  G4cout << "in loop over fHCIds, counter " << counter << G4endl;
256  }
257 #endif
258  auto hitsMap
259  = static_cast<G4THitsMap<G4double>*>(hce->GetHC(id));
260 
261  //G4cout << eventNumber << ".. go to fill ntuple " << counter + fFirstNtupleId << G4endl;
262 
263  // fill hits in ntuple
264  std::map<G4int, G4double*>::iterator it;
265  for ( it = hitsMap->GetMap()->begin(); it != hitsMap->GetMap()->end(); it++ ) {
266  fAnalysisManager->FillNtupleIColumn(counter + fFirstNtupleId, 0, eventNumber);
267  fAnalysisManager->FillNtupleIColumn(counter + fFirstNtupleId, 1, it->first);
268  fAnalysisManager->FillNtupleDColumn(counter + fFirstNtupleId, 2, *(it->second));
269  fAnalysisManager->AddNtupleRow(counter + fFirstNtupleId);
270  }
271  counter++;
272  }
273 
274 #ifdef G4VERBOSE
275  if ( fVerboseLevel > 1 ) {
276  G4cout << "--- done G4ScoreNtupleWriter::Fill" << G4endl;
277  }
278 #endif
279 }
