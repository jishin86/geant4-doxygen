49 G4double G4VKinkyStringDecay::GetLightConeGluonZ(G4double zmin, G4double zmax)
50 {
51  G4double z, yf;
52  do {
53  z = zmin + G4UniformRand()*(zmax-zmin);
54  yf = z*z +sqr(1 - z);
55  }
56  while (G4UniformRand() > yf); /* Loop checking, 07.08.2015, A.Ribon */
57  return z;
58 }

62 G4KineticTrackVector* G4VKinkyStringDecay::FragmentString(const G4ExcitedString& String)
63 {
64  G4LorentzVector Mom = String.GetGluon()->Get4Momentum();
65  G4ThreeVector Pos = String.GetGluon()->GetPosition();
66  G4int QuarkEncoding = theLongitudinalStringDecay->SampleQuarkFlavor();
67  G4ThreeVector Pquark=theLongitudinalStringDecay->SampleQuarkPt();
68  G4double Pt2 = Pquark.mag2();
69  G4double z = GetLightConeGluonZ(0, 1);
70  G4double w = Mom.e() + Mom.pz();
71  //... now compute quark longitudinal momentum and energy
72 
73  Pquark.setZ( (z*w - Pt2/(z*w))*0.5);
74  G4double E = (z*w + Pt2/(z*w))*0.5;
75 
76  G4Parton* AntiColor = new G4Parton(-QuarkEncoding);
77  AntiColor->SetPosition(Pos);
78  G4LorentzVector AntiColorMom(-Pquark, E);
79  AntiColor->Set4Momentum(AntiColorMom);
80  G4Parton* Color = new G4Parton(*String.GetColorParton());
81  G4ExcitedString Str1(Color, AntiColor, String.GetDirection());
82  G4KineticTrackVector* KTV1 = theLongitudinalStringDecay->FragmentString(Str1);
83 
84  Color = new G4Parton(QuarkEncoding);
85  Color->SetPosition(Pos);
86  G4LorentzVector ColorMom(Pquark, E);
87  Color->Set4Momentum(ColorMom);
88  AntiColor = new G4Parton(*String.GetAntiColorParton());
89  G4ExcitedString Str2(Color, AntiColor, String.GetDirection());
90  G4KineticTrackVector* KTV2 = theLongitudinalStringDecay->FragmentString(Str2);
91 
92  if (KTV1 && KTV2)
93  while(!KTV2->empty()) /* Loop checking, 07.08.2015, A.Ribon */
94  {
95  KTV1->push_back(KTV2->back());
96  KTV1->erase(KTV1->end()-1);
97  }
98  return KTV1;
99 }
