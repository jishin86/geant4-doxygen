74 {
75  // New event
76  G4int nEvt = evt->GetEventID();
77 
78  if(fSelected>0) {
79  for(G4int i=0; i<fSelected; ++i) {
80  if(nEvt == fSelectedEvents[i]) {
81  fUI->ApplyCommand("/random/saveThisEvent");
82  fUI->ApplyCommand("/tracking/verbose 2");
83  fDebugStarted = true;
84  break;
85  }
86  }
87  }
88 
89  // Initialize user actions
90  if(G4int(nEvt/fPrintModulo)*fPrintModulo == nEvt) {
91  G4cout << "EventAction: Event # "

93  }
94  HistoManager::GetPointer()->BeginOfEvent();
95 }
