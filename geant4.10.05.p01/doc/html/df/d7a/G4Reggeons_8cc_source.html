234 void G4Reggeons::CalculateXs()
235 {
236  Xtotal =0.; XtotalP=0.; XtotalR=0.;
237  Xelastic=0.; Xpr_Diff=0.; Xtr_Diff=0.; XDDiff=0.; G4double XDiff=0.;
238  Xinel =0.; Xnd=0.; XndP=0.; XndR=0.;
239 
240  G4double AmplitudeP(0.), AmplitudeR(0.);
241 
242  G4double B_max = 10.*fermi;
243  G4double dB = B_max/10000.;
244 
245  G4double B =-dB/2.;
246 
247  G4double chiP(0.), chiR(0.), chiRin(0.); // chiPin Pomeron inelastic phase is a data member
248  chiPin=0.;
249  for(G4int i=0; i<10000;i++)
250  {
251  B += dB;
252 
253  chiP = Chi_pomeron(1.,B); chiR = Chi_reggeon(1.,B);
254  chiPin = Chi_pomeron(2.,B); chiRin = Chi_reggeon(2.,B);
255 
256  AmplitudeP = (1.0/C_pomeron)*(1.0 - G4Exp(-chiP))*G4Exp(-chiR);
257  AmplitudeR = (1.0 - G4Exp(-chiR));
258 
259  Xtotal += 2 * (AmplitudeP + AmplitudeR) * B * dB;
260  XtotalP += 2 * (AmplitudeP + 0. ) * B * dB;
261  XtotalR += 2 * (0. + AmplitudeR) * B * dB;
262 
263  Xelastic += sqr(AmplitudeP + AmplitudeR) * B * dB;
264  Xpr_Diff += (Cpr_pomeron - 1.0) * sqr(AmplitudeP) * B * dB;
265  Xtr_Diff += (Ctr_pomeron - 1.0) * sqr(AmplitudeP) * B * dB;
266  XDiff += (Cpr_pomeron - 1.0) * (Ctr_pomeron - 1.0) * sqr(AmplitudeP) * B * dB;
267 
268  // ----------------------------------
269  AmplitudeP = (1.0/C_pomeron)*(1.0 - G4Exp(-chiPin))*G4Exp(-chiRin);
270  AmplitudeR = (1.0 - G4Exp(-chiRin));
271 
272  Xnd += (AmplitudeP + AmplitudeR) * B * dB;
273  XndP += (AmplitudeP + 0. ) * B * dB;
274  XndR += (0. + AmplitudeR) * B * dB;
275  }
276 
277  Xtotal *=twopi; XtotalP *=twopi; XtotalR *=twopi;
278  Xelastic *=twopi; Xpr_Diff *=twopi; Xtr_Diff *=twopi; XDiff *=twopi;
279  Xinel = Xtotal - Xelastic;
280  (void)Xinel; // To avoid compiler warning "variable not used"
281 
282  Xnd *=twopi; XndP *=twopi; XndR *=twopi;
283  XDDiff = XDiff-Xpr_Diff-Xtr_Diff;
284 
285  /*
286  G4cout<<"Total totalP totalR "<<Xtotal/millibarn <<" "<<XtotalP/millibarn <<" "<<XtotalR/millibarn<<" mb"<<G4endl;
287  G4cout<<"Elastic "<<Xelastic/millibarn <<G4endl;
288  G4cout<<"PrDiff TrDiff W_Diff "<<Xpr_Diff/millibarn<<" "<<Xtr_Diff/millibarn<<" "<<XDiff/millibarn<<G4endl;
289  G4cout<<"Inelastic "<<Xinel/millibarn <<G4endl;
290  G4cout<<"NonDiff Pom & Reg "<<Xnd/millibarn <<" "<<XndP/millibarn <<" "<<XndR/millibarn <<G4endl;
291  */
292 }

295 G4double G4Reggeons::Chi_pomeron(G4double Mult, G4double B)
296 {
297  G4double R2 = Rsquare_pomeron + Alphaprime_pomeron * G4Log(Sint/S0_pomeron);
298  G4double Eikonal = Mult * C_pomeron * Gamma_pomeron/R2 *
299  G4Pow::GetInstance()->powA(Sint/S0_pomeron, Alpha_pomeron -1.) *
300  G4Exp(-sqr(B)/4.0/R2/hbarc_squared);
301  return Eikonal;
302 }

305 G4double G4Reggeons::Chi_reggeon(G4double Mult, G4double B)
306 {
307  G4double R2F = Freggeon_Rsquare + Freggeon_Alphaprime * G4Log(Sint/S0_pomeron);
308  G4double R2W = Wreggeon_Rsquare + Wreggeon_Alphaprime * G4Log(Sint/S0_pomeron);
309 
310  G4double Eikonal = Mult * FParity * Freggeon_C * Freggeon_Gamma/R2F *
311  G4Pow::GetInstance()->powA(Sint/S0_pomeron, Freggeon_Alpha -1.) *
312  G4Exp(-sqr(B)/4.0/R2F/hbarc_squared);
313 
314  Eikonal+= Mult * WParity * Wreggeon_C * Wreggeon_Gamma/R2W *
315  G4Pow::GetInstance()->powA(Sint/S0_pomeron, Wreggeon_Alpha -1.) *
316  G4Exp(-sqr(B)/4.0/R2W/hbarc_squared);
317  return Eikonal;
318 }

338 void G4Reggeons::GetProbabilities(G4double B, G4int Mode,
339  G4double & Pint,
340  G4double & Pprd, G4double & Ptrd, G4double & Pdd,
341  G4double & Pnd, G4double & Pnvr)
342 {
343  // Puprose of the method is a calculation of inelastic interaction probability (Pint),
344  // probability of projectile diffraction (Pprd),
345  // probability of target diffraction (Ptrd),
346  // probability of double diffraction (Pdd ),
347  // probability of non-diffractive inter. (Pnd ),
348  // probability of quark-exc. inter. (Pnvr),
349  // number of cutted pomerons (NcutPomerons).
350  // The input parameters are B - impact parameter, and Mode = All/WITHOUT_R/NON_DIFF
351  //
352  if ( B > 2.* fermi ) { Pint=0.; Pprd=0.; Ptrd=0.; Pdd=0.; Pnd=0.; Pnvr=0.; return;}
353  // At large B for hN collisions it is better to return zero inter. probability
354 
355  G4double chiP = Chi_pomeron(1.,B); G4double chiR = Chi_reggeon(1.,B);
356  chiPin = Chi_pomeron(2.,B); G4double chiRin = Chi_reggeon(2.,B);
357  //chiPin is data member of the class
358 
359  G4double Exp_ChiR = G4Exp(-chiR);
360 
361  G4double AmplitudeP = (1.0/C_pomeron)*(1.0 - G4Exp(-chiP))*Exp_ChiR;
362  G4double AmplitudeR = (1.0 - Exp_ChiR);
363 
364  G4double AmplitudeP2, Apr_Diff, Atr_Diff, ADiff;
365 
366  // Aelastic = sqr(AmplitudeP + AmplitudeR);
367  AmplitudeP2 = sqr(AmplitudeP);
368  Apr_Diff = (Cpr_pomeron - 1.0) * AmplitudeP2;
369  Atr_Diff = (Ctr_pomeron - 1.0) * AmplitudeP2;
370  ADiff = (Cpr_pomeron - 1.0) * (Ctr_pomeron - 1.0) * AmplitudeP2;
371 
372  // ----------------------------------
373  Exp_ChiR = G4Exp(-chiRin);
374  AmplitudeP = (1.0/C_pomeron)*(1.0 - G4Exp(-chiPin))*Exp_ChiR;
375  AmplitudeR = (1.0 - Exp_ChiR);
376 
377  G4double And, AndP, AndR;
378 
379  And = (AmplitudeP + AmplitudeR);
380  AndP = (AmplitudeP + 0. );
381  AndR = (0. + AmplitudeR);
382 
383  // ----------------------------------
384  if ( Mode == ALL)
385  {
386  Pint = Apr_Diff + Atr_Diff + ADiff + And;
387  Pprd = Apr_Diff/Pint; // Probability of projectile diffraction
388  Ptrd = Atr_Diff/Pint; // Probability of target diffraction
389  Pdd = ADiff /Pint; // Probability of double diffraction
390  Pnd = AndP /Pint; // Probability of non-diffractive inelastic interaction
391  Pnvr = AndR /Pint; // Probability of non-vacuum reggeon (nvr) inelastic interaction
392  }
393  else if ( Mode == WITHOUT_R)
394  {
395  Pint = Apr_Diff + Atr_Diff + ADiff + AndP;
396  Pprd = Apr_Diff/Pint;
397  Ptrd = Atr_Diff/Pint;
398  Pdd = ADiff /Pint;
399  Pnd = AndP /Pint;
400  Pnvr = 0.;
401  }
402  else
403  { // Mode == NON_DIFF (of projectile)
404  Pint = Atr_Diff + AndP;
405  Pprd = 0.;
406  Ptrd = Atr_Diff/Pint;
407  Pdd = 0.;
408  Pnd = AndP /Pint;
409  Pnvr = 0.;
410  }
411  return;
412 }

415 G4int G4Reggeons::ncPomerons() // Non-complete Poisson distribution
416 {
417  if ( chiPin < 0.001 ) return 0; // At small average multiplicity of cutted pomerons
418  // it is better to return 0 to avoid problems with
419  // calculation exactness.
420  G4double ksi = G4UniformRand() * (1.0-G4Exp(-chiPin)) * G4Exp(chiPin);
421  G4double Term = chiPin;
422  G4double Sum = Term;
423  G4int nCuts = 1;
424 
425  while ( Sum < ksi) {
426  nCuts++;
427  Term *= chiPin/(G4double) nCuts;
428  Sum += Term;
429  }
430 
431  return nCuts;
432 }
