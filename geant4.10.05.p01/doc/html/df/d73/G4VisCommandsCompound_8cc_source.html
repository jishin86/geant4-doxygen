63 void G4VisCommandDrawTree::SetNewValue(G4UIcommand*, G4String newValue) {
64 
65  G4String pvname, system;
66  std::istringstream is(newValue);
67  is >> pvname >> system;
68 
69  // Note: The second parameter, "system", is intended to allow the user
70  // a choice of dedicated tree printing/displaying systems but at present
71  // the only such dedicated system is ASCIITree. It doesn't make sense to
72  // specify OGLSX, for example. So to avoid confusion we restrict this
73  // feature to systems that have "Tree" in the name or nickname.
74 
75  // Of course, some other systems, such as OGLSQt, have a tree browser
76  // built-in. The HepRApp offline browser also has a tree browser
77  // built in.
78 
79  if (!system.contains("Tree")) {
80  system = "ATree";
81  }
82 
83  G4VGraphicsSystem* keepSystem = fpVisManager->GetCurrentGraphicsSystem();
84  G4Scene* keepScene = fpVisManager->GetCurrentScene();
85  G4VSceneHandler* keepSceneHandler = fpVisManager->GetCurrentSceneHandler();
86  G4VViewer* keepViewer = fpVisManager->GetCurrentViewer();
87 
88  G4UImanager* UImanager = G4UImanager::GetUIpointer();
89  G4int keepVerbose = UImanager->GetVerboseLevel();
90  G4int newVerbose(0);
91  if (keepVerbose >= 2 ||
92  fpVisManager->GetVerbosity() >= G4VisManager::confirmations)
93  newVerbose = 2;
94  UImanager->SetVerboseLevel(newVerbose);
95  UImanager->ApplyCommand(G4String("/vis/open " + system));
96  if (fErrorCode == 0) {
97  UImanager->ApplyCommand(G4String("/vis/drawVolume " + pvname));
98  UImanager->ApplyCommand("/vis/viewer/flush");
99  if (keepViewer) {
100  if (fpVisManager->GetVerbosity() >= G4VisManager::warnings) {
101  G4cout << "Reverting to " << keepViewer->GetName() << G4endl;
102  }
103  fpVisManager->SetCurrentGraphicsSystem(keepSystem);
104  fpVisManager->SetCurrentScene(keepScene);
105  fpVisManager->SetCurrentSceneHandler(keepSceneHandler);
106  fpVisManager->SetCurrentViewer(keepViewer);
107  }
108  }
109  UImanager->SetVerboseLevel(keepVerbose);
110 }

150 void G4VisCommandDrawView::SetNewValue(G4UIcommand*, G4String newValue) {
151 
152  G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
153 
154  G4VViewer* currentViewer = fpVisManager->GetCurrentViewer();
155  if (!currentViewer) {
156  if (verbosity >= G4VisManager::warnings) {
157  G4cout <<
158  "WARNING: G4VisCommandsDrawView::SetNewValue: no current viewer."
159  << G4endl;
160  }
161  return;
162  }
163 
164  G4String thetaDeg;
165  G4String phiDeg;
166  G4String panRight;
167  G4String panUp;
168  G4String panUnit;
169  G4String zoomFactor;
170  G4String dolly;
171  G4String dollyUnit;
172  std::istringstream is(newValue);
173  is >> thetaDeg >> phiDeg >> panRight >> panUp >> panUnit
174  >> zoomFactor >> dolly >> dollyUnit;
175 
176  G4UImanager* UImanager = G4UImanager::GetUIpointer();
177  G4int keepVerbose = UImanager->GetVerboseLevel();
178  G4int newVerbose(0);
179  if (keepVerbose >= 2 ||
180  fpVisManager->GetVerbosity() >= G4VisManager::confirmations)
181  newVerbose = 2;
182  UImanager->SetVerboseLevel(newVerbose);
183  G4ViewParameters vp = currentViewer->GetViewParameters();
184  G4bool keepAutoRefresh = vp.IsAutoRefresh();
185  vp.SetAutoRefresh(false);
186  currentViewer->SetViewParameters(vp);
187  UImanager->ApplyCommand(
188  G4String("/vis/viewer/set/viewpointThetaPhi " + thetaDeg + " " + phiDeg + " deg"));
189  UImanager->ApplyCommand(
190  G4String("/vis/viewer/panTo " + panRight + " " + panUp + " " + panUnit));
191  UImanager->ApplyCommand(
192  G4String("/vis/viewer/zoomTo " + zoomFactor));
193  vp = currentViewer->GetViewParameters();
194  vp.SetAutoRefresh(keepAutoRefresh);
195  currentViewer->SetViewParameters(vp);
196  UImanager->ApplyCommand(
197  G4String("/vis/viewer/dollyTo " + dolly + " " + dollyUnit));
198  UImanager->SetVerboseLevel(keepVerbose);
199 }

232 void G4VisCommandDrawLogicalVolume::SetNewValue(G4UIcommand*, G4String newValue) {
233  G4VisManager::Verbosity verbosity = fpVisManager->GetVerbosity();
234  G4UImanager* UImanager = G4UImanager::GetUIpointer();
235  G4int keepVerbose = UImanager->GetVerboseLevel();
236  G4int newVerbose(0);
237  if (keepVerbose >= 2 || verbosity >= G4VisManager::confirmations)
238  newVerbose = 2;
239  UImanager->SetVerboseLevel(newVerbose);
240  G4VViewer* currentViewer = fpVisManager->GetCurrentViewer();
241  const G4ViewParameters& currentViewParams = currentViewer->GetViewParameters();
242  G4bool keepAutoRefresh = currentViewParams.IsAutoRefresh();
243  if (keepAutoRefresh) UImanager->ApplyCommand("/vis/viewer/set/autoRefresh false");
244  UImanager->ApplyCommand("/vis/scene/create");
245  UImanager->ApplyCommand(G4String("/vis/scene/add/logicalVolume " + newValue));
246  UImanager->ApplyCommand("/vis/sceneHandler/attach");
247  G4ViewParameters::DrawingStyle keepDrawingStyle = currentViewParams.GetDrawingStyle();
248  if(keepDrawingStyle != G4ViewParameters::wireframe) UImanager->ApplyCommand("/vis/viewer/set/style wireframe");
249  G4bool keepMarkerNotHidden = currentViewParams.IsMarkerNotHidden();
250  if (!keepMarkerNotHidden) UImanager->ApplyCommand("/vis/viewer/set/hiddenMarker false");
251  if (keepAutoRefresh) UImanager->ApplyCommand("/vis/viewer/set/autoRefresh true");
252  UImanager->SetVerboseLevel(keepVerbose);
253  if (verbosity >= G4VisManager::warnings) {
254  if (keepDrawingStyle != currentViewParams.GetDrawingStyle()) {
255  G4cout
256  << "Drawing style changed to wireframe. To restore previous style:";
257  G4String style, edge;
258  switch (keepDrawingStyle) {
259  case G4ViewParameters::wireframe:
260  style = "wireframe"; edge = "false"; break;
261  case G4ViewParameters::hlr:
262  style = "wireframe"; edge = "true"; break;
263  case G4ViewParameters::hsr:
264  style = "surface"; edge = "false"; break;
265  case G4ViewParameters::hlhsr:
266  style = "surface"; edge = "true"; break;
267  }
268  G4cout
269  << "\n /vis/viewer/set/style " + style
270  << "\n /vis/viewer/set/hiddenEdge " + edge
271  << G4endl;
272  }
273  if (keepMarkerNotHidden != currentViewParams.IsMarkerNotHidden()) {
274  G4cout
275  << "Markers changed to \"not hidden\". To restore previous condition:"
276  << "\n /vis/viewer/set/hiddenmarker true"
277  << G4endl;
278  }
279  }
280  static G4bool warned = false;
281  if (verbosity >= G4VisManager::confirmations && !warned) {
282  G4cout <<
283  "NOTE: For systems which are not \"auto-refresh\" you will need to"
284  "\n issue \"/vis/viewer/refresh\" or \"/vis/viewer/flush\"."
285  << G4endl;
286  warned = true;
287  }
288 }

363 void G4VisCommandOpen::SetNewValue (G4UIcommand*, G4String newValue) {
364  G4String systemName, windowSizeHint;
365  std::istringstream is(newValue);
366  is >> systemName >> windowSizeHint;
367  G4UImanager* UImanager = G4UImanager::GetUIpointer();
368  G4int keepVerbose = UImanager->GetVerboseLevel();
369  G4int newVerbose(0);
370  if (keepVerbose >= 2 ||
371  fpVisManager->GetVerbosity() >= G4VisManager::confirmations)
372  newVerbose = 2;
373  UImanager->SetVerboseLevel(newVerbose);
374  fErrorCode = UImanager->ApplyCommand(G4String("/vis/sceneHandler/create " + systemName));
375  if (fErrorCode == 0) {
376  UImanager->ApplyCommand(G4String("/vis/viewer/create ! ! " + windowSizeHint));
377  } else {
378  // Use set to get alphabetical order
379  std::set<G4String> candidates;
380  for (const auto gs: fpVisManager -> GetAvailableGraphicsSystems()) {
381  // Just list nicknames, but exclude FALLBACK nicknames
382  for (const auto& nickname: gs->GetNicknames()) {
383  if (!nickname.contains("FALLBACK")) {
384  candidates.insert(nickname);
385  }
386  }
387  }
388  G4cerr << "Candidates are:";
389  for (const auto& candidate: candidates) {
390  G4cerr << ' ' << candidate;
391  }
392  G4cerr << G4endl;
393  }
394  UImanager->SetVerboseLevel(keepVerbose);
395 }
