117 G4bool G4RDCompositeEMDataSet::LoadData(const G4String& argFileName)
118 {
119  CleanUpComponents();
120 
121  for (G4int z(minZ); z<maxZ; z++)
122  {
123  G4RDVEMDataSet* component = new G4RDEMDataSet(z, algorithm->Clone(), unitEnergies, unitData);
124  if (!component->LoadData(argFileName))
125  {
126  delete component;
127  return false;
128  }
129  AddComponent(component);
130  }
131  return true;
132 }
