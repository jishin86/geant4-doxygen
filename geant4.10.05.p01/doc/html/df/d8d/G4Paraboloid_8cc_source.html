63 // constructor - check parameters
64 
65 G4Paraboloid::G4Paraboloid(const G4String& pName,
66  G4double pDz,
67  G4double pR1,
68  G4double pR2)
69  : G4VSolid(pName), fRebuildPolyhedron(false), fpPolyhedron(0),
70  fSurfaceArea(0.), fCubicVolume(0.)
71 
72 {
73  if( (pDz <= 0.) || (pR2 <= pR1) || (pR1 < 0.) )
74  {
75  std::ostringstream message;
76  message << "Invalid dimensions. Negative Input Values or R1>=R2 - "
77  << GetName();
78  G4Exception("G4Paraboloid::G4Paraboloid()", "GeomSolids0002",
79  FatalErrorInArgument, message,
80  "Z half-length must be larger than zero or R1>=R2.");
81  }
82 
83  r1 = pR1;
84  r2 = pR2;
85  dz = pDz;
86 
87  // r1^2 = k1 * (-dz) + k2
88  // r2^2 = k1 * ( dz) + k2
89  // => r1^2 + r2^2 = k2 + k2 => k2 = (r2^2 + r1^2) / 2
90  // and r2^2 - r1^2 = k1 * dz - k1 * (-dz) => k1 = (r2^2 - r1^2) / 2 / dz
91 
92  k1 = (r2 * r2 - r1 * r1) / 2 / dz;
93  k2 = (r2 * r2 + r1 * r1) / 2;
94 }

213 
214 EInside G4Paraboloid::Inside(const G4ThreeVector& p) const
215 {
216  // First check is the point is above or below the solid.
217  //
218  if(std::fabs(p.z()) > dz + 0.5 * kCarTolerance) { return kOutside; }
219 
220  G4double rho2 = p.perp2(),
221  rhoSurfTimesTol2 = (k1 * p.z() + k2) * sqr(kCarTolerance),
222  A = rho2 - ((k1 *p.z() + k2) + 0.25 * kCarTolerance * kCarTolerance);
223 
224  if(A < 0 && sqr(A) > rhoSurfTimesTol2)
225  {
226  // Actually checking rho < radius of paraboloid at z = p.z().
227  // We're either inside or in lower/upper cutoff area.
228 
229  if(std::fabs(p.z()) > dz - 0.5 * kCarTolerance)
230  {
231  // We're in the upper/lower cutoff area, sides have a paraboloid shape
232  // maybe further checks should be made to make these nicer
233 
234  return kSurface;
235  }
236  else
237  {
238  return kInside;
239  }
240  }
241  else if(A <= 0 || sqr(A) < rhoSurfTimesTol2)
242  {
243  // We're in the parabolic surface.
244 
245  return kSurface;
246  }
247  else
248  {
249  return kOutside;
250  }
251 }

256 G4ThreeVector G4Paraboloid::SurfaceNormal( const G4ThreeVector& p) const
257 {
258  G4ThreeVector n(0, 0, 0);
259  if(std::fabs(p.z()) > dz + 0.5 * kCarTolerance)
260  {
261  // If above or below just return normal vector for the cutoff plane.
262 
263  n = G4ThreeVector(0, 0, p.z()/std::fabs(p.z()));
264  }
265  else if(std::fabs(p.z()) > dz - 0.5 * kCarTolerance)
266  {
267  // This means we're somewhere in the plane z = dz or z = -dz.
268  // (As far as the program is concerned anyway.
269 
270  if(p.z() < 0) // Are we in upper or lower plane?
271  {
272  if(p.perp2() > sqr(r1 + 0.5 * kCarTolerance))
273  {
274  n = G4ThreeVector(p.x(), p.y(), -k1 / 2).unit();
275  }
276  else if(r1 < 0.5 * kCarTolerance
277  || p.perp2() > sqr(r1 - 0.5 * kCarTolerance))
278  {
279  n = G4ThreeVector(p.x(), p.y(), 0.).unit()
280  + G4ThreeVector(0., 0., -1.).unit();
281  n = n.unit();
282  }
283  else
284  {
285  n = G4ThreeVector(0., 0., -1.);
286  }
287  }
288  else
289  {
290  if(p.perp2() > sqr(r2 + 0.5 * kCarTolerance))
291  {
292  n = G4ThreeVector(p.x(), p.y(), 0.).unit();
293  }
294  else if(r2 < 0.5 * kCarTolerance
295  || p.perp2() > sqr(r2 - 0.5 * kCarTolerance))
296  {
297  n = G4ThreeVector(p.x(), p.y(), 0.).unit()
298  + G4ThreeVector(0., 0., 1.).unit();
299  n = n.unit();
300  }
301  else
302  {
303  n = G4ThreeVector(0., 0., 1.);
304  }
305  }
306  }
307  else
308  {
309  G4double rho2 = p.perp2();
310  G4double rhoSurfTimesTol2 = (k1 * p.z() + k2) * sqr(kCarTolerance);
311  G4double A = rho2 - ((k1 *p.z() + k2)
312  + 0.25 * kCarTolerance * kCarTolerance);
313 
314  if(A < 0 && sqr(A) > rhoSurfTimesTol2)
315  {
316  // Actually checking rho < radius of paraboloid at z = p.z().
317  // We're inside.
318 
319  if(p.mag2() != 0) { n = p.unit(); }
320  }
321  else if(A <= 0 || sqr(A) < rhoSurfTimesTol2)
322  {
323  // We're in the parabolic surface.
324 
325  n = G4ThreeVector(p.x(), p.y(), - k1 / 2).unit();
326  }
327  else
328  {
329  n = G4ThreeVector(p.x(), p.y(), - k1 / 2).unit();
330  }
331  }
332 
333  if(n.mag2() == 0)
334  {
335  std::ostringstream message;
336  message << "No normal defined for this point p." << G4endl
337  << " p = " << 1 / mm * p << " mm";
338  G4Exception("G4Paraboloid::SurfaceNormal(p)", "GeomSolids1002",
339  JustWarning, message);
340  }
341  return n;
342 }

346 // Calculate distance to shape from outside, along normalised vector
347 // - return kInfinity if no intersection
348 //
349 
350 G4double G4Paraboloid::DistanceToIn( const G4ThreeVector& p,
351  const G4ThreeVector& v ) const
352 {
353  G4double rho2 = p.perp2(), paraRho2 = std::fabs(k1 * p.z() + k2);
354  G4double tol2 = kCarTolerance*kCarTolerance;
355  G4double tolh = 0.5*kCarTolerance;
356 
357  if(r2 && p.z() > - tolh + dz)
358  {
359  // If the points is above check for intersection with upper edge.
360 
361  if(v.z() < 0)
362  {
363  G4double intersection = (dz - p.z()) / v.z(); // With plane z = dz.
364  if(sqr(p.x() + v.x()*intersection)
365  + sqr(p.y() + v.y()*intersection) < sqr(r2 + 0.5 * kCarTolerance))
366  {
367  if(p.z() < tolh + dz)
368  { return 0; }
369  else
370  { return intersection; }
371  }
372  }
373  else // Direction away, no possibility of intersection
374  {
375  return kInfinity;
376  }
377  }
378  else if(r1 && p.z() < tolh - dz)
379  {
380  // If the points is belove check for intersection with lower edge.
381 
382  if(v.z() > 0)
383  {
384  G4double intersection = (-dz - p.z()) / v.z(); // With plane z = -dz.
385  if(sqr(p.x() + v.x()*intersection)
386  + sqr(p.y() + v.y()*intersection) < sqr(r1 + 0.5 * kCarTolerance))
387  {
388  if(p.z() > -tolh - dz)
389  {
390  return 0;
391  }
392  else
393  {
394  return intersection;
395  }
396  }
397  }
398  else // Direction away, no possibility of intersection
399  {
400  return kInfinity;
401  }
402  }
403 
404  G4double A = k1 / 2 * v.z() - p.x() * v.x() - p.y() * v.y(),
405  vRho2 = v.perp2(), intersection,
406  B = (k1 * p.z() + k2 - rho2) * vRho2;
407 
408  if ( ( (rho2 > paraRho2) && (sqr(rho2-paraRho2-0.25*tol2) > tol2*paraRho2) )
409  || (p.z() < - dz+kCarTolerance)
410  || (p.z() > dz-kCarTolerance) ) // Make sure it's safely outside.
411  {
412  // Is there a problem with squaring rho twice?
413 
414  if(vRho2<tol2) // Needs to be treated seperately.
415  {
416  intersection = ((rho2 - k2)/k1 - p.z())/v.z();
417  if(intersection < 0) { return kInfinity; }
418  else if(std::fabs(p.z() + v.z() * intersection) <= dz)
419  {
420  return intersection;
421  }
422  else
423  {
424  return kInfinity;
425  }
426  }
427  else if(A*A + B < 0) // No real intersections.
428  {
429  return kInfinity;
430  }
431  else
432  {
433  intersection = (A - std::sqrt(B + sqr(A))) / vRho2;
434  if(intersection < 0)
435  {
436  return kInfinity;
437  }
438  else if(std::fabs(p.z() + intersection * v.z()) < dz + tolh)
439  {
440  return intersection;
441  }
442  else
443  {
444  return kInfinity;
445  }
446  }
447  }
448  else if(sqr(rho2 - paraRho2 - .25 * tol2) <= tol2 * paraRho2)
449  {
450  // If this is true we're somewhere in the border.
451 
452  G4ThreeVector normal(p.x(), p.y(), -k1/2);
453  if(normal.dot(v) <= 0)
454  { return 0; }
455  else
456  { return kInfinity; }
457  }
458  else
459  {
460  std::ostringstream message;
461  if(Inside(p) == kInside)
462  {
463  message << "Point p is inside! - " << GetName() << G4endl;
464  }
465  else
466  {
467  message << "Likely a problem in this function, for solid: " << GetName()
468  << G4endl;
469  }
470  message << " p = " << p * (1/mm) << " mm" << G4endl
471  << " v = " << v * (1/mm) << " mm";
472  G4Exception("G4Paraboloid::DistanceToIn(p,v)", "GeomSolids1002",
473  JustWarning, message);
474  return 0;
475  }
476 }

479 //
480 // Calculate distance (<= actual) to closest surface of shape from outside
481 // - Return 0 if point inside
482 
483 G4double G4Paraboloid::DistanceToIn(const G4ThreeVector& p) const
484 {
485  G4double safz = -dz+std::fabs(p.z());
486  if(safz<0) { safz=0; }
487  G4double safr = kInfinity;
488 
489  G4double rho = p.x()*p.x()+p.y()*p.y();
490  G4double paraRho = (p.z()-k2)/k1;
491  G4double sqrho = std::sqrt(rho);
492 
493  if(paraRho<0)
494  {
495  safr=sqrho-r2;
496  if(safr>safz) { safz=safr; }
497  return safz;
498  }
499 
500  G4double sqprho = std::sqrt(paraRho);
501  G4double dRho = sqrho-sqprho;
502  if(dRho<0) { return safz; }
503 
504  G4double talf = -2.*k1*sqprho;
505  G4double tmp = 1+talf*talf;
506  if(tmp<0.) { return safz; }
507 
508  G4double salf = talf/std::sqrt(tmp);
509  safr = std::fabs(dRho*salf);
510  if(safr>safz) { safz=safr; }
511 
512  return safz;
513 }

516 //
517 // Calculate distance to surface of shape from 'inside'
518 
519 G4double G4Paraboloid::DistanceToOut(const G4ThreeVector& p,
520  const G4ThreeVector& v,
521  const G4bool calcNorm,
522  G4bool *validNorm,
523  G4ThreeVector *n ) const
524 {
525  G4double rho2 = p.perp2(), paraRho2 = std::fabs(k1 * p.z() + k2);
526  G4double vRho2 = v.perp2(), intersection;
527  G4double tol2 = kCarTolerance*kCarTolerance;
528  G4double tolh = 0.5*kCarTolerance;
529 
530  if(calcNorm) { *validNorm = false; }
531 
532  // We have that the particle p follows the line x = p + s * v
533  // meaning x = p.x() + s * v.x(), y = p.y() + s * v.y() and
534  // z = p.z() + s * v.z()
535  // The equation for all points on the surface (surface expanded for
536  // to include all z) x^2 + y^2 = k1 * z + k2 => .. =>
537  // => s = (A +- std::sqrt(A^2 + B)) / vRho2
538  // where:
539  //
540  G4double A = k1 / 2 * v.z() - p.x() * v.x() - p.y() * v.y();
541  //
542  // and:
543  //
544  G4double B = (-rho2 + paraRho2) * vRho2;
545 
546  if ( rho2 < paraRho2 && sqr(rho2 - paraRho2 - 0.25 * tol2) > tol2 * paraRho2
547  && std::fabs(p.z()) < dz - kCarTolerance)
548  {
549  // Make sure it's safely inside.
550 
551  if(v.z() > 0)
552  {
553  // It's heading upwards, check where it colides with the plane z = dz.
554  // When it does, is that in the surface of the paraboloid.
555  // z = p.z() + variable * v.z() for all points the particle can go.
556  // => variable = (z - p.z()) / v.z() so intersection must be:
557 
558  intersection = (dz - p.z()) / v.z();
559  G4ThreeVector ip = p + intersection * v; // Point of intersection.
560 
561  if(ip.perp2() < sqr(r2 + kCarTolerance))
562  {
563  if(calcNorm)
564  {
565  *n = G4ThreeVector(0, 0, 1);
566  if(r2 < tolh || ip.perp2() > sqr(r2 - tolh))
567  {
568  *n += G4ThreeVector(ip.x(), ip.y(), - k1 / 2).unit();
569  *n = n->unit();
570  }
571  *validNorm = true;
572  }
573  return intersection;
574  }
575  }
576  else if(v.z() < 0)
577  {
578  // It's heading downwards, check were it colides with the plane z = -dz.
579  // When it does, is that in the surface of the paraboloid.
580  // z = p.z() + variable * v.z() for all points the particle can go.
581  // => variable = (z - p.z()) / v.z() so intersection must be:
582 
583  intersection = (-dz - p.z()) / v.z();
584  G4ThreeVector ip = p + intersection * v; // Point of intersection.
585 
586  if(ip.perp2() < sqr(r1 + tolh))
587  {
588  if(calcNorm)
589  {
590  *n = G4ThreeVector(0, 0, -1);
591  if(r1 < tolh || ip.perp2() > sqr(r1 - tolh))
592  {
593  *n += G4ThreeVector(ip.x(), ip.y(), - k1 / 2).unit();
594  *n = n->unit();
595  }
596  *validNorm = true;
597  }
598  return intersection;
599  }
600  }
601 
602  // Now check for collisions with paraboloid surface.
603 
604  if(vRho2 == 0) // Needs to be treated seperately.
605  {
606  intersection = ((rho2 - k2)/k1 - p.z())/v.z();
607  if(calcNorm)
608  {
609  G4ThreeVector intersectionP = p + v * intersection;
610  *n = G4ThreeVector(intersectionP.x(), intersectionP.y(), -k1/2);
611  *n = n->unit();
612 
613  *validNorm = true;
614  }
615  return intersection;
616  }
617  else if( ((A <= 0) && (B >= sqr(A) * (sqr(vRho2) - 1))) || (A >= 0))
618  {
619  // intersection = (A + std::sqrt(B + sqr(A))) / vRho2;
620  // The above calculation has a precision problem:
621  // known problem of solving quadratic equation with small A
622 
623  A = A/vRho2;
624  B = (k1 * p.z() + k2 - rho2)/vRho2;
625  intersection = B/(-A + std::sqrt(B + sqr(A)));
626  if(calcNorm)
627  {
628  G4ThreeVector intersectionP = p + v * intersection;
629  *n = G4ThreeVector(intersectionP.x(), intersectionP.y(), -k1/2);
630  *n = n->unit();
631  *validNorm = true;
632  }
633  return intersection;
634  }
635  std::ostringstream message;
636  message << "There is no intersection between given line and solid!"
637  << G4endl
638  << " p = " << p << G4endl
639  << " v = " << v;
640  G4Exception("G4Paraboloid::DistanceToOut(p,v,...)", "GeomSolids1002",
641  JustWarning, message);
642 
643  return kInfinity;
644  }
645  else if ( (rho2 < paraRho2 + kCarTolerance
646  || sqr(rho2 - paraRho2 - 0.25 * tol2) < tol2 * paraRho2 )
647  && std::fabs(p.z()) < dz + tolh)
648  {
649  // If this is true we're somewhere in the border.
650 
651  G4ThreeVector normal = G4ThreeVector (p.x(), p.y(), -k1/2);
652 
653  if(std::fabs(p.z()) > dz - tolh)
654  {
655  // We're in the lower or upper edge
656  //
657  if( ((v.z() > 0) && (p.z() > 0)) || ((v.z() < 0) && (p.z() < 0)) )
658  { // If we're heading out of the object that is treated here
659  if(calcNorm)
660  {
661  *validNorm = true;
662  if(p.z() > 0)
663  { *n = G4ThreeVector(0, 0, 1); }
664  else
665  { *n = G4ThreeVector(0, 0, -1); }
666  }
667  return 0;
668  }
