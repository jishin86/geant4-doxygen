97 void G4Scene::CalculateExtent ()
98 {
99  G4BoundingSphereScene boundingSphereScene;
100 
101  for (size_t i = 0; i < fRunDurationModelList.size(); i++) {
102  if (fRunDurationModelList[i].fActive) {
103  G4VModel* model = fRunDurationModelList[i].fpModel;
104  if (model -> Validate()) { // Validates and also recomputes extent.
105  const G4VisExtent& thisExtent = model -> GetExtent ();
106  G4double thisRadius = thisExtent.GetExtentRadius ();
107  if (thisRadius > 0.) {
108  G4Point3D thisCentre = thisExtent.GetExtentCentre ();
109  thisCentre.transform (model -> GetTransformation ());
110  boundingSphereScene.AccrueBoundingSphere (thisCentre, thisRadius);
111  }
112  } else {
113  G4ExceptionDescription ed;
114  ed << "Invalid model \"" << model->GetGlobalDescription()
115  << "\".\n Not included in extent calculation.";
116  G4Exception
117  ("G4Scene::CalculateExtent",
118  "visman0201", JustWarning, ed);
119  }
120  }
121  }
122 
123  for (size_t i = 0; i < fEndOfEventModelList.size(); i++) {
124  if (fEndOfEventModelList[i].fActive) {
125  G4VModel* model = fEndOfEventModelList[i].fpModel;

128  G4double thisRadius = thisExtent.GetExtentRadius ();
129  if (thisRadius > 0.) {
130  G4Point3D thisCentre = thisExtent.GetExtentCentre ();
131  thisCentre.transform (model -> GetTransformation ());

141  }

145  for (size_t i = 0; i < fEndOfRunModelList.size(); i++) {

181 G4bool G4Scene::AddWorldIfEmpty (G4bool warn) {
182  G4bool successful = true;
183  if (IsEmpty ()) {
184  successful = false;
185  G4VPhysicalVolume* pWorld =
186  G4TransportationManager::GetTransportationManager ()
187  -> GetNavigatorForTracking () -> GetWorldVolume ();
188  if (pWorld) {
189  const G4VisAttributes* pVisAttribs =
190  pWorld -> GetLogicalVolume () -> GetVisAttributes ();
191  if (!pVisAttribs || pVisAttribs -> IsVisible ()) {
192  if (warn) {
193  G4cout <<
194  "Your \"world\" has no vis attributes or is marked as visible."
195  "\n For a better view of the contents, mark the world as"
196  " invisible, e.g.,"
197  "\n myWorldLogicalVol ->"
198  " SetVisAttributes (G4VisAttributes::GetInvisible());"
199  << G4endl;
200  }
201  }
202  successful = AddRunDurationModel (new G4PhysicalVolumeModel (pWorld));
203  // Note: default depth and no modeling parameters.
204  if (successful) {
205  if (warn) {
206  G4cout <<
207  "G4Scene::AddWorldIfEmpty: The scene was empty of run-duration models."
208  "\n \"world\" has been added.";
209  G4cout << G4endl;
210  }
211  }
212  }
213  }
214  return successful;
215 }
