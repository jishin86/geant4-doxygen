134 void G4EmDNAPhysics_option2::ConstructProcess()
135 {
136  if(verbose > 1) {
137  G4cout << "### " << GetPhysicsName() << " Construct Processes " << G4endl;
138  }
139  G4PhysicsListHelper* ph = G4PhysicsListHelper::GetPhysicsListHelper();
140 
141  auto myParticleIterator=GetParticleIterator();
142  myParticleIterator->reset();
143  while( (*myParticleIterator)() )
144  {
145  G4ParticleDefinition* particle = myParticleIterator->value();
146  G4String particleName = particle->GetParticleName();
147 
148  if (particleName == "e-") {
149 
150  G4DNAElectronSolvation* solvation =
151  new G4DNAElectronSolvation("e-_G4DNAElectronSolvation");
152  G4DNAOneStepThermalizationModel* therm =
153  new G4DNAOneStepThermalizationModel();
154  therm->SetHighEnergyLimit(7.4*eV); // limit of the Champion's model
155  solvation->SetEmModel(therm);
156  ph->RegisterProcess(solvation, particle);
157 
158  // *** Elastic scattering (two alternative models available) ***
159 
160  G4DNAElastic* theDNAElasticProcess = new G4DNAElastic("e-_G4DNAElastic");
161  theDNAElasticProcess->SetEmModel(new G4DNAChampionElasticModel());
162 
163  // or alternative model
164  //theDNAElasticProcess->SetEmModel(new G4DNAScreenedRutherfordElasticModel());
165 
166  ph->RegisterProcess(theDNAElasticProcess, particle);
167 
168  // *** Excitation ***
169  ph->RegisterProcess(new G4DNAExcitation("e-_G4DNAExcitation"), particle);
170 
171  // *** Ionisation ***
172  G4DNAIonisation* theDNAIonisationProcess = new G4DNAIonisation("e-_G4DNAIonisation");
173  G4DNABornIonisationModel* mod = new G4DNABornIonisationModel();
174  mod->SelectFasterComputation(true);
175  theDNAIonisationProcess->SetEmModel(mod);
176  ph->RegisterProcess(theDNAIonisationProcess, particle);
177 
178  // *** Vibrational excitation ***
179  ph->RegisterProcess(new G4DNAVibExcitation("e-_G4DNAVibExcitation"), particle);
180 
181  // *** Attachment ***
182  ph->RegisterProcess(new G4DNAAttachment("e-_G4DNAAttachment"), particle);
183 
184  } else if ( particleName == "proton" ) {
185 
186  ph->RegisterProcess(new G4DNAElastic("proton_G4DNAElastic"), particle);
187 
188  ph->RegisterProcess(new G4DNAExcitation("proton_G4DNAExcitation"), particle);
189 
190  G4DNAIonisation* theDNAIonisationProcess = new G4DNAIonisation("proton_G4DNAIonisation");
191 
192  G4VEmModel* mod1 = new G4DNARuddIonisationExtendedModel();
193  mod1->SetLowEnergyLimit(0*eV);
194  mod1->SetHighEnergyLimit(500*keV);
195 
196  G4DNABornIonisationModel* mod2 = new G4DNABornIonisationModel();
197  mod2->SetLowEnergyLimit(500*keV);
198  mod2->SetHighEnergyLimit(100*MeV);
199  mod2->SelectFasterComputation(true);
200 
201  theDNAIonisationProcess->SetEmModel(mod1);
202  theDNAIonisationProcess->SetEmModel(mod2);
203 
204  ph->RegisterProcess(theDNAIonisationProcess, particle);
205 
206  ph->RegisterProcess(new G4DNAChargeDecrease("proton_G4DNAChargeDecrease"), particle);
207 
208  } else if ( particleName == "hydrogen" ) {
209 
210  ph->RegisterProcess(new G4DNAElastic("hydrogen_G4DNAElastic"), particle);
211 
212  ph->RegisterProcess(new G4DNAExcitation("hydrogen_G4DNAExcitation"), particle);
213 
214  //ph->RegisterProcess(new G4DNAIonisation("hydrogen_G4DNAIonisation"), particle);
215  G4DNAIonisation* theDNAIonisationProcess = new G4DNAIonisation("hydrogen_G4DNAIonisation");
216  theDNAIonisationProcess->SetEmModel(new G4DNARuddIonisationExtendedModel());
217  ph->RegisterProcess(theDNAIonisationProcess, particle);
218 
219  ph->RegisterProcess(new G4DNAChargeIncrease("hydrogen_G4DNAChargeIncrease"), particle);
220 
221  } else if ( particleName == "alpha" ) {
222 
223  ph->RegisterProcess(new G4DNAElastic("alpha_G4DNAElastic"), particle);
224 
225  ph->RegisterProcess(new G4DNAExcitation("alpha_G4DNAExcitation"), particle);
226 
227  G4DNAIonisation* theDNAIonisationProcess = new G4DNAIonisation("alpha_G4DNAIonisation");
228  theDNAIonisationProcess->SetEmModel(new G4DNARuddIonisationExtendedModel());
229  ph->RegisterProcess(theDNAIonisationProcess, particle);
230 
231  ph->RegisterProcess(new G4DNAChargeDecrease("alpha_G4DNAChargeDecrease"), particle);
232 
233  } else if ( particleName == "alpha+" ) {
234 
235  ph->RegisterProcess(new G4DNAElastic("alpha+_G4DNAElastic"), particle);
236 
237  ph->RegisterProcess(new G4DNAExcitation("alpha+_G4DNAExcitation"), particle);
238 
239  G4DNAIonisation* theDNAIonisationProcess = new G4DNAIonisation("alpha+_G4DNAIonisation");
240  theDNAIonisationProcess->SetEmModel(new G4DNARuddIonisationExtendedModel());
241  ph->RegisterProcess(theDNAIonisationProcess, particle);
242 
243  ph->RegisterProcess(new G4DNAChargeDecrease("alpha+_G4DNAChargeDecrease"), particle);
244  ph->RegisterProcess(new G4DNAChargeIncrease("alpha+_G4DNAChargeIncrease"), particle);
245 
246  } else if ( particleName == "helium" ) {
247 
248  ph->RegisterProcess(new G4DNAElastic("helium_G4DNAElastic"), particle);
249 
250  ph->RegisterProcess(new G4DNAExcitation("helium_G4DNAExcitation"), particle);
251 
252  G4DNAIonisation* theDNAIonisationProcess = new G4DNAIonisation("helium_G4DNAIonisation");
253  theDNAIonisationProcess->SetEmModel(new G4DNARuddIonisationExtendedModel());
254  ph->RegisterProcess(theDNAIonisationProcess, particle);
255 
256  ph->RegisterProcess(new G4DNAChargeIncrease("helium_G4DNAChargeIncrease"), particle);
257 
258  } else if ( particleName == "GenericIon" ) {
259  ph->RegisterProcess(new G4DNAIonisation("GenericIon_G4DNAIonisation"), particle);
260  }
261 
262  // Warning : the following particles and processes are needed by EM Physics builders
263  // They are taken from the default Livermore Physics list
264  // These particles are currently not handled by Geant4-DNA
265 
266  // e+
267 
268  else if (particleName == "e+") {
269 
270  G4eMultipleScattering* msc = new G4eMultipleScattering();
271  msc->SetStepLimitType(fUseDistanceToBoundary);
272  G4eIonisation* eIoni = new G4eIonisation();
273  eIoni->SetStepFunction(0.2, 100*um);
274 
275  ph->RegisterProcess(msc, particle);
276  ph->RegisterProcess(eIoni, particle);
277  ph->RegisterProcess(new G4eBremsstrahlung(), particle);
278  ph->RegisterProcess(new G4eplusAnnihilation(), particle);
279 
280  } else if (particleName == "gamma") {
281 
282  // photoelectric effect - Livermore model only
283  G4PhotoElectricEffect* thePhotoElectricEffect = new G4PhotoElectricEffect();
284  thePhotoElectricEffect->SetEmModel(new G4LivermorePhotoElectricModel());
285  ph->RegisterProcess(thePhotoElectricEffect, particle);
286 
287  // Compton scattering - Livermore model only
288  G4ComptonScattering* theComptonScattering = new G4ComptonScattering();
289  theComptonScattering->SetEmModel(new G4LivermoreComptonModel());
290  ph->RegisterProcess(theComptonScattering, particle);
291 
292  // gamma conversion - Livermore model below 80 GeV
293  G4GammaConversion* theGammaConversion = new G4GammaConversion();
294  theGammaConversion->SetEmModel(new G4LivermoreGammaConversionModel());
295  ph->RegisterProcess(theGammaConversion, particle);
296 
297  // default Rayleigh scattering is Livermore
298  G4RayleighScattering* theRayleigh = new G4RayleighScattering();
299  ph->RegisterProcess(theRayleigh, particle);
300  }
301 
302  // Warning : end of particles and processes are needed by EM Physics builders
303 
304  }
305 
306  // Deexcitation
307  //
308  G4VAtomDeexcitation* de = new G4UAtomicDeexcitation();
309  G4LossTableManager::Instance()->SetAtomDeexcitation(de);
310 }
