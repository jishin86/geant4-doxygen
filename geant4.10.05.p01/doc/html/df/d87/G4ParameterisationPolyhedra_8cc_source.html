47 G4VParameterisationPolyhedra::
48 G4VParameterisationPolyhedra( EAxis axis, G4int nDiv, G4double width,
49  G4double offset, G4VSolid* msolid,
50  DivisionType divType )
51  : G4VDivisionParameterisation( axis, nDiv, width, offset, divType, msolid )
52 {
53  std::ostringstream message;
54 #ifdef G4MULTITHREADED
55  message << "Divisions for G4Polyhedra currently NOT supported in MT-mode."
56  << G4endl
57  << "Sorry! Solid: " << msolid->GetName();
58  G4Exception("G4VParameterisationPolyhedra::G4VParameterisationPolyhedra()",
59  "GeomDiv0001", FatalException, message);
60 #endif
61 
62  G4Polyhedra* msol = (G4Polyhedra*)(msolid);
63  if ((msolid->GetEntityType() != "G4ReflectedSolid") && (msol->IsGeneric()))
64  {
65  message << "Generic construct for G4Polyhedra NOT supported." << G4endl
66  << "Sorry! Solid: " << msol->GetName();
67  G4Exception("G4VParameterisationPolyhedra::G4VParameterisationPolyhedra()",
68  "GeomDiv0001", FatalException, message);
69  }
70  if (msolid->GetEntityType() == "G4ReflectedSolid")
71  {
72  // Get constituent solid
73  G4VSolid* mConstituentSolid
74  = ((G4ReflectedSolid*)msolid)->GetConstituentMovedSolid();
75  msol = (G4Polyhedra*)(mConstituentSolid);
76 
77  // Get parameters
78  G4int nofSides = msol->GetOriginalParameters()->numSide;
79  G4int nofZplanes = msol->GetOriginalParameters()->Num_z_planes;
80  G4double* zValues = msol->GetOriginalParameters()->Z_values;
81  G4double* rminValues = msol->GetOriginalParameters()->Rmin;
82  G4double* rmaxValues = msol->GetOriginalParameters()->Rmax;
83 
84  // Invert z values,
85  // convert radius parameters
86  G4double* rminValues2 = new G4double[nofZplanes];
87  G4double* rmaxValues2 = new G4double[nofZplanes];
88  G4double* zValuesRefl = new G4double[nofZplanes];
89  for (G4int i=0; i<nofZplanes; i++)
90  {
91  rminValues2[i] = rminValues[i] * ConvertRadiusFactor(*msol);
92  rmaxValues2[i] = rmaxValues[i] * ConvertRadiusFactor(*msol);
93  zValuesRefl[i] = - zValues[i];
94  }
95 
96  G4Polyhedra* newSolid
97  = new G4Polyhedra(msol->GetName(),
98  msol->GetStartPhi(),
99  msol->GetEndPhi() - msol->GetStartPhi(),
100  nofSides,
101  nofZplanes, zValuesRefl, rminValues2, rmaxValues2);
102 
103  delete [] rminValues2;
104  delete [] rmaxValues2;
105  delete [] zValuesRefl;
106 
107  msol = newSolid;
108  fmotherSolid = newSolid;
109  fReflectedSolid = true;
110  fDeleteSolid = true;
111  }
112 }

119 //--------------------------------------------------------------------------
120 G4double
121 G4VParameterisationPolyhedra::
122 ConvertRadiusFactor(const G4Polyhedra& phedra) const
123 {
124  G4double phiTotal = phedra.GetEndPhi() - phedra.GetStartPhi();
125  G4int nofSides = phedra.GetOriginalParameters()->numSide;
126 
127  if ( (phiTotal <=0) || (phiTotal >
128  2*pi+G4GeometryTolerance::GetInstance()->GetAngularTolerance()) )
129  { phiTotal = 2*pi; }
130 
131  return std::cos(0.5*phiTotal/nofSides);
132 }

251 ComputeDimensions( G4Polyhedra& phedra, const G4int copyNo,
252  const G4VPhysicalVolume* ) const
253 {
254  G4Polyhedra* msol = (G4Polyhedra*)(fmotherSolid);

259 
260  G4double width = 0.;

266  origparam.Rmax[ii] = origparamMother->Rmin[ii]+foffset+width*(copyNo+1);
267  }
268 
269  phedra.SetOriginalParameters(&origparam); // copy values & transfer pointers
270  phedra.Reset(); // reset to new solid parameters
271 
272 #ifdef G4DIVDEBUG
273  if( verbose >= -2 )
274  {
275  G4cout << "G4ParameterisationPolyhedraRho::ComputeDimensions()" << G4endl
276  << "-- Parametrised phedra copy-number: " << copyNo << G4endl;
277  phedra.DumpInfo();

472 //------------------------------------------------------------------------
473 G4double G4ParameterisationPolyhedraZ::GetR(G4double z,
474  G4double z1, G4double r1,
475  G4double z2, G4double r2) const
476 {
477  // Linear parameterisation:
478  // r = az + b
479  // a = (r1 - r2)/(z1-z2)
480  // b = r1 - a*z1
481 
482  return (r1-r2)/(z1-z2)*z + ( r1 - (r1-r2)/(z1-z2)*z1 ) ;
483 }

516 //---------------------------------------------------------------------
517 void G4ParameterisationPolyhedraZ::CheckParametersValidity()
518 {
519  G4VDivisionParameterisation::CheckParametersValidity();
520 
521  // Division will be following the mother polyhedra segments
522  if( fDivisionType == DivNDIV ) {
523  if( fOrigParamMother->Num_z_planes-1 != fnDiv ) {
524  std::ostringstream message;
525  message << "Configuration not supported." << G4endl
526  << "Division along Z will be done splitting in the defined"
527  << G4endl
528  << "Z planes, i.e, the number of division would be :"
529  << fOrigParamMother->Num_z_planes-1 << " instead of "
530  << fnDiv << " !";
531  G4Exception("G4ParameterisationPolyhedraZ::CheckParametersValidity()",
532  "GeomDiv0001", FatalException, message);
533  }
534  }
535 
536  // Division will be done within one polyhedra segment
537  // with applying given width and offset
538  if( fDivisionType == DivNDIVandWIDTH || fDivisionType == DivWIDTH ) {
539  // Check if divided region does not span over more
540  // than one z segment
541 
542  G4int isegstart = -1; // number of the segment containing start position
543  G4int isegend = -1; // number of the segment containing end position
544 
545  if ( ! fReflectedSolid ) {
546  // The start/end position of the divided region
547  G4double zstart
548  = fOrigParamMother->Z_values[0] + foffset;
549  G4double zend
550  = fOrigParamMother->Z_values[0] + foffset + fnDiv* fwidth;
551 
552  G4int counter = 0;
553  while ( isegend < 0 && counter < fOrigParamMother->Num_z_planes-1 ) {
554  // first segment
555  if ( zstart >= fOrigParamMother->Z_values[counter] &&
556  zstart < fOrigParamMother->Z_values[counter+1] ) {
557  isegstart = counter;
558  }
559  // last segment
560  if ( zend > fOrigParamMother->Z_values[counter] &&
561  zend <= fOrigParamMother->Z_values[counter+1] ) {
562  isegend = counter;
563  }
564  ++counter;
565  } // Loop checking, 06.08.2015, G.Cosmo
566  }
567  else {
568  // The start/end position of the divided region
569  G4double zstart
570  = fOrigParamMother->Z_values[0] - foffset;
571  G4double zend
572  = fOrigParamMother->Z_values[0] - ( foffset + fnDiv* fwidth);
573 
574  G4int counter = 0;
575  while ( isegend < 0 && counter < fOrigParamMother->Num_z_planes-1 ) {
576  // first segment
577  if ( zstart <= fOrigParamMother->Z_values[counter] &&
578  zstart > fOrigParamMother->Z_values[counter+1] ) {
579  isegstart = counter;
580  }
581  // last segment
582  if ( zend < fOrigParamMother->Z_values[counter] &&
583  zend >= fOrigParamMother->Z_values[counter+1] ) {
584  isegend = counter;
585  }
586  ++counter;
587  } // Loop checking, 06.08.2015, G.Cosmo
588  }
589 
590  if ( isegstart != isegend ) {
591  std::ostringstream message;
592  message << "Configuration not supported." << G4endl
593  << "Division with user defined width." << G4endl
594  << "Solid " << fmotherSolid->GetName() << G4endl
595  << "Divided region is not between two Z planes.";
596  G4Exception("G4ParameterisationPolyhedraZ::CheckParametersValidity()",
597  "GeomDiv0001", FatalException, message);
598  }
599 
600  fNSegment = isegstart;
601  }

604 //---------------------------------------------------------------------
605 void
606 G4ParameterisationPolyhedraZ::
607 ComputeTransformation( const G4int copyNo, G4VPhysicalVolume* physVol) const
608 {
609  if ( fDivisionType == DivNDIV ) {
610  // The position of the centre of copyNo-th mother polycone segment
611  G4double posi = ( fOrigParamMother->Z_values[copyNo]
612  + fOrigParamMother->Z_values[copyNo+1])/2;
613  physVol->SetTranslation( G4ThreeVector(0, 0, posi) );
614  }
615 
616  if ( fDivisionType == DivNDIVandWIDTH || fDivisionType == DivWIDTH ) {
617  // The position of the centre of copyNo-th division
618 
619  G4double posi = fOrigParamMother->Z_values[0];
620 
621  if ( ! fReflectedSolid )
622  posi += foffset + (2*copyNo + 1) * fwidth/2.;
623  else
624  posi -= foffset + (2*copyNo + 1) * fwidth/2.;
625 
626  physVol->SetTranslation( G4ThreeVector(0, 0, posi) );
627  }
628 
629  //----- calculate rotation matrix: unit
630 
631 #ifdef G4DIVDEBUG
632  if( verbose >= 2 )
633  {
634  G4cout << " G4ParameterisationPolyhedraZ - position: " << posi << G4endl
635  << " copyNo: " << copyNo << " - foffset: " << foffset/deg
636  << " - fwidth: " << fwidth/deg << G4endl;
637  }
638 #endif
639 
640  ChangeRotMatrix( physVol );
641 
642 #ifdef G4DIVDEBUG
643  if( verbose >= 2 )
644  {

654 void
655 G4ParameterisationPolyhedraZ::
656 ComputeDimensions( G4Polyhedra& phedra, const G4int copyNo,
657  const G4VPhysicalVolume* ) const
658 {
659  // Define division solid
660  G4PolyhedraHistorical origparam;
661  G4int nz = 2;
662  origparam.Num_z_planes = nz;
663  origparam.numSide = fOrigParamMother->numSide;
664  origparam.Start_angle = fOrigParamMother->Start_angle;
665  origparam.Opening_angle = fOrigParamMother->Opening_angle;
666 
667  // Define division solid z sections
668  origparam.Z_values = new G4double[nz];
669  origparam.Rmin = new G4double[nz];
670  origparam.Rmax = new G4double[nz];
671  origparam.Z_values[0] = - fwidth/2.;
672  origparam.Z_values[1] = fwidth/2.;
673 
674  if ( fDivisionType == DivNDIV ) {
675  // The position of the centre of copyNo-th mother polycone segment
676  G4double posi = ( fOrigParamMother->Z_values[copyNo]
677  + fOrigParamMother->Z_values[copyNo+1])/2;
678 
679  origparam.Z_values[0] = fOrigParamMother->Z_values[copyNo] - posi;
680  origparam.Z_values[1] = fOrigParamMother->Z_values[copyNo+1] - posi;
681  origparam.Rmin[0] = fOrigParamMother->Rmin[copyNo];
682  origparam.Rmin[1] = fOrigParamMother->Rmin[copyNo+1];
683  origparam.Rmax[0] = fOrigParamMother->Rmax[copyNo];
684  origparam.Rmax[1] = fOrigParamMother->Rmax[copyNo+1];
685  }
686 
687  if ( fDivisionType == DivNDIVandWIDTH || fDivisionType == DivWIDTH ) {
688  if ( ! fReflectedSolid ) {
689  origparam.Z_values[0] = - fwidth/2.;
690  origparam.Z_values[1] = fwidth/2.;
691 
692  // The position of the centre of copyNo-th division
693  G4double posi
694  = fOrigParamMother->Z_values[0] + foffset + (2*copyNo + 1) * fwidth/2.;
695 
696  // The first and last z sides z values
697  G4double zstart = posi - fwidth/2.;
698  G4double zend = posi + fwidth/2.;
699  origparam.Rmin[0] = GetRmin(zstart, fNSegment);
700  origparam.Rmax[0] = GetRmax(zstart, fNSegment);
701  origparam.Rmin[1] = GetRmin(zend, fNSegment);
702  origparam.Rmax[1] = GetRmax(zend, fNSegment);
703  }
704  else {
705  origparam.Z_values[0] = fwidth/2.;
706  origparam.Z_values[1] = - fwidth/2.;
707 
708  // The position of the centre of copyNo-th division
709  G4double posi
710  = fOrigParamMother->Z_values[0] - ( foffset + (2*copyNo + 1) * fwidth/2.);
711 
712  // The first and last z sides z values
713  G4double zstart = posi + fwidth/2.;
714  G4double zend = posi - fwidth/2.;
715  origparam.Rmin[0] = GetRmin(zstart, fNSegment);
716  origparam.Rmax[0] = GetRmax(zstart, fNSegment);
717  origparam.Rmin[1] = GetRmin(zend, fNSegment);
718  origparam.Rmax[1] = GetRmax(zend, fNSegment);
719  }
720 
721  // It can happen due to rounding errors
722  if ( origparam.Rmin[0] < 0.0 ) origparam.Rmin[0] = 0.0;
723  if ( origparam.Rmin[nz-1] < 0.0 ) origparam.Rmin[1] = 0.0;
724  }
725 
726  phedra.SetOriginalParameters(&origparam); // copy values & transfer pointers
727  phedra.Reset(); // reset to new solid parameters
728 
729 #ifdef G4DIVDEBUG
730  if( verbose >= 2 )
731  {
732  G4cout << "G4ParameterisationPolyhedraZ::ComputeDimensions()" << G4endl
733  << "-- Parametrised phedra copy-number: " << copyNo << G4endl;
734  phedra.DumpInfo();
735  }
736 #endif
737 }
