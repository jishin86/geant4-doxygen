71 void XrayFluoSimulation::RunSimulation(int argc,char* argv[])
72 {

74  G4Random::setTheEngine(new CLHEP::RanecuEngine);
75  G4Random::setTheSeed(dir);
76 
77 #ifdef G4MULTITHREADED
78  // Construct the default run manager
79  G4MTRunManager* runManager = new G4MTRunManager();
80  G4cout << "Using the MT Run Manager (G4MULTITHREADED=ON)" << G4endl;

83  G4cout << "Using the sequential Run Manager" << G4endl;
84 #endif
85 
86 
87  // chosing Geometry setup
88  G4int geometryNumber = 0;
89 
90  if (argc == 3){
91  geometryNumber = atoi(argv[2]);
92 
93  }
94  while ( (geometryNumber < 1) || (geometryNumber >4)) {
95  G4cout << "Please Select Simulation Geometrical Set-Up: "<< G4endl;
96  G4cout << "1 - Test Beam" << G4endl;
97  G4cout << "2 - Infinite Plane" << G4endl;
98  G4cout << "3 - Planet and Sun"<< G4endl;
99  G4cout << "4 - Phase-Space Production"<< G4endl;
100 
101  G4cin >> geometryNumber;
102  }
103 
104  // set analysis to have the messenger running...
105  XrayFluoAnalysisManager* analysis =
106  XrayFluoAnalysisManager::getInstance();
107 
108  // set mandatory initialization
109 
110  //Initialize geometry
111  if (geometryNumber == 1 || geometryNumber == 4) {
112  XrayFluoDetectorConstruction* testBeamDetector
113  = XrayFluoDetectorConstruction::GetInstance();
114  if (geometryNumber == 4) {
115  testBeamDetector->PhaseSpaceOn();
116  analysis->PhaseSpaceOn();
117  }
118  runManager->SetUserInitialization(testBeamDetector);
119  }
120  else if (geometryNumber == 2) {
121  XrayFluoPlaneDetectorConstruction*
122  planeDetector = XrayFluoPlaneDetectorConstruction::GetInstance();
123  runManager->SetUserInitialization(planeDetector);
124  }
125  else if (geometryNumber == 3) {
126  XrayFluoMercuryDetectorConstruction* mercuryDetector =
127  XrayFluoMercuryDetectorConstruction::GetInstance();
128  runManager->SetUserInitialization(mercuryDetector);
129  }
130 
131 
132  //Initialize physics
133  runManager->SetUserInitialization(new XrayFluoPhysicsList());
134 
135  //Initialize actions
136  runManager->SetUserInitialization
137  (new XrayFluoActionInitializer(geometryNumber));
138 
139 #ifdef G4VIS_USE
140  //visualization manager
141  G4VisManager* visManager = new G4VisExecutive;
142  visManager->Initialize();
143 #endif
144 
145  // get the pointer to the User Interface manager
146  G4UImanager* UImanager = G4UImanager::GetUIpointer();
147 
148  if (getenv("G4VIS_USE")) {
149  UImanager->ApplyCommand("/control/execute vis.mac");
150  }
151 
152  if (argc == 1) // Define UI session for interactive mode.
153  {
154  UImanager->ApplyCommand("/control/execute initInter.mac");
155 #ifdef G4UI_USE
156  G4UIExecutive* ui = new G4UIExecutive(argc, argv);
157  ui->SessionStart();
158  delete ui;
159 #endif
160  }
161  else // Batch mode
162  {
163  G4String command = "/control/execute ";
164  G4String fileName = argv[1];
165  UImanager->ApplyCommand(command+fileName);
166  }
167 
168  // job termination
169 #ifdef G4VIS_USE
170  delete visManager;
171  G4cout << "visManager deleted"<< G4endl;
172 #endif
173 
174 
175  delete runManager;
176 
177 
178 }
