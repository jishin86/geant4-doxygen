69 G4double G4HETCDeuteron::K(const G4Fragment & aFragment)
70 {
71  // Number of protons in emitted fragment
72  G4int Pa = theZ;
73  // Number of neutrons in emitted fragment
74  G4int Na = theA - Pa;
75 
76  G4double r = G4double(theResZ)/G4double(theResA);
77 
78  G4int P = aFragment.GetNumberOfParticles();
79  G4int H = aFragment.GetNumberOfHoles();
80 
81  G4double result = 0.0;
82  if (P > 1) {
83  result = 2.*(H*(H-1.0)*r*(r-1.0)+H*(Na*r+Pa*(1.0-r)) + Pa*Na)
84  /(P*(P-1.0)*r*(1.0 - r));
85  }
86  return std::max(0.0,result);
87 }
