112 void G4ModelingParameters::SetVisibleDensity (G4double visibleDensity) {
113  const G4double reasonableMaximum = 10.0 * g / cm3;
114  if (visibleDensity < 0 && fWarning) {
115  G4cout << "G4ModelingParameters::SetVisibleDensity: attempt to set negative "
116  "density - ignored." << G4endl;
117  }
118  else {
119  if (fVisibleDensity > reasonableMaximum && fWarning) {
120  G4cout << "G4ModelingParameters::SetVisibleDensity: density > "
121  << reasonableMaximum
122  << " g / cm3 - did you mean this?"
123  << G4endl;
124  }
125  fVisibleDensity = visibleDensity;
126  }
127 }

129 G4int G4ModelingParameters::SetNoOfSides (G4int nSides) {
130  const G4int nSidesMin = fpDefaultVisAttributes->GetMinLineSegmentsPerCircle();
131  if (nSides < nSidesMin) {
132  nSides = nSidesMin;
133  if (fWarning)
134  G4cout << "G4ModelingParameters::SetNoOfSides: attempt to set the"
135  "\nnumber of sides per circle < " << nSidesMin
136  << "; forced to" << nSides << G4endl;
137  }
138  fNoOfSides = nSides;
139  return fNoOfSides;
140 }

154 std::ostream& operator << (std::ostream& os, const G4ModelingParameters& mp)
155 {
156  os << "Modeling parameters (warning ";
157  if (mp.fWarning) os << "true";
158  else os << "false";
159  os << "):";
160 
161  const G4VisAttributes* va = mp.fpDefaultVisAttributes;
162  os << "\n Default vis. attributes: ";
163  if (va) os << *va;
164  else os << "none";
165 
166  os << "\n Current requested drawing style: ";
167  switch (mp.fDrawingStyle) {
168  case G4ModelingParameters::wf:
169  os << "wireframe"; break;
170  case G4ModelingParameters::hlr:
171  os << "hidden line removal (hlr)"; break;
172  case G4ModelingParameters::hsr:
173  os << "surface (hsr)"; break;
174  case G4ModelingParameters::hlhsr:
175  os << "surface and edges (hlhsr)"; break;
176  default: os << "unrecognised"; break;
177  }
178 
179  os << "\n Culling: ";
180  if (mp.fCulling) os << "on";
181  else os << "off";
182 
183  os << "\n Culling invisible objects: ";
184  if (mp.fCullInvisible) os << "on";
185  else os << "off";
186 
187  os << "\n Density culling: ";
188  if (mp.fDensityCulling) {
189  os << "on - invisible if density less than "
190  << mp.fVisibleDensity / (1. * g / cm3) << " g cm^-3";
191  }
192  else os << "off";
193 
194  os << "\n Culling daughters covered by opaque mothers: ";
195  if (mp.fCullCovered) os << "on";
196  else os << "off";
197 
198  os << "\n Colour by density: ";
199  if (mp.fCBDAlgorithmNumber <= 0) {
200  os << "inactive";
201  } else {
202  os << "Algorithm " << mp.fCBDAlgorithmNumber << ", Parameters:";
203  for (auto p: mp.fCBDParameters) {
204  os << ' ' << G4BestUnit(p,"Volumic Mass");
205  }
206  }
207 
208  os << "\n Explode factor: " << mp.fExplodeFactor
209  << " about centre: " << mp.fExplodeCentre;
210 
211  os << "\n No. of sides used in circle polygon approximation: "
212  << mp.fNoOfSides;
213 
214  os << "\n Section (DCUT) shape (G4VSolid) pointer: ";
215  if (!mp.fpSectionSolid) os << "non-";
216  os << "null";
217 
218  os << "\n Cutaway (DCUT) shape (G4VSolid) pointer: ";
219  if (!mp.fpCutawaySolid) os << "non-";
220  os << "null";
221 
222  os << "\n Event pointer: " << mp.fpEvent;
223 
224  os << "\n Vis attributes modifiers: ";
225  const std::vector<G4ModelingParameters::VisAttributesModifier>& vams =
226  mp.fVisAttributesModifiers;
227  if (vams.empty()) {
228  os << "None";
229  } else {
230  os << vams;
231  }
232 
233  return os;
234 }

236 G4bool G4ModelingParameters::operator !=
237 (const G4ModelingParameters& mp) const {
238 
239  if (
240  (fWarning != mp.fWarning) ||
241  (*fpDefaultVisAttributes != *mp.fpDefaultVisAttributes) ||
242  (fCulling != mp.fCulling) ||
243  (fCullInvisible != mp.fCullInvisible) ||
244  (fDensityCulling != mp.fDensityCulling) ||
245  (fCullCovered != mp.fCullCovered) ||
246  (fCBDAlgorithmNumber != mp.fCBDAlgorithmNumber) ||
247  (fExplodeFactor != mp.fExplodeFactor) ||
248  (fExplodeCentre != mp.fExplodeCentre) ||
249  (fNoOfSides != mp.fNoOfSides) ||
250  (fpSectionSolid != mp.fpSectionSolid) ||
251  (fpCutawaySolid != mp.fpCutawaySolid) ||
252  (fpEvent != mp.fpEvent)
253  )
254  return true;
255 
256  if (fDensityCulling &&
257  (fVisibleDensity != mp.fVisibleDensity)) return true;
258 
259  if (fCBDAlgorithmNumber > 0) {
260  if (fCBDParameters.size() != mp.fCBDParameters.size()) return true;
261  else if (fCBDParameters != mp.fCBDParameters) return true;
262  }
263 
264  if (fVisAttributesModifiers != mp.fVisAttributesModifiers)
265  return true;
266 
267  return false;
268 }

270 G4bool G4ModelingParameters::VisAttributesModifier::operator!=
271 (const G4ModelingParameters::VisAttributesModifier& rhs) const
272 {
273  if (fSignifier != rhs.fSignifier) return true;
274  if (fPVNameCopyNoPath != rhs.fPVNameCopyNoPath) return true;
275  switch (fSignifier) {
276  case G4ModelingParameters::VASVisibility:
277  if (fVisAtts.IsVisible() != rhs.fVisAtts.IsVisible())
278  return true;
279  break;
280  case G4ModelingParameters::VASDaughtersInvisible:
281  if (fVisAtts.IsDaughtersInvisible() !=
282  rhs.fVisAtts.IsDaughtersInvisible())
283  return true;
284  break;
285  case G4ModelingParameters::VASColour:
286  if (fVisAtts.GetColour() != rhs.fVisAtts.GetColour())
287  return true;
288  break;
289  case G4ModelingParameters::VASLineStyle:
290  if (fVisAtts.GetLineStyle() != rhs.fVisAtts.GetLineStyle())
291  return true;
292  break;
293  case G4ModelingParameters::VASLineWidth:
294  if (fVisAtts.GetLineWidth() != rhs.fVisAtts.GetLineWidth())
295  return true;
296  break;
297  case G4ModelingParameters::VASForceWireframe:
298  if (fVisAtts.GetForcedDrawingStyle() !=
299  rhs.fVisAtts.GetForcedDrawingStyle())
300  return true;
301  break;
302  case G4ModelingParameters::VASForceSolid:
303  if (fVisAtts.GetForcedDrawingStyle() !=
304  rhs.fVisAtts.GetForcedDrawingStyle())
305  return true;
306  break;
307  case G4ModelingParameters::VASForceAuxEdgeVisible:
308  if (fVisAtts.IsForceAuxEdgeVisible() !=
309  rhs.fVisAtts.IsForceAuxEdgeVisible() ||
310  fVisAtts.IsForcedAuxEdgeVisible() !=
311  rhs.fVisAtts.IsForcedAuxEdgeVisible())
312  return true;
313  break;
314  case G4ModelingParameters::VASForceLineSegmentsPerCircle:
315  if (fVisAtts.GetForcedLineSegmentsPerCircle() !=
316  rhs.fVisAtts.GetForcedLineSegmentsPerCircle())
317  return true;
318  break;
319  }
320  return false;
321 }

374  const std::vector<G4ModelingParameters::VisAttributesModifier>& vams)

376  std::vector<G4ModelingParameters::VisAttributesModifier>::const_iterator

382  iModifier->GetPVNameCopyNoPath();
383  os << '\n' << vamPath;
384  const G4VisAttributes& vamVisAtts = iModifier->GetVisAttributes();
385  const G4Colour& c = vamVisAtts.GetColour();
386  switch (iModifier->GetVisAttributesSignifier()) {
387  case G4ModelingParameters::VASVisibility:
388  os << " visibility ";

403  case G4ModelingParameters::VASColour:
404  os << " colour " << c;
405  break;
406  case G4ModelingParameters::VASLineStyle:
407  os << " lineStyle ";
408  switch (vamVisAtts.GetLineStyle()) {
409  case G4VisAttributes::unbroken:
410  os << "unbroken";
411  break;
412  case G4VisAttributes::dashed:
413  os << "dashed";
414  break;
415  case G4VisAttributes::dotted:

417  }
418  break;
419  case G4ModelingParameters::VASLineWidth:
420  os << " lineWidth "

422  break;
423  case G4ModelingParameters::VASForceWireframe:
424  if (vamVisAtts.GetForcedDrawingStyle() == G4VisAttributes::wireframe) {
