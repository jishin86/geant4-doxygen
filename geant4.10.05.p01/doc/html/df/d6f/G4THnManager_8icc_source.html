61 template <typename T>
62 T* G4THnManager<T>::GetTInFunction(G4int id,
63  G4String functionName, G4bool warn,
64  G4bool onlyIfActive) const
65 {
66  G4int index = id - fHnManager->GetFirstId();
67  if ( index < 0 || index >= G4int(fTVector.size()) ) {
68  if ( warn) {
69  G4String inFunction = "G4THnManager::";
70  inFunction += functionName;
71  G4ExceptionDescription description;
72  description << " " << "histogram " << id << " does not exist.";
73  G4Exception(inFunction, "Analysis_W011", JustWarning, description);
74  }
75  return nullptr;
76  }
77 
78  // Do not return histogram if inactive
79  if ( fState.GetIsActivation() && onlyIfActive && ( ! fHnManager->GetActivation(id) ) ) {
80  return nullptr;
81  }
82 
83  return fTVector[index];
84 }

88 G4int G4THnManager<T>::RegisterT(T* t, const G4String& name)
89 {
90  G4int index = fTVector.size();
91  fTVector.push_back(t);
92 
93  fHnManager->SetLockFirstId(true);
94  fNameIdMap[name] = index + fHnManager->GetFirstId();
95  return index + fHnManager->GetFirstId();

108  G4Exception(inFunction, "Analysis_W011", JustWarning, description);
109  }
