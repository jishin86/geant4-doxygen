89  // Find a value x between x1 and x2 at which ddxPower[PDF](x) = p.
90  // ddxPower = -1 -> CDF;
91  // ddxPower = 0 -> PDF
92  // ddxPower = 1 -> PDF'

96  // If not found in range, returns the nearest boundary.
