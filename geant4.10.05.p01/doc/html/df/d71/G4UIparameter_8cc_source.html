209 G4int G4UIparameter::
210 TypeCheck(const char* newValue)
211 {
212  G4String newValueString(newValue);
213  char type = toupper( parameterType );
214  switch(type) {
215  case 'D':
216  if( IsDouble(newValueString.data())== 0) {
217  G4cerr<<newValue<<": double value expected."
218  << G4endl;
219  return 0;
220  } break;
221  case 'I':
222  if( IsInt(newValueString.data(),20)== 0) {
223  G4cerr<<newValue<<": integer expected."
224  << G4endl;
225  return 0;
226  } break;
227  case 'S': break;
228  case 'B':
229  newValueString.toUpper();
230  if ( newValueString == "Y" || newValueString == "N"
231  ||newValueString == "YES" || newValueString == "NO"
232  ||newValueString == "1" || newValueString == "0"
233  ||newValueString == "T" || newValueString == "F"
234  ||newValueString == "TRUE" || newValueString == "FALSE")
235  return 1;
236  else {
237  G4cerr<<newValue<<": bool expected." << G4endl;
238  return 0;
239  }
240  default: ;
241  }
242  return 1;
243 }

279 IsDouble(const char* buf) // see state diagram for this spec.
280 {
281  const char* p= buf;
282  switch( *p) {
283  case '+': case '-': ++p;
284  if( isdigit(*p) ) {
285  while( isdigit( (G4int)(*p) )) { ++p; }
286  switch ( *p ) {
287  case '\0': return 1; //break;
288  case 'E': case 'e':
289  return ExpectExponent(++p ); //break;
290  case '.': ++p;
291  if( *p == '\0' ) return 1;
292  if( *p == 'e' || *p =='E' ) return ExpectExponent(++p );
293  if( isdigit(*p) ) {
294  while( isdigit( (G4int)(*p) )) { ++p; }
295  if( *p == '\0' ) return 1;
296  if( *p == 'e' || *p =='E') return ExpectExponent(++p);
297  } else return 0; break;
298  default: return 0;
299  }
300  }
301  if( *p == '.' ) { ++p;
302  if( isdigit(*p) ) {
303  while( isdigit( (G4int)(*p) )) { ++p; }
304  if( *p == '\0' ) return 1;
305  if( *p == 'e' || *p =='E') return ExpectExponent(++p);
306  }
307  }
308  break;
309  case '.': ++p;
310  if( isdigit(*p) ) {
311  while( isdigit( (G4int)(*p) )) { ++p; }
312  if( *p == '\0' ) return 1;
313  if( *p == 'e' || *p =='E' ) return ExpectExponent(++p);
314  } break;
315  default: // digit is expected
316  if( isdigit(*p) ) {
317  while( isdigit( (G4int)(*p) )) { ++p; }
318  if( *p == '\0' ) return 1;
319  if( *p == 'e' || *p =='E') return ExpectExponent(++p);
320  if( *p == '.' ) { ++p;
321  if( *p == '\0' ) return 1;
322  if( *p == 'e' || *p =='E') return ExpectExponent(++p);
323  if( isdigit(*p) ) {
324  while( isdigit( (G4int)(*p) )) { ++p; }
325  if( *p == '\0' ) return 1;
326  if( *p == 'e' || *p =='E') return ExpectExponent(++p);
327  }
328  }
329  }
330  }

348 yystype G4UIparameter::
349 LogicalORExpression(void)
350 {
351  yystype result;
352  yystype p;
353  p = LogicalANDExpression();
354  if( token != LOGICALOR) return p;
355  if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
356  G4cerr << "Parameter range: illegal type at '||'" << G4endl;
357  paramERR = 1;
358  }
359  result.I = p.I;
360  while (token == LOGICALOR)
361  {
362  token = Yylex();
363  p = LogicalANDExpression();
364  if( p.type == CONSTSTRING || p.type == IDENTIFIER ) {
365  G4cerr << "Parameter range: illegal type at '||'" <<G4endl;
366  paramERR = 1;
367  }
368  switch (p.type) {
369  case CONSTINT:
370  result.I += p.I;
371  result.type = CONSTINT; break;
372  case CONSTDOUBLE:
373  result.I += (p.D != 0.0);
374  result.type = CONSTINT; break;
375  default:
376  G4cerr << "Parameter range: unknown type"<<G4endl;
377  paramERR = 1;
378  }
379  }
380  return result;
381 }

577 G4int G4UIparameter::
578 Eval2(yystype arg1, G4int op, yystype arg2)
579 {
580  if( (arg1.type != IDENTIFIER) && (arg2.type != IDENTIFIER)) {
581  G4cerr << parameterName
582  << ": meaningless comparison "
583  << G4int(arg1.type) << " " << G4int(arg2.type) << G4endl;
584  paramERR = 1;
585  }
586  char type = toupper( parameterType );
587  if( arg1.type == IDENTIFIER) {
588  switch (type) {
589  case 'I':
590  if ( arg2.type == CONSTINT ) {
591  return CompareInt( newVal.I, op, arg2.I );
592  } else {
593  G4cerr << "integer operand expected for "
594  << parameterRange << '.'
595  << G4endl;
596  }
597  break;
598  case 'D':
599  if ( arg2.type == CONSTDOUBLE ) {
600  return CompareDouble( newVal.D, op, arg2.D );
601  } else
602  if ( arg2.type == CONSTINT ) { // integral promotion
603  return CompareDouble( newVal.D, op, arg2.I );
604  } break;
605  default: ;
606  }
607  }
608  if( arg2.type == IDENTIFIER) {
609  switch (type) {
610  case 'I':
611  if ( arg1.type == CONSTINT ) {
612  return CompareInt( arg1.I, op, newVal.I );
613  } else {
614  G4cerr << "integer operand expected for "
615  << parameterRange << '.'
616  << G4endl;
617  }
618  break;
619  case 'D':
620  if ( arg1.type == CONSTDOUBLE ) {
621  return CompareDouble( arg1.D, op, newVal.D );
622  } else
623  if ( arg1.type == CONSTINT ) { // integral promotion
624  return CompareDouble( arg1.I, op, newVal.D );
625  } break;
626  default: ;
627  }
628  }
629  G4cerr << "no param name is specified at the param range."<<G4endl;
630  return 0;
631 }

685 tokenNum G4UIparameter::
686 Yylex() // reads input and returns token number KR486
687 { // (returns EOF)
688  G4int c;
689  G4String buf;
690 
691  while(( c= G4UIpGetc())==' '|| c=='\t' || c== '\n' )
692  ;
693  if (c== EOF)
694  return (tokenNum)EOF; // KR488
695  buf= "";
696  if (isdigit(c) || c== '.') { // I or D
697  do {
698  buf += G4String((unsigned char)c);
699  c=G4UIpGetc();
700  } while (c=='.' || isdigit(c) ||
701  c=='e' || c=='E' || c=='+' || c=='-');
702  G4UIpUngetc(c);
703  const char* t = buf;
704  std::istringstream is(t);
705  if ( IsInt(buf.data(),20) ) {
