42 G4Run* G4MultiRunAction::GenerateRun() {
43  G4Run* aRun = nullptr;
44  for ( auto& ru : *this ) {
45  auto anotherRun = ru->GenerateRun();
46  if ( aRun != nullptr && anotherRun != nullptr ) {
47  G4Exception("G4MultiRunAction::GenerateRun()","Run0036",FatalException,
48  "More than one registered UserRunAction return an instance"\
49  " of G4Run, not allowed.");
50  return nullptr;
51  }
52  if( anotherRun != nullptr) aRun = anotherRun;
53  }
54  return aRun;
55 }
