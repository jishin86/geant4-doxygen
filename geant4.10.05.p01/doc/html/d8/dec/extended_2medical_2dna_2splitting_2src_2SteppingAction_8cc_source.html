62  "Transportation")

64  // Get if hit was an ionization and get energy deposit
65  //
66  G4double edepStep = theStep->GetTotalEnergyDeposit()/eV;
67  G4Region* currentRegion =
68  theStep->GetPreStepPoint()->GetTouchableHandle()->GetVolume()->
69  GetLogicalVolume()->GetRegion();
70 
71  if (edepStep > 0. && (fRegion == currentRegion ) )
72  {
73  const G4String processName = theStep->GetPostStepPoint()->
74  GetProcessDefinedStep()->GetProcessName();
75 
76  if ( "Wrappede-_G4DNAIonisation" == processName ||
77  "e-_G4DNAIonisation" == processName ) {
78  // Get the flag/label
