51 void SteppingAction::UserSteppingAction(const G4Step* aStep)
52 {

55 
56  Run* run = static_cast<Run*>(

58 
59  G4bool transmit = (endPoint->GetStepStatus() <= fGeomBoundary);
60  if (transmit) { run->CountProcesses(procName); }
61  else {
62  //count real processes and sum track length

71

76  G4double energy = endPoint->GetKineticEnergy();
77  analysisManager->FillH1(id,energy);
78

80  G4ThreeVector direction = endPoint->GetMomentumDirection();
81  G4double costeta = direction.x();
82  analysisManager->FillH1(id,costeta);
83  }
84 
85  //secondaries
86  //
87  const std::vector<const G4Track*>* secondary
88  = aStep->GetSecondaryInCurrentStep();
89  for (size_t lp=0; lp<(*secondary).size(); lp++) {
90  G4double charge = (*secondary)[lp]->GetDefinition()->GetPDGCharge();
91  if (charge != 0.) { id = 3; } else { id = 5; }
92  G4double energy = (*secondary)[lp]->GetKineticEnergy();
93  analysisManager->FillH1(id,energy);
94 
95  ++id;
96  G4ThreeVector direction = (*secondary)[lp]->GetMomentumDirection();
97  G4double costeta = direction.x();
98  analysisManager->FillH1(id,costeta);
99 
100  //energy tranferred to charged secondaries
101  if (charge != 0.) { run->SumeTransf(energy); }
102  }
103 
104  // kill event after first interaction
105  //
106  G4RunManager::GetRunManager()->AbortEvent();
107 }
