100 G4VParticleChange*
101 G4OpWLS::PostStepDoIt(const G4Track& aTrack, const G4Step& aStep)
102 {
103  aParticleChange.Initialize(aTrack);
104 
105  aParticleChange.ProposeTrackStatus(fStopAndKill);
106 
107  if (verboseLevel>0) {
108  G4cout << "\n** Photon absorbed! **" << G4endl;
109  }
110 
111  const G4Material* aMaterial = aTrack.GetMaterial();
112 
113  G4StepPoint* pPostStepPoint = aStep.GetPostStepPoint();
114 
115  G4MaterialPropertiesTable* aMaterialPropertiesTable =
116  aMaterial->GetMaterialPropertiesTable();
117  if (!aMaterialPropertiesTable)
118  return G4VDiscreteProcess::PostStepDoIt(aTrack, aStep);
119 
120  const G4MaterialPropertyVector* WLS_Intensity =
121  aMaterialPropertiesTable->GetProperty(kWLSCOMPONENT);
122 
123  if (!WLS_Intensity)
124  return G4VDiscreteProcess::PostStepDoIt(aTrack, aStep);
125 
126  G4int NumPhotons = 1;
127 
128  if (aMaterialPropertiesTable->ConstPropertyExists("WLSMEANNUMBERPHOTONS")) {
129 
130  G4double MeanNumberOfPhotons = aMaterialPropertiesTable->
131  GetConstProperty(kWLSMEANNUMBERPHOTONS);
132 
133  NumPhotons = G4int(G4Poisson(MeanNumberOfPhotons));
134 
135  if (NumPhotons <= 0) {
136 
137  // return unchanged particle and no secondaries
138 
139  aParticleChange.SetNumberOfSecondaries(0);
140 
141  return G4VDiscreteProcess::PostStepDoIt(aTrack, aStep);
142 
143  }
144 
145  }
146 
147  aParticleChange.SetNumberOfSecondaries(NumPhotons);
148 
149  G4double primaryEnergy = aTrack.GetDynamicParticle()->GetKineticEnergy();
150 
151  G4int materialIndex = aMaterial->GetIndex();
152 
153  // Retrieve the WLS Integral for this material
154  // new G4PhysicsOrderedFreeVector allocated to hold CII's
155 
156  G4double WLSTime = 0.*ns;
157  G4PhysicsOrderedFreeVector* WLSIntegral = 0;
158 
159  WLSTime = aMaterialPropertiesTable->
160  GetConstProperty(kWLSTIMECONSTANT);
161  WLSIntegral =
162  (G4PhysicsOrderedFreeVector*)((*theIntegralTable)(materialIndex));
163 
164  // Max WLS Integral
165 
166  G4double CIImax = WLSIntegral->GetMaxValue();
167 
168  G4int NumberOfPhotons = NumPhotons;
169 
170  for (G4int i = 0; i < NumPhotons; i++) {
171 
172  G4double sampledEnergy;
173 
174  // Make sure the energy of the secondary is less than that of the primary
175 
176  for (G4int j = 1; j <= 100; j++) {
177 
178  // Determine photon energy
179 
180  G4double CIIvalue = G4UniformRand()*CIImax;
181  sampledEnergy = WLSIntegral->GetEnergy(CIIvalue);
182 
183  if (verboseLevel>1) {
184  G4cout << "sampledEnergy = " << sampledEnergy << G4endl;
185  G4cout << "CIIvalue = " << CIIvalue << G4endl;
186  }
187 
188  if (sampledEnergy <= primaryEnergy) break;
189  }
190 
191  // If no such energy can be sampled, return one less secondary, or none
192 
193  if (sampledEnergy > primaryEnergy) {
194  if (verboseLevel>1)
195  G4cout << " *** One less WLS photon will be returned ***" << G4endl;
196  NumberOfPhotons--;
197  aParticleChange.SetNumberOfSecondaries(NumberOfPhotons);
198  if (NumberOfPhotons == 0) {
199  if (verboseLevel>1)
200  G4cout << " *** No WLS photon can be sampled for this primary ***"
201  << G4endl;
202  // return unchanged particle and no secondaries
203  return G4VDiscreteProcess::PostStepDoIt(aTrack, aStep);
204  }
205  continue;
206  }
207 
208  // Generate random photon direction
209 
210  G4double cost = 1. - 2.*G4UniformRand();
211  G4double sint = std::sqrt((1.-cost)*(1.+cost));
212 
213  G4double phi = twopi*G4UniformRand();
214  G4double sinp = std::sin(phi);
215  G4double cosp = std::cos(phi);
216 
217  G4double px = sint*cosp;
218  G4double py = sint*sinp;
219  G4double pz = cost;
220 
221  // Create photon momentum direction vector
222 
223  G4ParticleMomentum photonMomentum(px, py, pz);
224 
225  // Determine polarization of new photon
226 
227  G4double sx = cost*cosp;
228  G4double sy = cost*sinp;
229  G4double sz = -sint;
230 
231  G4ThreeVector photonPolarization(sx, sy, sz);
232 
233  G4ThreeVector perp = photonMomentum.cross(photonPolarization);
234 
235  phi = twopi*G4UniformRand();
236  sinp = std::sin(phi);
237  cosp = std::cos(phi);
238 
239  photonPolarization = cosp * photonPolarization + sinp * perp;
240 
241  photonPolarization = photonPolarization.unit();
242 
243  // Generate a new photon:
244 
245  G4DynamicParticle* aWLSPhoton =
246  new G4DynamicParticle(G4OpticalPhoton::OpticalPhoton(),
247  photonMomentum);
248  aWLSPhoton->SetPolarization
249  (photonPolarization.x(),
250  photonPolarization.y(),
251  photonPolarization.z());
252 
253  aWLSPhoton->SetKineticEnergy(sampledEnergy);
254 
255  // Generate new G4Track object:
256 
257  // Must give position of WLS optical photon
258 
259  G4double TimeDelay = WLSTimeGeneratorProfile->GenerateTime(WLSTime);
260  G4double aSecondaryTime = (pPostStepPoint->GetGlobalTime()) + TimeDelay;
261 
262  G4ThreeVector aSecondaryPosition = pPostStepPoint->GetPosition();
263 
264  G4Track* aSecondaryTrack =
265  new G4Track(aWLSPhoton,aSecondaryTime,aSecondaryPosition);
266 
267  aSecondaryTrack->SetTouchableHandle(aTrack.GetTouchableHandle());
268  // aSecondaryTrack->SetTouchableHandle((G4VTouchable*)0);
269 
270  aSecondaryTrack->SetParentID(aTrack.GetTrackID());
271 
272  aParticleChange.AddSecondary(aSecondaryTrack);
273  }
274 
275  if (verboseLevel>0) {
276  G4cout << "\n Exiting from G4OpWLS::DoIt -- NumberOfSecondaries = "
277  << aParticleChange.GetNumberOfSecondaries() << G4endl;
278  }
279 
280  return G4VDiscreteProcess::PostStepDoIt(aTrack, aStep);
281 }

286 void G4OpWLS::BuildPhysicsTable(const G4ParticleDefinition&)
287 {
288  if (theIntegralTable) {
289  theIntegralTable->clearAndDestroy();
290  delete theIntegralTable;
291  theIntegralTable = NULL;
292  }
293 
294  const G4MaterialTable* theMaterialTable =
295  G4Material::GetMaterialTable();
296  G4int numOfMaterials = G4Material::GetNumberOfMaterials();
297 
298  // create new physics table
299 
300  theIntegralTable = new G4PhysicsTable(numOfMaterials);
301 
302  // loop for materials
303 
304  for (G4int i=0 ; i < numOfMaterials; i++)
305  {
306  G4PhysicsOrderedFreeVector* aPhysicsOrderedFreeVector =
307  new G4PhysicsOrderedFreeVector();
308 
309  // Retrieve vector of WLS wavelength intensity for
310  // the material from the material's optical properties table.
311 
312  G4Material* aMaterial = (*theMaterialTable)[i];
313 
314  G4MaterialPropertiesTable* aMaterialPropertiesTable =
315  aMaterial->GetMaterialPropertiesTable();
316 
317  if (aMaterialPropertiesTable) {
318 
319  G4MaterialPropertyVector* theWLSVector =
320  aMaterialPropertiesTable->GetProperty(kWLSCOMPONENT);
321 
322  if (theWLSVector) {
323 
324  // Retrieve the first intensity point in vector
325  // of (photon energy, intensity) pairs
326 
327  G4double currentIN = (*theWLSVector)[0];
328 
329  if (currentIN >= 0.0) {
330 
331  // Create first (photon energy)
332 
333  G4double currentPM = theWLSVector->Energy(0);
334 
335  G4double currentCII = 0.0;
336 
337  aPhysicsOrderedFreeVector->
338  InsertValues(currentPM , currentCII);
339 
340  // Set previous values to current ones prior to loop
341 
342  G4double prevPM = currentPM;
343  G4double prevCII = currentCII;
344  G4double prevIN = currentIN;
345 
346  // loop over all (photon energy, intensity)
347  // pairs stored for this material
348 
349  for (size_t j = 1;
350  j < theWLSVector->GetVectorLength();
351  j++)
352  {
353  currentPM = theWLSVector->Energy(j);
354  currentIN = (*theWLSVector)[j];
355 
356  currentCII = 0.5 * (prevIN + currentIN);
357 
358  currentCII = prevCII +
359  (currentPM - prevPM) * currentCII;
360 
361  aPhysicsOrderedFreeVector->
362  InsertValues(currentPM, currentCII);
363 
364  prevPM = currentPM;
365  prevCII = currentCII;
366  prevIN = currentIN;
367  }
368  }
369  }
370  }
371  // The WLS integral for a given material
372  // will be inserted in the table according to the
373  // position of the material in the material table.
374 
375  theIntegralTable->insertAt(i,aPhysicsOrderedFreeVector);
376  }
377 }
