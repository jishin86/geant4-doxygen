67 void PrimaryGeneratorAction::GeneratePrimaries(G4Event* anEvent)
68 {
69  // This function is called at the begining of event
70 
71  // In order to avoid dependence of PrimaryGeneratorAction
72  // on DetectorConstruction class we get world volume
73  // from G4LogicalVolumeStore
74  //
75  G4double worldZHalfLength = 0.;
76  auto worldLV = G4LogicalVolumeStore::GetInstance()->GetVolume("World");
77 
78  // Check that the world volume has box shape
79  G4Box* worldBox = nullptr;
80  if ( worldLV ) {
81  worldBox = dynamic_cast<G4Box*>(worldLV->GetSolid());
82  }
83 
84  if ( worldBox ) {
85  worldZHalfLength = worldBox->GetZHalfLength();
86  }
87  else {
88  G4ExceptionDescription msg;
89  msg << "World volume of box shape not found." << G4endl;
90  msg << "Perhaps you have changed geometry." << G4endl;
91  msg << "The gun will be place in the center.";
92  G4Exception("PrimaryGeneratorAction::GeneratePrimaries()",
93  "MyCode0002", JustWarning, msg);
94  }
95 
96  // Set gun position
97  fParticleGun
98  ->SetParticlePosition(G4ThreeVector(0., 0., -worldZHalfLength));
99 
100  fParticleGun->GeneratePrimaryVertex(anEvent);
101 }
