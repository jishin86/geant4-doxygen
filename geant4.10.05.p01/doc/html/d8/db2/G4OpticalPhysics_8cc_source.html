112 void G4OpticalPhysics::PrintStatistics() const
113 {
114 // Print all processes activation and their parameters
115 
116  for ( G4int i=0; i<kNoProcess; i++ ) {
117  G4cout << " " << G4OpticalProcessName(i) << " process: ";
118  if ( ! fProcessUse[i] ) {
119  G4cout << "not used" << G4endl;
120  }
121  else {
122  G4cout << "used" << G4endl;
123  if ( i == kCerenkov ) {
124  G4cout << " Max number of photons per step: " << fMaxNumPhotons << G4endl;
125  G4cout << " Max beta change per step: " << fMaxBetaChange << G4endl;
126  if ( fProcessTrackSecondariesFirst[kCerenkov] ) {
127  G4cout << " Track secondaries first: activated" << G4endl;
128  }
129  else {
130  G4cout << " Track secondaries first: inactivated" << G4endl;
131  }
132  }
133  if ( i == kScintillation ) {
134  if (fScintillationByParticleType)
135  G4cout << " Scintillation by Particle Type: activated " << G4endl;
136  G4cout << " Yield factor: " << fYieldFactor << G4endl;
137  G4cout << " ExcitationRatio: " << fExcitationRatio << G4endl;
138  if ( fProcessTrackSecondariesFirst[kScintillation] ) {
139  G4cout << " Track secondaries first: activated" << G4endl;
140  }
141  else {
142  G4cout << " Track secondaries first: inactivated" << G4endl;
143  }
144  }
145  if ( i == kWLS ) {
146  G4cout << " WLS process time profile: " << fWLSTimeProfileName << G4endl;
147  }
148  }
149  }
150 }

161 void G4OpticalPhysics::ConstructProcess()
162 {
163  if(verboseLevel>0)
164  G4cout <<"G4OpticalPhysics:: Add Optical Physics Processes"<< G4endl;
165 
166  // A vector of optical processes
167  std::vector<G4VProcess*> OpProcesses;
168 
169  for ( G4int i=0; i<kNoProcess; i++ ) OpProcesses.push_back(nullptr);
170 
171  // Add Optical Processes
172 
173  fAbsorptionProcess = new G4OpAbsorption();
174  OpProcesses[kAbsorption] = fAbsorptionProcess;
175 
176  fRayleighProcess = new G4OpRayleigh();
177  OpProcesses[kRayleigh] = fRayleighProcess;
178 
179  fMieProcess = new G4OpMieHG();
180  OpProcesses[kMieHG] = fMieProcess;
181 
182  fBoundaryProcess = new G4OpBoundaryProcess();
183  fBoundaryProcess->SetInvokeSD(fInvokeSD);
184  OpProcesses[kBoundary] = fBoundaryProcess;
185 
186  fWLSProcess = new G4OpWLS();
187  fWLSProcess->UseTimeProfile(fWLSTimeProfileName);
188  OpProcesses[kWLS] = fWLSProcess;
189 
190  G4ProcessManager* pManager = nullptr;
191  pManager = G4OpticalPhoton::OpticalPhoton()->GetProcessManager();
192 
193  if (!pManager) {
194  G4ExceptionDescription ed;
195  ed << "Optical Photon without a Process Manager";
196  G4Exception("G4OpticalPhysics::ConstructProcess()","",
197  FatalException,ed);
198  return;
199  }
200 
201  for ( G4int i=kAbsorption; i<=kWLS; i++ ) {
202  if ( fProcessUse[i] ) {
203  pManager->AddDiscreteProcess(OpProcesses[i]);
204  }
205  }
206 
207  fScintillationProcess = new G4Scintillation();
208  fScintillationProcess->SetScintillationYieldFactor(fYieldFactor);
209  fScintillationProcess->SetScintillationExcitationRatio(fExcitationRatio);
210  fScintillationProcess->SetFiniteRiseTime(fFiniteRiseTime);
211  fScintillationProcess->SetScintillationByParticleType(fScintillationByParticleType);
212  fScintillationProcess->SetScintillationTrackInfo(fScintillationTrackInfo);
213  fScintillationProcess->SetTrackSecondariesFirst(fProcessTrackSecondariesFirst[kScintillation]);
214  fScintillationProcess->SetStackPhotons(fScintillationStackPhotons);
215  G4EmSaturation* emSaturation = G4LossTableManager::Instance()->EmSaturation();
216  fScintillationProcess->AddSaturation(emSaturation);
217  OpProcesses[kScintillation] = fScintillationProcess;
218 
219  fCerenkovProcess = new G4Cerenkov();
220  fCerenkovProcess->SetMaxNumPhotonsPerStep(fMaxNumPhotons);
221  fCerenkovProcess->SetMaxBetaChangePerStep(fMaxBetaChange);
222  fCerenkovProcess->SetTrackSecondariesFirst(fProcessTrackSecondariesFirst[kCerenkov]);
223  fCerenkovProcess->SetStackPhotons(fCerenkovStackPhotons);
224  OpProcesses[kCerenkov] = fCerenkovProcess;
225 
226  auto myParticleIterator=GetParticleIterator();
227  myParticleIterator->reset();
228 
229  while( (*myParticleIterator)() ){
230 
231  G4ParticleDefinition* particle = myParticleIterator->value();
232  G4String particleName = particle->GetParticleName();
233 
234  pManager = particle->GetProcessManager();
235  if (!pManager) {
236  G4ExceptionDescription ed;
237  ed << "Particle " << particleName << "without a Process Manager";
238  G4Exception("G4OpticalPhysics::ConstructProcess()","",
239  FatalException, ed);
240  return; // else coverity complains for pManager use below
241  }
242 
243  if( fCerenkovProcess->IsApplicable(*particle) &&
244  fProcessUse[kCerenkov] ) {
245  pManager->AddProcess(fCerenkovProcess);
246  pManager->SetProcessOrdering(fCerenkovProcess,idxPostStep);
247  }
248  if( fScintillationProcess->IsApplicable(*particle) &&
249  fProcessUse[kScintillation] ) {
250  pManager->AddProcess(fScintillationProcess);
251  pManager->SetProcessOrderingToLast(fScintillationProcess,idxAtRest);
252  pManager->SetProcessOrderingToLast(fScintillationProcess,idxPostStep);
253  }
254  }
255 
256  // Add verbose
257  for ( G4int i=0; i<kNoProcess; i++ ) {
258  if ( fProcessUse[i] ) OpProcesses[i]->SetVerboseLevel(verboseLevel);
259  }
260 
261  if (verboseLevel > 1) PrintStatistics();
262  if (verboseLevel > 0)
263  G4cout << "### " << namePhysics << " physics constructed." << G4endl;
264 }

440 void G4OpticalPhysics::Configure(G4OpticalProcessIndex index, G4bool isUse)
441 {
442  // Configure the physics constructor to use/not use a selected process.
443  // This method can only be called in PreInit> phase (before execution of
444  // ConstructProcess). The process is not added to particle's process manager
445  // and so it cannot be re-activated later in Idle> phase with the command
446  // /process/activate.
447 
448  if ( index >= kNoProcess ) return;
449  if ( fProcessUse[index] == isUse ) return;
450  fProcessUse[index] = isUse;
451 }
