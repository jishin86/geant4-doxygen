81 void PrimaryGeneratorAction::GeneratePrimaries(G4Event* anEvent)
82 {
83  //this function is called at the begining of event
84  //
85  G4double Zvertex;
86  if (G4UniformRand() < fMassWallRatio) {
87  Zvertex = 0.5*fCavityThickness + G4UniformRand()*fEffWallThick;
88  if (G4UniformRand() < 0.5) Zvertex = -Zvertex;
89  fNwall++;
90  } else {
91  Zvertex = (G4UniformRand() - 0.5)*fCavityThickness;
92  fNcavity++;
93  }
94 
95  fParticleGun->SetParticlePosition(G4ThreeVector(0., 0., Zvertex));
96  fParticleGun->GeneratePrimaryVertex(anEvent);
97 
98  //histograms
99  //
100  G4AnalysisManager* analysisManager = G4AnalysisManager::Instance();
101  analysisManager->FillH1(1,Zvertex);
102  analysisManager->FillH1(2,fParticleGun->GetParticleEnergy());
103 }
