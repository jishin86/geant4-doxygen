47 void ExN04SteppingAction::UserSteppingAction(const G4Step * theStep)
48 {
49  G4Track * theTrack = theStep->GetTrack();
50 
51  // check if it is alive
52  if(theTrack->GetTrackStatus()!=fAlive) { return; }
53 
54  // check if it is primary
55  if(theTrack->GetParentID()!=0) { return; }
56 
57  // check if it is NOT muon
58  G4ParticleDefinition * particleType = theTrack->GetDefinition();
59  if((particleType==G4MuonPlus::MuonPlusDefinition())
60  ||(particleType==G4MuonMinus::MuonMinusDefinition()))
61  { return; }
62 
63  // check if it is entering to the calorimeter volume
64  G4StepPoint * thePrePoint = theStep->GetPreStepPoint();
65  G4VPhysicalVolume * thePrePV = thePrePoint->GetPhysicalVolume();
66  G4String thePrePVname = thePrePV->GetName();
67  if(thePrePVname(0,4)=="calo") { return; }
68  G4StepPoint * thePostPoint = theStep->GetPostStepPoint();
69  G4VPhysicalVolume * thePostPV = thePostPoint->GetPhysicalVolume();
70  G4String thePostPVname = thePostPV->GetName();
71  if(thePostPVname(0,4)!="calo") { return; }
72 
73  // then suspend the track
74  theTrack->SetTrackStatus(fSuspend);
75 }
