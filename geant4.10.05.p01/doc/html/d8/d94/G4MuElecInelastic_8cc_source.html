83 void G4MuElecInelastic::InitialiseProcess(const G4ParticleDefinition* p)
84 {
85  if(!isInitialised)
86  {
87  isInitialised = true;
88  SetBuildTableFlag(false);
89  G4String name = p->GetParticleName();
90 
91  if(name == "e-")
92  {
93  if(!EmModel()) SetEmModel(new G4MuElecInelasticModel);
94  EmModel()->SetLowEnergyLimit(16.7*eV);
95  EmModel()->SetHighEnergyLimit(100*MeV);
96 
97  AddEmModel(1, EmModel());
98  }
99 
100  else if(name == "proton")
101  {
102  if(!EmModel()) SetEmModel(new G4MuElecInelasticModel);
103  EmModel()->SetLowEnergyLimit(50.*keV);
104  EmModel()->SetHighEnergyLimit(10*GeV);
105 
106  AddEmModel(1, EmModel());
107  }
108 
109  else if(name == "GenericIon")
110  {
111  if(!EmModel()) SetEmModel(new G4MuElecInelasticModel);
112  EmModel()->SetLowEnergyLimit(50.*keV);
113  EmModel()->SetHighEnergyLimit(10000.*GeV);
114 
115  AddEmModel(1, EmModel());
116  }
117  }
118 }
