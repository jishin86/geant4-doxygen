128 void G4GDMLMessenger::SetNewValue(G4UIcommand* command, G4String newValue)
129 {
130  if( command == StripCmd )
131  {
132  G4bool mode = StripCmd->GetNewBoolValue(newValue);
133  myParser->SetStripFlag(mode);
134  }
135 
136  if( command == AppendCmd )
137  {
138  pFlag = AppendCmd->GetNewBoolValue(newValue);
139  myParser->SetAddPointerToName(pFlag);
140  }
141 
142  if( command == ReaderCmd )
143  {
144  G4GeometryManager::GetInstance()->OpenGeometry();
145  myParser->Read(newValue);
146  G4RunManager::GetRunManager()->DefineWorldVolume(myParser->GetWorldVolume());
147  }
148 
149  if( command == RegionCmd )
150  {
151  G4bool mode = RegionCmd->GetNewBoolValue(newValue);
152  myParser->SetRegionExport(mode);
153  }
154 
155  if( command == EcutsCmd )
156  {
157  G4bool mode = EcutsCmd->GetNewBoolValue(newValue);
158  myParser->SetEnergyCutsExport(mode);
159  }
160 
161  if( command == SDCmd )
162  {
163  G4bool mode = SDCmd->GetNewBoolValue(newValue);
164  myParser->SetSDExport(mode);
165  }
166 
167  if( command == TopVolCmd )
168  {
169  topvol = G4LogicalVolumeStore::GetInstance()->GetVolume(newValue);
170  }
171 
172  if( command == WriterCmd )
173  {
174  myParser->Write(newValue, topvol, pFlag);
175  }
176 
177  if( command == ClearCmd )
178  {
179  myParser->Clear();
180  G4RunManager::GetRunManager()->ReinitializeGeometry(true);
181  }
182 }
