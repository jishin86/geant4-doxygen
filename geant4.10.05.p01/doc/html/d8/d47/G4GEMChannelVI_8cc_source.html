105 G4double G4GEMChannelVI::GetEmissionProbability(G4Fragment* fragment)
106 {
107  fragZ = fragment->GetZ_asInt();
108  fragA = fragment->GetA_asInt();
109  resZ = fragZ - Z;
110  resA = fragA - A;
111  G4double prob = 0.0;
112  if(resA < A || resA < resZ || resZ < 0 || (resA == A && resZ < Z)) {
113  return prob;
114  }
115 
116  resMassGround = G4NucleiProperties::GetNuclearMass(resA, resZ);
117  G4double exc = fragment->GetExcitationEnergy();
118  massFrag = fragment->GetGroundStateMass() + exc;
119  delta0 = pairingCorrection->GetPairingCorrection(fragA, fragZ);
120  eCBarrier = cBarrier->GetCoulombBarrier(resA, resZ, exc);
121 
122  maxExc = massFrag - massGround - resMassGround - eCBarrier - delta0;
123  if(maxExc < 0.0) { return prob; }
124 
125  resZ13 = fG4pow->Z13(resZ);
126  resA13 = fG4pow->Z13(resA);
127  delta1 = pairingCorrection->GetPairingCorrection(resA, resZ);
128 
129  G4double C = 0.0;
130  if(resA >= 50) {
131  C = -0.10/G4double(A);
132  } else if(resZ > 20) {
133  C = (0.123482-0.00534691*Z-0.0000610624*(Z*Z)+5.93719*1e-7*(Z*Z*Z)+
134  1.95687*1e-8*(Z*Z*Z*Z))/G4double(A);
135  }
136  if(0 == Z) {
137  alphaP = 0.76+1.93/resA13;
138  betaP = (1.66/(resA13*resA13)-0.05)*CLHEP::MeV/alphaP;
139  } else {
140  alphaP = 1.0 + C;
141  betaP = - eCBarrier;
142  }
143 
144  maxProb = 0.0;
145  G4double e0 = maxExc*0.5;
146 
147  // e is an excitation of emitted fragment
148  for (G4int i=0; i<NPOINTSGEM; ++i) {
149  prob += ws[i]*IntegratedProbability(e0*(xs[i] + 1.0));
150  }
151  prob *= coeff*e0*e0;
152 
153  /*
154  G4cout << "G4GEMChannelVI: Z= " << Z << " A= " << A
155  << " FragmentZ= " << aZ << " FragmentA= " << anA
156  << " Zres= " << ResidualZ << " Ares= " << ResidualA
157  << G4endl;
158  */
159 
160  //G4cout << "Prob= " << prob << G4endl;
161  return prob;
162 }

164 G4double G4GEMChannelVI::IntegratedProbability(G4double exc)
165 {
166  G4double e0 = (maxExc - exc)*0.5;
167 
168  G4double y;
169  G4double sum = 0.0;
170 
171  for (G4int i=0; i<NPOINTSGEM; ++i) {
172  y = ProbabilityDistributionFunction(exc, e0*(xs[i] + 1.0));
173  maxProb = std::max(maxProb, y);
174  sum += ws[i]*y;
175  }
176  return sum;
177 }

179 G4Fragment* G4GEMChannelVI::EmittedFragment(G4Fragment* theNucleus)
180 {
181  G4double exc;
182  G4double resExc;
183 
184  static G4double factor = 1.2;
185  maxProb *= factor;
186 
187  CLHEP::HepRandomEngine* rndm = G4Random::getTheEngine();
188  for(G4int i=0; i<100; ++i) {
189  do {
190  exc = maxExc*rndm->flat();
191  resExc = maxExc*rndm->flat();
192  } while (exc + resExc > maxExc);
193 
194  G4double prob = ProbabilityDistributionFunction(exc, resExc);
195  if(prob > maxProb && nWarn < 10) {
196  ++nWarn;
197  G4cout << "### G4GEMChannelVI::EmittedFragment WARNING: majoranta "
198  << maxProb << " is exceeded " << prob << "\n"
199  << " fragZ= " << fragZ << " fragA= " << fragA
200  << " Z= " << Z << " A= " << A
201  << " resZ= " << resZ << " resA= " << resA << "\n"
202  << " exc(MeV)= " << exc << " resExc(MeV)= " << resExc
203  << " maxExc(MeV)= " << maxExc << G4endl;
204  }
205  if(maxProb*rndm->flat() <= prob) { break; }
206  }
207  if(exc <= maxLevelE) {
208  exc = FindLevel(levelManager, exc, maxExc - resExc);
209  }
210  if(resA >= nData->GetMinA(resZ) && resA <= nData->GetMaxA(resZ)
211  && resExc < nData->GetMaxLevelEnergy(Z, A)) {
212  const G4LevelManager* lman = nData->GetLevelManager(Z, A);
213  if(lman) { resExc = FindLevel(lman, resExc, maxExc - exc); }
214  }
215 
216  G4LorentzVector lv0 = theNucleus->GetMomentum();
217  G4double mass1 = massGround + exc;
218  G4double mass2 = resMassGround + resExc;
219 
220  G4double e1 = 0.5*((massFrag - mass2)*(massFrag + mass2)
221  + mass1*mass1)/massFrag;
222 
223  G4double p1(0.0);
224  if(e1 > mass1) {
225  p1 = std::sqrt((e1 - mass1)*(e1 + mass1));
226  } else {
227  e1 = mass1;
228  }
229  G4ThreeVector v = G4RandomDirection();
230  G4LorentzVector lv1(p1*v, e1);
231 
232  G4ThreeVector boostVector = lv0.boostVector();
233  lv1.boost(boostVector);
234 
235  G4Fragment* frag = new G4Fragment(A, Z, lv1);
236 
237  G4double e2 = massFrag - e1;
238  if(e2 < mass2) {
239  e2 = mass2;
240  p1 = 0.0;
241  }
242  lv0.set(-v*p1, e2);
243  lv0.boost(boostVector);
244 
245  theNucleus->SetZandA_asInt(resZ, resA);
246  theNucleus->SetMomentum(lv0);
247 
248  return frag;
249 }

251 G4double
252 G4GEMChannelVI::ProbabilityDistributionFunction(G4double exc, G4double resExc)
253 {
254  G4double Ux = (2.5 + 150.0/G4double(resA))*CLHEP::MeV;
255  G4double Ex = Ux + delta1;
256  G4double T = 1.0/(std::sqrt(levelDensity/Ux) - 1.5/Ux);
257  G4double E0 = Ex - T*(G4Log(T) - G4Log(levelDensity)*0.25
258  - 1.25*G4Log(Ux) + 2.0*std::sqrt(levelDensity*Ux));
259 
260  G4double UxCN = (2.5 + 150.0/G4double(A))*CLHEP::MeV;
261  G4double ExCN = UxCN + delta0;
262  G4double TCN = 1.0/(std::sqrt(levelDensity/UxCN) - 1.5/UxCN);
263 
264  G4double mass1 = massGround + exc;
265  G4double mass2 = resMassGround + resExc;
266 
267  maxKinEnergy = 0.5*((massFrag - mass2)*(massFrag + mass2)
268  + mass1*mass1)/massFrag - mass1;
269  maxKinEnergy = std::max(maxKinEnergy, 0.0);
270 
271  G4double Width = 0.0;
272  G4double t = maxKinEnergy/T;
273  if ( maxKinEnergy < Ex ) {
274  Width = (I1(t,t)*T + (betaP+eCBarrier)*I0(t))/G4Exp(E0/T);
275 
276  } else {
277 
278  G4double tx = Ex/T;
279  G4double s0 = 2.0*std::sqrt(levelDensity*(maxKinEnergy-delta0));
280  G4double sx = 2.0*std::sqrt(levelDensity*(Ex-delta0));
281 
282  // VI: protection against FPE exception
283  if(s0 > 350.) { s0 = 350.; }
284 
285  G4double expE0T = G4Exp(E0/T);
286  G4double exps0 = G4Exp(s0);
287  static const G4double sqrt2 = std::sqrt(2.0);
288 
289  Width = I1(t,tx)*T/expE0T + I3(s0,sx)*exps0/(sqrt2*levelDensity);
290 
291  if (0 == Z) {
292  Width += (betaP+eCBarrier)*(I0(tx)/expE0T + 2.0*sqrt2*I2(s0,sx)*exps0);
293  }
294  }
295  Width *= alphaP*massFrag;
296 
297 
298  //JMQ 190709 fix on Rb and geometrical cross sections according to
299  // Furihata's paper (JAERI-Data/Code 2001-105, p6)
300  G4double Rb = 0.0;
301  if (A > 4) {
302  Rb = 1.12*(resA13 + A13) - 0.86*((resA13 + A13)/(resA13*A13))+2.85;
303  } else if (A > 1) {
304  Rb=1.5*(resA13 + A13);
305  } else {
306  Rb = 1.5*resA13;
307  }
308 
309  G4double ild;
310  if (exc < ExCN ) {
311  G4double E0CN = ExCN - TCN*(G4Log(TCN) - 0.25*G4Log(levelDensity)
312  - 1.25*G4Log(UxCN)
313  + 2.0*std::sqrt(levelDensity*UxCN));
314  ild = G4Exp((exc-E0CN)/TCN)/TCN;
315  } else {
316  G4double x = exc - delta0;
317  G4double x1 = std::sqrt(levelDensity*x);
318  ild = G4Exp(2*x1)/(x*std::sqrt(x1));
319  }
320 
321  Width *= (Rb*Rb/ild);
322  return Width;
323 }

325 G4double G4GEMChannelVI::FindLevel(const G4LevelManager* man,
326  G4double exc, G4double exclim)
327 {
328  size_t idx = man->NearestLowEdgeLevelIndex(exc);
329  size_t idxm = man->NumberOfTransitions();
330  G4double e1 = man->LevelEnergy(idx);
331  if(idx + 1 < idxm) {
332  G4double e2 = man->LevelEnergy(idx+1);
333  if(e2 <= exclim) {
334  G4int s1 = man->SpinTwo(idx)+1;
335  G4int s2 = man->SpinTwo(idx+1)+1;
336  G4double pr = (G4double)s1/(G4double)(s1 + s2);
337  pr = (exc - e1 <= e2 - exc) ? 1.0 - (1.0 - pr)*2*(exc - e1)/(e2 - e1) :
338  2*pr*(e2 - exc)/(e2 - e1);
339  exc = (G4UniformRand() < pr) ? e1 : e2;
340  } else {
341  exc = e1;
342  }
343  } else { exc = e1; }
344  return exc;
345 }

347 G4double G4GEMChannelVI::I3(G4double s0, G4double sx)
348 {
349  G4double s2 = s0*s0;
350  G4double sx2 = sx*sx;
351  G4double S = 1.0/std::sqrt(s0);
352  G4double S2 = S*S;
353  G4double Sx = 1.0/std::sqrt(sx);
354  G4double Sx2 = Sx*Sx;
355 
356  G4double p1 = S *(2.0 + S2 *( 4.0 + S2 *( 13.5 + S2 *( 60.0 + S2 * 325.125 ))));
357  G4double p2 = Sx*Sx2 *((s2-sx2)
358  + Sx2 *((1.5*s2+0.5*sx2)
359  + Sx2 *((3.75*s2+0.25*sx2)
360  + Sx2 *((12.875*s2+0.625*sx2)
361  + Sx2 *((59.0625*s2+0.9375*sx2)
362  + Sx2 *(324.8*s2+3.28*sx2))))));
363  p2 *= G4Exp(sx-s0);
364  return p1-p2;
365 }
