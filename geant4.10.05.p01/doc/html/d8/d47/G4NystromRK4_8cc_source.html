65 void G4NystromRK4::Stepper(const G4double y[],
66  const G4double dydx[],
67  G4double hstep,
68  G4double yOut[],
69  G4double yError[])
70 {
71  fInitialPoint = { y[0], y[1], y[2] };
72 
73  G4double field[3];
74 
75  constexpr G4double one_sixth= 1./6.;
76  const G4double S5 = 0.5 * hstep;
77  const G4double S4 = .25 * hstep;
78  const G4double S6 = hstep * one_sixth;
79 
80  const G4double momentum2 = getValue2(y, Value3D::Momentum);
81  if (notEquals(momentum2, fMomentum2))
82  {
83  fMomentum = std::sqrt(momentum2);
84  fMomentum2 = momentum2;
85  fInverseMomentum = 1. / fMomentum;
86  fCoefficient = GetFCof() * fInverseMomentum;
87  }
88 
89  // Point 1
90  const G4double K1[3] = {
91  fInverseMomentum * dydx[3],
92  fInverseMomentum * dydx[4],
93  fInverseMomentum * dydx[5]
94  };
95 
96  // Point2
97  G4double p[4] = {
98  y[0] + S5 * (dydx[0] + S4 * K1[0]),
99  y[1] + S5 * (dydx[1] + S4 * K1[1]),
100  y[2] + S5 * (dydx[2] + S4 * K1[2]),
101  y[7]
102  };
103 
104  GetFieldValue(p, field);
105 
106  const G4double A2[3] = {
107  dydx[0] + S5 * K1[0],
108  dydx[1] + S5 * K1[1],
109  dydx[2] + S5 * K1[2]
110  };
111 
112  const G4double K2[3] = {
113  (A2[1] * field[2] - A2[2] * field[1]) * fCoefficient,
114  (A2[2] * field[0] - A2[0] * field[2]) * fCoefficient,
115  (A2[0] * field[1] - A2[1] * field[0]) * fCoefficient
116  };
117 
118  fMidPoint = { p[0], p[1], p[2] };
119 
120  // Point 3 with the same magnetic field
121  const G4double A3[3] = {
122  dydx[0] + S5 * K2[0],
123  dydx[1] + S5 * K2[1],
124  dydx[2] + S5 * K2[2]
125  };
126 
127  const G4double K3[3] = {
128  (A3[1] * field[2] - A3[2] * field[1]) * fCoefficient,
129  (A3[2] * field[0] - A3[0] * field[2]) * fCoefficient,
130  (A3[0] * field[1] - A3[1] * field[0]) * fCoefficient
131  };
132 
133  // Point 4
134  p[0] = y[0] + hstep * (dydx[0] + S5 * K3[0]);
135  p[1] = y[1] + hstep * (dydx[1] + S5 * K3[1]);
136  p[2] = y[2] + hstep * (dydx[2] + S5 * K3[2]);
137 
138  GetFieldValue(p, field);
139 
140  const G4double A4[3] = {
141  dydx[0] + hstep * K3[0],
142  dydx[1] + hstep * K3[1],
143  dydx[2] + hstep * K3[2]
144  };
145 
146  const G4double K4[3] = {
147  (A4[1] * field[2] - A4[2] * field[1]) * fCoefficient,
148  (A4[2] * field[0] - A4[0] * field[2]) * fCoefficient,
149  (A4[0] * field[1] - A4[1] * field[0]) * fCoefficient
150  };
151 
152  // New position
153  yOut[0] = y[0] + hstep * (dydx[0] + S6 * (K1[0] + K2[0] + K3[0]));
154  yOut[1] = y[1] + hstep * (dydx[1] + S6 * (K1[1] + K2[1] + K3[1]));
155  yOut[2] = y[2] + hstep * (dydx[2] + S6 * (K1[2] + K2[2] + K3[2]));
156  // New direction
157  yOut[3] = dydx[0] + S6 * (K1[0] + K4[0] + 2. * (K2[0] + K3[0]));
158  yOut[4] = dydx[1] + S6 * (K1[1] + K4[1] + 2. * (K2[1] + K3[1]));
159  yOut[5] = dydx[2] + S6 * (K1[2] + K4[2] + 2. * (K2[2] + K3[2]));
160  // Pass Energy, time unchanged -- time is not integrated !!
161  yOut[6] = y[6];
162  yOut[7] = y[7];
163 
164  fEndPoint = { yOut[0], yOut[1], yOut[2] };
165 
166  // Errors
167  yError[3] = hstep * std::fabs(K1[0] - K2[0] - K3[0] + K4[0]);
168  yError[4] = hstep * std::fabs(K1[1] - K2[1] - K3[1] + K4[1]);
169  yError[5] = hstep * std::fabs(K1[2] - K2[2] - K3[2] + K4[2]);
170  yError[0] = hstep * yError[3];
171  yError[1] = hstep * yError[4];
172  yError[2] = hstep * yError[5];
173  yError[3] *= fMomentum;
174  yError[4] *= fMomentum;
175  yError[5] *= fMomentum;
176 
177  // Normalize momentum
178  const G4double normF = fMomentum / getValue(yOut, Value3D::Momentum);
179  yOut[3] *= normF;
180  yOut[4] *= normF;
181  yOut[5] *= normF;
182 
183  // My trial code:
184  // G4double endMom2 = yOut[3]*yOut[3]+yOut[4]*yOut[4]+yOut[5]*yOut[5];
185  // G4double normF = std::sqrt( startMom2 / endMom2 );
186 }

193 void G4NystromRK4::SetDistanceForConstantField(G4double length)
194 {
