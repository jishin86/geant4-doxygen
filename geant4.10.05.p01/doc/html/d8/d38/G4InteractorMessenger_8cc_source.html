166 G4bool GetValues (
167  G4String newValue
168 ,int paramn
169 ,G4String* params
170 )
171 {
172  char* value = STRDUP(newValue.data());
173  if(value==NULL) return false;
174  char* tok = strtok(value," ");
175  for( int i=0; i<paramn;i++ ) {
176  if(tok==NULL) {
177  STRDEL(value);
178  return false;
179  }
180  G4String token = tok;
181  if( token(0)=='"' ) {
182  while( token(token.length()-1) != '"' ) {
183  tok = strtok(NULL," ");
184  if( (tok==NULL) || (*tok=='\0')) {
185  STRDEL(value);
186  return false;
187  }
188  token += " ";
189  token += tok;
190  }
191  token = (G4String)token.strip(G4String::both,'"');
192  }
193  if( token.isNull() ) {
194  STRDEL(value);
195  return false;
196  } else {
197  params[i] = token;
198  }
199  tok = strtok(NULL," ");
200  }
201  STRDEL(value);
202  return true;
203 }
