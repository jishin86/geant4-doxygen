45 PurgMagTabulatedField3D::PurgMagTabulatedField3D(const char* filename,
46  double zOffset )

49 
50  double lenUnit= meter;
51  double fieldUnit= tesla;
52  G4cout << "\n-----------------------------------------------------------"
53  << "\n Magnetic field"

55 
56  G4cout << "\n ---> " "Reading the field grid from " << filename << " ... " << endl;
57 
58  //
59  //This is a thread-local class and we have to avoid that all workers open the
60  //file at the same time
61  G4AutoLock lock(&myPurgMagTabulatedField3DLock);
62 
63  ifstream file( filename ); // Open the file for reading.
64 
65  if (!file.is_open())
66  {
67  G4ExceptionDescription ed;
68  ed << "Could not open input file " << filename << G4endl;
69  G4Exception("PurgMagTabulatedField3D::PurgMagTabulatedField3D",
70  "pugmag001",FatalException,ed);
71  }
72 
73  // Ignore first blank line
74  char buffer[256];
75  file.getline(buffer,256);
76 
77  // Read table dimensions
78  file >> nx >> ny >> nz; // Note dodgy order
79 
80  G4cout << " [ Number of values x,y,z: "
81  << nx << " " << ny << " " << nz << " ] "
82  << endl;
83 
84  // Set up storage space for table
85  xField.resize( nx );
86  yField.resize( nx );
87  zField.resize( nx );
88  int ix, iy, iz;
89  for (ix=0; ix<nx; ix++) {
90  xField[ix].resize(ny);
91  yField[ix].resize(ny);
92  zField[ix].resize(ny);
93  for (iy=0; iy<ny; iy++) {
94  xField[ix][iy].resize(nz);
95  yField[ix][iy].resize(nz);
96  zField[ix][iy].resize(nz);
97  }
98  }
99 
100  // Ignore other header information
101  // The first line whose second character is '0' is considered to
102  // be the last line of the header.
103  do {
104  file.getline(buffer,256);
105  } while ( buffer[1]!='0');
106 
107  // Read in the data
108  double xval,yval,zval,bx,by,bz;
109  double permeability; // Not used in this example.
110  for (ix=0; ix<nx; ix++) {
111  for (iy=0; iy<ny; iy++) {
112  for (iz=0; iz<nz; iz++) {
113  file >> xval >> yval >> zval >> bx >> by >> bz >> permeability;
114  if ( ix==0 && iy==0 && iz==0 ) {
115  minx = xval * lenUnit;
116  miny = yval * lenUnit;
117  minz = zval * lenUnit;
118  }
119  xField[ix][iy][iz] = bx * fieldUnit;
120  yField[ix][iy][iz] = by * fieldUnit;
121  zField[ix][iy][iz] = bz * fieldUnit;
122  }
123  }
124  }
125  file.close();
126 
127  lock.unlock();
128 
129  maxx = xval * lenUnit;
130  maxy = yval * lenUnit;
131  maxz = zval * lenUnit;
132 
133  G4cout << "\n ---> ... done reading " << endl;
134 
135  // G4cout << " Read values of field from file " << filename << endl;
136  G4cout << " ---> assumed the order: x, y, z, Bx, By, Bz "
137  << "\n ---> Min values x,y,z: "
138  << minx/cm << " " << miny/cm << " " << minz/cm << " cm "
139  << "\n ---> Max values x,y,z: "
140  << maxx/cm << " " << maxy/cm << " " << maxz/cm << " cm "
141  << "\n ---> The field will be offset by " << zOffset/cm << " cm " << endl;
142 
143  // Should really check that the limits are not the wrong way around.
144  if (maxx < minx) {swap(maxx,minx); invertX = true;}
145  if (maxy < miny) {swap(maxy,miny); invertY = true;}
146  if (maxz < minz) {swap(maxz,minz); invertZ = true;}
147  G4cout << "\nAfter reordering if neccesary"
148  << "\n ---> Min values x,y,z: "
149  << minx/cm << " " << miny/cm << " " << minz/cm << " cm "
150  << " \n ---> Max values x,y,z: "
151  << maxx/cm << " " << maxy/cm << " " << maxz/cm << " cm ";
152 
153  dx = maxx - minx;
154  dy = maxy - miny;
155  dz = maxz - minz;
156  G4cout << "\n ---> Dif values x,y,z (range): "
157  << dx/cm << " " << dy/cm << " " << dz/cm << " cm in z "
158  << "\n-----------------------------------------------------------" << endl;
159 }

161 void PurgMagTabulatedField3D::GetFieldValue(const double point[4],
162  double *Bfield ) const
163 {
164 
165  double x = point[0];
166  double y = point[1];
167  double z = point[2] + fZoffset;
168 
169  // Check that the point is within the defined region
170  if ( x>=minx && x<=maxx &&
171  y>=miny && y<=maxy &&
172  z>=minz && z<=maxz ) {
173 
174  // Position of given point within region, normalized to the range
175  // [0,1]
176  double xfraction = (x - minx) / dx;
177  double yfraction = (y - miny) / dy;
178  double zfraction = (z - minz) / dz;
179 
180  if (invertX) { xfraction = 1 - xfraction;}
181  if (invertY) { yfraction = 1 - yfraction;}
182  if (invertZ) { zfraction = 1 - zfraction;}
183 
184  // Need addresses of these to pass to modf below.
185  // modf uses its second argument as an OUTPUT argument.
186  double xdindex, ydindex, zdindex;
187 
188  // Position of the point within the cuboid defined by the
189  // nearest surrounding tabulated points
190  double xlocal = ( std::modf(xfraction*(nx-1), &xdindex));
191  double ylocal = ( std::modf(yfraction*(ny-1), &ydindex));
192  double zlocal = ( std::modf(zfraction*(nz-1), &zdindex));
193 
194  // The indices of the nearest tabulated point whose coordinates
195  // are all less than those of the given point
196  int xindex = static_cast<int>(xdindex);
197  int yindex = static_cast<int>(ydindex);
198  int zindex = static_cast<int>(zdindex);
199 
200 
201 #ifdef DEBUG_INTERPOLATING_FIELD
202  G4cout << "Local x,y,z: " << xlocal << " " << ylocal << " " << zlocal << endl;
203  G4cout << "Index x,y,z: " << xindex << " " << yindex << " " << zindex << endl;
204  double valx0z0, mulx0z0, valx1z0, mulx1z0;
205  double valx0z1, mulx0z1, valx1z1, mulx1z1;
206  valx0z0= table[xindex ][0][zindex]; mulx0z0= (1-xlocal) * (1-zlocal);
207  valx1z0= table[xindex+1][0][zindex]; mulx1z0= xlocal * (1-zlocal);
208  valx0z1= table[xindex ][0][zindex+1]; mulx0z1= (1-xlocal) * zlocal;
209  valx1z1= table[xindex+1][0][zindex+1]; mulx1z1= xlocal * zlocal;
210 #endif
211 
212  // Full 3-dimensional version
213  Bfield[0] =
214  xField[xindex ][yindex ][zindex ] * (1-xlocal) * (1-ylocal) * (1-zlocal) +
215  xField[xindex ][yindex ][zindex+1] * (1-xlocal) * (1-ylocal) * zlocal +
216  xField[xindex ][yindex+1][zindex ] * (1-xlocal) * ylocal * (1-zlocal) +
217  xField[xindex ][yindex+1][zindex+1] * (1-xlocal) * ylocal * zlocal +
218  xField[xindex+1][yindex ][zindex ] * xlocal * (1-ylocal) * (1-zlocal) +
219  xField[xindex+1][yindex ][zindex+1] * xlocal * (1-ylocal) * zlocal +
220  xField[xindex+1][yindex+1][zindex ] * xlocal * ylocal * (1-zlocal) +
221  xField[xindex+1][yindex+1][zindex+1] * xlocal * ylocal * zlocal ;
222  Bfield[1] =
223  yField[xindex ][yindex ][zindex ] * (1-xlocal) * (1-ylocal) * (1-zlocal) +
224  yField[xindex ][yindex ][zindex+1] * (1-xlocal) * (1-ylocal) * zlocal +
225  yField[xindex ][yindex+1][zindex ] * (1-xlocal) * ylocal * (1-zlocal) +
226  yField[xindex ][yindex+1][zindex+1] * (1-xlocal) * ylocal * zlocal +
227  yField[xindex+1][yindex ][zindex ] * xlocal * (1-ylocal) * (1-zlocal) +
228  yField[xindex+1][yindex ][zindex+1] * xlocal * (1-ylocal) * zlocal +
229  yField[xindex+1][yindex+1][zindex ] * xlocal * ylocal * (1-zlocal) +
230  yField[xindex+1][yindex+1][zindex+1] * xlocal * ylocal * zlocal ;
231  Bfield[2] =
232  zField[xindex ][yindex ][zindex ] * (1-xlocal) * (1-ylocal) * (1-zlocal) +
233  zField[xindex ][yindex ][zindex+1] * (1-xlocal) * (1-ylocal) * zlocal +
234  zField[xindex ][yindex+1][zindex ] * (1-xlocal) * ylocal * (1-zlocal) +
235  zField[xindex ][yindex+1][zindex+1] * (1-xlocal) * ylocal * zlocal +
236  zField[xindex+1][yindex ][zindex ] * xlocal * (1-ylocal) * (1-zlocal) +
237  zField[xindex+1][yindex ][zindex+1] * xlocal * (1-ylocal) * zlocal +
238  zField[xindex+1][yindex+1][zindex ] * xlocal * ylocal * (1-zlocal) +
239  zField[xindex+1][yindex+1][zindex+1] * xlocal * ylocal * zlocal ;
240 
241  } else {
242  Bfield[0] = 0.0;
243  Bfield[1] = 0.0;
244  Bfield[2] = 0.0;
245  }
246 }
