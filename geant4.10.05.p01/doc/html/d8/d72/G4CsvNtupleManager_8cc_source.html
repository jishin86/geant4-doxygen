61 void G4CsvNtupleManager::CreateTNtuple(
62  G4TNtupleDescription<tools::wcsv::ntuple>* ntupleDescription,
63  const G4String& /*name*/, const G4String& title)
64 {
65  // Create ntuple if the file is open (what means here that
66  // a filename was already set)
67  if ( fFileManager->GetFileName().size() ) {
68  if ( fFileManager->CreateNtupleFile(ntupleDescription) ) {
69  ntupleDescription->fNtuple
70  = new tools::wcsv::ntuple(*(ntupleDescription->fFile));
71  // ntuple object is deleted when closing a file
72  (ntupleDescription->fNtuple)->set_title(title);
73  fNtupleVector.push_back(ntupleDescription->fNtuple);
74  }
75  }
76 }

93 void G4CsvNtupleManager::FinishTNtuple(
94  G4TNtupleDescription<tools::wcsv::ntuple>* ntupleDescription,
95  G4bool /*fromBooking*/)
96 {
97  if ( ! ntupleDescription->fNtuple ) {
98  CreateTNtupleFromBooking(ntupleDescription);
99  }
100 
101  // Write header if ntuple already exists
102  if ( ! WriteHeader(ntupleDescription->fNtuple) ) {
103  G4ExceptionDescription description;
104  description << " "
105  << "Writing ntuple header has failed. ";
106  G4Exception("G4CsvNtupleManager::FinishTNtuple()",
107  "Analysis_W022", JustWarning, description);
108  }
109 }

112 G4bool G4CsvNtupleManager::WriteHeader(tools::wcsv::ntuple* ntuple) const
113 {
114 // Write header if ntuple already exists and if this option is activated.
115 // When both Hippo and Commented headers are selected, only Commented
116 // header, which reading is supported.
117 // Return false only if an error occurred.
118 
119  if ( fIsCommentedHeader ) {
120  return ntuple->write_commented_header(G4cout);
121  }
122 
123  // write hippo header (if activated and if not commented header)
124  if ( fIsHippoHeader ) {
125  ntuple->write_hippo_header();
126  return true;
127  }
128 
129  return true;
130 }
