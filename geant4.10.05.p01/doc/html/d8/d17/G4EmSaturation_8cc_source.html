80 G4double G4EmSaturation::VisibleEnergyDeposition(
81  const G4ParticleDefinition* p,
82  const G4MaterialCutsCouple* couple,
83  G4double length,
84  G4double edep,
85  G4double niel) const
86 {
87  if(edep <= 0.0) { return 0.0; }
88 
89  G4double evis = edep;
90  G4double bfactor = couple->GetMaterial()->GetIonisation()->GetBirksConstant();
91 
92  if(bfactor > 0.0) {
93 
94  // atomic relaxations for gamma incident
95  if(22 == p->GetPDGEncoding()) {
96  //G4cout << "%% gamma edep= " << edep/keV << " keV " <<manager << G4endl;
97  evis /= (1.0 + bfactor*edep/
98  G4LossTableManager::Instance()->GetRange(electron,edep,couple));
99 
100  // energy loss
101  } else {
102 
103  // protections
104  G4double nloss = std::max(niel, 0.0);
105  G4double eloss = edep - nloss;
106 
107  // neutrons and neutral hadrons
108  if(0.0 == p->GetPDGCharge() || eloss < 0.0 || length <= 0.0) {
109  nloss = edep;
110  eloss = 0.0;
111  } else {
112 
113  // continues energy loss
114  eloss /= (1.0 + bfactor*eloss/length);
115  }
116  // non-ionizing energy loss
117  if(nloss > 0.0) {
118  G4int idx = couple->GetMaterial()->GetIndex();
119  G4double escaled = nloss*massFactors[idx];
120  /*
121  G4cout << "%% p edep= " << nloss/keV << " keV Escaled= "
122  << escaled << " MeV in " << couple->GetMaterial()->GetName()
123  << " " << p->GetParticleName()
124  << G4endl;
125  */
126  G4double range = G4LossTableManager::Instance()
127  ->GetRange(proton,escaled,couple)/effCharges[idx];
128  nloss /= (1.0 + bfactor*nloss/range);
129  }
130  evis = eloss + nloss;
131  }
132  }
133  return evis;
134 }

154 G4double G4EmSaturation::FindG4BirksCoefficient(const G4Material* mat)
155 {
156  if(0 == nG4Birks) { InitialiseG4materials(); }
157 
158  G4String name = mat->GetName();

175 void G4EmSaturation::InitialiseBirksCoefficient(const G4Material* mat)
176 {
177  // electron and proton should exist in any case
178  if(!electron) {
179  electron = G4ParticleTable::GetParticleTable()->FindParticle("e-");
180  proton = G4ParticleTable::GetParticleTable()->FindParticle("proton");
181  if(!electron || !proton) {
182  G4Exception("G4EmSaturation::InitialiseBirksCoefficient", "em0001",
183  FatalException, "both electron and proton should exist");
184  }
185  }
186 
187  G4double curBirks = mat->GetIonisation()->GetBirksConstant();
188 
189  G4String name = mat->GetName();
190 
191  // material has no Birks coeffitient defined
192  // seach in the Geant4 list
193  if(curBirks == 0.0) {
194  for(G4int j=0; j<nG4Birks; ++j) {
195  if(name == g4MatNames[j]) {
196  mat->GetIonisation()->SetBirksConstant(g4MatData[j]);
197  curBirks = g4MatData[j];
198  break;
199  }
200  }
201  }
202 
203  if(curBirks == 0.0) { return; }
204 
205  // compute mean mass ratio
206  G4double curRatio = 0.0;
207  G4double curChargeSq = 0.0;
208  G4double norm = 0.0;
209  const G4ElementVector* theElementVector = mat->GetElementVector();
210  const G4double* theAtomNumDensityVector = mat->GetVecNbOfAtomsPerVolume();
211  size_t nelm = mat->GetNumberOfElements();
212  for (size_t i=0; i<nelm; ++i) {
213  const G4Element* elm = (*theElementVector)[i];
214  G4double Z = elm->GetZ();
215  G4double w = Z*Z*theAtomNumDensityVector[i];
216  curRatio += w/nist->GetAtomicMassAmu(G4int(Z));
217  curChargeSq = Z*Z*w;
218  norm += w;
219  }
220  curRatio *= proton_mass_c2/norm;
221  curChargeSq /= norm;
222 
223  // store results
224  G4int idx = mat->GetIndex();
225  massFactors[idx] = curRatio;
226  effCharges[idx] = curChargeSq;
227 }

236  const G4Material* mat = (*mtable)[i];

239  G4cout << " " << mat->GetName() << " "

243  << " effCharge= " << effCharges[i] << G4endl;
244  }
245  }

264 {
265  nG4Birks = 4;
266  g4MatData.reserve(nG4Birks);
267 
268  // M.Hirschberg et al., IEEE Trans. Nuc. Sci. 39 (1992) 511
269  // SCSN-38 kB = 0.00842 g/cm^2/MeV; rho = 1.06 g/cm^3
270  g4MatNames.push_back("G4_POLYSTYRENE");
271  g4MatData.push_back(0.07943*mm/MeV);
272 
273  // C.Fabjan (private communication)
274  // kB = 0.006 g/cm^2/MeV; rho = 7.13 g/cm^3
275  g4MatNames.push_back("G4_BGO");
276  g4MatData.push_back(0.008415*mm/MeV);
277 
278  // A.Ribon analysis of publications
279  // Scallettar et al., Phys. Rev. A25 (1982) 2419.
280  // NIM A 523 (2004) 275.
281  // kB = 0.022 g/cm^2/MeV; rho = 1.396 g/cm^3;
282  // ATLAS Efield = 10 kV/cm provide the strongest effect
283  // kB = 0.1576*mm/MeV
284  // A. Kiryunin and P.Strizenec "Geant4 hadronic
285  // working group meeting " kB = 0.041/9.13 g/cm^2/MeV
286  g4MatNames.push_back("G4_lAr");
287  g4MatData.push_back(0.032*mm/MeV);

289  //G4_BARIUM_FLUORIDE
290  //G4_CESIUM_IODIDE
291  //G4_GEL_PHOTO_EMULSION
292  //G4_PHOTO_EMULSION
293  //G4_PLASTIC_SC_VINYLTOLUENE

295  //G4_STILBENE
296  //G4_lAr
297 
298  //G4_PbWO4 - CMS value
299  g4MatNames.push_back("G4_PbWO4");

302  //G4_Lucite

304 }
