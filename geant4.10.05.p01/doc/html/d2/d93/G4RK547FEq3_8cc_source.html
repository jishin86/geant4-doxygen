51 void G4RK547FEq3::makeStep(
52  const G4double yInput[],
53  const G4double dydx[],
54  const G4double hstep,
55  G4double yOutput[],
56  G4double* dydxOutput,
57  G4double* yError) const
58 {
59  G4double yTemp[G4FieldTrack::ncompSVEC];
60  for (int i = GetNumberOfVariables(); i < GetNumberOfStateVariables(); ++i){
61  yOutput[i] = yTemp[i] = yInput[i];
62  }
63 
64  G4double ak2[G4FieldTrack::ncompSVEC],
65  ak3[G4FieldTrack::ncompSVEC],
66  ak4[G4FieldTrack::ncompSVEC],
67  ak5[G4FieldTrack::ncompSVEC],
68  ak6[G4FieldTrack::ncompSVEC];
69 
70  const G4double
71  b21 = 11./45.,
72  b31 = 11./120., b32 = 11./40.,
73  b41 = 106865./87808., b42 = -408375./87808., b43 = 193875./43904.,
74  b51 = 79503./121000., b52 = -1053./440., b53 = 147753./56870.,
75  b54 = 27048./710875.,
76  b61 = 89303./78045., b62 = -2025./473., b63 = 994650./244541.,
77  b64 = -2547216./28122215., b65 = 475./2967.,
78  b71 = 1247./10890., b72 = 0., b73 = 57375./108053.,
79  b74 = -1229312./1962015., b75 = 125./207., b76 = 43./114.;
80 
81  const G4double
82  dc1 = b71 - 21487./185130.,
83  dc2 = b72 - 0.,
84  dc3 = b73 - 963225./1836901.,
85  dc4 = b74 + 39864832./33354255.,
86  dc5 = b75 - 2575./3519.,
87  dc6 = b76 - 4472./4845.,
88  dc7 = 0. + 1./10.;
89 
90  //RightHandSide(yInput, dydx);
91  for(int i = 0; i < GetNumberOfVariables(); ++i)
92  yTemp[i] = yInput[i] + hstep * b21 * dydx[i];
93 
94  RightHandSide(yTemp, ak2);
95  for(int i = 0; i < GetNumberOfVariables(); ++i)
96  yTemp[i] = yInput[i] + hstep * (b31 * dydx[i] + b32 * ak2[i]);
97 
98  RightHandSide(yTemp, ak3);
99  for(int i = 0;i < GetNumberOfVariables(); ++i)
100  yTemp[i] = yInput[i] + hstep * (b41 * dydx[i] + b42 * ak2[i] +
101  b43 * ak3[i]);
102 
103  RightHandSide(yTemp, ak4);
104  for(int i = 0; i < GetNumberOfVariables(); ++i)
105  yTemp[i] = yInput[i] + hstep * (b51 * dydx[i] + b52 * ak2[i] +
106  b53 * ak3[i] + b54 * ak4[i]);
107 
108  RightHandSide(yTemp, ak5);
109  for(int i = 0; i < GetNumberOfVariables(); ++i)
110  yTemp[i] = yInput[i] + hstep * (b61 * dydx[i] + b62 * ak2[i] +
111  b63 * ak3[i] + b64 * ak4[i] +
112  b65 * ak5[i]);
113 
114  RightHandSide(yTemp, ak6);
115  for(int i = 0; i < GetNumberOfVariables(); ++i)
116  yOutput[i] = yInput[i] + hstep * (b71 * dydx[i] + b72 * ak2[i] +
117  b73 * ak3[i] + b74 * ak4[i] +
118  b75 * ak5[i] + b76 * ak6[i]);
119 
120  if (dydxOutput && yError) {
121  RightHandSide(yOutput, dydxOutput);
122  for(int i = 0; i < GetNumberOfVariables(); ++i)
123  yError[i] = hstep * (dc1 * dydx[i] + dc2 * ak2[i] + dc3 * ak3[i] +
124  dc4 * ak4[i] + dc5 * ak5[i] + dc6 * ak6[i] +
125  dc7 * dydxOutput[i]);
126  }
127 }
