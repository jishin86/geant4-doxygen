56 G4ClassificationOfNewTrack G4AdjointStackingAction::ClassifyNewTrack(const G4Track * aTrack)
57 {
58  G4ClassificationOfNewTrack classification = fUrgent;
59  G4String partType = aTrack->GetParticleDefinition()->GetParticleType();
60  adjoint_mode = partType.contains(G4String("adjoint"));
61  if (!adjoint_mode ){
62  if (!reclassification_stage) classification = fWaiting;
63  else { //need to check if forwrad tracking can be continued use of
64  if (theAdjointTrackingAction->GetNbOfAdointTracksReachingTheExternalSurface()>0) {
65  if (theFwdStackingAction) classification = theFwdStackingAction->ClassifyNewTrack(aTrack);
66  }
67  else classification = fKill;
68  }
69  }
70  else if (theUserAdjointStackingAction) classification = theUserAdjointStackingAction->ClassifyNewTrack(aTrack);
71  return classification;
72 }
