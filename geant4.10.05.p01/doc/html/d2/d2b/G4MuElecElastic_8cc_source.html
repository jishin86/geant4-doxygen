75 void G4MuElecElastic::InitialiseProcess(const G4ParticleDefinition*)
76 {
77  if(!isInitialised)

80  SetBuildTableFlag(false);
81  if(!EmModel()) SetEmModel(new G4MuElecElasticModel);
