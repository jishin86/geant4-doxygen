106 void G4IStore::AddImportanceGeometryCell(G4double importance,
107  const G4GeometryCell &gCell)
108 {
109  if (importance < 0 ) {
110  Error("AddImportanceGeometryCell() - Invalid importance value given.");
111  }
112  if (!IsInWorld(gCell.GetPhysicalVolume()) ) {
113  Error("AddImportanceGeometryCell() - Physical volume not found!");

117  Error("AddImportanceGeometryCell() - Region already existing!");

152 G4double G4IStore::GetImportance(const G4VPhysicalVolume &aVolume,
153  G4int aRepNum) const
154 {
155 #ifdef G4MULTITHREADED
156  G4MUTEXLOCK(&G4IStore::IStoreMutex);
157 #endif
158  SetInternalIterator(G4GeometryCell(aVolume, aRepNum));
159  G4GeometryCellImportance::const_iterator gCellIterator = fCurrentIterator;
160  if (gCellIterator==fGeometryCelli.end()) {
161  Error("GetImportance() - Region does not exist!");
162  return 0.;

164  G4double importance_value = (*fCurrentIterator).second;
165 #ifdef G4MULTITHREADED
