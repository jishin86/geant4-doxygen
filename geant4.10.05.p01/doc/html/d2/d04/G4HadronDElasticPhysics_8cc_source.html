113 void G4HadronDElasticPhysics::ConstructProcess()
114 {
115  const G4double elimitAntiNuc = 100.1*MeV;
116  if(verbose > 1) {
117  G4cout << "### HadronDElasticPhysics Construct Processes "
118  << " for anti-neuclei "
119  << elimitAntiNuc/GeV << " GeV" << G4endl;
120  }
121 
122  G4AntiNuclElastic* anuc = new G4AntiNuclElastic();
123  anuc->SetMinEnergy(elimitAntiNuc);
124  G4CrossSectionElastic* anucxs =
125  new G4CrossSectionElastic(anuc->GetComponentCrossSection());
126 
127  G4HadronElastic* lhep0 = new G4HadronElastic();
128  G4HadronElastic* lhep1 = new G4HadronElastic();
129  lhep1->SetMaxEnergy(10.1*MeV);
130  G4HadronElastic* lhep2 = new G4HadronElastic();
131  lhep2->SetMaxEnergy(elimitAntiNuc);
132 
133  G4DiffuseElastic* model = nullptr;
134  G4HadronElasticProcess* hel = nullptr;
135 
136  auto myParticleIterator=GetParticleIterator();
137  myParticleIterator->reset();
138  while( (*myParticleIterator)() )
139  {
140  G4ParticleDefinition* particle = myParticleIterator->value();
141  G4ProcessManager* pmanager = particle->GetProcessManager();
142  G4String pname = particle->GetParticleName();
143  if(pname == "anti_lambda" ||
144  pname == "anti_neutron" ||
145  pname == "anti_omega-" ||
146  pname == "anti_sigma-" ||
147  pname == "anti_sigma+" ||
148  pname == "anti_xi-" ||
149  pname == "anti_xi0" ||
150  pname == "lambda" ||
151  pname == "omega-" ||
152  pname == "sigma-" ||
153  pname == "sigma+" ||
154  pname == "xi-" ||
155  pname == "alpha" ||
156  pname == "deuteron" ||
157  pname == "triton"
158  ) {
159 
160  hel = new G4HadronElasticProcess();
161  hel->RegisterMe(lhep0);
162  pmanager->AddDiscreteProcess(hel);
163  if(verbose > 1) {
164  G4cout << "### HadronDElasticPhysics: " << hel->GetProcessName()
165  << " added for " << particle->GetParticleName() << G4endl;
166  }
167 
168  } else if(pname == "proton") {
169 
170  hel = new G4HadronElasticProcess();
171  hel->AddDataSet(new G4BGGNucleonElasticXS(particle));
172  //hel->AddDataSet(new G4CHIPSElasticXS());
173  model = new G4DiffuseElastic();
174  hel->RegisterMe(lhep1);
175  hel->RegisterMe(model);
176  pmanager->AddDiscreteProcess(hel);
177  if(verbose > 1) {
178  G4cout << "### HadronDElasticPhysics: " << hel->GetProcessName()
179  << " added for " << particle->GetParticleName() << G4endl;
180  }
181 
182  } else if(pname == "neutron") {
183 
184  hel = new G4HadronElasticProcess();
185  hel->AddDataSet(G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4NeutronElasticXS::Default_Name()));
186  model = new G4DiffuseElastic();
187  hel->RegisterMe(lhep1);
188  hel->RegisterMe(model);
189  pmanager->AddDiscreteProcess(hel);
190  if(verbose > 1) {
191  G4cout << "### HadronDElasticPhysics: "
192  << hel->GetProcessName()
193  << " added for " << particle->GetParticleName() << G4endl;
194  }
195 
196  } else if (pname == "pi+" || pname == "pi-") {
197 
198  hel = new G4HadronElasticProcess();
199  hel->AddDataSet(new G4BGGPionElasticXS(particle));
200  model = new G4DiffuseElastic();
201  hel->RegisterMe(lhep1);
202  hel->RegisterMe(model);
203  pmanager->AddDiscreteProcess(hel);
204  if(verbose > 1) {
205  G4cout << "### HadronDElasticPhysics: " << hel->GetProcessName()
206  << " added for " << particle->GetParticleName() << G4endl;
207  }
208 
209  } else if(pname == "kaon-") {
210 
211  hel = new G4HadronElasticProcess();
212  hel->AddDataSet(G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4ChipsKaonMinusElasticXS::Default_Name()));
213  model = new G4DiffuseElastic();
214  hel->RegisterMe(lhep1);
215  hel->RegisterMe(model);
216  pmanager->AddDiscreteProcess(hel);
217  if(verbose > 1) {
218  G4cout << "### HadronElasticPhysics: " << hel->GetProcessName()
219  << " added for " << particle->GetParticleName() << G4endl;
220  }
221  } else if(pname == "kaon+") {
222 
223  hel = new G4HadronElasticProcess();
224  hel->AddDataSet(G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4ChipsKaonPlusElasticXS::Default_Name()));
225  model = new G4DiffuseElastic();
226  hel->RegisterMe(lhep1);
227  hel->RegisterMe(model);
228  pmanager->AddDiscreteProcess(hel);
229  if(verbose > 1) {
230  G4cout << "### HadronElasticPhysics: " << hel->GetProcessName()
231  << " added for " << particle->GetParticleName() << G4endl;
232  }
233  } else if(pname == "kaon0S" ||
234  pname == "kaon0L"
235  ) {
236 
237  hel = new G4HadronElasticProcess();
238  hel->AddDataSet(G4CrossSectionDataSetRegistry::Instance()->GetCrossSectionDataSet(G4ChipsKaonZeroElasticXS::Default_Name()));
239  model = new G4DiffuseElastic();
240  hel->RegisterMe(lhep1);
241  hel->RegisterMe(model);
242  pmanager->AddDiscreteProcess(hel);
243  if(verbose > 1) {
244  G4cout << "### HadronElasticPhysics: " << hel->GetProcessName()
245  << " added for " << particle->GetParticleName() << G4endl;
246  }
247  } else if(
248  pname == "anti_proton" ||
249  pname == "anti_alpha" ||
250  pname == "anti_deuteron" ||
251  pname == "anti_triton" ||
252  pname == "anti_He3" ) {
253 
254  hel = new G4HadronElasticProcess();
255  hel->AddDataSet(anucxs);
256  hel->RegisterMe(lhep2);
257  hel->RegisterMe(anuc);
258  pmanager->AddDiscreteProcess(hel);
259  }
260  }
261  if(verbose > 1) {
262  G4cout << "### HadronDElasticPhysics Construct Processes " << G4endl;
263  }
264 }
