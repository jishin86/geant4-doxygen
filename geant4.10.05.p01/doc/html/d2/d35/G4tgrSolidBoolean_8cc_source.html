43 G4tgrSolidBoolean::G4tgrSolidBoolean( const std::vector<G4String>& wl )
44 {
45  // :SOLID/:VOLU VOLU UNION/SUBS/INTERS VOLU1 VOLU2 ROTM POSX POSY POSZ
46 
47  if( wl.size() != 9 )
48  {
49  G4tgrUtils::DumpVS(wl, "G4tgrSolidBoolean::G4tgrSolidBoolean()");
50  G4Exception("G4tgrSolidBoolean::G4tgrSolidBoolean()", "InvalidInput",
51  FatalException, "Line read with less or more than 9 words.");
52  }
53 
54  //---------- Set name
55  theName = G4tgrUtils::GetString( wl[1] );
56 
57  G4tgrVolumeMgr* volmgr = G4tgrVolumeMgr::GetInstance();
58  const G4tgrSolid* sol1 = volmgr->FindSolid( G4tgrUtils::GetString( wl[3] ));
59  if( !sol1 )
60  {
61  sol1 = volmgr->FindVolume( G4tgrUtils::GetString( wl[3] ), 1)->GetSolid();
62  }
63  const G4tgrSolid* sol2 = volmgr->FindSolid( G4tgrUtils::GetString( wl[4] ));
64  if( !sol2 )
65  {
66  sol2 = volmgr->FindVolume( G4tgrUtils::GetString( wl[4] ), 1)->GetSolid();
67  }
68  theSolids.push_back( sol1 );
69  theSolids.push_back( sol2 );
70 
71  //---------- Set relative placement and rotation matrix
72  theRelativeRotMatName = G4tgrUtils::GetString( wl[5] );
73  theRelativePlace = G4ThreeVector( G4tgrUtils::GetDouble(wl[6]),
74  G4tgrUtils::GetDouble(wl[7]),
75  G4tgrUtils::GetDouble(wl[8]) );
76  //---------- Set solid type
77  G4String wl2 = wl[2];
78  for( size_t ii = 0; ii < wl2.length(); ii++ )
79  {
80  wl2[ii] = toupper( wl2[ii] );
81  }
82  theType = "Boolean_" + wl2;
83 
84 #ifdef G4VERBOSE
85  if( G4tgrMessenger::GetVerboseLevel() >= 1 )
86  {
87  G4cout << " Created " << *this << G4endl;
88  }
89 #endif
90 
91  G4tgrVolumeMgr::GetInstance()->RegisterMe( this );
92 }
