63 void B3aEventAction::EndOfEventAction(const G4Event* evt )
64 {
65  //Hits collections
66  //
67  G4HCofThisEvent* HCE = evt->GetHCofThisEvent();
68  if(!HCE) return;
69 
70  // Get hits collections IDs
71  if (fCollID_cryst < 0) {
72  G4SDManager* SDMan = G4SDManager::GetSDMpointer();
73  fCollID_cryst = SDMan->GetCollectionID("crystal/edep");
74  fCollID_patient = SDMan->GetCollectionID("patient/dose");
75  }
76 
77  //Energy in crystals : identify 'good events'
78  //
79  const G4double eThreshold = 500*keV;
80  G4int nbOfFired = 0;
81 
82  G4THitsMap<G4double>* evtMap =
83  (G4THitsMap<G4double>*)(HCE->GetHC(fCollID_cryst));
84 
85  std::map<G4int,G4double*>::iterator itr;
86  for (itr = evtMap->GetMap()->begin(); itr != evtMap->GetMap()->end(); itr++) {
88  G4double edep = *(itr->second);
89  if (edep > eThreshold) nbOfFired++;
91  }
92  if (nbOfFired == 2) fRunAction->CountEvent();
93 
94  //Dose deposit in patient
95  //
96  G4double dose = 0.;
97 
98  evtMap = (G4THitsMap<G4double>*)(HCE->GetHC(fCollID_patient));
99 
100  for (itr = evtMap->GetMap()->begin(); itr != evtMap->GetMap()->end(); itr++) {
102  dose = *(itr->second);
103  }
104  if (dose > 0.) fRunAction->SumDose(dose);
105 }
