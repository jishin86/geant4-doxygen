83 void IonHIJINGPhysics::ConstructProcess()
84 {
85  if(fWasActivated) { return; }
86  fWasActivated = true;
87 
88  G4double emin = 0.*MeV;
89  G4double emaxFTF = 25.*GeV;
90  G4double eminHIJ = 12.*GeV;
91  G4double emaxHIJ = G4HadronicParameters::Instance()->GetMaxEnergy();
92 
93  G4ExcitationHandler* handler = new G4ExcitationHandler();
94  G4PreCompoundModel* thePreCompound = new G4PreCompoundModel(handler);
95 
96  // Binary Cascade
97  theIonBC = new G4BinaryLightIonReaction(thePreCompound);
98  theIonBC->SetMinEnergy(0.0);
99  theIonBC->SetMaxEnergy(4*GeV); //4
100 
101  // FTFP
102  theBuilder = new G4FTFBuilder("FTFP",thePreCompound);
103  theFTFP = theBuilder->GetModel();
104  theFTFP->SetMinEnergy(2*GeV);
105  theFTFP->SetMaxEnergy(emaxFTF);
106 
107  //HIJING
108  fModel = new G4HIJING_Model();
109  fModel->SetMinEnergy( eminHIJ );
110  fModel->SetMaxEnergy( emaxHIJ );
111 
112  theNuclNuclData =
113  new G4CrossSectionInelastic( new G4ComponentGGNuclNuclXsc() );
114 
115  AddProcess("dInelastic", G4Deuteron::Deuteron(),false);
116  AddProcess("tInelastic",G4Triton::Triton(),false);
117  AddProcess("He3Inelastic",G4He3::He3(),true);
118  AddProcess("alphaInelastic", G4Alpha::Alpha(),true);
119  AddProcess("ionInelastic",G4GenericIon::GenericIon(),true);
120 
121  if(fVerbose > 1) {
122  G4cout << "IonHIJINGPhysics::ConstructProcess done! "
123  << G4endl;
124  }
125 }

130  G4ParticleDefinition* part, G4bool isIon)

133  G4ProcessManager* pManager = part->GetProcessManager();
134  pManager->AddDiscreteProcess(hadi);
135  hadi->AddDataSet(theNuclNuclData);
136  hadi->RegisterMe( theIonBC );
137  hadi->RegisterMe( theFTFP );
138  hadi->RegisterMe( fModel );
