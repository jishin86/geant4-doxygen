139 inline G4double G4GEMChannelVI::I2(G4double s0, G4double sx)
140 {
141  G4double S = 1.0/std::sqrt(s0);
142  G4double Sx = 1.0/std::sqrt(sx);
143 
144  G4double p1 = S*S*S*( 1.0 + S*S*( 1.5 + 3.75*S*S) );
145  G4double p2 = Sx*Sx*Sx*( 1.0 + Sx*Sx*( 1.5 + 3.75*Sx*Sx) )*G4Exp(sx-s0);
146 
147  return p1-p2;
148 }
