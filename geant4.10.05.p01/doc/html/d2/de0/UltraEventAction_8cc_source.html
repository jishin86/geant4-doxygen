110 #endif
111 
112  for(G4int iHit=0; iHit<nOptHits; iHit++){
113  G4double HitEnergy = (*OpticalHitsColl)[iHit]->GetEnergy() ;
114  man->FillH1(1,HitEnergy/eV);
115  }
116 
117  }

119  man->FillH1(2,nOptHits);
