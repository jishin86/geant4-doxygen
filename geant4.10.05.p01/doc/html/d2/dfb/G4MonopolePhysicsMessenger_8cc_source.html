48 G4MonopolePhysicsMessenger::G4MonopolePhysicsMessenger(G4MonopolePhysics* p)
49  : G4UImessenger(), fPhys(p)
50 {
51  fPhysicsDir = new G4UIdirectory("/monopole/");
52  fPhysicsDir->SetGuidance("histograms control");
53 
54  fPhysicsCmd = new G4UIcommand("/monopole/setup",this);
55  fPhysicsCmd->SetGuidance("Setup monopole");
56  //
57  G4UIparameter* qmag = new G4UIparameter("qmag",'d',false);
58  qmag->SetGuidance("Magnetic charge");
59  qmag->SetDefaultValue("1");
60  fPhysicsCmd->SetParameter(qmag);
61 
62  G4UIparameter* q = new G4UIparameter("qelec",'d',false);
63  q->SetGuidance("Electric charge charge");
64  q->SetDefaultValue("0");
65  fPhysicsCmd->SetParameter(q);
66  //
67  G4UIparameter* mass = new G4UIparameter("mass",'d',false);
68  mass->SetGuidance("mass");
69  mass->SetParameterRange("mass>0.");
70  qmag->SetDefaultValue("100");
71  fPhysicsCmd->SetParameter(mass);
72  //
73  G4UIparameter* unit = new G4UIparameter("unit",'s',false);
74  fPhysicsCmd->SetParameter(unit);
75  qmag->SetDefaultValue("GeV");
76  fPhysicsCmd->AvailableForStates(G4State_PreInit);
77 }

89 void G4MonopolePhysicsMessenger::SetNewValue(G4UIcommand* command,
90  G4String newValue)
91 {
92  if (command == fPhysicsCmd)
93  { G4double q, m; G4double mass;
94  G4String unts;
95  std::istringstream is(newValue);
96  is >> m >> q >> mass >> unts;
97  G4String unit = unts;
98  G4double vUnit = G4UIcommand::ValueOf(unit);
99  fPhys->SetMagneticCharge(m);
100  fPhys->SetElectricCharge(q);
101  fPhys->SetMonopoleMass(mass*vUnit);
102  }
103 }
