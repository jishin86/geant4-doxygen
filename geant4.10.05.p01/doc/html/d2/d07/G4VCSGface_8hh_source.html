33 // G4VCSGface
34 //
35 // Class description:
36 //
37 // Definition of the virtual base class G4VCSGface, one side (or face)
38 // of a CSG-like solid. It should be possible to build a CSG entirely out of
39 // connecting CSG faces.
40 //
41 // Each face has an inside and outside surface, the former represents
42 // the inside of the volume, the latter, the outside.
43 //
44 // Virtual members:
45 //
46 // -------------------------------------------------------------------
47 // Intersect( const G4ThreeVector &p, const G4ThreeVector &v,
48 // G4bool outGoing, G4double surfTolerance,
49 // G4double &distance, G4double &distFromSurface,
50 // G4ThreeVector &normal, G4bool &allBehind );
51 //
52 // p - (in) position
53 // v - (in) direction (assumed to be a unit vector)
54 // outgoing - (in) true, to consider only inside surfaces
55 // false, to consider only outside surfaces
56 // distance - (out) distance to intersection
57 // distFromSurface - (out) distance from surface (along surface normal),
58 // < 0 if the point is in front of the surface
59 // normal - (out) normal of surface at intersection point
60 // allBehind - (out) true, if entire surface is behind normal
61 //
62 // return value = true if there is an intersection,
63 // false if there is no intersection
64 // (all output arguments undefined)
65 //
66 // Determine the distance along a line to the face.
67 //
68 // -------------------------------------------------------------------
69 // Distance( const G4ThreeVector &p, const G4bool outgoing );
70 //
71 // p - (in) position
72 // outgoing - (in) true, to consider only inside surfaces
73 // false, to consider only outside surfaces
74 //
75 // return value = distance to closest surface satisifying requirements
76 // or kInfinity if no such surface exists
77 //
78 // Determine the distance of a point from either the inside or outside
79 // surfaces of the face.
80 //
81 // -------------------------------------------------------------------
82 // Inside( const G4ThreeVector &p, const G4double tolerance,
83 // G4double *bestDistance );
84 //
85 // p - (in) position
86 // tolerance - (in) tolerance defining the bounds of the "kSurface",
87 // nominally equal to kCarTolerance/2
88 // bestDistance - (out) distance to closest surface (in or out)
89 //
90 // return value = kInside if the point is closest to the inside surface
91 // kOutside if the point is closest to the outside surface
92 // kSurface if the point is withing tolerance of the surface
93 //
94 // Determine whether a point is inside, outside, or on the surface of
95 // the face.
96 //
97 // -------------------------------------------------------------------
98 // Normal( const G4ThreeVector &p, G4double *bestDistance );
99 //
100 // p - (in) position
101 // bestDistance - (out) distance to closest surface (in or out)
102 //
103 // return value = the normal of the surface nearest the point
104 //
105 // Return normal of surface closest to the point.
106 //
107 // -------------------------------------------------------------------
108 // Extent( const G4ThreeVector axis );
109 //
110 // axis - (in) unit vector defining direction
111 //
112 // return value = the largest point along the given axis of the
113 // the face's extent.
114 //
115 // -------------------------------------------------------------------
116 // CalculateExtent( const EAxis pAxis,
117 // const G4VoxelLimit &pVoxelLimit,
118 // const G4AffineTransform &pTransform,
119 // G4double &min, G4double &max )
120 //
121 // pAxis - (in) The x,y, or z axis in which to check
122 // the shapes 3D extent against
123 // pVoxelLimit - (in) Limits along x, y, and/or z axes
124 // pTransform - (in) A coordinate transformation on which
125 // to apply to the shape before testing
126 // min - (out) If the face has any point on its
127 // surface after tranformation and limits
128 // along pAxis that is smaller than the value
129 // of min, than it is used to replace min.
130 // Undefined if the return value is false.
131 // max - (out) Same as min, except for the largest
132 // point.
133 // Undefined if the return value is false.
134 //
135 // return value = true if anything remains of the face
136 //
137 // Calculate the extent of the face for the voxel navigator.
138 // In analogy with CalculateExtent for G4VCSGfaceted, this is
139 // done in the following steps:
140 //
141 // 1. Transform the face using pTranform, an arbitrary 3D
142 // rotation/offset/reflection
143 // 2. Clip the face to those boundaries as specified in
144 // pVoxelLimit. This may include limits in any number
145 // of x, y, or z axes.
146 // 3. For each part of the face that remains (there could
147 // be many separate pieces in general):
148 // 4. Check to see if the piece overlaps the currently
149 // existing limits along axis pAxis. For
150 // pVoxelLimit.IsLimited(pAxis) = false, there are
151 // no limits.
152 // 5. For a piece that does overlap, update min/max
153 // accordingly (within confines of pre-existing
154 // limits) along the direction pAxis.
155 // 6. If min/max were updated, return true
156 //
157 // -------------------------------------------------------------------
158 // G3VCSGface *Clone()
159 //
160 // This method is invoked by G4CSGfaceted during the copy constructor
161 // or the assignment operator. Its purpose is to return a pointer
162 // (of type G4VCSGface) to a duplicate copy of the face.
163 // The implementation is straight forward for inherited classes. Example:
164 //
165 // G4VCSGface G4PolySideFace::Clone() { return new G4PolySideFace(*this); }
166 //
167 // Of course, this assumes the copy constructor of G4PolySideFace is
168 // correctly implemented.
169 //
170 // Implementation notes:
171 // * distance.
172 // The meaning of distance includes the boundaries of the face.
173 // For example, for a rectangular, planer face:
174 //
175 // A | B | C
176 // | |
177 // -------+--------------+-----
178 // D | I | E
179 // | |
180 // -------+--------------+-----
181 // F | G | H
182 // | |
183 //
184 // A, C, F, and H: closest distance is the distance to
185 // the adjacent corner.
186 //
187 // B, D, E, and G: closest distance is the distance to
188 // the adjacent line.
189 //
190 // I: normal distance to plane
191 //
192 // For non-planer faces, one can use the normal to decide when
193 // a point falls off the edge and then act accordingly.
194 //
195 //
196 // Usage:
197 //
198 // A CSG shape can be defined by putting together any number of generic
199 // faces, as long as the faces cover the entire surface of the shape
200 // without overlapping.
201 //
202 // G4VSolid::CalculateExtent
203 //
204 // Define unit vectors along the specified transform axis.
205 // Use the inverse of the specified coordinate transformation to rotate
206 // these unit vectors. Loop over each face, call face->Extent, and save
207 // the maximum value.
208 //
209 // G4VSolid::Inside
210 //
211 // To decide if a point is inside, outside, or on the surface of the shape,
212 // loop through all faces, and find the answer from face->Inside which gives
213 // a value of "bestDistance" smaller than any other. While looping, if any
214 // face->Inside returns kSurface, this value can be returned immediately.
215 //
216 // EInside answer;
217 // G4VCSGface *face = faces;
218 // G4double best = kInfinity;
219 // do {
220 // G4double distance;
221 // EInside result = (*face)->Inside( p, kCarTolerance/2, distance );
222 // if (result == kSurface) return kSurface;
223 // if (distance < best) {
224 // best = distance;
225 // answer = result;
226 // }
227 // } while( ++face < faces + numFaces );
228 //
229 // return(answer);
230 //
231 // G4VSolid::SurfaceNormal
232 //
233 // Loop over all faces, call face->Normal, and return the normal to the face
234 // that is closest to the point.
235 //
236 // G4VSolid::DistanceToIn(p)
237 //
238 // Loop over all faces, invoking face->Distance with outgoing = false,
239 // and save the answer that is smallest.
240 //
241 // G4VSolid::DistanceToIn(p,v)
242 //
243 // Loop over all faces, invoking face->Intersect with outgoing = false,
244 // and save the answer that is smallest.
245 //
246 // G4VSolid::DistanceToOut(p)
247 //
248 // Loop over all faces, invoking face->Distance with outgoing = true,
249 // and save the answer that is smallest.
250 //
251 // G4VSolid::DistanceToOut(p,v)
252 //
253 // Loop over all faces, invoking face->Intersect with outgoing = true,
254 // and save the answer that is smallest. If there is more than one answer,
255 // or if allBehind is false for the one answer, return validNorm as false.
