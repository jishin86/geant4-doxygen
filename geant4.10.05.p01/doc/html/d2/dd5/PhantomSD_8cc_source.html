76 G4bool PhantomSD::ProcessHits(G4Step* aStep, G4TouchableHistory*)
77 {
78 
79  Run* run = static_cast<Run*>(
80  G4RunManager::GetRunManager()->GetNonConstCurrentRun());
81 
82  G4double edep = aStep->GetTotalEnergyDeposit();
83 
84  // only if there is energy deposition
85  if(0.0 < edep) {
86 
87  G4ThreeVector p1 = aStep->GetPreStepPoint()->GetPosition();
88  G4ThreeVector p2 = aStep->GetPostStepPoint()->GetPosition();
89  G4double x1 = p1.x();
90  G4double y1 = p1.y();
91  G4double z1 = p1.z() - fShiftZ;
92  G4double r1 = std::sqrt(x1*x1 + y1*y1);
93  G4double x2 = p2.x();
94  G4double y2 = p2.y();
95  G4double z2 = p2.z() - fShiftZ;
96  G4double r2 = std::sqrt(x2*x2 + y2*y2);
97  G4double x0 = 0.5*(x1 + x2);
98  G4double y0 = 0.5*(y1 + y2);
99  G4double z0 = 0.5*(z1 + z2);
100  G4double r0 = std::sqrt(x0*x0 + y0*y0);
101 
102  run->AddPhantomStep(edep,r1,z1,r2,z2,r0,z0);
103 
104  if(run->GetVerbose()) {
105  G4cout << "PhantomSD: energy = " << edep/MeV
106  << " MeV is deposited at the step at r1,z1= " << r1 << " " << z1
107  << "; r2,z2= " << r2 << " " << z2 << G4endl;
108  }
109  }
110 
111  return true;
112 }
