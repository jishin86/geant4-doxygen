67  G4String newValue)

70  DetectorConstruction::EMethod method;

72  method = DetectorConstruction::kWithDirectMatrix;

74  method = DetectorConstruction::kWithInverseMatrix;

77  else if ( newValue == "WithEulerAngles" )
78  method = DetectorConstruction::kWithEulerAngles;
79  else if ( newValue == "WithReflections" )

81  else return;
82  fDetectorConstruction->SetMethod(method);
