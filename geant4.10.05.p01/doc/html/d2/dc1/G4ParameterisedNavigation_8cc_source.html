77 G4double G4ParameterisedNavigation::
78  ComputeStep(const G4ThreeVector& localPoint,
79  const G4ThreeVector& localDirection,
80  const G4double currentProposedStepLength,
81  G4double& newSafety,
82  G4NavigationHistory& history,
83  G4bool& validExitNormal,
84  G4ThreeVector& exitNormal,
85  G4bool& exiting,
86  G4bool& entering,
87  G4VPhysicalVolume *(*pBlockedPhysical),
88  G4int& blockedReplicaNo)
89 {
90  G4VPhysicalVolume *motherPhysical, *samplePhysical;
91  G4VPVParameterisation *sampleParam;
92  G4LogicalVolume *motherLogical;
93  G4VSolid *motherSolid, *sampleSolid;
94  G4ThreeVector sampleDirection;
95  G4double ourStep=currentProposedStepLength, ourSafety;
96  G4double motherSafety, motherStep=DBL_MAX;
97  G4bool motherValidExitNormal=false;
98  G4ThreeVector motherExitNormal;
99 
100  G4int sampleNo;
101 
102  G4bool initialNode, noStep;
103  G4SmartVoxelNode *curVoxelNode;
104  G4int curNoVolumes, contentNo;
105  G4double voxelSafety;
106 
107  // Replication data
108  //
109  EAxis axis;
110  G4int nReplicas;
111  G4double width, offset;
112  G4bool consuming;
113 
114  motherPhysical = history.GetTopVolume();
115  motherLogical = motherPhysical->GetLogicalVolume();
116  motherSolid = motherLogical->GetSolid();
117 
118  //
119  // Compute mother safety
120  //
121 
122  motherSafety = motherSolid->DistanceToOut(localPoint);
123  ourSafety = motherSafety; // Working isotropic safety
124 
125 #ifdef G4VERBOSE
126  if ( fCheck )
127  {
128  if( motherSafety < 0.0 )
129  {
130  motherSolid->DumpInfo();
131  std::ostringstream message;
132  message << "Negative Safety In Voxel Navigation !" << G4endl
133  << " Current solid " << motherSolid->GetName()
134  << " gave negative safety: " << motherSafety << G4endl
135  << " for the current (local) point " << localPoint;
136  G4Exception("G4ParameterisedNavigation::ComputeStep()",
137  "GeomNav0003", FatalException, message);
138  }
139  if( motherSolid->Inside(localPoint)==kOutside )
140  {
141  std::ostringstream message;
142  message << "Point is outside Current Volume !" << G4endl
143  << " Point " << localPoint
144  << " is outside current volume " << motherPhysical->GetName()
145  << G4endl;
146  G4double estDistToSolid= motherSolid->DistanceToIn(localPoint);
147  G4cout << " Estimated isotropic distance to solid (distToIn)= "
148  << estDistToSolid;
149  if( estDistToSolid > 100.0 * motherSolid->GetTolerance() )
150  {
151  motherSolid->DumpInfo();
152  G4Exception("G4ParameterisedNavigation::ComputeStep()",
153  "GeomNav0003", FatalException, message,
154  "Point is far outside Current Volume !");
155  }
156  else
157  G4Exception("G4ParameterisedNavigation::ComputeStep()",
158  "GeomNav1002", JustWarning, message,
159  "Point is a little outside Current Volume.");
160  }
161 
162  // Compute early:
163  // a) to check whether point is (wrongly) outside
164  // (signaled if step < 0 or step == kInfinity )
165  // b) to check value against answer of daughters!
166  //
167  motherStep = motherSolid->DistanceToOut(localPoint,
168  localDirection,
169  true,
170  &motherValidExitNormal,
171  &motherExitNormal);
172 
173  if( (motherStep >= kInfinity) || (motherStep < 0.0) )
174  {
175  // Error - indication of being outside solid !!
176  //
177  fLogger->ReportOutsideMother(localPoint, localDirection, motherPhysical);
178 
179  ourStep = motherStep = 0.0;
180  exiting = true;
181  entering = false;
182 
183  // If we are outside the solid does the normal make sense?
184  validExitNormal = motherValidExitNormal;
185  exitNormal = motherExitNormal;
186 
187  *pBlockedPhysical= 0; // or motherPhysical ?
188  blockedReplicaNo= 0; // or motherReplicaNumber ?
189 
190  newSafety= 0.0;
191  return ourStep;
192  }
193  }
194 #endif
195 
196  initialNode = true;
197  noStep = true;
198 
199  // By definition, the parameterised volume is the first
200  // (and only) daughter of the mother volume
201  //
202  samplePhysical = motherLogical->GetDaughter(0);
203  samplePhysical->GetReplicationData(axis,nReplicas,width,offset,consuming);
204  fBList.Enlarge(nReplicas);
205  fBList.Reset();
206 
207  // Exiting normal optimisation
208  //
209  if (exiting && (*pBlockedPhysical==samplePhysical) && validExitNormal)
210  {
211  if (localDirection.dot(exitNormal)>=kMinExitingNormalCosine)
212  {
213  // Block exited daughter replica; Must be on boundary => zero safety
214  //
215  fBList.BlockVolume(blockedReplicaNo);
216  ourSafety = 0;
217  }
218  }
219  exiting = false;
220  entering = false;
221 
222  sampleParam = samplePhysical->GetParameterisation();
223 
224  // Loop over voxels & compute daughter safeties & intersections
225 
226  do
227  {
228  curVoxelNode = fVoxelNode;
229  curNoVolumes = curVoxelNode->GetNoContained();
230 
231  for ( contentNo=curNoVolumes-1; contentNo>=0; contentNo-- )
232  {
233  sampleNo = curVoxelNode->GetVolume(contentNo);
234  if ( !fBList.IsBlocked(sampleNo) )
235  {
236  fBList.BlockVolume(sampleNo);
237 
238  // Call virtual methods, and copy information if needed
239  //
240  sampleSolid = IdentifyAndPlaceSolid( sampleNo, samplePhysical,
241  sampleParam );
242 
243  G4AffineTransform sampleTf(samplePhysical->GetRotation(),
244  samplePhysical->GetTranslation());
245  sampleTf.Invert();
246  const G4ThreeVector samplePoint = sampleTf.TransformPoint(localPoint);
247  const G4double sampleSafety = sampleSolid->DistanceToIn(samplePoint);
248  if ( sampleSafety<ourSafety )
249  {
250  ourSafety = sampleSafety;
251  }
252  if ( sampleSafety<=ourStep )
253  {
254  sampleDirection = sampleTf.TransformAxis(localDirection);
255  G4double sampleStep =
256  sampleSolid->DistanceToIn(samplePoint, sampleDirection);
257  if ( sampleStep<=ourStep )
258  {
259  ourStep = sampleStep;
260  entering = true;
261  exiting = false;
262  *pBlockedPhysical = samplePhysical;
263  blockedReplicaNo = sampleNo;
264 #ifdef G4VERBOSE
265  // Check to see that the resulting point is indeed in/on volume.
266  // This check could eventually be made only for successful
267  // candidate.
268 
269  if ( ( fCheck ) && ( sampleStep < kInfinity ) )
270  {
271  G4ThreeVector intersectionPoint;
272  intersectionPoint= samplePoint + sampleStep * sampleDirection;
273  EInside insideIntPt= sampleSolid->Inside(intersectionPoint);
274  if( insideIntPt != kSurface )
275  {
276  G4int oldcoutPrec = G4cout.precision(16);
277  std::ostringstream message;
278  message << "Navigator gets conflicting response from Solid."
279  << G4endl
280  << " Inaccurate solid DistanceToIn"
281  << " for solid " << sampleSolid->GetName() << G4endl
282  << " Solid gave DistanceToIn = "
283  << sampleStep << " yet returns " ;
284  if( insideIntPt == kInside )
285  message << "-kInside-";
286  else if( insideIntPt == kOutside )
287  message << "-kOutside-";
288  else
289  message << "-kSurface-";
290  message << " for this point !" << G4endl
291  << " Point = " << intersectionPoint
292  << G4endl;
293  if ( insideIntPt != kInside )
294  message << " DistanceToIn(p) = "
295  << sampleSolid->DistanceToIn(intersectionPoint);
296  if ( insideIntPt != kOutside )
297  message << " DistanceToOut(p) = "
298  << sampleSolid->DistanceToOut(intersectionPoint);
299  G4Exception("G4ParameterisedNavigation::ComputeStep()",
300  "GeomNav1002", JustWarning, message);
301  G4cout.precision(oldcoutPrec);
302  }
303  }
304 #endif
305  }
306  }
307  }
308  }
309 
310  if ( initialNode )
311  {
312  initialNode = false;
313  voxelSafety = ComputeVoxelSafety(localPoint,axis);
314  if ( voxelSafety<ourSafety )
315  {
316  ourSafety = voxelSafety;
317  }
318  if ( currentProposedStepLength<ourSafety )
319  {
320  // Guaranteed physics limited
321  //
322  noStep = false;
323  entering = false;
324  exiting = false;
325  *pBlockedPhysical = 0;
326  ourStep = kInfinity;
327  }
328  else
329  {
330  // Consider intersection with mother solid
331  //
332  if ( motherSafety<=ourStep )
333  {
334  if ( !fCheck )
335  {
336  motherStep = motherSolid->DistanceToOut(localPoint,
337  localDirection,
338  true,
339  &motherValidExitNormal,
340  &motherExitNormal);
341  }
342 
343  if( ( motherStep < 0.0 ) || ( motherStep >= kInfinity) )
344  {
345 #ifdef G4VERBOSE
346  fLogger->ReportOutsideMother(localPoint, localDirection, motherPhysical);
347 #endif
348  ourStep = motherStep = 0.0;
349  // Rely on the code below to set the remaining state, i.e.
350  // exiting, entering, exitNormal & validExitNormal,
351  // pBlockedPhysical etc.
352  }
353 #ifdef G4VERBOSE
354  if( motherValidExitNormal && ( fCheck || (motherStep<=ourStep)) )
355  {
356  fLogger->CheckAndReportBadNormal(motherExitNormal,
357  localPoint, localDirection,
358  motherStep, motherSolid,
359  "From motherSolid::DistanceToOut");
360  }
361 #endif
362  if ( motherStep<=ourStep )
363  {
364  ourStep = motherStep;
365  exiting = true;
366  entering = false;
367  if ( validExitNormal )
368  {
369  const G4RotationMatrix *rot = motherPhysical->GetRotation();
370  if (rot)
371  {
372  exitNormal *= rot->inverse();
373  }
374  }
375  }
376  else
377  {
378  validExitNormal = false;
379  }
380  }
381  }
382  newSafety=ourSafety;
383  }
384  if (noStep)
385  {
386  noStep = LocateNextVoxel(localPoint, localDirection, ourStep, axis);
387  }
388  } while (noStep);
389 
390  return ourStep;
391 }

397 G4double
398 G4ParameterisedNavigation::ComputeSafety(const G4ThreeVector& localPoint,
399  const G4NavigationHistory& history,
400  const G4double )
401 {
402  G4VPhysicalVolume *motherPhysical, *samplePhysical;
403  G4VPVParameterisation *sampleParam;
404  G4LogicalVolume *motherLogical;
405  G4VSolid *motherSolid, *sampleSolid;
406  G4double motherSafety, ourSafety;
407  G4int sampleNo, curVoxelNodeNo;
408 
409  G4SmartVoxelNode *curVoxelNode;
410  G4int curNoVolumes, contentNo;
411  G4double voxelSafety;
412 
413  // Replication data
414  //
415  EAxis axis;
416  G4int nReplicas;
417  G4double width, offset;
418  G4bool consuming;
419 
420  motherPhysical = history.GetTopVolume();
421  motherLogical = motherPhysical->GetLogicalVolume();
422  motherSolid = motherLogical->GetSolid();
423 
424  //
425  // Compute mother safety
426  //
427 
428  motherSafety = motherSolid->DistanceToOut(localPoint);
429  ourSafety = motherSafety; // Working isotropic safety
430 
431  //
432  // Compute daughter safeties
433  //
434 
435  // By definition, parameterised volumes exist as first
436  // daughter of the mother volume
437  //
438  samplePhysical = motherLogical->GetDaughter(0);
439  samplePhysical->GetReplicationData(axis, nReplicas,
440  width, offset, consuming);
441  sampleParam = samplePhysical->GetParameterisation();
442 
443  // Look inside the current Voxel only at the current point
444  //
445  if ( axis==kUndefined ) // 3D case: current voxel node is retrieved
446  { // from G4VoxelNavigation.
447  curVoxelNode = fVoxelNode;
448  }
449  else // 1D case: current voxel node is computed here.
450  {
451  curVoxelNodeNo = G4int((localPoint(fVoxelAxis)
452  -fVoxelHeader->GetMinExtent()) / fVoxelSliceWidth );
453  curVoxelNode = fVoxelHeader->GetSlice(curVoxelNodeNo)->GetNode();
454  fVoxelNodeNo = curVoxelNodeNo;
455  fVoxelNode = curVoxelNode;
456  }
457  curNoVolumes = curVoxelNode->GetNoContained();
458 
459  for ( contentNo=curNoVolumes-1; contentNo>=0; contentNo-- )
460  {
461  sampleNo = curVoxelNode->GetVolume(contentNo);
462 
463  // Call virtual methods, and copy information if needed
464  //
465  sampleSolid= IdentifyAndPlaceSolid( sampleNo,samplePhysical,sampleParam );
466 
467  G4AffineTransform sampleTf(samplePhysical->GetRotation(),
468  samplePhysical->GetTranslation());
469  sampleTf.Invert();
470  const G4ThreeVector samplePoint = sampleTf.TransformPoint(localPoint);
471  G4double sampleSafety = sampleSolid->DistanceToIn(samplePoint);
472  if ( sampleSafety<ourSafety )
473  {
474  ourSafety = sampleSafety;
475  }
476  }
477 
478  voxelSafety = ComputeVoxelSafety(localPoint,axis);
479  if ( voxelSafety<ourSafety )
480  {
481  ourSafety=voxelSafety;
482  }
483 
484  return ourSafety;
485 }

487 // ********************************************************************
488 // ComputeVoxelSafety
489 //
490 // Computes safety from specified point to collected voxel boundaries
491 // using already located point.
492 // ********************************************************************
493 //
494 G4double G4ParameterisedNavigation::
495 ComputeVoxelSafety(const G4ThreeVector& localPoint,
496  const EAxis pAxis) const
497 {
498  // If no best axis is specified, adopt default
499  // strategy as for placements
500  //
501  if ( pAxis==kUndefined )
502  {
503  return G4VoxelNavigation::ComputeVoxelSafety(localPoint);
504  }
505 
506  G4double voxelSafety, plusVoxelSafety, minusVoxelSafety;
507  G4double curNodeOffset, minCurCommonDelta, maxCurCommonDelta;
508  G4int minCurNodeNoDelta, maxCurNodeNoDelta;
509 
510  // Compute linear intersection distance to boundaries of max/min
511  // to collected nodes at current level
512  //
513  curNodeOffset = fVoxelNodeNo*fVoxelSliceWidth;
514  minCurCommonDelta = localPoint(fVoxelAxis)
515  - fVoxelHeader->GetMinExtent()-curNodeOffset;
516  maxCurNodeNoDelta = fVoxelNode->GetMaxEquivalentSliceNo()-fVoxelNodeNo;
517  minCurNodeNoDelta = fVoxelNodeNo-fVoxelNode->GetMinEquivalentSliceNo();
518  maxCurCommonDelta = fVoxelSliceWidth-minCurCommonDelta;
519  plusVoxelSafety = minCurNodeNoDelta*fVoxelSliceWidth+minCurCommonDelta;
520  minusVoxelSafety = maxCurNodeNoDelta*fVoxelSliceWidth+maxCurCommonDelta;
521  voxelSafety = std::min(plusVoxelSafety,minusVoxelSafety);
522 
523  if ( voxelSafety<0 )
524  {
525  voxelSafety = 0;
526  }
527 
528  return voxelSafety;
529 }

531 // ********************************************************************
532 // LocateNextVoxel
533 //
534 // Finds the next voxel from the current voxel and point
535 // in the specified direction.
536 //
537 // Returns false if all voxels considered
538 // true otherwise
539 // [current Step ends inside same voxel or leaves all voxels]
540 // ********************************************************************
541 //
542 G4bool G4ParameterisedNavigation::
543 LocateNextVoxel( const G4ThreeVector& localPoint,
544  const G4ThreeVector& localDirection,
545  const G4double currentStep,
546  const EAxis pAxis)
547 {
548  // If no best axis is specified, adopt default
549  // location strategy as for placements
550  //
551  if ( pAxis==kUndefined )
552  {
553  return G4VoxelNavigation::LocateNextVoxel(localPoint,
554  localDirection,
555  currentStep);
556  }
557 
558  G4bool isNewVoxel;
559  G4int newNodeNo;
560  G4double minVal, maxVal, curMinExtent, curCoord;
561 
562  curMinExtent = fVoxelHeader->GetMinExtent();
563  curCoord = localPoint(fVoxelAxis)+currentStep*localDirection(fVoxelAxis);
564  minVal = curMinExtent+fVoxelNode->GetMinEquivalentSliceNo()*fVoxelSliceWidth;
565  isNewVoxel = false;
566 
567  if ( minVal<=curCoord )
568  {
569  maxVal = curMinExtent
570  + (fVoxelNode->GetMaxEquivalentSliceNo()+1)*fVoxelSliceWidth;
571  if ( maxVal<curCoord )
572  {
573  newNodeNo = fVoxelNode->GetMaxEquivalentSliceNo()+1;
574  if ( newNodeNo<fVoxelHeader->GetNoSlices() )
575  {
576  fVoxelNodeNo = newNodeNo;
577  fVoxelNode = fVoxelHeader->GetSlice(newNodeNo)->GetNode();
578  isNewVoxel = true;
579  }
580  }
581  }
582  else
583  {
584  newNodeNo = fVoxelNode->GetMinEquivalentSliceNo()-1;
585 
586  // Must locate from newNodeNo no and down to setup stack and fVoxelNode
587  // Repeat or earlier code...
588  //
589  if ( newNodeNo>=0 )
590  {
591  fVoxelNodeNo = newNodeNo;
592  fVoxelNode = fVoxelHeader->GetSlice(newNodeNo)->GetNode();
593  isNewVoxel = true;
594  }
595  }
596  return isNewVoxel;
597 }

604 G4ParameterisedNavigation::LevelLocate( G4NavigationHistory& history,

610  G4ThreeVector& localPoint )

617  G4VSolid *pSolid;
618  G4ThreeVector samplePoint;

621  motherPhysical = history.GetTopVolume();

623  motherVoxelHeader = motherLogical->GetVoxelHeader();

641  //

660  else

662  // Enter this daughter
663  //
664  localPoint = samplePoint;
665

667  //

669 
670  // Set the correct solid and material in Logical Volume
671  //
672  G4LogicalVolume *pLogical = pPhysical->GetLogicalVolume();
673  pLogical->SetSolid(pSolid);
674  pLogical->UpdateMaterial(pParam->ComputeMaterial(replicaNo,
675  pPhysical, &parentTouchable) );
676  return true;
677  }
678  }
679  }
680  return false;
