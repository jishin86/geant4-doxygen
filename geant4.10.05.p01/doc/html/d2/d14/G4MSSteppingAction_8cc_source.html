57 void G4MSSteppingAction::UserSteppingAction(const G4Step* aStep)
58 {
59  G4StepPoint* preStepPoint = aStep->GetPreStepPoint();
60  G4Region* region = preStepPoint->GetPhysicalVolume()->GetLogicalVolume()->GetRegion();
61 
62  if(regionSensitive && (region!=theRegion)) return;
63 
64  G4double stlen = aStep->GetStepLength();
